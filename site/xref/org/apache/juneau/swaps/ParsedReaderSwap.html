28  /**
29   * Transforms the contents of a {@link Reader} into an {@code Object}.
30   *
31   * <h5 class='topic'>Description</h5>
32   *
33   * The {@code Reader} must contain JSON, Juneau-generated XML (output from {@link XmlSerializer}), or Juneau-generated
34   * HTML (output from {@link JsonSerializer}) in order to be parsed correctly.
35   *
36   * <p>
37   * Useful for serializing models that contain {@code Readers} created by {@code RestCall} instances.
38   *
39   * <p>
40   * This is a one-way transform, since {@code Readers} cannot be reconstituted.
41   *
42   * <h5 class='topic'>Behavior-specific subclasses</h5>
43   *
44   * The following direct subclasses are provided for convenience:
45   * <ul>
46   * 	<li>{@link Json} - Parses JSON text.
47   * 	<li>{@link Xml} - Parses XML text.
48   * 	<li>{@link Html} - Parses HTML text.
49   * 	<li>{@link PlainText} - Parses plain text.
50   * </ul>
51   *
52   * <h5 class='section'>See Also:</h5><ul>
53   * 	<li class='link'><a class="doclink" href="../../../../index.html#jm.Swaps">Swaps</a>
54  
55   * </ul>
56   */
57  public class ParsedReaderSwap extends ObjectSwap<Reader,Object> {
58  
59  	/** Reader transform for reading JSON text. */
60  	public static class Json extends ParsedReaderSwap {
61  		/** Constructor */
62  		public Json() {
63  			super(JsonParser.DEFAULT);
64  		}
65  	}
66  
67  	/** Reader transform for reading XML text. */
68  	public static class Xml extends ParsedReaderSwap {
69  		/** Constructor */
70  		public Xml() {
71  			super(XmlParser.DEFAULT);
72  		}
73  	}
74  
75  	/** Reader transform for reading HTML text. */
76  	public static class Html extends ParsedReaderSwap {
77  		/** Constructor */
78  		public Html() {
79  			super(HtmlParser.DEFAULT);
80  		}
81  	}
82  
83  	/** Reader transform for reading plain text. */
84  	public static class PlainText extends ParsedReaderSwap {
85  		/** Constructor */
86  		public PlainText() {
87  			super(null);
88  		}
89  	}
90  
91  	/** Reader transform for reading plain text. */
92  	public static class Uon extends ParsedReaderSwap {
93  		/** Constructor */
94  		public Uon() {
95  			super(UonParser.DEFAULT);
96  		}
97  	}
98  
99  	/** Reader transform for reading plain text. */
100 	public static class UrlEncoding extends ParsedReaderSwap {
101 		/** Constructor */
102 		public UrlEncoding() {
103 			super(UrlEncodingParser.DEFAULT);
104 		}
105 	}
106 
107 	/** The parser to use to parse the contents of the Reader. */
108 	private ReaderParser parser;
109 
110 	/**
111 	 * @param parser The parser to use to convert the contents of the reader to Java objects.
112 	 */
113 	public ParsedReaderSwap(ReaderParser parser) {
114 		this.parser = parser;
115 	}
116 
117 	/**
118 	 * Converts the specified {@link Reader} to an {@link Object} whose type is determined by the contents of the reader.
119 	 */
120 	@Override /* ObjectSwap */
121 	public Object swap(BeanSession session, Reader o) throws Exception {
122 		if (parser == null)
123 			return read(o);
124 		return parser.parse(o, Object.class);
125 	}
126 }
