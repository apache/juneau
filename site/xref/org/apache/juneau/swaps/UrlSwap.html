13  package org.apache.juneau.swaps;
14  
15  import java.net.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.swap.*;
19  
20  /**
21   * Transforms {@link URL} objects to and from {@link String Strings}.
22   *
23   * <h5 class='section'>See Also:</h5><ul>
24   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SwapBasics">Swaps</a>
25   * </ul>
26   */
27  public class UrlSwap extends StringSwap<URL> {
28  
29  	@Override /* ObjectSwap */
30  	public String swap(BeanSession session, URL o) throws Exception {
31  		return o.toString();
32  	}
33  
34  	@Override /* ObjectSwap */
35  	public URL unswap(BeanSession session, String o, ClassMeta<?> hint) throws Exception {
36  		return new URI(o).toURL();
37  	}
38  }
