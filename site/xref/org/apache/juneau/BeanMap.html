35  /**
36   * Java bean wrapper class.
37   *
38   * <h5 class='topic'>Description</h5>
39   *
40   * A wrapper that wraps Java bean instances inside of a {@link Map} interface that allows properties on the wrapped
41   * object can be accessed using the {@link Map#get(Object) get()} and {@link Map#put(Object,Object) put()} methods.
42   *
43   * <p>
44   * Use the {@link BeanContext} class to create instances of this class.
45   *
46   * <h5 class='topic'>Bean property order</h5>
47   *
48   * The order of the properties returned by the {@link Map#keySet() keySet()} and {@link Map#entrySet() entrySet()}
49   * methods are as follows:
50   * <ul class='spaced-list'>
51   * 	<li>
52   * 		If {@link Bean @Bean} annotation is specified on class, then the order is the same as the list of properties
53   * 		in the annotation.
54   * 	<li>
55   * 		If {@link Bean @Bean} annotation is not specified on the class, then the order is the same as that returned
56   * 		by the {@link java.beans.BeanInfo} class (i.e. ordered by definition in the class).
57   * </ul>
58   *
59   * <h5 class='topic'>POJO swaps</h5>
60   *
61   * If {@link ObjectSwap ObjectSwaps} are defined on the class types of the properties of this bean or the bean properties
62   * themselves, the {@link #get(Object)} and {@link #put(String, Object)} methods will automatically transform the
63   * property value to and from the serialized form.
64   *
65   * <h5 class='section'>See Also:</h5><ul>
66  
67   * </ul>
68   *
69   * @param <T> Specifies the type of object that this map encapsulates.
70   */
71  public class BeanMap<T> extends AbstractMap<String,Object> implements Delegate<T> {

100 	 * Instance of this class are instantiated through the BeanContext class.
101 	 *
102 	 * @param session The bean session object that created this bean map.
103 	 * @param bean The bean to wrap inside this map.
104 	 * @param meta The metadata associated with the bean class.
105 	 */
106 	protected BeanMap(BeanSession session, T bean, BeanMeta<T> meta) {
107 		this.session = session;
108 		this.bean = bean;
109 		this.meta = meta;
110 		if (meta.constructorArgs.length > 0)

112 		this.typePropertyName = session.getBeanTypePropertyName(meta.classMeta);

116 	 * Add a value to a collection or array property.
117 	 *
118 	 * <p>
119 	 * As a general rule, adding to arrays is not recommended since the array must be recreate each time this method is
120 	 * called.
121 	 *
122 	 * @param property Property name or child-element name (if {@link org.apache.juneau.xml.annotation.Xml#childName() @Xml(childName)} is specified).
123 	 * @param value The value to add to the collection or array.
124 	 */
125 	public void add(String property, Object value) {
126 		BeanPropertyMeta p = getPropertyMeta(property);
127 		if (p == null) {
128 			if (meta.ctx.isIgnoreUnknownBeanProperties())
129 				return;
130 			throw new BeanRuntimeException(meta.c, "Bean property ''{0}'' not found.", property);
131 		}
132 		p.add(this, property, value);
133 	}

135 	@Override /* Overridden from Map */
136 	public boolean containsKey(Object property) {
137 		// JUNEAU-248: Match the behavior of keySet() - only check properties map, not hiddenProperties
138 		String key = emptyIfNull(property);
139 		if (meta.properties.containsKey(key) && ! "*".equals(key))
140 			return true;
141 		if (meta.dynaProperty != null) {
142 			try {
143 				return meta.dynaProperty.getDynaMap(bean).containsKey(key);
144 			} catch (Exception e) {
145 				throw new BeanRuntimeException(e);
146 			}
147 		}
148 		return false;
149 	}

151 	/**
152 	 * Returns all the properties associated with the bean.
153 	 *
154 	 * @return A new set.
155 	 */
156 	@Override
157 	public Set<Entry<String,Object>> entrySet() {
158 
159 		// If this bean has a dyna-property, then we need to construct the entire set before returning.
160 		// Otherwise, we can create an iterator without a new data structure.
161 		if (meta.dynaProperty != null) {
162 			Set<Entry<String,Object>> s = set();
163 			forEachProperty(x -> true, x -> {
164 				if (x.isDyna()) {
165 					try {
166 						x.getDynaMap(bean).entrySet().forEach(y -> s.add(new BeanMapEntry(this, x, y.getKey())));
167 					} catch (Exception e) {
168 						throw new BeanRuntimeException(e);
169 					}
170 				} else {
171 					s.add(new BeanMapEntry(this, x, x.getName()));
172 				}
173 			});
174 			return s;
175 		}
176 
177 		// Construct our own anonymous set to implement this function.
178 		Set<Entry<String,Object>> s = new AbstractSet<>() {
179 
180 			// Get the list of properties from the meta object.
181 			// Note that the HashMap.values() method caches results, so this collection
182 			// will really only be constructed once per bean type since the underlying
183 			// map never changes.
184 			final Collection<BeanPropertyMeta> pSet = getProperties();
185 
186 			@Override /* Overridden from Set */
187 			public Iterator<java.util.Map.Entry<String,Object>> iterator() {
188 
189 				// Construct our own anonymous iterator that uses iterators against the meta.properties
190 				// map to maintain position.  This prevents us from having to construct any of our own
191 				// collection objects.
192 				return new Iterator<>() {
193 
194 					final Iterator<BeanPropertyMeta> pIterator = pSet.iterator();
195 
196 					@Override /* Overridden from Iterator */
197 					public boolean hasNext() {
198 						return pIterator.hasNext();
199 					}
200 
201 					@Override /* Overridden from Iterator */
202 					public Map.Entry<String,Object> next() {
203 						return new BeanMapEntry(BeanMap.this, pIterator.next(), null);
204 					}
205 
206 					@Override /* Overridden from Iterator */
207 					public void remove() {
208 						throw new UnsupportedOperationException("Cannot remove item from iterator.");
209 					}
210 				};
211 			}
212 
213 			@Override /* Overridden from Set */
214 			public int size() {
215 				return pSet.size();
216 			}
217 		};
218 
219 		return s;
220 	}

223 	 * Performs an action on each property in this bean map.
224 	 *
225 	 * @param filter The filter to apply to properties.
226 	 * @param action The action.
227 	 * @return This object.
228 	 */
229 	public BeanMap<T> forEachProperty(Predicate<BeanPropertyMeta> filter, Consumer<BeanPropertyMeta> action) {
230 		for (BeanPropertyMeta bpm : meta.propertyArray)
231 			if (filter.test(bpm))
232 				action.accept(bpm);
233 		return this;
234 	}

236 	/**
237 	 * Invokes all the getters on this bean and consumes the results.
238 	 *
239 	 * @param valueFilter Filter to apply to value before applying action.
240 	 * @param action The action to perform.
241 	 * @return The list of all bean property values.
242 	 */
243 	public BeanMap<T> forEachValue(Predicate<Object> valueFilter, BeanPropertyConsumer action) {
244 
245 		// Normal bean.
246 		if (meta.dynaProperty == null) {
247 			forEachProperty(BeanPropertyMeta::canRead, bpm -> {
248 				try {
249 					Object val = bpm.get(this, null);
250 					if (valueFilter.test(val))
251 						action.apply(bpm, bpm.getName(), val, null);
252 				} catch (Error e) {
253 					// Errors should always be uncaught.
254 					throw e;
255 				} catch (Throwable t) {
256 					action.apply(bpm, bpm.getName(), null, t);
257 				}
258 			});
259 
260 			// Bean with dyna properties.
261 		} else {
262 			Map<String,BeanPropertyValue> actions = (meta.sortProperties ? CollectionUtils.sortedMap() : map());
263 
264 			forEachProperty(x -> ! x.isDyna(), bpm -> {
265 				try {
266 					actions.put(bpm.getName(), new BeanPropertyValue(bpm, bpm.getName(), bpm.get(this, null), null));
267 				} catch (Error e) {
268 					// Errors should always be uncaught.
269 					throw e;
270 				} catch (Throwable t) {
271 					actions.put(bpm.getName(), new BeanPropertyValue(bpm, bpm.getName(), null, t));
272 				}
273 			});
274 
275 			forEachProperty(BeanPropertyMeta::isDyna, bpm -> {
276 				try {
277 					// TODO - This is kind of inefficient.
278 					Map<String,Object> dynaMap = bpm.getDynaMap(bean);
279 					if (dynaMap != null) {
280 						dynaMap.forEach((k, v) -> {
281 							Object val = bpm.get(this, k);
282 							actions.put(k, new BeanPropertyValue(bpm, k, val, null));
283 						});
284 					}
285 				} catch (Exception e) {
286 					e.printStackTrace();
287 				}
288 			});
289 
290 			actions.forEach((k, v) -> {
291 				if (valueFilter.test(v.getValue()))
292 					action.apply(v.getMeta(), v.getName(), v.getValue(), v.getThrown());
293 			});
294 		}
295 
296 		return this;
297 	}

299 	/**
300 	 * Gets a property on the bean.
301 	 *
302 	 * <p>
303 	 * If there is a {@link ObjectSwap} associated with this bean property or bean property type class, then this method
304 	 * will return the transformed value.
305 	 * For example, if the bean property type class is a {@link Date} and the bean property has the
306 	 * {@link org.apache.juneau.swaps.TemporalDateSwap.IsoInstant} swap associated with it through the
307 	 * {@link Swap#value() @Swap(value)} annotation, this method will return a String containing an
308 	 * ISO8601 date-time string value.
309 	 *
310 	 * <h5 class='section'>Example:</h5>
311 	 * <p class='bjava'>
312 	 * 	<jc>// Construct a bean with a 'birthDate' Date field</jc>
313 	 * 	Person <jv>person</jv> = <jk>new</jk> Person();
314 	 * 	<jv>person</jv>.setBirthDate(<jk>new</jk> Date(1, 2, 3, 4, 5, 6));
315 	 *
316 	 * 	<jc>// Create a bean context and add the ISO8601 date-time swap</jc>
317 	 * 	BeanContext <jv>beanContext</jv> = BeanContext.<jsm>create</jsm>().swaps(DateSwap.ISO8601DT.<jk>class</jk>).build();
318 	 *
319 	 * 	<jc>// Wrap our bean in a bean map</jc>
320 	 * 	BeanMap&lt;Person&gt; <jv>beanMap</jv> = <jv>beanContext</jv>.toBeanMap(<jv>person</jv>);
321 	 *
322 	 * 	<jc>// Get the field as a string (i.e. "'1901-03-03T04:05:06-5000'")</jc>
323 	 * 	String <jv>birthDate</jv> = <jv>beanMap</jv>.get(<js>"birthDate"</js>);
324 	 * </p>
325 	 *
326 	 * @param property The name of the property to get.
327 	 * @return The property value.
328 	 * @throws RuntimeException if any of the following occur.
329 	 * 	<ol>
330 	 * 		<li>BeanMapEntry does not exist on the underlying object.
331 	 * 		<li>Security settings prevent access to the underlying object getter method.
332 	 * 		<li>An exception occurred inside the getter method.
333 	 * 	</ol>
334 	 */
335 	@Override /* Overridden from Map */
336 	public Object get(Object property) {
337 		String pName = s(property);
338 		BeanPropertyMeta p = getPropertyMeta(pName);
339 		if (p == null)
340 			return meta.onReadProperty(this.bean, pName, null);
341 		return p.get(this, pName);
342 	}

345 	 * Same as {@link #get(Object)} but casts the value to the specific type.
346 	 *
347 	 * @param <T2> The type to cast to.
348 	 * @param property The name of the property to get.
349 	 * @param c The type to cast to.
350 	 * @return The property value.
351 	 * @throws RuntimeException if any of the following occur.
352 	 * 	<ol>
353 	 * 		<li>BeanMapEntry does not exist on the underlying object.
354 	 * 		<li>Security settings prevent access to the underlying object getter method.
355 	 * 		<li>An exception occurred inside the getter method.
356 	 * 	</ol>
357 	 * @throws ClassCastException if property is not the specified type.
358 	 */
359 	@SuppressWarnings("unchecked")
360 	public <T2> T2 get(String property, Class<T2> c) {
361 		String pName = s(property);
362 		BeanPropertyMeta p = getPropertyMeta(pName);
363 		if (p == null)
364 			return (T2)meta.onReadProperty(this.bean, pName, null);
365 		return (T2)p.get(this, pName);
366 	}

368 	/**
369 	 * Returns the wrapped bean object.
370 	 *
371 	 * <p>
372 	 * Triggers bean creation if bean has read-only properties set through a constructor defined by the
373 	 * {@link Beanc @Beanc} annotation.
374 	 *
375 	 * @return The inner bean object.
376 	 */
377 	public T getBean() {
378 		T b = getBean(true);
379 
380 		// If we have any arrays that need to be constructed, do it now.
381 		if (arrayPropertyCache != null) {
382 			arrayPropertyCache.forEach((k, v) -> {
383 				try {
384 					getPropertyMeta(k).setArray(b, v);
385 				} catch (Exception e1) {
386 					throw asRuntimeException(e1);
387 				}
388 			});
389 			arrayPropertyCache = null;
390 		}
391 
392 		// Initialize any null Optional<X> fields.
393 		for (BeanPropertyMeta pMeta : this.meta.propertyArray) {
394 			ClassMeta<?> cm = pMeta.getClassMeta();
395 			if (cm.isOptional() && pMeta.get(this, pMeta.getName()) == null)
396 				pMeta.set(this, pMeta.getName(), cm.getOptionalDefault());
397 		}
398 		// Do the same for hidden fields.
399 		this.meta.hiddenProperties.forEach((k, v) -> {
400 			ClassMeta<?> cm = v.getClassMeta();
401 			if (cm.isOptional() && v.get(this, v.getName()) == null)
402 				v.set(this, v.getName(), cm.getOptionalDefault());
403 		});
404 
405 		return b;
406 	}

408 	/**
409 	 * Returns the wrapped bean object.
410 	 *
411 	 * <p>
412 	 * If <c>create</c> is <jk>false</jk>, then this method may return <jk>null</jk> if the bean has read-only
413 	 * properties set through a constructor defined by the {@link Beanc @Beanc} annotation.
414 	 *
415 	 * <p>
416 	 * This method does NOT always return the bean in it's final state.
417 	 * Array properties temporary stored as ArrayLists are not finalized until the {@link #getBean()} method is called.
418 	 *
419 	 * @param create If bean hasn't been instantiated yet, then instantiate it.
420 	 * @return The inner bean object.
421 	 */
422 	public T getBean(boolean create) {
423 		/** If this is a read-only bean, then we need to create it. */
424 		if (bean == null && create && meta.constructorArgs.length > 0) {
425 			String[] props = meta.constructorArgs;
426 			ConstructorInfo c = meta.constructor;
427 			Object[] args = new Object[props.length];
428 			for (int i = 0; i < props.length; i++)
429 				args[i] = propertyCache.remove(props[i]);
430 			try {
431 				bean = c.<T>invoke(args);
432 				propertyCache.forEach((k, v) -> put(k, v));
433 				propertyCache = null;
434 			} catch (IllegalArgumentException e) {
435 				throw new BeanRuntimeException(e, meta.classMeta.innerClass, "IllegalArgumentException occurred on call to class constructor ''{0}'' with argument types ''{1}''", c.getSimpleName(),
436 					Json5Serializer.DEFAULT.toString(ClassUtils.getClasses(args)));
437 			} catch (Exception e) {
438 				throw new BeanRuntimeException(e);
439 			}
440 		}
441 		return bean;
442 	}

524 	 * Returns the names of all properties associated with the bean.
525 	 *
526 	 * <p>
527 	 * The returned set is unmodifiable.
528 	 */
529 	@Override /* Overridden from Map */
530 	public Set<String> keySet() {
531 		if (meta.dynaProperty == null)
532 			return meta.properties.keySet();
533 		Set<String> l = set();
534 		meta.properties.forEach((k, v) -> {
535 			if (! "*".equals(k))
536 				l.add(k);
537 		});
538 		try {
539 			l.addAll(meta.dynaProperty.getDynaMap(bean).keySet());
540 		} catch (Exception e) {
541 			throw new BeanRuntimeException(e);
542 		}
543 		return l;
544 	}

593 	 * Sets a property on the bean.
594 	 *
595 	 * <p>
596 	 * If there is a {@link ObjectSwap} associated with this bean property or bean property type class, then you must pass
597 	 * in a transformed value.
598 	 * For example, if the bean property type class is a {@link Date} and the bean property has the
599 	 * {@link org.apache.juneau.swaps.TemporalDateSwap.IsoInstant} swap associated with it through the
600 	 * {@link Swap#value() @Swap(value)} annotation, the value being passed in must be
601 	 * a String containing an ISO8601 date-time string value.
602 	 *
603 	 * <h5 class='section'>Example:</h5>
604 	 * <p class='bjava'>
605 	 * 	<jc>// Construct a bean with a 'birthDate' Date field</jc>
606 	 * 	Person <jv>person</jv> = <jk>new</jk> Person();
607 	 *
608 	 * 	<jc>// Create a bean context and add the ISO8601 date-time swap</jc>
609 	 * 	BeanContext <jv>beanContext</jv> = BeanContext.<jsm>create</jsm>().swaps(DateSwap.ISO8601DT.<jk>class</jk>).build();
610 	 *
