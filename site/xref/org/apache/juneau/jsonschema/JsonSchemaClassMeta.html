17  package org.apache.juneau.jsonschema;
18  
19  import static org.apache.juneau.common.utils.ThrowableUtils.*;
20  
21  import java.util.function.*;
22  
23  import org.apache.juneau.*;
24  import org.apache.juneau.annotation.*;
25  import org.apache.juneau.collections.*;
26  import org.apache.juneau.parser.*;
27  
28  /**
29   * Metadata on classes specific to the JSON-Schema serializer and pulled from the {@link Schema @Schema} annotation on
30   * the class.
31   *
32   * <h5 class='section'>See Also:</h5><ul>
33   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JsonSchemaDetails">JSON-Schema Support</a>
34  
35   * </ul>
36   */
37  public class JsonSchemaClassMeta extends ExtendedClassMeta {
38  
39  	private final JsonMap schema = new JsonMap();
40  
41  	/**
42  	 * Constructor.
43  	 *
44  	 * @param cm The class that this annotation is defined on.
45  	 * @param mp JSON-schema metadata provider (for finding information about other artifacts).
46  	 */
47  	public JsonSchemaClassMeta(ClassMeta<?> cm, JsonSchemaMetaProvider mp) {
48  		super(cm);
49  		Consumer<Schema> c = x -> {
50  			try {
51  				schema.append(SchemaAnnotation.asMap(x));
52  			} catch (ParseException e) {
53  				throw asRuntimeException(e);
54  			}
55  		};
56  		cm.forEachAnnotation(Schema.class, x -> true, c);
57  	}
58  
59  	/**
60  	 * Returns the {@link Schema @Schema} annotation defined on the class.
61  	 *
62  	 * @return The value of the annotation, or <jk>null</jk> if not specified.
63  	 */
64  	protected JsonMap getSchema() {
65  		return schema;
66  	}
67  }
