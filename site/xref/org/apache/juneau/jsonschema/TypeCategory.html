13  package org.apache.juneau.jsonschema;
14  
15  import static org.apache.juneau.common.utils.Utils.*;
16  
17  import java.util.*;
18  
19  import org.apache.juneau.common.utils.*;
20  
21  /**
22   * Represents possible values for JSONSCHEMA_addExamplesTo and JSONSCHEMA_addDescriptionsTo.
23   *
24   * <ul class='javatree'>
25   * 	<li class='jm'>{@link JsonSchemaGenerator.Builder#addExamplesTo(TypeCategory...)}
26   * 	<li class='jm'>{@link JsonSchemaGenerator.Builder#addDescriptionsTo(TypeCategory...)}
27   * </ul>
28   *
29   * <h5 class='section'>See Also:</h5><ul>
30   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JsonSchemaDetails">JSON-Schema Support</a>
31   * </ul>
32   */
33  public enum TypeCategory {
34  
35  	/** Beans */
36  	BEAN,
37  
38  	/** Map */
39  	MAP,
40  
41  	/** List/Set */
42  	COLLECTION,
43  
44  	/** Array */
45  	ARRAY,
46  
47  	/** Boolean (including primitives) */
48  	BOOLEAN,
49  
50  	/** Short/Integer/Long/Float/Double (including primitives) */
51  	NUMBER,
52  
53  	/** String/CharSequence/Character */
54  	STRING,
55  
56  	/** Enums */
57  	ENUM,
58  
59  	/** Anything else */
60  	OTHER,
61  
62  	/** Anything */
63  	ANY;
64  
65  	/**
66  	 * Parses a comma-delimited list of values into a set of {@link TypeCategory} values.
67  	 *
68  	 * @param s The comma-delimited string.
69  	 * @return A comma-delimited list of values into a set of {@link TypeCategory} values.
70  	 */
71  	public static Set<TypeCategory> parse(String s) {
72  		if (s == null || s.isEmpty())
73  			return Collections.emptySet();
74  		Set<TypeCategory> set = set();
75  		Utils.split(s, x -> set.add(valueOf(x.toUpperCase())));
76  		return set;
77  	}
78  
79  	/**
80  	 * Parses a comma-delimited list of values into an array of {@link TypeCategory} values.
81  	 *
82  	 * @param s The comma-delimited string.
83  	 * @return A comma-delimited list of values into an array of {@link TypeCategory} values.
84  	 */
85  	public static TypeCategory[] parseArray(String s) {
86  		if (s == null || s.isEmpty())
87  			return new TypeCategory[0];
88  		List<TypeCategory> list = list();
89  		Utils.split(s, x -> list.add(valueOf(x.toUpperCase())));
90  		return list.toArray(new TypeCategory[list.size()]);
91  	}
92  }
