17  package org.apache.juneau.annotation;
18  
19  import static org.apache.juneau.common.utils.CollectionUtils.*;
20  
21  import java.lang.annotation.*;
22  
23  /**
24   * Builder for {@link TargetedAnnotationImpl} objects.
25   *
26   * <h5 class='section'>See Also:</h5><ul>
27   * </ul>
28   *
29   * @param <B> The actual builder class.
30   */
31  public class TargetedAnnotationBuilder<B extends TargetedAnnotationBuilder<B>> extends AnnotationBuilder<B> {
32  
33  	String[] on = {};
34  
35  	/**
36  	 * Constructor.
37  	 *
38  	 * @param annotationType The annotation type of the annotation implementation class.
39  	 */
40  	public TargetedAnnotationBuilder(Class<? extends Annotation> annotationType) {
41  		super(annotationType);
42  	}
43  
44  	/**
45  	 * The targets this annotation applies to.
46  	 *
47  	 * @param values The targets this annotation applies to.
48  	 * @return This object.
49  	 */
50  	public B on(String...values) {
51  		for (var v : values)
52  			on = addAll(on, v);
53  		return asThis();
54  	}
55  }
