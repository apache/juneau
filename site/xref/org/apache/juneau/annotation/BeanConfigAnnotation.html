18  
19  import static org.apache.juneau.common.utils.CollectionUtils.*;
20  
21  import java.util.*;
22  
23  import org.apache.juneau.*;
24  import org.apache.juneau.common.reflect.*;

28  /**
29   * Utility classes and methods for the {@link BeanConfig @BeanConfig} annotation.
30   *
31   * <h5 class='section'>See Also:</h5><ul>
32   * </ul>
33   */
34  public class BeanConfigAnnotation {
35  
36  	/**
37  	 * Applies {@link BeanConfig} annotations to a {@link org.apache.juneau.BeanContext.Builder}.
38  	 */
39  	public static class Applier extends AnnotationApplier<BeanConfig,BeanContext.Builder> {
40  
41  		/**
42  		 * Constructor.
43  		 *
44  		 * @param vr The resolver for resolving values in annotations.
45  		 */
46  		public Applier(VarResolverSession vr) {
47  			super(BeanConfig.class, BeanContext.Builder.class, vr);
48  		}
49  
50  		@Override
51  		public void apply(AnnotationInfo<BeanConfig> ai, BeanContext.Builder b) {
52  			BeanConfig a = ai.inner();
53  
54  			// @formatter:off
55  			string(a.beanClassVisibility()).map(Visibility::valueOf).ifPresent(x -> b.beanClassVisibility(x));
56  			string(a.beanConstructorVisibility()).map(Visibility::valueOf).ifPresent(x -> b.beanConstructorVisibility(x));
57  			string(a.beanFieldVisibility()).map(Visibility::valueOf).ifPresent(x -> b.beanFieldVisibility(x));
58  			string(a.beanMethodVisibility()).map(Visibility::valueOf).ifPresent(x -> b.beanMethodVisibility(x));
59  			bool(a.beanMapPutReturnsOldValue()).ifPresent(x -> b.beanMapPutReturnsOldValue(x));
60  			bool(a.beansRequireDefaultConstructor()).ifPresent(x -> b.beansRequireDefaultConstructor(x));
61  			bool(a.beansRequireSerializable()).ifPresent(x -> b.beansRequireSerializable(x));
62  			bool(a.beansRequireSettersForGetters()).ifPresent(x -> b.beansRequireSettersForGetters(x));
63  			bool(a.disableBeansRequireSomeProperties()).ifPresent(x -> b.disableBeansRequireSomeProperties(x));
64  			bool(a.debug()).ifPresent(x -> b.debug(x));
65  			bool(a.findFluentSetters()).ifPresent(x -> b.findFluentSetters(x));
66  			bool(a.ignoreInvocationExceptionsOnGetters()).ifPresent(x -> b.ignoreInvocationExceptionsOnGetters(x));
67  			bool(a.ignoreInvocationExceptionsOnSetters()).ifPresent(x -> b.ignoreInvocationExceptionsOnSetters(x));
68  			bool(a.disableIgnoreMissingSetters()).ifPresent(x -> b.disableIgnoreMissingSetters(x));
69  			bool(a.disableIgnoreTransientFields()).ifPresent(x -> b.disableIgnoreTransientFields(x));
70  			bool(a.ignoreUnknownBeanProperties()).ifPresent(x -> b.ignoreUnknownBeanProperties(x));
71  			bool(a.ignoreUnknownEnumValues()).ifPresent(x -> b.ignoreUnknownEnumValues(x));
72  			bool(a.disableIgnoreUnknownNullBeanProperties()).ifPresent(x -> b.disableIgnoreUnknownNullBeanProperties(x));
73  			bool(a.sortProperties()).ifPresent(x -> b.sortProperties(x));
74  			bool(a.useEnumNames()).ifPresent(x -> b.useEnumNames(x));
75  			bool(a.disableInterfaceProxies()).ifPresent(x -> b.disableInterfaceProxies(x));
76  			bool(a.useJavaBeanIntrospector()).ifPresent(x -> b.useJavaBeanIntrospector(x));
77  			string(a.typePropertyName()).ifPresent(x -> b.typePropertyName(x));
78  			string(a.locale()).map(Locale::forLanguageTag).ifPresent(x -> b.locale(x));
79  			string(a.mediaType()).map(MediaType::of).ifPresent(x -> b.mediaType(x));
80  			string(a.timeZone()).map(TimeZone::getTimeZone).ifPresent(x -> b.timeZone(x));
81  			classes(a.dictionary()).ifPresent(x -> b.beanDictionary(x));
82  			classes(a.dictionary_replace()).ifPresent(x -> { b.beanDictionary().clear(); b.beanDictionary(x);});
83  			classes(a.swaps()).ifPresent(x -> b.swaps(x));
84  			classes(a.swaps_replace()).ifPresent(x -> { b.swaps().clear(); b.swaps(x);});
85  			classes(a.notBeanClasses()).ifPresent(x -> b.notBeanClasses(x));
86  			classes(a.notBeanClasses_replace()).ifPresent(x -> { b.notBeanClasses().clear(); b.notBeanClasses(x);});
87  			type(a.propertyNamer()).ifPresent(x -> b.propertyNamer(x));
88  			l(a.interfaces()).stream().map(x -> BeanAnnotation.create(x).interfaceClass(x).build()).forEach(x -> b.annotations(x));
89  			strings(a.notBeanPackages()).ifPresent(x -> b.notBeanPackages(x));
90  			strings(a.notBeanPackages_replace()).ifPresent(x -> {b.notBeanPackages().clear(); b.notBeanPackages(x);});
91  			// @formatter:on
92  		}
93  	}
94  }
