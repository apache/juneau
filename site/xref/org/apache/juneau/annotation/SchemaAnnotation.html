42  	/**
43  	 * Applies targeted {@link Schema} annotations to a {@link org.apache.juneau.Context.Builder}.
44  	 */
45  	public static class Apply extends AnnotationApplier<Schema,Context.Builder> {
46  
47  		/**
48  		 * Constructor.
49  		 *
50  		 * @param vr The resolver for resolving values in annotations.
51  		 */
52  		public Apply(VarResolverSession vr) {
53  			super(Schema.class, Context.Builder.class, vr);
54  		}
55  
56  		@Override
57  		public void apply(AnnotationInfo<Schema> ai, Context.Builder b) {
58  			Schema a = ai.inner();
59  			if (isEmptyArray(a.on(), a.onClass()))
60  				return;
61  			b.annotations(a);
62  		}
63  	}
