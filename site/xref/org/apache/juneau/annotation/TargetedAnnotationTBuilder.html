13  package org.apache.juneau.annotation;
14  
15  import java.lang.annotation.*;
16  
17  import org.apache.juneau.internal.*;
18  
19  /**
20   * An implementation of an annotation that has an <code>on</code> value targeting classes/methods/fields/constructors.
21   *
22   * <h5 class='section'>See Also:</h5><ul>
23   * </ul>
24   *
25   * @param <B> The actual builder class.
26   */
27  @FluentSetters
28  public class TargetedAnnotationTBuilder<B extends TargetedAnnotationTBuilder<B>> extends TargetedAnnotationBuilder<B> {
29  
30  	Class<?>[] onClass = {};
31  
32  	/**
33  	 * Constructor.
34  	 *
35  	 * @param annotationType The annotation type of the annotation implementation class.
36  	 */
37  	public TargetedAnnotationTBuilder(Class<? extends Annotation> annotationType) {
38  		super(annotationType);
39  	}
40  
41  	/**
42  	 * Appends the classes that this annotation applies to.
43  	 *
44  	 * @param value The values to append.
45  	 * @return This object.
46  	 */
47  	@FluentSetter
48  	public B on(Class<?>... value) {
49  		for (Class<?> v : value)
50  			on = ArrayUtils.append(on, v.getName());
51  		return asThis();
52  	}
53  
54  	/**
55  	 * Appends the classes that this annotation applies to.
56  	 *
57  	 * @param value The values to append.
58  	 * @return This object.
59  	 */
60  	@SuppressWarnings("unchecked")
61  	@FluentSetter
62  	public B onClass(Class<?>... value) {
63  		for (Class<?> v : value)
64  			onClass = ArrayUtils.append(onClass, v);
65  		return asThis();
66  	}
67  
68  	// <FluentSetters>
69  
70  
71  	// </FluentSetters>
72  }
