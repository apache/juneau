28  /**
29   * Provides an {@link InvocationHandler} for creating dynamic proxy instances of bean interfaces.
30   *
31   * <p>
32   * This class enables the creation of bean instances from interfaces without requiring concrete implementations.
33   * When the {@code useInterfaceProxies} setting is enabled in {@link BeanContext}, this handler is used to create
34   * proxy instances that implement bean interfaces.
35   *
36   * <p>
37   * The handler stores bean property values in an internal map and intercepts method calls to:
38   * <ul>
39   * 	<li><b>Getter methods</b> - Returns values from the internal property map</li>
40   * 	<li><b>Setter methods</b> - Stores values in the internal property map</li>
41   * 	<li><b>{@code equals(Object)}</b> - Compares property maps, with special handling for other proxy instances</li>
42   * 	<li><b>{@code hashCode()}</b> - Returns hash code based on the property map</li>
43   * 	<li><b>{@code toString()}</b> - Serializes the property map to JSON</li>
44   * </ul>
45   *
46   * <p>
47   * When comparing two proxy instances using {@code equals()}, if both are created with {@code BeanProxyInvocationHandler},
48   * the comparison is optimized by directly comparing their internal property maps rather than converting to {@link BeanMap}.
49   *
50   * <h5 class='section'>Example:</h5>
51   * <p class='bjava'>
52   * 	<jc>// Define an interface</jc>
53   * 	<jk>public interface</jk> Person {
54   * 		<jk>String</jk> <jsm>getName</jsm>();
55   * 		<jk>void</jk> <jsm>setName</jsm>(<jk>String</jk> name);
56   * 		<jk>int</jk> <jsm>getAge</jsm>();
57   * 		<jk>void</jk> <jsm>setAge</jsm>(<jk>int</jk> age);
58   * 	}
59   *
60   * 	<jc>// Create a proxy instance</jc>
61   * 	<jk>var</jk> bc = <jsm>BeanContext</jsm>.<jsm>create</jsm>().<jsm>useInterfaceProxies</jsm>().<jsm>build</jsm>();
62   * 	<jk>var</jk> person = bc.<jsm>getClassMeta</jsm>(Person.<jk>class</jk>).<jsm>newInstance</jsm>();
63   *
64   * 	<jc>// Use it like a regular bean</jc>
65   * 	person.<jsm>setName</jsm>(<js>"John"</js>);
66   * 	person.<jsm>setAge</jsm>(25);
67   * 	<jk>var</jk> name = person.<jsm>getName</jsm>(); <jc>// Returns "John"</jc>
68   * </p>
69   *
70   * @param <T> The interface class type
71   * @see BeanContext#isUseInterfaceProxies()
72   * @see BeanMeta#getBeanProxyInvocationHandler()
73   * @see Proxy#newProxyInstance(ClassLoader, Class[], InvocationHandler)
74   */
75  public class BeanProxyInvocationHandler<T> implements InvocationHandler {
76  
77  	private final BeanMeta<T> meta;						// The BeanMeta for this instance
78  	private Map<String,Object> beanProps;		// The map of property names to bean property values.
79  
80  	/**
81  	 * Constructor.
82  	 *
83  	 * @param meta The bean metadata for the interface. Must not be <jk>null</jk>.
84  	 */
85  	public BeanProxyInvocationHandler(BeanMeta<T> meta) {
86  		this.meta = meta;
87  		this.beanProps = new HashMap<>();
88  	}
89  
90  	/**
91  	 * Handles method invocations on the proxy instance.
92  	 *
93  	 * <p>
94  	 * This method intercepts all method calls on the proxy and routes them appropriately:
95  	 * <ul>
96  	 * 	<li>If the method is {@code equals(Object)}, compares property maps</li>
97  	 * 	<li>If the method is {@code hashCode()}, returns the hash code of the property map</li>
98  	 * 	<li>If the method is {@code toString()}, serializes the property map to JSON</li>
99  	 * 	<li>If the method is a getter (identified via {@link BeanMeta#getGetterProps()}), returns the property value</li>
100 	 * 	<li>If the method is a setter (identified via {@link BeanMeta#getSetterProps()}), stores the property value</li>
101 	 * 	<li>Otherwise, throws {@link UnsupportedOperationException}</li>
102 	 * </ul>
103 	 *
104 	 * <p>
105 	 * The {@code equals()} method has special optimization: when comparing two proxy instances created with this handler,
106 	 * it directly compares their internal property maps using {@link Proxy#getInvocationHandler(Object)} to access the
107 	 * other instance's handler.
108 	 *
109 	 * @param proxy The proxy instance on which the method was invoked
110 	 * @param method The method that was invoked
111 	 * @param args The arguments passed to the method, or <jk>null</jk> if no arguments
112 	 * @return The return value of the method invocation
113 	 * @throws UnsupportedOperationException If the method is not a supported bean method (getter, setter, equals, hashCode, or toString)
114 	 */
115 	@Override /* Overridden from InvocationHandler */
116 	public Object invoke(Object proxy, Method method, Object[] args) {
117 		var mi = info(method);
118 		if (mi.hasName("equals") && mi.hasParameterTypes(Object.class)) {
119 			var arg = args[0];
120 			if (arg == null)
121 				return false;
122 			if (proxy == arg)
123 				return true;
124 			if (eq(proxy.getClass(), arg.getClass())) {
125 				var ih = Proxy.getInvocationHandler(arg);
126 				if (ih instanceof BeanProxyInvocationHandler ih2) {
127 					return beanProps.equals(ih2.beanProps);
128 				}
129 			}
130 			return eq(beanProps, meta.getBeanContext().toBeanMap(arg));
131 		}
132 
133 		if (mi.hasName("hashCode") && mi.getParameterCount() == 0)
134 			return Integer.valueOf(this.beanProps.hashCode());
135 
136 		if (mi.hasName("toString") && mi.getParameterCount() == 0)
137 			return Json5Serializer.DEFAULT.toString(this.beanProps);
138 
139 		var prop = meta.getGetterProps().get(method);
140 		if (nn(prop))
141 			return beanProps.get(prop);
142 
143 		prop = meta.getSetterProps().get(method);
144 		if (nn(prop)) {
145 			beanProps.put(prop, args[0]);
146 			return null;
147 		}
148 
149 		throw unsupportedOp("Unsupported bean method.  method=''{0}''", method);
150 	}
151 }
