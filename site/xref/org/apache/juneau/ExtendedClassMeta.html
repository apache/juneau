13  package org.apache.juneau;
14  
15  /**
16   * Defines extended language-specific metadata associated with a class.
17   *
18   * <h5 class='section'>See Also:</h5><ul>
19   * </ul>
20   */
21  public class ExtendedClassMeta extends ExtendedMeta {
22  
23  	private final ClassMeta<?> cm;
24  
25  	/**
26  	 * Constructor.
27  	 *
28  	 * @param cm The metadata of the class we're extending.
29  	 * @throws BeanRuntimeException If any error occurred trying to construct the metadata.
30  	 */
31  	public ExtendedClassMeta(ClassMeta<?> cm) throws BeanRuntimeException {
32  		this.cm = cm;
33  	}
34  
35  	/**
36  	 * Returns the class metadata that was passed into the constructor.
37  	 *
38  	 * @return The class metadata that was passed into the constructor.
39  	 */
40  	protected ClassMeta<?> getClassMeta() {
41  		return cm;
42  	}
43  
44  	/**
45  	 * Convenience method for calling {@link ClassMeta#getInnerClass()} on the metadata passed to the constructor.
46  	 *
47  	 * @return The class wrapped by this metadata.
48  	 */
49  	protected Class<?> getInnerClass() {
50  		return cm.getInnerClass();
51  	}
52  }
