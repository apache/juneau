201 	/*
202 	 * Converts a Collection into an integer-indexed map.
203 	 */
204 	private static Map<Integer,Object> getCollectionMap(Collection<?> c) {
205 		Map<Integer,Object> m = new TreeMap<>();
206 		IntValue i = IntValue.create();
207 		c.forEach(o -> m.put(i.getAndIncrement(), o));
208 		return m;
209 	}
210 
211 	/*
212 	 * Converts an array into an integer-indexed map.
213 	 */
214 	private static Map<Integer,Object> getCollectionMap(Object array) {
215 		Map<Integer,Object> m = new TreeMap<>();
216 		for (int i = 0; i < Array.getLength(array); i++)
217 			m.put(i, Array.get(array, i));
218 		return m;
219 	}

233 	/*
234 	 * Workhorse method. Determines the type of object, and then calls the appropriate type-specific serialization method.
235 	 */
236 	private SerializerWriter serializeAnything(UonWriter out, Object o) throws IOException, SerializeException {
237 
238 		ClassMeta<?> aType;			// The actual type
239 		ClassMeta<?> sType;			// The serialized type
240 
241 		ClassMeta<?> eType = getExpectedRootType(o);
242 		aType = push2("root", o, eType);
243 		indent--;
244 		if (aType == null)
245 			aType = object();
246 
247 		sType = aType;
248 		var typeName = getBeanTypeName(this, eType, aType, null);
249 
250 		// Swap if necessary
251 		var swap = aType.getSwap(this);
252 		if (swap != null) {
253 			o = swap(swap, o);
254 			sType = swap.getSwapClassMeta(this);
255 
256 			// If the getSwapClass() method returns Object, we need to figure out
257 			// the actual type now.
258 			if (sType.isObject())
259 				sType = getClassMetaForObject(o);
260 		}
261 
262 		if (sType.isMap()) {
263 			if (o instanceof BeanMap)
264 				serializeBeanMap(out, (BeanMap)o, typeName);
265 			else
266 				serializeMap(out, (Map)o, sType);
267 		} else if (sType.isBean()) {
268 			serializeBeanMap(out, toBeanMap(o), typeName);
269 		} else if (sType.isCollection() || sType.isArray()) {
270 			var m = sType.isCollection() ? getCollectionMap((Collection)o) : getCollectionMap(o);
271 			serializeCollectionMap(out, m, getClassMeta(Map.class, Integer.class, Object.class));
272 		} else if (sType.isReader()) {
273 			pipe((Reader)o, out);
274 		} else if (sType.isInputStream()) {
275 			pipe((InputStream)o, out);
276 		} else {
277 			// All other types can't be serialized as key/value pairs, so we create a
278 			// mock key/value pair with a "_value" key.
279 			out.append("_value=");
280 			pop();
281 			super.serializeAnything(out, o, null, null, null);
282 			return out;
283 		}
284 
285 		pop();
286 		return out;
287 	}

289 	private SerializerWriter serializeBeanMap(UonWriter out, BeanMap<?> m, String typeName) throws SerializeException {
290 		Flag addAmp = Flag.create();
291 
292 		if (typeName != null) {
293 			var pm = m.getMeta().getTypeProperty();
294 			out.appendObject(pm.getName(), true).append('=').appendObject(typeName, false);
295 			addAmp.set();
296 		}
297 
298 		Predicate<Object> checkNull = x -> isKeepNullProperties() || x != null;
299 		m.forEachValue(checkNull, (pMeta, key, value, thrown) -> {
300 			ClassMeta<?> cMeta = pMeta.getClassMeta();
301 			ClassMeta<?> sMeta = cMeta.getSerializedClassMeta(this);
302 
303 			if (thrown != null)
304 				onBeanGetterException(pMeta, thrown);
305 
306 			if (canIgnoreValue(sMeta, key, value))
307 				return;
308 
309 			if (value != null && shouldUseExpandedParams(pMeta)) {
310 				// Transformed object array bean properties may be transformed resulting in ArrayLists,
311 				// so we need to check type if we think it's an array.
312 				if (sMeta.isCollection() || value instanceof Collection) {
313 					((Collection<?>)value).forEach(x -> {
314 						addAmp.ifSet(() -> out.cr(indent).append('&')).set();
315 						out.appendObject(key, true).append('=');
316 						super.serializeAnything(out, x, cMeta.getElementType(), key, pMeta);
317 					});
318 				} else /* array */ {
319 					for (int i = 0; i < Array.getLength(value); i++) {
320 						addAmp.ifSet(() -> out.cr(indent).append('&')).set();
321 						out.appendObject(key, true).append('=');
322 						super.serializeAnything(out, Array.get(value, i), cMeta.getElementType(), key, pMeta);
323 					}
324 				}
325 			} else {
326 				addAmp.ifSet(() -> out.cr(indent).append('&')).set();
327 				out.appendObject(key, true).append('=');
328 				super.serializeAnything(out, value, cMeta, key, pMeta);
329 			}
330 		});
331 
332 		return out;
333 	}

335 	private SerializerWriter serializeCollectionMap(UonWriter out, Map<?,?> m, ClassMeta<?> type) throws SerializeException {
336 
337 		ClassMeta<?> valueType = type.getValueType();
338 
339 		Flag addAmp = Flag.create();
340 
341 		m.forEach((k, v) -> {
342 			addAmp.ifSet(() -> out.cr(indent).append('&')).set();
343 			out.append(k).append('=');
344 			super.serializeAnything(out, v, valueType, null, null);
345 		});
346 
347 		return out;
348 	}

350 	private SerializerWriter serializeMap(UonWriter out, Map m, ClassMeta<?> type) throws SerializeException {
351 
352 		ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
353 
354 		Flag addAmp = Flag.create();
355 
356 		forEachEntry(m, e -> {
357 			var key = generalize(e.getKey(), keyType);
358 			var value = e.getValue();
359 
360 			if (shouldUseExpandedParams(value)) {
361 				if (value instanceof Collection) {
362 					((Collection<?>)value).forEach(x -> {
363 						addAmp.ifSet(() -> out.cr(indent).append('&')).set();
364 						out.appendObject(key, true).append('=');
365 						super.serializeAnything(out, x, null, Utils.s(key), null);
366 					});
367 				} else /* array */ {
368 					for (int i = 0; i < Array.getLength(value); i++) {
369 						addAmp.ifSet(() -> out.cr(indent).append('&')).set();
370 						out.appendObject(key, true).append('=');
371 						super.serializeAnything(out, Array.get(value, i), null, Utils.s(key), null);
372 					}
373 				}
374 			} else {
375 				addAmp.ifSet(() -> out.cr(indent).append('&')).set();
376 				out.appendObject(key, true).append('=');
377 				super.serializeAnything(out, value, valueType, (key == null ? null : key.toString()), null);
378 			}
379 		});
380 
381 		return out;
382 	}

384 	/*
385 	 * Returns <jk>true</jk> if the specified bean property should be expanded as multiple key-value pairs.
386 	 */
387 	private boolean shouldUseExpandedParams(BeanPropertyMeta pMeta) {
388 		ClassMeta<?> cm = pMeta.getClassMeta().getSerializedClassMeta(this);
389 		if (cm.isCollectionOrArray()) {
390 			if (isExpandedParams() || getUrlEncodingClassMeta(pMeta.getBeanMeta().getClassMeta()).isExpandedParams())
391 				return true;
392 		}
393 		return false;
394 	}
395 
396 	/*
397 	 * Returns <jk>true</jk> if the specified value should be represented as an expanded parameter list.
398 	 */
399 	private boolean shouldUseExpandedParams(Object value) {
400 		if (value == null || ! isExpandedParams())
401 			return false;
402 		ClassMeta<?> cm = getClassMetaForObject(value).getSerializedClassMeta(this);
403 		if (cm.isCollectionOrArray()) {
404 			if (isExpandedParams())
405 				return true;
406 		}
407 		return false;
408 	}

411 	protected void doSerialize(SerializerPipe out, Object o) throws IOException, SerializeException {
