204 	 * Returns <jk>true</jk> if the specified bean property should be expanded as multiple key-value pairs.

207 	 * @return <jk>true</jk> if the specified bean property should be expanded as multiple key-value pairs.

209 	public final boolean shouldUseExpandedParams(BeanPropertyMeta pMeta) {
210 		ClassMeta<?> cm = pMeta.getClassMeta().getSerializedClassMeta(this);
211 		if (cm.isCollectionOrArray()) {
212 			if (isExpandedParams() || getUrlEncodingClassMeta(pMeta.getBeanMeta().getClassMeta()).isExpandedParams())
213 				return true;
214 		}
215 		return false;
216 	}

218 	private <T> T parseAnything(ClassMeta<T> eType, UonReader r, Object outer) throws IOException, ParseException, ExecutableException {
219 
220 		if (eType == null)
221 			eType = (ClassMeta<T>)object();
222 		ObjectSwap<T,Object> swap = (ObjectSwap<T,Object>)eType.getSwap(this);
223 		BuilderSwap<T,Object> builder = (BuilderSwap<T,Object>)eType.getBuilderSwap(this);
224 		ClassMeta<?> sType = null;
225 		if (nn(builder))
226 			sType = builder.getBuilderClassMeta(this);
227 		else if (nn(swap))
228 			sType = swap.getSwapClassMeta(this);
229 		else
230 			sType = eType;
231 
232 		if (sType.isOptional())
233 			return (T)opt(parseAnything(eType.getElementType(), r, outer));
234 
235 		int c = r.peekSkipWs();
236 		if (c == '?')
237 			r.read();  // NOSONAR - skip leading '?'.
238 
239 		Object o;
240 
241 		if (sType.isObject()) {
242 			JsonMap m = new JsonMap(this);
243 			parseIntoMap2(r, m, getClassMeta(Map.class, String.class, Object.class), outer);
244 			if (m.containsKey("_value"))
245 				o = m.get("_value");
246 			else
247 				o = cast(m, null, eType);
248 		} else if (sType.isMap()) {
249 			Map m = (sType.canCreateNewInstance() ? (Map)sType.newInstance() : newGenericMap(sType));
250 			o = parseIntoMap2(r, m, sType, m);
251 		} else if (nn(builder)) {
252 			BeanMap m = toBeanMap(builder.create(this, eType));
253 			m = parseIntoBeanMap(r, m);
254 			o = m == null ? null : builder.build(this, m.getBean(), eType);
255 		} else if (sType.canCreateNewBean(outer)) {
256 			BeanMap m = newBeanMap(outer, sType.getInnerClass());
257 			m = parseIntoBeanMap(r, m);
258 			o = m == null ? null : m.getBean();
259 		} else if (sType.isCollection() || sType.isArray() || sType.isArgs()) {
260 			// ?1=foo&2=bar...
261 			Collection c2 = ((sType.isArray() || sType.isArgs()) || ! sType.canCreateNewInstance(outer)) ? new JsonList(this) : (Collection)sType.newInstance();
262 			Map<Integer,Object> m = new TreeMap<>();
263 			parseIntoMap2(r, m, sType, c2);
264 			c2.addAll(m.values());
265 			if (sType.isArray())
266 				o = CollectionUtils.toArray(c2, sType.getElementType().getInnerClass());
267 			else if (sType.isArgs())
268 				o = c2.toArray(new Object[c2.size()]);
269 			else
270 				o = c2;
271 		} else {
272 			// It could be a non-bean with _type attribute.
273 			JsonMap m = new JsonMap(this);
274 			parseIntoMap2(r, m, getClassMeta(Map.class, String.class, Object.class), outer);
275 			if (m.containsKey(getBeanTypePropertyName(eType)))
276 				o = cast(m, null, eType);
277 			else if (m.containsKey("_value"))
278 				o = convertToType(m.get("_value"), sType);
279 			else if (nn(sType.getProxyInvocationHandler())) {
280 				o = newBeanMap(outer, sType.getInnerClass()).load(m).getBean();
281 			} else {
282 				if (nn(sType.getNotABeanReason()))
283 					throw new ParseException(this, "Class ''{0}'' could not be instantiated as application/x-www-form-urlencoded.  Reason: ''{1}''", sType, sType.getNotABeanReason());
284 				throw new ParseException(this, "Malformed application/x-www-form-urlencoded input for class ''{0}''.", sType);
285 			}
286 		}
287 
288 		if (nn(swap) && nn(o))
289 			o = unswap(swap, o, eType);
290 
291 		if (nn(outer))
292 			setParent(eType, o, outer);
293 
294 		return (T)o;
295 	}

297 	private <T> BeanMap<T> parseIntoBeanMap(UonReader r, BeanMap<T> m) throws IOException, ParseException, ExecutableException {
298 
299 		int c = r.peekSkipWs();
300 		if (c == -1)
301 			return m;
302 
303 		// S1: Looking for attrName start.
304 		// S2: Found attrName end, looking for =.
305 		// S3: Found =, looking for valStart.
306 		// S4: Looking for , or }
307 
308 		boolean isInEscape = false;
309 
310 		var state = S1;
311 		String currAttr = "";
312 		mark();
313 		try {
314 			while (c != -1) {
315 				c = r.read();
316 				if (! isInEscape) {
317 					if (state == S1) {
318 						if (c == -1) {
319 							return m;
320 						}
321 						r.unread();
322 						mark();
323 						currAttr = parseAttrName(r, true);
324 						if (currAttr == null)  // Value was '%00'
325 							return null;
326 						state = S2;
327 					} else if (state == S2) {
328 						if (c == '\u0002')
329 							state = S3;
330 						else if (c == -1 || c == '\u0001') {
331 							m.put(currAttr, null);
332 							if (c == -1)
333 								return m;
334 							state = S1;
335 						}
336 					} else if (state == S3) {
337 						if (c == -1 || c == '\u0001') {
338 							if (! currAttr.equals(getBeanTypePropertyName(m.getClassMeta()))) {
339 								BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
340 								if (pMeta == null) {
341 									onUnknownProperty(currAttr, m, null);
342 									unmark();
343 								} else {
344 									unmark();
345 									setCurrentProperty(pMeta);
346 									// In cases of "&foo=", create an empty instance of the value if createable.
347 									// Otherwise, leave it null.
348 									ClassMeta<?> cm = pMeta.getClassMeta();
349 									if (cm.canCreateNewInstance()) {
350 										try {
351 											pMeta.set(m, currAttr, cm.newInstance());
352 										} catch (BeanRuntimeException e) {
353 											onBeanSetterException(pMeta, e);
354 											throw e;
355 										}
356 									}
357 									setCurrentProperty(null);
358 								}
359 							}
360 							if (c == -1)
361 								return m;
362 							state = S1;
363 						} else {
364 							if (! currAttr.equals(getBeanTypePropertyName(m.getClassMeta()))) {
365 								BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
366 								if (pMeta == null) {
367 									onUnknownProperty(currAttr, m, parseAnything(object(), r.unread(), m.getBean(false), true, null));
368 									unmark();
369 								} else {
370 									unmark();
371 									setCurrentProperty(pMeta);
372 									if (shouldUseExpandedParams(pMeta)) {
373 										ClassMeta et = pMeta.getClassMeta().getElementType();
374 										Object value = parseAnything(et, r.unread(), m.getBean(false), true, pMeta);
375 										setName(et, value, currAttr);
376 										try {
377 											pMeta.add(m, currAttr, value);
378 										} catch (BeanRuntimeException e) {
379 											onBeanSetterException(pMeta, e);
380 											throw e;
381 										}
382 									} else {
383 										ClassMeta<?> cm = pMeta.getClassMeta();
384 										Object value = parseAnything(cm, r.unread(), m.getBean(false), true, pMeta);
385 										setName(cm, value, currAttr);
386 										try {
387 											pMeta.set(m, currAttr, value);
388 										} catch (BeanRuntimeException e) {
389 											onBeanSetterException(pMeta, e);
390 											throw e;
391 										}
392 									}
393 									setCurrentProperty(null);
394 								}
395 							}
396 							state = S4;
397 						}
398 					} else if (state == S4) {
399 						if (c == '\u0001')
400 							state = S1;
401 						else if (c == -1) {
402 							return m;
403 						}
404 					}
405 				}
406 				isInEscape = (c == '\\' && ! isInEscape);
407 			}
408 			if (state == S1)
409 				throw new ParseException(this, "Could not find attribute name on object.");
410 			if (state == S2)
411 				throw new ParseException(this, "Could not find '=' following attribute name on object.");
412 			if (state == S3)
413 				throw new ParseException(this, "Could not find value following '=' on object.");
414 			if (state == S4)
415 				throw new ParseException(this, "Could not find end of object.");
416 		} finally {
417 			unmark();
418 		}
419 
420 		return null; // Unreachable.
421 	}

423 	private <K,V> Map<K,V> parseIntoMap2(UonReader r, Map<K,V> m, ClassMeta<?> type, Object outer) throws IOException, ParseException, ExecutableException {
424 
425 		ClassMeta<K> keyType = (ClassMeta<K>)(type.isArgs() || type.isCollectionOrArray() ? getClassMeta(Integer.class) : type.getKeyType());
426 
427 		int c = r.peekSkipWs();
428 		if (c == -1)
429 			return m;
430 
431 		// S1: Looking for attrName start.
432 		// S2: Found attrName end, looking for =.
433 		// S3: Found =, looking for valStart.
434 		// S4: Looking for & or end.
435 
436 		boolean isInEscape = false;
437 
438 		var state = S1;
439 		int argIndex = 0;
440 		K currAttr = null;
441 		while (c != -1) {
442 			c = r.read();
443 			if (! isInEscape) {
444 				if (state == S1) {
445 					if (c == -1)
446 						return m;
447 					r.unread();
448 					Object attr = parseAttr(r, true);
449 					currAttr = attr == null ? null : convertAttrToType(m, trim(attr.toString()), keyType);
450 					state = S2;
451 					c = 0; // Avoid isInEscape if c was '\'
452 				} else if (state == S2) {
453 					if (c == '\u0002')
454 						state = S3;
455 					else if (c == -1 || c == '\u0001') {
456 						m.put(currAttr, null);
457 						if (c == -1)
458 							return m;
459 						state = S1;
460 					}
461 				} else if (state == S3) {
462 					if (c == -1 || c == '\u0001') {
463 						ClassMeta<V> valueType = (ClassMeta<V>)(type.isArgs() ? type.getArg(argIndex++) : type.isCollectionOrArray() ? type.getElementType() : type.getValueType());
464 						V value = convertAttrToType(m, "", valueType);
465 						m.put(currAttr, value);
466 						if (c == -1)
467 							return m;
468 						state = S1;
469 					} else {
470 						// For performance, we bypass parseAnything for string values.
471 						ClassMeta<V> valueType = (ClassMeta<V>)(type.isArgs() ? type.getArg(argIndex++) : type.isCollectionOrArray() ? type.getElementType() : type.getValueType());
472 						V value = (V)(valueType.isString() ? super.parseString(r.unread(), true) : super.parseAnything(valueType, r.unread(), outer, true, null));
473 
474 						// If we already encountered this parameter, turn it into a list.
475 						if (m.containsKey(currAttr) && valueType.isObject()) {
476 							Object v2 = m.get(currAttr);
477 							if (! (v2 instanceof JsonList)) {
478 								v2 = new JsonList(v2).setBeanSession(this);
479 								m.put(currAttr, (V)v2);
480 							}
481 							((JsonList)v2).add(value);
482 						} else {
483 							m.put(currAttr, value);
484 						}
485 						state = S4;
486 						c = 0; // Avoid isInEscape if c was '\'
487 					}
488 				} else if (state == S4) {
489 					if (c == '\u0001')
490 						state = S1;
491 					else if (c == -1) {
492 						return m;
493 					}
494 				}
495 			}
496 			isInEscape = (c == '\\' && ! isInEscape);
497 		}
498 		if (state == S1)
499 			throw new ParseException(this, "Could not find attribute name on object.");
500 		if (state == S2)
501 			throw new ParseException(this, "Could not find '=' following attribute name on object.");
502 		if (state == S3)
503 			throw new ParseException(this, "Dangling '=' found in object entry");
504 		if (state == S4)
505 			throw new ParseException(this, "Could not find end of object.");
506 
507 		return null; // Unreachable.
508 	}

510 	@Override /* Overridden from ParserSession */
511 	protected <T> T doParse(ParserPipe pipe, ClassMeta<T> type) throws IOException, ParseException, ExecutableException {
512 		try (UonReader r = getUonReader(pipe, true)) {
513 			return parseAnything(type, r, getOuter());
514 		}
515 	}

517 	@Override /* Overridden from ReaderParserSession */
518 	protected <K,V> Map<K,V> doParseIntoMap(ParserPipe pipe, Map<K,V> m, Type keyType, Type valueType) throws Exception {
519 		try (UonReader r = getUonReader(pipe, true)) {
520 			if (r.peekSkipWs() == '?')
521 				r.read();  // NOSONAR - skip leading '?'.
522 			m = parseIntoMap2(r, m, getClassMeta(Map.class, keyType, valueType), null);
523 			return m;
524 		}
