13  package org.apache.juneau.urlencoding;
14  
15  import org.apache.juneau.*;
16  import org.apache.juneau.urlencoding.annotation.*;
17  
18  /**
19   * Metadata on classes specific to the URL-Encoding serializers and parsers pulled from the {@link UrlEncoding @UrlEncoding} annotation on the class.
20   *
21   * <h5 class='section'>See Also:</h5><ul>
22   * 	<li class='link'><a class="doclink" href="../../../../index.html#jm.UrlEncodingDetails">URL-Encoding Details</a>
23   * </ul>
24   */
25  public class UrlEncodingClassMeta extends ExtendedClassMeta {
26  
27  	private final boolean expandedParams;
28  
29  	/**
30  	 * Constructor.
31  	 *
32  	 * @param cm The class that this annotation is defined on.
33  	 * @param mp URL-encoding metadata provider (for finding information about other artifacts).
34  	 */
35  	public UrlEncodingClassMeta(ClassMeta<?> cm, UrlEncodingMetaProvider mp) {
36  		super(cm);
37  
38  		Value<Boolean> expandedParams = Value.empty();
39  		cm.forEachAnnotation(UrlEncoding.class, UrlEncoding::expandedParams, x -> expandedParams.set(true));
40  		this.expandedParams = expandedParams.orElse(false);
41  	}
42  
43  	/**
44  	 * Returns the {@link UrlEncoding#expandedParams()} annotation defined on the class.
45  	 *
46  	 * @return The value of the {@link UrlEncoding#expandedParams()} annotation.
47  	 */
48  	protected boolean isExpandedParams() {
49  		return expandedParams;
50  	}
51  }
