17  package org.apache.juneau.urlencoding;
18  
19  import org.apache.juneau.*;
20  import org.apache.juneau.common.collections.*;
21  import org.apache.juneau.urlencoding.annotation.*;
22  
23  /**
24   * Metadata on classes specific to the URL-Encoding serializers and parsers pulled from the {@link UrlEncoding @UrlEncoding} annotation on the class.
25   *
26   * <h5 class='section'>See Also:</h5><ul>
27   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/UrlEncodingBasics">URL-Encoding Basics</a>
28   * </ul>
29   */
30  public class UrlEncodingClassMeta extends ExtendedClassMeta {
31  
32  	private final boolean expandedParams;
33  
34  	/**
35  	 * Constructor.
36  	 *
37  	 * @param cm The class that this annotation is defined on.
38  	 * @param mp URL-encoding metadata provider (for finding information about other artifacts).
39  	 */
40  	public UrlEncodingClassMeta(ClassMeta<?> cm, UrlEncodingMetaProvider mp) {
41  		super(cm);
42  
43  		var expandedParams = Value.<Boolean>empty();
44  		cm.forEachAnnotation(UrlEncoding.class, UrlEncoding::expandedParams, x -> expandedParams.set(true));
45  		this.expandedParams = expandedParams.orElse(false);
46  	}
47  
48  	/**
49  	 * Returns the {@link UrlEncoding#expandedParams()} annotation defined on the class.
50  	 *
51  	 * @return The value of the {@link UrlEncoding#expandedParams()} annotation.
52  	 */
53  	protected boolean isExpandedParams() { return expandedParams; }
54  }
