610 	/**
611 	 * Used to resolve <c>ClassMetas</c> of type <c>Collection</c> and <c>Map</c> that have
612 	 * <c>ClassMeta</c> values that themselves could be collections or maps.
613 	 *
614 	 * <p>
615 	 * <c>Collection</c> meta objects are assumed to be followed by zero or one meta objects indicating the
616 	 * element type.
617 	 *
618 	 * <p>
619 	 * <c>Map</c> meta objects are assumed to be followed by zero or two meta objects indicating the key and value
620 	 * types.
621 	 *
622 	 * <p>
623 	 * The array can be arbitrarily long to indicate arbitrarily complex data structures.
624 	 *
625 	 * <h5 class='section'>Examples:</h5>
626 	 * <ul>
627 	 * 	<li><code>getClassMeta(String.<jk>class</jk>);</code> - A normal type.
628 	 * 	<li><code>getClassMeta(List.<jk>class</jk>);</code> - A list containing objects.
629 	 * 	<li><code>getClassMeta(List.<jk>class</jk>, String.<jk>class</jk>);</code> - A list containing strings.
630 	 * 	<li><code>getClassMeta(LinkedList.<jk>class</jk>, String.<jk>class</jk>);</code> - A linked-list containing
631 	 * 		strings.
632 	 * 	<li><code>getClassMeta(LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);</code> -
633 	 * 		A linked-list containing linked-lists of strings.
634 	 * 	<li><code>getClassMeta(Map.<jk>class</jk>);</code> - A map containing object keys/values.
635 	 * 	<li><code>getClassMeta(Map.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);</code> - A map
636 	 * 		containing string keys/values.
637 	 * 	<li><code>getClassMeta(Map.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);</code> -
638 	 * 		A map containing string keys and values of lists containing beans.
639 	 * </ul>
640 	 *

1161 		if (to == null)
1162 			to = (ClassMeta<T>)object();
1163 
1164 		try {
1165 			// Handle the case of a null value.
1166 			if (value == null) {
1167 
1168 				// If it's a primitive, then use the converters to get the default value for the primitive type.
1169 				if (to.isPrimitive())
1170 					return to.getPrimitiveDefault();
1171 
1172 				// Otherwise, just return null.
1173 				return to.isOptional() ? (T)to.getOptionalDefault() : null;

1176 			if (to.isOptional() && (! (value instanceof Optional)))

1178 
1179 			Class<T> tc = to.getInnerClass();
1180 
1181 			// If no conversion needed, then just return the value.
1182 			// Don't include maps or collections, because child elements may need conversion.
1183 			if (tc.isInstance(value))
1184 				if (! ((to.isMap() && to.getValueType().isNotObject()) || ((to.isCollection() || to.isOptional()) && to.getElementType().isNotObject())))

1188 		if (nn(swap)) {

1191 			if (nc.isParentOf(tc) && fc.isParentOf(value.getClass()))
1192 				return (T)swap.unswap(this, value, to);
1193 			ClassMeta fcm = getClassMeta(fc.inner());
1194 				if (fcm.isNumber() && value instanceof Number) {
1195 					value = convertToMemberType(null, value, fc.inner());

1199

1204 			var fc = swap.getSwapClass();
