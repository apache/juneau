13  package org.apache.juneau.internal;
14  
15  import java.util.concurrent.locks.*;
16  
17  /**
18   * An extension of {@link ReentrantReadWriteLock} with convenience methods for creating
19   * auto-closeable locks.
20   */
21  public class SimpleReadWriteLock extends ReentrantReadWriteLock {
22  	private static final long serialVersionUID = 1L;
23  
24  	//-----------------------------------------------------------------------------------------------------------------
25  	// Static
26  	//-----------------------------------------------------------------------------------------------------------------
27  
28  	/**
29  	 * A no-op lock.
30  	 */
31  	public static SimpleReadWriteLock NO_OP = new SimpleReadWriteLock() {
32  		private static final long serialVersionUID = 1L;
33  
34  		@Override
35  		public SimpleLock write() {
36  			return SimpleLock.NO_OP;
37  		}
38  		@Override
39  		public SimpleLock read(){
40  			return SimpleLock.NO_OP;
41  		}
42  	};
43  
44  	//-----------------------------------------------------------------------------------------------------------------
45  	// Instance
46  	//-----------------------------------------------------------------------------------------------------------------
47  
48  	/**
49  	 * Constructor.
50  	 */
51  	public SimpleReadWriteLock() {
52  	}
53  
54  	/**
55  	 * Constructor
56  	 *
57  	 * @param fair <jk>true</jk> if this lock should use a fair ordering policy.
58  	 */
59  	public SimpleReadWriteLock(boolean fair) {
60  		super(fair);
61  	}
62  
63  	/**
64  	 * Construct a write lock.
65  	 *
66  	 * @return A new closeable write lock.
67  	 */
68  	public SimpleLock write(){
69  		return new SimpleLock(writeLock());
70  	}
71  
72  	/**
73  	 * Construct a read lock.
74  	 *
75  	 * @return A new closeable read lock.
76  	 */
77  	public SimpleLock read(){
78  		return new SimpleLock(readLock());
79  	}
80  }
