13  package org.apache.juneau.internal;
14  
15  import java.util.*;
16  
17  import org.apache.juneau.common.internal.*;
18  
19  /**
20   * Encapsulates multiple collections so they can be iterated over as if they were all part of the same collection.
21   *
22   * <h5 class='section'>See Also:</h5><ul>
23  
24   * </ul>
25   *
26   * @param <E> The object type of this set.
27   */
28  public class MultiSet<E> extends AbstractSet<E> {
29  
30  	/** Inner collections. */
31  	final Collection<E>[] l;
32  
33  	/**
34  	 * Create a new Set that consists as a coalesced set of the specified collections.
35  	 *
36  	 * @param c Zero or more collections to add to this set.
37  	 */
38  	@SafeVarargs
39  	public MultiSet(Collection<E>...c) {
40  		Utils.assertArgNotNull("c", c);
41  		for (Collection<E> cc : c)
42  			Utils.assertArgNotNull("c", cc);
43  		l = c;
44  	}
45  
46  	/**
47  	 * Iterates over all entries in all collections.
48  	 */
49  	@Override /* Set */
50  	public Iterator<E> iterator() {
51  		return new Iterator<>() {
52  			int i = 0;
53  			Iterator<E> i2 = (l.length > 0 ? l[i++].iterator() : null);
54  
55  			@Override /* Iterator */
56  			public boolean hasNext() {
57  				if (i2 == null)
58  					return false;
59  				if (i2.hasNext())
60  					return true;
61  				for (int j = i; j < l.length; j++)
62  					if (l[j].size() > 0)
63  						return true;
64  				return false;
65  			}
66  
67  			@Override /* Iterator */
68  			public E next() {
69  				if (i2 == null)
70  					throw new NoSuchElementException();
71  				while (! i2.hasNext()) {
72  					if (i >= l.length)
73  						throw new NoSuchElementException();
74  					i2 = l[i++].iterator();
75  				}
76  				return i2.next();
77  			}
78  
79  			@Override /* Iterator */
80  			public void remove() {
81  				if (i2 == null)
82  					throw new NoSuchElementException();
83  				i2.remove();
84  			}
85  		};
86  	}
87  
88  	/**
89  	 * Enumerates over all entries in all collections.
90  	 *
91  	 * @return An enumeration wrapper around this set.
92  	 */
93  	public Enumeration<E> enumerator() {
94  		return Collections.enumeration(this);
95  	}
96  
97  	@Override /* Set */
98  	public int size() {
99  		int i = 0;
100 		for (Collection<E> c : l)
101 			i += c.size();
102 		return i;
103 	}
104 }
