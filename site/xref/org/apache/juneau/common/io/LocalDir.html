28  /**
29   * Represents a directory that can be located either on the classpath or in the file system.
30   *
31   * <p>
32   * This class provides a unified interface for working with directories regardless of their location,
33   * allowing code to transparently access files within directories from either the classpath (as resources)
34   * or the file system. This is particularly useful in applications that need to work with directory-based
35   * resources in both development (file system) and production (packaged JAR) environments.
36   *
37   * <h5 class='section'>Features:</h5>
38   * <ul class='spaced-list'>
39   * 	<li>Unified directory access - works with both classpath resources and file system directories
40   * 	<li>File resolution - can resolve files within the directory using relative paths
41   * 	<li>Transparent resolution - automatically resolves files based on construction parameters
42   * 	<li>Immutable - directory location cannot be changed after construction
43   * </ul>
44   *
45   * <h5 class='section'>Use Cases:</h5>
46   * <ul class='spaced-list'>
47   * 	<li>Accessing template directories that may be on classpath or file system
48   * 	<li>Loading configuration files from directories in both development and production
49   * 	<li>Resolving resources within package directories
50   * 	<li>Applications that need to support both embedded and external directory access
51   * </ul>
52   *
53   * <h5 class='section'>Usage:</h5>
54   * <p class='bjava'>
55   * 	<jc>// Classpath directory</jc>
56   * 	LocalDir <jv>classpathDir</jv> = <jk>new</jk> LocalDir(MyClass.<jk>class</jk>, <js>"templates"</js>);
57   * 	LocalFile <jv>file</jv> = <jv>classpathDir</jv>.resolve(<js>"index.html"</js>);
58   *
59   * 	<jc>// File system directory</jc>
60   * 	LocalDir <jv>fsDir</jv> = <jk>new</jk> LocalDir(Paths.get(<js>"/var/config"</js>));
61   * 	LocalFile <jv>file2</jv> = <jv>fsDir</jv>.resolve(<js>"app.properties"</js>);
62   *
63   * 	<jc>// Package directory (null or empty path)</jc>
64   * 	LocalDir <jv>packageDir</jv> = <jk>new</jk> LocalDir(MyClass.<jk>class</jk>, <jk>null</jk>);
65   * 	LocalFile <jv>file3</jv> = <jv>packageDir</jv>.resolve(<js>"resource.txt"</js>);
66   * </p>
67   *
68   * <h5 class='section'>Path Resolution:</h5>
69   * <p>
70   * The {@link #resolve(String)} method resolves files within the directory using relative paths.
71   * For classpath directories, the path resolution follows Java resource path conventions:
72   * <ul class='spaced-list'>
73   * 	<li><jk>null</jk> or empty string - resolves relative to the class's package
74   * 	<li>Absolute path (starts with <js>'/'</js>) - resolves relative to classpath root
75   * 	<li>Relative path - resolves relative to the specified classpath path
76   * </ul>
77   *
78   * <h5 class='section'>Thread Safety:</h5>
79   * <p>
80   * This class is immutable and therefore thread-safe. Multiple threads can safely access a LocalDir
81   * instance concurrently without synchronization.
82   *
83   * <h5 class='section'>See Also:</h5><ul>
84   * 	<li class='jc'>{@link LocalFile} - File counterpart for individual file access
85   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauCommonIO">juneau-common-io</a>
86   * </ul>
87   */
88  public class LocalDir {
89  
90  	/**
91  	 * Validates that the specified classpath resource exists and is a file.
92  	 * Note that the behavior of Class.getResource(path) is different when pointing to directories on the classpath.
93  	 * When packaged as a jar, calling Class.getResource(path) on a directory returns null.
94  	 * When unpackaged, calling Class.getResource(path) on a directory returns a URL starting with "file:".
95  	 * We perform a test to make the behavior the same regardless of whether we're packaged or not.
96  	 */
97  	private static boolean isClasspathFile(URL url) {
98  		return safeSupplier(() -> {
99  			if (url == null)
100 				return false;
101 			var uri = url.toURI();
102 			if (uri.toString().startsWith("file:"))
103 				if (Files.isDirectory(Paths.get(uri)))
104 					return false;
105 			return true;
106 		});
107 	}

109 	private final Class<?> clazz;
110 	private final String clazzPath;
111 	private final Path path;
112 
113 	private final int hashCode;

115 	/**
116 	 * Constructor for classpath directory.
117 	 *
118 	 * <p>
119 	 * Creates a LocalDir that references a directory on the classpath, relative to the specified class.
120 	 * The path resolution follows Java resource path conventions.
121 	 *
122 	 * <h5 class='section'>Path Examples:</h5>
123 	 * <p class='bjava'>
124 	 * 	<jc>// Package directory (null or empty)</jc>
125 	 * 	LocalDir <jv>dir1</jv> = <jk>new</jk> LocalDir(MyClass.<jk>class</jk>, <jk>null</jk>);
126 	 * 	<jc>// Resolves files in same package as MyClass</jc>
127 	 *
128 	 * 	<jc>// Absolute path from classpath root</jc>
129 	 * 	LocalDir <jv>dir2</jv> = <jk>new</jk> LocalDir(MyClass.<jk>class</jk>, <js>"/com/example/templates"</js>);
130 	 * 	<jc>// Resolves files from classpath root</jc>
131 	 *
132 	 * 	<jc>// Relative path from class package</jc>
133 	 * 	LocalDir <jv>dir3</jv> = <jk>new</jk> LocalDir(MyClass.<jk>class</jk>, <js>"templates"</js>);
134 	 * 	<jc>// Resolves files relative to MyClass package</jc>
135 	 * </p>
136 	 *
137 	 * @param clazz The class used to retrieve resources. Must not be <jk>null</jk>.
138 	 * @param clazzPath The subpath. Can be any of the following:
139 	 *                  <ul>
140 	 *                  	<li><jk>null</jk> or an empty string - Package of the class
141 	 *                  	<li>Absolute path (starts with <js>'/'</js>) - Relative to root package
142 	 *                  	<li>Relative path (does not start with <js>'/'</js>) - Relative to class package
143 	 *                  </ul>
144 	 */
145 	public LocalDir(Class<?> clazz, String clazzPath) {
146 		this.clazz = assertArgNotNull("clazz", clazz);
147 		this.clazzPath = "/".equals(clazzPath) ? "/" : StringUtils.nullIfEmpty(trimTrailingSlashes(clazzPath));
148 		this.path = null;
149 		this.hashCode = HashCode.of(clazz, clazzPath);
150 	}
151 
152 	/**
153 	 * Constructor for file system directory.
154 	 *
155 	 * <p>
156 	 * Creates a LocalDir that references a directory on the file system using a {@link Path}.
157 	 *
158 	 * <h5 class='section'>Example:</h5>
159 	 * <p class='bjava'>
160 	 * 	<jc>// Absolute path</jc>
161 	 * 	LocalDir <jv>dir1</jv> = <jk>new</jk> LocalDir(Paths.get(<js>"/var/config"</js>));
162 	 *
163 	 * 	<jc>// Relative path</jc>
164 	 * 	LocalDir <jv>dir2</jv> = <jk>new</jk> LocalDir(Paths.get(<js>"data/templates"</js>));
165 	 *
166 	 * 	<jc>// From File object</jc>
167 	 * 	File <jv>f</jv> = <jk>new</jk> File(<js>"output"</js>);
168 	 * 	LocalDir <jv>dir3</jv> = <jk>new</jk> LocalDir(<jv>f</jv>.toPath());
169 	 * </p>
170 	 *
171 	 * @param path Filesystem directory location. Must not be <jk>null</jk>.
172 	 */
173 	public LocalDir(Path path) {
174 		this.clazz = null;
175 		this.clazzPath = null;
176 		this.path = assertArgNotNull("path", path);
177 		this.hashCode = path.hashCode();
178 	}

183 	}

190 	/**
191 	 * Resolves a file within this directory using the specified relative path.
192 	 *
193 	 * <p>
194 	 * This method attempts to locate a file within the directory. If the file exists and is readable,
195 	 * a {@link LocalFile} instance is returned. If the file does not exist or is not accessible,
196 	 * <jk>null</jk> is returned.
197 	 *
198 	 * <p>
199 	 * For classpath directories, the path is resolved according to the directory's path type:
200 	 * <ul class='spaced-list'>
201 	 * 	<li>Package directory (null clazzPath) - path is kept relative
202 	 * 	<li>Root directory ("/") - path is made absolute if not already
203 	 * 	<li>Absolute clazzPath - resolved path is made absolute
204 	 * 	<li>Relative clazzPath - resolved path remains relative
205 	 * </ul>
206 	 *
207 	 * <h5 class='section'>Example:</h5>
208 	 * <p class='bjava'>
209 	 * 	LocalDir <jv>dir</jv> = <jk>new</jk> LocalDir(MyClass.<jk>class</jk>, <js>"templates"</js>);
210 	 *
211 	 * 	<jc>// Resolve file in directory</jc>
212 	 * 	LocalFile <jv>file</jv> = <jv>dir</jv>.resolve(<js>"index.html"</js>);
213 	 * 	<jk>if</jk> (<jv>file</jv> != <jk>null</jk>) {
214 	 * 		InputStream <jv>is</jv> = <jv>file</jv>.read();
215 	 * 	}
216 	 *
217 	 * 	<jc>// Resolve file in subdirectory</jc>
218 	 * 	LocalFile <jv>file2</jv> = <jv>dir</jv>.resolve(<js>"pages/about.html"</js>);
219 	 * </p>
220 	 *
221 	 * <h5 class='section'>Security Note:</h5>
222 	 * <p>
223 	 * This method does not perform path validation or security checks (e.g., checking for path
224 	 * traversal attacks or malformed values). The caller is responsible for ensuring the path
225 	 * is safe and valid.
226 	 *
227 	 * @param path The relative path to the file to resolve within this directory.
228 	 *             Must be a non-null relative path.
229 	 * @return A {@link LocalFile} instance if the file exists and is readable, or <jk>null</jk> if it does not.
230 	 */
231 	public LocalFile resolve(String path) {
232 		assertArgNotNull("path", path);
233 		if (nn(clazz)) {
234 			String p;
235 			if (clazzPath == null) {
236 				// Relative to class package - keep path relative
237 				p = path;
238 			} else if ("/".equals(clazzPath)) {
239 				// Root - make path absolute
240 				p = path.startsWith("/") ? path : "/" + path;
241 			} else if (clazzPath.startsWith("/")) {
242 				// Absolute clazzPath - make resolved path absolute
243 				p = clazzPath + '/' + path;
244 			} else {
245 				// Relative clazzPath - keep resolved path relative
246 				p = clazzPath + '/' + path;
247 			}
248 			if (isClasspathFile(clazz.getResource(p)))
249 				return new LocalFile(clazz, p);
250 		} else {
251 			var p = this.path.resolve(path);
252 			if (Files.isReadable(p) && ! Files.isDirectory(p))
253 				return new LocalFile(p);
254 		}
255 		return null;
256 	}
257 
258 	@Override /* Overridden from Object */
259 	public String toString() {
260 		if (clazz == null)
261 			return path.toString();
262 		return cn(clazz) + ":" + clazzPath;
263 	}
