27  /**
28   * An implementation of an annotation that can be dynamically applied to classes, methods, fields, and constructors at runtime.
29   *
30   * <p>
31   * This class extends {@link AnnotationObject} to provide support for annotations that have an <c>on</c> property which
32   * allows them to be targeted to specific program elements based on their fully-qualified names.
33   *
34   * <h5 class='section'>Overview:</h5>
35   * <p>
36   * Applied annotations allow you to define annotation properties in external configuration or programmatically at runtime,
37   * rather than requiring them to be declared directly on classes, methods, fields, or constructors at compile-time.
38   *
39   * <p>
40   * This is particularly useful for:
41   * <ul class='spaced-list'>
42   * 	<li>Configuration-driven behavior - Define annotation properties in config files without modifying source code
43   * 	<li>Dynamic application - Apply annotations to third-party classes where you can't modify the source
44   * 	<li>Runtime customization - Change annotation properties based on runtime conditions
45   * 	<li>Centralized configuration - Define annotation properties for multiple classes in one place
46   * </ul>
47   *
48   * <h5 class='section'>Targeting with <c>on</c>:</h5>
49   * <p>
50   * The base {@link Builder#on(String...)} method accepts fully-qualified names in the following formats:
51   * <ul class='spaced-list'>
52   * 	<li><js>"com.example.MyClass"</js> - Target a specific class
53   * 	<li><js>"com.example.MyClass.myMethod"</js> - Target a specific method
54   * 	<li><js>"com.example.MyClass.myField"</js> - Target a specific field
55   * 	<li><js>"com.example.MyClass(java.lang.String,int)"</js> - Target a specific constructor
56   * </ul>
57   *
58   * <h5 class='section'>Builder Variants:</h5>
59   * <p>
60   * This class provides specialized builder variants for different targeting scenarios:
61   * <ul class='javatree'>
62   * 	<li class='jc'>{@link Builder} - Base builder with string-based targeting via {@link Builder#on(String...)}
63   * 	<li class='jc'>{@link BuilderT} - For targeting classes
64   * 		<ul>
65   * 			<li class='jm'>{@link BuilderT#on(Class...) on(Class...)} - Target by class, stored as strings
66   * 			<li class='jm'>{@link BuilderT#on(ClassInfo...) on(ClassInfo...)} - Target by ClassInfo, stored as strings
67   * 			<li class='jm'>{@link BuilderT#onClass(Class...) onClass(Class...)} - Target by class, stored as Class objects
68   * 			<li class='jm'>{@link BuilderT#onClass(ClassInfo...) onClass(ClassInfo...)} - Target by ClassInfo, stored as Class objects
69   * 		</ul>
70   * 	<li class='jc'>{@link BuilderM} - For targeting methods
71   * 		<ul>
72   * 			<li class='jm'>{@link BuilderM#on(Method...) on(Method...)} - Target by Method object
73   * 			<li class='jm'>{@link BuilderM#on(MethodInfo...) on(MethodInfo...)} - Target by MethodInfo object
74   * 		</ul>
75   * 	<li class='jc'>{@link BuilderC} - For targeting constructors
76   * 		<ul>
77   * 			<li class='jm'>{@link BuilderC#on(Constructor...) on(Constructor...)} - Target by Constructor object
78   * 			<li class='jm'>{@link BuilderC#on(ConstructorInfo...) on(ConstructorInfo...)} - Target by ConstructorInfo object
79   * 		</ul>
80   * 	<li class='jc'>{@link BuilderMF} - For targeting methods and fields
81   * 		<ul>
82   * 			<li class='jm'>{@link BuilderMF#on(Method...) on(Method...)} - Target by Method object
83   * 			<li class='jm'>{@link BuilderMF#on(MethodInfo...) on(MethodInfo...)} - Target by MethodInfo object
84   * 			<li class='jm'>{@link BuilderMF#on(Field...) on(Field...)} - Target by Field object
85   * 			<li class='jm'>{@link BuilderMF#on(FieldInfo...) on(FieldInfo...)} - Target by FieldInfo object
86   * 		</ul>
87   * 	<li class='jc'>{@link BuilderTM} - For targeting classes and methods
88   * 	<li class='jc'>{@link BuilderTMF} - For targeting classes, methods, and fields
89   * 	<li class='jc'>{@link BuilderTMFC} - For targeting classes, methods, fields, and constructors
90   * </ul>
91   *
92   * <h5 class='section'>Example:</h5>
93   * <p class='bjava'>
94   * 	<jc>// Create a Bean annotation that applies to MyClass</jc>
95   * 	BeanAnnotation <jv>annotation</jv> = BeanAnnotation
96   * 		.<jsm>create</jsm>(MyClass.<jk>class</jk>)
97   * 		.sort(<jk>true</jk>)
98   * 		.build();
99   *
100  * 	<jc>// Or target by string name</jc>
101  * 	BeanAnnotation <jv>annotation2</jv> = BeanAnnotation
102  * 		.<jsm>create</jsm>()
103  * 		.on(<js>"com.example.MyClass"</js>)
104  * 		.sort(<jk>true</jk>)
105  * 		.build();
106  * </p>
107  *
108  * <h5 class='section'>Notes:</h5>
109  * <ul class='spaced-list'>
110  * 	<li>The {@link #on()} method returns the string-based targets
111  * 	<li>Subclasses may provide additional {@code onClass()}, {@code onMethod()}, etc. methods for type-safe access
112  * 	<li>All builder methods return the builder for method chaining
113  * 	<li>Hashcode is calculated lazily on first access and then cached for performance
114  * </ul>
115  *
116  * <h5 class='section'>See Also:</h5><ul>
117  * 	<li class='link'><a class="doclink" href="../../../../../overview-summary.html#juneau-common.Annotations">Overview &gt; juneau-common &gt; Annotations</a>
118  * </ul>
119  */
120 public class AppliedAnnotationObject extends AnnotationObject {

127 	 * Builder for {@link AppliedAnnotationObject} objects.
128 	 *
129 	 * <p>
130 	 * Provides string-based targeting via the {@link #on(String...)} method.
131 	 *
132 	 * <h5 class='section'>Example:</h5>
133 	 * <p class='bjava'>
134 	 * 	<jc>// Target specific classes and methods by name</jc>
135 	 * 	MyAnnotation <jv>annotation</jv> = MyAnnotation
136 	 * 		.<jsm>create</jsm>()
137 	 * 		.on(<js>"com.example.MyClass"</js>)
138 	 * 		.on(<js>"com.example.MyClass.myMethod"</js>)
139 	 * 		.build();
140 	 * </p>
141 	 *
142 	 * <h5 class='section'>See Also:</h5><ul>
143 	 * 	<li class='jc'>{@link BuilderT} - For type-safe class targeting
144 	 * 	<li class='jc'>{@link BuilderM} - For method targeting
145 	 * 	<li class='jc'>{@link BuilderC} - For constructor targeting
146 	 * 	<li class='jc'>{@link BuilderMF} - For method and field targeting
147 	 * </ul>

151 		String[] on = {};
152 
153 		/**
154 		 * Constructor.
155 		 *
156 		 * @param annotationType The annotation type of the annotation implementation class.
157 		 */
158 		public Builder(Class<? extends Annotation> annotationType) {
159 			super(annotationType);
160 		}
161 
162 		/**
163 		 * The targets this annotation applies to.
164 		 *
165 		 * @param values The targets this annotation applies to.
166 		 * @return This object.
167 		 */
168 		public Builder on(String...values) {
169 			for (var v : values)
170 				on = addAll(on, v);
171 			return this;
172 		}

177 	 *
178 	 * <p>
179 	 * Adds type-safe class targeting methods to the base builder:
180 	 * <ul class='javatree'>
181 	 * 	<li class='jm'>{@link #on(Class...) on(Class...)} - Target by class, stored as strings in {@link AppliedAnnotationObject#on()}
182 	 * 	<li class='jm'>{@link #on(ClassInfo...) on(ClassInfo...)} - Target by ClassInfo, stored as strings
183 	 * 	<li class='jm'>{@link #onClass(Class...) onClass(Class...)} - Target by class, stored as Class objects in {@link AppliedOnClassAnnotationObject#onClass()}
184 	 * 	<li class='jm'>{@link #onClass(ClassInfo...) onClass(ClassInfo...)} - Target by ClassInfo, stored as Class objects
185 	 * </ul>
186 	 *
187 	 * <h5 class='section'>Example:</h5>
188 	 * <p class='bjava'>

190 	 * 	MyAnnotation <jv>annotation</jv> = MyAnnotation
191 	 * 		.<jsm>create</jsm>()
192 	 * 		.on(MyClass.<jk>class</jk>, MyOtherClass.<jk>class</jk>)  <jc>// Stored as strings</jc>
193 	 * 		.onClass(ThirdClass.<jk>class</jk>)  <jc>// Stored as Class object</jc>
194 	 * 		.build();
195 	 * </p>
196 	 *
197 	 * <h5 class='section'>Notes:</h5>
198 	 * <ul class='spaced-list'>
199 	 * 	<li>Use {@link #on(Class...)} when you want string-based matching (e.g., for configuration)
200 	 * 	<li>Use {@link #onClass(Class...)} when you need direct Class object references
201 	 * 	<li>Both can be used together on the same builder
202 	 * </ul>

611 	 * Builder for applied annotations targeting classes, methods, fields, and constructors.
612 	 *
613 	 * <p>
614 	 * The most comprehensive builder variant, combining all targeting capabilities from {@link BuilderT}, {@link BuilderM},
615 	 * field targeting, and {@link BuilderC}, providing complete targeting for all program elements.
616 	 *
617 	 * <h5 class='section'>Example:</h5>
