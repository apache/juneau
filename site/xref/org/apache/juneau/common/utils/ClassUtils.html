44  	/**
45  	 * Determines whether the specified collection supports modification operations (e.g., {@code add()}, {@code remove()}).
46  	 *
47  	 * <p>
48  	 * This method performs a heuristic check based on the collection's class name to determine if it's likely modifiable.
49  	 * It checks whether the class name contains indicators of immutability such as "Immutable", "Unmodifiable",
50  	 * or "Arrays$ArrayList" (which represents the unmodifiable list returned by {@link Arrays#asList(Object...)}).
51  	 *
52  	 * <p>
53  	 * Results are cached for performance, so repeated calls for the same collection type are very fast.
54  	 *
55  	 * <h5 class='section'>Examples:</h5>
56  	 * <p class='bjava'>
57  	 * 	<jc>// Modifiable collections</jc>
58  	 * 	canAddTo(<jk>new</jk> ArrayList&lt;&gt;());        <jc>// true</jc>
59  	 * 	canAddTo(<jk>new</jk> LinkedList&lt;&gt;());       <jc>// true</jc>
60  	 * 	canAddTo(<jk>new</jk> HashSet&lt;&gt;());          <jc>// true</jc>
61  	 *
62  	 * 	<jc>// Unmodifiable collections</jc>
63  	 * 	canAddTo(Collections.unmodifiableList(...));    <jc>// false</jc>
64  	 * 	canAddTo(Collections.unmodifiableSet(...));     <jc>// false</jc>
65  	 * 	canAddTo(Arrays.asList(<js>"a"</js>, <js>"b"</js>));               <jc>// false</jc>
66  	 * 	canAddTo(List.of(<js>"a"</js>, <js>"b"</js>));                     <jc>// false (ImmutableCollections)</jc>
67  	 * </p>
68  	 *
69  	 * <p>
70  	 * <b>Note:</b> This is a heuristic check based on naming conventions. It does not attempt to actually
71  	 * modify the collection, so it never throws exceptions. However, it may produce false positives for
72  	 * custom collection implementations with misleading names.
73  	 *
74  	 * @param value The collection to check. Must not be <jk>null</jk>.
75  	 * @return <jk>true</jk> if the collection is likely modifiable, <jk>false</jk> if it's likely unmodifiable.
76  	 * @throws IllegalArgumentException If value is <jk>null</jk>.
77  	 */
78  	public static boolean canAddTo(Collection<?> value) {
79  		assertArgNotNull("value", value);
80  		return canAddTo(value.getClass());
81  	}

156 	 * Extracts generic type parameter mappings from a class's superclass declaration and adds them to a type map.
157 	 *
158 	 * <p>
159 	 * This method builds a mapping between type variables (e.g., <c>T</c>, <c>K</c>, <c>V</c>) and their actual
160 	 * type arguments in the inheritance hierarchy. It's used to resolve generic type parameters when navigating
161 	 * through class hierarchies.
162 	 *
163 	 * <h5 class='section'>Example:</h5>
164 	 * <p class='bjava'>
165 	 * 	<jc>// Given these classes:</jc>
166 	 * 	<jk>class</jk> MyList&lt;T&gt; <jk>extends</jk> ArrayList&lt;T&gt; {}
167 	 * 	<jk>class</jk> StringList <jk>extends</jk> MyList&lt;String&gt; {}
168 	 *
169 	 * 	<jc>// When called on StringList, this would extract:</jc>
170 	 * 	<jc>//   T (from MyList) -&gt; String</jc>
171 	 * </p>
172 	 *
173 	 * <p>
174 	 * The method handles transitive type mappings by checking if an actual type argument is itself a type variable
175 	 * that's already been mapped, and resolving it to its final type.
176 	 *
177 	 * @param typeMap
178 	 * 	The map to populate with type variable to actual type mappings.
179 	 * 	<br>Existing entries are used to resolve transitive mappings.
180 	 * @param c
181 	 * 	The class whose generic superclass should be examined for type parameters.
182 	 */
183 	public static void extractTypes(Map<Type,Type> typeMap, Class<?> c) {
184 		var gs = c.getGenericSuperclass();
185 		if (gs instanceof ParameterizedType gs2) {
186 			var typeParameters = ((Class<?>)gs2.getRawType()).getTypeParameters();
187 			var actualTypeArguments = gs2.getActualTypeArguments();
188 			for (var i = 0; i < typeParameters.length; i++) {
189 				if (typeMap.containsKey(actualTypeArguments[i]))
190 					actualTypeArguments[i] = typeMap.get(actualTypeArguments[i]);
191 				typeMap.put(typeParameters[i], actualTypeArguments[i]);
192 			}
193 		}
194 	}

209 	/**
210 	 * Matches arguments to a list of parameter types.
211 	 *
212 	 * <p>
213 	 * This method intelligently matches a variable number of arguments to a fixed set of parameter types,
214 	 * handling cases where arguments may be provided in a different order, or where some arguments are
215 	 * missing or extra arguments are provided. This is particularly useful for reflective method/constructor
216 	 * invocation where parameter order flexibility is desired.
217 	 *
218 	 * <h5 class='section'>Matching Rules:</h5>
219 	 * <ul>
220 	 * 	<li>If arguments already match parameter types in order and count, they are returned as-is (fast path)
221 	 * 	<li>Otherwise, each parameter type is matched with the first compatible argument
222 	 * 	<li>Extra arguments are ignored
223 	 * 	<li>Missing parameters are left as <jk>null</jk>
224 	 * 	<li>Primitive types are automatically matched with their wrapper equivalents
225 	 * 	<li>Type hierarchy is respected (subclasses match parent parameters)
226 	 * </ul>
227 	 *
228 	 * <h5 class='section'>Examples:</h5>
229 	 * <p class='bjava'>
230 	 * 	<jc>// Already in correct order - fast path returns original array</jc>
231 	 * 	Class&lt;?&gt;[] types = {String.<jk>class</jk>, Integer.<jk>class</jk>};
232 	 * 	Object[] args = {<js>"hello"</js>, 42};
233 	 * 	Object[] result = getMatchingArgs(types, args);
234 	 * 	<jc>// Returns: ["hello", 42]</jc>
235 	 *
236 	 * 	<jc>// Arguments in wrong order - method reorders them</jc>
237 	 * 	Class&lt;?&gt;[] types = {Integer.<jk>class</jk>, String.<jk>class</jk>};
238 	 * 	Object[] args = {<js>"hello"</js>, 42};
239 	 * 	Object[] result = getMatchingArgs(types, args);
240 	 * 	<jc>// Returns: [42, "hello"]</jc>
241 	 *
242 	 * 	<jc>// Extra arguments are ignored</jc>
243 	 * 	Class&lt;?&gt;[] types = {String.<jk>class</jk>};
244 	 * 	Object[] args = {<js>"hello"</js>, 42, <jk>true</jk>};
245 	 * 	Object[] result = getMatchingArgs(types, args);
246 	 * 	<jc>// Returns: ["hello"]</jc>
247 	 *
248 	 * 	<jc>// Missing arguments become null</jc>
249 	 * 	Class&lt;?&gt;[] types = {String.<jk>class</jk>, Integer.<jk>class</jk>, Boolean.<jk>class</jk>};
250 	 * 	Object[] args = {<js>"hello"</js>};
251 	 * 	Object[] result = getMatchingArgs(types, args);
252 	 * 	<jc>// Returns: ["hello", null, null]</jc>
253 	 *
254 	 * 	<jc>// Handles primitive types and their wrappers</jc>
255 	 * 	Class&lt;?&gt;[] types = {<jk>int</jk>.<jk>class</jk>, String.<jk>class</jk>};
256 	 * 	Object[] args = {<js>"hello"</js>, 42};  <jc>// Integer object matches int.class</jc>
257 	 * 	Object[] result = getMatchingArgs(types, args);
258 	 * 	<jc>// Returns: [42, "hello"]</jc>
259 	 *
260 	 * 	<jc>// Respects type hierarchy - subclasses match parent types</jc>
261 	 * 	Class&lt;?&gt;[] types = {Number.<jk>class</jk>, String.<jk>class</jk>};
262 	 * 	Object[] args = {<js>"hello"</js>, 42};  <jc>// Integer extends Number</jc>
263 	 * 	Object[] result = getMatchingArgs(types, args);
264 	 * 	<jc>// Returns: [42, "hello"]</jc>
265 	 * </p>
266 	 *
267 	 * <p>
268 	 * This method is used internally by {@link ClassInfo}, {@link MethodInfo}, and {@link ConstructorInfo}
269 	 * to provide flexible parameter matching during reflective invocation.
270 	 *
271 	 * @param paramTypes The parameter types to match against. Must not be <jk>null</jk>.
272 	 * @param args The arguments to match to the parameter types. Can be empty or contain <jk>null</jk> values.
273 	 * @return
274 	 * 	An array of arguments matched to the parameter types. The returned array will always have
275 	 * 	the same length as {@code paramTypes}. Returns the original {@code args} array if it already
276 	 * 	matches (fast path optimization).
277 	 */
278 	public static Object[] getMatchingArgs(Class<?>[] paramTypes, Object...args) {
279 		assertArgNotNull("args", args);
280 		var needsShuffle = paramTypes.length != args.length;
281 		if (! needsShuffle) {
282 			for (var i = 0; i < paramTypes.length; i++) {
283 				if (! paramTypes[i].isInstance(args[i]))
284 					needsShuffle = true;
285 			}
286 		}
287 		if (! needsShuffle)
288 			return args;
289 		var params = new Object[paramTypes.length];
290 		for (var i = 0; i < paramTypes.length; i++) {
291 			var pt = info(paramTypes[i]).getWrapperIfPrimitive();
292 			for (var arg : args) {
293 				if (nn(arg) && pt.isParentOf(arg.getClass())) {
294 					params[i] = arg;
295 					break;
296 				}
297 			}
298 		}
299 		return params;
300 	}

302 	/**
303 	 * Returns the generic parameter type at the specified index for a class that extends a parameterized type.
304 	 *
305 	 * @param c The class to examine.
306 	 * @param index The zero-based index of the parameter to retrieve.
307 	 * @param pt The parameterized superclass or interface.
308 	 * @return The parameter type at the specified index.
309 	 * @throws IllegalArgumentException If the class is not a subclass of the parameterized type or if the index is invalid.
310 	 */
311 	public static Class<?> getParameterType(Class<?> c, int index, Class<?> pt) {
312 		assertArgsNotNull("pt", pt, "c", c);
313 
314 		// We need to make up a mapping of type names.
315 		var typeMap = new HashMap<Type,Type>();
316 		var cc = c;
317 		while (pt != cc.getSuperclass()) {
318 			extractTypes(typeMap, cc);
319 			cc = cc.getSuperclass();
320 			assertArg(nn(cc), "Class ''{0}'' is not a subclass of parameterized type ''{1}''", cns(c), cns(pt));
321 		}
322 
323 		var gsc = cc.getGenericSuperclass();
324 
325 		assertArg(gsc instanceof ParameterizedType, "Class ''{0}'' is not a parameterized type", cns(pt));
326 
327 		var cpt = (ParameterizedType)gsc;
328 		var atArgs = cpt.getActualTypeArguments();
329 		assertArg(index < atArgs.length, "Invalid type index. index={0}, argsLength={1}", index, atArgs.length);
330 		var actualType = cpt.getActualTypeArguments()[index];
331 
332 		if (typeMap.containsKey(actualType))
333 			actualType = typeMap.get(actualType);
334 
335 		if (actualType instanceof Class actualType2) {
336 			return actualType2;
337 
338 		}
339 		if (actualType instanceof GenericArrayType actualType2) {
340 			var gct = actualType2.getGenericComponentType();
341 			if (gct instanceof ParameterizedType gct2)
342 				return Array.newInstance((Class<?>)gct2.getRawType(), 0).getClass();
343 		} else if (actualType instanceof TypeVariable<?> actualType3) {
344 			var nestedOuterTypes = new LinkedList<Class<?>>();
345 			for (var ec = cc.getEnclosingClass(); nn(ec); ec = ec.getEnclosingClass()) {
346 				var outerClass = cc.getClass();
347 				nestedOuterTypes.add(outerClass);
348 				var outerTypeMap = new HashMap<Type,Type>();
349 				extractTypes(outerTypeMap, outerClass);
350 				for (var entry : outerTypeMap.entrySet()) {
351 					var key = entry.getKey();
352 					var value = entry.getValue();
353 					if ((key instanceof TypeVariable<?> key2) && (key2.getName().equals(actualType3.getName()) && isInnerClass(key2.getGenericDeclaration(), actualType3.getGenericDeclaration()))) {
354 						if (value instanceof Class<?> value2)
355 							return value2;
356 						actualType3 = (TypeVariable<?>)entry.getValue();
357 					}
358 				}
359 			}
360 		} else if (actualType instanceof ParameterizedType actualType2) {
361 			return (Class<?>)actualType2.getRawType();
362 		}
363 		throw illegalArg("Could not resolve variable ''{0}'' to a type.", actualType.getTypeName());
364 	}

366 	/**
367 	 * Attempts to unwrap a proxy object and return the underlying "real" class.
368 	 *
369 	 * <p>
370 	 * This method handles several common proxy types used in Java applications:
371 	 * <ul>
372 	 * 	<li><b>JDK Dynamic Proxies</b> - Created via {@link Proxy#newProxyInstance}
373 	 * 	<li><b>CGLIB Proxies</b> - Created by Spring's CGLIB enhancer (class name contains <js>"$$EnhancerBySpringCGLIB$$"</js>)
374 	 * 	<li><b>Javassist Proxies</b> - Created by Javassist proxy factory (class name contains <js>"_$$_javassist"</js> or <js>"_$$_jvst"</js>)
375 	 * 	<li><b>ByteBuddy Proxies</b> - Created by ByteBuddy (class name contains <js>"$ByteBuddy$"</js>)
376 	 * </ul>
377 	 *
378 	 * <h5 class='section'>Example:</h5>
379 	 * <p class='bjava'>
380 	 * 	<jc>// Get the real class behind a Spring proxy</jc>
381 	 * 	<ja>@Service</ja>
382 	 * 	<jk>class</jk> MyService {}
383 	 *
384 	 * 	MyService proxy = applicationContext.getBean(MyService.<jk>class</jk>);
385 	 * 	Class&lt;?&gt; realClass = ClassUtils.<jsm>getProxyFor</jsm>(proxy);
386 	 * 	<jc>// Returns MyService.class instead of the CGLIB proxy class</jc>
387 	 * </p>
388 	 *
389 	 * <h5 class='section'>Notes:</h5>
390 	 * <ul class='spaced-list'>
391 	 * 	<li>For JDK dynamic proxies, returns the first interface implemented by the proxy
392 	 * 	<li>For CGLIB/Javassist/ByteBuddy proxies, returns the superclass
393 	 * 	<li>For Spring CGLIB proxies with a <c>getTargetClass()</c> method, invokes that method for more accurate results
394 	 * 	<li>Returns <jk>null</jk> if the object is not a recognized proxy type
395 	 * </ul>
396 	 *
397 	 * @param o The object to unwrap. Can be <jk>null</jk>.
398 	 * @return
399 	 * 	The underlying class, or <jk>null</jk> if:
400 	 * 	<ul>
401 	 * 		<li>The object is <jk>null</jk>
402 	 * 		<li>The object is not a recognized proxy type
403 	 * 		<li>The proxy cannot be unwrapped
404 	 * 	</ul>
405 	 */
406 	public static Class<?> getProxyFor(Object o) {
407 		if (o == null)
408 			return null;
409 
410 		var c = o.getClass();
411 		var s = c.getName();
412 
413 		// Quick exit: Most classes aren't proxies, and proxies typically have '$' in their name
414 		if (s.indexOf('$') == -1)
415 			return null;
416 
417 		// JDK Dynamic Proxy: Created via Proxy.newProxyInstance()
418 		// These implement interfaces and are instances of java.lang.reflect.Proxy
419 		if (Proxy.isProxyClass(c)) {
420 			var interfaces = c.getInterfaces();
421 			return interfaces.length > 0 ? interfaces[0] : null;
422 		}
423 
424 		// CGLIB Proxy: Spring's CGLIB enhancer creates subclasses
425 		// Pattern: com.example.MyClass$$EnhancerBySpringCGLIB$$abc123
426 		if (s.contains("$$EnhancerBySpringCGLIB$$")) {
427 			// Try to invoke getTargetClass() if available (Spring specific)
428 			var v = Value.<Class<?>>empty();
429 			info(c).getPublicMethods().stream().filter(m -> m.hasName("getTargetClass") && m.getParameterCount() == 0 && m.hasReturnType(Class.class)).forEach(m -> safe(() -> v.set(m.invoke(o))));
430 			return v.isPresent() ? v.get() : c.getSuperclass();
431 		}
432 
433 		// Javassist Proxy: Created by Javassist ProxyFactory
434 		// Pattern: com.example.MyClass_$$_javassist_123 or com.example.MyClass_$$_jvst123
435 		// ByteBuddy Proxy: Created by ByteBuddy framework
436 		// Pattern: com.example.MyClass$ByteBuddy$abc123
437 		if (s.contains("_$$_javassist") || s.contains("_$$_jvst") || s.contains("$ByteBuddy$")) {
438 			return c.getSuperclass();
439 		}
440 
441 		// Not a recognized proxy type
442 		return null;
443 	}

445 	/**
446 	 * Checks if one generic declaration (typically a class) is an inner class of another.
447 	 *
448 	 * <p>
449 	 * This method walks up the enclosing class hierarchy of the potential inner class to determine if it's
450 	 * nested within the outer class at any level. It's used during generic type resolution to determine if
451 	 * type variables from different scopes refer to the same logical type parameter.
452 	 *
453 	 * <h5 class='section'>Example:</h5>
454 	 * <p class='bjava'>
455 	 * 	<jc>// Given these classes:</jc>
456 	 * 	<jk>class</jk> Outer&lt;T&gt; {
457 	 * 		<jk>class</jk> Inner&lt;T&gt; {
458 	 * 			<jc>// The T here could refer to either Outer.T or Inner.T</jc>
459 	 * 		}
460 	 * 	}
461 	 *
462 	 * 	<jc>// This method helps determine the relationship:</jc>
463 	 * 	isInnerClass(Outer.<jk>class</jk>, Inner.<jk>class</jk>);  <jc>// Returns true</jc>
464 	 * 	isInnerClass(Inner.<jk>class</jk>, Outer.<jk>class</jk>);  <jc>// Returns false</jc>
465 	 * </p>
466 	 *
467 	 * @param od
468 	 * 	The potential outer declaration (typically an outer class).
469 	 * @param id
470 	 * 	The potential inner declaration (typically an inner class).
471 	 * @return
472 	 * 	<jk>true</jk> if <c>id</c> is nested within <c>od</c> at any level, <jk>false</jk> otherwise.
473 	 * 	<br>Returns <jk>false</jk> if either parameter is not a <c>Class</c>.
474 	 */
475 	public static boolean isInnerClass(GenericDeclaration od, GenericDeclaration id) {
476 		if (od instanceof Class<?> oc && id instanceof Class<?> ic) {
477 			while (nn(ic = ic.getEnclosingClass()))
478 				if (ic == oc)
479 					return true;
480 		}
481 		return false;
482 	}

485 	 * Returns the generic parameter type of the Value type.
486 	 *
487 	 * @param t The type to find the parameter type of.
488 	 * @return The parameter type of the value, or <jk>null</jk> if the type is not a subclass of <c>Value</c>.
489 	 */
490 	public static Type getValueParameterType(Type t) {
491 		if (t instanceof ParameterizedType t2) {
492 			if (t2.getRawType() == Value.class) {

612 	/**
613 	 * Returns the simple qualified class name for the specified object.
614 	 *
615 	 * <p>
616 	 * This returns the simple class name including outer class names, but without the package.
617 	 * Inner class separators ($) are replaced with dots (.).
618 	 * Array types are properly formatted with brackets.
619 	 *
620 	 * <h5 class='section'>Examples:</h5>
621 	 * <p class='bjava'>
622 	 * 	<jc>// Regular classes</jc>
623 	 * 	simpleQualifiedClassName(String.<jk>class</jk>);                     <jc>// "String"</jc>
624 	 * 	simpleQualifiedClassName(<jk>new</jk> HashMap&lt;&gt;());                   <jc>// "HashMap"</jc>

626 	 * 	<jc>// Inner classes</jc>
627 	 * 	simpleQualifiedClassName(Map.Entry.<jk>class</jk>);                  <jc>// "Map.Entry"</jc>
628 	 * 	simpleQualifiedClassName(Outer.Inner.Deep.<jk>class</jk>);           <jc>// "Outer.Inner.Deep"</jc>
629 	 *
630 	 * 	<jc>// Primitives</jc>
631 	 * 	simpleQualifiedClassName(<jk>int</jk>.<jk>class</jk>);                         <jc>// "int"</jc>
632 	 * 	simpleQualifiedClassName(<jk>boolean</jk>.<jk>class</jk>);                     <jc>// "boolean"</jc>
633 	 *
634 	 * 	<jc>// Object arrays</jc>
635 	 * 	simpleQualifiedClassName(String[].<jk>class</jk>);                   <jc>// "String[]"</jc>
636 	 * 	simpleQualifiedClassName(Map.Entry[].<jk>class</jk>);                <jc>// "Map.Entry[]"</jc>
637 	 * 	simpleQualifiedClassName(String[][].<jk>class</jk>);                 <jc>// "String[][]"</jc>
638 	 *
639 	 * 	<jc>// Primitive arrays</jc>
640 	 * 	simpleQualifiedClassName(<jk>int</jk>[].<jk>class</jk>);                       <jc>// "int[]"</jc>
641 	 * 	simpleQualifiedClassName(<jk>boolean</jk>[][].<jk>class</jk>);                 <jc>// "boolean[][]"</jc>
642 	 *
643 	 * 	<jc>// Null</jc>
644 	 * 	simpleQualifiedClassName(<jk>null</jk>);                             <jc>// null</jc>
645 	 * </p>
646 	 *
647 	 * @param value The object to get the simple qualified class name for.
648 	 * @return The simple qualified name of the class or <jk>null</jk> if the value was null.
649 	 */
650 	public static String classNameSimpleQualified(Object value) {
651 		if (value == null)
652 			return null;
653 		var clazz = value instanceof Class<?> ? (Class<?>)value : value.getClass();
654 
655 		// Handle array types by recursively getting component type
656 		if (clazz.isArray()) {
657 			return classNameSimpleQualified(clazz.getComponentType()) + "[]";
658 		}
659 
660 		// Handle non-array types
661 		var className = clazz.getName();
662 		var lastDot = className.lastIndexOf('.');
663 		var simpleName = lastDot == -1 ? className : className.substring(lastDot + 1);
664 		return simpleName.replace('$', '.');
665 	}
