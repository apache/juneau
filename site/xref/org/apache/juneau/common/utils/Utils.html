95  	 * Compares two objects for equality.
96  	 *
97  	 * <p>
98  	 * Nulls are always considered less-than unless both are null.
99  	 *
100 	 * @param o1 Object 1.
101 	 * @param o2 Object 2.
102 	 * @return
103 	 * 	<c>-1</c>, <c>0</c>, or <c>1</c> if <c>o1</c> is less-than, equal, or greater-than <c>o2</c>.
104 	 *	<br><c>0</c> if objects are not of the same type or do not implement the {@link Comparable} interface.
105 	 */
106 	@SuppressWarnings({ "rawtypes", "unchecked" })
107 	public static int compare(Object o1, Object o2) {
108 		if (o1 == null) {
109 			if (o2 == null)
110 				return 0;
111 			return -1;
112 		} else if (o2 == null) {
113 			return 1;
114 		}
115 
116 		if (o1.getClass() == o2.getClass() && o1 instanceof Comparable)
117 			return ((Comparable)o1).compareTo(o2);
118 
119 		return 0;
120 	}

190 	 * Tests two objects for equality, gracefully handling nulls and arrays.
191 	 *
192 	 * @param <T> The value types.
193 	 * @param o1 Object 1.
194 	 * @param o2 Object 2.
195 	 * @return <jk>true</jk> if both objects are equal based on the {@link Object#equals(Object)} method.
196 	 */
197 	public static <T> boolean eq(T o1, T o2) {
198 		if (isArray(o1) && isArray(o2)) {
199 			int l1 = Array.getLength(o1), l2 = Array.getLength(o2);
200 			if (l1 != l2)
201 				return false;
202 			for (int i = 0; i < l1; i++)
203 				if (ne(Array.get(o1, i), Array.get(o2, i)))
204 					return false;
205 			return true;
206 		}
207 		return Objects.equals(o1, o2);
208 	}

211 	 * Tests two objects for equality, gracefully handling nulls.
212 	 *
213 	 * Allows you to simplify object comparison without sacrificing efficiency.
214 	 *
215 	 * Example:
216 	 * <code>
217 	 * 	public boolean equals(Object o)
218 	 * 		return eq(this, (Role)o, (x,y)-&gt;eq(x.id,y.id) &amp;&amp; eq(x.name,y.name) &amp;&amp; eq(x.created,y.created) &amp;&amp; eq(x.createdBy,y.createdBy));
219 	 * 	}
220 	 * </code>
221 	 *
222 	 * @param <T> Object 1 type.
223 	 * @param <U> Object 2 type.
224 	 * @param o1 Object 1.
225 	 * @param o2 Object 2.
226 	 * @param test The test to use for equality.
227 	 * @return <jk>true</jk> if both objects are equal based on the test.
228 	 */
229 	public static <T,U> boolean eq(T o1, U o2, BiPredicate<T,U> test) {
230 		if (o1 == null) {
231 			return o2 == null;
232 		}
233 		if (o2 == null) {
234 			return false;
235 		}
236 		if (o1 == o2) {
237 			return true;
238 		}
239 		return test.test(o1, o2);
240 	}

320 	 * Converts a string containing glob-style wildcard characters to a regular expression {@link java.util.regex.Pattern}.
321 	 *
322 	 * <p>This method converts glob-style patterns to regular expressions with the following mappings:
323 	 * <ul>
324 	 *   <li>{@code *} matches any sequence of characters (including none)</li>
325 	 *   <li>{@code ?} matches exactly one character</li>
326 	 *   <li>All other characters are treated literally</li>
327 	 * </ul>
328 	 *
329 	 * <h5 class='section'>Example:</h5>
330 	 * <p class='bjava'>
331 	 *   <jk>var</jk> <jv>pattern</jv> = <jsm>getGlobMatchPattern</jsm>(<js>"user_*_temp"</js>);
332 	 *   <jk>boolean</jk> <jv>matches</jv> = <jv>pattern</jv>.matcher(<js>"user_alice_temp"</js>).matches();  <jc>// true</jc>
333 	 *   <jv>matches</jv> = <jv>pattern</jv>.matcher(<js>"user_bob_temp"</js>).matches();    <jc>// true</jc>
334 	 *   <jv>matches</jv> = <jv>pattern</jv>.matcher(<js>"admin_alice_temp"</js>).matches(); <jc>// false</jc>
335 	 * </p>
336 	 *
337 	 * @param s The glob-style wildcard pattern string.
338 	 * @return A compiled {@link java.util.regex.Pattern} object, or <jk>null</jk> if the input string is <jk>null</jk>.
339 	 */
340 	public static java.util.regex.Pattern getGlobMatchPattern(String s) {
341 		return getGlobMatchPattern(s, 0);
342 	}

344 	/**
345 	 * Converts a string containing glob-style wildcard characters to a regular expression {@link java.util.regex.Pattern} with flags.
346 	 *
347 	 * <p>This method converts glob-style patterns to regular expressions with the following mappings:
348 	 * <ul>
349 	 *   <li>{@code *} matches any sequence of characters (including none)</li>
350 	 *   <li>{@code ?} matches exactly one character</li>
351 	 *   <li>All other characters are treated literally</li>
352 	 * </ul>
353 	 *
354 	 * <h5 class='section'>Example:</h5>
355 	 * <p class='bjava'>
356 	 *   <jc>// Case-insensitive matching</jc>
357 	 *   <jk>var</jk> <jv>pattern</jv> = <jsm>getGlobMatchPattern</jsm>(<js>"USER_*"</js>, Pattern.<jsf>CASE_INSENSITIVE</jsf>);
358 	 *   <jk>boolean</jk> <jv>matches</jv> = <jv>pattern</jv>.matcher(<js>"user_alice"</js>).matches();  <jc>// true</jc>
359 	 * </p>
360 	 *
361 	 * @param s The glob-style wildcard pattern string.
362 	 * @param flags Regular expression flags (see {@link java.util.regex.Pattern} constants).
363 	 * @return A compiled {@link java.util.regex.Pattern} object, or <jk>null</jk> if the input string is <jk>null</jk>.
364 	 */
365 	public static java.util.regex.Pattern getGlobMatchPattern(String s, int flags) {
366 		if (s == null)
367 			return null;
368 		var sb = new StringBuilder();
369 		sb.append("\\Q");
370 		for (var i = 0; i < s.length(); i++) {
371 			var c = s.charAt(i);
372 			if (c == '*')
373 				sb.append("\\E").append(".*").append("\\Q");
374 			else if (c == '?')
375 				sb.append("\\E").append(".").append("\\Q");
376 			else
377 				sb.append(c);
378 		}
379 		sb.append("\\E");
380 		return java.util.regex.Pattern.compile(sb.toString(), flags);
381 	}

398 	 *
399 	 * @param pattern The message pattern using <js>{0}</js>, <js>{1}</js>, etc. placeholders.
400 	 * @param args The arguments to substitute into the pattern placeholders.
401 	 * @return A {@link Supplier} that will format the string when {@code get()} is called.

409 	 * Calculates a hash code for the specified values.
410 	 *
411 	 * <p>
412 	 * This method handles annotations specially by delegating to {@link AnnotationUtils#hash(Annotation)}
413 	 * to ensure consistent hashing according to the {@link java.lang.annotation.Annotation#hashCode()} contract.
414 	 * For non-annotation values, it uses {@link Objects#hashCode(Object)}.
415 	 *
416 	 * @param values The values to hash.
417 	 * @return A hash code value for the given values.
418 	 * @see AnnotationUtils#hash(Annotation)
419 	 */
420 	public static final int hash(Object...values) {
421 		if (values == null)
422 			return 0;
423 		var result = 1;
424 		for (var value : values) {
425 			result = 31 * result + (value instanceof java.lang.annotation.Annotation ? AnnotationUtils.hash((java.lang.annotation.Annotation)value) : Objects.hashCode(value));
426 		}
427 		return result;
428 	}
