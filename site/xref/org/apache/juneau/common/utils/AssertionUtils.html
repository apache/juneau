21  /**
22   * Utility methods for argument validation and assertion.
23   *
24   * <p>
25   * This class provides static utility methods for validating method arguments and throwing
26   * {@link IllegalArgumentException} when validation fails. These methods are designed to
27   * simplify common parameter validation patterns and provide consistent error messages.
28   *
29   * <h5 class='section'>Features:</h5>
30   * <ul>
31   *   <li><b>Null checking:</b> Validate single or multiple arguments are not null
32   *   <li><b>Boolean expressions:</b> Assert arbitrary boolean conditions with custom messages
33   *   <li><b>String validation:</b> Check for null or blank strings
34   *   <li><b>Type checking:</b> Validate class array types
35   * </ul>
36   *
37   * <h5 class='section'>Usage:</h5>
38   * <p class='bjava'>
39   *   <jk>import static</jk> org.apache.juneau.common.utils.AssertionUtils.*;
40   *
41   *   <jk>public</jk> <jk>void</jk> setFooBar(String <jv>foo</jv>, String <jv>bar</jv>) {
42   *       <jc>// Validate multiple arguments at once</jc>
43   *       <jsm>assertArgsNotNull</jsm>(<js>"foo"</js>, <jv>foo</jv>, <js>"bar"</js>, <jv>bar</jv>);
44   *
45   *       <jc>// Validate custom condition</jc>
46   *       <jsm>assertArg</jsm>(<jv>foo</jv>.length() &gt; 0, <js>"Foo cannot be empty"</js>);
47   *   }
48   * </p>
49   *
50   * <h5 class='section'>See Also:</h5>
51   * <ul>
52   *   <li class='link'><a class="doclink" href='../../../../../index.html#juneau-common.utils'>Overview &gt; juneau-common.utils</a>
53   * </ul>
54   */
55  public class AssertionUtils {

57  	/**
58  	 * Throws an {@link IllegalArgumentException} if the specified expression is <jk>false</jk>.
59  	 *
60  	 * <h5 class='section'>Example:</h5>
61  	 * <p class='bjava'>
62  	 * 	<jk>import static</jk> org.apache.juneau.common.utils.AssertionUtils.*;
63  	 *
64  	 *	<jk>public</jk> String setFoo(List&lt;String&gt; <jv>foo</jv>) {
65  	 *		<jsm>assertArg</jsm>(<jv>foo</jv> != <jk>null</jk> &amp;&amp; ! <jv>foo</jv>.isEmpty(), <js>"'foo' cannot be null or empty."</js>);
66  	 *		...
67  	 *	}
68  	 * </p>
69  	 *
70  	 * @param expression The boolean expression to check.
71  	 * @param msg The exception message.
72  	 * @param args The exception message args.
73  	 * @throws IllegalArgumentException Constructed exception.
74  	 */
75  	public static final void assertArg(boolean expression, String msg, Object...args) throws IllegalArgumentException {
76  		if (! expression)
77  			throw new IllegalArgumentException(MessageFormat.format(msg, args));
78  	}

80  	/**
81  	 * Throws an {@link IllegalArgumentException} if the specified argument is <jk>null</jk>.
82  	 *
83  	 * <h5 class='section'>Example:</h5>
84  	 * <p class='bjava'>
85  	 * 	<jk>import static</jk> org.apache.juneau.common.utils.AssertionUtils.*;
86  	 *
87  	 *	<jk>public</jk> String setFoo(String <jv>foo</jv>) {
88  	 *		<jsm>assertArgNotNull</jsm>(<js>"foo"</js>, <jv>foo</jv>);
89  	 *		...
90  	 *	}
91  	 * </p>
92  	 *
93  	 * @param <T> The argument data type.
94  	 * @param name The argument name.
95  	 * @param o The object to check.
96  	 * @return The same argument.
97  	 * @throws IllegalArgumentException Constructed exception.
98  	 */
99  	public static final <T> T assertArgNotNull(String name, T o) throws IllegalArgumentException {
100 		assertArg(o != null, "Argument ''{0}'' cannot be null.", name);
101 		return o;
102 	}

104 	/**
105 	 * Throws an {@link IllegalArgumentException} if any of the specified arguments are <jk>null</jk>.
106 	 *
107 	 * <h5 class='section'>Example:</h5>
108 	 * <p class='bjava'>
109 	 * 	<jk>import static</jk> org.apache.juneau.common.utils.AssertionUtils.*;
110 	 *
111 	 *	<jk>public</jk> String setFooBar(String <jv>foo</jv>, String <jv>bar</jv>) {
112 	 *		<jsm>assertArgsNotNull</jsm>(<js>"foo"</js>, <jv>foo</jv>, <js>"bar"</js>, <jv>bar</jv>);
113 	 *		...
114 	 *	}
115 	 * </p>
116 	 *
117 	 * @param name1 The first argument name.
118 	 * @param o1 The first object to check.
119 	 * @param name2 The second argument name.
120 	 * @param o2 The second object to check.
121 	 * @throws IllegalArgumentException Constructed exception.
122 	 */
123 	public static final void assertArgsNotNull(String name1, Object o1, String name2, Object o2) throws IllegalArgumentException {
124 		assertArg(o1 != null, "Argument ''{0}'' cannot be null.", name1);
125 		assertArg(o2 != null, "Argument ''{0}'' cannot be null.", name2);
126 	}

129 	 * Throws an {@link IllegalArgumentException} if any of the specified arguments are <jk>null</jk>.
130 	 *
131 	 * <h5 class='section'>Example:</h5>

133 	 * 	<jk>import static</jk> org.apache.juneau.common.utils.AssertionUtils.*;

135 	 *	<jk>public</jk> String setFooBarBaz(String <jv>foo</jv>, String <jv>bar</jv>, String <jv>baz</jv>) {

137 	 *		...
138 	 *	}
139 	 * </p>
140 	 *
141 	 * @param name1 The first argument name.
142 	 * @param o1 The first object to check.
143 	 * @param name2 The second argument name.
144 	 * @param o2 The second object to check.
145 	 * @param name3 The third argument name.
146 	 * @param o3 The third object to check.
147 	 * @throws IllegalArgumentException Constructed exception.
148 	 */
149 	public static final void assertArgsNotNull(String name1, Object o1, String name2, Object o2, String name3, Object o3) throws IllegalArgumentException {

151 		assertArg(o2 != null, "Argument ''{0}'' cannot be null.", name2);
152 		assertArg(o3 != null, "Argument ''{0}'' cannot be null.", name3);
153 	}

179 	 * @param o1 The first object to check.
180 	 * @param name2 The second argument name.

198 	/**
199 	 * Throws an {@link IllegalArgumentException} if the specified string is <jk>null</jk> or blank.
200 	 *
201 	 * @param name The argument name.
202 	 * @param o The object to check.
203 	 * @return The same object.
204 	 * @throws IllegalArgumentException Thrown if the specified string is <jk>null</jk> or blank.
205 	 */
206 	@SuppressWarnings("null")
207 	public static final String assertArgNotNullOrBlank(String name, String o) throws IllegalArgumentException {
208 		assertArg(o != null, "Argument ''{0}'' cannot be null.", name);
209 		assertArg(! o.isBlank(), "Argument ''{0}'' cannot be blank.", name);
210 		return o;
211 	}
212 
213 	/**
214 	 * Throws an {@link IllegalArgumentException} if the specified value doesn't have all subclasses of the specified type.
215 	 *
216 	 * @param <E> The element type.
217 	 * @param name The argument name.
218 	 * @param type The expected parent class.
219 	 * @param value The array value being checked.
220 	 * @return The value cast to the specified array type.
221 	 * @throws IllegalArgumentException Constructed exception.
222 	 */
223 	@SuppressWarnings("unchecked")
224 	public static final <E> Class<E>[] assertClassArrayArgIsType(String name, Class<E> type, Class<?>[] value) throws IllegalArgumentException {
225 		for (var i = 0; i < value.length; i++)
226 			if (! type.isAssignableFrom(value[i]))
227 				throw new IllegalArgumentException("Arg " + name + " did not have arg of type " + type.getName() + " at index " + i + ": " + value[i].getName());
228 		return (Class<E>[])value;
229 	}

231 	/**
232 	 * Throws an {@link IllegalArgumentException} if the specified varargs array or any of its elements are <jk>null</jk>.
233 	 *
234 	 * @param <T> The element type.
235 	 * @param name The argument name.
236 	 * @param o The object to check.
237 	 * @return The same object.
238 	 * @throws IllegalArgumentException Thrown if the specified varargs array or any of its elements are <jk>null</jk>.
239 	 */
240 	@SuppressWarnings("null")
241 	public static final <T> T[] assertVarargsNotNull(String name, T[] o) throws IllegalArgumentException {
242 		assertArg(o != null, "Argument ''{0}'' cannot be null.", name);
243 		for (int i = 0; i < o.length; i++)
244 			assertArg(o[i] != null, "Argument ''{0}'' parameter {1} cannot be null.", name, i);
245 		return o;
246 	}
