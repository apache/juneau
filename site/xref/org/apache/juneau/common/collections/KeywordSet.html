17  package org.apache.juneau.common.collections;
18  
19  import java.util.*;
20  
21  /**
22   * A specialized set for storing and efficiently looking up language keywords.
23   *
24   * <p>
25   * This class provides a lightweight, immutable container optimized for fast keyword lookups using binary search.
26   * Keywords are stored in a sorted array, making lookups O(log n) efficient. This is particularly useful for
27   * parsers, compilers, and syntax highlighters that need to frequently check if identifiers are reserved keywords.
28   *
29   * <h5 class='section'>Features:</h5>
30   * <ul class='spaced-list'>
31   * 	<li>Immutable after construction - thread-safe for concurrent reads
32   * 	<li>Efficient O(log n) lookups using binary search
33   * 	<li>Compact memory footprint using sorted array
34   * 	<li>Automatic rejection of null and single-character strings
35   * </ul>
36   *
37   * <h5 class='section'>Implementation Details:</h5>
38   * <p>
39   * Keywords are sorted lexicographically during construction and stored in an internal array. The {@link #contains(String)}
40   * method uses {@link Arrays#binarySearch(Object[], Object)} for efficient lookups. Strings shorter than 2 characters
41   * are automatically rejected without performing a search, as most programming languages don't have single-character
42   * keywords.
43   *
44   * <h5 class='section'>Examples:</h5>
45   * <p class='bjava'>
46   * 	<jc>// Create a keyword set for Java reserved words</jc>
47   * 	KeywordSet <jv>javaKeywords</jv> = <jk>new</jk> KeywordSet(
48   * 		<js>"abstract"</js>, <js>"assert"</js>, <js>"boolean"</js>, <js>"break"</js>, <js>"byte"</js>,
49   * 		<js>"case"</js>, <js>"catch"</js>, <js>"char"</js>, <js>"class"</js>, <js>"const"</js>
50   * 		<jc>// ... more keywords</jc>
51   * 	);
52   *
53   * 	<jc>// Check if identifiers are keywords</jc>
54   * 	<jk>if</jk> (<jv>javaKeywords</jv>.contains(<js>"class"</js>)) {
55   * 		<jc>// Handle keyword</jc>
56   * 	}
57   *
58   * 	<jc>// Safe handling of edge cases</jc>
59   * 	<jsm>assertFalse</jsm>(<jv>javaKeywords</jv>.contains(<jk>null</jk>));     <jc>// Returns false</jc>
60   * 	<jsm>assertFalse</jsm>(<jv>javaKeywords</jv>.contains(<js>"a"</js>));      <jc>// Single char - returns false</jc>
61   * 	<jsm>assertFalse</jsm>(<jv>javaKeywords</jv>.contains(<js>"myVar"</js>));  <jc>// Not a keyword</jc>
62   *
63   * 	<jc>// Use in a parser/lexer</jc>
64   * 	<jk>class</jk> Lexer {
65   * 		<jk>private static final</jk> KeywordSet KEYWORDS = <jk>new</jk> KeywordSet(
66   * 			<js>"if"</js>, <js>"else"</js>, <js>"while"</js>, <js>"for"</js>, <js>"return"</js>
67   * 		);
68   *
69   * 		<jk>boolean</jk> isKeyword(String token) {
70   * 			<jk>return</jk> KEYWORDS.contains(token);
71   * 		}
72   * 	}
73   * </p>
74   *
75   * <h5 class='section'>Use Cases:</h5>
76   * <ul class='spaced-list'>
77   * 	<li>Programming language parsers - checking if tokens are reserved words
78   * 	<li>Syntax highlighters - identifying keywords for special formatting
79   * 	<li>Code analyzers - distinguishing keywords from identifiers
80   * 	<li>Template engines - recognizing template keywords
81   * 	<li>Query languages - validating reserved words
82   * </ul>
83   *
84   * <h5 class='section'>Notes:</h5><ul>
85   * 	<li class='note'>
86   * 		This class is <b>immutable and thread-safe</b> after construction. Multiple threads can safely call
87   * 		{@link #contains(String)} concurrently.
88   * 	<li class='note'>
89   * 		Keywords are compared using exact string matching (case-sensitive). For case-insensitive matching,
90   * 		normalize your keywords and input strings to the same case.
91   * 	<li class='note'>
92   * 		The minimum keyword length is 2 characters. Single-character strings are automatically rejected.
93   * 	<li class='note'>
94   * 		Consider creating {@link KeywordSet} instances as static final constants to avoid repeated construction.
95   * </ul>
96   *
97   * <h5 class='section'>See Also:</h5><ul>
98   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauCommonCollections">juneau-common-collections</a>
99   * </ul>
100  */
101 public class KeywordSet {
102 	final String[] store;
103 
104 	/**
105 	 * Creates a new keyword set with the specified keywords.
106 	 *
107 	 * <p>
108 	 * Keywords are automatically sorted during construction. Duplicate keywords are allowed but provide
109 	 * no benefit. For best performance, pass unique keywords.
110 	 *
111 	 * <h5 class='section'>Example:</h5>
112 	 * <p class='bjava'>
113 	 * 	<jc>// Create a keyword set for SQL keywords</jc>
114 	 * 	KeywordSet <jv>sql</jv> = <jk>new</jk> KeywordSet(
115 	 * 		<js>"SELECT"</js>, <js>"FROM"</js>, <js>"WHERE"</js>, <js>"INSERT"</js>, <js>"UPDATE"</js>,
116 	 * 		<js>"DELETE"</js>, <js>"CREATE"</js>, <js>"DROP"</js>, <js>"TABLE"</js>, <js>"INDEX"</js>
117 	 * 	);
118 	 *
119 	 * 	<jc>// Keywords can be passed in any order</jc>
120 	 * 	KeywordSet <jv>keywords</jv> = <jk>new</jk> KeywordSet(<js>"zebra"</js>, <js>"apple"</js>, <js>"banana"</js>);
121 	 * 	<jsm>assertTrue</jsm>(<jv>keywords</jv>.contains(<js>"apple"</js>));  <jc>// Sorted internally</jc>
122 	 * </p>
123 	 *
124 	 * @param keywords The keywords to store. Can be empty but not <jk>null</jk>. Individual keywords can be any non-null string.
125 	 */
126 	public KeywordSet(String...keywords) {
127 		this.store = keywords;
128 		Arrays.sort(store);
129 	}
130 
131 	/**
132 	 * Checks if the specified string is a keyword in this set.
133 	 *
134 	 * <p>
135 	 * This method performs an O(log n) binary search on the sorted keyword array. Null strings and
136 	 * strings with fewer than 2 characters are automatically rejected without performing a search.
137 	 *
138 	 * <h5 class='section'>Example:</h5>
139 	 * <p class='bjava'>
140 	 * 	KeywordSet <jv>keywords</jv> = <jk>new</jk> KeywordSet(<js>"class"</js>, <js>"interface"</js>, <js>"enum"</js>);
141 	 *
142 	 * 	<jc>// Standard checks</jc>
143 	 * 	<jsm>assertTrue</jsm>(<jv>keywords</jv>.contains(<js>"class"</js>));      <jc>// Keyword exists</jc>
144 	 * 	<jsm>assertFalse</jsm>(<jv>keywords</jv>.contains(<js>"MyClass"</js>));   <jc>// Not a keyword</jc>
145 	 *
146 	 * 	<jc>// Edge cases handled gracefully</jc>
147 	 * 	<jsm>assertFalse</jsm>(<jv>keywords</jv>.contains(<jk>null</jk>));        <jc>// null returns false</jc>
148 	 * 	<jsm>assertFalse</jsm>(<jv>keywords</jv>.contains(<js>""</js>));          <jc>// Empty string returns false</jc>
149 	 * 	<jsm>assertFalse</jsm>(<jv>keywords</jv>.contains(<js>"a"</js>));         <jc>// Single char returns false</jc>
150 	 *
151 	 * 	<jc>// Case-sensitive matching</jc>
152 	 * 	<jsm>assertTrue</jsm>(<jv>keywords</jv>.contains(<js>"class"</js>));
153 	 * 	<jsm>assertFalse</jsm>(<jv>keywords</jv>.contains(<js>"CLASS"</js>));     <jc>// Different case</jc>
154 	 * </p>
155 	 *
156 	 * <h5 class='section'>Performance:</h5>
157 	 * <ul>
158 	 * 	<li>Time complexity: O(log n) using binary search
159 	 * 	<li>Space complexity: O(1) - no additional memory allocated
160 	 * 	<li>Short-circuit: Strings with length &lt; 2 return immediately without searching
161 	 * </ul>
162 	 *
163 	 * @param s The string to check. Can be <jk>null</jk>.
164 	 * @return <jk>true</jk> if the string exists in this keyword set, <jk>false</jk> if it doesn't exist,
165 	 *         is <jk>null</jk>, or has fewer than 2 characters.
166 	 */
167 	public boolean contains(String s) {
168 		if (s == null || s.length() < 2)
169 			return false;
170 		return Arrays.binarySearch(store, s) >= 0;
171 	}
172 }
