30  /**
31   * A fluent builder for constructing {@link Set} instances with various configuration options.
32   *
33   * <p>
34   * This builder provides a flexible and type-safe way to construct sets with support for adding elements,
35   * collections, arrays, sorting, and applying modifiers like unmodifiable or sparse modes. Sets automatically
36   * handle duplicates - adding the same element multiple times will result in only one occurrence in the final set.
37   *
38   * <h5 class='section'>Features:</h5>
39   * <ul class='spaced-list'>
40   * 	<li>Fluent API - all methods return <c>this</c> for method chaining
41   * 	<li>Multiple add methods - single elements, varargs, collections, arrays
42   * 	<li>Arbitrary input support - automatic type conversion with {@link #addAny(Object...)}
43   * 	<li>Conditional adding - {@link #addIf(boolean, Object)} for conditional elements
44   * 	<li>Sorting support - natural order or custom {@link Comparator}
45   * 	<li>Sparse mode - return <jk>null</jk> for empty sets
46   * 	<li>Unmodifiable mode - create immutable sets
47   * 	<li>Custom converters - type conversion via {@link Converter}
48   * 	<li>Automatic deduplication - duplicate elements are automatically removed
49   * </ul>
50   *
51   * <h5 class='section'>Examples:</h5>
52   * <p class='bjava'>
53   * 	<jc>// Basic usage</jc>
54   * 	Set&lt;String&gt; <jv>set</jv> = SetBuilder.<jsm>create</jsm>(String.<jk>class</jk>)
55   * 		.add(<js>"apple"</js>, <js>"banana"</js>, <js>"cherry"</js>)
56   * 		.build();
57   *
58   * 	<jc>// Automatic deduplication</jc>
59   * 	Set&lt;Integer&gt; <jv>unique</jv> = SetBuilder.<jsm>create</jsm>(Integer.<jk>class</jk>)
60   * 		.add(1, 2, 3, 2, 1)  <jc>// Duplicates ignored</jc>
61   * 		.build();  <jc>// Contains: 1, 2, 3</jc>
62   *
63   * 	<jc>// With sorting</jc>
64   * 	Set&lt;String&gt; <jv>sorted</jv> = SetBuilder.<jsm>create</jsm>(String.<jk>class</jk>)
65   * 		.add(<js>"zebra"</js>, <js>"apple"</js>, <js>"banana"</js>)
66   * 		.sorted()
67   * 		.build();  <jc>// Returns TreeSet in natural order</jc>
68   *
69   * 	<jc>// Conditional elements</jc>
70   * 	Set&lt;String&gt; <jv>features</jv> = SetBuilder.<jsm>create</jsm>(String.<jk>class</jk>)
71   * 		.add(<js>"basic"</js>)
72   * 		.addIf(<jv>hasPremium</jv>, <js>"premium"</js>)
73   * 		.addIf(<jv>hasEnterprise</jv>, <js>"enterprise"</js>)
74   * 		.build();
75   *
76   * 	<jc>// Immutable set</jc>
77   * 	Set&lt;String&gt; <jv>immutable</jv> = SetBuilder.<jsm>create</jsm>(String.<jk>class</jk>)
78   * 		.add(<js>"read"</js>, <js>"only"</js>)
79   * 		.unmodifiable()
80   * 		.build();
81   *
82   * 	<jc>// From multiple sources</jc>
83   * 	Set&lt;Integer&gt; <jv>existing</jv> = Set.of(1, 2, 3);
84   * 	Set&lt;Integer&gt; <jv>combined</jv> = SetBuilder.<jsm>create</jsm>(Integer.<jk>class</jk>)
85   * 		.addAll(<jv>existing</jv>)
86   * 		.add(4, 5, 6)
87   * 		.build();
88   *
89   * 	<jc>// Sparse mode - returns null when empty</jc>
90   * 	Set&lt;String&gt; <jv>maybeNull</jv> = SetBuilder.<jsm>create</jsm>(String.<jk>class</jk>)
91   * 		.sparse()
92   * 		.build();  <jc>// Returns null, not empty set</jc>
93   * </p>
94   *
95   * <h5 class='section'>Thread Safety:</h5>
96   * <p>
97   * This class is <b>not thread-safe</b>. Each builder instance should be used by a single thread or
98   * properly synchronized.
99   *
100  * <h5 class='section'>See Also:</h5><ul>
101  * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauCommonCollections">juneau-common-collections</a>
102  * 	<li class='jc'>{@link ListBuilder}
103  * 	<li class='jc'>{@link MapBuilder}
104  * </ul>
105  *
106  * @param <E> The element type.
107  */
108 public class SetBuilder<E> {
109 
110 	private Set<E> set;
111 	private boolean unmodifiable, sparse;
112 	private Comparator<E> comparator;
113 
114 	private Class<E> elementType;
115 	private List<Converter> converters;
116 
117 	/**
118 	 * Static creator.
119 	 *
120 	 * @param <E> The element type.
121 	 * @param elementType The element type.
122 	 * @param elementTypeArgs Optional element type arguments.
123 	 * @return A new builder.
124 	 */
125 	public static <E> SetBuilder<E> create(Class<E> elementType) {
126 		return new SetBuilder<>(elementType);
127 	}
128 
129 	/**
130 	 * Constructor.
131 	 *
132 	 * @param elementType The element type.
133 	 * @param elementTypeArgs The element type generic arguments if there are any.
134 	 */
135 	public SetBuilder(Class<E> elementType) {
136 		this.elementType = elementType;
137 	}
138 
139 	/**
140 	 * Specifies the set to append to.
141 	 *
142 	 * <p>
143 	 * If not specified, uses a new {@link LinkedHashSet}.
144 	 *
145 	 * @param set The set to append to.
146 	 * @return This object.
147 	 */
148 	public SetBuilder<E> to(Set<E> set) {
149 		this.set = set;
150 		return this;
151 	}
152 
153 	/**
154 	 * Adds a single value to this set.
155 	 *
156 	 * @param value The value to add to this set.
157 	 * @return This object.
158 	 */
159 	public SetBuilder<E> add(E value) {
160 		if (set == null)
161 			set = new LinkedHashSet<>();
162 		set.add(value);
163 		return this;
164 	}
165 
166 	/**
167 	 * Adds multiple values to this set.
168 	 *
169 	 * @param values The values to add to this set.
170 	 * @return This object.
171 	 */
172 	@SuppressWarnings("unchecked")
173 	public SetBuilder<E> add(E...values) {
174 		for (var v : values)
175 			add(v);
176 		return this;
177 	}
178 
179 	/**
180 	 * Appends the contents of the specified collection into this set.
181 	 *
182 	 * <p>
183 	 * This is a no-op if the value is <jk>null</jk>.
184 	 *
185 	 * @param value The collection to add to this set.
186 	 * @return This object.
187 	 */
188 	public SetBuilder<E> addAll(Collection<E> value) {
189 		if (nn(value)) {
190 			if (set == null)
191 				set = new LinkedHashSet<>(value);
192 			else
193 				set.addAll(value);
194 		}
195 		return this;
196 	}
197 
198 	/**
199 	 * Adds arbitrary values to this set.
200 	 *
201 	 * <p>
202 	 * Objects can be any of the following:
203 	 * <ul>
204 	 * 	<li>The same type or convertible to the element type of this set.
205 	 * 	<li>Collections or arrays of anything on this set.
206 	 * 	<li>JSON array strings parsed and convertible to the element type of this set.
207 	 * </ul>
208 	 *
209 	 * @param values The values to add.
210 	 * @return This object.
211 	 */
212 	public SetBuilder<E> addAny(Object...values) {
213 		if (elementType == null)
214 			throw new IllegalStateException("Unknown element type. Cannot use this method.");
215 		if (nn(values)) {
216 			for (var o : values) {
217 				if (nn(o)) {
218 					if (o instanceof Collection) {
219 						((Collection<?>)o).forEach(x -> addAny(x));
220 					} else if (isArray(o)) {
221 						for (int i = 0; i < Array.getLength(o); i++)
222 							addAny(Array.get(o, i));
223 					} else if (elementType.isInstance(o)) {
224 						add(elementType.cast(o));
225 					} else {
226 						if (nn(converters)) {
227 							var e = converters.stream().map(x -> x.convertTo(elementType, o)).filter(x -> nn(x)).findFirst().orElse(null);
228 							if (nn(e)) {
229 								add(e);
230 							} else {
231 								var l = converters.stream().map(x -> x.convertTo(List.class, o)).filter(x -> nn(x)).findFirst().orElse(null);
232 								if (nn(l))
233 									addAny(l);
234 								else
235 									throw rex("Object of type {0} could not be converted to type {1}", cn(o), cn(elementType));
236 							}
237 						}
238 					}
239 				}
240 			}
241 		}
242 		return this;
243 	}
244 
245 	/**
246 	 * Registers value converters that can adapt incoming values in {@link #addAny(Object...)}.
247 	 *
248 	 * @param values Converters to register. Ignored if {@code null}.
249 	 * @return This object.
250 	 */
251 	public SetBuilder<E> converters(Converter...values) {
252 		if (values.length == 0)
253 			return this;
254 		if (converters == null)
255 			converters = list();
256 		converters.addAll(l(values));
257 		return this;
258 	}
259 
260 	/**
261 	 * Adds a value to this set if the specified flag is true.
262 	 *
263 	 * @param flag The flag.
264 	 * @param value The value.
265 	 * @return This object.
266 	 */
267 	public SetBuilder<E> addIf(boolean flag, E value) {
268 		if (flag)
269 			add(value);
270 		return this;
271 	}
272 
273 	/**
274 	 * Adds entries to this set via JSON array strings.
275 	 *
276 	 * @param values The JSON array strings to parse and add to this set.
277 	 * @return This object.
278 	 */
279 	public SetBuilder<E> addJson(String...values) {
280 		return addAny((Object[])values);
281 	}
282 
283 	/**
284 	 * Builds the set.
285 	 *
286 	 * @return A set conforming to the settings on this builder.
287 	 */
288 	/**
289 	 * Builds the set.
290 	 *
291 	 * <p>
292 	 * Applies sorting/unmodifiable/sparse options.
293 	 *
294 	 * @return The built set or {@code null} if {@link #sparse()} is set and the set is empty.
295 	 */
296 	public Set<E> build() {
297 		if (sparse) {
298 			if (nn(set) && set.isEmpty())
299 				set = null;
300 		} else {
301 			if (set == null)
302 				set = new LinkedHashSet<>(0);
303 		}
304 		if (nn(set)) {
305 			if (nn(comparator)) {
306 				Set<E> s = new TreeSet<>(comparator);
307 				s.addAll(set);
308 				set = s;
309 			}
310 			if (unmodifiable)
311 				set = unmodifiableSet(set);
312 		}
313 		return set;
314 	}
315 
316 	/**
317 	 * Forces the existing set to be copied instead of appended to.
318 	 *
319 	 * @return This object.
320 	 */
321 	public SetBuilder<E> copy() {
322 		if (nn(set))
323 			set = new LinkedHashSet<>(set);
324 		return this;
325 	}

327 	/**
328 	 * Specifies the element type on this list.
329 	 *
330 	 * @param value The element type.
331 	 * @return This object.
332 	 */
333 	public SetBuilder<E> elementType(Class<E> value) {
334 		this.elementType = value;
335 		return this;
336 	}
337 
338 	/**
339 	 * Converts the set into a {@link SortedSet}.
340 	 *
341 	 * @return This object.
342 	 */
343 	@SuppressWarnings("unchecked")
344 	public SetBuilder<E> sorted() {
345 		return sorted((Comparator<E>)Comparator.naturalOrder());
346 	}
347 
348 	/**
349 	 * Converts the set into a {@link SortedSet} using the specified comparator.
350 	 *
351 	 * @param comparator The comparator to use for sorting.
352 	 * @return This object.
353 	 */
354 	public SetBuilder<E> sorted(Comparator<E> comparator) {
355 		this.comparator = comparator;
356 		return this;
357 	}
358 
359 	/**
360 	 * When specified, the {@link #build()} method will return <jk>null</jk> if the set is empty.
361 	 *
362 	 * <p>
363 	 * Otherwise {@link #build()} will never return <jk>null</jk>.
364 	 *
365 	 * @return This object.
366 	 */
367 	public SetBuilder<E> sparse() {
368 		this.sparse = true;
369 		return this;
370 	}
371 
372 	/**
373 	 * When specified, {@link #build()} will return an unmodifiable set.
374 	 *
375 	 * @return This object.
376 	 */
377 	public SetBuilder<E> unmodifiable() {
378 		this.unmodifiable = true;
379 		return this;
380 	}
