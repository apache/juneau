17  package org.apache.juneau.common.collections;
18  
19  import static org.apache.juneau.common.utils.AssertionUtils.*;
20  
21  import java.util.concurrent.*;
22  
23  import org.apache.juneau.common.function.*;
24  
25  /**
26   * A thread-safe concurrent hash map that uses composite five-part keys for lookups.
27   *
28   * <p>
29   * This class extends {@link ConcurrentHashMap} to provide efficient storage and retrieval of values
30   * indexed by five separate key components. It's useful when you need to look up values based on a
31   * combination of five keys without creating intermediate wrapper objects.
32   *
33   * <h5 class='section'>Features:</h5>
34   * <ul class='spaced-list'>
35   * 	<li><b>Five-Part Keys:</b> Values are indexed by five keys (K1, K2, K3, K4, K5) instead of a single key
36   * 	<li><b>Content-Based Keys:</b> Keys use content-based equality via {@link Tuple5}
37   * 	<li><b>Array Support:</b> Arrays can be used as key components with proper content-based hashing and equality
38   * 	<li><b>Thread-Safe:</b> Inherits all thread-safety guarantees from {@link ConcurrentHashMap}
39   * 	<li><b>Null Keys Supported:</b> All key parts can be <jk>null</jk>
40   * 	<li><b>Optional Caching:</b> Can be disabled to always invoke a supplier function instead of caching
41   * 	<li><b>Supplier Integration:</b> Optionally provides automatic value computation via a {@link Function5}
42   * </ul>
43   *
44   * <h5 class='section'>Usage:</h5>
45   * <p class='bjava'>
46   * 	<jc>// Create a map for caching complex queries</jc>
47   * 	ConcurrentHashMap5Key&lt;String,String,String,String,String,ResultSet&gt; <jv>queryCache</jv> =
48   * 		<jk>new</jk> ConcurrentHashMap5Key&lt;&gt;();
49   *
50   * 	<jc>// Store results: environment, database, schema, table, operation</jc>
51   * 	<jv>queryCache</jv>.put(<js>"prod"</js>, <js>"db1"</js>, <js>"public"</js>, <js>"users"</js>, <js>"select"</js>, <jv>results</jv>);
52   *
53   * 	<jc>// Retrieve results</jc>
54   * 	ResultSet <jv>cached</jv> = <jv>queryCache</jv>.get(<js>"prod"</js>, <js>"db1"</js>, <js>"public"</js>, <js>"users"</js>, <js>"select"</js>);
55   * </p>
56   *
57   * <h5 class='section'>With Supplier:</h5>
58   * <p class='bjava'>
59   * 	<jc>// Create a map that automatically executes queries</jc>
60   * 	ConcurrentHashMap5Key&lt;String,String,String,String,String,ResultSet&gt; <jv>cache</jv> =
61   * 		<jk>new</jk> ConcurrentHashMap5Key&lt;&gt;(<jk>false</jk>, (env, db, schema, table, op) -&gt;
62   * 			queryService.execute(env, db, schema, table, op)
63   * 		);
64   *
65   * 	<jc>// Automatically executes and caches if not present</jc>
66   * 	ResultSet <jv>results</jv> = <jv>cache</jv>.get(<js>"prod"</js>, <js>"db1"</js>, <js>"public"</js>, <js>"users"</js>, <js>"select"</js>);
67   * </p>
68   *
69   * <h5 class='section'>Disabled Mode:</h5>
70   * <p>
71   * When constructed with <c>disabled=<jk>true</jk></c>, the map doesn't actually store any values.
72   * Instead, it always invokes the supplier function (if provided) or returns <jk>null</jk>.
73   * This is useful for testing or development scenarios where caching should be bypassed.
74   * </p>
75   *
76   * <p class='bjava'>
77   * 	<jc>// Create a disabled cache (useful for testing)</jc>
78   * 	ConcurrentHashMap5Key&lt;String,String,String,String,String,ResultSet&gt; <jv>cache</jv> =
79   * 		<jk>new</jk> ConcurrentHashMap5Key&lt;&gt;(<jk>true</jk>, (env, db, schema, table, op) -&gt;
80   * 			queryService.execute(env, db, schema, table, op)
81   * 		);
82   *
83   * 	<jc>// Always calls queryService.execute(), never caches</jc>
84   * 	ResultSet <jv>results</jv> = <jv>cache</jv>.get(<js>"prod"</js>, <js>"db1"</js>, <js>"public"</js>, <js>"users"</js>, <js>"select"</js>);
85   * </p>
86   *
87   * <h5 class='section'>Common Use Cases:</h5>
88   * <ul class='spaced-list'>
89   * 	<li>Caching database query results with complex composite keys
90   * 	<li>Storing multi-dimensional analytics data
91   * 	<li>Mapping highly granular configuration settings
92   * 	<li>Tracking detailed audit logs or activity records
93   * 	<li>Caching results from complex multi-parameter computations
94   * </ul>
95   *
96   * <h5 class='section'>See Also:</h5>
97   * <ul>
98   * 	<li class='jc'>{@link ConcurrentHashMap1Key}
99   * 	<li class='jc'>{@link ConcurrentHashMap2Key}
100  * 	<li class='jc'>{@link ConcurrentHashMap3Key}
101  * 	<li class='jc'>{@link ConcurrentHashMap4Key}
102  * 	<li class='link'><a class="doclink" href="../../../../../index.html#juneau-common">Overview &gt; juneau-common</a>
103  * </ul>
104  *
105  * @param <K1> The first key component type. Can be an array type for content-based key matching.
106  * @param <K2> The second key component type. Can be an array type for content-based key matching.
107  * @param <K3> The third key component type. Can be an array type for content-based key matching.
108  * @param <K4> The fourth key component type. Can be an array type for content-based key matching.
109  * @param <K5> The fifth key component type. Can be an array type for content-based key matching.
110  * @param <V> The value type.
111  * @serial exclude
112  */
113 public class ConcurrentHashMap5Key<K1,K2,K3,K4,K5,V> extends ConcurrentHashMap<Tuple5<K1,K2,K3,K4,K5>,V> {
114 
115 	private static final long serialVersionUID = 1L;
116 
117 	/**
118 	 * Retrieves the value associated with the specified five-part key.
119 	 *
120 	 * @param key1 First key component. Must not be <jk>null</jk>.
121 	 * @param key2 Second key component. Must not be <jk>null</jk>.
122 	 * @param key3 Third key component. Must not be <jk>null</jk>.
123 	 * @param key4 Fourth key component. Must not be <jk>null</jk>.
124 	 * @param key5 Fifth key component. Must not be <jk>null</jk>.
125 	 * @return The value associated with the key, or <jk>null</jk> if not found.
126 	 * @throws IllegalArgumentException if any key is <jk>null</jk>.
127 	 */
128 	public V get(K1 key1, K2 key2, K3 key3, K4 key4, K5 key5) {
129 		assertArgsNotNull("key1", key1, "key2", key2, "key3", key3, "key4", key4, "key5", key5);
130 		return super.get(Tuple5.of(key1, key2, key3, key4, key5));
131 	}
132 
133 	/**
134 	 * Associates the specified value with the specified five-part key in this map.
135 	 *
136 	 * @param key1 First key component. Must not be <jk>null</jk>.
137 	 * @param key2 Second key component. Must not be <jk>null</jk>.
138 	 * @param key3 Third key component. Must not be <jk>null</jk>.
139 	 * @param key4 Fourth key component. Must not be <jk>null</jk>.
140 	 * @param key5 Fifth key component. Must not be <jk>null</jk>.
141 	 * @param value The value to associate with the key.
142 	 * @return The previous value associated with the key, or <jk>null</jk> if there was no mapping.
143 	 * @throws IllegalArgumentException if any key is <jk>null</jk>.
144 	 */
145 	public V put(K1 key1, K2 key2, K3 key3, K4 key4, K5 key5, V value) {
146 		assertArgsNotNull("key1", key1, "key2", key2, "key3", key3, "key4", key4, "key5", key5);
147 		return super.put(Tuple5.of(key1, key2, key3, key4, key5), value);
148 	}
149 }
