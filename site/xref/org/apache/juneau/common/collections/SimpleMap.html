17  package org.apache.juneau.common.collections;
18  
19  import java.lang.reflect.*;
20  import java.util.*;
21  
22  import org.apache.juneau.common.utils.*;
23  
24  /**
25   * An instance of a <c>Map</c> where the keys and values are simple arrays.
26   *
27   * <p>
28   * Typically more efficient than <c>HashMaps</c> for small maps (e.g. &lt;10 entries).
29   *
30   * <p>
31   * Does not support adding or removing entries.
32   *
33   * <p>
34   * Setting values overwrites the value on the underlying value array.
35   *
36   * <h5 class='section'>See Also:</h5><ul>
37  
38   * </ul>
39   *
40   * @param <K> The key type.
41   * @param <V> The value type.
42   */
43  public class SimpleMap<K,V> extends AbstractMap<K,V> {
44  
45  	class SimpleMapEntry implements Map.Entry<K,V> {
46  
47  		private int index;
48  
49  		SimpleMapEntry(int index) {
50  			this.index = index;
51  		}
52  
53  		@Override /* Overridden from Map.Entry */
54  		public K getKey() { return keys[index]; }
55  
56  		@Override /* Overridden from Map.Entry */
57  		public V getValue() { return values[index]; }
58  
59  		@Override /* Overridden from Map.Entry */
60  		public V setValue(V val) {
61  			V v = values[index];
62  			values[index] = val;
63  			return v;
64  		}
65  	}
66  
67  	final K[] keys;
68  	final V[] values;
69  
70  	final SimpleMapEntry[] entries;
71  
72  	/**
73  	 * Constructor.
74  	 *
75  	 * @param keys The map keys.  Must not be <jk>null</jk>.
76  	 * @param values The map values.  Must not be <jk>null</jk>.
77  	 */
78  	@SuppressWarnings("unchecked")
79  	public SimpleMap(K[] keys, V[] values) {
80  		Utils.assertArgNotNull("keys", keys);
81  		Utils.assertArgNotNull("values", values);
82  		Utils.assertArg(keys.length == values.length, "keys ''{0}'' and values ''{1}'' array lengths differ", keys.length, values.length);
83  
84  		this.keys = keys;
85  		this.values = values;
86  		entries = (SimpleMapEntry[])Array.newInstance(SimpleMapEntry.class, keys.length);
87  		for (int i = 0; i < keys.length; i++) {
88  			Utils.assertArg(keys[i] != null, "Keys array cannot contain a null value.");
89  			entries[i] = new SimpleMapEntry(i);
90  		}
91  	}
92  
93  	@Override /* Overridden from Map */
94  	public Set<Map.Entry<K,V>> entrySet() {
95  		return ArrayUtils.asSet(entries);
96  	}
97  
98  	@Override /* Overridden from Map */
99  	public V get(Object key) {
100 		for (int i = 0; i < keys.length; i++)
101 			if (keys[i].equals(key))
102 				return values[i];
103 		return null;
104 	}
105 
106 	@Override /* Overridden from Map */
107 	public Set<K> keySet() {
108 		return ArrayUtils.asSet(keys);
109 	}
110 
111 	@Override /* Overridden from Map */
112 	public V put(K key, V value) {
113 		for (int i = 0; i < keys.length; i++) {
114 			if (keys[i].equals(key)) {
115 				V v = values[i];
116 				values[i] = value;
117 				return v;
118 			}
119 		}
120 		throw new IllegalArgumentException("No key '" + key + "' defined in map");
121 	}
122 }
