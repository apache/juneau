17  package org.apache.juneau.common.collections;
18  
19  import java.util.concurrent.*;
20  
21  import org.apache.juneau.common.function.*;
22  import org.apache.juneau.common.utils.*;
23  
24  /**
25   * A thread-safe concurrent hash map that uses composite four-part keys for lookups.
26   *
27   * <p>
28   * This class extends {@link ConcurrentHashMap} to provide efficient storage and retrieval of values
29   * indexed by four separate key components. It's useful when you need to look up values based on a
30   * combination of four keys without creating intermediate wrapper objects.
31   *
32   * <h5 class='section'>Features:</h5>
33   * <ul class='spaced-list'>
34   * 	<li><b>Four-Part Keys:</b> Values are indexed by four keys (K1, K2, K3, K4) instead of a single key
35   * 	<li><b>Thread-Safe:</b> Inherits all thread-safety guarantees from {@link ConcurrentHashMap}
36   * 	<li><b>Null Keys Supported:</b> All key parts can be <jk>null</jk>
37   * 	<li><b>Optional Caching:</b> Can be disabled to always invoke a supplier function instead of caching
38   * 	<li><b>Supplier Integration:</b> Optionally provides automatic value computation via a {@link Function4}
39   * </ul>
40   *
41   * <h5 class='section'>Usage:</h5>
42   * <p class='bjava'>
43   * 	<jc>// Create a map for caching permissions</jc>
44   * 	ConcurrentHashMap4Key&lt;String,String,String,String,Boolean&gt; <jv>permissions</jv> =
45   * 		<jk>new</jk> ConcurrentHashMap4Key&lt;&gt;();
46   *
47   * 	<jc>// Store permissions: tenant, user, resource, action</jc>
48   * 	<jv>permissions</jv>.put(<js>"tenant1"</js>, <js>"user123"</js>, <js>"document"</js>, <js>"read"</js>, <jk>true</jk>);
49   * 	<jv>permissions</jv>.put(<js>"tenant1"</js>, <js>"user123"</js>, <js>"document"</js>, <js>"write"</js>, <jk>false</jk>);
50   *
51   * 	<jc>// Check permission</jc>
52   * 	Boolean <jv>canRead</jv> = <jv>permissions</jv>.get(<js>"tenant1"</js>, <js>"user123"</js>, <js>"document"</js>, <js>"read"</js>);
53   * </p>
54   *
55   * <h5 class='section'>With Supplier:</h5>
56   * <p class='bjava'>
57   * 	<jc>// Create a map that automatically checks permissions</jc>
58   * 	ConcurrentHashMap4Key&lt;String,String,String,String,Boolean&gt; <jv>permCache</jv> =
59   * 		<jk>new</jk> ConcurrentHashMap4Key&lt;&gt;(<jk>false</jk>, (tenant, user, resource, action) -&gt;
60   * 			permissionService.check(tenant, user, resource, action)
61   * 		);
62   *
63   * 	<jc>// Automatically checks and caches if not present</jc>
64   * 	Boolean <jv>allowed</jv> = <jv>permCache</jv>.get(<js>"tenant1"</js>, <js>"user123"</js>, <js>"document"</js>, <js>"write"</js>);
65   * </p>
66   *
67   * <h5 class='section'>Disabled Mode:</h5>
68   * <p>
69   * When constructed with <c>disabled=<jk>true</jk></c>, the map doesn't actually store any values.
70   * Instead, it always invokes the supplier function (if provided) or returns <jk>null</jk>.
71   * This is useful for testing or development scenarios where caching should be bypassed.
72   * </p>
73   *
74   * <p class='bjava'>
75   * 	<jc>// Create a disabled cache (useful for testing)</jc>
76   * 	ConcurrentHashMap4Key&lt;String,String,String,String,Boolean&gt; <jv>permCache</jv> =
77   * 		<jk>new</jk> ConcurrentHashMap4Key&lt;&gt;(<jk>true</jk>, (tenant, user, resource, action) -&gt;
78   * 			permissionService.check(tenant, user, resource, action)
79   * 		);
80   *
81   * 	<jc>// Always calls permissionService.check(), never caches</jc>
82   * 	Boolean <jv>allowed</jv> = <jv>permCache</jv>.get(<js>"tenant1"</js>, <js>"user123"</js>, <js>"document"</js>, <js>"write"</js>);
83   * </p>
84   *
85   * <h5 class='section'>Common Use Cases:</h5>
86   * <ul class='spaced-list'>
87   * 	<li>Caching permissions indexed by tenant, user, resource, and action
88   * 	<li>Storing metrics indexed by environment, service, instance, and metric name
89   * 	<li>Mapping configuration by tenant, environment, service, and config key
90   * 	<li>Tracking analytics by date, source, campaign, and metric
91   * 	<li>Caching query results indexed by database, schema, table, and query type
92   * </ul>
93   *
94   * <h5 class='section'>Key Hashing:</h5>
95   * <p>
96   * The composite key is hashed using the formula:
97   * <p class='bjava'>
98   * 	hash = 31 * (31 * (31 * k1.hashCode() + k2.hashCode()) + k3.hashCode()) + k4.hashCode()
99   * </p>
100  * <p>
101  * Null keys are treated as having a hash code of 0. Keys are considered equal if all components
102  * are equal according to {@link Object#equals(Object)}.
103  *
104  * <h5 class='section'>Thread Safety:</h5>
105  * <p>
106  * This class inherits all thread-safety properties from {@link ConcurrentHashMap}. Multiple threads
107  * can safely read and write to the map concurrently without external synchronization.
108  *
109  * <h5 class='section'>Performance:</h5>
110  * <ul class='spaced-list'>
111  * 	<li>Average O(1) lookup and insertion time (inherited from {@link ConcurrentHashMap})
112  * 	<li>Minimal object allocation: composite keys are created only during put/get operations
113  * 	<li>Lock-free reads for existing entries (inherited from {@link ConcurrentHashMap})
114  * </ul>
115  *
116  * <h5 class='section'>See Also:</h5>
117  * <ul>
118  * 	<li class='jc'>{@link ConcurrentHashMap2Key}
119  * 	<li class='jc'>{@link ConcurrentHashMap3Key}
120  * 	<li class='jc'>{@link ConcurrentHashMap5Key}
121  * 	<li class='link'><a class="doclink" href="../../../../../index.html#juneau-common">Overview &gt; juneau-common</a>
122  * </ul>
123  *
124  * @param <K1> The first key component type.
125  * @param <K2> The second key component type.
126  * @param <K3> The third key component type.
127  * @param <K4> The fourth key component type.
128  * @param <V> The value type.
129  * @serial exclude
130  */
131 public class ConcurrentHashMap4Key<K1,K2,K3,K4,V> extends ConcurrentHashMap<Tuple4<K1,K2,K3,K4>,V> {
132 
133 	private static final long serialVersionUID = 1L;
134 
135 	/**
136 	 * Retrieves the value associated with the specified four-part key.
137 	 *
138 	 * @param key1 First key component. Must not be <jk>null</jk>.
139 	 * @param key2 Second key component. Must not be <jk>null</jk>.
140 	 * @param key3 Third key component. Must not be <jk>null</jk>.
141 	 * @param key4 Fourth key component. Must not be <jk>null</jk>.
142 	 * @return The value associated with the key, or <jk>null</jk> if not found.
143 	 * @throws IllegalArgumentException if any key is <jk>null</jk>.
144 	 */
145 	public V get(K1 key1, K2 key2, K3 key3, K4 key4) {
146 		AssertionUtils.assertArgsNotNull("key1", key1, "key2", key2, "key3", key3, "key4", key4);
147 		return super.get(Tuple4.of(key1, key2, key3, key4));
148 	}
149 
150 	/**
151 	 * Associates the specified value with the specified four-part key in this map.
152 	 *
153 	 * @param key1 First key component. Must not be <jk>null</jk>.
154 	 * @param key2 Second key component. Must not be <jk>null</jk>.
155 	 * @param key3 Third key component. Must not be <jk>null</jk>.
156 	 * @param key4 Fourth key component. Must not be <jk>null</jk>.
157 	 * @param value The value to associate with the key.
158 	 * @return The previous value associated with the key, or <jk>null</jk> if there was no mapping.
159 	 * @throws IllegalArgumentException if any key is <jk>null</jk>.
160 	 */
161 	public V put(K1 key1, K2 key2, K3 key3, K4 key4, V value) {
162 		AssertionUtils.assertArgsNotNull("key1", key1, "key2", key2, "key3", key3, "key4", key4);
163 		return super.put(Tuple4.of(key1, key2, key3, key4), value);
164 	}
165 }
