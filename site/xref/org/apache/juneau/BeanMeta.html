40  /**
41   * Encapsulates all access to the properties of a bean class (like a souped-up {@link java.beans.BeanInfo}).
42   *
43   * <h5 class='topic'>Description</h5>
44   *
45   * Uses introspection to find all the properties associated with this class.  If the {@link Bean @Bean} annotation
46   * 	is present on the class, then that information is used to determine the properties on the class.
47   * Otherwise, the {@code BeanInfo} functionality in Java is used to determine the properties on the class.
48   *
49   * <h5 class='topic'>Bean property ordering</h5>
50   *
51   * The order of the properties are as follows:
52   * <ul class='spaced-list'>
53   * 	<li>
54   * 		If {@link Bean @Bean} annotation is specified on class, then the order is the same as the list of properties
55   * 		in the annotation.
56   * 	<li>
57   * 		If {@link Bean @Bean} annotation is not specified on the class, then the order is based on the following.
58   * 		<ul>
59   * 			<li>Public fields (same order as {@code Class.getFields()}).
60   * 			<li>Properties returned by {@code BeanInfo.getPropertyDescriptors()}.
61   * 			<li>Non-standard getters/setters with {@link Beanp @Beanp} annotation defined on them.
62   * 		</ul>
63   * </ul>
64   *
65   *
66   * @param <T> The class type that this metadata applies to.
67   */
68  public class BeanMeta<T> {

110 		/*
111 		 * Returns true if this method matches the class type of the specified property.
112 		 * Only meant to be used for setters.
113 		 */
114 		private boolean matchesPropertyType(BeanPropertyMeta.Builder b) {
115 			if (b == null)
116 				return false;
117 
118 			// Don't do further validation if this is the "*" bean property.
119 			if ("*".equals(b.name))
120 				return true;
121 
122 			// Get the bean property type from the getter/field.
123 			var pt = (Class<?>)null;
124 			if (nn(b.getter))
125 				pt = b.getter.getReturnType();
126 			else if (nn(b.field))
127 				pt = b.field.getType();
128 
129 			// Matches if only a setter is defined.
130 			if (pt == null)
131 				return true;
132 
133 			// Doesn't match if not same type or super type as getter/field.
134 			if (! type.isParentOf(pt))
135 				return false;
136 
137 			// If a setter was previously set, only use this setter if it's a closer
138 			// match (e.g. prev type is a superclass of this type).
139 			if (b.setter == null)
140 				return true;
141 
142 			return type.isStrictChildOf(b.setter.getParameterTypes()[0]);
143 		}
144 	}

154 	/**
155 	 * Creates a {@link BeanMeta} instance for the specified class metadata.
156 	 *
157 	 * <p>
158 	 * This is a factory method that attempts to create bean metadata for a class. If the class is determined to be a bean,
159 	 * the returned {@link BeanMetaValue} will contain the {@link BeanMeta} instance and a <jk>null</jk> reason.
160 	 * If the class is not a bean, the returned value will contain <jk>null</jk> for the bean metadata and a non-null
161 	 * string explaining why it's not a bean.
162 	 *
163 	 * <h5 class='section'>Parameters:</h5>
164 	 * <ul class='spaced-list'>
165 	 * 	<li><b>cm</b> - The class metadata for the class to create bean metadata for.
166 	 * 	<li><b>bf</b> - Optional bean filter to apply. Can be <jk>null</jk>.
167 	 * 	<li><b>pNames</b> - Explicit list of property names and order. If <jk>null</jk>, properties are determined automatically.
168 	 * 	<li><b>implClassConstructor</b> - Optional constructor to use if one cannot be found. Can be <jk>null</jk>.
169 	 * </ul>
170 	 *
171 	 * <h5 class='section'>Return Value:</h5>
172 	 * <p>
173 	 * Returns a {@link BeanMetaValue} containing:
174 	 * <ul>
175 	 * 	<li><b>beanMeta</b> - The bean metadata if the class is a bean, or <jk>null</jk> if it's not.
176 	 * 	<li><b>notABeanReason</b> - A string explaining why the class is not a bean, or <jk>null</jk> if it is a bean.
177 	 * </ul>
178 	 *
179 	 * <h5 class='section'>Exception Handling:</h5>
180 	 * <p>
181 	 * If a {@link RuntimeException} is thrown during bean metadata creation, it is caught and the exception message
182 	 * is returned as the <c>notABeanReason</c> with <jk>null</jk> for the bean metadata.
183 	 *
184 	 * <h5 class='section'>Example:</h5>
185 	 * <p class='bjava'>
186 	 * 	<jc>// Create bean metadata for a class</jc>
187 	 * 	ClassMeta&lt;Person&gt; <jv>cm</jv> = <jv>beanContext</jv>.getClassMeta(Person.<jk>class</jk>);
188 	 * 	BeanMetaValue&lt;Person&gt; <jv>result</jv> = BeanMeta.<jsm>create</jsm>(<jv>cm</jv>, <jk>null</jk>, <jk>null</jk>, <jk>null</jk>);
189 	 *
190 	 * 	<jc>// Check if it's a bean</jc>
191 	 * 	<jk>if</jk> (<jv>result</jv>.beanMeta() != <jk>null</jk>) {
192 	 * 		BeanMeta&lt;Person&gt; <jv>bm</jv> = <jv>result</jv>.beanMeta();
193 	 * 		<jc>// Use the bean metadata...</jc>
194 	 * 	} <jk>else</jk> {
195 	 * 		String <jv>reason</jv> = <jv>result</jv>.notABeanReason();
196 	 * 		<jc>// Handle the case where it's not a bean...</jc>
197 	 * 	}
198 	 * </p>
199 	 *
200 	 * @param <T> The class type.
201 	 * @param cm The class metadata for the class to create bean metadata for.
202 	 * @param implClass
203 	 * @param bf Optional bean filter to apply. Can be <jk>null</jk>.
204 	 * @param implClassConstructor Optional constructor to use if one cannot be found. Can be <jk>null</jk>.
205 	 * @return A {@link BeanMetaValue} containing the bean metadata (if successful) or a reason why it's not a bean.
206 	 */
207 	public static <T> BeanMetaValue<T> create(ClassMeta<T> cm, ClassInfo implClass) {
208 		try {
209 			var bc = cm.getBeanContext();
210 			var ap = bc.getAnnotationProvider();
211 
212 			// Sanity checks first.
213 			if (bc.isNotABean(cm))
214 				return notABean("Class matches exclude-class list");
215 
216 			if (bc.isBeansRequireSerializable() && ! cm.isChildOf(Serializable.class) && ! ap.has(Bean.class, cm))
217 				return notABean("Class is not serializable");
218 
219 			if (ap.has(BeanIgnore.class, cm))
220 				return notABean("Class is annotated with @BeanIgnore");
221 
222 			if ((! bc.getBeanClassVisibility().isVisible(cm.getModifiers()) || cm.isAnonymousClass()) && ! ap.has(Bean.class, cm))
223 				return notABean("Class is not public");
224 
225 			var bm = new BeanMeta<>(cm, findBeanFilter(cm), null, implClass);
226 
227 			if (nn(bm.notABeanReason))
228 				return notABean(bm.notABeanReason);
229 
230 			return new BeanMetaValue<>(bm, null);
231 		} catch (RuntimeException e) {
232 			return new BeanMetaValue<>(null, e.getMessage());
233 		}
234 	}

237 	 * Extracts the property name from {@link Beanp @Beanp} or {@link Name @Name} annotations.
238 	 *
239 	 * <p>
240 	 * If {@link Name @Name} annotations are present, returns the value from the last one.
241 	 * Otherwise, searches through {@link Beanp @Beanp} annotations and returns the first non-empty
242 	 * {@link Beanp#value() value()} or {@link Beanp#name() name()} found.
243 	 *
244 	 * @param p List of {@link Beanp @Beanp} annotations.
245 	 * @param n List of {@link Name @Name} annotations.
246 	 * @return The property name, or <jk>null</jk> if no name is found.
247 	 */
248 	private static String bpName(List<Beanp> p, List<Name> n) {
249 		if (p.isEmpty() && n.isEmpty())
250 			return null;
251 		if (! n.isEmpty())
252 			return last(n).value();
253 
254 		var name = Value.of(p.isEmpty() ? null : "");
255 		p.forEach(x -> {
256 			if (! x.value().isEmpty())
257 				name.set(x.value());
258 			if (! x.name().isEmpty())
259 				name.set(x.name());
260 		});
261 
262 		return name.orElse(null);
263 	}

265 	/*
266 	 * Finds and creates the bean filter for the specified class metadata.
267 	 *
268 	 * <p>
269 	 * Searches for {@link Bean @Bean} annotations on the class and its parent classes/interfaces. If found, creates a
270 	 * {@link BeanFilter} that applies the configuration from those annotations.
271 	 *
272 	 * <p>
273 	 * When multiple {@link Bean @Bean} annotations are found (e.g., on a parent class and a child class), they are
274 	 * applied in reverse order (parent classes first, then child classes). This ensures that child class annotations
275 	 * override parent class annotations, allowing child classes to customize or extend the bean configuration.
276 	 *
277 	 * <p>
278 	 * The bean filter controls various aspects of bean serialization and parsing, such as:
279 	 * <ul>
280 	 * 	<li>Property inclusion/exclusion lists
281 	 * 	<li>Property ordering and sorting
282 	 * 	<li>Type name mapping for dictionary lookups
283 	 * 	<li>Fluent setter detection
284 	 * 	<li>Read-only and write-only property definitions
285 	 * </ul>
286 	 *
287 	 * @param <T> The class type.
288 	 * @param cm The class metadata to find the filter for.
289 	 * @return The bean filter, or <jk>null</jk> if no {@link Bean @Bean} annotations are found on the class or its hierarchy.
290 	 * @see Bean
291 	 * @see BeanFilter
292 	 */
293 	private static <T> BeanFilter findBeanFilter(ClassMeta<T> cm) {
294 		var ap = cm.getBeanContext().getAnnotationProvider();
295 		var l = ap.find(Bean.class, cm);
296 		if (l.isEmpty())
297 			return null;
298 		return BeanFilter.create(cm).applyAnnotations(reverse(l.stream().map(AnnotationInfo::inner).toList())).build();
299 	}

301 	/*
302 	 * Extracts the property name from a single {@link Beanp @Beanp} or {@link Name @Name} annotation.
303 	 *
304 	 * <p>
305 	 * For {@link Beanp @Beanp} annotations, returns the first non-empty value found in the following order:
306 	 * <ol>
307 	 * 	<li>{@link Beanp#name() name()}
308 	 * 	<li>{@link Beanp#value() value()}
309 	 * </ol>
310 	 *
311 	 * <p>
312 	 * For {@link Name @Name} annotations, returns the {@link Name#value() value()}.
313 	 *
314 	 * <p>
315 	 * This method is used to extract property names from individual annotations, typically when processing
316 	 * annotation lists in stream operations.
317 	 *
318 	 * @param ai The annotation info containing either a {@link Beanp @Beanp} or {@link Name @Name} annotation.
319 	 * @return The property name extracted from the annotation, or <jk>null</jk> if no name is found.
320 	 * @see #bpName(List, List)
321 	 */
322 	private static String name(AnnotationInfo<?> ai) {
323 		if (ai.isType(Beanp.class)) {
324 			var p = ai.cast(Beanp.class).inner();
325 			if (isNotEmpty(p.name()))
326 				return p.name();
327 			if (isNotEmpty(p.value()))
328 				return p.value();
329 		} else {
330 			var n = ai.cast(Name.class).inner();
331 			if (isNotEmpty(n.value()))
332 				return n.value();
333 		}
334 		return null;
335 	}

365 	/**
366 	 * Constructor.
367 	 *
368 	 * <p>
369 	 * Creates a new {@link BeanMeta} instance for the specified class metadata. This constructor performs
370 	 * introspection to discover all bean properties, methods, and fields in the class hierarchy.
371 	 *
372 	 * <p>
373 	 * The bean metadata is built by:
374 	 * <ul>
375 	 * 	<li>Finding all bean fields in the class hierarchy
376 	 * 	<li>Finding all bean methods (getters, setters, extraKeys) in the class hierarchy
377 	 * 	<li>Determining the appropriate constructor for bean instantiation
378 	 * 	<li>Building the class hierarchy for property discovery
379 	 * 	<li>Creating the bean registry for dictionary name resolution
380 	 * 	<li>Validating and filtering properties based on bean filter settings
381 	 * </ul>
382 	 *
383 	 * @param cm The class metadata for the bean class.
384 	 * @param bf Optional bean filter to apply. Can be <jk>null</jk>.
385 	 * @param pNames Explicit list of property names and order. If <jk>null</jk>, properties are determined automatically.
386 	 * @param implClass Optional implementation class constructor to use if one cannot be found. Can be <jk>null</jk>.
387 	 */
388 	@SuppressWarnings("rawtypes")
389 	protected BeanMeta(ClassMeta<T> cm, BeanFilter bf, String[] pNames, ClassInfo implClass) {
390 		classMeta = cm;
391 		beanContext = cm.getBeanContext();
392 		beanFilter = bf;
393 		implClassConstructor = opt(implClass).map(x -> x.getPublicConstructor(x2 -> x2.hasNumParameters(0)).orElse(null)).orElse(null);
394 		fluentSetters = beanContext.isFindFluentSetters() || (nn(bf) && bf.isFluentSetters());
395 		stopClass = opt(bf).map(x -> (Class)x.getStopClass()).orElse(Object.class);
396 		beanRegistry = memoize(()->findBeanRegistry());
397 		classHierarchy = memoize(()->findClassHierarchy());
398 		beanConstructor = findBeanConstructor();
399 
400 		// Local variables for initialization
401 		var ap = beanContext.getAnnotationProvider();
402 		var c = cm.inner();
403 		var ci = cm;
404 		var _notABeanReason = (String)null;
405 		var _properties = Value.<Map<String,BeanPropertyMeta>>empty();
406 		var _hiddenProperties = CollectionUtils.<String,BeanPropertyMeta>map();
407 		var _getterProps = CollectionUtils.<Method,String>map();
408 		var _setterProps = CollectionUtils.<Method,String>map();
409 		var _dynaProperty = Value.<BeanPropertyMeta>empty();
410 		var _sortProperties = false;
411 		var ba = ap.find(Bean.class, cm);
412 		var propertyNamer = opt(bf).map(x -> x.getPropertyNamer()).orElse(beanContext.getPropertyNamer());
413 
414 		this.typePropertyName = ba.stream().map(x -> x.inner().typePropertyName()).filter(Utils::isNotEmpty).findFirst().orElseGet(() -> beanContext.getBeanTypePropertyName());
415 
416 		// Check if constructor is required but not found
417 		if (! beanConstructor.constructor().isPresent() && bf == null && beanContext.isBeansRequireDefaultConstructor())
418 			_notABeanReason = "Class does not have the required no-arg constructor";
419 
420 		var bfo = opt(bf);
421 		var fixedBeanProps = bfo.map(x -> x.getProperties()).orElse(sete());
422 
423 		try {
424 			Map<String,BeanPropertyMeta.Builder> normalProps = map();  // NOAI
425 
426 			// First populate the properties with those specified in the bean annotation to
427 			// ensure that ordering first.
428 			fixedBeanProps.forEach(x -> normalProps.put(x, BeanPropertyMeta.builder(this, x)));
429 
430 			if (beanContext.isUseJavaBeanIntrospector()) {
431 				var c2 = bfo.map(x -> (Class)x.getInterfaceClass()).filter(Objects::nonNull).orElse(c);
432 				var bi = (BeanInfo)null;
433 				if (! c2.isInterface())
434 					bi = Introspector.getBeanInfo(c2, stopClass);
435 				else
436 					bi = Introspector.getBeanInfo(c2, null);
437 				if (nn(bi)) {
438 					for (var pd : bi.getPropertyDescriptors()) {
439 						normalProps.computeIfAbsent(pd.getName(), n -> BeanPropertyMeta.builder(this, n)).setGetter(pd.getReadMethod()).setSetter(pd.getWriteMethod());
440 					}
441 				}
442 
443 			} else /* Use 'better' introspection */ {
444 
445 				findBeanFields().forEach(x -> {
446 					var name = ap.find(x).stream()
447 						.filter(x2 -> x2.isType(Beanp.class) || x2.isType(Name.class))
448 						.map(x2 -> name(x2))
449 						.filter(Objects::nonNull)
450 						.findFirst()
451 						.orElse(propertyNamer.getPropertyName(x.getName()));
452 					if (nn(name)) {
453 						normalProps.computeIfAbsent(name, n->BeanPropertyMeta.builder(this, n)).setField(x.inner());
454 					}
455 				});
456 
457 				var bms = findBeanMethods();
458 
459 				// Iterate through all the getters.
460 				bms.forEach(x -> {
461 					var pn = x.propertyName;
462 					var m = x.method;
463 					var mi = info(m);
464 					var bpm = normalProps.computeIfAbsent(pn, k -> new BeanPropertyMeta.Builder(this, k));
465 
466 					if (x.methodType == GETTER) {
467 						// Two getters.  Pick the best.
468 						if (nn(bpm.getter)) {
469 							if (! ap.has(Beanp.class, mi) && ap.has(Beanp.class, info(bpm.getter))) {
470 								m = bpm.getter;  // @Beanp annotated method takes precedence.
471 							} else if (m.getName().startsWith("is") && bpm.getter.getName().startsWith("get")) {
472 								m = bpm.getter;  // getX() overrides isX().
473 							}
474 						}
475 						bpm.setGetter(m);
476 					}
477 				});
478 
479 				// Now iterate through all the setters.
480 				bms.stream().filter(x -> eq(x.methodType, SETTER)).forEach(x -> {
481 					var bpm = normalProps.get(x.propertyName);
482 					if (x.matchesPropertyType(bpm))
483 						bpm.setSetter(x.method);
484 				});
485 
486 				// Now iterate through all the extraKeys.
487 				bms.stream().filter(x -> eq(x.methodType, EXTRAKEYS)).forEach(x -> normalProps.get(x.propertyName).setExtraKeys(x.method));
488 			}
489 
490 			var typeVarImpls = ClassUtils.findTypeVarImpls(c);
491 
492 			// Eliminate invalid properties, and set the contents of getterProps and setterProps.
493 			var readOnlyProps = bfo.map(x -> x.getReadOnlyProperties()).orElse(sete());
494 			var writeOnlyProps = bfo.map(x -> x.getWriteOnlyProperties()).orElse(sete());
495 			for (var i = normalProps.values().iterator(); i.hasNext();) {
496 				var p = i.next();
497 				try {
498 					if (p.field == null)
499 						p.setInnerField(findInnerBeanField(p.name));
500 
501 					if (p.validate(beanContext, beanRegistry.get(), typeVarImpls, readOnlyProps, writeOnlyProps)) {
502 
503 						if (nn(p.getter))
504 							_getterProps.put(p.getter, p.name);
505 
506 						if (nn(p.setter))
507 							_setterProps.put(p.setter, p.name);
508 
509 					} else {
510 						i.remove();
511 					}
512 				} catch (ClassNotFoundException e) {
513 					throw bex(c, lm(e));
514 				}
515 			}
516 
517 			// Check for missing properties.
518 			fixedBeanProps.stream().filter(x -> ! normalProps.containsKey(x)).findFirst().ifPresent(x -> { throw bex(c, "The property ''{0}'' was defined on the @Bean(properties=X) annotation of class ''{1}'' but was not found on the class definition.", x, ci.getNameSimple()); });
519 
520 			// Mark constructor arg properties.

1018 	 * 		<ul>

1025 	 * 		<ul>

1027 	 * 			<li><c>withX(value)</c> - fluent setter pattern (returns the bean type)
1028 	 * 			<li>Methods annotated with {@link Beanp @Beanp} or {@link Name @Name}
1029 	 * 			<li>Methods with {@link Beanp @Beanp} annotation with value <js>"*"</js> that accept a Map
1030 	 * 			<li>Fluent setters (if enabled) - methods that return the bean type and accept one parameter
1031 	 * 		</ul>
1032 	 * 	<li><b>ExtraKeys:</b> Methods with {@link Beanp @Beanp} annotation with value <js>"*"</js> that return a Collection
1033 	 * </ul>

1035 	 * <p>
1036 	 * Methods are filtered based on:
1037 	 * <ul>

1043 	 * </ul>

1046 	 * Property names are determined from:
1047 	 * <ul>
1048 	 * 	<li>{@link Beanp @Beanp} or {@link Name @Name} annotations (if present)
1049 	 * 	<li>Otherwise, derived from the method name using the provided {@link PropertyNamer}
1050 	 * </ul>

1052 	 * @return A list of {@link BeanMethod} objects representing all found bean methods.
1053 	 */

1058 		var v = beanContext.getBeanMethodVisibility();

1065 					continue;
1066

1068

1070 					continue;
1071 
1072 				if (mm.stream().anyMatch(m2 -> ap.find(Transient.class, m2, SELF).stream().map(x -> x.inner().value()).findFirst().orElse(false)))
1073 					continue;
1074 
1075 				var beanps = ap.find(Beanp.class, m).stream().map(AnnotationInfo::inner).toList();
1076 				var names = ap.find(Name.class, m).stream().map(AnnotationInfo::inner).toList();
1077 
1078 				if (! (m.isVisible(v) || isNotEmpty(beanps) || isNotEmpty(names)))
1079 					continue;
1080 
1081 				var n = m.getSimpleName();
1082 
1083 				var params = m.getParameters();

1204 	 * <p>

1210 	 * <ol>

1239 	 * <p>

1263 	 * Finds the dictionary name (type name) for this bean class.
1264 	 *
1265 	 * <p>
1266 	 * The dictionary name is used in serialized forms to identify the specific type of a bean instance, enabling

1268 	 * that are part of an inheritance hierarchy.
1269 	 *
1270 	 * <p>
1271 	 * The dictionary name is determined by searching in the following order of precedence:
1272 	 * <ol>

1278 	 * 		second one, skipping the class itself) and checks if any of their bean registries contain a type name
1279 	 * 		for this class.
1280 	 * 	<li><b>{@link Bean @Bean} annotation:</b> If the class has a {@link Bean @Bean} annotation with a non-empty
1281 	 * 		{@link Bean#typeName() typeName()}, that value is used.

1287 	 * <js>"_type"</js>) so that parsers can determine the correct class to instantiate when deserializing.
