38  /**
39   * Encapsulates all access to the properties of a bean class (like a souped-up {@link java.beans.BeanInfo}).
40   *
41   * <h5 class='topic'>Description</h5>
42   *
43   * Uses introspection to find all the properties associated with this class.  If the {@link Bean @Bean} annotation
44   * 	is present on the class, then that information is used to determine the properties on the class.
45   * Otherwise, the {@code BeanInfo} functionality in Java is used to determine the properties on the class.
46   *
47   * <h5 class='topic'>Bean property ordering</h5>
48   *
49   * The order of the properties are as follows:
50   * <ul class='spaced-list'>
51   * 	<li>
52   * 		If {@link Bean @Bean} annotation is specified on class, then the order is the same as the list of properties
53   * 		in the annotation.
54   * 	<li>
55   * 		If {@link Bean @Bean} annotation is not specified on the class, then the order is based on the following.
56   * 		<ul>
57   * 			<li>Public fields (same order as {@code Class.getFields()}).
58   * 			<li>Properties returned by {@code BeanInfo.getPropertyDescriptors()}.
59   * 			<li>Non-standard getters/setters with {@link Beanp @Beanp} annotation defined on them.
60   * 		</ul>
61   * </ul>
62   *
63   * <h5 class='section'>See Also:</h5><ul>
64  
65   * </ul>
66   *
67   * @param <T> The class type that this metadata applies to.
68   */
69  public class BeanMeta<T> {

74  	private static class BeanMethod {
75  		String propertyName;
76  		MethodType methodType;
77  		Method method;
78  		ClassInfo type;
79  
80  		BeanMethod(String propertyName, MethodType type, Method method) {
81  			this.propertyName = propertyName;
82  			this.methodType = type;
83  			this.method = method;
84  			if (type == MethodType.SETTER)
85  				this.type = ClassInfo.of(method.getParameterTypes()[0]);
86  			else
87  				this.type = ClassInfo.of(method.getReturnType());
88  		}
89  
90  		@Override /* Overridden from Object */
91  		public String toString() {
92  			return method.toString();
93  		}
94  
95  		/*
96  		 * Returns true if this method matches the class type of the specified property.
97  		 * Only meant to be used for setters.
98  		 */
99  		boolean matchesPropertyType(BeanPropertyMeta.Builder b) {
100 			if (b == null)
101 				return false;
102 
103 			// Don't do further validation if this is the "*" bean property.
104 			if ("*".equals(b.name))
105 				return true;
106 
107 			// Get the bean property type from the getter/field.
108 			Class<?> pt = null;
109 			if (nn(b.getter))
110 				pt = b.getter.getReturnType();
111 			else if (nn(b.field))
112 				pt = b.field.getType();
113 
114 			// Matches if only a setter is defined.
115 			if (pt == null)
116 				return true;
117 
118 			// Doesn't match if not same type or super type as getter/field.
119 			if (! type.isParentOf(pt))
120 				return false;
121 
122 			// If a setter was previously set, only use this setter if it's a closer
123 			// match (e.g. prev type is a superclass of this type).
124 			if (b.setter == null)
125 				return true;
126 
127 			return type.isStrictChildOf(b.setter.getParameterTypes()[0]);
128 		}
129 	}

158 		private String findDictionaryName(ClassMeta<?> cm) {
159 			BeanRegistry br = cm.getBeanRegistry();
160 			if (nn(br)) {
161 				String s = br.getTypeName(this.classMeta);
162 				if (nn(s))
163 					return s;
164 			}
165 			Class<?> pcm = cm.innerClass.getSuperclass();
166 			if (nn(pcm)) {
167 				String s = findDictionaryName(ctx.getClassMeta(pcm));
168 				if (nn(s))
169 					return s;
170 			}
171 			for (var icm : cm.innerClass.getInterfaces()) {
172 				String s = findDictionaryName(ctx.getClassMeta(icm));
173 				if (nn(s))
174 					return s;
175 			}
176 			return null;
177 		}

179 		/*
180 		 * Returns the property name of the specified field if it's a valid property.
181 		 * Returns null if the field isn't a valid property.
182 		 */
183 		private String findPropertyName(Field f) {
184 			List<Beanp> lp = list();
185 			List<Name> ln = list();
186 			// Inline Context.forEachAnnotation() calls
187 			ctx.getAnnotationProvider().xfind(Beanp.class, f).map(x -> x.inner()).filter(x -> true).forEach(x -> lp.add(x));
188 			ctx.getAnnotationProvider().xfind(Name.class, f).map(x -> x.inner()).filter(x -> true).forEach(x -> ln.add(x));
189 			String name = bpName(lp, ln);
190 			if (isNotEmpty(name))
191 				return name;
192 			return propertyNamer.getPropertyName(f.getName());
193 		}

195 		String init(BeanMeta<T> beanMeta) {
196 			var c = classMeta.getInnerClass();
197 			var ci = classMeta.getInfo();
198 			var ap = ctx.getAnnotationProvider();
199 
200 			try {
201 				var conVis = ctx.getBeanConstructorVisibility();
202 				var cVis = ctx.getBeanClassVisibility();
203 				var mVis = ctx.getBeanMethodVisibility();
204 				var fVis = ctx.getBeanFieldVisibility();
205 
206 				List<Class<?>> bdClasses = list();
207 				if (nn(beanFilter) && nn(beanFilter.getBeanDictionary()))
208 					addAll(bdClasses, beanFilter.getBeanDictionary());
209 
210 				Value<String> typeName = Value.empty();
211 				classMeta.forEachAnnotation(Bean.class, x -> isNotEmpty(x.typeName()), x -> typeName.set(x.typeName()));
212 				if (typeName.isPresent())
213 					bdClasses.add(classMeta.innerClass);
214 				this.beanRegistry = new BeanRegistry(ctx, null, bdClasses.toArray(new Class<?>[bdClasses.size()]));
215 
216 				Value<String> typePropertyName = Value.empty();
217 				classMeta.forEachAnnotation(Bean.class, x -> isNotEmpty(x.typePropertyName()), x -> typePropertyName.set(x.typePropertyName()));
218 				this.typePropertyName = typePropertyName.orElseGet(() -> ctx.getBeanTypePropertyName());
219 
220 				fluentSetters = (ctx.isFindFluentSetters() || (nn(beanFilter) && beanFilter.isFluentSetters()));
221 
222 				// If @Bean.interfaceClass is specified on the parent class, then we want
223 				// to use the properties defined on that class, not the subclass.
224 				Class<?> c2 = (nn(beanFilter) && nn(beanFilter.getInterfaceClass()) ? beanFilter.getInterfaceClass() : c);
225 
226 				Class<?> stopClass = (nn(beanFilter) ? beanFilter.getStopClass() : Object.class);
227 				if (stopClass == null)
228 					stopClass = Object.class;
229 
230 				Map<String,BeanPropertyMeta.Builder> normalProps = map();
231 
232 				var hasBean = ap.has(Bean.class, ci);
233 				var hasBeanIgnore = ap.has(BeanIgnore.class, ci);
234 
235 				/// See if this class matches one the patterns in the exclude-class list.
236 				if (ctx.isNotABean(c))
237 					return "Class matches exclude-class list";
238 
239 				if (! hasBean && ! (cVis.isVisible(c.getModifiers()) || c.isAnonymousClass()))
240 					return "Class is not public";
241 
242 				if (hasBeanIgnore)
243 					return "Class is annotated with @BeanIgnore";
244 
245 				// Make sure it's serializable.
246 				if (beanFilter == null && ctx.isBeansRequireSerializable() && ! ci.isChildOf(Serializable.class))
247 					return "Class is not serializable";
248 
249 				// Look for @Beanc constructor on public constructors.
250 				ci.getPublicConstructors().stream().filter(x -> ap.has(Beanc.class, x)).forEach(x -> {
251 					if (nn(constructor))
252 						throw new BeanRuntimeException(c, "Multiple instances of '@Beanc' found.");
253 					constructor = x;
254 					constructorArgs = new String[0];
255 					ap.find(Beanc.class, x).map(x2 -> x2.inner().properties()).filter(StringUtils::isNotBlank).forEach(z -> constructorArgs = splita(z));
256 					if (! x.hasNumParameters(constructorArgs.length)) {
257 						if (constructorArgs.length != 0)
258 						throw new BeanRuntimeException(c, "Number of properties defined in '@Beanc' annotation does not match number of parameters in constructor.");
259 						constructorArgs = new String[x.getParameterCount()];
260 						var i = IntegerValue.create();
261 						x.getParameters().forEach(pi -> {
262 							String pn = pi.getName();
263 							if (pn == null)
264 								throw new BeanRuntimeException(c, "Could not find name for parameter #{0} of constructor ''{1}''", i, x.getFullName());
265 							constructorArgs[i.getAndIncrement()] = pn;
266 						});
267 					}
268 					constructor.setAccessible();
269 				});
270 
271 				// Look for @Beanc on all other constructors.
272 				if (constructor == null) {
273 					ci.getDeclaredConstructors().stream().filter(x -> ctx.getAnnotationProvider().xfind(Beanc.class, x.inner()).findAny().isPresent()).forEach(x -> {
274 						if (nn(constructor))
275 							throw new BeanRuntimeException(c, "Multiple instances of '@Beanc' found.");
276 						constructor = x;
277 						constructorArgs = new String[0];
278 						ctx.getAnnotationProvider().xfind(Beanc.class, x.inner()).map(x2 -> x2.inner()).filter(y -> ! y.properties().isEmpty()).forEach(z -> constructorArgs = splita(z.properties()));
279 						if (! x.hasNumParameters(constructorArgs.length)) {
280 							if (constructorArgs.length != 0)
281 							throw new BeanRuntimeException(c, "Number of properties defined in '@Beanc' annotation does not match number of parameters in constructor.");
282 						constructorArgs = new String[x.getParameterCount()];
283 						var i = IntegerValue.create();
284 						x.getParameters().forEach(y -> {
285 							String pn = y.getName();
286 							if (pn == null)
287 								throw new BeanRuntimeException(c, "Could not find name for parameter #{0} of constructor ''{1}''", i, x.getFullName());
288 								constructorArgs[i.getAndIncrement()] = pn;
289 							});
290 						}
291 						constructor.setAccessible();
292 					});
293 				}
294 
295 				// If this is an interface, look for impl classes defined in the context.
296 				if (constructor == null)
297 					constructor = implClassConstructor;
298 
299 				if (constructor == null)
300 					constructor = ci.getNoArgConstructor(hasBean ? Visibility.PRIVATE : conVis).orElse(null);
301 
302 				if (constructor == null && beanFilter == null && ctx.isBeansRequireDefaultConstructor())
303 					return "Class does not have the required no-arg constructor";
304 
305 				if (nn(constructor))
306 					constructor.setAccessible();
307 
308 				// Explicitly defined property names in @Bean annotation.
309 				Set<String> fixedBeanProps = set();
310 				Set<String> bpi = set();
311 				Set<String> bpx = set();
312 				Set<String> bpro = set();
313 				Set<String> bpwo = set();
314 
315 				Set<String> filterProps = set();  // Names of properties defined in @Bean(properties)
316 
317 				if (nn(beanFilter)) {
318 
319 					Set<String> bfbpi = beanFilter.getProperties();
320 
321 					filterProps.addAll(bfbpi);
322 
323 					// Get the 'properties' attribute if specified.
324 					if (bpi.isEmpty())
325 						fixedBeanProps.addAll(bfbpi);
326 
327 					if (nn(beanFilter.getPropertyNamer()))
328 						propertyNamer = beanFilter.getPropertyNamer();
329 
330 					bpro.addAll(beanFilter.getReadOnlyProperties());
331 					bpwo.addAll(beanFilter.getWriteOnlyProperties());
332 				}
333 
334 				fixedBeanProps.addAll(bpi);
335 
336 				if (propertyNamer == null)
337 					propertyNamer = ctx.getPropertyNamer();
338 
339 				// First populate the properties with those specified in the bean annotation to
340 				// ensure that ordering first.
341 				fixedBeanProps.forEach(x -> normalProps.put(x, BeanPropertyMeta.builder(beanMeta, x)));
342 
343 				if (ctx.isUseJavaBeanIntrospector()) {
344 					BeanInfo bi = null;
345 					if (! c2.isInterface())
346 						bi = Introspector.getBeanInfo(c2, stopClass);
347 					else
348 						bi = Introspector.getBeanInfo(c2, null);
349 					if (nn(bi)) {
350 						for (var pd : bi.getPropertyDescriptors()) {
351 							String name = pd.getName();
352 							if (! normalProps.containsKey(name))
353 								normalProps.put(name, BeanPropertyMeta.builder(beanMeta, name));
354 							normalProps.get(name).setGetter(pd.getReadMethod()).setSetter(pd.getWriteMethod());
355 						}
356 					}
357 
358 				} else /* Use 'better' introspection */ {
359 
360 					findBeanFields(ctx, c2, stopClass, fVis).forEach(x -> {
361 						String name = findPropertyName(x);
362 						if (nn(name)) {
363 							if (! normalProps.containsKey(name))
364 								normalProps.put(name, BeanPropertyMeta.builder(beanMeta, name));
365 							normalProps.get(name).setField(x);
366 						}
367 					});
368 
369 					List<BeanMethod> bms = findBeanMethods(ctx, c2, stopClass, mVis, propertyNamer, fluentSetters);
370 
371 					// Iterate through all the getters.
372 					bms.forEach(x -> {
373 						String pn = x.propertyName;
374 						Method m = x.method;
375 						if (! normalProps.containsKey(pn))
376 							normalProps.put(pn, new BeanPropertyMeta.Builder(beanMeta, pn));
377 						BeanPropertyMeta.Builder bpm = normalProps.get(pn);
378 						if (x.methodType == GETTER) {
379 							// Two getters.  Pick the best.
380 							if (nn(bpm.getter)) {
381 
382 								if (! ctx.hasAnnotation(Beanp.class, m) && ctx.hasAnnotation(Beanp.class, bpm.getter))
383 									m = bpm.getter;  // @Beanp annotated method takes precedence.
384 
385 								else if (m.getName().startsWith("is") && bpm.getter.getName().startsWith("get"))
386 									m = bpm.getter;  // getX() overrides isX().
387 							}
388 							bpm.setGetter(m);
389 						}
390 					});
391 
392 					// Now iterate through all the setters.
393 					bms.forEach(x -> {
394 						if (x.methodType == SETTER) {
395 							BeanPropertyMeta.Builder bpm = normalProps.get(x.propertyName);
396 							if (x.matchesPropertyType(bpm))
397 								bpm.setSetter(x.method);
398 						}
399 					});
400 
401 					// Now iterate through all the extraKeys.
402 					bms.forEach(x -> {
403 						if (x.methodType == EXTRAKEYS) {
404 							BeanPropertyMeta.Builder bpm = normalProps.get(x.propertyName);
405 							bpm.setExtraKeys(x.method);
406 						}
407 					});
408 				}
409 
410 				typeVarImpls = map();
411 				findTypeVarImpls(c, typeVarImpls);
412 				if (typeVarImpls.isEmpty())
413 					typeVarImpls = null;
414 
415 				// Eliminate invalid properties, and set the contents of getterProps and setterProps.
416 				for (Iterator<BeanPropertyMeta.Builder> i = normalProps.values().iterator(); i.hasNext();) {
417 					BeanPropertyMeta.Builder p = i.next();
418 					try {
419 						if (p.field == null)
420 							p.setInnerField(findInnerBeanField(ctx, c, stopClass, p.name));
421 
422 						if (p.validate(ctx, beanRegistry, typeVarImpls, bpro, bpwo)) {
423 
424 							if (nn(p.getter))
425 								getterProps.put(p.getter, p.name);
426 
427 							if (nn(p.setter))
428 								setterProps.put(p.setter, p.name);
429 
430 						} else {
431 							i.remove();
432 						}
433 					} catch (ClassNotFoundException e) {
434 						throw new BeanRuntimeException(c, lm(e));
435 					}
436 				}
437 
438 				// Check for missing properties.
439 				fixedBeanProps.forEach(x -> {
440 					if (! normalProps.containsKey(x))
441 						throw new BeanRuntimeException(c, "The property ''{0}'' was defined on the @Bean(properties=X) annotation of class ''{1}'' but was not found on the class definition.", x,
442 							ci.getNameSimple());
443 				});
444 
445 				// Mark constructor arg properties.
446 				for (var fp : constructorArgs) {
447 					BeanPropertyMeta.Builder m = normalProps.get(fp);
448 					if (m == null)
449 						throw new BeanRuntimeException(c, "The property ''{0}'' was defined on the @Beanc(properties=X) annotation but was not found on the class definition.", fp);
450 					m.setAsConstructorArg();
451 				}
452 
453 				// Make sure at least one property was found.
454 				if (beanFilter == null && ctx.isBeansRequireSomeProperties() && normalProps.isEmpty())
455 					return "No properties detected on bean class";
456 
457 				sortProperties = (ctx.isSortProperties() || (nn(beanFilter) && beanFilter.isSortProperties())) && fixedBeanProps.isEmpty();
458 
459 				properties = sortProperties ? sortedMap() : map();
460 
461 				if (nn(beanFilter) && nn(beanFilter.getTypeName()))
462 					dictionaryName = beanFilter.getTypeName();
463 				if (dictionaryName == null)
464 					dictionaryName = findDictionaryName(this.classMeta);
465 
466 				normalProps.forEach((k, v) -> {
467 					BeanPropertyMeta pMeta = v.build();
468 					if (pMeta.isDyna())
469 						dynaProperty = pMeta;
470 					properties.put(k, pMeta);
471 				});
472 
473 				// If a beanFilter is defined, look for inclusion and exclusion lists.
474 				if (nn(beanFilter)) {
475 
476 					// Eliminated excluded properties if BeanFilter.excludeKeys is specified.
477 					Set<String> bfbpi = beanFilter.getProperties();
478 					Set<String> bfbpx = beanFilter.getExcludeProperties();
479 
480 					if (bpi.isEmpty() && ! bfbpi.isEmpty()) {
481 						// Only include specified properties if BeanFilter.includeKeys is specified.
482 						// Note that the order must match includeKeys.
483 						Map<String,BeanPropertyMeta> properties2 = map();
484 						bfbpi.forEach(x -> {
485 							if (properties.containsKey(x))
486 								properties2.put(x, properties.remove(x));
487 						});
488 						hiddenProperties.putAll(properties);
489 						properties = properties2;
490 					}
491 					if (bpx.isEmpty() && ! bfbpx.isEmpty()) {
492 						bfbpx.forEach(x -> hiddenProperties.put(x, properties.remove(x)));
493 					}
494 				}
495 
496 				if (! bpi.isEmpty()) {
497 					Map<String,BeanPropertyMeta> properties2 = map();
498 					bpi.forEach(x -> {
499 						if (properties.containsKey(x))
500 							properties2.put(x, properties.remove(x));
501 					});
502 					hiddenProperties.putAll(properties);
503 					properties = properties2;
504 				}
505 
506 				bpx.forEach(x -> hiddenProperties.put(x, properties.remove(x)));
507 
508 				if (nn(pNames)) {
509 					Map<String,BeanPropertyMeta> properties2 = map();
510 					for (var k : pNames) {
511 						if (properties.containsKey(k))
512 							properties2.put(k, properties.get(k));
513 						else
514 							hiddenProperties.put(k, properties.get(k));
515 					}
516 					properties = properties2;
517 				}
518 
519 			} catch (BeanRuntimeException e) {
520 				throw e;
521 			} catch (Exception e) {
522 				return "Exception:  " + getStackTrace(e);
523 			}
524 
525 			return null;
526 		}
527 	}

538 	private static void forEachClass(ClassInfo c, Class<?> stopClass, Consumer<ClassInfo> consumer) {
539 		ClassInfo sc = c.getSuperclass();
540 		if (nn(sc) && ! sc.is(stopClass))
541 			forEachClass(sc, stopClass, consumer);
542 		c.getInterfaces().forEach(x -> forEachClass(x, stopClass, consumer));
543 		consumer.accept(c);
544 	}

546 	static final String bpName(List<Beanp> p, List<Name> n) {
547 		if (p.isEmpty() && n.isEmpty())
548 			return null;
549 		if (! n.isEmpty())
550 			return last(n).value();
551 
552 		Value<String> name = Value.of(p.isEmpty() ? null : "");
553 		p.forEach(x -> {
554 			if (! x.value().isEmpty())
555 				name.set(x.value());
556 			if (! x.name().isEmpty())
557 				name.set(x.name());
558 		});
559 
560 		return name.orElse(null);
561 	}

563 	static final Collection<Field> findBeanFields(BeanContext ctx, Class<?> c, Class<?> stopClass, Visibility v) {
564 		List<Field> l = new LinkedList<>();
565 		boolean noIgnoreTransients = ! ctx.isIgnoreTransientFields();
566 		forEachClass(ClassInfo.of(c), stopClass, c2 -> {
567 			// @formatter:off
568 			c2.getDeclaredFields().stream()
569 				.filter(x -> x.isNotStatic()
570 				&& (x.isNotTransient() || noIgnoreTransients)
571 				&& (! x.hasAnnotation(Transient.class) || noIgnoreTransients)

573 				&& (v.isVisible(x.inner()) || ctx.getAnnotationProvider().xfind(Beanp.class, x.inner()).findAny().isPresent()))

576 			// @formatter:on

581 	/*
582 	 * Find all the bean methods on this class.
583 	 *
584 	 * @param c The transformed class.
585 	 * @param stopClass Don't look above this class in the hierarchy.
586 	 * @param v The minimum method visibility.
587 	 * @param fixedBeanProps Only include methods whose properties are in this list.
588 	 * @param pn Use this property namer to determine property names from the method names.
589 	 */
590 	static final List<BeanMethod> findBeanMethods(BeanContext ctx, Class<?> c, Class<?> stopClass, Visibility v, PropertyNamer pn, boolean fluentSetters) {
591 		List<BeanMethod> l = new LinkedList<>();
592 
593 		forEachClass(ClassInfo.of(c), stopClass, c2 -> {
594 			for (var m : c2.getDeclaredMethods()) {
595 				if (m.isStatic() || m.isBridge() || m.getParameterCount() > 2 || m.getMatchingMethods().stream().anyMatch(m2 -> ctx.getAnnotationProvider().xfind(BeanIgnore.class, m2.inner()).findFirst().isPresent()))
596 					continue;
597 				Transient t = m.getMatchingMethods().stream()
598 					.map(m2 -> ctx.getAnnotationProvider().xfind(Transient.class, m2.inner()).map(x -> x.inner()).filter(x -> true).findFirst().orElse(null))
599 					.filter(Objects::nonNull)
600 					.findFirst()
601 					.orElse(null);
602 				if (nn(t) && t.value())
603 					continue;
604 
605 				List<Beanp> lp = list();
606 				List<Name> ln = list();
607 				ctx.getAnnotationProvider().xfind(Beanp.class, m.inner()).map(x -> x.inner()).filter(x -> true).forEach(x -> lp.add(x));
608 				ctx.getAnnotationProvider().xfind(Name.class, m.inner()).map(x -> x.inner()).filter(x -> true).forEach(x -> ln.add(x));
609 
610 				// If this method doesn't have @Beanp or @Name, check if it overrides a parent method that does
611 				// This ensures property names are inherited correctly, preventing duplicate property definitions
612 				inheritParentAnnotations(ctx, m, c, stopClass, lp, ln);
613 
614 				if (! (m.isVisible(v) || isNotEmpty(lp) || isNotEmpty(ln)))
615 					continue;
616 
617 			String n = m.getSimpleName();
618 
619 			List<ParameterInfo> params = m.getParameters();
620 			ClassInfo rt = m.getReturnType();
621 			MethodType methodType = UNKNOWN;
622 			String bpName = bpName(lp, ln);
623 
624 			if (params.isEmpty()) {
625 					if ("*".equals(bpName)) {
626 						if (rt.isChildOf(Collection.class)) {
627 							methodType = EXTRAKEYS;
628 						} else if (rt.isChildOf(Map.class)) {
629 							methodType = GETTER;
630 						}
631 						n = bpName;
632 					} else if (n.startsWith("get") && (! rt.is(Void.TYPE))) {
633 						methodType = GETTER;
634 						n = n.substring(3);
635 					} else if (n.startsWith("is") && (rt.is(Boolean.TYPE) || rt.is(Boolean.class))) {
636 						methodType = GETTER;
637 						n = n.substring(2);
638 					} else if (nn(bpName)) {
639 						methodType = GETTER;
640 						if (bpName.isEmpty()) {
641 							if (n.startsWith("get"))
642 								n = n.substring(3);
643 							else if (n.startsWith("is"))
644 								n = n.substring(2);
645 							bpName = n;
646 						} else {
647 							n = bpName;
648 						}
649 				}
650 			} else if (params.size() == 1) {
651 				if ("*".equals(bpName)) {
652 					if (params.get(0).getParameterType().isChildOf(Map.class)) {
653 						methodType = SETTER;
654 						n = bpName;
655 					} else if (params.get(0).getParameterType().is(String.class)) {
656 						methodType = GETTER;
657 							n = bpName;
658 						}
659 					} else if (n.startsWith("set") && (rt.isParentOf(c) || rt.is(Void.TYPE))) {
660 						methodType = SETTER;
661 						n = n.substring(3);
662 					} else if (n.startsWith("with") && (rt.isParentOf(c))) {
663 						methodType = SETTER;
664 						n = n.substring(4);
665 					} else if (nn(bpName)) {
666 						methodType = SETTER;
667 						if (bpName.isEmpty()) {
668 							if (n.startsWith("set"))
669 								n = n.substring(3);
670 							bpName = n;
671 						} else {
672 							n = bpName;
673 						}
674 					} else if (fluentSetters && rt.isParentOf(c)) {
675 						methodType = SETTER;
676 				}
677 			} else if (params.size() == 2) {
678 				if ("*".equals(bpName) && params.get(0).getParameterType().is(String.class)) {
679 						if (n.startsWith("set") && (rt.isParentOf(c) || rt.is(Void.TYPE))) {
680 							methodType = SETTER;
681 						} else {
682 							methodType = GETTER;
683 						}
684 						n = bpName;
685 					}
686 				}
687 				n = pn.getPropertyName(n);
688 
689 				if ("*".equals(bpName) && methodType == UNKNOWN)
690 					throw new BeanRuntimeException(c, "Found @Beanp(\"*\") but could not determine method type on method ''{0}''.", m.getSimpleName());
691 
692 				if (methodType != UNKNOWN) {
693 					if (nn(bpName) && ! bpName.isEmpty())
694 						n = bpName;
695 					if (nn(n))
696 						l.add(new BeanMethod(n, methodType, m.inner()));
697 				}
698 			}
699 		});
700 		return l;
701 	}

720 	/**
721 	 * Recursively determines the classes represented by parameterized types in the class hierarchy of the specified
722 	 * type, and puts the results in the specified map.
723 	 *
724 	 * <p>
725 	 * For example, given the following classes...
726 	 * <p class='bjava'>
727 	 * 	<jk>public static class</jk> BeanA&lt;T&gt; {
728 	 * 		<jk>public</jk> T <jf>x</jf>;
729 	 * 	}
730 	 * 	<jk>public static class</jk> BeanB <jk>extends</jk> BeanA&lt;Integer> {...}
731 	 * </p>
732 	 * <p>
733 	 * 	...calling this method on {@code BeanB.class} will load the following data into {@code m} indicating
734 	 * 	that the {@code T} parameter on the BeanA class is implemented with an {@code Integer}:
735 	 * <p class='bcode'>
736 	 * 	{BeanA.class:[Integer.class]}
737 	 * </p>
738 	 *
739 	 * <h5 class='section'>Known Limitations:</h5>
740 	 * <p>
741 	 * This code doesn't currently properly handle the following situation with nested generic bounds:
742 	 * <p class='bjava'>
743 	 * 	<jk>public static class</jk> BeanB&lt;T <jk>extends</jk> Number&gt; <jk>extends</jk> BeanA&lt;>;
744 	 * 	<jk>public static class</jk> BeanC <jk>extends</jk> BeanB&lt;Integer>;
745 	 * </p>
746 	 *
747 	 * <p>
748 	 * When called on {@code BeanC}, the type variable will be resolved as {@code Number}, not {@code Integer}.
749 	 * This limitation exists because the intermediate type parameter bound information is lost during type resolution
750 	 *
751 	 * @param t The type we're recursing.
752 	 * @param m Where the results are loaded.
753 	 */
754 	static final void findTypeVarImpls(Type t, Map<Class<?>,Class<?>[]> m) {
755 		if (t instanceof Class<?> c) {
756 			findTypeVarImpls(c.getGenericSuperclass(), m);
757 			for (var ci : c.getGenericInterfaces())
758 				findTypeVarImpls(ci, m);
759 		} else if (t instanceof ParameterizedType pt) {
760 			Type rt = pt.getRawType();
761 			if (rt instanceof Class) {
762 				Type[] gImpls = pt.getActualTypeArguments();
763 				Class<?>[] gTypes = new Class[gImpls.length];
764 				for (int i = 0; i < gImpls.length; i++) {
765 					Type gt = gImpls[i];
766 					if (gt instanceof Class<?> c)
767 						gTypes[i] = c;
768 					else if (gt instanceof TypeVariable<?> tv) {
769 						for (var upperBound : tv.getBounds())
770 							if (upperBound instanceof Class)
771 								gTypes[i] = (Class<?>)upperBound;
772 					}
773 				}
774 				m.put((Class<?>)rt, gTypes);
775 				findTypeVarImpls(pt.getRawType(), m);
776 			}
777 		}
778 	}

780 	/**
781 	 * Finds @Beanp and @Name annotations from parent methods if this method overrides a parent.
782 	 * This ensures that property names are inherited from parent methods, preventing duplicate
783 	 * property definitions when a child overrides a @Beanp-annotated parent method.
784 	 *
785 	 * @param ctx The bean context.
786 	 * @param method The method to check.
787 	 * @param c The current class being inspected.
788 	 * @param stopClass Don't look above this class in the hierarchy.
789 	 * @param lp List to populate with @Beanp annotations (input/output parameter).
790 	 * @param ln List to populate with @Name annotations (input/output parameter).
791 	 */
792 	static final void inheritParentAnnotations(BeanContext ctx, MethodInfo method, Class<?> c, Class<?> stopClass, List<Beanp> lp, List<Name> ln) {
793 		// If this method already has @Beanp or @Name annotations, don't look for parent annotations
794 		if (! lp.isEmpty() || ! ln.isEmpty())
795 			return;
796 
797 	String methodName = method.getSimpleName();
798 	List<ParameterInfo> params = method.getParameters();
799 
800 	// Walk up the class hierarchy looking for a matching parent method with @Beanp or @Name
801 	var currentClass = ClassInfo.of(c);
802 	ClassInfo sc = currentClass.getSuperclass();
803 
804 	while (nn(sc) && ! sc.is(stopClass) && ! sc.is(Object.class)) {
805 		// Look for a method with the same signature in the parent class
806 		for (var parentMethod : sc.getDeclaredMethods()) {
807 			if (parentMethod.getSimpleName().equals(methodName) && params.size() == parentMethod.getParameters().size()) {
808 
809 				// Check if parameter types match
810 				boolean paramsMatch = true;
811 				List<ParameterInfo> parentParams = parentMethod.getParameters();
812 				for (int i = 0; i < params.size(); i++) {
813 					if (! params.get(i).getParameterType().is(parentParams.get(i).getParameterType().inner())) {
814 						paramsMatch = false;
815 						break;
816 					}
817 				}
818 
819 					if (paramsMatch) {
820 						// Found a matching parent method - check for @Beanp and @Name annotations
821 						ctx.getAnnotationProvider().xfind(Beanp.class, parentMethod.inner()).map(x -> x.inner()).filter(x -> true).forEach(x -> lp.add(x));
822 						ctx.getAnnotationProvider().xfind(Name.class, parentMethod.inner()).map(x -> x.inner()).filter(x -> true).forEach(x -> ln.add(x));
823 
824 						// If we found annotations, we're done
825 						if (! lp.isEmpty() || ! ln.isEmpty())
826 							return;
827 					}
828 				}
829 			}
830 
831 			// Move to the next superclass
832 			sc = sc.getSuperclass();
833 		}
834 	}
