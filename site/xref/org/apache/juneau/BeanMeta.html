38  /**
39   * Encapsulates all access to the properties of a bean class (like a souped-up {@link java.beans.BeanInfo}).
40   *
41   * <h5 class='topic'>Description</h5>
42   *
43   * Uses introspection to find all the properties associated with this class.  If the {@link Bean @Bean} annotation
44   * 	is present on the class, then that information is used to determine the properties on the class.
45   * Otherwise, the {@code BeanInfo} functionality in Java is used to determine the properties on the class.
46   *
47   * <h5 class='topic'>Bean property ordering</h5>
48   *
49   * The order of the properties are as follows:
50   * <ul class='spaced-list'>
51   * 	<li>
52   * 		If {@link Bean @Bean} annotation is specified on class, then the order is the same as the list of properties
53   * 		in the annotation.
54   * 	<li>
55   * 		If {@link Bean @Bean} annotation is not specified on the class, then the order is based on the following.
56   * 		<ul>
57   * 			<li>Public fields (same order as {@code Class.getFields()}).
58   * 			<li>Properties returned by {@code BeanInfo.getPropertyDescriptors()}.
59   * 			<li>Non-standard getters/setters with {@link Beanp @Beanp} annotation defined on them.
60   * 		</ul>
61   * </ul>
62   *
63   * <h5 class='section'>See Also:</h5><ul>
64  
65   * </ul>
66   *
67   * @param <T> The class type that this metadata applies to.
68   */
69  public class BeanMeta<T> {

74  	private static class BeanMethod {
75  		String propertyName;
76  		MethodType methodType;
77  		Method method;
78  		ClassInfo type;
79  
80  		BeanMethod(String propertyName, MethodType type, Method method) {
81  			this.propertyName = propertyName;
82  			this.methodType = type;
83  			this.method = method;
84  			if (type == MethodType.SETTER)
85  				this.type = ClassInfo.of(method.getParameterTypes()[0]);
86  			else
87  				this.type = ClassInfo.of(method.getReturnType());
88  		}
89  
90  		@Override /* Overridden from Object */
91  		public String toString() {
92  			return method.toString();
93  		}
94  
95  		/*
96  		 * Returns true if this method matches the class type of the specified property.
97  		 * Only meant to be used for setters.
98  		 */
99  		boolean matchesPropertyType(BeanPropertyMeta.Builder b) {
100 			if (b == null)
101 				return false;
102 
103 			// Don't do further validation if this is the "*" bean property.
104 			if ("*".equals(b.name))
105 				return true;
106 
107 			// Get the bean property type from the getter/field.
108 			Class<?> pt = null;
109 			if (nn(b.getter))
110 				pt = b.getter.getReturnType();
111 			else if (nn(b.field))
112 				pt = b.field.getType();
113 
114 			// Matches if only a setter is defined.
115 			if (pt == null)
116 				return true;
117 
118 			// Doesn't match if not same type or super type as getter/field.
119 			if (! type.isParentOf(pt))
120 				return false;
121 
122 			// If a setter was previously set, only use this setter if it's a closer
123 			// match (e.g. prev type is a superclass of this type).
124 			if (b.setter == null)
125 				return true;
126 
127 			return type.isStrictChildOf(b.setter.getParameterTypes()[0]);
128 		}
129 	}

131 	private static class Builder<T> {
132 		ClassMeta<T> classMeta;
133 		BeanContext ctx;
134 		BeanFilter beanFilter;
135 		String[] pNames;
136 		Map<String,BeanPropertyMeta> properties;
137 		Map<String,BeanPropertyMeta> hiddenProperties = map();
138 		Map<Method,String> getterProps = map();
139 		Map<Method,String> setterProps = map();
140 		BeanPropertyMeta dynaProperty;

148 		boolean sortProperties, fluentSetters;

150 		Builder(ClassMeta<T> classMeta, BeanContext ctx, BeanFilter beanFilter, String[] pNames, ConstructorInfo implClassConstructor) {
151 			this.classMeta = classMeta;
152 			this.ctx = ctx;
153 			this.beanFilter = beanFilter;
154 			this.pNames = pNames;
155 			this.implClassConstructor = implClassConstructor;

158 		private String findDictionaryName(ClassMeta<?> cm) {
159 			BeanRegistry br = cm.getBeanRegistry();
160 			if (nn(br)) {
161 				String s = br.getTypeName(this.classMeta);
162 				if (nn(s))
163 					return s;
164 			}
165 			Class<?> pcm = cm.innerClass.getSuperclass();
166 			if (nn(pcm)) {
167 				String s = findDictionaryName(ctx.getClassMeta(pcm));
168 				if (nn(s))
169 					return s;
170 			}
171 			for (var icm : cm.innerClass.getInterfaces()) {
172 				String s = findDictionaryName(ctx.getClassMeta(icm));
173 				if (nn(s))
174 					return s;
175 			}
176 			return null;
177 		}

179 		/*
180 		 * Returns the property name of the specified field if it's a valid property.
181 		 * Returns null if the field isn't a valid property.
182 		 */
183 		private String findPropertyName(Field f) {
184 			List<Beanp> lp = list();
185 			List<Name> ln = list();
186 			ctx.forEachAnnotation(Beanp.class, f, x -> true, x -> lp.add(x));
187 			ctx.forEachAnnotation(Name.class, f, x -> true, x -> ln.add(x));
188 			String name = bpName(lp, ln);
189 			if (isNotEmpty(name))
190 				return name;
191 			return propertyNamer.getPropertyName(f.getName());
192 		}

194 		String init(BeanMeta<T> beanMeta) {
195 			Class<?> c = classMeta.getInnerClass();
196 			ClassInfo ci = classMeta.getInfo();
197 
198 			try {
199 				// @formatter:off
200 				Visibility
201 					conVis = ctx.getBeanConstructorVisibility(),
202 					cVis = ctx.getBeanClassVisibility(),
203 					mVis = ctx.getBeanMethodVisibility(),
204 					fVis = ctx.getBeanFieldVisibility();
205 				// @formatter:on
206 
207 				List<Class<?>> bdClasses = list();
208 				if (nn(beanFilter) && nn(beanFilter.getBeanDictionary()))
209 					addAll(bdClasses, beanFilter.getBeanDictionary());
210 
211 				Value<String> typeName = Value.empty();
212 				classMeta.forEachAnnotation(Bean.class, x -> isNotEmpty(x.typeName()), x -> typeName.set(x.typeName()));
213 				if (typeName.isPresent())
214 					bdClasses.add(classMeta.innerClass);
215 				this.beanRegistry = new BeanRegistry(ctx, null, bdClasses.toArray(new Class<?>[bdClasses.size()]));
216 
217 				Value<String> typePropertyName = Value.empty();
218 				classMeta.forEachAnnotation(Bean.class, x -> isNotEmpty(x.typePropertyName()), x -> typePropertyName.set(x.typePropertyName()));
219 				this.typePropertyName = typePropertyName.orElseGet(() -> ctx.getBeanTypePropertyName());
220 
221 				fluentSetters = (ctx.isFindFluentSetters() || (nn(beanFilter) && beanFilter.isFluentSetters()));
222 
223 				// If @Bean.interfaceClass is specified on the parent class, then we want
224 				// to use the properties defined on that class, not the subclass.
225 				Class<?> c2 = (nn(beanFilter) && nn(beanFilter.getInterfaceClass()) ? beanFilter.getInterfaceClass() : c);
226 
227 				Class<?> stopClass = (nn(beanFilter) ? beanFilter.getStopClass() : Object.class);
228 				if (stopClass == null)
229 					stopClass = Object.class;
230 
231 				Map<String,BeanPropertyMeta.Builder> normalProps = map();
232 
233 				boolean hasBean = ci.hasAnnotation(ctx, Bean.class);
234 				boolean hasBeanIgnore = ci.hasAnnotation(ctx, BeanIgnore.class);
235 
236 				/// See if this class matches one the patterns in the exclude-class list.
237 				if (ctx.isNotABean(c))
238 					return "Class matches exclude-class list";
239 
240 				if (! hasBean && ! (cVis.isVisible(c.getModifiers()) || c.isAnonymousClass()))
241 					return "Class is not public";
242 
243 				if (hasBeanIgnore)
244 					return "Class is annotated with @BeanIgnore";
245 
246 				// Make sure it's serializable.
247 				if (beanFilter == null && ctx.isBeansRequireSerializable() && ! ci.isChildOf(Serializable.class))
248 					return "Class is not serializable";
249 
250 				// Look for @Beanc constructor on public constructors.
251 				ci.forEachPublicConstructor(x -> x.hasAnnotation(ctx, Beanc.class), x -> {
252 					if (nn(constructor))
253 						throw new BeanRuntimeException(c, "Multiple instances of '@Beanc' found.");
254 					constructor = x;
255 					constructorArgs = new String[0];
256 					ctx.forEachAnnotation(Beanc.class, x.inner(), y -> ! y.properties().isEmpty(), z -> constructorArgs = splita(z.properties()));
257 					if (! x.hasNumParams(constructorArgs.length)) {
258 						if (constructorArgs.length != 0)
259 							throw new BeanRuntimeException(c, "Number of properties defined in '@Beanc' annotation does not match number of parameters in constructor.");
260 						constructorArgs = new String[x.getParamCount()];
261 						IntegerValue i = IntegerValue.create();
262 						x.forEachParam(null, pi -> {
263 							String pn = pi.getName();
264 							if (pn == null)
265 								throw new BeanRuntimeException(c, "Could not find name for parameter #{0} of constructor ''{1}''", i, x.getFullName());
266 							constructorArgs[i.getAndIncrement()] = pn;
267 						});
268 					}
269 					constructor.setAccessible();
270 				});
271 
272 				// Look for @Beanc on all other constructors.
273 				if (constructor == null) {
274 					ci.forEachDeclaredConstructor(x -> x.hasAnnotation(ctx, Beanc.class), x -> {
275 						if (nn(constructor))
276 							throw new BeanRuntimeException(c, "Multiple instances of '@Beanc' found.");
277 						constructor = x;
278 						constructorArgs = new String[0];
279 						ctx.forEachAnnotation(Beanc.class, x.inner(), y -> ! y.properties().isEmpty(), z -> constructorArgs = splita(z.properties()));
280 						if (! x.hasNumParams(constructorArgs.length)) {
281 							if (constructorArgs.length != 0)
282 								throw new BeanRuntimeException(c, "Number of properties defined in '@Beanc' annotation does not match number of parameters in constructor.");
283 							constructorArgs = new String[x.getParamCount()];
284 							IntegerValue i = IntegerValue.create();
285 							x.forEachParam(null, y -> {
286 								String pn = y.getName();
287 								if (pn == null)
288 									throw new BeanRuntimeException(c, "Could not find name for parameter #{0} of constructor ''{1}''", i, x.getFullName());
289 								constructorArgs[i.getAndIncrement()] = pn;
290 							});
291 						}
292 						constructor.setAccessible();
293 					});
294 				}
295 
296 				// If this is an interface, look for impl classes defined in the context.
297 				if (constructor == null)
298 					constructor = implClassConstructor;
299 
300 				if (constructor == null)
301 					constructor = ci.getNoArgConstructor(hasBean ? Visibility.PRIVATE : conVis);
302 
303 				if (constructor == null && beanFilter == null && ctx.isBeansRequireDefaultConstructor())
304 					return "Class does not have the required no-arg constructor";
305 
306 				if (nn(constructor))
307 					constructor.setAccessible();
308 
309 				// Explicitly defined property names in @Bean annotation.
310 				Set<String> fixedBeanProps = set();
311 				Set<String> bpi = set();
312 				Set<String> bpx = set();
313 				Set<String> bpro = set();
314 				Set<String> bpwo = set();
315 
316 				Set<String> filterProps = set();  // Names of properties defined in @Bean(properties)
317 
318 				if (nn(beanFilter)) {
319 
320 					Set<String> bfbpi = beanFilter.getProperties();
321 
322 					filterProps.addAll(bfbpi);
323 
324 					// Get the 'properties' attribute if specified.
325 					if (bpi.isEmpty())
326 						fixedBeanProps.addAll(bfbpi);
327 
328 					if (nn(beanFilter.getPropertyNamer()))
329 						propertyNamer = beanFilter.getPropertyNamer();
330 
331 					bpro.addAll(beanFilter.getReadOnlyProperties());
332 					bpwo.addAll(beanFilter.getWriteOnlyProperties());
333 				}
334 
335 				fixedBeanProps.addAll(bpi);
336 
337 				if (propertyNamer == null)
338 					propertyNamer = ctx.getPropertyNamer();
339 
340 				// First populate the properties with those specified in the bean annotation to
341 				// ensure that ordering first.
342 				fixedBeanProps.forEach(x -> normalProps.put(x, BeanPropertyMeta.builder(beanMeta, x)));
343 
344 				if (ctx.isUseJavaBeanIntrospector()) {
345 					BeanInfo bi = null;
346 					if (! c2.isInterface())
347 						bi = Introspector.getBeanInfo(c2, stopClass);
348 					else
349 						bi = Introspector.getBeanInfo(c2, null);
350 					if (nn(bi)) {
351 						for (var pd : bi.getPropertyDescriptors()) {
352 							String name = pd.getName();
353 							if (! normalProps.containsKey(name))
354 								normalProps.put(name, BeanPropertyMeta.builder(beanMeta, name));
355 							normalProps.get(name).setGetter(pd.getReadMethod()).setSetter(pd.getWriteMethod());
356 						}
357 					}
358 
359 				} else /* Use 'better' introspection */ {
360 
361 					findBeanFields(ctx, c2, stopClass, fVis).forEach(x -> {
362 						String name = findPropertyName(x);
363 						if (nn(name)) {
364 							if (! normalProps.containsKey(name))
365 								normalProps.put(name, BeanPropertyMeta.builder(beanMeta, name));
366 							normalProps.get(name).setField(x);
367 						}
368 					});
369 
370 					List<BeanMethod> bms = findBeanMethods(ctx, c2, stopClass, mVis, propertyNamer, fluentSetters);
371 
372 					// Iterate through all the getters.
373 					bms.forEach(x -> {
374 						String pn = x.propertyName;
375 						Method m = x.method;
376 						if (! normalProps.containsKey(pn))
377 							normalProps.put(pn, new BeanPropertyMeta.Builder(beanMeta, pn));
378 						BeanPropertyMeta.Builder bpm = normalProps.get(pn);
379 						if (x.methodType == GETTER) {
380 							// Two getters.  Pick the best.
381 							if (nn(bpm.getter)) {
382 
383 								if (! ctx.hasAnnotation(Beanp.class, m) && ctx.hasAnnotation(Beanp.class, bpm.getter))
384 									m = bpm.getter;  // @Beanp annotated method takes precedence.
385 
386 								else if (m.getName().startsWith("is") && bpm.getter.getName().startsWith("get"))
387 									m = bpm.getter;  // getX() overrides isX().
388 							}
389 							bpm.setGetter(m);
390 						}
391 					});
392 
393 					// Now iterate through all the setters.
394 					bms.forEach(x -> {
395 						if (x.methodType == SETTER) {
396 							BeanPropertyMeta.Builder bpm = normalProps.get(x.propertyName);
397 							if (x.matchesPropertyType(bpm))
398 								bpm.setSetter(x.method);
399 						}
400 					});
401 
402 					// Now iterate through all the extraKeys.
403 					bms.forEach(x -> {
404 						if (x.methodType == EXTRAKEYS) {
405 							BeanPropertyMeta.Builder bpm = normalProps.get(x.propertyName);
406 							bpm.setExtraKeys(x.method);
407 						}
408 					});
409 				}
410 
411 				typeVarImpls = map();
412 				findTypeVarImpls(c, typeVarImpls);
413 				if (typeVarImpls.isEmpty())
414 					typeVarImpls = null;
415 
416 				// Eliminate invalid properties, and set the contents of getterProps and setterProps.
417 				for (Iterator<BeanPropertyMeta.Builder> i = normalProps.values().iterator(); i.hasNext();) {
418 					BeanPropertyMeta.Builder p = i.next();
419 					try {
420 						if (p.field == null)
421 							p.setInnerField(findInnerBeanField(ctx, c, stopClass, p.name));
422 
423 						if (p.validate(ctx, beanRegistry, typeVarImpls, bpro, bpwo)) {
424 
425 							if (nn(p.getter))
426 								getterProps.put(p.getter, p.name);
427 
428 							if (nn(p.setter))
429 								setterProps.put(p.setter, p.name);
430 
431 						} else {
432 							i.remove();
433 						}
434 					} catch (ClassNotFoundException e) {
435 						throw new BeanRuntimeException(c, lm(e));
436 					}
437 				}
438 
439 				// Check for missing properties.
440 				fixedBeanProps.forEach(x -> {
441 					if (! normalProps.containsKey(x))
442 						throw new BeanRuntimeException(c, "The property ''{0}'' was defined on the @Bean(properties=X) annotation of class ''{1}'' but was not found on the class definition.", x,
443 							ci.getSimpleName());
444 				});
445 
446 				// Mark constructor arg properties.
447 				for (var fp : constructorArgs) {
448 					BeanPropertyMeta.Builder m = normalProps.get(fp);
449 					if (m == null)
450 						throw new BeanRuntimeException(c, "The property ''{0}'' was defined on the @Beanc(properties=X) annotation but was not found on the class definition.", fp);
451 					m.setAsConstructorArg();
452 				}
453 
454 				// Make sure at least one property was found.
455 				if (beanFilter == null && ctx.isBeansRequireSomeProperties() && normalProps.isEmpty())
456 					return "No properties detected on bean class";
457 
458 				sortProperties = (ctx.isSortProperties() || (nn(beanFilter) && beanFilter.isSortProperties())) && fixedBeanProps.isEmpty();
459 
460 				properties = sortProperties ? sortedMap() : map();
461 
462 				if (nn(beanFilter) && nn(beanFilter.getTypeName()))
463 					dictionaryName = beanFilter.getTypeName();
464 				if (dictionaryName == null)
465 					dictionaryName = findDictionaryName(this.classMeta);
466 
467 				normalProps.forEach((k, v) -> {
468 					BeanPropertyMeta pMeta = v.build();
469 					if (pMeta.isDyna())
470 						dynaProperty = pMeta;
471 					properties.put(k, pMeta);
472 				});
473 
474 				// If a beanFilter is defined, look for inclusion and exclusion lists.
475 				if (nn(beanFilter)) {
476 
477 					// Eliminated excluded properties if BeanFilter.excludeKeys is specified.
478 					Set<String> bfbpi = beanFilter.getProperties();
479 					Set<String> bfbpx = beanFilter.getExcludeProperties();
480 
481 					if (bpi.isEmpty() && ! bfbpi.isEmpty()) {
482 						// Only include specified properties if BeanFilter.includeKeys is specified.
483 						// Note that the order must match includeKeys.
484 						Map<String,BeanPropertyMeta> properties2 = map();
485 						bfbpi.forEach(x -> {
486 							if (properties.containsKey(x))
487 								properties2.put(x, properties.remove(x));
488 						});
489 						hiddenProperties.putAll(properties);
490 						properties = properties2;
491 					}
492 					if (bpx.isEmpty() && ! bfbpx.isEmpty()) {
493 						bfbpx.forEach(x -> hiddenProperties.put(x, properties.remove(x)));
494 					}
495 				}
496 
497 				if (! bpi.isEmpty()) {
498 					Map<String,BeanPropertyMeta> properties2 = map();
499 					bpi.forEach(x -> {
500 						if (properties.containsKey(x))
501 							properties2.put(x, properties.remove(x));
502 					});
503 					hiddenProperties.putAll(properties);
504 					properties = properties2;
505 				}
506 
507 				bpx.forEach(x -> hiddenProperties.put(x, properties.remove(x)));
508 
509 				if (nn(pNames)) {
510 					Map<String,BeanPropertyMeta> properties2 = map();
511 					for (var k : pNames) {
512 						if (properties.containsKey(k))
513 							properties2.put(k, properties.get(k));
514 						else
515 							hiddenProperties.put(k, properties.get(k));
516 					}
517 					properties = properties2;
518 				}
519 
520 			} catch (BeanRuntimeException e) {
521 				throw e;
522 			} catch (Exception e) {
523 				return "Exception:  " + getStackTrace(e);
524 			}
525 
526 			return null;
527 		}
528 	}

539 	private static void forEachClass(ClassInfo c, Class<?> stopClass, Consumer<ClassInfo> consumer) {
540 		ClassInfo sc = c.getSuperclass();
541 		if (nn(sc) && ! sc.is(stopClass))
542 			forEachClass(sc, stopClass, consumer);
543 		c.getInterfaces().forEach(x -> forEachClass(x, stopClass, consumer));
544 		consumer.accept(c);
545 	}

547 	static final String bpName(List<Beanp> p, List<Name> n) {
548 		if (p.isEmpty() && n.isEmpty())
549 			return null;
550 		if (! n.isEmpty())
551 			return last(n).value();
552 
553 		Value<String> name = Value.of(p.isEmpty() ? null : "");
554 		p.forEach(x -> {
555 			if (! x.value().isEmpty())
556 				name.set(x.value());
557 			if (! x.name().isEmpty())
558 				name.set(x.name());
559 		});
560 
561 		return name.orElse(null);
562 	}

564 	static final Collection<Field> findBeanFields(BeanContext ctx, Class<?> c, Class<?> stopClass, Visibility v) {
565 		List<Field> l = new LinkedList<>();
566 		boolean noIgnoreTransients = ! ctx.isIgnoreTransientFields();
567 		forEachClass(ClassInfo.of(c), stopClass, c2 -> {
568 			// @formatter:off
569 			c2.forEachDeclaredField(
570 				x -> x.isNotStatic()
571 				&& (x.isNotTransient() || noIgnoreTransients)
572 				&& (x.hasNoAnnotation(Transient.class) || noIgnoreTransients)
573 				&& x.hasNoAnnotation(ctx, BeanIgnore.class)
574 				&& (v.isVisible(x.inner()) || x.hasAnnotation(ctx, Beanp.class)),
575 				x -> l.add(x.inner())
576 			);
577 			// @formatter:on
578 		});
579 		return l;
580 	}

582 	/*
583 	 * Find all the bean methods on this class.
584 	 *
585 	 * @param c The transformed class.
586 	 * @param stopClass Don't look above this class in the hierarchy.
587 	 * @param v The minimum method visibility.
588 	 * @param fixedBeanProps Only include methods whose properties are in this list.
589 	 * @param pn Use this property namer to determine property names from the method names.
590 	 */
591 	static final List<BeanMethod> findBeanMethods(BeanContext ctx, Class<?> c, Class<?> stopClass, Visibility v, PropertyNamer pn, boolean fluentSetters) {
592 		List<BeanMethod> l = new LinkedList<>();
593 
594 		forEachClass(ClassInfo.of(c), stopClass, c2 -> {
595 			for (var m : c2.getDeclaredMethods()) {
596 				if (m.isStatic() || m.isBridge() || m.getParamCount() > 2 || m.hasAnnotation(ctx, BeanIgnore.class))
597 					continue;
598 				Transient t = m.getAnnotation(ctx, Transient.class);
599 				if (nn(t) && t.value())
600 					continue;
601 
602 				List<Beanp> lp = list();
603 				List<Name> ln = list();
604 				ctx.forEachAnnotation(Beanp.class, m.inner(), x -> true, x -> lp.add(x));
605 				ctx.forEachAnnotation(Name.class, m.inner(), x -> true, x -> ln.add(x));
606 
607 				// If this method doesn't have @Beanp or @Name, check if it overrides a parent method that does
608 				// This ensures property names are inherited correctly, preventing duplicate property definitions
609 				inheritParentAnnotations(ctx, m, c, stopClass, lp, ln);
610 
611 				if (! (m.isVisible(v) || isNotEmpty(lp) || isNotEmpty(ln)))
612 					continue;
613 
614 				String n = m.getSimpleName();
615 
616 				List<ClassInfo> pt = m.getParamTypes();
617 				ClassInfo rt = m.getReturnType();
618 				MethodType methodType = UNKNOWN;
619 				String bpName = bpName(lp, ln);
620 
621 				if (pt.isEmpty()) {
622 					if ("*".equals(bpName)) {
623 						if (rt.isChildOf(Collection.class)) {
624 							methodType = EXTRAKEYS;
625 						} else if (rt.isChildOf(Map.class)) {
626 							methodType = GETTER;
627 						}
628 						n = bpName;
629 					} else if (n.startsWith("get") && (! rt.is(Void.TYPE))) {
630 						methodType = GETTER;
631 						n = n.substring(3);
632 					} else if (n.startsWith("is") && (rt.is(Boolean.TYPE) || rt.is(Boolean.class))) {
633 						methodType = GETTER;
634 						n = n.substring(2);
635 					} else if (nn(bpName)) {
636 						methodType = GETTER;
637 						if (bpName.isEmpty()) {
638 							if (n.startsWith("get"))
639 								n = n.substring(3);
640 							else if (n.startsWith("is"))
641 								n = n.substring(2);
642 							bpName = n;
643 						} else {
644 							n = bpName;
645 						}
646 					}
647 				} else if (pt.size() == 1) {
648 					if ("*".equals(bpName)) {
649 						if (pt.get(0).isChildOf(Map.class)) {
650 							methodType = SETTER;
651 							n = bpName;
652 						} else if (pt.get(0).is(String.class)) {
653 							methodType = GETTER;
654 							n = bpName;
655 						}
656 					} else if (n.startsWith("set") && (rt.isParentOf(c) || rt.is(Void.TYPE))) {
657 						methodType = SETTER;
658 						n = n.substring(3);
659 					} else if (n.startsWith("with") && (rt.isParentOf(c))) {
660 						methodType = SETTER;
661 						n = n.substring(4);
662 					} else if (nn(bpName)) {
663 						methodType = SETTER;
664 						if (bpName.isEmpty()) {
665 							if (n.startsWith("set"))
666 								n = n.substring(3);
667 							bpName = n;
668 						} else {
669 							n = bpName;
670 						}
671 					} else if (fluentSetters && rt.isParentOf(c)) {
672 						methodType = SETTER;
673 					}
674 				} else if (pt.size() == 2) {
675 					if ("*".equals(bpName) && pt.get(0).is(String.class)) {
676 						if (n.startsWith("set") && (rt.isParentOf(c) || rt.is(Void.TYPE))) {
677 							methodType = SETTER;
678 						} else {
679 							methodType = GETTER;
680 						}
681 						n = bpName;
682 					}
683 				}
684 				n = pn.getPropertyName(n);
685 
686 				if ("*".equals(bpName) && methodType == UNKNOWN)
687 					throw new BeanRuntimeException(c, "Found @Beanp(\"*\") but could not determine method type on method ''{0}''.", m.getSimpleName());
688 
689 				if (methodType != UNKNOWN) {
690 					if (nn(bpName) && ! bpName.isEmpty())
691 						n = bpName;
692 					if (nn(n))
693 						l.add(new BeanMethod(n, methodType, m.inner()));
694 				}
695 			}
696 		});
697 		return l;
698 	}

719 	/**
720 	 * Recursively determines the classes represented by parameterized types in the class hierarchy of the specified
721 	 * type, and puts the results in the specified map.
722 	 *
723 	 * <p>
724 	 * For example, given the following classes...
725 	 * <p class='bjava'>
726 	 * 	<jk>public static class</jk> BeanA&lt;T&gt; {
727 	 * 		<jk>public</jk> T <jf>x</jf>;
728 	 * 	}
729 	 * 	<jk>public static class</jk> BeanB <jk>extends</jk> BeanA&lt;Integer> {...}
730 	 * </p>
731 	 * <p>
732 	 * 	...calling this method on {@code BeanB.class} will load the following data into {@code m} indicating
733 	 * 	that the {@code T} parameter on the BeanA class is implemented with an {@code Integer}:
734 	 * <p class='bcode'>
735 	 * 	{BeanA.class:[Integer.class]}
736 	 * </p>
737 	 *
738 	 * <h5 class='section'>Known Limitations:</h5>
739 	 * <p>
740 	 * This code doesn't currently properly handle the following situation with nested generic bounds:
741 	 * <p class='bjava'>
742 	 * 	<jk>public static class</jk> BeanB&lt;T <jk>extends</jk> Number&gt; <jk>extends</jk> BeanA&lt;>;
743 	 * 	<jk>public static class</jk> BeanC <jk>extends</jk> BeanB&lt;Integer>;
744 	 * </p>
745 	 *
746 	 * <p>
747 	 * When called on {@code BeanC}, the type variable will be resolved as {@code Number}, not {@code Integer}.
748 	 * This limitation exists because the intermediate type parameter bound information is lost during type resolution
749 	 *
750 	 * @param t The type we're recursing.
751 	 * @param m Where the results are loaded.
752 	 */
753 	static final void findTypeVarImpls(Type t, Map<Class<?>,Class<?>[]> m) {
754 		if (t instanceof Class<?> c) {
755 			findTypeVarImpls(c.getGenericSuperclass(), m);
756 			for (var ci : c.getGenericInterfaces())
757 				findTypeVarImpls(ci, m);
758 		} else if (t instanceof ParameterizedType pt) {
759 			Type rt = pt.getRawType();
760 			if (rt instanceof Class) {
761 				Type[] gImpls = pt.getActualTypeArguments();
762 				Class<?>[] gTypes = new Class[gImpls.length];
763 				for (int i = 0; i < gImpls.length; i++) {
764 					Type gt = gImpls[i];
765 					if (gt instanceof Class<?> c)
766 						gTypes[i] = c;
767 					else if (gt instanceof TypeVariable<?> tv) {
768 						for (var upperBound : tv.getBounds())
769 							if (upperBound instanceof Class)
770 								gTypes[i] = (Class<?>)upperBound;
771 					}
772 				}
773 				m.put((Class<?>)rt, gTypes);
774 				findTypeVarImpls(pt.getRawType(), m);
775 			}
776 		}
777 	}

779 	/**
780 	 * Finds @Beanp and @Name annotations from parent methods if this method overrides a parent.
781 	 * This ensures that property names are inherited from parent methods, preventing duplicate
782 	 * property definitions when a child overrides a @Beanp-annotated parent method.
783 	 *
784 	 * @param ctx The bean context.
785 	 * @param method The method to check.
786 	 * @param c The current class being inspected.
787 	 * @param stopClass Don't look above this class in the hierarchy.
788 	 * @param lp List to populate with @Beanp annotations (input/output parameter).
789 	 * @param ln List to populate with @Name annotations (input/output parameter).
790 	 */
791 	static final void inheritParentAnnotations(BeanContext ctx, MethodInfo method, Class<?> c, Class<?> stopClass, List<Beanp> lp, List<Name> ln) {
792 		// If this method already has @Beanp or @Name annotations, don't look for parent annotations
793 		if (! lp.isEmpty() || ! ln.isEmpty())
794 			return;
795 
796 		String methodName = method.getSimpleName();
797 		List<ClassInfo> paramTypes = method.getParamTypes();
798 
799 		// Walk up the class hierarchy looking for a matching parent method with @Beanp or @Name
800 		ClassInfo currentClass = ClassInfo.of(c);
801 		ClassInfo sc = currentClass.getSuperclass();
802 
803 		while (nn(sc) && ! sc.is(stopClass) && ! sc.is(Object.class)) {
804 			// Look for a method with the same signature in the parent class
805 			for (var parentMethod : sc.getDeclaredMethods()) {
806 				if (parentMethod.getSimpleName().equals(methodName) && paramTypes.size() == parentMethod.getParamTypes().size()) {
807 
808 					// Check if parameter types match
809 					boolean paramsMatch = true;
810 					List<ClassInfo> parentParamTypes = parentMethod.getParamTypes();
811 					for (int i = 0; i < paramTypes.size(); i++) {
812 						if (! paramTypes.get(i).is(parentParamTypes.get(i).inner())) {
813 							paramsMatch = false;
814 							break;
815 						}
816 					}
817 
818 					if (paramsMatch) {
819 						// Found a matching parent method - check for @Beanp and @Name annotations
820 						ctx.forEachAnnotation(Beanp.class, parentMethod.inner(), x -> true, x -> lp.add(x));
821 						ctx.forEachAnnotation(Name.class, parentMethod.inner(), x -> true, x -> ln.add(x));
822 
823 						// If we found annotations, we're done
824 						if (! lp.isEmpty() || ! ln.isEmpty())
825 							return;
826 					}
827 				}
828 			}
829 
830 			// Move to the next superclass
831 			sc = sc.getSuperclass();
832 		}
833 	}
