13  package org.apache.juneau.http.resource;
14  
15  import org.apache.http.*;
16  import org.apache.juneau.http.header.*;
17  
18  /**
19   * An extension of an {@link HttpEntity} that also includes arbitrary headers.
20   *
21   * <p>
22   * While {@link HttpEntity} beans support <c>Content-Type</c>, <c>Content-Encoding</c>, and <c>Content-Length</c>
23   * headers, this interface allows you to add any number of arbitrary headers to an entity.
24   *
25   * <p>
26   * For example, you might want to be able to create an entity with a <c>Cache-Control</c> header.
27   *
28   * <p class='bjava'>
29   * 	<jk>import static</jk> org.apache.juneau.http.HttpResources.*;
30   *
31   *	<jc>// Create a resource with dynamic content and a no-cache header.</jc>
32   * 	HttpResource <jv>myResource</jv> = <jsm>stringResource</jsm>(()-&gt;<jsm>getMyContent</jsm>(), ContentType.<jsf>TEXT_PLAIN</jsf>)
33   * 		.header(<js>"Cache-Control"</js>, <js>"no-cache"</js>)
34   * 		.build();
35   * </p>
36   *
37   * <h5 class='section'>See Also:</h5><ul>
38   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestCommonBasics">juneau-rest-common</a>
39   * </ul>
40   */
41  public interface HttpResource extends HttpEntity {
42  
43  	/**
44  	 * Returns the list of headers associated with this resource.
45  	 *
46  	 * <p>
47  	 * Note that this typically does NOT include headers associated with {@link HttpEntity}
48  	 * (e.g. <c>Content-Type</c>, <c>Content-Encoding</c>, and <c>Content-Length</c>).
49  	 *
50  	 * @return The list of headers associated with this resource.
51  	 */
52  	HeaderList getHeaders();
53  }
