31  /**
32   * A basic {@link org.apache.http.HttpEntity} implementation with additional features.
33   *
34   * Provides the following features:
35   * <ul class='spaced-list'>
36   * 	<li>
37   * 		Caching.
38   * 	<li>
39   * 		Fluent setters.
40   * 	<li>
41   * 		Fluent assertions.
42   * 	<li>
43   * 		Externally-supplied/dynamic content.
44   * </ul>
45   *
46   * <h5 class='section'>See Also:</h5><ul>
47   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestCommonBasics">juneau-rest-common Basics</a>
48   * </ul>
49   */
50  @BeanIgnore
51  @SuppressWarnings("resource")
52  public class BasicHttpEntity implements HttpEntity {
53  	/**
54  	 * An empty HttpEntity.
55  	 */
56  	public static final BasicHttpEntity EMPTY = new BasicHttpEntity().setUnmodifiable();
57  	private boolean cached, chunked, unmodifiable;
58  	private Object content;
59  	private Supplier<?> contentSupplier;
60  	private ContentType contentType;
61  	private ContentEncoding contentEncoding;
62  	private Charset charset;
63  	private long contentLength = -1;
64  	private int maxLength = -1;
65  
66  	/**
67  	 * Constructor.
68  	 */
69  	public BasicHttpEntity() {
70  	}
71  
72  	/**
73  	 * Copy constructor.
74  	 *
75  	 * @param copyFrom The bean being copied.
76  	 */
77  	public BasicHttpEntity(BasicHttpEntity copyFrom) {
78  		this.cached = copyFrom.cached;
79  		this.chunked = copyFrom.chunked;
80  		this.content = copyFrom.content;
81  		this.contentSupplier = copyFrom.contentSupplier;
82  		this.contentType = copyFrom.contentType;
83  		this.contentEncoding = copyFrom.contentEncoding;
84  		this.contentLength = copyFrom.contentLength;
85  		this.charset = copyFrom.charset;
86  		this.maxLength = copyFrom.maxLength;
87  		this.unmodifiable = false;
88  	}
89  
90  	/**
91  	 * Constructor.
92  	 *
93  	 * @param contentType The entity content type.
94  	 * @param content The entity content.
95  	 */
96  	public BasicHttpEntity(ContentType contentType, Object content) {
97  		this.contentType = contentType;
98  		this.content = content;
99  	}

101 	/**
102 	 * Converts the contents of this entity as a byte array.
103 	 *
104 	 * <p>
105 	 * Note that this may exhaust the content on non-repeatable, non-cached entities.
106 	 *
107 	 * @return The contents of this entity as a byte array.
108 	 * @throws IOException If a problem occurred while trying to read the content.
109 	 */
110 	public byte[] asBytes() throws IOException {
111 		return readBytes(getContent());
112 	}

120 	 * @throws IOException If a problem occurred while trying to read the byte array.
121 	 */

123 		return new FluentByteArrayAssertion<>(asBytes(), this);
124 	}

407 	/**
408 	 * Same as {@link #asBytes()} but wraps {@link IOException IOExceptions} inside a {@link RuntimeException}.
409 	 *
410 	 * @return The contents of this entity as a byte array.
411 	 */
412 	protected byte[] asSafeBytes() {
413 		try {
414 			return asBytes();
415 		} catch (IOException e) {
416 			throw new UncheckedIOException(e);
417 		}
418 	}
419 
420 	/**
421 	 * Throws an {@link UnsupportedOperationException} if the unmodifiable flag is set on this bean.
422 	 */
423 	protected final void assertModifiable() {
424 		if (unmodifiable)
425 			throw new UnsupportedOperationException("Bean is read-only");
426 	}

428 	/**
429 	 * Returns the content of this entity.
430 	 *
431 	 * @param <T> The value type.
432 	 * @param def The default value if <jk>null</jk>.
433 	 * @return The content object.
434 	 */
435 	@SuppressWarnings("unchecked")
436 	protected <T> T contentOrElse(T def) {
437 		Object o = content;
438 		if (o == null && contentSupplier != null)
439 			o = contentSupplier.get();
440 		return (o == null ? def : (T)o);
441 	}

445 	 *
446 	 * <p>
447 	 * Externally supplied content generally means the content length cannot be reliably determined
448 	 * based on the content.
449 	 *
450 	 * @return <jk>true</jk> if the contents of this entity are provided through an external supplier.
451 	 */
452 	protected boolean isSupplied() {
453 		return contentSupplier != null;
