15  import static org.apache.juneau.common.internal.ThrowableUtils.*;
16  
17  import java.lang.reflect.*;
18  import java.util.concurrent.*;
19  
20  import org.apache.juneau.common.internal.*;
21  import org.apache.juneau.http.annotation.*;
22  import org.apache.juneau.httppart.*;
23  import org.apache.juneau.reflect.*;

25  /**
26   * Holds metadata about http part beans (POJOs that get serialized as HTTP parts such as form data or query parameters).
27   *
28   * <p>
29   * HTTP part beans are typically annotated with {@link Query @Query}, {@link FormData @FormData}, or {@link Path @Path}
30   * although it's not an absolute requirement.
31   *
32   * <p>
33   * HTTP part beans must have one of the following public constructors:
34   * <ul>
35   * 	<li><c><jk>public</jk> X(String <jv>partValue</jv>)</c>
36   * 	<li><c><jk>public</jk> X(Object <jv>partValue</jv>)</c>
37   * 	<li><c><jk>public</jk> X(String <jv>partName</jv>, String <jv>partValue</jv>)</c>
38   * 	<li><c><jk>public</jk> X(String <jv>partName</jv>, Object <jv>partValue</jv>)</c>
39   * </ul>
40   *
41   * <p>
42   * <h5 class='figure'>Example</h5>
43   * <p class='bjava'>
44   * 	<jc>// Our part bean.</jc>
45   * 	<ja>@Query</ja>(<js>"foo"</js>)
46   * 	<jk>public class</jk> FooPart <jk>extends</jk> BasicStringPart {
47   *
48   * 		<jk>public</jk> FooPart(String <jv>partValue</jv>) {
49   * 			<jk>super</jk>(<js>"foo"</js>, <jv>partValue</jv>);
50   * 		}
51   *  }
52   *
53   *  <jc>// Code to retrieve a part bean from a part list in a request.</jc>
54   *  PartList <jv>parts</jv> = <jv>httpRequest</jv>.getFormDataList();
55   *  FooPart <jv>foo</jv> = <jv>parts</jv>.get(FooPart.<jk>class</jk>);
56   * </p>
57   *
58   * <h5 class='section'>See Also:</h5><ul>
59   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestCommonBasics">juneau-rest-common Basics</a>
60   * </ul>
61   *
62   * @param <T> The HTTP part bean type.
63   */
64  public class PartBeanMeta<T> {
65  
66  	private static final ConcurrentHashMap<Class<?>,PartBeanMeta<?>> CACHE = new ConcurrentHashMap<>();
67  
68  	private final Class<T> type;
69  	private final Constructor<T> constructor;
70  	private final HttpPartSchema schema;
71  
72  	/**
73  	 * Finds the part bean meta for the specified type.
74  	 *
75  	 * @param <T> The part bean type.
76  	 * @param type The part bean type.
77  	 * @return The metadata, or <jk>null</jk> if a valid constructor could not be found.
78  	 */
79  	@SuppressWarnings("unchecked")
80  	public static <T> PartBeanMeta<T> of(Class<T> type) {
81  		PartBeanMeta<?> m = CACHE.get(type);
82  		if (m == null) {
83  			m = new PartBeanMeta<>(type);
84  			CACHE.put(type, m);
85  		}
86  		return (PartBeanMeta<T>)m;
87  	}
88  
89  	private PartBeanMeta(Class<T> type) {
90  		this.type = type;
91  
92  		ClassInfo ci = ClassInfo.of(type);
93  
94  		ConstructorInfo cci = ci.getPublicConstructor(x -> x.hasParamTypes(String.class));
95  		if (cci == null)
96  			cci = ci.getPublicConstructor(x -> x.hasParamTypes(Object.class));
97  		if (cci == null)
98  			cci = ci.getPublicConstructor(x -> x.hasParamTypes(String.class, String.class));
99  		if (cci == null)
100 			cci = ci.getPublicConstructor(x -> x.hasParamTypes(String.class, Object.class));
101 		constructor = cci == null ? null : cci.inner();
102 
103 		if (ci.hasAnnotation(Query.class))
104 			this.schema = HttpPartSchema.create(Query.class, type);
105 		else if (ci.hasAnnotation(FormData.class))
106 			this.schema = HttpPartSchema.create(FormData.class, type);
107 		else if (ci.hasAnnotation(Path.class))
108 			this.schema = HttpPartSchema.create(Path.class, type);
109 		else
110 			this.schema = HttpPartSchema.create(org.apache.juneau.http.annotation.Header.class, type);
111 	}
112 
113 	/**
114 	 * Returns schema information about this part.
115 	 *
116 	 * <p>
117 	 * This is information pulled from {@link Query @Query}, {@link FormData @FormData}, or {@link Path @Path} annotations
118 	 * on the class.
119 	 *
120 	 * @return The schema information.
121 	 */
122 	public HttpPartSchema getSchema() {
123 		return schema;
124 	}
125 
126 	/**
127 	 * Constructs a part bean with the specified name or value.
128 	 *
129 	 * <p>
130 	 * Can only be used on beans where the part name is known.
131 	 *
132 	 * @param value
133 	 * 	The part value.
134 	 * @return A newly constructed bean.
135 	 */
136 	public T construct(Object value) {
137 		return construct(null, value);
138 	}
139 
140 	/**
141 	 * Constructs a part bean with the specified name or value.
142 	 *
143 	 * @param name
144 	 * 	The part name.
145 	 * 	<br>If <jk>null</jk>, uses the value pulled from the {@link org.apache.juneau.http.annotation.Header#name() @Header(name)} or
146 	 * 	{@link org.apache.juneau.http.annotation.Header#value() @Header(value)} annotations.
147 	 * @param value
148 	 * 	The part value.
149 	 * @return A newly constructed bean.
150 	 * @throws UnsupportedOperationException If bean could not be constructed (e.g. couldn't find a constructor).
151 	 */
152 	public T construct(String name, Object value) {
153 
154 		if (constructor == null)
155 			throw new UnsupportedOperationException("Constructor for type "+type.getName()+" could not be found.");
156 
157 		if (name == null)
158 			name = schema.getName();
159 
160 		Class<?>[] pt = constructor.getParameterTypes();
161 		Object[] args = new Object[pt.length];
162 		if (pt.length == 1) {
163 			args[0] = pt[0] == String.class ? Utils.s(value) : value;
164 		} else {
165 			if (name == null)
166 				throw new UnsupportedOperationException("Constructor for type "+type.getName()+" requires a name as the first argument.");
167 			args[0] = name;
168 			args[1] = pt[1] == String.class ? Utils.s(value) : value;
169 		}
170 
171 		try {
172 			return constructor.newInstance(args);
173 		} catch (Exception e) {
174 			throw asRuntimeException(e);
175 		}
176 	}
177 }
