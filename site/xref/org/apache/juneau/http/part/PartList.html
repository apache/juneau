31  /**
32   * An simple list of HTTP parts (form-data, query-parameters, path-parameters).
33   *
34   * <h5 class='figure'>Example</h5>
35   * <p class='bjava'>
36   * 	PartList <jv>parts</jv> = PartList
37   * 		.<jsm>create</jsm>()
38   * 		.append(MyPart.<jsm>of</jsm>(<js>"foo"</js>))
39   * 		.append(<js>"Bar"</js>, ()-&gt;<jsm>getDynamicValueFromSomewhere</jsm>());
40   * </p>
41   *
42   * <h5 class='section'>See Also:</h5><ul>

94  	/**
95  	 * Creates a new {@link PartList} initialized with the specified name/value pairs.
96  	 *
97  	 * <h5 class='figure'>Example</h5>
98  	 * <p class='bjava'>
99  	 * 	PartList <jv>parts</jv> = PartList.<jsm>ofPairs</jsm>(<js>"foo"</js>, 1, <js>"bar"</js>, <jk>true</jk>);
100 	 * </p>
101 	 *
102 	 * @param pairs
103 	 * 	Initial list of pairs.
104 	 * 	<br>Must be an even number of parameters representing key/value pairs.
105 	 * @throws RuntimeException If odd number of parameters were specified.
106 	 * @return A new instance.
107 	 */
108 	public static PartList ofPairs(String... pairs) {
109 		PartList x = new PartList();
110 		if (pairs == null)
111 			pairs = new String[0];
112 		if (pairs.length % 2 != 0)
113 			throw new IllegalArgumentException("Odd number of parameters passed into PartList.ofPairs()");
114 		for (int i = 0; i < pairs.length; i+=2)
115 			x.add(BasicPart.of(pairs[i], pairs[i+1]));
116 		return x;
117 	}

479 	 * If no part with the same name is found the given part is added to the end of the list.

484 	@FluentSetter
485 	public PartList set(NameValuePair value) {
486 		if (value != null) {
487 			boolean replaced = false;
488 			for (int i = 0, j = size(); i < j; i++) {
489 				NameValuePair x = get(i);
490 				if (eq(x.getName(), value.getName())) {
491 					if (replaced) {
492 						remove(i);
493 						j--;
494 					} else {
495 						set(i, value);
496 						replaced = true;
497 					}
498 				}
499 			}
500 
501 			if (! replaced)
502 				add(value);
503 		}
504 
505 		return this;
506 	}

617 	/**
618 	 * Gets a part representing all of the part values with the given name.
619 	 *
620 	 * <p>
621 	 * If more that one part with the given name exists the values will be combined with <js>", "</js> as per
622 	 * <a href='https://tools.ietf.org/html/rfc2616#section-4.2'>RFC 2616 Section 4.2</a>.
623 	 *
624 	 * @param name The part name.
625 	 * @return A part with a condensed value, or {@link Optional#empty()} if no parts by the given name are present
626 	 */
627 	public Optional<NameValuePair> get(String name) {
628 
629 		NameValuePair first = null;
630 		List<NameValuePair> rest = null;
631 		for (NameValuePair x : this) {
632 			if (eq(x.getName(), name)) {
633 				if (first == null)
634 					first = x;
635 				else {
636 					if (rest == null)
637 						rest = Utils.list();
638 					rest.add(x);
639 				}
640 			}
641 		}
642 
643 		if (first == null)
644 			return opte();
645 
646 		if (rest == null)
647 			return Utils.opt(first);
648 
649 		CharArrayBuffer sb = new CharArrayBuffer(128);
650 		sb.append(first.getValue());
651 		for (NameValuePair element : rest) {
652 			sb.append(',');
653 			sb.append(element.getValue());
654 		}
655 
656 		return Utils.opt(new BasicStringPart(name, sb.toString()));
657 	}

659 	/**
660 	 * Gets a part representing all of the part values with the given name.
661 	 *
662 	 * <p>
663 	 * If more that one part with the given name exists the values will be combined with <js>", "</js> as per
664 	 * <a href='https://tools.ietf.org/html/rfc2616#section-4.2'>RFC 2616 Section 4.2</a>.
665 	 *
666 	 * <p>
667 	 * The implementation class must have a public constructor taking in one of the following argument lists:
668 	 * <ul>
669 	 * 	<li><c>X(String <jv>value</jv>)</c>
670 	 * 	<li><c>X(Object <jv>value</jv>)</c>
671 	 * 	<li><c>X(String <jv>name</jv>, String <jv>value</jv>)</c>
672 	 * 	<li><c>X(String <jv>name</jv>, Object <jv>value</jv>)</c>
673 	 * </ul>
674 	 *
675 	 * <h5 class='figure'>Example</h5>
676 	 * <p class='bjava'>
677 	 * 	BasicIntegerPart <jv>age</jv> = <jv>partList</jv>.get(<js>"age"</js>, BasicIntegerPart.<jk>class</jk>);
678 	 * </p>
679 	 *
680 	 * @param <T> The part implementation class.
681 	 * @param name The part name.
682 	 * @param type The part implementation class.
683 	 * @return A part with a condensed value or <jk>null</jk> if no parts by the given name are present
684 	 */
685 	public <T> Optional<T> get(String name, Class<T> type) {
686 
687 		NameValuePair first = null;
688 		List<NameValuePair> rest = null;
689 		for (NameValuePair x : this) {
690 			if (eq(x.getName(), name)) {
691 				if (first == null)
692 					first = x;
693 				else {
694 					if (rest == null)
695 						rest = Utils.list();
696 					rest.add(x);
697 				}
698 			}
699 		}
700 
701 		if (first == null)
