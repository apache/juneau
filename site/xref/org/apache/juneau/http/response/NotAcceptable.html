31  /**
32   * Exception representing an HTTP 406 (Not Acceptable).
33   *
34   * <br>
35   * The requested resource is capable of generating only content not acceptable according to the Accept headers sent in the request.
36   *
37   * <h5 class='section'>See Also:</h5><ul>
38   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestCommonBasics">juneau-rest-common Basics</a>
39   * </ul>
40   *
41   * @serial exclude
42   */
43  @Response
44  @StatusCode(STATUS_CODE)
45  @Schema(description=REASON_PHRASE)
46  public class NotAcceptable extends BasicHttpException {
47  	private static final long serialVersionUID = 1L;
48  
49  	/** HTTP status code */
50  	public static final int STATUS_CODE = 406;
51  
52  	/** Reason phrase */
53  	public static final String REASON_PHRASE = "Not Acceptable";
54  
55  	/** Default status line */
56  	private static final BasicStatusLine STATUS_LINE = BasicStatusLine.create(STATUS_CODE, REASON_PHRASE);
57  
58  	/** Reusable unmodifiable instance */
59  	public static final NotAcceptable INSTANCE = new NotAcceptable().setUnmodifiable();
60  
61  	/**
62  	 * Constructor.
63  	 */
64  	public NotAcceptable() {
65  		this((Throwable)null, REASON_PHRASE);
66  	}
67  
68  	/**
69  	 * Constructor.
70  	 *
71  	 * <p>
72  	 * This is the constructor used when parsing an HTTP response.
73  	 *
74  	 * @param response The HTTP response to copy from.  Must not be <jk>null</jk>.
75  	 * @throws AssertionError If HTTP response status code does not match what was expected.
76  	 */
77  	public NotAcceptable(HttpResponse response) {
78  		super(response);
79  		assertStatusCode(response);
80  	}
81  
82  	/**
83  	 * Constructor.
84  	 *
85  	 * @param msg The message.  Can be <jk>null</jk>.
86  	 * @param args Optional {@link MessageFormat}-style arguments in the message.
87  	 */
88  	public NotAcceptable(String msg, Object...args) {
89  		this((Throwable)null, msg, args);
90  	}
91  
92  	/**
93  	 * Constructor.
94  	 *
95  	 * @param cause The cause.  Can be <jk>null</jk>.
96  	 */
97  	public NotAcceptable(Throwable cause) {
98  		this(cause, cause == null ? REASON_PHRASE : cause.getMessage());
99  	}
100 
101 	/**
102 	 * Constructor.
103 	 *
104 	 * @param cause The caused-by exception.  Can be <jk>null</jk>.
105 	 * @param msg The message.  Can be <jk>null</jk>.
106 	 * @param args The message arguments.
107 	 */
108 	public NotAcceptable(Throwable cause, String msg, Object...args) {
109 		super(STATUS_CODE, cause, msg, args);
110 		setStatusLine(STATUS_LINE.copy());
111 	}
112 
113 	/**
114 	 * Copy constructor.
115 	 *
116 	 * @param copyFrom The bean to copy.
117 	 */
118 	protected NotAcceptable(NotAcceptable copyFrom) {
119 		super(copyFrom);
120 	}
121 
122 	/**
123 	 * Creates a modifiable copy of this bean.
124 	 *
125 	 * @return A new modifiable bean.
126 	 */
127 	public NotAcceptable copy() {
128 		return new NotAcceptable(this);
129 	}
