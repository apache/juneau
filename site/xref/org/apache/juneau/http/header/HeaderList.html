34  /**
35   * A simple list of HTTP headers with various convenience methods.
36   *
37   * <h5 class='figure'>Example</h5>
38   * <p class='bjava'>
39   * 	HeaderList <jv>headers</jv> = HeaderList
40   * 		.<jsm>create</jsm>()
41   * 		.append(Accept.<jsm>of</jsm>(<js>"text/xml"</js>))
42   * 		.append(<js>"Content-Type"</js>, ()-&gt;<jsm>getDynamicContentTypeFromSomewhere</jsm>());
43   * </p>
44   *
45   * <p>
46   * Convenience creators are provided for creating lists with minimal code:
47   * <p class='bjava'>
48   * 	HeaderList <jv>headers</jv> = HeaderList.<jsm>of</jsm>(Accept.<jsf>TEXT_XML</jsf>, ContentType.<jsf>TEXT_XML</jsf>);
49   * </p>
50   *
51   * <p>
52   * Static methods are provided on {@link HttpHeaders} to further simplify creation of header lists.
53   * <p class='bjava'>
54   * 	<jk>import static</jk> org.apache.juneau.http.HttpHeaders.*;
55   *
56   * 	HeaderList <jv>headers</jv> = <jsm>headerList</jsm>(<jsm>accept</jsm>(<js>"text/xml"</js>), <jsm>contentType</jsm>(<js>"text/xml"</js>));
57   * </p>
58   *
59   * <h5 class='section'>See Also:</h5><ul>
60   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestCommonBasics">juneau-rest-common Basics</a>
61   * </ul>
62   */

106 	/**
107 	 * Creates a new {@link HeaderList} initialized with the specified name/value pairs.
108 	 *
109 	 * <h5 class='figure'>Example</h5>
110 	 * <p class='bjava'>
111 	 * 	HeaderList <jv>headers</jv> = HeaderList.<jsm>ofPairs</jsm>(<js>"Accept"</js>, <js>"text/xml"</js>, <js>"Content-Type"</js>, <js>"text/xml"</js>);
112 	 * </p>
113 	 *
114 	 * @param pairs
115 	 * 	Initial list of pairs.
116 	 * 	<br>Must be an even number of parameters representing key/value pairs.
117 	 * @throws RuntimeException If odd number of parameters were specified.
118 	 * @return A new instance.
119 	 */
120 	public static HeaderList ofPairs(String...pairs) {
121 		var x = new HeaderList();
122 		if (pairs == null)
123 			pairs = new String[0];
124 		if (pairs.length % 2 != 0)
125 			throw new IllegalArgumentException("Odd number of parameters passed into HeaderList.ofPairs()");
126 		for (int i = 0; i < pairs.length; i += 2)
127 			x.add(BasicHeader.of(pairs[i], pairs[i + 1]));
128 		return x;
129 	}

312 	 *

314 	 * Same as {@link #get(String, Class)} but the header name is pulled from the {@link org.apache.juneau.http.annotation.Header#name()} or

316 	 *
317 	 * <h5 class='figure'>Example</h5>
318 	 * <p class='bjava'>
319 	 * 	Age <jv>age</jv> = headerList.get(Age.<jk>class</jk>);
320 	 * </p>
321 	 *
322 	 * @param <T> The return type.
323 	 * @param type The header implementation class.
324 	 * @return A header with a condensed value or <jk>null</jk> if no headers by the given name are present
325 	 */
326 	public <T> Optional<T> get(Class<T> type) {
327 		assertArgNotNull("type", type);
328 
329 		String name = HeaderBeanMeta.of(type).getSchema().getName();
330 		assertArg(name != null, "Header name could not be found on bean type ''{0}''", type.getName());
331 
332 		return get(name, type);
333 	}

335 	/**
336 	 * Gets a header representing all of the header values with the given name.
337 	 *
338 	 * <p>
339 	 * If more that one header with the given name exists the values will be combined with <js>", "</js> as per
340 	 * <a href='https://tools.ietf.org/html/rfc2616#section-4.2'>RFC 2616 Section 4.2</a>.
341 	 *
342 	 * @param name The header name.
343 	 * @return A header with a condensed value, or {@link Optional#empty()} if no headers by the given name are present
344 	 */
345 	public Optional<Header> get(String name) {
346 
347 		Header first = null;
348 		List<Header> rest = null;
349 		for (Header x : this) {
350 			if (eq(x.getName(), name)) {
351 				if (first == null)
352 					first = x;
353 				else {
354 					if (rest == null)
355 						rest = list();
356 					rest.add(x);
357 				}
358 			}
359 		}
360 
361 		if (first == null)
362 			return opte();
363 
364 		if (rest == null)
365 			return opt(first);
366 
367 		var sb = new CharArrayBuffer(128);
368 		sb.append(first.getValue());
369 		for (Header element : rest) {
370 			sb.append(", ");
371 			sb.append(element.getValue());
372 		}
373 
374 		return opt(new BasicHeader(name, sb.toString()));
375 	}

377 	/**
378 	 * Gets a header representing all of the header values with the given name.
379 	 *
380 	 * <p>
381 	 * If more that one header with the given name exists the values will be combined with <js>", "</js> as per
382 	 * <a href='https://tools.ietf.org/html/rfc2616#section-4.2'>RFC 2616 Section 4.2</a>.
383 	 *
384 	 * <p>
385 	 * The implementation class must have a public constructor taking in one of the following argument lists:
386 	 * <ul>
387 	 * 	<li><c>X(String <jv>value</jv>)</c>
388 	 * 	<li><c>X(Object <jv>value</jv>)</c>
389 	 * 	<li><c>X(String <jv>name</jv>, String <jv>value</jv>)</c>
390 	 * 	<li><c>X(String <jv>name</jv>, Object <jv>value</jv>)</c>
391 	 * </ul>
392 	 *
393 	 * <h5 class='figure'>Example</h5>
394 	 * <p class='bjava'>
395 	 * 	BasicIntegerHeader <jv>age</jv> = headerList.get(<js>"Age"</js>, BasicIntegerHeader.<jk>class</jk>);
396 	 * </p>
397 	 *
398 	 * @param <T> The header implementation class.
399 	 * @param name The header name.
400 	 * @param type The header implementation class.
401 	 * @return A header with a condensed value or <jk>null</jk> if no headers by the given name are present
402 	 */
403 	public <T> Optional<T> get(String name, Class<T> type) {
404 
405 		Header first = null;
406 		List<Header> rest = null;
407 		for (Header x : this) {
408 			if (eq(x.getName(), name)) {
409 				if (first == null)
410 					first = x;
411 				else {
412 					if (rest == null)
413 						rest = list();
414 					rest.add(x);
415 				}
416 			}
417 		}
418 
419 		if (first == null)
420 			return opte();
421 
422 		if (rest == null)
423 			return opt(HeaderBeanMeta.of(type).construct(name, first.getValue()));
424 
425 		var sb = new CharArrayBuffer(128);
426 		sb.append(first.getValue());
427 		for (Header element : rest) {
428 			sb.append(", ");
429 			sb.append(element.getValue());
430 		}
431 
432 		return opt(HeaderBeanMeta.of(type).construct(name, sb.toString()));
433 	}
