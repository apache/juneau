44  /**
45   * A wrapper class around the {@link Class} object that provides cached information about that class.
46   *
47   * <p>
48   * Instances of this class can be created through the {@link BeanContext#getClassMeta(Class)} method.
49   *
50   * <p>
51   * The {@link BeanContext} class will cache and reuse instances of this class except for the following class types:
52   * <ul>
53   * 	<li>Arrays
54   * 	<li>Maps with non-Object key/values.
55   * 	<li>Collections with non-Object key/values.
56   * </ul>
57   *
58   * <p>
59   * This class is tied to the {@link BeanContext} class because it's that class that makes the determination of what is
60   * a bean.
61   *
62   * <h5 class='section'>See Also:</h5><ul>
63  
64   * </ul>
65   *
66   * @param <T> The class type of the wrapped class.
67   */
68  @Bean(properties = "innerClass,classCategory,elementType,keyType,valueType,notABeanReason,initException,beanMeta")
69  public class ClassMeta<T> implements Type {

71  	@SuppressWarnings({ "unchecked", "rawtypes", "hiding" })
72  	private class ClassMetaBuilder<T> {
73  		Class<T> innerClass;
74  		ClassInfo ci;
75  		Class<? extends T> implClass;
76  		BeanContext beanContext;
77  		ClassCategory cc = ClassCategory.OTHER;
78  		boolean isDelegate = false, isMemberClass = false, isAbstract = false;
79  		Method fromStringMethod = null;
80  		Setter parentPropertyMethod = null, namePropertyMethod = null;
81  		ConstructorInfo noArgConstructor = null, stringConstructor = null;
82  		Object primitiveDefault = null;
83  		Map<String,Method> publicMethods = CollectionUtils.map();
84  		ClassMeta<?> keyType = null, valueType = null, elementType = null;
85  		String typePropertyName = null, notABeanReason = null, dictionaryName = null;
86  		Throwable initException = null;
87  		BeanMeta beanMeta = null;
88  		List<ObjectSwap> swaps = list();
89  		BuilderSwap builderSwap;
90  		InvocationHandler invocationHandler = null;
91  		BeanRegistry beanRegistry = null;
92  		ObjectSwap<?,?>[] childSwaps;
93  		ConcurrentHashMap<Class<?>,ObjectSwap<?,?>> childSwapMap, childUnswapMap;
94  		Method exampleMethod;
95  		Field exampleField;
96  		String example;
97  		Mutater<String,T> stringMutater;
98  		BiMap.Builder<Object,String> enumValues;

100 		ClassMetaBuilder(Class<T> innerClass, BeanContext beanContext, ObjectSwap<T,?>[] swaps, ObjectSwap<?,?>[] childSwaps) {
101 			this.innerClass = innerClass;
102 			this.beanContext = beanContext;
103 			BeanContext bc = beanContext;
104 
105 			this.childSwaps = childSwaps;
106 			if (childSwaps == null) {
107 				this.childSwapMap = null;
108 				this.childUnswapMap = null;
109 			} else {
110 				this.childSwapMap = new ConcurrentHashMap<>();
111 				this.childUnswapMap = new ConcurrentHashMap<>();
112 			}
113 
114 			Class<T> c = innerClass;
115 			ci = ClassInfo.of(c);
116 
117 			if (c.isPrimitive()) {
118 				if (c == Boolean.TYPE)
119 					cc = BOOLEAN;
120 				else if (c == Byte.TYPE || c == Short.TYPE || c == Integer.TYPE || c == Long.TYPE || c == Float.TYPE || c == Double.TYPE) {
121 					if (c == Float.TYPE || c == Double.TYPE)
122 						cc = DECIMAL;
123 					else
124 						cc = NUMBER;
125 				} else if (c == Character.TYPE)
126 					cc = CHAR;
127 				else if (c == void.class || c == Void.class)
128 					cc = VOID;
129 			} else {
130 				if (ci.isChildOf(Delegate.class))
131 					isDelegate = true;
132 
133 				if (c == Object.class)
134 					cc = OBJ;
135 				else if (c.isEnum())
136 					cc = ENUM;
137 				else if (c.equals(Class.class))
138 					cc = ClassCategory.CLASS;
139 				else if (ci.isChildOf(Method.class))
140 					cc = METHOD;
141 				else if (ci.isChildOf(CharSequence.class)) {
142 					if (c.equals(String.class))
143 						cc = STR;
144 					else
145 						cc = CHARSEQ;
146 				} else if (ci.isChildOf(Number.class)) {
147 					if (ci.isChildOfAny(Float.class, Double.class))
148 						cc = DECIMAL;
149 					else
150 						cc = NUMBER;
151 				} else if (ci.isChildOf(Collection.class))
152 					cc = COLLECTION;
153 				else if (ci.isChildOf(Map.class)) {
154 					if (ci.isChildOf(BeanMap.class))
155 						cc = BEANMAP;
156 					else
157 						cc = MAP;
158 				} else if (c == Character.class)
159 					cc = CHAR;
160 				else if (c == Boolean.class)
161 					cc = BOOLEAN;
162 				else if (ci.isChildOfAny(Date.class, Calendar.class))
163 					cc = DATE;
164 				else if (c.isArray())
165 					cc = ARRAY;
166 				else if (ci.isChildOfAny(URL.class, URI.class) || ci.hasAnnotation(bc, Uri.class))
167 					cc = URI;
168 				else if (ci.isChildOf(Reader.class))
169 					cc = READER;
170 				else if (ci.isChildOf(InputStream.class))
171 					cc = INPUTSTREAM;
172 				else if (ci.is(Optional.class))
173 					cc = OPTIONAL;
174 			}
175 
176 			isMemberClass = ci.isMemberClass() && ci.isNotStatic();
177 
178 			// Find static fromString(String) or equivalent method.
179 			// fromString() must be checked before valueOf() so that Enum classes can create their own
180 			//		specialized fromString() methods to override the behavior of Enum.valueOf(String).
181 			// valueOf() is used by enums.
182 			// parse() is used by the java logging Level class.
183 			// forName() is used by Class and Charset
184 			String[] fromStringMethodNames = {"fromString","fromValue","valueOf","parse","parseString","forName","forString"};
185 			// @formatter:off
186 			fromStringMethod = Utils.opt(
187 				ci.getPublicMethod(
188 					x -> x.isStatic()
189 					&& x.isNotDeprecated()
190 					&& x.hasReturnType(c)
191 					&& x.hasParamTypes(String.class)
192 					&& ArrayUtils.contains(x.getName(), fromStringMethodNames))
193 			).map(MethodInfo::inner)
194 			.orElse(null);
195 			// @formatter:on
196 
197 			// Find example() method if present.
198 			// @formatter:off
199 			exampleMethod = Utils.opt(
200 				ci.getPublicMethod(
201 					x -> x.isStatic()
202 					&& x.isNotDeprecated()
203 					&& x.hasName("example")
204 					&& x.hasFuzzyParamTypes(BeanSession.class))
205 			).map(MethodInfo::inner)
206 			.orElse(null);
207 			// @formatter:on
208 
209 			ci.forEachAllField(x -> x.hasAnnotation(bc, ParentProperty.class), x -> {
210 				if (x.isStatic())
211 					throw new ClassMetaRuntimeException(c, "@ParentProperty used on invalid field ''{0}''.  Must be static.", x);
212 				parentPropertyMethod = new Setter.FieldSetter(x.accessible().inner());
213 			});
214 
215 			ci.forEachAllField(x -> x.hasAnnotation(bc, NameProperty.class), x -> {
216 				if (x.isStatic())
217 					throw new ClassMetaRuntimeException(c, "@NameProperty used on invalid field ''{0}''.  Must be static.", x);
218 				namePropertyMethod = new Setter.FieldSetter(x.accessible().inner());
219 			});
220 
221 			ci.forEachDeclaredField(x -> x.hasAnnotation(bc, Example.class), x -> {
222 				if (! (x.isStatic() && ci.isParentOf(x.getType().inner())))
223 					throw new ClassMetaRuntimeException(c, "@Example used on invalid field ''{0}''.  Must be static and an instance of the type.", x);
224 				exampleField = x.accessible().inner();
225 			});
226 
227 			// Find @NameProperty and @ParentProperty methods if present.
228 			List<MethodInfo> methods = ci.getMethods();
229 			for (int i = methods.size() - 1; i >= 0; i--) {
230 				MethodInfo m = methods.get(i);
231 				if (m.hasAnnotation(bc, ParentProperty.class)) {
232 					if (m.isStatic() || ! m.hasNumParams(1))
233 						throw new ClassMetaRuntimeException(c, "@ParentProperty used on invalid method ''{0}''.  Must not be static and have one argument.", m);
234 					m.setAccessible();
235 					parentPropertyMethod = new Setter.MethodSetter(m.inner());
236 				}
237 				if (m.hasAnnotation(bc, NameProperty.class)) {
238 					if (m.isStatic() || ! m.hasNumParams(1))
239 						throw new ClassMetaRuntimeException(c, "@NameProperty used on invalid method ''{0}''.  Must not be static and have one argument.", m);
240 					m.setAccessible();
241 					namePropertyMethod = new Setter.MethodSetter(m.inner());
242 				}
243 			}
244 
245 			ci.forEachDeclaredMethod(m -> m.hasAnnotation(bc, Example.class), m -> {
246 				if (! (m.isStatic() && m.hasFuzzyParamTypes(BeanSession.class) && ci.isParentOf(m.getReturnType().inner())))
247 					throw new ClassMetaRuntimeException(c, "@Example used on invalid method ''{0}''.  Must be static and return an instance of the declaring class.", m.toString());
248 				m.setAccessible();
249 				exampleMethod = m.inner();
250 			});
251 
252 			// Note:  Primitive types are normally abstract.
253 			isAbstract = ci.isAbstract() && ci.isNotPrimitive();
254 
255 			// Find constructor(String) method if present.
256 			ci.forEachPublicConstructor(cs -> cs.isPublic() && cs.isNotDeprecated(), cs -> {
257 				List<ClassInfo> pt = cs.getParamTypes();
258 				if (pt.size() == (isMemberClass ? 1 : 0) && c != Object.class && ! isAbstract) {
259 					noArgConstructor = cs;
260 				} else if (pt.size() == (isMemberClass ? 2 : 1)) {
261 					ClassInfo arg = pt.get(isMemberClass ? 1 : 0);
262 					if (arg.is(String.class))
263 						stringConstructor = cs;
264 				}
265 			});
266 
267 			primitiveDefault = ci.getPrimitiveDefault();
268 
269 			// @formatter:off
270 			ci.forEachPublicMethod(
271 				MethodInfo::isNotDeprecated,
272 				x -> publicMethods.put(x.getSignature(), x.inner())
273 			);
274 			// @formatter:on
275 
276 			BeanFilter beanFilter = findBeanFilter(bc);
277 			MarshalledFilter marshalledFilter = findMarshalledFilter(bc);
278 
279 			CollectionUtils.addAll(this.swaps, swaps);
280 
281 			if (bc != null)
282 				this.builderSwap = BuilderSwap.findSwapFromObjectClass(bc, c, bc.getBeanConstructorVisibility(), bc.getBeanMethodVisibility());
283 
284 			findSwaps(this.swaps, bc);
285 
286 			if (beanFilter != null) {
287 				example = beanFilter.getExample();
288 				implClass = (Class<? extends T>)beanFilter.getImplClass();
289 			}
290 
291 			if (marshalledFilter != null) {
292 				if (example == null)
293 					example = marshalledFilter.getExample();
294 				if (implClass == null)
295 					implClass = (Class<? extends T>)marshalledFilter.getImplClass();
296 			}
297 
298 			if (innerClass != Object.class) {
299 				ClassInfo x = implClass == null ? ci : ClassInfo.of(implClass);
300 				noArgConstructor = x.getPublicConstructor(ConstructorInfo::hasNoParams);
301 			}
302 
303 			try {
304 
305 				// If this is an array, get the element type.
306 				if (cc == ARRAY)
307 					elementType = findClassMeta(innerClass.getComponentType());
308 
309 				// If this is a MAP, see if it's parameterized (e.g. AddressBook extends HashMap<String,Person>)
310 				else if (cc == MAP) {
311 					ClassMeta[] parameters = findParameters();
312 					if (parameters != null && parameters.length == 2) {
313 						keyType = parameters[0];
314 						valueType = parameters[1];
315 					} else {
316 						keyType = findClassMeta(Object.class);
317 						valueType = findClassMeta(Object.class);
318 					}
319 				}
320 
321 				// If this is a COLLECTION, see if it's parameterized (e.g. AddressBook extends LinkedList<Person>)
322 				else if (cc == COLLECTION || cc == OPTIONAL) {
323 					ClassMeta[] parameters = findParameters();
324 					if (parameters != null && parameters.length == 1) {
325 						elementType = parameters[0];
326 					} else {
327 						elementType = findClassMeta(Object.class);
328 					}
329 				}
330 
331 				// If the category is unknown, see if it's a bean.
332 				// Note that this needs to be done after all other initialization has been done.
333 				else if (cc == OTHER) {
334 
335 					BeanMeta newMeta = null;
336 					try {
337 						newMeta = new BeanMeta(ClassMeta.this, bc, beanFilter, null, implClass == null ? null : noArgConstructor);
338 						notABeanReason = newMeta.notABeanReason;
339 
340 						// Always get these even if it's not a bean:
341 						beanRegistry = newMeta.beanRegistry;
342 						typePropertyName = newMeta.typePropertyName;
343 
344 					} catch (RuntimeException e) {
345 						notABeanReason = e.getMessage();
346 						throw e;
347 					}
348 					if (notABeanReason == null)
349 						beanMeta = newMeta;
350 				}
351 
352 			} catch (NoClassDefFoundError e) {
353 				initException = e;
354 			} catch (RuntimeException e) {
355 				initException = e;
356 				throw e;
357 			}
358 
359 			if (beanMeta != null)
360 				dictionaryName = beanMeta.getDictionaryName();
361 
362 			if (beanMeta != null && bc != null && bc.isUseInterfaceProxies() && innerClass.isInterface())
363 				invocationHandler = new BeanProxyInvocationHandler<T>(beanMeta);
364 
365 			if (bc != null) {
366 				bc.forEachAnnotation(Bean.class, c, x -> true, x -> {
367 					if (x.dictionary().length != 0)
368 						beanRegistry = new BeanRegistry(bc, null, x.dictionary());
369 					// This could be a non-bean POJO with a type name.
370 					if (dictionaryName == null && ! x.typeName().isEmpty())
371 						dictionaryName = x.typeName();
372 				});
373 			}
374 
375 			if (example == null && bc != null) {
376 				bc.forEachAnnotation(Example.class, c, x -> ! x.value().isEmpty(), x -> example = x.value());
377 			}
378 
379 			if (example == null) {
380 				switch (cc) {
381 					case BOOLEAN:
382 						example = "true";
383 						break;
384 					case CHAR:
385 						example = "a";
386 						break;
387 					case CHARSEQ:
388 					case STR:
389 						example = "foo";
390 						break;
391 					case DECIMAL:
392 						if (isFloat())
393 							example = "1.0";
394 						else if (isDouble())
395 							example = "1.0";
396 						break;
397 					case ENUM:
398 						Iterator<? extends Enum> i = EnumSet.allOf((Class<? extends Enum>)c).iterator();
399 						if (i.hasNext())
400 							example = beanContext.isUseEnumNames() ? i.next().name() : i.next().toString();
401 						break;
402 					case NUMBER:
403 						if (isShort())
404 							example = "1";
405 						else if (isInteger())
406 							example = "1";
407 						else if (isLong())
408 							example = "1";
409 						break;
410 					case URI:
411 					case ARGS:
412 					case ARRAY:
413 					case BEANMAP:
414 					case CLASS:
415 					case COLLECTION:
416 					case DATE:
417 					case INPUTSTREAM:
418 					case MAP:
419 					case METHOD:
420 					case OBJ:
421 					case OTHER:
422 					case READER:
423 					case OPTIONAL:
424 					case VOID:
425 						break;
426 					default:
427 						break;
428 				}
429 			}
430 
431 			this.stringMutater = Mutaters.get(String.class, c);
432 
433 			if (cc == ENUM) {
434 				Class<? extends Enum> ec = (Class<? extends Enum<?>>)c;
435 				boolean useEnumNames = bc != null && bc.isUseEnumNames();
436 				enumValues = BiMap.create();
437 				enumValues.unmodifiable();
438 				stream(ec.getEnumConstants()).forEach(x -> enumValues.add(x, useEnumNames ? x.name() : x.toString()));
439 			}
440 		}

442 		private ObjectSwap<T,?> createSwap(Swap s) {
443 			Class<?> c = s.value();
444 			if (ClassUtils2.isVoid(c))
445 				c = s.impl();
446 			ClassInfo ci = ClassInfo.of(c);
447 
448 			if (ci.isChildOf(ObjectSwap.class)) {
449 				ObjectSwap ps = BeanCreator.of(ObjectSwap.class).type(c).run();
450 				if (s.mediaTypes().length > 0)
451 					ps.forMediaTypes(MediaType.ofAll(s.mediaTypes()));
452 				if (! s.template().isEmpty())
453 					ps.withTemplate(s.template());
454 				return ps;
455 			}
456 
457 			if (ci.isChildOf(Surrogate.class)) {
458 				List<SurrogateSwap<?,?>> l = SurrogateSwap.findObjectSwaps(c, beanContext);
459 				if (! l.isEmpty())
460 					return (ObjectSwap<T,?>)l.iterator().next();
461 			}
462 
463 			throw new ClassMetaRuntimeException(c, "Invalid swap class ''{0}'' specified.  Must extend from ObjectSwap or Surrogate.", c);
464 		}

467 			try {
468 				List<Bean> ba = info.getAnnotations(bc, Bean.class);

471 			} catch (Exception e) {
472 				throw asRuntimeException(e);
473 			}
474 			return null;

496 		private void findSwaps(List<ObjectSwap> l, BeanContext bc) {
497 
498 			if (bc != null)
499 				bc.forEachAnnotation(Swap.class, innerClass, x -> true, x -> l.add(createSwap(x)));
500 
501 			ObjectSwap defaultSwap = DefaultSwaps.find(ci);
502 			if (defaultSwap == null)
503 				defaultSwap = AutoObjectSwap.find(bc, ci);
504 			if (defaultSwap == null)
505 				defaultSwap = AutoNumberSwap.find(bc, ci);
506 			if (defaultSwap == null)
507 				defaultSwap = AutoMapSwap.find(bc, ci);
508 			if (defaultSwap == null)
509 				defaultSwap = AutoListSwap.find(bc, ci);
510 			if (defaultSwap != null)
511 				l.add(defaultSwap);
512 		}

520 	/**
521 	 * Generated classes shouldn't be cacheable to prevent needlessly filling up the cache.
522 	 */
523 	private static boolean isCacheable(Class<?> c) {
524 		String n = c.getName();
525 		char x = n.charAt(n.length() - 1);  // All generated classes appear to end with digits.
526 		if (x >= '0' && x <= '9') {
527 			if (n.indexOf("$$") != -1 || n.startsWith("sun") || n.startsWith("com.sun") || n.indexOf("$Proxy") != -1)
528 				return false;
529 		}
530 		return true;
531 	}

533 	/**
534 	 * Locates the no-arg constructor for the specified class.
535 	 *
536 	 * <p>
537 	 * Constructor must match the visibility requirements specified by parameter 'v'.
538 	 * If class is abstract, always returns <jk>null</jk>.
539 	 * Note that this also returns the 1-arg constructor for non-static member classes.
540 	 *
541 	 * @param <T> The class from which to locate the no-arg constructor.
542 	 * @param c The class from which to locate the no-arg constructor.
543 	 * @param v The minimum visibility.
544 	 * @return The constructor, or <jk>null</jk> if no no-arg constructor exists with the required visibility.
545 	 */
546 	@SuppressWarnings({ "unchecked" })
547 	protected static <T> Constructor<? extends T> findNoArgConstructor(Class<?> c, Visibility v) {
548 		ClassInfo ci = ClassInfo.of(c);
549 		if (ci.isAbstract())
550 			return null;
551 		boolean isMemberClass = ci.isMemberClass() && ci.isNotStatic();
552 		// @formatter:off
553 		ConstructorInfo cc = ci.getPublicConstructor(
554 			x -> x.isVisible(v)
555 			&& x.isNotDeprecated()
556 			&& x.hasNumParams(isMemberClass ? 1 : 0)
557 		);
558 		// @formatter:on
559 		if (cc != null)
560 			return (Constructor<? extends T>)v.transform(cc.inner());
561 		return null;
562 	}

566 	private final Class<? extends T> implClass;             // The implementation class to use if this is an interface.
567 	private final ClassCategory cc;                         // The class category.
568 	private final Method fromStringMethod;                  // The static valueOf(String) or fromString(String) or forString(String) method (if it has one).
569 	private final ConstructorInfo noArgConstructor,                                    // The no-arg constructor for this class (if it has one).
