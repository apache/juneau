179 	private static Locale toLocale(String lang) {
180 		var country = "";
181 		var i = lang.indexOf('-');
182 		if (i > -1) {
183 			country = lang.substring(i + 1).trim();
184 			lang = lang.substring(0, i).trim();
185 		}
186 		return new Locale(lang, country);
187 	}

213 	 * Constructor.
214 	 */
215 	RestRequest(RestOpContext opContext, RestSession session) throws Exception {
216 		super(session.getRequest());
217 		this.session = session;
218 		this.opContext = opContext;
219 
220 		inner = session.getRequest();
221 		context = session.getContext();
222 
223 		attrs = new RequestAttributes(this);
224 
225 		queryParams = new RequestQueryParams(this, session.getQueryParams(), true);
226 
227 		headers = new RequestHeaders(this, queryParams, false);
228 
229 		content = new RequestContent(this);
230 
231 		if (context.isAllowContentParam()) {
232 			String b = queryParams.get("content").asString().orElse(null);
233 			if (nn(b)) {
234 				headers.set("Content-Type", UonSerializer.DEFAULT.getResponseContentType());
235 				content.mediaType(MediaType.UON).parser(UonParser.DEFAULT).content(b.getBytes(UTF8));
236 			}
237 		}
238 
239 		pathParams = new RequestPathParams(session, this, true);
240 
241 		beanSession = opContext.getBeanContext().getSession();
242 
243 		partParserSession = opContext.getPartParser().getPartSession();
244 
245 		pathParams.parser(partParserSession);
246 
247 		// @formatter:off
248 		queryParams
249 			.addDefault(opContext.getDefaultRequestQueryData().getAll())
250 			.parser(partParserSession);
251 
252 		headers
253 			.addDefault(opContext.getDefaultRequestHeaders().getAll())
254 			.addDefault(context.getDefaultRequestHeaders().getAll())
255 			.parser(partParserSession);
256 
257 		content
258 			.encoders(opContext.getEncoders())
259 			.parsers(opContext.getParsers())
260 			.maxInput(opContext.getMaxInput());
261 
262 		attrs
263 			.addDefault(opContext.getDefaultRequestAttributes())
264 			.addDefault(context.getDefaultRequestAttributes());
265 		// @formatter:on
266 
267 		if (isDebug())
268 			inner = CachingHttpServletRequest.wrap(inner);
269 	}

453 	 * Returns the URI authority portion of the request.
454 	 *
455 	 * @return The URI authority portion of the request.
456 	 */
457 	public String getAuthorityPath() {
458 		if (authorityPath == null)
459 			authorityPath = context.getUriAuthority();
460 		if (authorityPath == null) {
461 			String scheme = inner.getScheme();
462 			int port = inner.getServerPort();
463 			StringBuilder sb = new StringBuilder(inner.getScheme()).append("://").append(inner.getServerName());
464 			if (! (port == 80 && "http".equals(scheme) || port == 443 && "https".equals(scheme)))
465 				sb.append(':').append(port);
466 			authorityPath = sb.toString();
467 		}
468 		return authorityPath;
469 	}

479 	 * Returns the charset specified on the <c>Content-Type</c> header, or <js>"UTF-8"</js> if not specified.
480 	 *
481 	 * @return The charset to use to decode the request content.
482 	 */
483 	public Charset getCharset() {
484 		if (charset == null) {
485 			// Determine charset
486 			// NOTE:  Don't use super.getCharacterEncoding() because the spec is implemented inconsistently.
487 			// Jetty returns the default charset instead of null if the character is not specified on the request.
488 			String h = getHeaderParam("Content-Type").orElse(null);
489 			if (nn(h)) {
490 				int i = h.indexOf(";charset=");
491 				if (i > 0)
492 					charset = Charset.forName(h.substring(i + 9).trim());
493 			}
494 			if (charset == null)
495 				charset = opContext.getDefaultCharset();
496 			if (charset == null)
497 				charset = Charset.forName("UTF-8");
498 		}
499 		return charset;
500 	}

1154 	 * Same as {@link #getRequest(Class)} but used on pre-instantiated {@link RequestBeanMeta} objects.
1155 	 *
1156 	 * @param <T> The request bean interface to instantiate.
1157 	 * @param rbm The metadata about the request bean interface to create.
1158 	 * @return A new request bean proxy for this REST request.
1159 	 */
1160 	public <T> T getRequest(final RequestBeanMeta rbm) {
1161 		try {
1162 			Class<T> c = (Class<T>)rbm.getClassMeta().getInnerClass();
1163 			final BeanSession bs = getBeanSession();
1164 			final BeanMeta<T> bm = bs.getBeanMeta(c);
1165 			return (T)Proxy.newProxyInstance(c.getClassLoader(), new Class[] { c }, (InvocationHandler)(proxy, method, args) -> {
1166 				RequestBeanPropertyMeta pm = rbm.getProperty(method.getName());
1167 				if (nn(pm)) {
1168 					HttpPartParserSession pp = pm.getParser(getPartParserSession());
1169 					HttpPartSchema schema = pm.getSchema();
1170 					String name = pm.getPartName();
1171 					ClassMeta<?> type = bs.getClassMeta(method.getGenericReturnType());
1172 					HttpPartType pt = pm.getPartType();
1173 					if (pt == HttpPartType.BODY)
1174 						return getContent().setSchema(schema).as(type);
1175 					if (pt == QUERY)
1176 						return getQueryParam(name).parser(pp).schema(schema).as(type).orElse(null);
1177 					if (pt == FORMDATA)
1178 						return getFormParam(name).parser(pp).schema(schema).as(type).orElse(null);
1179 					if (pt == HEADER)
1180 						return getHeaderParam(name).parser(pp).schema(schema).as(type).orElse(null);
1181 					if (pt == PATH)
1182 						return getPathParam(name).parser(pp).schema(schema).as(type).orElse(null);
1183 				}
1184 				return null;
1185 			});
1186 		} catch (Exception e) {
1187 			throw asRuntimeException(e);
1188 		}
1189 	}
