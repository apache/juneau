32  /**
33   * Resolves method parameters annotated with {@link HasFormData} on {@link RestOp}-annotated Java methods.
34   *
35   * <p>
36   * The parameter value is resolved using:
37   * <p class='bjava'>
38   * 	<jv>opSession</jv>
39   * 		.{@link RestOpSession#getRequest() getRequest}()
40   * 		.{@link RestRequest#getFormParams() getFormParams}()
41   * 		.{@link RequestFormParams#contains(String) contains}(<jv>name</jv>);
42   * </p>
43   *
44   * <p>
45   * The parameter type can be a <jk>boolean</jk> or anything convertible from a <jk>boolean</jk>.
46   *
47   * <h5 class='section'>See Also:</h5><ul>
48   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JavaMethodParameters">Java Method Parameters</a>
49   * </ul>
50   */
51  public class HasFormDataArg implements RestOpArg {
52  
53  	/**
54  	 * Static creator.
55  	 *
56  	 * @param paramInfo The Java method parameter being resolved.
57  	 * @return A new {@link HasFormDataArg}, or <jk>null</jk> if the parameter is not annotated with {@link HasFormData}.
58  	 */
59  	public static HasFormDataArg create(ParameterInfo paramInfo) {
60  		if (paramInfo.hasAnnotation(HasFormData.class))
61  			return new HasFormDataArg(paramInfo);
62  		return null;
63  	}
64  
65  	private static String getName(HasFormData x) {
66  		return firstNonEmpty(x.name(), x.value());
67  	}
68  
69  	private static boolean hasName(HasFormData x) {
70  		return isNotEmpty(x.name()) || isNotEmpty(x.value());
71  	}
72  
73  	private final String name;
74  
75  	private final Type type;
76  
77  	/**
78  	 * Constructor.
79  	 *
80  	 * @param pi The Java method parameter being resolved.
81  	 */
82  	protected HasFormDataArg(ParameterInfo pi) {
83  		Value<String> _name = Value.empty();
84  		pi.forEachAnnotation(HasFormData.class, HasFormDataArg::hasName, x -> _name.set(getName(x)));
85  		this.name = _name.orElseThrow(() -> new ArgException(pi, "@HasFormData used without name or value"));
86  		this.type = pi.getParameterType().innerType();
87  	}
88  
89  	@Override /* Overridden from RestOpArg */
90  	public Object resolve(RestOpSession opSession) throws Exception {
91  		RestRequest req = opSession.getRequest();
92  		BeanSession bs = req.getBeanSession();
93  		return bs.convertToType(req.getFormParams().contains(name), bs.getClassMeta(type));
94  	}
95  }
