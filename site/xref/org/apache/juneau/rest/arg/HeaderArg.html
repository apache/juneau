37  /**
38   * Resolves method parameters and parameter types annotated with {@link Header} on {@link RestOp}-annotated Java methods.
39   *
40   * <p>
41   * This includes any of the following predefined request header types:
42   *
43   * <ul class='javatree condensed'>
44   * 	<li class='jc'>{@link Accept}
45   * 	<li class='jc'>{@link AcceptCharset}
46   * 	<li class='jc'>{@link AcceptEncoding}
47   * 	<li class='jc'>{@link AcceptLanguage}
48   * 	<li class='jc'>{@link AcceptRanges}
49   * 	<li class='jc'>{@link Authorization}
50   * 	<li class='jc'>{@link CacheControl}
51   * 	<li class='jc'>{@link ClientVersion}
52   * 	<li class='jc'>{@link Connection}
53   * 	<li class='jc'>{@link ContentDisposition}
54   * 	<li class='jc'>{@link ContentEncoding}
55   * 	<li class='jc'>{@link ContentLength}
56   * 	<li class='jc'>{@link ContentType}
57   * 	<li class='jc'>{@link org.apache.juneau.http.header.Date}
58   * 	<li class='jc'>{@link Debug}
59   * 	<li class='jc'>{@link Expect}
60   * 	<li class='jc'>{@link Forwarded}
61   * 	<li class='jc'>{@link From}
62   * 	<li class='jc'>{@link Host}
63   * 	<li class='jc'>{@link IfMatch}
64   * 	<li class='jc'>{@link IfModifiedSince}
65   * 	<li class='jc'>{@link IfNoneMatch}
66   * 	<li class='jc'>{@link IfRange}
67   * 	<li class='jc'>{@link IfUnmodifiedSince}
68   * 	<li class='jc'>{@link MaxForwards}
69   * 	<li class='jc'>{@link NoTrace}
70   * 	<li class='jc'>{@link Origin}
71   * 	<li class='jc'>{@link Pragma}
72   * 	<li class='jc'>{@link ProxyAuthorization}
73   * 	<li class='jc'>{@link Range}
74   * 	<li class='jc'>{@link Referer}
75   * 	<li class='jc'>{@link TE}
76   * 	<li class='jc'>{@link Thrown}
77   * 	<li class='jc'>{@link Upgrade}
78   * 	<li class='jc'>{@link UserAgent}
79   * 	<li class='jc'>{@link Warning}
80   * </ul>
81   *
82   * <p>
83   * The parameter value is resolved using:
84   * <p class='bjava'>
85   * 	<jv>opSession</jv>
86   * 		.{@link RestOpSession#getRequest() getRequest}()
87   * 		.{@link RestRequest#getHeaders() getHeaders}();
88   * </p>
89   *
90   * <p>
91   * {@link HttpPartSchema schema} is derived from the {@link Header} annotation.
92   *
93   * <p>
94   * If the {@link Schema#collectionFormat()} value is {@link HttpPartCollectionFormat#MULTI}, then the data type can be a {@link Collection} or array.
95   *
96   * <h5 class='section'>See Also:</h5><ul>
97   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JavaMethodParameters">Java Method Parameters</a>
98   * </ul>
99   */
100 public class HeaderArg implements RestOpArg {
101 	private final HttpPartParser partParser;
102 	private final HttpPartSchema schema;
103 	private final boolean multi;
104 	private final String name, def;
105 	private final ClassInfo type;
106 
107 	/**
108 	 * Static creator.
109 	 *
110 	 * @param paramInfo The Java method parameter being resolved.
111 	 * @param annotations The annotations to apply to any new part parsers.
112 	 * @return A new {@link HeaderArg}, or <jk>null</jk> if the parameter is not annotated with {@link Header}.
113 	 */
114 	public static HeaderArg create(ParamInfo paramInfo, AnnotationWorkList annotations) {
115 		if ((!paramInfo.getParameterType().is(Value.class)) && (paramInfo.hasAnnotation(Header.class) || paramInfo.getParameterType().hasAnnotation(Header.class)))
116 			return new HeaderArg(paramInfo, annotations);
117 		return null;
118 	}
119 
120 	/**
121 	 * Constructor.
122 	 *
123 	 * @param pi The Java method parameter being resolved.
124 	 * @param annotations The annotations to apply to any new part parsers.
125 	 */
126 	protected HeaderArg(ParamInfo pi, AnnotationWorkList annotations) {
127 		// Get the header name from the parameter
128 		this.name = findName(pi).orElseThrow(() -> new ArgException(pi, "@Header used without name or value"));
129 
130 		// Check for class-level defaults and merge if found
131 		Header mergedHeader = getMergedHeader(pi, name);
132 
133 		// Use merged header annotation for all lookups
134 		this.def = mergedHeader != null && !mergedHeader.def().isEmpty() ? mergedHeader.def() : findDef(pi).orElse(null);
135 		this.type = pi.getParameterType();
136 		this.schema = mergedHeader != null ? HttpPartSchema.create(mergedHeader) : HttpPartSchema.create(Header.class, pi);
137 		Class<? extends HttpPartParser> pp = schema.getParser();
138 		this.partParser = pp != null ? HttpPartParser.creator().type(pp).apply(annotations).create() : null;
139 		this.multi = schema.getCollectionFormat() == HttpPartCollectionFormat.MULTI;
140 
141 		if (multi && ! type.isCollectionOrArray())
142 			throw new ArgException(pi, "Use of multipart flag on @Header parameter that is not an array or Collection");
143 	}
144 
145 	/**
146 	 * Gets the merged @Header annotation combining class-level and parameter-level values.
147 	 *
148 	 * @param pi The parameter info.
149 	 * @param paramName The header name.
150 	 * @return Merged annotation, or null if no class-level defaults exist.
151 	 */
152 	private static Header getMergedHeader(ParamInfo pi, String paramName) {
153 		// Get the declaring class
154 		ClassInfo declaringClass = pi.getMethod().getDeclaringClass();
155 		if (declaringClass == null)
156 			return null;
157 
158 		// Find @Rest annotation on the class
159 		Rest restAnnotation = declaringClass.getAnnotation(Rest.class);
160 		if (restAnnotation == null)
161 			return null;
162 
163 		// Find matching @Header from class-level headerParams array
164 		Header classLevelHeader = null;
165 		for (Header h : restAnnotation.headerParams()) {
166 			String hName = firstNonEmpty(h.name(), h.value());
167 			if (paramName.equals(hName)) {
168 				classLevelHeader = h;
169 				break;
170 			}
171 		}
172 
173 		if (classLevelHeader == null)
174 			return null;
175 
176 		// Get parameter-level @Header
177 		Header paramHeader = pi.getAnnotation(Header.class);
178 		if (paramHeader == null)
179 			paramHeader = pi.getParameterType().getAnnotation(Header.class);
180 
181 		if (paramHeader == null) {
182 			// No parameter-level @Header, use class-level as-is
183 			return classLevelHeader;
184 		}
185 
186 		// Merge the two annotations: parameter-level takes precedence
187 		return mergeAnnotations(classLevelHeader, paramHeader);
188 	}
189 
190 	/**
191 	 * Merges two @Header annotations, with param-level taking precedence over class-level.
192 	 *
193 	 * @param classLevel The class-level default.
194 	 * @param paramLevel The parameter-level override.
195 	 * @return Merged annotation.
196 	 */
197 	private static Header mergeAnnotations(Header classLevel, Header paramLevel) {
198 		return HeaderAnnotation.create()
199 			.name(firstNonEmpty(paramLevel.name(), paramLevel.value(), classLevel.name(), classLevel.value()))
200 			.value(firstNonEmpty(paramLevel.value(), paramLevel.name(), classLevel.value(), classLevel.name()))
201 			.def(firstNonEmpty(paramLevel.def(), classLevel.def()))
202 			.description(paramLevel.description().length > 0 ? paramLevel.description() : classLevel.description())
203 			.parser(paramLevel.parser() != HttpPartParser.Void.class ? paramLevel.parser() : classLevel.parser())
204 			.serializer(paramLevel.serializer() != HttpPartSerializer.Void.class ? paramLevel.serializer() : classLevel.serializer())
205 			.schema(mergeSchemas(classLevel.schema(), paramLevel.schema()))
206 			.build();
207 	}
208 
209 	/**
210 	 * Merges two @Schema annotations, with param-level taking precedence.
211 	 *
212 	 * @param classLevel The class-level default.
213 	 * @param paramLevel The parameter-level override.
214 	 * @return Merged annotation.
215 	 */
216 	private static Schema mergeSchemas(Schema classLevel, Schema paramLevel) {
217 		// If parameter has a non-default schema, use it; otherwise use class-level
218 		if (!SchemaAnnotation.empty(paramLevel))
219 			return paramLevel;
220 		return classLevel;
221 	}
222 
223 	@SuppressWarnings({ "rawtypes", "unchecked" })
224 	@Override /* RestOpArg */
225 	public Object resolve(RestOpSession opSession) throws Exception {
226 		RestRequest req = opSession.getRequest();
227 		HttpPartParserSession ps = partParser == null ? req.getPartParserSession() : partParser.getPartSession();
228 		RequestHeaders rh = req.getHeaders();
229 		BeanSession bs = req.getBeanSession();
230 		ClassMeta<?> cm = bs.getClassMeta(type.innerType());
231 
232 		if (multi) {
233 			Collection c = cm.isArray() ? list() : (Collection)(cm.canCreateNewInstance() ? cm.newInstance() : new JsonList());
234 			rh.stream(name).map(x -> x.parser(ps).schema(schema).as(cm.getElementType()).orElse(null)).forEach(x -> c.add(x));
235 			return cm.isArray() ? ArrayUtils.toArray(c, cm.getElementType().getInnerClass()) : c;
236 		}
237 
238 		if (cm.isMapOrBean() && isOneOf(name, "*", "")) {
239 			JsonMap m = new JsonMap();
240 			rh.forEach(x -> m.put(x.getName(), x.parser(ps).schema(schema == null ? null : schema.getProperty(x.getName())).as(cm.getValueType()).orElse(null)));
241 			return req.getBeanSession().convertToType(m, cm);
242 		}
243 
244 		return rh.getLast(name).parser(ps).schema(schema).def(def).as(type.innerType()).orElse(null);
245 	}
