13  package org.apache.juneau.rest.arg;
14  
15  import java.lang.reflect.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.http.annotation.*;
19  import org.apache.juneau.httppart.bean.*;
20  import org.apache.juneau.reflect.*;
21  import org.apache.juneau.rest.*;
22  import org.apache.juneau.rest.annotation.*;
23  
24  /**
25   * Resolves method parameters and parameter types annotated with {@link Response} on {@link RestOp}-annotated Java methods.
26   *
27   * <p>
28   * The parameter value must be of type {@link Value} that accepts a value that is then set via:
29   * <p class='bjava'>
30   * 	<jv>opSession</jv>
31   * 		.{@link RestOpSession#getResponse() getResponse}()
32   * 		.{@link RestResponse#setContent(Object) setOutput}(<jv>value</jv>);
33   * </p>
34   *
35   * <h5 class='section'>See Also:</h5><ul>
36   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/ResponseBeans">@Response Beans</a>
37   * </ul>
38   */
39  public class ResponseBeanArg implements RestOpArg {
40  	final ResponseBeanMeta meta;
41  	private final Type type;
42  
43  	/**
44  	 * Static creator.
45  	 *
46  	 * @param paramInfo The Java method parameter being resolved.
47  	 * @param annotations The annotations to apply to any new part parsers.
48  	 * @return A new {@link ResponseBeanArg}, or <jk>null</jk> if the parameter is not annotated with {@link Response}.
49  	 */
50  	public static ResponseBeanArg create(ParamInfo paramInfo, AnnotationWorkList annotations) {
51  		if (paramInfo.hasAnnotation(Response.class) || paramInfo.getParameterType().hasAnnotation(Response.class))
52  			return new ResponseBeanArg(paramInfo, annotations);
53  		return null;
54  	}
55  
56  	/**
57  	 * Constructor.
58  	 *
59  	 * @param paramInfo The Java method parameter being resolved.
60  	 * @param annotations The annotations to apply to any new part parsers.
61  	 */
62  	protected ResponseBeanArg(ParamInfo paramInfo, AnnotationWorkList annotations) {
63  		this.type = paramInfo.getParameterType().innerType();
64  		this.meta = ResponseBeanMeta.create(paramInfo, annotations);
65  		Class<?> c = type instanceof Class ? (Class<?>)type : type instanceof ParameterizedType ? (Class<?>)((ParameterizedType)type).getRawType() : null;
66  		if (c != Value.class)
67  			throw new ArgException(paramInfo, "Type must be Value<?> on parameter annotated with @Response annotation");
68  	}
69  
70  	@SuppressWarnings({ "unchecked", "rawtypes" })
71  	@Override /* RestOpArg */
72  	public Object resolve(final RestOpSession opSession) throws Exception {
73  		Value<Object> v = new Value();
74  		v.listener(o -> {
75          	RestRequest req = opSession.getRequest();
76          	RestResponse res = opSession.getResponse();
77          	ResponseBeanMeta meta = req.getOpContext().getResponseBeanMeta(o);
78          	if (meta == null)
79          		meta = ResponseBeanArg.this.meta;
80          	res.setResponseBeanMeta(meta);
81          	res.setContent(o);
82          });
83  		return v;
84  	}
85  }
