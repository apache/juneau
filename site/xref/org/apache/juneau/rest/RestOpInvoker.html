17  package org.apache.juneau.rest;
18  
19  import java.lang.reflect.*;
20  
21  import org.apache.juneau.http.response.*;
22  import org.apache.juneau.reflect.*;
23  import org.apache.juneau.rest.arg.*;
24  import org.apache.juneau.rest.stats.*;
25  
26  /**
27   * A specialized invoker for methods that are called during a servlet request.
28   *
29   * <h5 class='section'>See Also:</h5><ul>
30   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestContext">RestContext</a>
31   * </ul>
32   */
33  public class RestOpInvoker extends MethodInvoker {
34  
35  	private final RestOpArg[] opArgs;
36  
37  	/**
38  	 * Constructor.
39  	 *
40  	 * @param m The method being wrapped.
41  	 * @param opArgs The parameter resolvers.
42  	 * @param stats The instrumentor.
43  	 */
44  	public RestOpInvoker(Method m, RestOpArg[] opArgs, MethodExecStats stats) {
45  		super(m, stats);
46  		this.opArgs = opArgs;
47  	}
48  
49  	/**
50  	 * Invokes this method from the specified {@link RestSession}.
51  	 *
52  	 * @param opSession The REST call.
53  	 * @throws Throwable If an error occurred during either parameter resolution or method invocation.
54  	 */
55  	public void invoke(RestOpSession opSession) throws Throwable {
56  		Object[] args = new Object[opArgs.length];
57  		for (int i = 0; i < opArgs.length; i++) {
58  			ParamInfo pi = inner().getParam(i);
59  			try {
60  				args[i] = opArgs[i].resolve(opSession);
61  			} catch (BasicHttpException e) {
62  				throw e;
63  			} catch (Exception e) {
64  				throw new BadRequest(e, "Could not resolve parameter {0} of type ''{1}'' on method ''{2}''.", i, pi.getParameterType(), getFullName());
65  			}
66  		}
67  		try {
68  			RestSession session = opSession.getRestSession();
69  			RestRequest req = opSession.getRequest();
70  			RestResponse res = opSession.getResponse();
71  
72  			Object output = super.invoke(session.getResource(), args);
73  
74  			// Handle manual call to req.setDebug().
75  			Boolean debug = req.getAttribute("Debug").as(Boolean.class).orElse(null);
76  			if (debug == Boolean.TRUE) {
77  				session.debug(true);
78  			} else if (debug == Boolean.FALSE) {
79  				session.debug(false);
80  			}
81  
82  			if (! inner().hasReturnType(Void.TYPE))
83  				if (output != null || ! res.getOutputStreamCalled())
84  					res.setContent(output);
85  
86  		} catch (IllegalAccessException | IllegalArgumentException e) {
87  			throw new InternalServerError(e, "Error occurred invoking method ''{0}''.", inner().getFullName());
88  		} catch (InvocationTargetException e) {
89  			RestResponse res = opSession.getResponse();
90  			Throwable e2 = e.getTargetException();
91  			res.setStatus(500);  // May be overridden later.
92  			res.setContent(opSession.getRestContext().convertThrowable(e2));
93  		}
94  	}
95  }
