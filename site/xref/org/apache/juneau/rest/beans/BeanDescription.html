13  package org.apache.juneau.rest.beans;
14  
15  import org.apache.juneau.*;
16  import org.apache.juneau.annotation.*;
17  
18  /**
19   * Simple serializable bean description.
20   *
21   * <p>
22   * Given a particular class type, this serializes the class into the fully-qualified class name and the properties
23   * associated with the class.
24   *
25   * <p>
26   * Useful for rendering simple information about a bean during REST OPTIONS requests.
27   *
28   * <h5 class='section'>See Also:</h5><ul>
29   * 	<li class='link'><a class="doclink" href="../../../../../index.html#jrs.UtilityBeans">Utility Beans</a>
30  
31   * </ul>
32   */
33  @Bean(properties="type,properties")
34  public final class BeanDescription {
35  
36  	//-----------------------------------------------------------------------------------------------------------------
37  	// Static
38  	//-----------------------------------------------------------------------------------------------------------------
39  
40  	/**
41  	 * Static creator.
42  	 *
43  	 * @param c The bean being described.
44  	 * @return A new bean description.
45  	 */
46  	public static BeanDescription of(Class<?> c) {
47  		return new BeanDescription(c);
48  	}
49  
50  	//-----------------------------------------------------------------------------------------------------------------
51  	// Instance
52  	//-----------------------------------------------------------------------------------------------------------------
53  
54  	/** The bean class type. */
55  	public String type;
56  
57  	/** The bean properties. */
58  	public BeanPropertyDescription[] properties;
59  
60  	/**
61  	 * Constructor
62  	 *
63  	 * @param c The bean class type.
64  	 */
65  	public BeanDescription(Class<?> c) {
66  		type = c.getName();
67  		BeanMeta<?> bm = BeanContext.DEFAULT.getBeanMeta(c);
68  		if (bm == null)
69  			throw new BasicRuntimeException("Class ''{0}'' is not a valid bean.", c);
70  		properties = new BeanPropertyDescription[bm.getPropertyMetas().size()];
71  		int i = 0;
72  		for (BeanPropertyMeta pm : bm.getPropertyMetas())
73  			properties[i++] = new BeanPropertyDescription(pm.getName(), pm.getClassMeta());
74  	}
75  
76  	/**
77  	 * Information about a bean property.
78  	 */
79  	public static class BeanPropertyDescription {
80  
81  		/** The bean property name. */
82  		public String name;
83  
84  		/** The bean property filtered class type. */
85  		public String type;
86  
87  		/**
88  		 * Constructor.
89  		 *
90  		 * @param name The bean property name.
91  		 * @param type The bean property class type.
92  		 */
93  		public BeanPropertyDescription(String name, ClassMeta<?> type) {
94  			this.name = name;
95  			this.type = type.getSerializedClassMeta(null).toString();
96  		}
97  	}
98  }
