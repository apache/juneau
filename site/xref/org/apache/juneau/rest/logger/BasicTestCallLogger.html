25  /**
26   * Default implementation of a {@link CallLogger} that only logs REST call errors unless no-log is enabled on the request.
27   *
28   * <p>
29   * Useful for REST tests where you know that a particular call is going to produce an error response and you don't want that
30   * response to be logged producing noisy test output.
31   *
32   * <p>
33   * Requests can be tagged as no-log (meaning don't log if there's an error) in any of the following ways:
34   * <ul>
35   * 	<li>A <js>"No-Trace: true"</js> header.
36   * 	<li>A <js>"noTrace=true"</js> query parameter.
37   * 	<li>A <js>"NoTrace"</js> request attribute with a string value of <js>"true"</js>.
38   * </ul>
39   *
40   * <h5 class='section'>Configured Settings:</h5>
41   * <ul>
42   * 	<li>Logs to the {@link RestContext#getLogger() context logger}.
43   * 	<li>Only calls with status code &gt;=400 will be logged.
44   * 	<li>Logs full request and response entity.
45   * </ul>
46   *
47   * <h5 class='section'>See Also:</h5><ul>
48   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/LoggingAndDebugging">Logging / Debugging</a>
49   * </ul>
50   */
51  public class BasicTestCallLogger extends CallLogger {
52  
53  	/**
54  	 * Constructor.
55  	 *
56  	 * @param beanStore The bean store containing injectable beans for this logger.
57  	 */
58  	public BasicTestCallLogger(BeanStore beanStore) {
59  		super(beanStore);
60  	}
61  
62  	@Override
63  	protected Builder init(BeanStore beanStore) {
64  		return super.init(beanStore)
65  			.normalRules(  // Rules when debugging is not enabled.
66  				CallLoggerRule.create(beanStore)  // Log 500+ errors with status-line and header information.
67  					.statusFilter(x -> x >= 400)
68  					.level(SEVERE)
69  					.requestDetail(HEADER)
70  					.responseDetail(HEADER)
71  					.enabled(CONDITIONAL)
72  					.enabledPredicate(x -> ! isNoTrace(x))  // Only log if it's not a no-trace request.
73  					.logStackTrace()
74  					.build(),
75  				CallLoggerRule.create(beanStore)  // Log 400-500 errors with just status-line information.
76  					.statusFilter(x -> x >= 400)
77  					.level(WARNING)
78  					.requestDetail(STATUS_LINE)
79  					.responseDetail(STATUS_LINE)
80  					.enabled(CONDITIONAL)
81  					.enabledPredicate(x -> ! isNoTrace(x))  // Only log if it's not a no-trace request.
82  					.logStackTrace()
83  					.build()
84  			)
85  			.debugRules(  // Rules when debugging is enabled.
86  				CallLoggerRule.create(beanStore)  // Log everything with full details.
87  					.level(SEVERE)
88  					.requestDetail(ENTITY)
89  					.responseDetail(ENTITY)
90  					.logStackTrace()
91  					.build()
92  			)
93  		;
94  	}
95  
96  	private static boolean isNoTrace(HttpServletRequest req) {
97  		Object o = req.getAttribute("NoTrace");
98  		if (o != null)
99  			return "true".equalsIgnoreCase(o.toString());
100 		String s = req.getHeader("No-Trace");
101 		if (s != null)
102 			return "true".equalsIgnoreCase(s);
103 		return emptyIfNull(req.getQueryString()).contains("noTrace=true");
104 	}
105 }
