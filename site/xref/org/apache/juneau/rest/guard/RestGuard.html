13  package org.apache.juneau.rest.guard;
14  
15  import org.apache.juneau.rest.*;
16  import org.apache.juneau.rest.annotation.*;
17  import org.apache.juneau.http.response.*;
18  
19  /**
20   * REST method guard.
21   *
22   * <h5 class='topic'>Description</h5>
23   *
24   * Implements a guard mechanism for REST method calls that allows requests to be rejected before invocation of the REST
25   * method.
26   * For example, guards can be used to ensure that only administrators can call certain methods.
27   *
28   * <p>
29   * Guards are applied to REST methods declaratively through the {@link Rest#guards() @Rest(guards)} or
30   * {@link RestOp#guards() @RestOp(guards)} annotations.
31   *
32   * <p>
33   * If multiple guards are specified, ALL guards must pass in order for the request to proceed.
34   *
35   * <h5 class='topic'>How to implement</h5>
36   *
37   * Typically, guards will be used for permissions checking on the user making the request, but it can also be used for
38   * other purposes like pre-call validation of a request.
39   *
40   * <p>
41   * Implementers should simply throw a {@link BasicHttpException} from the {@link #guard(RestRequest, RestResponse)}
42   * method to abort processing on the current request.
43   *
44   * <p>
45   * Guards must implement a no-args constructor.
46   *
47   * <h5 class='topic'>Example usage:</h5>
48   * <p class='bjava'>
49   * 	<jk>public</jk> MyResource <jk>extends</jk> BasicRestServlet {
50   *
51   * 		<jc>// Delete method with guard that only allows Billy to call it.</jc>
52   * 		<ja>@RestDelete</ja>(guards=BillyGuard.<jk>class</jk>)
53   * 		<jk>public</jk> doDelete(RestRequest <jv>req</jv>, RestResponse <jv>res</jv>) <jk>throws</jk> Exception {...}
54   * 	}
55   * </p>
56   *
57   * <h5 class='topic'>Example implementation:</h5>
58   * <p class='bjava'>
59   * 	<jc>// Define a guard that only lets Billy make a request</jc>
60   * 	<jk>public</jk> BillyGuard <jk>extends</jk> RestGuard {
61   *
62   * 		<ja>@Override</ja>
63   * 		<jk>public boolean</jk> isRequestAllowed(RestRequest <jv>req</jv>) {
64   * 			<jk>return</jk> <jv>req</jv>.getUserPrincipal().getName().contains(<js>"Billy"</js>);
65   * 		}
66   * 	}
67   * </p>
68   *
69   * <h5 class='section'>See Also:</h5><ul>
70   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/Guards">Guards</a>
71   * </ul>
72   */
73  public abstract class RestGuard {
74  
75  	/**
76  	 * Checks the current HTTP request and throws a {@link BasicHttpException} if the guard does not permit the request.
77  	 *
78  	 * <p>
79  	 * By default, throws an <jsf>SC_FORBIDDEN</jsf> exception if {@link #isRequestAllowed(RestRequest)} returns
80  	 * <jk>false</jk>.
81  	 *
82  	 * <p>
83  	 * Subclasses are free to override this method to tailor the behavior of how to handle unauthorized requests.
84  	 *
85  	 * @param req The servlet request.
86  	 * @param res The servlet response.
87  	 * @throws BasicHttpException Thrown to abort processing on current request.
88  	 * @return
89  	 * 	<jk>true</jk> if request can proceed.
90  	 * 	Specify <jk>false</jk> if you're doing something like a redirection to a login page.
91  	 */
92  	public boolean guard(RestRequest req, RestResponse res) throws BasicHttpException {
93  		if (! isRequestAllowed(req))
94  			throw new Forbidden("Access denied by guard");
95  		return true;
96  	}
97  
98  	/**
99  	 * Returns <jk>true</jk> if the specified request can pass through this guard.
100 	 *
101 	 * @param req The servlet request.
102 	 * @return <jk>true</jk> if the specified request can pass through this guard.
103 	 */
104 	public abstract boolean isRequestAllowed(RestRequest req);
105 }
