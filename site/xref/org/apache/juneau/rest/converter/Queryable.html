24  /**
25   * Converter for enabling of search/view/sort/page support on response objects returned by a <c>@RestOp</c>-annotated method.
26   *
27   * <p>
28   * When enabled, objects in a POJO tree can be filtered using the functionality described in the {@link ObjectSearcher},
29   * {@link ObjectViewer}, {@link ObjectSorter}, and {@link ObjectPaginator} classes.
30   *
31   * <p>
32   * The following HTTP request parameters are available for tabular data (e.g. {@code Collections} of {@code Maps},
33   * arrays of beans, etc...):
34   * <ul class='spaced-list'>
35   * 	<li>
36   * 		<c>&amp;s=</c> Search arguments.
37   * 		<br>Comma-delimited list of key/value pairs representing column names and search tokens.
38   * 		<br>Example:
39   * 		<p class='burlenc'>
40   * 	&amp;s=name=Bill*,birthDate&gt;2000
41   * 		</p>
42   * 	<li>
43   * 		<c>&amp;v=</c> Visible columns.
44   * 		<br>Comma-delimited list of column names to display.
45   * 		<br>Example:
46   * 		<p class='burlenc'>
47   * 	&amp;v=name,birthDate
48   * 		</p>
49   * 	<li>
50   * 		<c>&amp;o=</c> Sort commands.
51   * 		<br>Comma-delimited list of columns to sort by.
52   * 		<br>Column names can be suffixed with <js>'+'</js> or <js>'-'</js> to indicate ascending or descending order.
53   * 		<br>The default is ascending order.
54   * 		<br>Example:
55   * 		<p class='burlenc'>
56   * 	&amp;o=name,birthDate-
57   * 		</p>
58   * 	<li>
59   * 		<c>&amp;i=</c> Case-insensitive parameter.
60   * 		<br>Boolean flag for case-insensitive matching on the search parameters.
61   * 	<li>
62   * 		<c>&amp;p=</c> - Position parameter.
63   * 		<br>Only return rows starting at the specified index position (zero-indexed).
64   * 		<br>Default is {@code 0}.
65   * 	<li>
66   * 		<c>&amp;l=</c> Limit parameter.
67   * 		<br>Only return the specified number of rows.
68   * 		<br>Default is {@code 0} (meaning return all rows).
69   * </ul>
70   *
71   * <h5 class='section'>See Also:</h5><ul>
72   * 	<li class='jc'>{@link ObjectSearcher} - Additional information on searching POJO models.
73   * 	<li class='jc'>{@link ObjectViewer} - Additional information on filtering POJO models.
74   * 	<li class='jc'>{@link ObjectSorter} - Additional information on sorting POJO models.
75   * 	<li class='jc'>{@link ObjectPaginator} - Additional information on paginating POJO models.
76   * 	<li class='jm'>{@link org.apache.juneau.rest.RestOpContext.Builder#converters()} - Registering converters with REST resources.
77   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/Converters">Converters</a>
78   * </ul>
79   */
80  public class Queryable implements RestConverter {
81  
82  	/**
83  	 * Swagger parameters for this converter.
84  	 */
85  	public static final String SWAGGER_PARAMS=""
86  		+ "{"
87  			+ "in:'query',"
88  			+ "name:'s',"
89  			+ "description:'"
90  				+ "Search.\n"
91  				+ "Key/value pairs representing column names and search tokens.\n"
92  				+ "\\'*\\' and \\'?\\' can be used as meta-characters in string fields.\n"
93  				+ "\\'>\\', \\'>=\\', \\'<\\', and \\'<=\\' can be used as limits on numeric and date fields.\n"
94  				+ "Date fields can be matched with partial dates (e.g. \\'2018\\' to match any date in the year 2018)."
95  			+ "',"
96  			+ "type:'array',"
97  			+ "collectionFormat:'csv',"
98  			+ "examples:{example:'?s=Bill*,birthDate>2000'}"
99  		+ "},"
100 		+ "{"
101 			+ "in:'query',"
102 			+ "name:'v',"
103 			+ "description:'"
104 				+ "View.\n"
105 				+ "Column names to display."
106 			+ "',"
107 			+ "type:'array',"
108 			+ "collectionFormat:'csv',"
109 			+ "examples:{example:'?v=name,birthDate'}"
110 		+ "},"
111 		+ "{"
112 			+ "in:'query',"
113 			+ "name:'o',"
114 			+ "description:'"
115 				+ "Order by.\n"
116 				+ "Columns to sort by.\n"
117 				+ "Column names can be suffixed with \\'+\\' or \\'-\\' to indicate ascending or descending order.\n"
118 				+ "The default is ascending order."
119 			+ "',"
120 			+ "type:'array',"
121 			+ "collectionFormat:'csv',"
122 			+ "examples:{example:'?o=name,birthDate-'}"
123 		+ "},"
124 		+ "{"
125 			+ "in:'query',"
126 			+ "name:'p',"
127 			+ "description:'"
128 				+ "Position.\n"
129 				+ "Only return rows starting at the specified index position (zero-indexed).\n"
130 				+ "Default is 0"
131 			+ "',"
132 			+ "type:'integer',"
133 			+ "examples:{example:'?p=100'}"
134 		+ "},"
135 		+ "{"
136 			+ "in:'query',"
137 			+ "name:'l',"
138 			+ "description:'"
139 				+ "Limit.\n"
140 				+ "Only return the specified number of rows.\n"
141 				+ "Default is 0 (meaning return all rows)."
142 			+ "',"
143 			+ "type:'integer',"
144 			+ "examples:{example:'?l=100'}"
145 		+ "}"
146 	;
147 
148 	@Override /* Overridden from RestConverter */
149 	public Object convert(RestRequest req, Object o) {
150 		if (o == null)
151 			return null;
152 
153 		Value<Object> v = Value.of(o);
154 		RequestQueryParams params = req.getQueryParams();
155 		BeanSession bs = req.getBeanSession();
156 
157 		params.getSearchArgs().ifPresent(x -> v.set(ObjectSearcher.create().run(bs, v.get(), x)));
158 		params.getSortArgs().ifPresent(x -> v.set(ObjectSorter.create().run(bs, v.get(), x)));
159 		params.getViewArgs().ifPresent(x -> v.set(ObjectViewer.create().run(bs, v.get(), x)));
160 		params.getPageArgs().ifPresent(x -> v.set(ObjectPaginator.create().run(bs, v.get(), x)));
161 		return v.get();
162 	}
163 }
