113 			if (! map.containsKey(httpMethodName))
114 				map.put(httpMethodName, new TreeSet<>());
115 			map.get(httpMethodName).add(value);

117 			return this;

141 	private final Map<String,List<RestOpContext>> map;
142 	private RestOpContext[] list;
143 
144 	/**
145 	 * Constructor.
146 	 *
147 	 * @param builder The builder containing the settings for this object.
148 	 */
149 	public RestOperations(Builder builder) {
150 		Map<String,List<RestOpContext>> m = map();
151 		for (Map.Entry<String,TreeSet<RestOpContext>> e : builder.map.entrySet())
152 			m.put(e.getKey(), listFrom(e.getValue()));
153 		this.map = m;
154 		this.list = Utils.array(builder.set, RestOpContext.class);
155 	}
156 
157 	/**
158 	 * Finds the method that should handle the specified call.
159 	 *
160 	 * @param session The HTTP call.
161 	 * @return The method that should handle the specified call.
162 	 * @throws MethodNotAllowed If no methods implement the requested HTTP method.
163 	 * @throws PreconditionFailed At least one method was found but it didn't match one or more matchers.
164 	 * @throws NotFound HTTP method match was found but matching path was not.
165 	 */
166 	public RestOpContext findOperation(RestSession session) throws MethodNotAllowed, PreconditionFailed, NotFound {
167 		String m = session.getMethod();
168 
169 		int rc = 0;
170 		if (map.containsKey(m)) {
171 			for (RestOpContext oc : map.get(m)) {
172 				int mrc = oc.match(session);
173 				if (mrc == 2)
174 					return oc;
175 				rc = Math.max(rc, mrc);
176 			}
177 		}
178 
179 		if (map.containsKey("*")) {
180 			for (RestOpContext oc : map.get("*")) {
181 				int mrc = oc.match(session);
182 				if (mrc == 2)
183 					return oc;
184 				rc = Math.max(rc, mrc);
185 			}
186 		}
187 
188 		// If no paths matched, see if the path matches any other methods.
189 		// Note that we don't want to match against "/*" patterns such as getOptions().
190 		if (rc == 0) {
191 			for (RestOpContext oc : list) {
192 				if (! oc.getPathPattern().endsWith("/*")) {
193 					int orc = oc.match(session);
194 					if (orc == 2)
195 						throw new MethodNotAllowed();
196 				}
197 			}
198 		}
199 
200 		if (rc == 1)
201 			throw new PreconditionFailed("Method ''{0}'' not found on resource on path ''{1}'' with matching matcher.", m, session.getPathInfo());
202 
203 		throw new NotFound("Java method matching path ''{0}'' not found on resource ''{1}''.", session.getPathInfo(), className(session.getResource()));
204 	}
