19  import static java.util.stream.Collectors.toList;
20  import static org.apache.juneau.common.utils.Utils.*;
21  
22  import java.util.*;
23  
24  import org.apache.juneau.*;
25  import org.apache.juneau.cp.*;

28  /**
29   * A list of {@link RestMatcher} objects.
30   *
31   * <h5 class='section'>See Also:</h5><ul>
32   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestOpAnnotatedMethodBasics">@RestOp-Annotated Method Basics</a>
33   * </ul>
34   */
35  public class RestMatcherList {
36  
37  	//-----------------------------------------------------------------------------------------------------------------
38  	// Static
39  	//-----------------------------------------------------------------------------------------------------------------
40  
41  	/**
42  	 * Static creator.
43  	 *
44  	 * @param beanStore The bean store to use for creating beans.
45  	 * @return A new builder for this object.
46  	 */
47  	public static Builder create(BeanStore beanStore) {
48  		return new Builder(beanStore);
49  	}
50  
51  	//-----------------------------------------------------------------------------------------------------------------
52  	// Builder
53  	//-----------------------------------------------------------------------------------------------------------------
54  
55  	/**
56  	 * Builder class.
57  	 */
58  	@FluentSetters
59  	public static class Builder extends BeanBuilder<RestMatcherList> {
60  
61  		List<BeanCreator<RestMatcher>> entries;
62  
63  		/**
64  		 * Constructor.
65  		 *
66  		 * @param beanStore The bean store to use for creating beans.
67  		 */
68  		protected Builder(BeanStore beanStore) {
69  			super(RestMatcherList.class, beanStore);
70  			entries = list();
71  		}
72  
73  		@Override /* BeanBuilder */
74  		protected RestMatcherList buildDefault() {
75  			return new RestMatcherList(this);
76  		}
77  
78  		//-------------------------------------------------------------------------------------------------------------
79  		// Properties
80  		//-------------------------------------------------------------------------------------------------------------
81  
82  		/**
83  		 * Appends the specified rest matcher classes to the list.
84  		 *
85  		 * @param values The values to add.
86  		 * @return This object.
87  		 */
88  		@SuppressWarnings("unchecked")
89  		public Builder append(Class<? extends RestMatcher>...values) {
90  			for (Class<? extends RestMatcher> v : values)
91  				entries.add(beanStore().createBean(RestMatcher.class).type(v));
92  			return this;
93  		}
94  
95  		/**
96  		 * Appends the specified rest matcher objects to the list.
97  		 *
98  		 * @param values The values to add.
99  		 * @return This object.
100 		 */
101 		public Builder append(RestMatcher...values) {
102 			for (RestMatcher v : values)
103 				entries.add(beanStore().createBean(RestMatcher.class).impl(v));
104 			return this;
105 		}
106 
107 		// <FluentSetters>
108 
109 		@Override /* GENERATED - org.apache.juneau.BeanBuilder */
110 		public Builder impl(Object value) {
111 			super.impl(value);
112 			return this;
113 		}
114 
115 		@Override /* GENERATED - org.apache.juneau.BeanBuilder */
116 		public Builder type(Class<?> value) {
117 			super.type(value);
118 			return this;
119 		}
120 
121 		// </FluentSetters>
122 	}
123 
124 	//-----------------------------------------------------------------------------------------------------------------
125 	// Instance
126 	//-----------------------------------------------------------------------------------------------------------------
127 
128 	private final RestMatcher[] optionalEntries;
129 	private final RestMatcher[] requiredEntries;
130 
131 	/**
132 	 * Constructor.
133 	 *
134 	 * @param builder The builder containing the contents for this list.
135 	 */
136 	protected RestMatcherList(Builder builder) {
137 		List<RestMatcher> l =
138 			builder
139 				.entries
140 				.stream()
141 				.map(BeanCreator::run)
142 				.collect(toList());
143 		optionalEntries = l.stream().filter(x -> ! x.required()).toArray(RestMatcher[]::new);
144 		requiredEntries = l.stream().filter(RestMatcher::required).toArray(RestMatcher[]::new);
145 	}
146 
147 	/**
148 	 * Returns the entries in this list that are specified as optional.
149 	 *
150 	 * @return An unmodifiable list of entries in this list that are specified as optional.
151 	 */
152 	public RestMatcher[] getOptionalEntries() {
153 		return optionalEntries;
154 	}
155 
156 	/**
157 	 * Returns the entries in this list that are specified as required.
158 	 *
159 	 * @return An unmodifiable list of entries in this list that are specified as required.
160 	 */
161 	public RestMatcher[] getRequiredEntries() {
162 		return requiredEntries;
163 	}
164 }
