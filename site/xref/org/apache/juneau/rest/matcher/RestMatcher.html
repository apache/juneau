13  package org.apache.juneau.rest.matcher;
14  
15  import jakarta.servlet.http.*;
16  
17  import org.apache.juneau.rest.annotation.*;
18  
19  /**
20   * Class used for defining method-level matchers using the {@link RestOp#matchers() @RestOp(matchers)} annotation.
21   *
22   * <p>
23   * Matchers are used to allow multiple Java methods to handle requests assigned to the same URL path pattern, but
24   * differing based on some request attribute, such as a specific header value.
25   * For example, matchers can be used to provide two different methods for handling requests from two different client
26   * versions.
27   *
28   * <p>
29   * Java methods with matchers associated with them are always attempted before Java methods without matchers.
30   * This allows a 'default' method to be defined to handle requests where no matchers match.
31   *
32   * <p>
33   * When multiple matchers are specified on a method, only one matcher is required to match.
34   * This is opposite from the {@link RestOp#guards() @RestOp(guards)} annotation, where all guards are required to match in order to
35   * execute the method.
36   *
37   * <h5 class='section'>Example:</h5>
38   * <p class='bjava'>
39   * 	<jk>public class</jk> MyResource <jk>extends</jk> BasicRestServlet {
40   *
41   * 		<ja>@RestGet</ja>(path=<js>"/foo"</js>, matchers=IsDNT.<jk>class</jk>)
42   * 		<jk>public</jk> Object doGetWithDNT() {
43   * 			<jc>// Handle request with Do-Not-Track specified</jc>
44   * 		}
45   *
46   * 		<ja>@RestGet</ja>(<js>"/foo"</js>)
47   * 		<jk>public</jk> Object doGetWithoutDNT() {
48   * 			<jc>// Handle request without Do-Not-Track specified</jc>
49   * 		}
50   * 	}
51   *
52   * 	<jk>public class</jk> IsDNT <jk>extends</jk> RestMatcher {
53   * 		<ja>@Override</ja>
54   * 		<jk>public boolean</jk> matches(HttpServletRequest <jv>req</jv>) {
55   * 			<jk>return</jk> <js>"1"</js>.equals(<jv>req</jv>.getHeader(<js>"DNT"</js>));
56   * 		}
57   * 	}
58   * </p>
59   *
60   * <p>
61   * Instances must provide one of the following public constructors:
62   * <ul>
63   * 	<li>No-args.
64   * 	<li>The following args: <c>Object resource, Method javaMethod</c>.
65   * 		<br>This gives access to the servlet/resource and Java method it's applied to.
66   * </ul>
67   *
68   * <h5 class='section'>See Also:</h5><ul>
69   * 	<li class='link'><a class="doclink" href="../../../../../index.html#jrs.RestOpAnnotatedMethods">@RestOp-Annotated Methods</a>
70   * </ul>
71   */
72  public abstract class RestMatcher {
73  
74  	/**
75  	 * Returns <jk>true</jk> if the specified request matches this matcher.
76  	 *
77  	 * @param req The servlet request.
78  	 * @return <jk>true</jk> if the specified request matches this matcher.
79  	 */
80  	public abstract boolean matches(HttpServletRequest req);
81  
82  	/**
83  	 * Returns <jk>true</jk> if this matcher is required to match in order for the method to be invoked.
84  	 *
85  	 * <p>
86  	 * If <jk>false</jk>, then only one of the matchers must match.
87  	 *
88  	 * @return <jk>true</jk> if this matcher is required to match in order for the method to be invoked.
89  	 */
90  	public boolean required() {
91  		return false;
92  	}
93  }
