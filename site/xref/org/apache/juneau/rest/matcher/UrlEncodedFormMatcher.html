13  package org.apache.juneau.rest.matcher;
14  
15  import jakarta.servlet.http.*;
16  
17  /**
18   * Predefined matcher for matching requests with content type <js>"application/x-www-form-urlencoded"</js>.
19   *
20   * <h5 class='section'>See Also:</h5><ul>
21   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestOpAnnotatedMethodBasics">@RestOp-Annotated Method Basics</a>
22   * </ul>
23   */
24  public class UrlEncodedFormMatcher extends RestMatcher {
25  
26  	@Override /* RestMatcher */
27  	public boolean matches(HttpServletRequest req) {
28  		String contentType = req.getContentType();
29  		return contentType != null && contentType.equals("application/x-www-form-urlencoded");
30  	}
31  }
