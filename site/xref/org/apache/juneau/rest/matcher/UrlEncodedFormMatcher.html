17  package org.apache.juneau.rest.matcher;
18  
19  import jakarta.servlet.http.*;
20  
21  /**
22   * Predefined matcher for matching requests with content type <js>"application/x-www-form-urlencoded"</js>.
23   *
24   * <h5 class='section'>See Also:</h5><ul>
25   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestOpAnnotatedMethodBasics">@RestOp-Annotated Method Basics</a>
26   * </ul>
27   */
28  public class UrlEncodedFormMatcher extends RestMatcher {
29  
30  	@Override /* RestMatcher */
31  	public boolean matches(HttpServletRequest req) {
32  		String contentType = req.getContentType();
33  		return contentType != null && contentType.equals("application/x-www-form-urlencoded");
34  	}
35  }
