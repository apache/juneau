13  package org.apache.juneau.rest.client;
14  
15  import java.io.*;
16  
17  import org.apache.http.*;
18  import org.apache.http.client.*;
19  import org.apache.http.client.methods.*;
20  import org.apache.http.protocol.*;
21  
22  /**
23   * Interface that allows you to override the handling of HTTP requests.
24   *
25   * <p>
26   * Providing this implementation is the equivalent to overriding the {@link RestClient#execute(HttpHost,HttpRequest,HttpContext)}.
27   * <br>This can also be accomplished by providing your own {@link RestClient.Builder#connectionManager(org.apache.http.conn.HttpClientConnectionManager) connection manager}
28   * or subclassing {@link RestClient}, but this provides a simpler way of handling the requests yourself.
29   *
30   * <p>
31   * The constructor on the implementation class can optionally take in any of the following parameters:
32   * <ul>
33   * 	<li>{@link RestClient} - The client using this handler.
34   * </ul>
35   *
36   * <p>
37   * The {@link BasicRestCallHandler} shows an example of a simple pass-through handler.  Note that you must handle
38   * the case where {@link HttpHost} is null.
39   *
40   * <p class='bjava'>
41   * 	<jk>public class</jk> BasicRestCallHandler <jk>implements</jk> RestCallHandler {
42   *
43   * 		<jk>private final</jk> RestClient <jf>client</jf>;
44   *
45   * 		<jk>public</jk> BasicRestCallHandler(RestClient <jv>client</jv>) {
46   * 			<jk>this</jk>.<jf>client</jf> = <jv>client</jv>;
47   * 		}
48   *
49   * 		<ja>@Override</ja>
50   * 		<jk>public</jk> HttpResponse run(HttpHost <jv>target</jv>, HttpRequest <jv>request</jv>, HttpContext <jv>context</jv>) <jk>throws</jk> IOException {
51   * 			<jk>if</jk> (<jv>target</jv> == <jk>null</jk>)
52   * 				<jk>return</jk> <jf>client</jf>.execute((HttpUriRequest)<jv>request</jv>, <jv>context</jv>);
53   * 			<jk>return</jk> <jf>client</jf>.execute(<jv>target</jv>, <jv>request</jv>, <jv>context</jv>);
54   * 		}
55   * 	}
56   * </p>
57   *
58   * <h5 class='section'>See Also:</h5><ul>
59   * 	<li class='jm'>{@link RestClient.Builder#callHandler()}
60   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestClientBasics">juneau-rest-client</a>
61   * </ul>
62   */
63  public interface RestCallHandler {
64  
65  	//-----------------------------------------------------------------------------------------------------------------
66  	// Instance
67  	//-----------------------------------------------------------------------------------------------------------------
68  
69  	/**
70  	 * Execute the specified request.
71  	 *
72  	 * <p>
73  	 * Subclasses can override this method to provide specialized handling.
74  	 *
75  	 * @param target The target host for the request.
76  	 * 	<br>Implementations may accept <jk>null</jk> if they can still determine a route, for example to a default
77  	 * 		target or by inspecting the request.
78  	 * @param request The request to execute.  Must be an instance of {@link HttpUriRequest} if the target is <jk>null</jk>.
79  	 * @param context The context to use for the execution, or <jk>null</jk> to use the default context.
80  	 * @return
81  	 * 	The response to the request.
82  	 * 	<br>This is always a final response, never an intermediate response with an 1xx status code.
83  	 * 	<br>Whether redirects or authentication challenges will be returned or handled automatically depends on the
84  	 * 		implementation and configuration of this client.
85  	 * @throws IOException In case of a problem or the connection was aborted.
86  	 * @throws ClientProtocolException In case of an http protocol error.
87  	 */
88  	HttpResponse run(HttpHost target, HttpRequest request, HttpContext context) throws ClientProtocolException, IOException;
89  }
