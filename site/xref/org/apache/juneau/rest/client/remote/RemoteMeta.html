34  /**
35   * Contains the meta-data about a REST proxy class.
36   *
37   * <p>
38   * Captures the information in {@link org.apache.juneau.http.remote.Remote @Remote} and {@link org.apache.juneau.http.remote.RemoteOp @RemoteOp} annotations for
39   * caching and reuse.
40   *
41   * <h5 class='section'>See Also:</h5><ul>
42   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestProxyBasics">REST Proxy Basics</a>
43   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestClientBasics">juneau-rest-client Basics</a>
44   * </ul>
45   */
46  public class RemoteMeta {
47  
48  	private static String resolve(String s) {
49  		return VarResolver.DEFAULT.resolve(s);
50  	}
51  
52  	private final Map<Method,RemoteOperationMeta> operations;
53  
54  	private final HeaderList headers;
55  
56  	/**
57  	 * Constructor.
58  	 *
59  	 * @param c The interface class annotated with a {@link org.apache.juneau.http.remote.Remote @Remote} annotation (optional).
60  	 */
61  	public RemoteMeta(Class<?> c) {
62  		var path = "";
63  
64  		var ci = ClassInfo.of(c);
65  		var remotes = rstream(ci.getAnnotations(Remote.class).toList()).map(AnnotationInfo::inner).toList();
66  
67  		var versionHeader = "Client-Version";
68  		var clientVersion = (String)null;
69  		var headers = HeaderList.create().resolving();
70  
71  		for (var r : remotes) {
72  			if (isNotEmpty(r.path()))
73  				path = trimSlashes(resolve(r.path()));
74  			for (var h : r.headers())
75  				headers.append(stringHeader(resolve(h)));
76  			if (isNotEmpty(r.version()))
77  				clientVersion = resolve(r.version());
78  			if (isNotEmpty(r.versionHeader()))
79  				versionHeader = resolve(r.versionHeader());
80  			if (isNotVoid(r.headerList())) {
81  				try {
82  					headers.append(r.headerList().getDeclaredConstructor().newInstance().getAll());
83  				} catch (Exception e) {
84  					throw rex(e, "Could not instantiate HeaderSupplier class");
85  				}
86  			}
87  		}
88  
89  		if (nn(clientVersion))
90  			headers.append(stringHeader(versionHeader, clientVersion));
91  
92  		Map<Method,RemoteOperationMeta> operations = map();
93  		var path2 = path;
94  		ci.getPublicMethods().stream().forEach(x -> operations.put(x.inner(), new RemoteOperationMeta(path2, x.inner(), "GET")));
95  
96  		this.operations = u(operations);
97  		this.headers = headers;
98  	}
99  
100 	/**
101 	 * Returns the headers to set on all requests.
102 	 *
103 	 * @return The headers to set on all requests.
104 	 */
105 	public HeaderList getHeaders() { return headers; }
106 
107 	/**
108 	 * Returns the metadata about the specified operation on this resource proxy.
109 	 *
110 	 * @param m The method to look up.
111 	 * @return Metadata about the method or <jk>null</jk> if no metadata was found.
112 	 */
113 	public RemoteOperationMeta getOperationMeta(Method m) {
114 		return operations.get(m);
115 	}
116 }
