32  /**
33   * Represents the metadata about the returned object of a method on a remote proxy interface.
34   *
35   * <h5 class='section'>See Also:</h5><ul>
36   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestProxyBasics">REST Proxy Basics</a>
37   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestClientBasics">juneau-rest-client Basics</a>
38   * </ul>
39   */
40  public class RemoteOperationReturn {
41  
42  	private final Type returnType;
43  	private final RemoteReturn returnValue;
44  	private final ResponseBeanMeta meta;
45  	private boolean isFuture, isCompletableFuture;
46  
47  	RemoteOperationReturn(MethodInfo m) {
48  		ClassInfo rt = m.getReturnType();
49  
50  		AnnotationList al = m.getAnnotationList(REMOTE_OP_GROUP);
51  		if (al.isEmpty())
52  			al = m.getReturnType().unwrap(Value.class, Optional.class).getAnnotationList(REMOTE_OP_GROUP);
53  
54  		RemoteReturn rv = null;
55  
56  		if (rt.is(Future.class)) {
57  			isFuture = true;
58  			rt = ClassInfo.of(((ParameterizedType)rt.innerType()).getActualTypeArguments()[0]);
59  		} else if (rt.is(CompletableFuture.class)) {
60  			isCompletableFuture = true;
61  			rt = ClassInfo.of(((ParameterizedType)rt.innerType()).getActualTypeArguments()[0]);
62  		}
63  
64  		if (rt.is(void.class) || rt.is(Void.class)) {
65  			rv = RemoteReturn.NONE;
66  		} else {
67  			Value<RemoteReturn> v = Value.of(RemoteReturn.BODY);
68  			al.forEachValue(RemoteReturn.class, "returns", x -> true, x -> v.set(x));
69  			rv = v.get();
70  		}
71  
72  		if (rt.hasAnnotation(Response.class) && rt.isInterface()) {
73  			this.meta = ResponseBeanMeta.create(m, AnnotationWorkList.create());
74  			rv = RemoteReturn.BEAN;
75  		} else {
76  			this.meta = null;
77  		}
78  
79  		this.returnType = rt.innerType();
80  		this.returnValue = rv;
81  	}
82  
83  	/**
84  	 * Returns schema information about the HTTP part.
85  	 *
86  	 * @return Schema information about the HTTP part, or <jk>null</jk> if not found.
87  	 */
88  	public ResponseBeanMeta getResponseBeanMeta() { return meta; }
89  
90  	/**
91  	 * Returns the class type of the method return.
92  	 *
93  	 * @return The class type of the method return.
94  	 */
95  	public Type getReturnType() { return returnType; }
96  
97  	/**
98  	 * Specifies whether the return value is the body of the request or the HTTP status.
99  	 *
100 	 * @return The type of value returned.
101 	 */
102 	public RemoteReturn getReturnValue() { return returnValue; }
103 
104 	/**
105 	 * Returns <jk>true</jk> if the return is wrapped in a {@link CompletableFuture}.
106 	 *
107 	 * @return <jk>true</jk> if the return is wrapped in a {@link CompletableFuture}.
108 	 */
109 	public boolean isCompletableFuture() { return isCompletableFuture; }
110 
111 	/**
112 	 * Returns <jk>true</jk> if the return is wrapped in a {@link Future}.
113 	 *
114 	 * @return <jk>true</jk> if the return is wrapped in a {@link Future}.
115 	 */
116 	public boolean isFuture() { return isFuture; }
117 }
