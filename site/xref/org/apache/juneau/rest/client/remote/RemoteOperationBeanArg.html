13  package org.apache.juneau.rest.client.remote;
14  
15  import org.apache.juneau.http.annotation.*;
16  import org.apache.juneau.httppart.bean.*;
17  
18  /**
19   * Represents the metadata about an {@link Request}-annotated argument of a method on a REST proxy class.
20   *
21   * <h5 class='section'>See Also:</h5><ul>
22   * 	<li class='link'><a class="doclink" href="../../../../../../index.html#jrc.Proxies">REST Proxies</a>
23   * 	<li class='link'><a class="doclink" href="../../../../../../index.html#juneau-rest-client">juneau-rest-client</a>
24   * </ul>
25   */
26  public final class RemoteOperationBeanArg {
27  
28  	private final int index;
29  	private final RequestBeanMeta meta;
30  
31  	RemoteOperationBeanArg(int index, RequestBeanMeta meta) {
32  		this.index = index;
33  		this.meta = meta;
34  	}
35  
36  	/**
37  	 * Returns the index of the parameter in the method that is a request bean.
38  	 *
39  	 * @return The index of the parameter in the method that is a request bean.
40  	 */
41  	public int getIndex() {
42  		return index;
43  	}
44  
45  	/**
46  	 * Returns metadata on the request bean.
47  	 *
48  	 * @return Metadata about the bean.
49  	 */
50  	public RequestBeanMeta getMeta() {
51  		return meta;
52  	}
53  }
