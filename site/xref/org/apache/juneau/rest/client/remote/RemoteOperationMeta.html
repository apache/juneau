38  /**
39   * Contains the meta-data about a Java method on a REST proxy class.
40   *
41   * <p>
42   * Captures the information in {@link RemoteOp @RemoteOp} annotations for caching and reuse.
43   *
44   * <h5 class='section'>See Also:</h5><ul>
45   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestProxyBasics">REST Proxy Basics</a>
46   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestClientBasics">juneau-rest-client Basics</a>
47   * </ul>
48   */
49  public class RemoteOperationMeta {
50  
51  	private static class Builder {
52  		String httpMethod, fullPath, path;
53  		List<RemoteOperationArg> pathArgs = new LinkedList<>(), queryArgs = new LinkedList<>(), headerArgs = new LinkedList<>(), formDataArgs = new LinkedList<>();
54  		List<RemoteOperationBeanArg> requestArgs = new LinkedList<>();
55  		RemoteOperationArg bodyArg;
56  		RemoteOperationReturn methodReturn;
57  		Map<String,String> pathDefaults = new LinkedHashMap<>(), queryDefaults = new LinkedHashMap<>(), headerDefaults = new LinkedHashMap<>(), formDataDefaults = new LinkedHashMap<>();
58  		String contentDefault = null;
59  
60  		Builder(String parentPath, Method m, String defaultMethod) {
61  
62  			MethodInfo mi = MethodInfo.of(m);
63  
64  			AnnotationList al = mi.getAnnotationList(REMOTE_OP_GROUP);
65  			if (al.isEmpty())
66  				al = mi.getReturnType().unwrap(Value.class, Optional.class).getAnnotationList(REMOTE_OP_GROUP);
67  
68  			Value<String> _httpMethod = Value.empty(), _path = Value.empty();
69  			al.stream().map(x -> x.getName().substring(6).toUpperCase()).filter(x -> ! x.equals("OP")).forEach(x -> _httpMethod.set(x));
70  			al.forEachValue(String.class, "method", NOT_EMPTY, x -> _httpMethod.set(x.trim().toUpperCase()));
71  			al.forEachValue(String.class, "path", NOT_EMPTY, x -> _path.set(x.trim()));
72  			httpMethod = _httpMethod.orElse("").trim();
73  			path = _path.orElse("").trim();
74  
75  			Value<String> value = Value.empty();
76  			al.forEach(RemoteOp.class, x -> isNotEmpty(x.inner().value().trim()), x -> value.set(x.inner().value().trim()));
77  
78  			if (value.isPresent()) {
79  				String v = value.get();
80  				int i = v.indexOf(' ');
81  				if (i == -1) {
82  					httpMethod = v;
83  				} else {
84  					httpMethod = v.substring(0, i).trim();
85  					path = v.substring(i).trim();
86  				}
87  			} else {
88  				al.forEach(x -> ! x.isType(RemoteOp.class) && isNotEmpty(x.getValue(String.class, "value", NOT_EMPTY).orElse("").trim()),
89  					x -> value.set(x.getValue(String.class, "value", NOT_EMPTY).get().trim()));
90  				if (value.isPresent())
91  					path = value.get();
92  			}
93  
94  			if (path.isEmpty()) {
95  				path = HttpUtils.detectHttpPath(m, StringUtils.nullIfEmpty(httpMethod));
96  			}
97  			if (httpMethod.isEmpty())
98  				httpMethod = HttpUtils.detectHttpMethod(m, true, defaultMethod);
99  
100 			path = trimSlashes(path);
101 
102 			if (! isOneOf(httpMethod, "DELETE", "GET", "POST", "PUT", "OPTIONS", "HEAD", "CONNECT", "TRACE", "PATCH"))
103 				throw new RemoteMetadataException(m,
104 					"Invalid value specified for @RemoteOp(httpMethod) annotation: '" + httpMethod + "'.  Valid values are [DELETE,GET,POST,PUT,OPTIONS,HEAD,CONNECT,TRACE,PATCH].");
105 
106 			methodReturn = new RemoteOperationReturn(mi);
107 
108 			fullPath = path.indexOf("://") != -1 ? path : (parentPath.isEmpty() ? urlEncodePath(path) : (trimSlashes(parentPath) + '/' + urlEncodePath(path)));
109 
110 			mi.getParams().forEach(x -> {
111 				RemoteOperationArg rma = RemoteOperationArg.create(x);
112 				if (rma != null) {
113 					HttpPartType pt = rma.getPartType();
114 					if (pt == HEADER)
115 						headerArgs.add(rma);
116 					else if (pt == QUERY)
117 						queryArgs.add(rma);
118 					else if (pt == FORMDATA)
119 						formDataArgs.add(rma);
120 					else if (pt == PATH)
121 						pathArgs.add(rma);
122 					else
123 						bodyArg = rma;
124 				}
125 				RequestBeanMeta rmba = RequestBeanMeta.create(x, AnnotationWorkList.create());
126 				if (rmba != null) {
127 					requestArgs.add(new RemoteOperationBeanArg(x.getIndex(), rmba));
128 				}
129 			});
130 
131 			// Process method-level annotations for defaults (9.2.0)
132 			// Note: We need to handle both individual annotations and repeated annotation arrays
133 			processHeaderDefaults(mi, headerDefaults);
134 			processQueryDefaults(mi, queryDefaults);
135 			processFormDataDefaults(mi, formDataDefaults);
136 			processPathDefaults(mi, pathDefaults);
137 			processContentDefaults(mi);
138 		}
139 
140 		// Helper methods to process method-level annotations with defaults (9.2.0)
141 		// These handle both individual annotations and repeated annotation arrays
142 
143 		private void processContentDefaults(MethodInfo mi) {
144 			mi.getAnnotationList().forEach(Content.class, null, x -> {
145 				Content c = x.inner();
146 				String def = c.def();
147 				if (isNotEmpty(def)) {
148 					contentDefault = def;
149 				}
150 			});
151 			mi.getAnnotationList().forEach(ContentAnnotation.Array.class, null, x -> {
152 				for (Content c : x.inner().value()) {
153 					String def = c.def();
154 					if (isNotEmpty(def)) {
155 						contentDefault = def;
156 					}
157 				}
158 			});
159 		}
160 
161 		private static void processFormDataDefaults(MethodInfo mi, Map<String,String> defaults) {
162 			mi.getAnnotationList().forEach(FormData.class, null, x -> {
163 				FormData fd = x.inner();
164 				String name = firstNonEmpty(fd.name(), fd.value());
165 				String def = fd.def();
166 				if (isNotEmpty(name) && isNotEmpty(def)) {
167 					defaults.put(name, def);
168 				}
169 			});
170 			mi.getAnnotationList().forEach(FormDataAnnotation.Array.class, null, x -> {
171 				for (FormData fd : x.inner().value()) {
172 					String name = firstNonEmpty(fd.name(), fd.value());
173 					String def = fd.def();
174 					if (isNotEmpty(name) && isNotEmpty(def)) {
175 						defaults.put(name, def);
176 					}
177 				}
178 			});
179 		}
180 
181 		private static void processHeaderDefaults(MethodInfo mi, Map<String,String> defaults) {
182 			// Check for individual @Header annotations
183 			mi.getAnnotationList().forEach(Header.class, null, x -> {
184 				Header h = x.inner();
185 				String name = firstNonEmpty(h.name(), h.value());
186 				String def = h.def();
187 				if (isNotEmpty(name) && isNotEmpty(def)) {
188 					defaults.put(name, def);
189 				}
190 			});
191 			// Check for @Header.Array (repeated annotations)
192 			mi.getAnnotationList().forEach(HeaderAnnotation.Array.class, null, x -> {
193 				for (Header h : x.inner().value()) {
194 					String name = firstNonEmpty(h.name(), h.value());
195 					String def = h.def();
196 					if (isNotEmpty(name) && isNotEmpty(def)) {
197 						defaults.put(name, def);
198 					}
199 				}
200 			});
201 		}

247 	private final RemoteOperationBeanArg[] requestArgs;
248 	private final RemoteOperationArg contentArg;
249 	private final RemoteOperationReturn methodReturn;

251 	private final Class<?>[] exceptions;

257 	/**
258 	 * Constructor.
259 	 *
260 	 * @param parentPath The absolute URI of the REST interface backing the interface proxy.
261 	 * @param m The Java method.
262 	 * @param defaultMethod The default HTTP method if not specified through annotation.
263 	 */
264 	public RemoteOperationMeta(final String parentPath, Method m, String defaultMethod) {
265 		Builder b = new Builder(parentPath, m, defaultMethod);
266 		this.httpMethod = b.httpMethod;
267 		this.fullPath = b.fullPath;
268 		this.pathArgs = b.pathArgs.toArray(new RemoteOperationArg[b.pathArgs.size()]);
269 		this.queryArgs = b.queryArgs.toArray(new RemoteOperationArg[b.queryArgs.size()]);
270 		this.formDataArgs = b.formDataArgs.toArray(new RemoteOperationArg[b.formDataArgs.size()]);
271 		this.headerArgs = b.headerArgs.toArray(new RemoteOperationArg[b.headerArgs.size()]);
272 		this.requestArgs = b.requestArgs.toArray(new RemoteOperationBeanArg[b.requestArgs.size()]);
273 		this.contentArg = b.bodyArg;
274 		this.methodReturn = b.methodReturn;
275 		this.exceptions = m.getExceptionTypes();
276 		this.pathDefaults = Collections.unmodifiableMap(b.pathDefaults);
277 		this.queryDefaults = Collections.unmodifiableMap(b.queryDefaults);
278 		this.headerDefaults = Collections.unmodifiableMap(b.headerDefaults);
279 		this.formDataDefaults = Collections.unmodifiableMap(b.formDataDefaults);
280 		this.contentDefault = b.contentDefault;
281 	}
