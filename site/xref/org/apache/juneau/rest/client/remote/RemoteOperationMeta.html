37  /**
38   * Contains the meta-data about a Java method on a REST proxy class.
39   *
40   * <p>
41   * Captures the information in {@link RemoteOp @RemoteOp} annotations for caching and reuse.
42   *
43   * <h5 class='section'>See Also:</h5><ul>
44   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestProxyBasics">REST Proxy Basics</a>
45   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestClientBasics">juneau-rest-client Basics</a>
46   * </ul>
47   */
48  public class RemoteOperationMeta {
49  
50  	private final String httpMethod;
51  	private final String fullPath;
52  	private final RemoteOperationArg[] pathArgs, queryArgs, headerArgs, formDataArgs;
53  	private final RemoteOperationBeanArg[] requestArgs;
54  	private final RemoteOperationArg contentArg;
55  	private final RemoteOperationReturn methodReturn;
56  	private final Class<?>[] exceptions;
57  
58  	/**
59  	 * Constructor.
60  	 *
61  	 * @param parentPath The absolute URI of the REST interface backing the interface proxy.
62  	 * @param m The Java method.
63  	 * @param defaultMethod The default HTTP method if not specified through annotation.
64  	 */
65  	public RemoteOperationMeta(final String parentPath, Method m, String defaultMethod) {
66  		Builder b = new Builder(parentPath, m, defaultMethod);
67  		this.httpMethod = b.httpMethod;
68  		this.fullPath = b.fullPath;
69  		this.pathArgs = b.pathArgs.toArray(new RemoteOperationArg[b.pathArgs.size()]);
70  		this.queryArgs = b.queryArgs.toArray(new RemoteOperationArg[b.queryArgs.size()]);
71  		this.formDataArgs = b.formDataArgs.toArray(new RemoteOperationArg[b.formDataArgs.size()]);
72  		this.headerArgs = b.headerArgs.toArray(new RemoteOperationArg[b.headerArgs.size()]);
73  		this.requestArgs = b.requestArgs.toArray(new RemoteOperationBeanArg[b.requestArgs.size()]);
74  		this.contentArg = b.bodyArg;
75  		this.methodReturn = b.methodReturn;
76  		this.exceptions = m.getExceptionTypes();
77  	}
78  
79  	private static class Builder {
80  		String httpMethod, fullPath, path;
81  		List<RemoteOperationArg>
82  			pathArgs = new LinkedList<>(),
83  			queryArgs = new LinkedList<>(),
84  			headerArgs = new LinkedList<>(),
85  			formDataArgs = new LinkedList<>();
86  		List<RemoteOperationBeanArg>
87  			requestArgs = new LinkedList<>();
88  		RemoteOperationArg bodyArg;
89  		RemoteOperationReturn methodReturn;
90  
91  		Builder(String parentPath, Method m, String defaultMethod) {
92  
93  			MethodInfo mi = MethodInfo.of(m);
94  
95  			AnnotationList al = mi.getAnnotationList(REMOTE_OP_GROUP);
96  			if (al.isEmpty())
97  				al = mi.getReturnType().unwrap(Value.class,Optional.class).getAnnotationList(REMOTE_OP_GROUP);
98  
99  			Value<String> _httpMethod = Value.empty(), _path = Value.empty();
100 			al.stream().map(x -> x.getName().substring(6).toUpperCase()).filter(x -> ! x.equals("OP")).forEach(x -> _httpMethod.set(x));
101 			al.forEachValue(String.class, "method", NOT_EMPTY, x -> _httpMethod.set(x.trim().toUpperCase()));
102 			al.forEachValue(String.class, "path", NOT_EMPTY, x-> _path.set(x.trim()));
103 			httpMethod = _httpMethod.orElse("").trim();
104 			path = _path.orElse("").trim();
105 
106 			Value<String> value = Value.empty();
107 			al.forEach(RemoteOp.class, x -> isNotEmpty(x.inner().value().trim()), x -> value.set(x.inner().value().trim()));
108 
109 			if (value.isPresent()) {
110 				String v = value.get();
111 				int i = v.indexOf(' ');
112 				if (i == -1) {
113 					httpMethod = v;
114 				} else {
115 					httpMethod = v.substring(0, i).trim();
116 					path = v.substring(i).trim();
117 				}
118 			} else {
119 				al.forEach(x -> ! x.isType(RemoteOp.class) && isNotEmpty(x.getValue(String.class, "value", NOT_EMPTY).orElse("").trim()),x -> value.set(x.getValue(String.class, "value", NOT_EMPTY).get().trim()));
120 				if (value.isPresent())
121 					path = value.get();
122 			}
123 
124 			if (path.isEmpty()) {
125 				path = HttpUtils.detectHttpPath(m, Utils.nullIfEmpty3(httpMethod));
126 			}
127 			if (httpMethod.isEmpty())
128 				httpMethod = HttpUtils.detectHttpMethod(m, true, defaultMethod);
129 
130 			path = trimSlashes(path);
131 
132 			if (! isOneOf(httpMethod, "DELETE", "GET", "POST", "PUT", "OPTIONS", "HEAD", "CONNECT", "TRACE", "PATCH"))
133 				throw new RemoteMetadataException(m,
134 					"Invalid value specified for @RemoteOp(httpMethod) annotation: '"+httpMethod+"'.  Valid values are [DELETE,GET,POST,PUT,OPTIONS,HEAD,CONNECT,TRACE,PATCH].");
135 
136 			methodReturn = new RemoteOperationReturn(mi);
137 
138 			fullPath = path.indexOf("://") != -1 ? path : (parentPath.isEmpty() ? urlEncodePath(path) : (trimSlashes(parentPath) + '/' + urlEncodePath(path)));
139 
140 			mi.getParams().forEach(x -> {
141 				RemoteOperationArg rma = RemoteOperationArg.create(x);
142 				if (rma != null) {
143 					HttpPartType pt = rma.getPartType();
144 					if (pt == HEADER)
145 						headerArgs.add(rma);
146 					else if (pt == QUERY)
147 						queryArgs.add(rma);
148 					else if (pt == FORMDATA)
149 						formDataArgs.add(rma);
150 					else if (pt == PATH)
151 						pathArgs.add(rma);
152 					else
153 						bodyArg = rma;
154 				}
155 				RequestBeanMeta rmba = RequestBeanMeta.create(x, AnnotationWorkList.create());
156 				if (rmba != null) {
157 					requestArgs.add(new RemoteOperationBeanArg(x.getIndex(), rmba));
158 				}
159 			});
160 		}
161 	}

163 	/**
164 	 * Returns the value of the {@link RemoteOp#method() @RemoteOp(method)} annotation on this Java method.
165 	 *
166 	 * @return The value of the annotation, never <jk>null</jk>.
167 	 */
168 	public String getHttpMethod() {
169 		return httpMethod;
170 	}
171 
172 	/**
173 	 * Returns the absolute URI of the REST interface invoked by this Java method.
174 	 *
175 	 * @return The absolute URI of the REST interface, never <jk>null</jk>.
176 	 */
177 	public String getFullPath() {
178 		return fullPath;
179 	}

181 	/**
182 	 * Performs an action on the {@link Path @Path} annotated arguments on this Java method.
183 	 *
184 	 * @param action The action to perform.
185 	 * @return This object.
186 	 */
187 	public RemoteOperationMeta forEachPathArg(Consumer<RemoteOperationArg> action) {
188 		for (RemoteOperationArg a : pathArgs)
189 			action.accept(a);
190 		return this;
191 	}
