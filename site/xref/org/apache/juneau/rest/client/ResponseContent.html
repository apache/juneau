47  /**
48   * Represents the body of an HTTP response.
49   *
50   * <p>
51   * An extension of an HttpClient {@link HttpEntity} that provides various support for converting the body to POJOs and
52   * other convenience methods.
53   *
54   * <h5 class='section'>See Also:</h5><ul>
55   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestClientBasics">juneau-rest-client Basics</a>
56   * </ul>
57   */
58  @SuppressWarnings("resource")
59  public class ResponseContent implements HttpEntity {
60  
61  	private static final HttpEntity NULL_ENTITY = new HttpEntity() {
62  
63  		@Override
64  		public void consumeContent() throws IOException {}
65  
66  		@Override
67  		public InputStream getContent() throws IOException, UnsupportedOperationException { return new ByteArrayInputStream(new byte[0]); }
68  
69  		@Override
70  		public Header getContentEncoding() { return ResponseHeader.NULL_HEADER; }
71  
72  		@Override
73  		public long getContentLength() { return -1; }
74  
75  		@Override
76  		public Header getContentType() { return ResponseHeader.NULL_HEADER; }
77  
78  		@Override
79  		public boolean isChunked() { return false; }
80  
81  		@Override
82  		public boolean isRepeatable() { return false; }
83  
84  		@Override
85  		public boolean isStreaming() { return false; }
86  
87  		@Override
88  		public void writeTo(OutputStream outstream) throws IOException {}
89  	};

102 	 * Constructor.
103 	 *
104 	 * @param client The client used to build this request.
105 	 * @param request The request object.
106 	 * @param response The response object.
107 	 * @param parser The parser to use to consume the body.  Can be <jk>null</jk>.

109 	public ResponseContent(RestClient client, RestRequest request, RestResponse response, Parser parser) {

117 	/**
118 	 * Same as {@link #as(Type,Type...)} except optimized for a non-parameterized class.
119 	 *
120 	 * <p>
121 	 * This is the preferred parse method for simple types since you don't need to cast the results.
122 	 *
123 	 * <h5 class='section'>Examples:</h5>
124 	 * <p class='bjava'>
125 	 * 	<jc>// Parse into a string.</jc>
126 	 * 	String <jv>string</jv> = <jv>client</jv>.get(<jsf>URI</jsf>).run().getContent().as(String.<jk>class</jk>);
127 	 *
128 	 * 	<jc>// Parse into a bean.</jc>
129 	 * 	MyBean <jv>bean</jv> = <jv>client</jv>.get(<jsf>URI</jsf>).run().getContent().as(MyBean.<jk>class</jk>);
130 	 *
131 	 * 	<jc>// Parse into a bean array.</jc>
132 	 * 	MyBean[] <jv>beanArray</jv> = <jv>client</jv>.get(<jsf>URI</jsf>).run().getContent().as(MyBean[].<jk>class</jk>);
133 	 *
134 	 * 	<jc>// Parse into a linked-list of objects.</jc>
135 	 * 	List <jv>list</jv> = <jv>client</jv>.get(<jsf>URI</jsf>).run().getContent().as(LinkedList.<jk>class</jk>);
136 	 *
137 	 * 	<jc>// Parse into a map of object keys/values.</jc>
138 	 * 	Map <jv>map</jv> = <jv>client</jv>.get(<jsf>URI</jsf>).run().getContent().as(TreeMap.<jk>class</jk>);
139 	 * </p>
140 	 *
141 	 * <h5 class='section'>Notes:</h5><ul>
142 	 * 	<li class='note'>
143 	 * 		You can also specify any of the following types:
144 	 * 		<ul class='compact'>
145 	 * 			<li>{@link ResponseContent}/{@link HttpEntity} - Returns access to this object.
146 	 * 			<li>{@link Reader} - Returns access to the raw reader of the response.
147 	 * 			<li>{@link InputStream} - Returns access to the raw input stream of the response.
148 	 * 			<li>{@link HttpResource} - Response will be converted to an {@link BasicResource}.
149 	 * 			<li>Any type that takes in an {@link HttpResponse} object.
150 	 * 		</ul>
151 	 * 	<li class='note'>
152 	 *		If {@link #cache()} or {@link RestResponse#cacheContent()} has been called, this method can be can be called multiple times and/or combined with
153 	 *		other methods that retrieve the content of the response.  Otherwise a {@link RestCallException}
154 	 *		with an inner {@link IllegalStateException} will be thrown.
155 	 * 	<li class='note'>
156 	 * 		The input stream is automatically closed after this call.
157 	 * </ul>
158 	 *
159 	 * @param <T>
160 	 * 	The class type of the object being created.
161 	 * 	See {@link #as(Type,Type...)} for details.
162 	 * @param type The object type to create.
163 	 * @return The parsed object.
164 	 * @throws RestCallException
165 	 * 	If the input contains a syntax error or is malformed, or is not valid for the specified type, or if a connection
166 	 * 	error occurred.
167 	 */
168 	public <T> T as(Class<T> type) throws RestCallException {
169 		return as(getClassMeta(type));
170 	}

173 	 * Same as {@link #as(Class)} except allows you to predefine complex data types using the {@link ClassMeta} API.
174 	 *
175 	 * <h5 class='section'>Examples:</h5>
176 	 * <p class='bjava'>
177 	 * 	BeanContext <jv>beanContext</jv> = BeanContext.<jsf>DEFAULT</jsf>;
178 	 *
179 	 * 	<jc>// Parse into a linked-list of strings.</jc>
180 	 *	ClassMeta&lt;List&lt;String&gt;&gt; <jv>cm1</jv> = <jv>beanContext</jv>.getClassMeta(LinkedList.<jk>class</jk>, String.<jk>class</jk>);
181 	 * 	List&lt;String&gt; <jv>list1</jv> = <jv>client</jv>.get(<jsf>URI</jsf>).run().getContent().as(<jv>cm1</jv>);
182 	 *
183 	 * 	<jc>// Parse into a linked-list of beans.</jc>
184 	 *	ClassMeta&lt;List&lt;String&gt;&gt; <jv>cm2</jv> = <jv>beanContext</jv>.getClassMeta(LinkedList.<jk>class</jk>, MyBean.<jk>class</jk>);
185 	 * 	List&lt;MyBean&gt; <jv>list2</jv> = <jv>client</jv>.get(<jsf>URI</jsf>).run().getContent().as(<jv>cm2</jv>);
186 	 *
187 	 * 	<jc>// Parse into a linked-list of linked-lists of strings.</jc>
188 	 *	ClassMeta&lt;List&lt;String&gt;&gt; <jv>cm3</jv> = <jv>beanContext</jv>.getClassMeta(LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
189 	 * 	List&lt;List&lt;String&gt;&gt; <jv>list3</jv> = <jv>client</jv>.get(<jsf>URI</jsf>).run().getContent().as(<jv>cm3</jv>);
190 	 *
191 	 * 	<jc>// Parse into a map of string keys/values.</jc>
192 	 *	ClassMeta&lt;List&lt;String&gt;&gt; <jv>cm4</jv> = <jv>beanContext</jv>.getClassMeta(TreeMap.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);
193 	 * 	Map&lt;String,String&gt; <jv>map4</jv> = <jv>client</jv>.get(<jsf>URI</jsf>).run().getContent().as(<jv>cm4</jv>);
194 	 *
195 	 * 	<jc>// Parse into a map containing string keys and values of lists containing beans.</jc>
196 	 *	ClassMeta&lt;List&lt;String&gt;&gt; <jv>cm5</jv> = <jv>beanContext</jv>.getClassMeta(TreeMap.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);
197 	 * 	Map&lt;String,List&lt;MyBean&gt;&gt; <jv>map5</jv> = <jv>client</jv>.get(<jsf>URI</jsf>).run().getContent().as(<jv>cm5</jv>);
198 	 * </p>
199 	 *
200 	 * <h5 class='section'>Notes:</h5><ul>
201 	 * 	<li class='note'>
202 	 *		If {@link #cache()} or {@link RestResponse#cacheContent()} has been called, this method can be can be called multiple times and/or combined with
203 	 *		other methods that retrieve the content of the response.  Otherwise a {@link RestCallException}
204 	 *		with an inner {@link IllegalStateException} will be thrown.
205 	 * 	<li class='note'>
206 	 * 		The input stream is automatically closed after this call.
207 	 * </ul>
208 	 *
209 	 * @param <T> The class type of the object to create.
210 	 * @param type The object type to create.
211 	 * @return The parsed object.
212 	 * @throws RestCallException
213 	 * 	<ul>
214 	 * 		<li>If the input contains a syntax error or is malformed, or is not valid for the specified type.
215 	 * 		<li>If a connection error occurred.
216 	 * 	</ul>
217 	 * @see BeanSession#getClassMeta(Class) for argument syntax for maps and collections.
218 	 */
219 	@SuppressWarnings("unchecked")
220 	public <T> T as(ClassMeta<T> type) throws RestCallException {
221 		try {
222 			if (type.is(ResponseContent.class) || type.is(HttpEntity.class))
223 				return (T)this;
224 
225 			if (type.is(Reader.class))
226 				return (T)asReader();
227 
228 			if (type.is(InputStream.class))
229 				return (T)asInputStream();
230 
231 			if (type.is(HttpResponse.class))
232 				return (T)response;
233 
234 			if (type.is(HttpResource.class))
235 				type = (ClassMeta<T>)getClassMeta(BasicResource.class);
236 
237 			ConstructorInfo ci = type.getInfo().getPublicConstructor(x -> x.hasParamTypes(HttpResponse.class));
238 			if (nn(ci)) {
239 				try {
240 					return (T)ci.invoke(response);
241 				} catch (ExecutableException e) {
242 					throw toRuntimeException(e);
243 				}
244 			}
245 
246 			var ct = firstNonEmpty(response.getHeader("Content-Type").orElse("text/plain"));
247 
248 			if (parser == null)
249 				parser = client.getMatchingParser(ct);
250 
251 			MediaType mt = MediaType.of(ct);
252 
253 			if (parser == null || (mt.toString().contains("text/plain") && ! parser.canHandle(ct))) {
254 				if (type.hasStringMutater())
255 					return type.getStringMutater().mutate(asString());
256 			}
257 
258 			if (nn(parser)) {
259 				try (Closeable in = parser.isReaderParser() ? asReader() : asInputStream()) {
260 
261 					// @formatter:off
262 					T t = parser
263 						.createSession()
264 						.properties(JsonMap.create().inner(request.getSessionProperties()))
265 						.locale(response.getLocale())
266 						.mediaType(mt)
267 						.schema(schema)
268 						.build()
269 						.parse(in, type);
270 					// @formatter:on
271 
272 					// Some HTTP responses have no body, so try to create these beans if they've got no-arg constructors.
273 					if (t == null && ! type.is(String.class)) {
274 						ConstructorInfo c = type.getInfo().getPublicConstructor(ConstructorInfo::hasNoParams);
275 						if (nn(c)) {
276 							try {
277 								return c.<T>invoke();
278 							} catch (ExecutableException e) {
279 								throw new ParseException(e);
280 							}
281 						}
282 					}
283 
284 					return t;
285 				}
286 			}
287 
288 			if (type.hasReaderMutater())
289 				return type.getReaderMutater().mutate(asReader());
290 
291 			if (type.hasInputStreamMutater())
292 				return type.getInputStreamMutater().mutate(asInputStream());
293 
294 			ct = response.getStringHeader("Content-Type").orElse(null);
295 
296 			if (ct == null && client.hasParsers())
297 				throw new ParseException("Content-Type not specified in response header.  Cannot find appropriate parser.");
298 
299 			throw new ParseException("Unsupported media-type in request header ''Content-Type'': ''{0}''", ct);
300 
301 		} catch (ParseException | IOException e) {
302 			response.close();
303 			throw new RestCallException(response, e, "Could not parse response body.");
304 		}
305 	}

307 	/**
308 	 * Parses HTTP body into the specified object type.
309 	 *
310 	 * <p>
311 	 * The type can be a simple type (e.g. beans, strings, numbers) or parameterized type (collections/maps).
312 	 *
313 	 * <h5 class='section'>Examples:</h5>
314 	 * <p class='bjava'>
315 	 * 	<jc>// Parse into a linked-list of strings.</jc>
316 	 * 	List&lt;String&gt; <jv>list1</jv> = <jv>client</jv>
317 	 * 		.get(<jsf>URI</jsf>)
318 	 * 		.run()
319 	 * 		.getContent().as(LinkedList.<jk>class</jk>, String.<jk>class</jk>);
320 	 *
321 	 * 	<jc>// Parse into a linked-list of beans.</jc>
322 	 * 	List&lt;MyBean&gt; <jv>list2</jv> = <jv>client</jv>
323 	 * 		.get(<jsf>URI</jsf>)
324 	 * 		.run()
325 	 * 		.getContent().as(LinkedList.<jk>class</jk>, MyBean.<jk>class</jk>);
326 	 *
327 	 * 	<jc>// Parse into a linked-list of linked-lists of strings.</jc>
328 	 * 	List&lt;List&lt;String&gt;&gt; <jv>list3</jv> = <jv>client</jv>
329 	 * 		.get(<jsf>URI</jsf>)
330 	 * 		.run()
331 	 * 		.getContent().as(LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
332 	 *
333 	 * 	<jc>// Parse into a map of string keys/values.</jc>
334 	 * 	Map&lt;String,String&gt; <jv>map1</jv> = <jv>client</jv>
335 	 * 		.get(<jsf>URI</jsf>)
336 	 * 		.run()
337 	 * 		.getContent().as(TreeMap.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);
338 	 *
339 	 * 	<jc>// Parse into a map containing string keys and values of lists containing beans.</jc>
340 	 * 	Map&lt;String,List&lt;MyBean&gt;&gt; <jv>map2</jv> = <jv>client</jv>
341 	 * 		.get(<jsf>URI</jsf>)
342 	 * 		.run()
343 	 * 		.getContent().as(TreeMap.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);
344 	 * </p>
345 	 *
346 	 * <p>
347 	 * <c>Collection</c> classes are assumed to be followed by zero or one objects indicating the element type.
348 	 *
349 	 * <p>
350 	 * <c>Map</c> classes are assumed to be followed by zero or two meta objects indicating the key and value types.
351 	 *
352 	 * <p>
353 	 * The array can be arbitrarily long to indicate arbitrarily complex data structures.
354 	 *
355 	 * <h5 class='section'>Notes:</h5><ul>
356 	 * 	<li class='note'>
357 	 * 		Use the {@link #as(Class)} method instead if you don't need a parameterized map/collection.
358 	 * 	<li class='note'>
359 	 * 		You can also specify any of the following types:
360 	 * 		<ul class='compact'>
361 	 * 			<li>{@link ResponseContent}/{@link HttpEntity} - Returns access to this object.
362 	 * 			<li>{@link Reader} - Returns access to the raw reader of the response.
363 	 * 			<li>{@link InputStream} - Returns access to the raw input stream of the response.
364 	 * 			<li>{@link HttpResource} - Response will be converted to an {@link BasicResource}.
365 	 * 			<li>Any type that takes in an {@link HttpResponse} object.
366 	 * 		</ul>
367 	 * 	<li class='note'>
368 	 *		If {@link #cache()} or {@link RestResponse#cacheContent()} has been called, this method can be can be called multiple times and/or combined with
369 	 *		other methods that retrieve the content of the response.  Otherwise a {@link RestCallException}
370 	 *		with an inner {@link IllegalStateException} will be thrown.
371 	 * 	<li class='note'>
372 	 * 		The input stream is automatically closed after this call.
373 	 * </ul>
374 	 *
375 	 * @param <T> The class type of the object to create.
376 	 * @param type
377 	 * 	The object type to create.
378 	 * 	<br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
379 	 * @param args
380 	 * 	The type arguments of the class if it's a collection or map.
381 	 * 	<br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
382 	 * 	<br>Ignored if the main type is not a map or collection.
383 	 * @return The parsed object.
384 	 * @throws RestCallException
385 	 * 	<ul>
386 	 * 		<li>If the input contains a syntax error or is malformed, or is not valid for the specified type.
387 	 * 		<li>If a connection error occurred.
388 	 * 	</ul>
389 	 * @see BeanSession#getClassMeta(Class) for argument syntax for maps and collections.
390 	 */
391 	public <T> T as(Type type, Type...args) throws RestCallException {
392 		return as(getClassMeta(type, args));
393 	}

409 	/**
410 	 * Returns the HTTP response message body as a byte array.
411 	 *
412 	 * 	The HTTP response message body reader, never <jk>null</jk>.
413 	 * 	<br>For responses without a body(e.g. HTTP 204), returns an empty array.
414 	 *
415 	 * @return The HTTP response body as a byte array.
416 	 * @throws RestCallException If an exception occurred.
417 	 */
418 	public byte[] asBytes() throws RestCallException {
419 		if (body == null) {
420 			try {
421 				if (entity instanceof BasicHttpEntity) {
422 					body = ((BasicHttpEntity)entity).asBytes();
423 				} else {
424 					body = readBytes(entity.getContent());
425 				}
426 			} catch (IOException e) {
427 				throw new RestCallException(response, e, "Could not read response body.");
428 			} finally {
429 				response.close();
430 			}
431 		}
432 		return body;
433 	}

582 					response.close();
