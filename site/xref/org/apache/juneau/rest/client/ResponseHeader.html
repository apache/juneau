41   * Represents a single header on an HTTP response.
42   *
43   * <p>
44   * An extension of an HttpClient {@link Header} that provides various support for converting the header to POJOs and
45   * other convenience methods.

73  	/**
74  	 * Constructor.
75  	 * @param name The header name.
76  	 * @param request The request object.
77  	 * @param response The response object.
78  	 * @param header The wrapped header.  Can be <jk>null</jk>.
79  	 * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
80  	 */
81  	public ResponseHeader(String name, RestRequest request, RestResponse response, Header header) {
82  		super(name, header == null ? null : header.getValue());
83  		this.request = request;
84  		this.response = response;
85  		this.elements = header == null ? new HeaderElement[0] : header.getElements();
86  		parser(null);
87  	}

100 	/**
101 	 * Converts this header to the specified type.
102 	 *
103 	 * @param <T> The type to convert to.
104 	 * @param type The type to convert to.
105 	 * @return The converted type, or <jk>null</jk> if header is not present.
106 	 */
107 	public <T> Optional<T> as(ClassMeta<T> type) {
108 		try {
109 			return opt(parser.parse(HEADER, schema, getValue(), type));
110 		} catch (ParseException e) {
111 			throw new BasicRuntimeException(e, "Could not parse response header {0}.", getName());
112 		}
113 	}

246 	/**
247 	 * Returns the value of this header as a {@link BasicHeader}.
248 	 *
249 	 * @param c The subclass of {@link BasicHeader} to instantiate.
250 	 * @param <T> The subclass of {@link BasicHeader} to instantiate.
251 	 * @return The value of this header as a string, never <jk>null</jk>.
252 	 */
253 	public <T extends BasicHeader> T asHeader(Class<T> c) {
254 		try {
255 			ClassInfo ci = ClassInfo.of(c);
256 			ConstructorInfo cc = ci.getPublicConstructor(x -> x.hasParamTypes(String.class));
257 			if (nn(cc))
258 				return cc.invoke(getValue());
259 			cc = ci.getPublicConstructor(x -> x.hasParamTypes(String.class, String.class));
260 			if (nn(cc))
261 				return cc.invoke(getName(), getValue());
262 		} catch (Throwable e) {
263 			if (e instanceof ExecutableException)
264 				e = e.getCause();
265 			throw asRuntimeException(e);
266 		}
267 		throw new BasicRuntimeException("Could not determine a method to construct type {0}", cn(c));
268 	}

404 	 * Provides the ability to perform fluent-style assertions on this response header.
405 	 *
406 	 * <h5 class='section'>Examples:</h5>
407 	 * <p class='bjava'>
408 	 * 	<jc>// Validates the content type header is provided.</jc>
409 	 * 	<jv>client</jv>
410 	 * 		.get(<jsf>URI</jsf>)
411 	 * 		.run()
412 	 * 		.getHeader(<js>"Content-Type"</js>).assertValue().exists();
413 	 *
414 	 * 	<jc>// Validates the content type is JSON.</jc>
415 	 * 	<jv>client</jv>
416 	 * 		.get(<jsf>URI</jsf>)
417 	 * 		.run()
418 	 * 		.getHeader(<js>"Content-Type"</js>).assertValue().equals(<js>"application/json"</js>);
419 	 *
420 	 * 	<jc>// Validates the content type is JSON using test predicate.</jc>
421 	 * 	<jv>client</jv>
422 	 * 		.get(<jsf>URI</jsf>)
423 	 * 		.run()
424 	 * 		.getHeader(<js>"Content-Type"</js>).assertValue().is(<jv>x</jv> -&gt; <jv>x</jv>.equals(<js>"application/json"</js>));
425 	 *
426 	 * 	<jc>// Validates the content type is JSON by just checking for substring.</jc>
427 	 * 	<jv>client</jv>
428 	 * 		.get(<jsf>URI</jsf>)
429 	 * 		.run()
430 	 * 		.getHeader(<js>"Content-Type"</js>).assertValue().contains(<js>"json"</js>);
431 	 *
432 	 * 	<jc>// Validates the content type is JSON using regular expression.</jc>
433 	 * 	<jv>client</jv>
434 	 * 		.get(<jsf>URI</jsf>)
435 	 * 		.run()
436 	 * 		.getHeader(<js>"Content-Type"</js>).assertValue().isPattern(<js>".*json.*"</js>);
437 	 *
438 	 * 	<jc>// Validates the content type is JSON using case-insensitive regular expression.</jc>
439 	 * 	<jv>client</jv>
440 	 * 		.get(<jsf>URI</jsf>)
441 	 * 		.run()
442 	 * 		.getHeader(<js>"Content-Type"</js>).assertValue().isPattern(<js>".*json.*"</js>, <jsf>CASE_INSENSITIVE</jsf>);
443 	 * </p>
444 	 *
445 	 * <p>
446 	 * The assertion test returns the original response object allowing you to chain multiple requests like so:
447 	 * <p class='bjava'>
448 	 * 	<jc>// Validates the header and converts it to a bean.</jc>

450 	 * 		.get(<jsf>URI</jsf>)
451 	 * 		.run()
452 	 * 		.getHeader(<js>"Content-Type"</js>).assertValue().isNotEmpty()
453 	 * 		.getHeader(<js>"Content-Type"</js>).assertValue().isPattern(<js>".*json.*"</js>)

455 	 * </p>
456 	 *
457 	 * @return A new fluent assertion object.
458 	 */

461 	}
