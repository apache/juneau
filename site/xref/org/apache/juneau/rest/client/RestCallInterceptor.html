13  package org.apache.juneau.rest.client;
14  
15  /**
16   * Used to intercept http connection responses to allow modification of that response before processing and for
17   * listening for call lifecycle events.
18   *
19   * <p>
20   * The {@link BasicRestCallInterceptor} is provided as an adapter class for implementing this interface.
21   *
22   * <p>
23   * Note that the {@link RestClient} class itself implements this interface so you can achieve the same results by
24   * overriding the methods on the client class as well.
25   *
26   * <h5 class='figure'>Example:</h5>
27   * <p class='bjava'>
28   * 	<jc>// Specialized client that adds a header to every request.</jc>
29   * 	<jk>public class</jk> MyRestClient <jk>extends</jk> RestClient {
30   * 		<ja>@Override</ja>
31   * 		<jk>public void</jk> onInit(RestRequest <jv>req</jv>) {
32   * 			<jv>req</jv>.header(<js>"Foo"</js>, <js>"bar"</js>);
33   * 		}
34   * 	}
35   *
36   *	<jc>// Instantiate the client.</jc>
37   *	MyRestClient <jv>client</jv> = RestClient
38   *		.<jsm>create</jsm>()
39   *		.json()
40   *		.build(MyRestClient.<jk>class</jk>);
41   * </p>
42   *
43   * <h5 class='section'>See Also:</h5><ul>
44   * 	<li class='jm'>{@link RestClient.Builder#interceptors(Object...)}
45   * 	<li class='link'><a class="doclink" href="../../../../../index.html#juneau-rest-client">juneau-rest-client</a>
46   * </ul>
47   */
48  public interface RestCallInterceptor {
49  
50  	/**
51  	 * Called immediately after {@link RestRequest} object is created and all headers/query/form-data has been
52  	 * copied from the client to the request object.
53  	 *
54  	 * @param req The HTTP request object.
55  	 * @throws Exception
56  	 * 	Any exception can be thrown.
57  	 * 	<br>If not a {@link RestCallException} or {@link RuntimeException}, will be wrapped in a {@link RestCallException}.
58  	 */
59  	void onInit(RestRequest req) throws Exception;
60  
61  	/**
62  	 * Called immediately after an HTTP response has been received.
63  	 *
64  	 * @param req The HTTP request object.
65  	 * @param res The HTTP response object.
66  	 * @throws Exception
67  	 * 	Any exception can be thrown.
68  	 * 	<br>If not a {@link RestCallException} or {@link RuntimeException}, will be wrapped in a {@link RestCallException}.
69  	 */
70  	void onConnect(RestRequest req, RestResponse res) throws Exception;
71  
72  	/**
73  	 * Called when the response body is consumed.
74  	 *
75  	 * @param req The request object.
76  	 * @param res The response object.
77  	 * @throws RestCallException Error occurred during call.
78  	 * @throws Exception
79  	 * 	Any exception can be thrown.
80  	 * 	<br>If not a {@link RestCallException} or {@link RuntimeException}, will be wrapped in a {@link RestCallException}.
81  	 */
82  	void onClose(RestRequest req, RestResponse res) throws Exception;
83  }
