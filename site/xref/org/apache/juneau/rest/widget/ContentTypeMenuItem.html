30  /**
31   * Widget that returns back a list of hyperlinks for rendering the contents of a page in a variety of content types.
32   *
33   * <p>
34   * The variable it resolves is <js>"$W{ContentTypeMenuItem}"</js>.
35   *
36   * <p>
37   * An example of this widget can be found in the <c>PetStoreResource</c> in the examples that provides
38   * a drop-down menu item for rendering all other supported content types in plain text:
39   * <p class='bjava'>
40   * 	<ja>@RestGet</ja>(<js>"/"</js>)
41   *  <ja>@HtmlDocConfig</ja>(
42   * 		widgets={
43   * 			ContentTypeMenuItem.<jk>class</jk>,
44   * 		},
45   * 		navlinks={
46   * 			<js>"up: ..."</js>,
47   * 			<js>"options: ..."</js>,
48   * 			<js>"$W{QueryMenuItem}"</js>,
49   * 			<js>"$W{ContentTypeMenuItem}"</js>,
50   * 			<js>"$W{ThemeMenuItem}"</js>,
51   * 			<js>"source: ..."</js>
52   * 		}
53   * 	)
54   * 	<jk>public</jk> Collection&lt;Pet&gt; getPets() {
55   * </p>
56   *
57   * <h5 class='section'>See Also:</h5><ul>
58   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HtmlPredefinedWidgets">Predefined Widgets</a>
59   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HtmlWidgets">Widgets</a>
60   * </ul>
61   */
62  public class ContentTypeMenuItem extends MenuItemWidget {
63  
64  	@Override /* Overridden from MenuItemWidget */
65  	public Div getContent(RestRequest req, RestResponse res) {
66  		Div div = div();
67  		Set<MediaType> l = new TreeSet<>();
68  		for (Serializer s : req.getOpContext().getSerializers().getSerializers())
69  			l.add(s.getPrimaryMediaType());
70  		for (MediaType mt : l) {
71  			URI uri = req.getUri(true, CollectionUtils.<String,Object>map("plainText", "true", "Accept", mt.toString()));
72  			div.children(a(uri, mt), br());
73  		}
74  		return div;
75  	}
76  
77  	@Override /* Overridden from MenuItemWidget */
78  	public String getLabel(RestRequest req, RestResponse res) {
79  		return "content-type";
80  	}
81  }
