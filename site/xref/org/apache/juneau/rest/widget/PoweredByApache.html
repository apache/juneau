13  package org.apache.juneau.rest.widget;
14  
15  import org.apache.juneau.*;
16  import org.apache.juneau.rest.*;
17  
18  /**
19   * Widget that places a powered-by-Apache message on the page.
20   *
21   * <p>
22   * The variable it resolves is <js>"$W{PoweredByApache}"</js>.
23   *
24   * <p>
25   * It produces a simple Apache icon floating on the right.
26   * Typically it's used in the footer of the page, as shown below in the <c>RootResources</c> from the examples:
27   *
28   * <p class='bjava'>
29   * 	<ja>@Rest</ja>(
30   * 		path=<js>"/"</js>,
31   * 		title=<js>"Root resources"</js>,
32   * 		description=<js>"Example of a router resource page."</js>
33   * 	)
34   *  <ja>@HtmlDocConfig</ja>(
35   * 		widgets={
36   * 			PoweredByApache.<jk>class</jk>
37   * 		},
38   * 		footer=<js>"$W{PoweredByApache}"</js>
39   * 	)
40   * </p>
41   *
42   * <p>
43   * It renders the following image:
44   * <img class='bordered' src='doc-files/PoweredByApacheWidget.png'>
45   *
46   * <h5 class='section'>See Also:</h5><ul>
47   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HtmlPredefinedWidgets">Predefined Widgets</a>
48   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HtmlWidgets">Widgets</a>
49   * </ul>
50   */
51  public class PoweredByApache extends Widget {
52  
53  	/**
54  	 * Returns an Apache image tag hyperlinked to <js>"http://apache.org"</js>
55  	 */
56  	@Override /* Widget */
57  	public String getHtml(RestRequest req, RestResponse res) {
58  		UriResolver r = req.getUriResolver();
59  		return "<a href='http://apache.org'><img style='float:right;padding-right:20px;height:32px' src='"+r.resolve("servlet:/htdocs/asf.png")+"'>";
60  	}
61  }
