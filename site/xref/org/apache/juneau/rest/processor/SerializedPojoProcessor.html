32  /**
33   * Response handler for plain-old Java objects.
34   *
35   * <h5 class='section'>See Also:</h5><ul>
36   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/ResponseProcessors">Response Processors</a>
37   * </ul>
38   */
39  @SuppressWarnings("resource")
40  public class SerializedPojoProcessor implements ResponseProcessor {
41  
42  	@Override /* Overridden from ResponseProcessor */
43  	public int process(RestOpSession opSession) throws IOException, NotAcceptable, BasicHttpException {
44  		RestRequest req = opSession.getRequest();
45  		RestResponse res = opSession.getResponse();
46  		SerializerMatch sm = res.getSerializerMatch().orElse(null);
47  		HttpPartSchema schema = res.getContentSchema().orElse(null);
48  
49  		Object o = res.getContent(Object.class);
50  
51  		if (nn(sm)) {
52  			try {
53  				Serializer s = sm.getSerializer();
54  				MediaType mediaType = res.getMediaType();
55  				if (mediaType == null)
56  					mediaType = sm.getMediaType();
57  
58  				MediaType responseType = s.getResponseContentType();
59  				if (responseType == null)
60  					responseType = mediaType;
61  
62  				if (req.isPlainText())
63  					res.setHeader(ContentType.TEXT_PLAIN);
64  				else
65  					res.setHeader(ContentType.of(responseType.toString()));
66  
67  				// @formatter:off
68  				SerializerSession session = s
69  					.createSession()
70  					.properties(req.getAttributes().asMap())
71  					.javaMethod(req.getOpContext().getJavaMethod())
72  					.locale(req.getLocale())
73  					.timeZone(req.getTimeZone().orElse(null))
74  					.mediaType(mediaType)
75  					.apply(WriterSerializerSession.Builder.class, x -> x.streamCharset(res.getCharset()).useWhitespace(req.isPlainText() ? true : null))
76  					.schema(schema)
77  					.debug(req.isDebug() ? true : null)
78  					.uriContext(req.getUriContext())
79  					.resolver(req.getVarResolverSession())
80  					.build();
81  				// @formatter:on
82  
83  				for (var h : session.getResponseHeaders().entrySet())
84  					res.addHeader(h.getKey(), h.getValue());
85  
86  				if (! session.isWriterSerializer()) {
87  					if (req.isPlainText()) {
88  						FinishablePrintWriter w = res.getNegotiatedWriter();
89  						var baos = new ByteArrayOutputStream();
90  						session.serialize(o, baos);
91  						w.write(toSpacedHex(baos.toByteArray()));
92  						w.flush();
93  						w.finish();
94  					} else {
95  						FinishableServletOutputStream os = res.getNegotiatedOutputStream();
96  						session.serialize(o, os);
97  						os.flush();
98  						os.finish();
99  					}
100 				} else {
101 					FinishablePrintWriter w = res.getNegotiatedWriter();
102 					session.serialize(o, w);
103 					w.flush();
104 					w.finish();
105 				}
106 			} catch (SerializeException e) {
107 				throw new InternalServerError(e);
108 			}
109 			return FINISHED;
110 		}
111 
112 		if (o == null)
113 			return FINISHED;
114 
115 		throw new NotAcceptable("Unsupported media-type in request header ''Accept'': ''{0}''\n\tSupported media-types: {1}", req.getHeaderParam("Accept").orElse(""),
116 			Json5.of(res.getOpContext().getSerializers().getSupportedMediaTypes()));
117 	}
118 }
