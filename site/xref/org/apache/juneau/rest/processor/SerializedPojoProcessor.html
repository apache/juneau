15  import java.io.*;
16  import java.util.*;

18  import org.apache.juneau.rest.*;
19  import org.apache.juneau.rest.util.*;
20  import org.apache.juneau.serializer.*;
21  import org.apache.juneau.*;

28  /**
29   * Response handler for plain-old Java objects.
30   *
31   * <h5 class='section'>See Also:</h5><ul>
32   * 	<li class='link'><a class="doclink" href="../../../../../index.html#jrs.ResponseProcessors">Response Processors</a>
33   * </ul>
34   */
35  public final class SerializedPojoProcessor implements ResponseProcessor {
36  
37  	@Override /* ResponseProcessor */
38  	public int process(RestOpSession opSession) throws IOException, NotAcceptable, BasicHttpException {
39  		RestRequest req = opSession.getRequest();
40  		RestResponse res = opSession.getResponse();
41  		SerializerMatch sm = res.getSerializerMatch().orElse(null);
42  		HttpPartSchema schema = res.getContentSchema().orElse(null);
43  
44  		Object o = res.getContent(Object.class);
45  
46  		if (sm != null) {
47  			try {
48  				Serializer s = sm.getSerializer();
49  				MediaType mediaType = res.getMediaType();
50  				if (mediaType == null)
51  					mediaType = sm.getMediaType();
52  
53  				MediaType responseType = s.getResponseContentType();
54  				if (responseType == null)
55  					responseType = mediaType;
56  
57  				if (req.isPlainText())
58  					res.setHeader(ContentType.TEXT_PLAIN);
59  				else
60  					res.setHeader(ContentType.of(responseType.toString()));
61  
62  				SerializerSession session = s
63  					.createSession()
64  					.properties(req.getAttributes().asMap())
65  					.javaMethod(req.getOpContext().getJavaMethod())
66  					.locale(req.getLocale())
67  					.timeZone(req.getTimeZone().orElse(null))
68  					.mediaType(mediaType)
69  					.apply(WriterSerializerSession.Builder.class, x -> x.streamCharset(res.getCharset()).useWhitespace(req.isPlainText() ? true : null))
70  					.schema(schema)
71  					.debug(req.isDebug() ? true : null)
72  					.uriContext(req.getUriContext())
73  					.resolver(req.getVarResolverSession())
74  					.build();
75  
76  				for (Map.Entry<String,String> h : session.getResponseHeaders().entrySet())
77  					res.addHeader(h.getKey(), h.getValue());
78  
79  				if (! session.isWriterSerializer()) {
80  					if (req.isPlainText()) {
81  						FinishablePrintWriter w = res.getNegotiatedWriter();
82  						ByteArrayOutputStream baos = new ByteArrayOutputStream();
83  						session.serialize(o, baos);
84  						w.write(StringUtils.toSpacedHex(baos.toByteArray()));
85  						w.flush();
86  						w.finish();
87  					} else {
88  						FinishableServletOutputStream os = res.getNegotiatedOutputStream();
89  						session.serialize(o, os);
90  						os.flush();
91  						os.finish();
92  					}
93  				} else {
94  					FinishablePrintWriter w = res.getNegotiatedWriter();
95  					session.serialize(o, w);
96  					w.flush();
97  					w.finish();
98  				}
99  			} catch (SerializeException e) {
100 				throw new InternalServerError(e);
101 			}
102 			return FINISHED;
103 		}
104 
105 		if (o == null)
106 			return FINISHED;
107 
108 		throw new NotAcceptable(
109 			"Unsupported media-type in request header ''Accept'': ''{0}''\n\tSupported media-types: {1}",
110 			req.getHeaderParam("Accept").orElse(""), Json5.of(res.getOpContext().getSerializers().getSupportedMediaTypes())
111 		);
112 	}
113 }
