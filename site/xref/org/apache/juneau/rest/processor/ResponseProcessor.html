17  package org.apache.juneau.rest.processor;
18  
19  import java.io.*;
20  
21  import org.apache.juneau.http.response.*;
22  import org.apache.juneau.rest.*;
23  import org.apache.juneau.rest.annotation.*;
24  
25  import jakarta.servlet.http.*;
26  
27  /**
28   * Defines the interface for processors that convert POJOs to appropriate HTTP responses.
29   *
30   * <p>
31   * The REST Server API uses the concept of registered response processors for converting objects returned by REST
32   * methods or set through {@link RestResponse#setContent(Object)} into appropriate HTTP responses.
33   *
34   * <p>
35   * Response processors can be associated with REST resources via the following:
36   * <ul>
37   * 	<li class='ja'>{@link Rest#responseProcessors}
38   * 	<li class='jm'>{@link org.apache.juneau.rest.RestContext.Builder#responseProcessors()}
39   * </ul>
40   *
41   * <p>
42   * Response processors can be used to process POJOs that cannot normally be handled through Juneau serializers, or
43   * because it's simply easier to define response processors for special cases.
44   *
45   * <p>
46   * The following example shows how to create a response processor to handle special <c>Foo</c> objects outside the
47   * normal Juneau architecture.
48   * <p class='bjava'>
49   * 	<ja>@Rest</ja>(
50   * 		path=<js>"/example"</js>,
51   * 		responseProcessors=FooProcessor.<jk>class</jk>
52   * 	)
53   * 	<jk>public class</jk> Example <jk>extends</jk> BasicRestServlet {
54   *
55   * 		<ja>@RestGet</ja>(<js>"/"</js>)
56   * 		<jk>public</jk> Foo test1() {
57   * 			<jk>return new</jk> Foo(<js>"123"</js>);
58   * 		}
59   *
60   * 		<jk>public static class</jk> FooProcessor <jk>implements</jk> ResponseProcessor {
61   * 			<ja>@Override</ja>
62   * 			<jk>public int</jk> process(RestOpSession <jv>opSession</jv>) {
63   *
64   * 				RestResponse <jv>res</jv> = <jv>opSession</jv>.getRestResponse();
65   * 				Foo <jv>foo</jv> = <jv>res</jv>.getOutput(Foo.<jk>class</jk>);
66   *
67   * 				<jk>if</jk> (<jv>foo</jv> == <jk>null</jk>)
68   * 					<jk>return</jk> <jsf>NEXT</jsf>;  <jc>// Let the next processor handle it.</jc>
69   *
70   * 				<jc>// Set some headers and content.</jc>
71   * 				<jv>res</jv>.setHeader(<js>"Foo-ID"</js>, <jv>foo</jv>.getId());
72   * 				<jv>res</jv>.getWriter().write(<js>"foo.id="</js> + <jv>foo</jv>.getId());
73   *
74   * 				<jk>return</jk> <jsf>FINISHED</jsf>;  <jc>// We handled it.</jc>
75   * 			}
76   * 		}
77   * 	}
78   * </p>
79   *
80   * <h5 class='section'>See Also:</h5><ul>
81   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/ResponseProcessors">Response Processors</a>
82   * </ul>
83   */
84  public interface ResponseProcessor {
85  
86  	/**
87  	 * Return code indicating to proceed to the next response processor in the chain.
88  	 */
89  	int NEXT = 0;
90  
91  	/**
92  	 * Return code indicating that processing is complete and to exit the chain.
93  	 */
94  	int FINISHED = 1;
95  
96  	/**
97  	 * Return code indicating to restart processing the chain from the beginning.
98  	 */
99  	int RESTART = 2;
100 
101 	/**
102 	 * Process this response if possible.
103 	 *
104 	 * @param opSession The HTTP call.
105 	 * @return One of the following codes:
106 	 * 	<ul>
107 	 * 		<li><c>0</c> - The processor could not handle the request.
108 	 * 		<li><c>1</c> - The processor was able to fully handle the request.
109 	 * 		<li><c>2</c> - The processor was able to partially handle the request by replacing the output.
110 	 * 			The response processors should start over.
111 	 * 	</ul>
112 	 * @throws IOException
113 	 * 	If low-level exception occurred on output stream.
114 	 * 	<br>Results in a {@link HttpServletResponse#SC_INTERNAL_SERVER_ERROR} error.
115 	 * @throws BasicHttpException
116 	 * 	If some other exception occurred.
117 	 * 	<br>Can be used to provide an appropriate HTTP response code and message.
118 	 */
119 	int process(RestOpSession opSession) throws IOException, BasicHttpException;
120 }
