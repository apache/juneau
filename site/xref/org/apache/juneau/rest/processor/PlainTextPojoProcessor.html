17  package org.apache.juneau.rest.processor;
18  
19  import java.io.*;
20  
21  import org.apache.juneau.common.utils.*;
22  import org.apache.juneau.http.header.*;
23  import org.apache.juneau.http.response.*;
24  import org.apache.juneau.rest.*;
25  import org.apache.juneau.rest.util.*;
26  
27  /**
28   * Response handler for plain-old Java objects when a serializer match is not found and they're asking for plain/text or anything.
29   *
30   * <h5 class='section'>See Also:</h5><ul>
31   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/ResponseProcessors">Response Processors</a>
32   * </ul>
33   */
34  public class PlainTextPojoProcessor implements ResponseProcessor {
35  
36  	@Override /* ResponseProcessor */
37  	public int process(RestOpSession opSession) throws IOException, NotAcceptable, BasicHttpException {
38  		RestRequest req = opSession.getRequest();
39  		RestResponse res = opSession.getResponse();
40  		String accept = req.getHeaderParam("Accept").orElse("*/*");
41  
42  		if (res.getSerializerMatch().isPresent() || ! (Utils.isEmpty(accept) || accept.startsWith("text/plain") || accept.contains("*/*")))
43  			return NEXT;
44  
45  		Object o = res.getContent(Object.class);
46  
47  		if (Utils.isEmpty(res.getContentType()))
48  			res.setHeader(ContentType.TEXT_PLAIN);
49  
50  		FinishablePrintWriter w = res.getNegotiatedWriter();
51  		if (o == null)
52  			w.append("null");
53  		else
54  			w.append(req.getBeanSession().getClassMetaForObject(o).toString(o));
55  		w.flush();
56  		w.finish();
57  
58  		return FINISHED;
59  	}
60  }
