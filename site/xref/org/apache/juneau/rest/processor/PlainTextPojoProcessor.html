13  package org.apache.juneau.rest.processor;
14  
15  import java.io.*;
16  
17  import org.apache.juneau.common.utils.*;
18  import org.apache.juneau.http.header.*;
19  import org.apache.juneau.http.response.*;
20  import org.apache.juneau.rest.*;
21  import org.apache.juneau.rest.util.*;
22  
23  /**
24   * Response handler for plain-old Java objects when a serializer match is not found and they're asking for plain/text or anything.
25   *
26   * <h5 class='section'>See Also:</h5><ul>
27   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/ResponseProcessors">Response Processors</a>
28   * </ul>
29   */
30  public class PlainTextPojoProcessor implements ResponseProcessor {
31  
32  	@Override /* ResponseProcessor */
33  	public int process(RestOpSession opSession) throws IOException, NotAcceptable, BasicHttpException {
34  		RestRequest req = opSession.getRequest();
35  		RestResponse res = opSession.getResponse();
36  		String accept = req.getHeaderParam("Accept").orElse("*/*");
37  
38  		if (res.getSerializerMatch().isPresent() || ! (Utils.isEmpty(accept) || accept.startsWith("text/plain") || accept.contains("*/*")))
39  			return NEXT;
40  
41  		Object o = res.getContent(Object.class);
42  
43  		if (Utils.isEmpty(res.getContentType()))
44  			res.setHeader(ContentType.TEXT_PLAIN);
45  
46  		FinishablePrintWriter w = res.getNegotiatedWriter();
47  		if (o == null)
48  			w.append("null");
49  		else
50  			w.append(req.getBeanSession().getClassMetaForObject(o).toString(o));
51  		w.flush();
52  		w.finish();
53  
54  		return FINISHED;
55  	}
56  }
