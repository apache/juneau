15  import org.apache.juneau.rest.*;
16  import org.apache.juneau.rest.annotation.*;
17  import org.apache.juneau.rest.beans.*;
18  import org.apache.juneau.rest.config.*;

20  /**
21   * Specialized subclass of {@link BasicRestServlet} for showing "group" pages.
22   *
23   * <p>
24   * Meant as a base class for top-level REST resources in servlet containers.
25   *
26   * <p>
27   * Provides support for JSON, XML, HTML, URL-Encoding, UON, XML, OpenAPI, and MessagePack.  See {@link BasicUniversalConfig}
28   * for details.
29   *
30   * <p>
31   * Implements the basic REST endpoints defined in {@link BasicRestOperations} and {@link BasicGroupOperations}.
32   *
33   * <p>
34   * Children are attached to this resource using the {@link Rest#children() @Rest(children)} annotation.
35   *
36   * <h5 class='section'>See Also:</h5><ul>
37   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestAnnotatedClassBasics">@Rest-Annotated Classes</a>
38   * </ul>
39   *
40   * @serial exclude
41   */
42  @Rest
43  public abstract class BasicRestServletGroup extends BasicRestServlet implements BasicGroupOperations {
44  	private static final long serialVersionUID = 1L;
45  
46  
47  	@Override /* BasicGroupOperations */
48  	public ChildResourceDescriptions getChildren(RestRequest req) {
49  		return ChildResourceDescriptions.of(req);
50  	}
51  }
