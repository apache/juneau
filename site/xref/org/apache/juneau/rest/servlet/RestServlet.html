40  /**
41   * Servlet implementation of a REST resource.
42   *
43   * <p>
44   * 	The {@link RestServlet} class is the entry point for your REST resources.
45   * 	It extends directly from <l>HttpServlet</l> and is deployed like any other servlet.
46   * </p>
47   * <p>
48   * 	When the servlet <l>init()</l> method is called, it triggers the code to find and process the <l>@Rest</l>
49   * 	annotations on that class and all child classes.
50   * 	These get constructed into a {@link RestContext} object that holds all the configuration
51   * 	information about your resource in a read-only object.
52   * </p>
53   *
54   * <h5 class='section'>Notes:</h5><ul>
55   * 	<li class='note'>
56   * 		Users will typically extend from {@link BasicRestServlet} or {@link BasicRestServletGroup}
57   * 		instead of this class directly.
58   * </ul>
59   *
60   * <h5 class='section'>See Also:</h5><ul>
61   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestAnnotatedClassBasics">@Rest-Annotated Class Basics</a>
62   * </ul>
63   *
64   * @serial exclude
65   */
66  public abstract class RestServlet extends HttpServlet {
67  
68  	private static final long serialVersionUID = 1L;
69  
70  	private AtomicReference<RestContext> context = new AtomicReference<>();
71  	private AtomicReference<Exception> initException = new AtomicReference<>();
72  
73  	@Override /* Overridden from GenericServlet */
74  	public synchronized void destroy() {
75  		if (context.get() != null)
76  			context.get().destroy();
77  		super.destroy();
78  	}
79  
80  	/**
81  	 * Returns the read-only context object that contains all the configuration information about this resource.
82  	 *
83  	 * <p>
84  	 * This object is <jk>null</jk> during the call to {@link #init(ServletConfig)} but is populated by the time
85  	 * {@link #init()} is called.
86  	 *
87  	 * <p>
88  	 * Resource classes that don't extend from {@link RestServlet} can add the following method to their class to get
89  	 * access to this context object:
90  	 * <p class='bjava'>
91  	 * 	<jk>public void</jk> init(RestServletContext <jv>context</jv>) <jk>throws</jk> Exception;
92  	 * </p>
93  	 *
94  	 * @return The context information on this servlet.
95  	 */
96  	public synchronized RestContext getContext() {
97  		RestContext rc = context.get();
98  		if (rc == null)
99  			throw new InternalServerError("RestContext object not set on resource.");
100 		return rc;
101 	}
102 
103 	/**
104 	 * Returns the path for this resource as defined by the @Rest(path) annotation or RestContext.Builder.path(String) method
105 	 * concatenated with those on all parent classes.
106 	 *
107 	 * @return The path defined on this servlet, or an empty string if not specified.
108 	 */
109 	public synchronized String getPath() {
110 		RestContext context = this.context.get();
111 		if (context != null)
112 			return context.getFullPath();
113 		ClassInfo ci = ClassInfo.of(getClass());
114 		Value<String> path = Value.empty();
115 		ci.forEachAnnotation(Rest.class, x -> isNotEmpty(x.path()), x -> path.set(trimSlashes(x.path())));
116 		return path.orElse("");
117 	}
118 
119 	/**
120 	 * Returns the current thread-local HTTP request.
121 	 *
122 	 * @return The current thread-local HTTP request, or <jk>null</jk> if it wasn't created.
123 	 */
124 	public synchronized RestRequest getRequest() { return getContext().getLocalSession().getOpSession().getRequest(); }
125 
126 	/**
127 	 * Returns the current thread-local HTTP response.
128 	 *
129 	 * @return The current thread-local HTTP response, or <jk>null</jk> if it wasn't created.
130 	 */
131 	public synchronized RestResponse getResponse() { return getContext().getLocalSession().getOpSession().getResponse(); }
132 
133 	@Override /* Overridden from Servlet */
134 	public synchronized void init(ServletConfig servletConfig) throws ServletException {
135 		try {
136 			if (context.get() != null)
137 				return;
138 			super.init(servletConfig);
139 			context.set(RestContext.create(this.getClass(), null, servletConfig).init(() -> this).build());
140 			context.get().postInit();
141 			context.get().postInitChildFirst();
142 		} catch (ServletException e) {
143 			initException.set(e);
144 			log(SEVERE, e, "Servlet init error on class ''{0}''", ClassUtils.className(this));
145 			throw e;
146 		} catch (BasicHttpException e) {
147 			initException.set(e);
148 			log(SEVERE, e, "Servlet init error on class ''{0}''", ClassUtils.className(this));
149 		} catch (Throwable e) {
150 			initException.set(new InternalServerError(e));
151 			log(SEVERE, e, "Servlet init error on class ''{0}''", ClassUtils.className(this));
152 		}
153 	}
154 
155 	/**
156 	 * Log a message.
157 	 *
158 	 * <p>
159 	 * Subclasses can intercept the handling of these messages by overriding {@link #doLog(Level, Throwable, Supplier)}.
160 	 *
161 	 * @param level The log level.
162 	 * @param msg The message to log.
163 	 * @param args Optional {@link MessageFormat}-style arguments.
164 	 */
165 	public void log(Level level, String msg, Object...args) {
166 		doLog(level, null, () -> StringUtils.format(msg, args));
167 	}
168 
169 	/**
170 	 * Log a message.
171 	 *
172 	 * <p>
173 	 * Subclasses can intercept the handling of these messages by overriding {@link #doLog(Level, Throwable, Supplier)}.
174 	 *
175 	 * @param level The log level.
176 	 * @param cause The cause.
177 	 * @param msg The message to log.
178 	 * @param args Optional {@link MessageFormat}-style arguments.
179 	 */
180 	public void log(Level level, Throwable cause, String msg, Object...args) {
181 		doLog(level, cause, () -> StringUtils.format(msg, args));
182 	}
183 
184 	/**
185 	 * The main service method.
186 	 *
187 	 * <p>
188 	 * Subclasses can optionally override this method if they want to tailor the behavior of requests.
189 	 */
190 	@Override /* Overridden from Servlet */
191 	public void service(HttpServletRequest r1, HttpServletResponse r2) throws ServletException, InternalServerError, IOException {
192 		try {
193 			if (initException.get() != null)
194 				throw initException.get();
195 			if (context.get() == null)
196 				throw new InternalServerError(
197 					"Servlet {0} not initialized.  init(ServletConfig) was not called.  This can occur if you've overridden this method but didn't call super.init(RestConfig).", ClassUtils.className(this));
198 			getContext().execute(this, r1, r2);
199 
200 		} catch (Throwable e) {
201 			r2.sendError(SC_INTERNAL_SERVER_ERROR, e.getLocalizedMessage());
202 		}
203 	}
204 
205 	/**
206 	 * Main logger method.
207 	 *
208 	 * <p>
209 	 * The default behavior logs a message to the Java logger of the class name.
210 	 *
211 	 * <p>
212 	 * Subclasses can override this method to implement their own logger handling.
213 	 *
214 	 * @param level The log level.
215 	 * @param cause Optional throwable.
216 	 * @param msg The message to log.
217 	 */
218 	protected void doLog(Level level, Throwable cause, Supplier<String> msg) {
219 		RestContext c = context.get();
220 		Logger logger = c == null ? null : c.getLogger();
221 		if (logger == null)
222 			logger = Logger.getLogger(ClassUtils.className(this));
223 		logger.log(level, cause, msg);
224 	}
225 
226 	/**
227 	 * Sets the context object for this servlet.
228 	 *
229 	 * <p>
230 	 * This method is effectively a no-op if {@link #init(ServletConfig)} has already been called.
231 	 *
232 	 * @param context Sets the context object on this servlet.
233 	 * @throws ServletException If error occurred during initialization.
234 	 */
235 	protected void setContext(RestContext context) throws ServletException {
236 		if (this.context.get() == null) {
237 			super.init(context.getBuilder());
238 			this.context.set(context);
239 		}
240 	}
