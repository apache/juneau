33  /**
34   * Represents the headers in an HTTP request.
35   *
36   * <p>
37   * 	The {@link RequestHeaders} object is the API for accessing the headers of an HTTP request.
38   * 	It can be accessed by passing it as a parameter on your REST Java method:
39   * </p>
40   * <p class='bjava'>
41   * 	<ja>@RestPost</ja>(...)
42   * 	<jk>public</jk> Object myMethod(RequestHeaders <jv>headers</jv>) {...}
43   * </p>
44   *
45   * <h5 class='figure'>Example:</h5>
46   * <p class='bjava'>
47   * 	<ja>@RestPost</ja>(...)
48   * 	<jk>public</jk> Object myMethod(RequestHeaders <jv>headers</jv>) {
49   *
50   * 		<jc>// Add a default value.</jc>
51   * 		<jv>headers</jv>.addDefault(<js>"ETag"</js>, <jsf>DEFAULT_UUID</jsf>);
52   *
53   * 		<jc>// Get a header value as a POJO.</jc>
54   * 		UUID <jv>etag</jv> = <jv>headers</jv>.get(<js>"ETag"</js>).as(UUID.<jk>class</jk>).get();
55   *
56   * 		<jc>// Get a header as a standard HTTP part.</jc>
57   * 		ContentType <jv>contentType</jv> = <jv>headers</jv>.get(ContentType.<jk>class</jk>).orElse(ContentType.<jsf>TEXT_XML</jsf>);
58   * 	}
59   * </p>
60   *
61   * <p>
62   * 	Some important methods on this class are:
63   * </p>
64   * <ul class='javatree'>
65   * 	<li class='jc'>{@link RequestHeaders}
66   * 	<ul class='spaced-list'>
67   * 		<li>Methods for retrieving headers:
68   * 		<ul class='javatreec'>
69   * 			<li class='jm'>{@link RequestHeaders#contains(String) contains(String)}
70   * 			<li class='jm'>{@link RequestHeaders#containsAny(String...) containsAny(String...)}
71   * 			<li class='jm'>{@link RequestHeaders#get(Class) get(Class)}
72   * 			<li class='jm'>{@link RequestHeaders#get(String) get(String)}
73   * 			<li class='jm'>{@link RequestHeaders#getAll(String) getAll(String)}
74   * 			<li class='jm'>{@link RequestHeaders#getFirst(String) getFirst(String)}
75   * 			<li class='jm'>{@link RequestHeaders#getLast(String) getLast(String)}
76   * 		</ul>
77   * 		<li>Methods overridding headers:
78   * 		<ul class='javatreec'>
79   * 			<li class='jm'>{@link RequestHeaders#add(Header...) add(Header...)}
80   * 			<li class='jm'>{@link RequestHeaders#add(String, Object) add(String, Object)}
81   * 			<li class='jm'>{@link RequestHeaders#addDefault(Header...) addDefault(Header...)}
82   * 			<li class='jm'>{@link RequestHeaders#addDefault(List) addDefault(List)}
83   * 			<li class='jm'>{@link RequestHeaders#addDefault(String,String) addDefault(String,String)}
84   * 			<li class='jm'>{@link RequestHeaders#remove(String) remove(String)}
85   * 			<li class='jm'>{@link RequestHeaders#set(Header...) set(Header...)}
86   * 			<li class='jm'>{@link RequestHeaders#set(String,Object) set(String,Object)}
87   * 		</ul>
88   * 		<li>Other methods:
89   * 		<ul class='javatreec'>
90   * 			<li class='jm'>{@link RequestHeaders#copy() copy()}
91   * 			<li class='jm'>{@link RequestHeaders#isEmpty() isEmpty()}
92   * 			<li class='jm'>{@link RequestHeaders#subset(String...) subset(String...)}
93   * 		</ul>
94   * 	</ul>
95   * </ul>
96   *
97   * <p>
98   * Entries are stored in a case-insensitive map unless overridden via the constructor.
99   *
100  * <h5 class='section'>See Also:</h5><ul>
101  * 	<li class='jc'>{@link RequestHeader}
102  * 	<li class='ja'>{@link org.apache.juneau.http.annotation.Header}
103  * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HttpParts">HTTP Parts</a>
104  * </ul>
105  */
106 public class RequestHeaders extends ArrayList<RequestHeader> {
107 
108 	private static final long serialVersionUID = 1L;
109 
110 	private final RestRequest req;
111 	private boolean caseSensitive;
112 	private final VarResolverSession vs;
113 
114 	private HttpPartParserSession parser;
115 
116 	/**
117 	 * Constructor.
118 	 *
119 	 * @param req The request creating this bean.
120 	 * @param query The query parameters on the request (used for overloaded header values).
121 	 * @param caseSensitive Whether case-sensitive name matching is enabled.
122 	 */
123 	public RequestHeaders(RestRequest req, RequestQueryParams query, boolean caseSensitive) {
124 		this.req = req;
125 		this.caseSensitive = caseSensitive;
126 		this.vs = req.getVarResolverSession();
127 
128 		for (Enumeration<String> e = req.getHttpServletRequest().getHeaderNames(); e.hasMoreElements();) {
129 			String name = e.nextElement();
130 			for (Enumeration<String> ve = req.getHttpServletRequest().getHeaders(name); ve.hasMoreElements();) {
131 				add(new RequestHeader(req, name, ve.nextElement()));
132 			}
133 		}
134 
135 		// Parameters defined on the request URL overwrite existing headers.
136 		Set<String> allowedHeaderParams = req.getContext().getAllowedHeaderParams();
137 		query.forEach(p -> {
138 			String name = p.getName();
139 			String key = key(name);
140 			if (allowedHeaderParams.contains(key) || allowedHeaderParams.contains("*")) {
141 				set(name, p.getValue());
142 			}
143 		});
144 	}
145 
146 	/**
147 	 * Copy constructor.
148 	 */
149 	private RequestHeaders(RequestHeaders copyFrom) {
150 		req = copyFrom.req;
151 		caseSensitive = copyFrom.caseSensitive;
152 		parser = copyFrom.parser;
153 		addAll(copyFrom);
154 		vs = copyFrom.vs;
155 	}
156 
157 	/**
158 	 * Subset constructor.
159 	 */
160 	private RequestHeaders(RequestHeaders copyFrom, String...names) {
161 		this.req = copyFrom.req;
162 		caseSensitive = copyFrom.caseSensitive;
163 		parser = copyFrom.parser;
164 		vs = copyFrom.vs;
165 		for (String n : names)
166 			copyFrom.stream().filter(x -> eq(x.getName(), n)).forEach(this::add);
167 	}

172 	 * @param value The new value for this setting.

177 		forEach(x -> x.parser(parser));
178 		return this;

196 	/**
197 	 * Adds default entries to these headers.
198 	 *
199 	 * <p>
200 	 * Similar to {@link #set(String, Object)} but doesn't override existing values.
201 	 *
202 	 * @param pairs The default entries.  Must not be <jk>null</jk>.
203 	 * @return This object.
204 	 */
205 	public RequestHeaders addDefault(List<Header> pairs) {
206 		Utils.assertArgNotNull("pairs", pairs);
207 		for (Header p : pairs) {
208 			String name = p.getName();
209 			Stream<RequestHeader> l = stream(name);
210 			boolean hasAllBlanks = l.allMatch(x -> Utils.isEmpty(x.getValue()));
211 			if (hasAllBlanks) {
212 				removeAll(getAll(name));
213 				add(new RequestHeader(req, name, vs.resolve(p.getValue())));
214 			}
215 		}
216 		return this;
217 	}
218 
219 	/**
220 	 * Adds default entries to these headers.
221 	 *
222 	 * <p>
223 	 * Similar to {@link #set(String, Object)} but doesn't override existing values.
224 	 *
225 	 * @param pairs The default entries.  Must not be <jk>null</jk>.
226 	 * @return This object.
227 	 */
228 	public RequestHeaders addDefault(Header...pairs) {
229 		return addDefault(alist(pairs));
230 	}

232 	/**
233 	 * Adds a default entry to the request headers.
234 	 *
235 	 * @param name The name.
236 	 * @param value The value.
237 	 * @return This object.
238 	 */
239 	public RequestHeaders addDefault(String name, String value) {
240 		return addDefault(BasicStringHeader.of(name, value));
241 	}

243 	/**
244 	 * Adds a request header value.
245 	 *
246 	 * <p>
247 	 * Header is added to the end.
248 	 * <br>Existing headers with the same name are not changed.
249 	 *
250 	 * @param name The header name.  Must not be <jk>null</jk>.
251 	 * @param value The header value.  Can be <jk>null</jk>.
252 	 * @return This object.
253 	 */
254 	public RequestHeaders add(String name, Object value) {
255 		Utils.assertArgNotNull("name", name);
256 		add(new RequestHeader(req, name, Utils.s(value)).parser(parser));
257 		return this;
258 	}
259 
260 	/**
261 	 * Adds request header values.
262 	 *
263 	 * <p>
264 	 * Headers are added to the end.
265 	 * <br>Existing headers with the same name are not changed.
266 	 *
267 	 * @param headers The header objects.  Must not be <jk>null</jk>.
268 	 * @return This object.
269 	 */
270 	public RequestHeaders add(Header...headers) {
271 		Utils.assertArgNotNull("headers", headers);
272 		for (Header h : headers)
273 			if (h != null)
274 				add(h.getName(), h.getValue());
275 		return this;
276 	}

278 	/**
279 	 * Sets a request header value.
280 	 *
281 	 * <p>
282 	 * Header is added to the end.
283 	 * <br>Any previous headers with the same name are removed.
284 	 *
285 	 * @param name The header name.  Must not be <jk>null</jk>.
286 	 * @param value
287 	 * 	The header value.
288 	 * 	<br>Converted to a string using {@link Object#toString()}.
289 	 * 	<br>Can be <jk>null</jk>.
290 	 * @return This object.
291 	 */
292 	public RequestHeaders set(String name, Object value) {
293 		Utils.assertArgNotNull("name", name);
294 		set(new RequestHeader(req, name, Utils.s(value)).parser(parser));
295 		return this;
296 	}
297 
298 	/**
299 	 * Sets request header values.
300 	 *
301 	 * <p>
302 	 * Headers are added to the end.
303 	 * <br>Any previous headers with the same name are removed.
304 	 *
305 	 * @param headers The header to set.  Must not be <jk>null</jk> or contain <jk>null</jk>.
306 	 * @return This object.
307 	 */
308 	public RequestHeaders set(Header...headers) {
309 		Utils.assertArgNotNull("headers", headers);
310 		for (Header h : headers)
311 			remove(h);
312 		for (Header h : headers)
313 			add(h);
314 		return this;
315 	}

318 	 * Remove header by name.
319 	 *
320 	 * @param name The header names.  Must not be <jk>null</jk>.
321 	 * @return This object.
322 	 */
323 	public RequestHeaders remove(String name) {
324 		Utils.assertArgNotNull("name", name);
325 		removeIf(x -> eq(x.getName(), name));
326 		return this;
327 	}

344 	 * Returns <jk>true</jk> if the header with the specified name is present.

353 	/**
354 	 * Returns <jk>true</jk> if the header with any of the specified names are present.
355 	 *
356 	 * @param names The header names.  Must not be <jk>null</jk>.
357 	 * @return <jk>true</jk> if the header with any of the specified names are present.
358 	 */
359 	public boolean containsAny(String...names) {
360 		Utils.assertArgNotNull("names", names);
361 		for (String n : names)
362 			if (stream(n).findAny().isPresent())
363 				return true;
364 		return false;
365 	}

367 	/**
368 	 * Returns all headers with the specified name.
369 	 *
370 	 * @param name The header name.
371 	 * @return The list of all headers with matching names.  Never <jk>null</jk>.
372 	 */
373 	public List<RequestHeader> getAll(String name) {
374 		return stream(name).collect(toList());
375 	}

377 	/**
378 	 * Returns all headers with the specified name.
379 	 *
380 	 * @param name The header name.
381 	 * @return The stream of all headers with matching names.  Never <jk>null</jk>.
382 	 */
383 	public Stream<RequestHeader> stream(String name) {
384 		return stream().filter(x -> eq(x.getName(), name));
385 	}

387 	/**
388 	 * Returns all headers in sorted order.
389 	 *
390 	 * @return The stream of all headers in sorted order.
391 	 */
392 	public Stream<RequestHeader> getSorted() {
393 		Comparator<RequestHeader> x;
394 		if (caseSensitive)
395 			x = Comparator.comparing(RequestHeader::getName);
396 		else
397 			x = (x1,x2) -> String.CASE_INSENSITIVE_ORDER.compare(x1.getName(), x2.getName());
398 		return stream().sorted(x);
399 	}

401 	/**
402 	 * Returns all the unique header names in this list.
403 	 * @return The list of all unique header names in this list.
404 	 */
405 	public List<String> getNames() {
406 		return stream().map(RequestHeader::getName).map(x -> caseSensitive ? x : x.toLowerCase()).distinct().collect(toList());
407 	}

409 	/**
410 	 * Returns the first header with the specified name.
411 	 *
412 	 * <p>
413 	 * Note that this method never returns <jk>null</jk> and that {@link RequestHeader#isPresent()} can be used
414 	 * to test for the existence of the header.
415 	 *
416 	 * @param name The header name.  Must not be <jk>null</jk>.
417 	 * @return The header.  Never <jk>null</jk>.
418 	 */
419 	public RequestHeader getFirst(String name) {
420 		Utils.assertArgNotNull("name", name);
421 		return stream(name).findFirst().orElseGet(()->new RequestHeader(req, name, null).parser(parser));
422 	}
423 
424 	/**
425 	 * Returns the last header with the specified name.
426 	 *
427 	 * <p>
428 	 * Note that this method never returns <jk>null</jk> and that {@link RequestHeader#isPresent()} can be used
429 	 * to test for the existence of the header.
430 	 *
431 	 * @param name The header name.  Must not be <jk>null</jk>.
432 	 * @return The header.  Never <jk>null</jk>.
433 	 */
434 	public RequestHeader getLast(String name) {
435 		Utils.assertArgNotNull("name", name);
436 		Value<RequestHeader> v = Value.empty();
437 		stream(name).forEach(x -> v.set(x));
438 		return v.orElseGet(() -> new RequestHeader(req, name, null).parser(parser));
439 	}
440 
441 	/**
442 	 * Returns the condensed header with the specified name.
443 	 *
444 	 * <p>
445 	 * If multiple headers are present, they will be combined into a single comma-delimited list.
446 	 *
447 	 * @param name The header name.
448 	 * @return The header, never <jk>null</jk>.
449 	 */
450 	public RequestHeader get(String name) {
451 		List<RequestHeader> l = getAll(name);
452 		if (l.isEmpty())
453 			return new RequestHeader(req, name, null).parser(parser);
454 		if (l.size() == 1)
455 			return l.get(0);
456 		StringBuilder sb = new StringBuilder(128);
457 		for (int i = 0, j = l.size(); i < j; i++) {
458 			if (i > 0)
459 				sb.append(", ");
460 			sb.append(l.get(i).getValue());
461 		}
462 		return new RequestHeader(req, name, sb.toString()).parser(parser);
463 	}

465 	/**
466 	 * Returns the header as the specified bean type.
467 	 *
468 	 * <p>
469 	 * Type must have a name specified via the {@link org.apache.juneau.http.annotation.Header} annotation
470 	 * and a public constructor that takes in either <c>value</c> or <c>name,value</c> as strings.
471 	 *
472 	 * @param <T> The bean type to create.
473 	 * @param type The bean type to create.
474 	 * @return The bean, never <jk>null</jk>.
475 	 */
476 	public <T> Optional<T> get(Class<T> type) {
477 		ClassMeta<T> cm = req.getBeanSession().getClassMeta(type);
478 		String name = HttpParts.getName(HEADER, cm).orElseThrow(()->new BasicRuntimeException("@Header(name) not found on class {0}", className(type)));
479 		return get(name).as(type);
480 	}

495 	private String key(String name) {
496 		return caseSensitive ? name : name.toLowerCase();
497 	}
498 
499 	private boolean eq(String s1, String s2) {
500 		if (caseSensitive)
501 			return Utils.eq(s1, s2);
502 		return Utils.eqic(s1, s2);
503 	}
504 
505 	@Override /* Object */
506 	public String toString() {
507 		JsonMap m = new JsonMap();
508 		for (String n : getNames())
509 			m.put(n, get(n).asString().orElse(null));
510 		return m.asJson();
511 	}
