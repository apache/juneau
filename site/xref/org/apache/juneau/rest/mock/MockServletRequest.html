126 	public MockServletRequest applyOverrides(HttpRequest req) {
127 
128 		if (req instanceof MockRestRequest) {
129 			MockRestRequest mreq = (MockRestRequest)req;
130 			mreq.getAttributeMap().forEach(this::attribute);
131 			mreq.getRequestDispatcherMap().forEach(this::requestDispatcher);
132 			if (nn(mreq.getCharacterEncoding()))
133 				characterEncoding(mreq.getCharacterEncoding());
134 			if (nn(mreq.getProtocol()))
135 				protocol(mreq.getProtocol());
136 			if (nn(mreq.getScheme()))
137 				scheme(mreq.getScheme());
138 			if (nn(mreq.getServerName()))
139 				serverName(mreq.getServerName());
140 			if (nn(mreq.getRemoteAddr()))
141 				remoteAddr(mreq.getRemoteAddr());
142 			if (nn(mreq.getRemoteHost()))
143 				remoteHost(mreq.getRemoteHost());
144 			if (nn(mreq.getLocalName()))
145 				localName(mreq.getLocalName());
146 			if (nn(mreq.getLocalAddr()))
147 				localAddr(mreq.getLocalAddr());
148 			if (nn(mreq.getPathInfo()))
149 				pathInfo(mreq.getPathInfo());
150 			if (nn(mreq.getPathTranslated()))
151 				pathTranslated(mreq.getPathTranslated());
152 			if (nn(mreq.getContextPath()))
153 				contextPath(mreq.getContextPath());
154 			if (nn(mreq.getQueryString()))
155 				queryString(mreq.getQueryString());
156 			if (nn(mreq.getRemoteUser()))
157 				remoteUser(mreq.getRemoteUser());
158 			if (nn(mreq.getRequestedSessionId()))
159 				requestedSessionId(mreq.getRequestedSessionId());
160 			if (nn(mreq.getRequestURI()))
161 				requestURI(mreq.getRequestURI());
162 			if (nn(mreq.getServletPath()))
163 				servletPath(mreq.getServletPath());
164 			if (nn(mreq.getAuthType()))
165 				authType(mreq.getAuthType());
166 			if (nn(mreq.getServerPort()))
167 				serverPort(mreq.getServerPort());
168 			if (nn(mreq.getRemotePort()))
169 				remotePort(mreq.getRemotePort());
170 			if (nn(mreq.getLocalPort()))
171 				localPort(mreq.getLocalPort());
172 			if (nn(mreq.getLocale()))
173 				locale(mreq.getLocale());
174 			if (nn(mreq.getServletContext()))
175 				servletContext(mreq.getServletContext());
176 			if (nn(mreq.getDispatcherType()))
177 				dispatcherType(mreq.getDispatcherType());
178 			if (nn(mreq.getCookies()))

188 		return this;
189 	}

235 	/**
236 	 * Fluent setter.
237 	 *
238 	 * @param value
239 	 * 	The body of the request.
240 	 * 	<br>Can be any of the following data types:
241 	 * 	<ul>
242 	 * 		<li><code><jk>byte</jk>[]</code>
243 	 * 		<li>{@link Reader}
244 	 * 		<li>{@link InputStream}
245 	 * 		<li>{@link CharSequence}
246 	 * 	</ul>
247 	 * 	Any other types are converted to a string using the <c>toString()</c> method.
248 	 * @return This object.
249 	 */
250 	public MockServletRequest content(Object value) {
251 		try {
252 			if (value instanceof byte[])
253 				this.content = (byte[])value;
254 			else if (value instanceof Reader)
255 				this.content = readBytes((Reader)value);
256 			else if (value instanceof InputStream)
257 				this.content = readBytes((InputStream)value);
258 			else if (value instanceof CharSequence)
259 				this.content = ((CharSequence)value).toString().getBytes();
260 			else if (nn(value))
261 				this.content = value.toString().getBytes();
262 		} catch (IOException e) {
263 			throw asRuntimeException(e);
264 		}
265 		return this;
266 	}

445 	@Override /* Overridden from HttpServletRequest */
446 	public String getQueryString() {
447 		if (queryString == null) {
448 			if (queryDataMap.isEmpty())
449 				queryString = "";
450 			else {
451 				StringBuilder sb = new StringBuilder();
452 				queryDataMap.forEach((k, v) -> {
453 					if (v == null)
454 						sb.append(sb.length() == 0 ? "" : "&").append(StringUtils.urlEncode(k));
455 					else
456 						for (String v2 : v)
457 							sb.append(sb.length() == 0 ? "" : "&").append(StringUtils.urlEncode(k)).append('=').append(StringUtils.urlEncode(v2));
458 				});
459 				queryString = sb.toString();
460 			}
461 		}
462 		return isEmpty(queryString) ? null : queryString;
463 	}
