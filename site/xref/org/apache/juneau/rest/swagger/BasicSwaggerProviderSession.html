57  /**
58   * A single session of generating a Swagger document.
59   *
60   * <h5 class='section'>See Also:</h5><ul>
61   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauBeanSwagger2">juneau-bean-swagger-v2</a>
62   * </ul>
63   */
64  @SuppressWarnings("resource")
65  public class BasicSwaggerProviderSession {
66  
67  	private static Set<Integer> getCodes(List<StatusCode> la, Integer def) {
68  		Set<Integer> codes = new TreeSet<>();
69  		for (StatusCode a : la) {
70  			for (int i : a.value())
71  				codes.add(i);
72  		}
73  		if (codes.isEmpty() && def != null)
74  			codes.add(def);
75  		return codes;
76  	}
77  
78  	private static JsonMap newMap(JsonMap om) {
79  		if (om == null)
80  			return new JsonMap();
81  		return om.modifiable();
82  	}
83  
84  	private static JsonList nullIfEmpty(JsonList l) {
85  		return (l == null || l.isEmpty() ? null : l);
86  	}
87  
88  	private static JsonMap nullIfEmpty(JsonMap m) {
89  		return (m == null || m.isEmpty() ? null : m);
90  	}
91  
92  	static String joinnl(String[]...s) {
93  		for (String[] ss : s) {
94  			if (ss.length != 0)
95  				return StringUtils.joinnl(ss).trim();
96  		}
97  		return "";
98  	}

120 	 * @param messages The messages to use for finding localized strings.
121 	 * @param vr The variable resolver to use for resolving variables in the swagger.
122 	 * @param js The JSON-schema generator to use for stuff like examples.
123 	 */
124 	public BasicSwaggerProviderSession(RestContext context, Locale locale, FileFinder ff, Messages messages, VarResolverSession vr, JsonSchemaGeneratorSession js) {
125 		this.context = context;
126 		this.c = context.getResourceClass();
127 		this.rci = ClassInfo.of(c);
128 		this.ff = ff;
129 		this.mb = messages;
130 		this.vr = vr;
131 		this.js = js;
132 		this.locale = locale;
133 	}

135 	/**
136 	 * Generates the swagger.
137 	 *
138 	 * @return A new {@link Swagger} object.
139 	 * @throws Exception If an error occurred producing the Swagger.
140 	 */
141 	public Swagger getSwagger() throws Exception {
142 		// @formatter:off
143 
144 		InputStream is = ff.getStream(rci.getSimpleName() + ".json", locale).orElse(null);
145 
146 		Predicate<String> ne = Utils::isNotEmpty;
147 		Predicate<Collection<?>> nec = Utils::isNotEmpty;
148 		Predicate<Map<?,?>> nem = Utils::isNotEmpty;
149 
150 		// Load swagger JSON from classpath.
151 		JsonMap omSwagger = Json5.DEFAULT.read(is, JsonMap.class);
152 		if (omSwagger == null)
153 			omSwagger = new JsonMap();
154 
155 		// Combine it with @Rest(swagger)
156 		for (Rest rr : rci.getAnnotations(context, Rest.class)) {
157 
158 			JsonMap sInfo = omSwagger.getMap("info", true);
159 
160 			sInfo
161 				.appendIf(ne, "title",
162 					firstNonEmpty(
163 						sInfo.getString("title"),
164 						resolve(rr.title())
165 					)
166 				)
167 				.appendIf(ne, "description",
168 					firstNonEmpty(
169 						sInfo.getString("description"),
170 						resolve(rr.description())
171 					)
172 				);
173 
174 			org.apache.juneau.rest.annotation.Swagger r = rr.swagger();
175 
176 			omSwagger.append(parseMap(r.value(), "@Swagger(value) on class {0}", c));
177 
178 			if (! SwaggerAnnotation.empty(r)) {
179 				JsonMap info = omSwagger.getMap("info", true);
180 
181 				info
182 					.appendIf(ne, "title", resolve(r.title()))
183 					.appendIf(ne, "description", resolve(r.description()))
184 					.appendIf(ne, "version", resolve(r.version()))
185 					.appendIf(ne, "termsOfService", resolve(r.termsOfService()))
186 					.appendIf(nem, "contact",
187 						merge(
188 							info.getMap("contact"),
189 							toMap(r.contact(), "@Swagger(contact) on class {0}", c)
190 						)
191 					)
192 					.appendIf(nem, "license",
193 						merge(
194 							info.getMap("license"),
195 							toMap(r.license(), "@Swagger(license) on class {0}", c)
196 						)
197 					);
198 			}
199 
200 			omSwagger
201 				.appendIf(nem, "externalDocs",
202 					merge(
203 						omSwagger.getMap("externalDocs"),
204 						toMap(r.externalDocs(), "@Swagger(externalDocs) on class {0}", c)
205 					)
206 				)
207 				.appendIf(nec, "tags",
208 					merge(
209 						omSwagger.getList("tags"),
210 						toList(r.tags(), "@Swagger(tags) on class {0}", c)
211 					)
212 				);
213 		}
214 
215 		omSwagger.appendIf(nem, "externalDocs", parseMap(mb.findFirstString("externalDocs"), "Messages/externalDocs on class {0}", c));
216 
217 		JsonMap info = omSwagger.getMap("info", true);
218 
219 		info
220 			.appendIf(ne, "title", resolve(mb.findFirstString("title")))
221 			.appendIf(ne, "description", resolve(mb.findFirstString("description")))
222 			.appendIf(ne, "version", resolve(mb.findFirstString("version")))
223 			.appendIf(ne, "termsOfService", resolve(mb.findFirstString("termsOfService")))
224 			.appendIf(nem, "contact", parseMap(mb.findFirstString("contact"), "Messages/contact on class {0}", c))
225 			.appendIf(nem, "license", parseMap(mb.findFirstString("license"), "Messages/license on class {0}", c));
226 
227 		if (info.isEmpty())
228 			omSwagger.remove("info");
229 
230 		JsonList
231 			produces = omSwagger.getList("produces", true),
232 			consumes = omSwagger.getList("consumes", true);
233 		if (consumes.isEmpty())
234 			consumes.addAll(context.getConsumes());
235 		if (produces.isEmpty())
236 			produces.addAll(context.getProduces());
237 
238 		Map<String,JsonMap> tagMap = CollectionUtils.map();
239 		if (omSwagger.containsKey("tags")) {
240 			for (JsonMap om : omSwagger.getList("tags").elements(JsonMap.class)) {
241 				String name = om.getString("name");
242 				if (name == null)
243 					throw new SwaggerException(null, "Tag definition found without name in swagger JSON.");
244 				tagMap.put(name, om);
245 			}
246 		}
247 
248 		String s = mb.findFirstString("tags");
249 		if (s != null) {
250 			for (JsonMap m : parseListOrCdl(s, "Messages/tags on class {0}", c).elements(JsonMap.class)) {
251 				String name = m.getString("name");
252 				if (name == null)
253 					throw new SwaggerException(null, "Tag definition found without name in resource bundle on class {0}", c);
254 				if (tagMap.containsKey(name))
255 					tagMap.get(name).putAll(m);
256 				else
257 					tagMap.put(name, m);
258 			}
259 		}
260 
261 		// Load our existing bean definitions into our session.
262 		JsonMap definitions = omSwagger.getMap("definitions", true);
263 		for (String defId : definitions.keySet())
264 			js.addBeanDef(defId, new JsonMap(definitions.getMap(defId)));
265 
266 		// Iterate through all the @RestOp methods.
267 		for (RestOpContext sm : context.getRestOperations().getOpContexts()) {
268 
269 			BeanSession bs = sm.getBeanContext().getSession();
270 
271 			Method m = sm.getJavaMethod();
272 			MethodInfo mi = MethodInfo.of(m);
273 			AnnotationList al = mi.getAnnotationList(REST_OP_GROUP);
274 			String mn = m.getName();
275 
276 			// Get the operation from the existing swagger so far.
277 			JsonMap op = getOperation(omSwagger, sm.getPathPattern(), sm.getHttpMethod().toLowerCase());
278 
279 			// Add @RestOp(swagger)
280 			Value<OpSwagger> _ms = Value.empty();
281 			al.forEachValue(OpSwagger.class, "swagger", OpSwaggerAnnotation::notEmpty, x -> _ms.set(x));
282 			OpSwagger ms = _ms.orElseGet(() -> OpSwaggerAnnotation.create().build());
283 
284 			op.append(parseMap(ms.value(), "@OpSwagger(value) on class {0} method {1}", c, m));
285 			op.appendIf(ne, "operationId",
286 				firstNonEmpty(
287 					resolve(ms.operationId()),
288 					op.getString("operationId"),
289 					mn
290 				)
291 			);
292 
293 			Value<String> _summary = Value.empty();
294 			al.forEachValue(String.class, "summary", NOT_EMPTY, x -> _summary.set(x));
295 			op.appendIf(ne, "summary",
296 				firstNonEmpty(
297 					resolve(ms.summary()),
298 					resolve(mb.findFirstString(mn + ".summary")),
299 					op.getString("summary"),
300 					resolve(_summary.orElse(null))
301 				)
302 			);
303 
304 			Value<String[]> _description = Value.empty();
305 			al.forEachValue(String[].class, "description", x -> x.length > 0, x -> _description.set(x));
306 			op.appendIf(ne, "description",
307 				firstNonEmpty(
308 					resolve(ms.description()),
309 					resolve(mb.findFirstString(mn + ".description")),
310 					op.getString("description"),
311 					resolve(_description.orElse(new String[0]))
312 				)
313 			);
314 			op.appendIf(ne, "deprecated",
315 				firstNonEmpty(
316 					resolve(ms.deprecated()),
317 					(m.getAnnotation(Deprecated.class) != null || m.getDeclaringClass().getAnnotation(Deprecated.class) != null) ? "true" : null
318 				)
319 			);
320 			op.appendIf(nec, "tags",
321 				merge(
322 					parseListOrCdl(mb.findFirstString(mn + ".tags"), "Messages/tags on class {0} method {1}", c, m),
323 					parseListOrCdl(ms.tags(), "@OpSwagger(tags) on class {0} method {1}", c, m)
324 				)
325 			);
326 			op.appendIf(nec, "schemes",
327 				merge(
328 					parseListOrCdl(mb.findFirstString(mn + ".schemes"), "Messages/schemes on class {0} method {1}", c, m),
329 					parseListOrCdl(ms.schemes(), "@OpSwagger(schemes) on class {0} method {1}", c, m)
330 				)
331 			);
332 			op.appendIf(nec, "consumes",
333 				firstNonEmpty(
334 					parseListOrCdl(mb.findFirstString(mn + ".consumes"), "Messages/consumes on class {0} method {1}", c, m),
335 					parseListOrCdl(ms.consumes(), "@OpSwagger(consumes) on class {0} method {1}", c, m)
336 				)
337 			);
338 			op.appendIf(nec, "produces",
339 				firstNonEmpty(
340 					parseListOrCdl(mb.findFirstString(mn + ".produces"), "Messages/produces on class {0} method {1}", c, m),
341 					parseListOrCdl(ms.produces(), "@OpSwagger(produces) on class {0} method {1}", c, m)
342 				)
343 			);
344 			op.appendIf(nec, "parameters",
345 				merge(
346 					parseList(mb.findFirstString(mn + ".parameters"), "Messages/parameters on class {0} method {1}", c, m),
347 					parseList(ms.parameters(), "@OpSwagger(parameters) on class {0} method {1}", c, m)
348 				)
349 			);
350 			op.appendIf(nem, "responses",
351 				merge(
352 					parseMap(mb.findFirstString(mn + ".responses"), "Messages/responses on class {0} method {1}", c, m),
353 					parseMap(ms.responses(), "@OpSwagger(responses) on class {0} method {1}", c, m)
354 				)
355 			);
356 			op.appendIf(nem, "externalDocs",
357 				merge(
358 					op.getMap("externalDocs"),
359 					parseMap(mb.findFirstString(mn + ".externalDocs"), "Messages/externalDocs on class {0} method {1}", c, m),
360 					toMap(ms.externalDocs(), "@OpSwagger(externalDocs) on class {0} method {1}", c, m)
361 				)
362 			);
363 
364 			if (op.containsKey("tags"))
365 				for (String tag : op.getList("tags").elements(String.class))
366 					if (! tagMap.containsKey(tag))
367 						tagMap.put(tag, JsonMap.of("name", tag));
368 
369 			JsonMap paramMap = new JsonMap();
370 			if (op.containsKey("parameters"))
371 				for (JsonMap param : op.getList("parameters").elements(JsonMap.class))
372 					paramMap.put(param.getString("in") + '.' + ("body".equals(param.getString("in")) ? "body" : param.getString("name")), param);
373 
374 			// Finally, look for parameters defined on method.
375 			for (ParamInfo mpi : mi.getParams()) {
376 
377 				ClassInfo pt = mpi.getParameterType();
378 				Type type = pt.innerType();
379 
380 				if (mpi.hasAnnotation(Content.class) || pt.hasAnnotation(Content.class)) {
381 					JsonMap param = paramMap.getMap(BODY + ".body", true).append("in", BODY);
382 					JsonMap schema = getSchema(param.getMap("schema"), type, bs);
383 					mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(schema, x));
384 					mpi.forEachAnnotation(Content.class, x -> true, x -> merge(schema, x.schema()));
385 					pushupSchemaFields(BODY, param, schema);
386 					param.appendIf(nem, "schema", schema);
387 					param.putIfAbsent("required", true);
388 					addBodyExamples(sm, param, false, type, locale);
389 
390 				} else if (mpi.hasAnnotation(Query.class) || pt.hasAnnotation(Query.class)) {
391 					String name = QueryAnnotation.findName(mpi).orElse(null);
392 					JsonMap param = paramMap.getMap(QUERY + "." + name, true).append("name", name).append("in", QUERY);
393 					mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(param, x));
394 					mpi.forEachAnnotation(Query.class, x -> true, x -> merge(param, x.schema()));
395 					pushupSchemaFields(QUERY, param, getSchema(param.getMap("schema"), type, bs));
396 					addParamExample(sm, param, QUERY, type);
397 
398 				} else if (mpi.hasAnnotation(FormData.class) || pt.hasAnnotation(FormData.class)) {
399 					String name = FormDataAnnotation.findName(mpi).orElse(null);
400 					JsonMap param = paramMap.getMap(FORM_DATA + "." + name, true).append("name", name).append("in", FORM_DATA);
401 					mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(param, x));
402 					mpi.forEachAnnotation(FormData.class, x -> true, x -> merge(param, x.schema()));
403 					pushupSchemaFields(FORM_DATA, param, getSchema(param.getMap("schema"), type, bs));
404 					addParamExample(sm, param, FORM_DATA, type);
405 
406 				} else if (mpi.hasAnnotation(Header.class) || pt.hasAnnotation(Header.class)) {
407 					String name = HeaderAnnotation.findName(mpi).orElse(null);
408 					JsonMap param = paramMap.getMap(HEADER + "." + name, true).append("name", name).append("in", HEADER);
409 					mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(param, x));
410 					mpi.forEachAnnotation(Header.class, x -> true, x -> merge(param, x.schema()));
411 					pushupSchemaFields(HEADER, param, getSchema(param.getMap("schema"), type, bs));
412 					addParamExample(sm, param, HEADER, type);
413 
414 				} else if (mpi.hasAnnotation(Path.class) || pt.hasAnnotation(Path.class)) {
415 					String name = PathAnnotation.findName(mpi).orElse(null);
416 					JsonMap param = paramMap.getMap(PATH + "." + name, true).append("name", name).append("in", PATH);
417 					mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(param, x));
418 					mpi.forEachAnnotation(Path.class, x -> true, x -> merge(param, x.schema()));
419 					pushupSchemaFields(PATH, param, getSchema(param.getMap("schema"), type, bs));
420 					addParamExample(sm, param, PATH, type);
421 					param.putIfAbsent("required", true);
422 				}
423 			}
424 
425 			if (! paramMap.isEmpty())
426 				op.put("parameters", paramMap.values());
427 
428 			JsonMap responses = op.getMap("responses", true);
429 
430 			for (ClassInfo eci : mi.getExceptionTypes()) {
431 				if (eci.hasAnnotation(Response.class)) {
432 					List<Response> la = eci.getAnnotations(context, Response.class);
433 					List<StatusCode> la2 = eci.getAnnotations(context, StatusCode.class);
434 					Set<Integer> codes = getCodes(la2, 500);
435 					for (Response a : la) {
436 						for (Integer code : codes) {
437 							JsonMap om = responses.getMap(String.valueOf(code), true);
438 							merge(om, a);
439 							JsonMap schema = getSchema(om.getMap("schema"), m.getGenericReturnType(), bs);
440 							eci.forEachAnnotation(Schema.class, x -> true, x -> merge(schema, x));
441 							pushupSchemaFields(RESPONSE, om, schema);
442 							om.appendIf(nem, "schema", schema);
443 						}
444 					}
445 					List<MethodInfo> methods = eci.getMethods();
446 					for (int i = methods.size() - 1; i >= 0; i--) {
447 						MethodInfo ecmi = methods.get(i);
448 						Header a = ecmi.getAnnotation(Header.class);
449 						if (a == null)
450 							a = ecmi.getReturnType().unwrap(Value.class, Optional.class).getAnnotation(Header.class);
451 						if (a != null && ! isMulti(a)) {
452 							String ha = a.name();
453 							for (Integer code : codes) {
454 								JsonMap header = responses.getMap(String.valueOf(code), true).getMap("headers", true).getMap(ha, true);
455 								ecmi.forEachAnnotation(context, Schema.class, x -> true, x -> merge(header, x));
456 								ecmi.getReturnType().unwrap(Value.class, Optional.class).forEachAnnotation(Schema.class, x -> true, x -> merge(header, x));
457 								pushupSchemaFields(RESPONSE_HEADER, header, getSchema(header.getMap("schema"), ecmi.getReturnType().unwrap(Value.class, Optional.class).innerType(), bs));
458 							}
459 						}
460 					}
461 				}
462 			}
463 
464 			if (mi.hasAnnotation(Response.class) || mi.getReturnType().unwrap(Value.class, Optional.class).hasAnnotation(Response.class)) {
465 				List<Response> la = list();
466 				mi.forEachAnnotation(context, Response.class, x -> true, x -> la.add(x));
467 				List<StatusCode> la2 = list();
468 				mi.forEachAnnotation(context, StatusCode.class, x -> true, x -> la2.add(x));
469 				Set<Integer> codes = getCodes(la2, 200);
470 				for (Response a : la) {
471 					for (Integer code : codes) {
472 						JsonMap om = responses.getMap(String.valueOf(code), true);
473 						merge(om, a);
474 						JsonMap schema = getSchema(om.getMap("schema"), m.getGenericReturnType(), bs);
475 						mi.forEachAnnotation(context, Schema.class, x -> true, x -> merge(schema, x));
476 						pushupSchemaFields(RESPONSE, om, schema);
477 						om.appendIf(nem, "schema", schema);
478 						addBodyExamples(sm, om, true, m.getGenericReturnType(), locale);
479 					}
480 				}
481 				if (mi.getReturnType().hasAnnotation(Response.class)) {
482 					List<MethodInfo> methods = mi.getReturnType().getMethods();
483 					for (int i = methods.size() - 1; i >= 0; i--) {
484 						MethodInfo ecmi = methods.get(i);
485 						if (ecmi.hasAnnotation(Header.class)) {
486 							Header a = ecmi.getAnnotation(Header.class);
487 							String ha = a.name();
488 							if (! isMulti(a)) {
489 								for (Integer code : codes) {
490 									JsonMap header = responses.getMap(String.valueOf(code), true).getMap("headers", true).getMap(ha, true);
491 									ecmi.forEachAnnotation(context, Schema.class, x -> true, x -> merge(header, x));
492 									ecmi.getReturnType().unwrap(Value.class, Optional.class).forEachAnnotation(Schema.class, x -> true, x -> merge(header, x));
493 									merge(header, a.schema());
494 									pushupSchemaFields(RESPONSE_HEADER, header, getSchema(header, ecmi.getReturnType().innerType(), bs));
495 								}
496 							}
497 						}
498 					}
499 				}
500 			} else if (m.getGenericReturnType() != void.class) {
501 				JsonMap om = responses.getMap("200", true);
502 				ClassInfo pt2 = ClassInfo.of(m.getGenericReturnType());
503 				JsonMap schema = getSchema(om.getMap("schema"), m.getGenericReturnType(), bs);
504 				pt2.forEachAnnotation(Schema.class, x -> true, x -> merge(schema, x));
505 				pushupSchemaFields(RESPONSE, om, schema);
506 				om.appendIf(nem, "schema", schema);
507 				addBodyExamples(sm, om, true, m.getGenericReturnType(), locale);
508 			}
509 
510 			// Finally, look for Value @Header parameters defined on method.
511 			for (ParamInfo mpi : mi.getParams()) {
512 
513 				ClassInfo pt = mpi.getParameterType();
514 
515 				if (pt.is(Value.class) && (mpi.hasAnnotation(Header.class) || pt.hasAnnotation(Header.class))) {
516 					List<Header> la = list();
517 					mpi.forEachAnnotation(Header.class, x -> true, x -> la.add(x));
518 					pt.forEachAnnotation(Header.class, x -> true, x -> la.add(x));
519 					List<StatusCode> la2 = list();
520 					mpi.forEachAnnotation(StatusCode.class, x -> true, x -> la2.add(x));
521 					pt.forEachAnnotation(StatusCode.class, x -> true, x -> la2.add(x));
522 					Set<Integer> codes = getCodes(la2, 200);
523 					String name = HeaderAnnotation.findName(mpi).orElse(null);
524 					Type type = Value.unwrap(mpi.getParameterType().innerType());
525 					for (Header a : la) {
526 						if (! isMulti(a)) {
527 							for (Integer code : codes) {
528 								JsonMap header = responses.getMap(String.valueOf(code), true).getMap("headers", true).getMap(name, true);
529 								mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(header, x));
530 								merge(header, a.schema());
531 								pushupSchemaFields(RESPONSE_HEADER, header, getSchema(header, type, bs));
532 							}
533 						}
534 					}
535 
536 				} else if (mpi.hasAnnotation(Response.class) || pt.hasAnnotation(Response.class)) {
537 					List<Response> la = list();
538 					mpi.forEachAnnotation(Response.class, x -> true, x -> la.add(x));
539 					pt.forEachAnnotation(Response.class, x -> true, x -> la.add(x));
540 					List<StatusCode> la2 = list();
541 					mpi.forEachAnnotation(StatusCode.class, x -> true, x -> la2.add(x));
542 					pt.forEachAnnotation(StatusCode.class, x -> true, x -> la2.add(x));
543 					Set<Integer> codes = getCodes(la2, 200);
544 					Type type = Value.unwrap(mpi.getParameterType().innerType());
545 					for (Response a : la) {
546 						for (Integer code : codes) {
547 							JsonMap om = responses.getMap(String.valueOf(code), true);
548 							merge(om, a);
549 							JsonMap schema = getSchema(om.getMap("schema"), type, bs);
550 							mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(schema, x));
551 							la.forEach(x -> merge(schema, x.schema()));
552 							pushupSchemaFields(RESPONSE, om, schema);
553 							om.appendIf(nem, "schema", schema);
554 						}
555 					}
556 				}
557 			}
558 
559 			// Add default response descriptions.
560 			for (Map.Entry<String,Object> e : responses.entrySet()) {
561 				String key = e.getKey();
562 				JsonMap val = responses.getMap(key);
563 				if (StringUtils.isDecimal(key))
564 					val.appendIfAbsentIf(ne, "description", RestUtils.getHttpResponseText(Integer.parseInt(key)));
565 			}
566 
567 			if (responses.isEmpty())
568 				op.remove("responses");
569 			else
570 				op.put("responses", new TreeMap<>(responses));
571 
572 			if (! op.containsKey("consumes")) {
573 				List<MediaType> mConsumes = sm.getSupportedContentTypes();
574 				if (! mConsumes.equals(consumes))
575 					op.put("consumes", mConsumes);
576 			}
577 
578 			if (! op.containsKey("produces")) {
579 				List<MediaType> mProduces = sm.getSupportedAcceptTypes();
580 				if (! mProduces.equals(produces))
581 					op.put("produces", mProduces);
582 			}
583 		}
584 
585 		if (js.getBeanDefs() != null)
586 			for (Map.Entry<String,JsonMap> e : js.getBeanDefs().entrySet())
587 				definitions.put(e.getKey(), fixSwaggerExtensions(e.getValue()));
588 		if (definitions.isEmpty())
589 			omSwagger.remove("definitions");
590 
591 		if (! tagMap.isEmpty())
592 			omSwagger.put("tags", tagMap.values());
593 
594 		if (consumes.isEmpty())
595 			omSwagger.remove("consumes");
596 		if (produces.isEmpty())
597 			omSwagger.remove("produces");
598 
599 		try {
600 			String swaggerJson = Json5Serializer.DEFAULT_READABLE.toString(omSwagger);
601 			return jp.parse(swaggerJson, Swagger.class);
602 		} catch (Exception e) {
603 			throw new ServletException("Error detected in swagger.", e);
604 		}
605 		// @formatter:on
606 	}

608 	private void addBodyExamples(RestOpContext sm, JsonMap piri, boolean response, Type type, Locale locale) throws Exception {
609 
610 		String sex = piri.getString("example");
611 
612 		if (sex == null) {
613 			JsonMap schema = resolveRef(piri.getMap("schema"));
614 			if (schema != null)
615 				sex = schema.getString("example", schema.getString("example"));
616 		}

618 		if (Utils.isEmpty(sex))
619 			return;
620 
621 		Object example = null;
622 		if (isJson(sex)) {
623 			example = jp.parse(sex, type);
624 		} else {
625 			ClassMeta<?> cm = js.getClassMeta(type);
626 			if (cm.hasStringMutater()) {

629 		}

642 				if (s2 != null) {
643 					try {
644 						// @formatter:off
645 						String eVal = s2
646 							.createSession()
647 							.locale(locale)
648 							.mediaType(mt)

658 			}
659 		}

985 			return (JsonMap)o;
986 		throw new SwaggerException(null, "Unexpected data type ''{0}''.  Expected JsonMap or String.", ClassUtils2.className(o));
