60  /**
61   * A single session of generating a Swagger document.
62   *
63   * <h5 class='section'>See Also:</h5><ul>
64   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauBeanSwagger2">juneau-bean-swagger-v2</a>
65   * </ul>
66   */
67  @SuppressWarnings("resource")
68  public class BasicSwaggerProviderSession {
69  
70  	private static Set<Integer> getCodes(List<StatusCode> la, Integer def) {
71  		var codes = new TreeSet<Integer>();
72  		for (var a : la) {
73  			for (int i : a.value())
74  				codes.add(i);
75  		}
76  		if (codes.isEmpty() && nn(def))
77  			codes.add(def);
78  		return codes;
79  	}
80  
81  	private static JsonMap newMap(JsonMap om) {
82  		if (om == null)
83  			return new JsonMap();
84  		return om.modifiable();
85  	}
86  
87  	private static JsonList nullIfEmpty(JsonList l) {
88  		return (l == null || l.isEmpty() ? null : l);
89  	}
90  
91  	private static JsonMap nullIfEmpty(JsonMap m) {
92  		return (m == null || m.isEmpty() ? null : m);
93  	}
94  
95  	static String joinnl(String[]...s) {
96  		for (var ss : s) {
97  			if (ss.length != 0)
98  				return StringUtils.joinnl(ss).trim();
99  		}
100 		return "";
101 	}

127 	public BasicSwaggerProviderSession(RestContext context, Locale locale, FileFinder ff, Messages messages, VarResolverSession vr, JsonSchemaGeneratorSession js) {
128 		this.context = context;
129 		this.c = context.getResourceClass();
130 		this.rci = ClassInfo.of(c);
131 		this.ff = ff;
132 		this.mb = messages;
133 		this.vr = vr;
134 		this.js = js;
135 		this.locale = locale;
136 	}

138 	/**
139 	 * Generates the swagger.
140 	 *
141 	 * @return A new {@link Swagger} object.
142 	 * @throws Exception If an error occurred producing the Swagger.
143 	 */
144 	public Swagger getSwagger() throws Exception {
145 		// @formatter:off
146 
147 		InputStream is = ff.getStream(rci.getNameSimple() + ".json", locale).orElse(null);
148 
149 		Predicate<String> ne = Utils::isNotEmpty;
150 		Predicate<Collection<?>> nec = Utils::isNotEmpty;
151 		Predicate<Map<?,?>> nem = Utils::isNotEmpty;
152 
153 		// Load swagger JSON from classpath.
154 		JsonMap omSwagger = Json5.DEFAULT.read(is, JsonMap.class);
155 		if (omSwagger == null)
156 			omSwagger = new JsonMap();
157 
158 		// Combine it with @Rest(swagger)
159 		for (var rr : rci.getAnnotations(context, Rest.class)) {
160 
161 			JsonMap sInfo = omSwagger.getMap("info", true);
162 
163 			sInfo
164 				.appendIf(ne, "title",
165 					firstNonEmpty(
166 						sInfo.getString("title"),
167 						resolve(rr.title())
168 					)
169 				)
170 				.appendIf(ne, "description",
171 					firstNonEmpty(
172 						sInfo.getString("description"),
173 						resolve(rr.description())
174 					)
175 				);
176 
177 			org.apache.juneau.rest.annotation.Swagger r = rr.swagger();
178 
179 			omSwagger.append(parseMap(r.value(), "@Swagger(value) on class {0}", c));
180 
181 			if (! SwaggerAnnotation.empty(r)) {
182 				JsonMap info = omSwagger.getMap("info", true);
183 
184 				info
185 					.appendIf(ne, "title", resolve(r.title()))
186 					.appendIf(ne, "description", resolve(r.description()))
187 					.appendIf(ne, "version", resolve(r.version()))
188 					.appendIf(ne, "termsOfService", resolve(r.termsOfService()))
189 					.appendIf(nem, "contact",
190 						merge(
191 							info.getMap("contact"),
192 							toMap(r.contact(), "@Swagger(contact) on class {0}", c)
193 						)
194 					)
195 					.appendIf(nem, "license",
196 						merge(
197 							info.getMap("license"),
198 							toMap(r.license(), "@Swagger(license) on class {0}", c)
199 						)
200 					);
201 			}
202 
203 			omSwagger
204 				.appendIf(nem, "externalDocs",
205 					merge(
206 						omSwagger.getMap("externalDocs"),
207 						toMap(r.externalDocs(), "@Swagger(externalDocs) on class {0}", c)
208 					)
209 				)
210 				.appendIf(nec, "tags",
211 					merge(
212 						omSwagger.getList("tags"),
213 						toList(r.tags(), "@Swagger(tags) on class {0}", c)
214 					)
215 				);
216 		}
217 
218 		omSwagger.appendIf(nem, "externalDocs", parseMap(mb.findFirstString("externalDocs"), "Messages/externalDocs on class {0}", c));
219 
220 		JsonMap info = omSwagger.getMap("info", true);
221 
222 		info
223 			.appendIf(ne, "title", resolve(mb.findFirstString("title")))
224 			.appendIf(ne, "description", resolve(mb.findFirstString("description")))
225 			.appendIf(ne, "version", resolve(mb.findFirstString("version")))
226 			.appendIf(ne, "termsOfService", resolve(mb.findFirstString("termsOfService")))
227 			.appendIf(nem, "contact", parseMap(mb.findFirstString("contact"), "Messages/contact on class {0}", c))
228 			.appendIf(nem, "license", parseMap(mb.findFirstString("license"), "Messages/license on class {0}", c));
229 
230 		if (info.isEmpty())
231 			omSwagger.remove("info");
232 
233 		JsonList
234 			produces = omSwagger.getList("produces", true),
235 			consumes = omSwagger.getList("consumes", true);
236 		if (consumes.isEmpty())
237 			consumes.addAll(context.getConsumes());
238 		if (produces.isEmpty())
239 			produces.addAll(context.getProduces());
240 
241 		Map<String,JsonMap> tagMap = map();
242 		if (omSwagger.containsKey("tags")) {
243 			for (var om : omSwagger.getList("tags").elements(JsonMap.class)) {
244 				String name = om.getString("name");
245 				if (name == null)
246 					throw new SwaggerException(null, "Tag definition found without name in swagger JSON.");
247 				tagMap.put(name, om);
248 			}
249 		}
250 
251 		String s = mb.findFirstString("tags");
252 		if (nn(s)) {
253 			for (var m : parseListOrCdl(s, "Messages/tags on class {0}", c).elements(JsonMap.class)) {
254 				String name = m.getString("name");
255 				if (name == null)
256 					throw new SwaggerException(null, "Tag definition found without name in resource bundle on class {0}", c);
257 				if (tagMap.containsKey(name))
258 					tagMap.get(name).putAll(m);
259 				else
260 					tagMap.put(name, m);
261 			}
262 		}
263 
264 		// Load our existing bean definitions into our session.
265 		JsonMap definitions = omSwagger.getMap("definitions", true);
266 		for (var defId : definitions.keySet())
267 			js.addBeanDef(defId, new JsonMap(definitions.getMap(defId)));
268 
269 		// Iterate through all the @RestOp methods.
270 		for (var sm : context.getRestOperations().getOpContexts()) {
271 
272 			BeanSession bs = sm.getBeanContext().getSession();
273 
274 			Method m = sm.getJavaMethod();
275 			var mi = MethodInfo.of(m);
276 			AnnotationList al = rstream(mi.getAllAnnotationInfos()).filter(REST_OP_GROUP).collect(Collectors.toCollection(AnnotationList::new));
277 			String mn = m.getName();
278 
279 			// Get the operation from the existing swagger so far.
280 			JsonMap op = getOperation(omSwagger, sm.getPathPattern(), sm.getHttpMethod().toLowerCase());
281 
282 			// Add @RestOp(swagger)
283 			Value<OpSwagger> _ms = Value.empty();
284 			al.forEachValue(OpSwagger.class, "swagger", OpSwaggerAnnotation::notEmpty, x -> _ms.set(x));
285 			OpSwagger ms = _ms.orElseGet(() -> OpSwaggerAnnotation.create().build());
286 
287 			op.append(parseMap(ms.value(), "@OpSwagger(value) on class {0} method {1}", c, m));
288 			op.appendIf(ne, "operationId",
289 				firstNonEmpty(
290 					resolve(ms.operationId()),
291 					op.getString("operationId"),
292 					mn
293 				)
294 			);
295 
296 			Value<String> _summary = Value.empty();
297 			al.forEachValue(String.class, "summary", NOT_EMPTY, x -> _summary.set(x));
298 			op.appendIf(ne, "summary",
299 				firstNonEmpty(
300 					resolve(ms.summary()),
301 					resolve(mb.findFirstString(mn + ".summary")),
302 					op.getString("summary"),
303 					resolve(_summary.orElse(null))
304 				)
305 			);
306 
307 			Value<String[]> _description = Value.empty();
308 			al.forEachValue(String[].class, "description", x -> x.length > 0, x -> _description.set(x));
309 			op.appendIf(ne, "description",
310 				firstNonEmpty(
311 					resolve(ms.description()),
312 					resolve(mb.findFirstString(mn + ".description")),
313 					op.getString("description"),
314 					resolve(_description.orElse(new String[0]))
315 				)
316 			);
317 			op.appendIf(ne, "deprecated",
318 				firstNonEmpty(
319 					resolve(ms.deprecated()),
320 					(nn(m.getAnnotation(Deprecated.class)) || nn(ClassInfo.of(m.getDeclaringClass()).getAnnotationInfos(Deprecated.class).findFirst().map(AnnotationInfo::inner).orElse(null))) ? "true" : null
321 				)
322 			);
323 			op.appendIf(nec, "tags",
324 				merge(
325 					parseListOrCdl(mb.findFirstString(mn + ".tags"), "Messages/tags on class {0} method {1}", c, m),
326 					parseListOrCdl(ms.tags(), "@OpSwagger(tags) on class {0} method {1}", c, m)
327 				)
328 			);
329 			op.appendIf(nec, "schemes",
330 				merge(
331 					parseListOrCdl(mb.findFirstString(mn + ".schemes"), "Messages/schemes on class {0} method {1}", c, m),
332 					parseListOrCdl(ms.schemes(), "@OpSwagger(schemes) on class {0} method {1}", c, m)
333 				)
334 			);
335 			op.appendIf(nec, "consumes",
336 				firstNonEmpty(
337 					parseListOrCdl(mb.findFirstString(mn + ".consumes"), "Messages/consumes on class {0} method {1}", c, m),
338 					parseListOrCdl(ms.consumes(), "@OpSwagger(consumes) on class {0} method {1}", c, m)
339 				)
340 			);
341 			op.appendIf(nec, "produces",
342 				firstNonEmpty(
343 					parseListOrCdl(mb.findFirstString(mn + ".produces"), "Messages/produces on class {0} method {1}", c, m),
344 					parseListOrCdl(ms.produces(), "@OpSwagger(produces) on class {0} method {1}", c, m)
345 				)
346 			);
347 			op.appendIf(nec, "parameters",
348 				merge(
349 					parseList(mb.findFirstString(mn + ".parameters"), "Messages/parameters on class {0} method {1}", c, m),
350 					parseList(ms.parameters(), "@OpSwagger(parameters) on class {0} method {1}", c, m)
351 				)
352 			);
353 			op.appendIf(nem, "responses",
354 				merge(
355 					parseMap(mb.findFirstString(mn + ".responses"), "Messages/responses on class {0} method {1}", c, m),
356 					parseMap(ms.responses(), "@OpSwagger(responses) on class {0} method {1}", c, m)
357 				)
358 			);
359 			op.appendIf(nem, "externalDocs",
360 				merge(
361 					op.getMap("externalDocs"),
362 					parseMap(mb.findFirstString(mn + ".externalDocs"), "Messages/externalDocs on class {0} method {1}", c, m),
363 					toMap(ms.externalDocs(), "@OpSwagger(externalDocs) on class {0} method {1}", c, m)
364 				)
365 			);
366 
367 			if (op.containsKey("tags"))
368 				for (var tag : op.getList("tags").elements(String.class))
369 					if (! tagMap.containsKey(tag))
370 						tagMap.put(tag, JsonMap.of("name", tag));
371 
372 			var paramMap = new JsonMap();
373 			if (op.containsKey("parameters"))
374 				for (var param : op.getList("parameters").elements(JsonMap.class))
375 					paramMap.put(param.getString("in") + '.' + ("body".equals(param.getString("in")) ? "body" : param.getString("name")), param);
376 
377 			// Finally, look for parameters defined on method.
378 			for (var mpi : mi.getParameters()) {
379 
380 				ClassInfo pt = mpi.getParameterType();
381 				Type type = pt.innerType();
382 
383 				if (mpi.hasAnnotation(Content.class) || pt.hasAnnotation(Content.class)) {
384 					JsonMap param = paramMap.getMap(BODY + ".body", true).append("in", BODY);
385 					JsonMap schema = getSchema(param.getMap("schema"), type, bs);
386 					mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(schema, x));
387 					mpi.forEachAnnotation(Content.class, x -> true, x -> merge(schema, x.schema()));
388 					pushupSchemaFields(BODY, param, schema);
389 					param.appendIf(nem, "schema", schema);
390 					param.putIfAbsent("required", true);
391 					addBodyExamples(sm, param, false, type, locale);
392 
393 				} else if (mpi.hasAnnotation(Query.class) || pt.hasAnnotation(Query.class)) {
394 					String name = QueryAnnotation.findName(mpi).orElse(null);
395 					JsonMap param = paramMap.getMap(QUERY + "." + name, true).append("name", name).append("in", QUERY);
396 					mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(param, x));
397 					mpi.forEachAnnotation(Query.class, x -> true, x -> merge(param, x.schema()));
398 					pushupSchemaFields(QUERY, param, getSchema(param.getMap("schema"), type, bs));
399 					addParamExample(sm, param, QUERY, type);
400 
401 				} else if (mpi.hasAnnotation(FormData.class) || pt.hasAnnotation(FormData.class)) {
402 					String name = FormDataAnnotation.findName(mpi).orElse(null);
403 					JsonMap param = paramMap.getMap(FORM_DATA + "." + name, true).append("name", name).append("in", FORM_DATA);
404 					mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(param, x));
405 					mpi.forEachAnnotation(FormData.class, x -> true, x -> merge(param, x.schema()));
406 					pushupSchemaFields(FORM_DATA, param, getSchema(param.getMap("schema"), type, bs));
407 					addParamExample(sm, param, FORM_DATA, type);
408 
409 				} else if (mpi.hasAnnotation(Header.class) || pt.hasAnnotation(Header.class)) {
410 					String name = HeaderAnnotation.findName(mpi).orElse(null);
411 					JsonMap param = paramMap.getMap(HEADER + "." + name, true).append("name", name).append("in", HEADER);
412 					mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(param, x));
413 					mpi.forEachAnnotation(Header.class, x -> true, x -> merge(param, x.schema()));
414 					pushupSchemaFields(HEADER, param, getSchema(param.getMap("schema"), type, bs));
415 					addParamExample(sm, param, HEADER, type);
416 
417 				} else if (mpi.hasAnnotation(Path.class) || pt.hasAnnotation(Path.class)) {
418 					String name = PathAnnotation.findName(mpi).orElse(null);
419 					JsonMap param = paramMap.getMap(PATH + "." + name, true).append("name", name).append("in", PATH);
420 					mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(param, x));
421 					mpi.forEachAnnotation(Path.class, x -> true, x -> merge(param, x.schema()));
422 					pushupSchemaFields(PATH, param, getSchema(param.getMap("schema"), type, bs));
423 					addParamExample(sm, param, PATH, type);
424 					param.putIfAbsent("required", true);
425 				}
426 			}
427 
428 			if (! paramMap.isEmpty())
429 				op.put("parameters", paramMap.values());
430 
431 			JsonMap responses = op.getMap("responses", true);
432 
433 			for (var eci : mi.getExceptionTypes()) {
434 				if (eci.hasAnnotation(Response.class)) {
435 					List<Response> la = eci.getAnnotations(context, Response.class);
436 					List<StatusCode> la2 = eci.getAnnotations(context, StatusCode.class);
437 					Set<Integer> codes = getCodes(la2, 500);
438 					for (var a : la) {
439 						for (var code : codes) {
440 							JsonMap om = responses.getMap(String.valueOf(code), true);
441 							merge(om, a);
442 							JsonMap schema = getSchema(om.getMap("schema"), m.getGenericReturnType(), bs);
443 							eci.forEachAnnotation(Schema.class, x -> true, x -> merge(schema, x));
444 							pushupSchemaFields(RESPONSE, om, schema);
445 							om.appendIf(nem, "schema", schema);
446 						}
447 					}
448 					List<MethodInfo> methods = eci.getMethods();
449 					for (int i = methods.size() - 1; i >= 0; i--) {
450 						MethodInfo ecmi = methods.get(i);
451 						Header a = ecmi.getAnnotationInfos(Header.class).findFirst().map(AnnotationInfo::inner).orElse(null);
452 						if (a == null)
453 							a = ecmi.getReturnType().unwrap(Value.class, Optional.class).getAnnotationInfos(Header.class).findFirst().map(AnnotationInfo::inner).orElse(null);
454 						if (nn(a) && ! isMulti(a)) {
455 							String ha = a.name();
456 							for (var code : codes) {
457 								JsonMap header = responses.getMap(String.valueOf(code), true).getMap("headers", true).getMap(ha, true);
458 								ecmi.forEachAnnotation(context, Schema.class, x -> true, x -> merge(header, x));
459 								ecmi.getReturnType().unwrap(Value.class, Optional.class).forEachAnnotation(Schema.class, x -> true, x -> merge(header, x));
460 								pushupSchemaFields(RESPONSE_HEADER, header, getSchema(header.getMap("schema"), ecmi.getReturnType().unwrap(Value.class, Optional.class).innerType(), bs));
461 							}
462 						}
463 					}
464 				}
465 			}
466 
467 			if (mi.hasAnnotation(Response.class) || mi.getReturnType().unwrap(Value.class, Optional.class).hasAnnotation(Response.class)) {
468 				List<Response> la = list();
469 				mi.forEachAnnotation(context, Response.class, x -> true, x -> la.add(x));
470 				List<StatusCode> la2 = list();
471 				mi.forEachAnnotation(context, StatusCode.class, x -> true, x -> la2.add(x));
472 				Set<Integer> codes = getCodes(la2, 200);
473 				for (var a : la) {
474 					for (var code : codes) {
475 						JsonMap om = responses.getMap(String.valueOf(code), true);
476 						merge(om, a);
477 						JsonMap schema = getSchema(om.getMap("schema"), m.getGenericReturnType(), bs);
478 						mi.forEachAnnotation(context, Schema.class, x -> true, x -> merge(schema, x));
479 						pushupSchemaFields(RESPONSE, om, schema);
480 						om.appendIf(nem, "schema", schema);
481 						addBodyExamples(sm, om, true, m.getGenericReturnType(), locale);
482 					}
483 				}
484 				if (mi.getReturnType().hasAnnotation(Response.class)) {
485 					List<MethodInfo> methods = mi.getReturnType().getMethods();
486 					for (int i = methods.size() - 1; i >= 0; i--) {
487 						MethodInfo ecmi = methods.get(i);
488 						if (ecmi.hasAnnotation(Header.class)) {
489 							Header a = ecmi.getAnnotationInfos(Header.class).findFirst().map(AnnotationInfo::inner).orElse(null);
490 							String ha = a.name();
491 							if (! isMulti(a)) {
492 								for (var code : codes) {
493 									JsonMap header = responses.getMap(String.valueOf(code), true).getMap("headers", true).getMap(ha, true);
494 									ecmi.forEachAnnotation(context, Schema.class, x -> true, x -> merge(header, x));
495 									ecmi.getReturnType().unwrap(Value.class, Optional.class).forEachAnnotation(Schema.class, x -> true, x -> merge(header, x));
496 									merge(header, a.schema());
497 									pushupSchemaFields(RESPONSE_HEADER, header, getSchema(header, ecmi.getReturnType().innerType(), bs));
498 								}
499 							}
500 						}
501 					}
502 				}
503 			} else if (m.getGenericReturnType() != void.class) {
504 				JsonMap om = responses.getMap("200", true);
505 				var pt2 = ClassInfo.of(m.getGenericReturnType());
506 				JsonMap schema = getSchema(om.getMap("schema"), m.getGenericReturnType(), bs);
507 				pt2.forEachAnnotation(Schema.class, x -> true, x -> merge(schema, x));
508 				pushupSchemaFields(RESPONSE, om, schema);
509 				om.appendIf(nem, "schema", schema);
510 				addBodyExamples(sm, om, true, m.getGenericReturnType(), locale);
511 			}
512 
513 			// Finally, look for Value @Header parameters defined on method.
514 			for (var mpi : mi.getParameters()) {
515 
516 				ClassInfo pt = mpi.getParameterType();
517 
518 				if (pt.is(Value.class) && (mpi.hasAnnotation(Header.class) || pt.hasAnnotation(Header.class))) {
519 					List<Header> la = list();
520 					mpi.forEachAnnotation(Header.class, x -> true, x -> la.add(x));
521 					pt.forEachAnnotation(Header.class, x -> true, x -> la.add(x));
522 					List<StatusCode> la2 = list();
523 					mpi.forEachAnnotation(StatusCode.class, x -> true, x -> la2.add(x));
524 					pt.forEachAnnotation(StatusCode.class, x -> true, x -> la2.add(x));
525 					Set<Integer> codes = getCodes(la2, 200);
526 					String name = HeaderAnnotation.findName(mpi).orElse(null);
527 					Type type = Value.unwrap(mpi.getParameterType().innerType());
528 					for (var a : la) {
529 						if (! isMulti(a)) {
530 							for (var code : codes) {
531 								JsonMap header = responses.getMap(String.valueOf(code), true).getMap("headers", true).getMap(name, true);
532 								mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(header, x));
533 								merge(header, a.schema());
534 								pushupSchemaFields(RESPONSE_HEADER, header, getSchema(header, type, bs));
535 							}
536 						}
537 					}
538 
539 				} else if (mpi.hasAnnotation(Response.class) || pt.hasAnnotation(Response.class)) {
540 					List<Response> la = list();
541 					mpi.forEachAnnotation(Response.class, x -> true, x -> la.add(x));
542 					pt.forEachAnnotation(Response.class, x -> true, x -> la.add(x));
543 					List<StatusCode> la2 = list();
544 					mpi.forEachAnnotation(StatusCode.class, x -> true, x -> la2.add(x));
545 					pt.forEachAnnotation(StatusCode.class, x -> true, x -> la2.add(x));
546 					Set<Integer> codes = getCodes(la2, 200);
547 					Type type = Value.unwrap(mpi.getParameterType().innerType());
548 					for (var a : la) {
549 						for (var code : codes) {
550 							JsonMap om = responses.getMap(String.valueOf(code), true);
551 							merge(om, a);
552 							JsonMap schema = getSchema(om.getMap("schema"), type, bs);
553 							mpi.forEachAnnotation(Schema.class, x -> true, x -> merge(schema, x));
554 							la.forEach(x -> merge(schema, x.schema()));
555 							pushupSchemaFields(RESPONSE, om, schema);
556 							om.appendIf(nem, "schema", schema);
557 						}
558 					}
559 				}
560 			}
561 
562 			// Add default response descriptions.
563 			for (var e : responses.entrySet()) {
564 				String key = e.getKey();
565 				JsonMap val = responses.getMap(key);
566 				if (StringUtils.isDecimal(key))
567 					val.appendIfAbsentIf(ne, "description", RestUtils.getHttpResponseText(Integer.parseInt(key)));
568 			}
569 
570 			if (responses.isEmpty())
571 				op.remove("responses");
572 			else
573 				op.put("responses", new TreeMap<>(responses));
574 
575 			if (! op.containsKey("consumes")) {
576 				List<MediaType> mConsumes = sm.getSupportedContentTypes();
577 				if (! mConsumes.equals(consumes))
578 					op.put("consumes", mConsumes);
579 			}
580 
581 			if (! op.containsKey("produces")) {
582 				List<MediaType> mProduces = sm.getSupportedAcceptTypes();
583 				if (! mProduces.equals(produces))
584 					op.put("produces", mProduces);
585 			}
586 		}
587 
588 		if (nn(js.getBeanDefs()))
589 			for (var e : js.getBeanDefs().entrySet())
590 				definitions.put(e.getKey(), fixSwaggerExtensions(e.getValue()));
591 		if (definitions.isEmpty())
592 			omSwagger.remove("definitions");
593 
594 		if (! tagMap.isEmpty())
595 			omSwagger.put("tags", tagMap.values());
596 
597 		if (consumes.isEmpty())
598 			omSwagger.remove("consumes");
599 		if (produces.isEmpty())
600 			omSwagger.remove("produces");
601 
602 		try {
603 			String swaggerJson = Json5Serializer.DEFAULT_READABLE.toString(omSwagger);
604 			return jp.parse(swaggerJson, Swagger.class);
605 		} catch (Exception e) {
606 			throw new ServletException("Error detected in swagger.", e);
607 		}

609 	}
