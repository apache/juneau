25  /**
26   * Specialized subclass of {@link BasicSpringRestServlet} for showing "group" pages.
27   *
28   * <p>
29   * Meant as a base class for top-level REST resources in Spring Boot environments.
30   *
31   * <p>
32   * Provides support for JSON, XML, HTML, URL-Encoding, UON, XML, OpenAPI, and MessagePack.  See {@link BasicUniversalConfig}
33   * for details.
34   *
35   * <p>
36   * Implements the basic REST endpoints defined in {@link BasicRestOperations}.
37   *
38   * <p>
39   * Children are attached to this resource using the {@link Rest#children() @Rest(children)} annotation.
40   *
41   * <h5 class='section'>See Also:</h5><ul>
42   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestServerSpringbootBasics">juneau-rest-server-springboot Basics</a>
43   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestAnnotatedClassBasics">@Rest-Annotated Class Basics</a>
44   * </ul>
45   *
46   * @serial exclude
47   */
48  @Rest
49  public abstract class BasicSpringRestServletGroup extends BasicSpringRestServlet implements BasicGroupOperations {
50  	private static final long serialVersionUID = 1L;
51  
52  	@Override /* BasicGroupOperations */
53  	public ChildResourceDescriptions getChildren(RestRequest req) {
54  		return ChildResourceDescriptions.of(req);
55  	}
56  }
