31  /**
32   * Identifies a REST DELETE operation Java method on a {@link RestServlet} implementation class.
33   *
34   * <p>
35   * This is a specialized subtype of <c><ja>{@link RestOp @RestOp}(method=<jsf>DELETE</jsf>)</c>.
36   *
37   * <h5 class='section'>See Also:</h5><ul>
38   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/RestOpAnnotatedMethodBasics">@RestOp-Annotated Method Basics</a>
39  
40   * </ul>
41   */
42  @Target(METHOD)
43  @Retention(RUNTIME)
44  @Inherited
45  @ContextApply(RestDeleteAnnotation.RestOpContextApply.class)
46  @AnnotationGroup(RestOp.class)
47  public @interface RestDelete {
48  
49  	/**
50  	 * Specifies whether this method can be called based on the client version.
51  	 *
52  	 * <p>
53  	 * The client version is identified via the HTTP request header identified by
54  	 * {@link Rest#clientVersionHeader() @Rest(clientVersionHeader)} which by default is <js>"Client-Version"</js>.
55  	 *
56  	 * <p>
57  	 * This is a specialized kind of {@link RestMatcher} that allows you to invoke different Java methods for the same
58  	 * method/path based on the client version.
59  	 *
60  	 * <p>
61  	 * The format of the client version range is similar to that of OSGi versions.
62  	 *
63  	 * <p>
64  	 * In the following example, the Java methods are mapped to the same HTTP method and URL <js>"/foobar"</js>.
65  	 * <p class='bjava'>
66  	 * 	<jc>// Call this method if Client-Version is at least 2.0.
67  	 * 	// Note that this also matches 2.0.1.</jc>
68  	 * 	<ja>@RestDelete</ja>(path=<js>"/foobar"</js>, clientVersion=<js>"2.0"</js>)
69  	 * 	<jk>public</jk> Object method1()  {...}
70  	 *
71  	 * 	<jc>// Call this method if Client-Version is at least 1.1, but less than 2.0.</jc>
72  	 * 	<ja>@RestDelete</ja>(path=<js>"/foobar"</js>, clientVersion=<js>"[1.1,2.0)"</js>)
73  	 * 	<jk>public</jk> Object method2()  {...}
74  	 *
75  	 * 	<jc>// Call this method if Client-Version is less than 1.1.</jc>
76  	 * 	<ja>@RestDelete</ja>(path=<js>"/foobar"</js>, clientVersion=<js>"[0,1.1)"</js>)
77  	 * 	<jk>public</jk> Object method3()  {...}
78  	 * </p>
79  	 *
80  	 * <p>
81  	 * It's common to combine the client version with transforms that will convert new POJOs into older POJOs for
82  	 * backwards compatibility.
83  	 * <p class='bjava'>
84  	 * 	<jc>// Call this method if Client-Version is at least 2.0.</jc>
85  	 * 	<ja>@RestDelete</ja>(path=<js>"/foobar"</js>, clientVersion=<js>"2.0"</js>)
86  	 * 	<jk>public void</jk> newMethod()  {...}
87  	 *
88  	 * 	<jc>// Call this method if Client-Version is at least 1.1, but less than 2.0.</jc>
89  	 * 	<ja>@RestDelete</ja>(path=<js>"/foobar"</js>, clientVersion=<js>"[1.1,2.0)"</js>})
90  	 * 	<jk>public void</jk> oldMethod() {...}
91  	 * </p>
92  	 *
93  	 * <p>
94  	 * Note that in the previous example, we're returning the exact same POJO, but using a transform to convert it into
95  	 * an older form.
96  	 * The old method could also just return back a completely different object.
97  	 * The range can be any of the following:
98  	 * <ul>
99  	 * 	<li><js>"[0,1.0)"</js> = Less than 1.0.  1.0 and 1.0.0 does not match.
100 	 * 	<li><js>"[0,1.0]"</js> = Less than or equal to 1.0.  Note that 1.0.1 will match.
101 	 * 	<li><js>"1.0"</js> = At least 1.0.  1.0 and 2.0 will match.
102 	 * </ul>
103 	 *
104 	 * <h5 class='section'>See Also:</h5><ul>
105 	 * 	<li class='jm'>{@link org.apache.juneau.rest.RestContext.Builder#clientVersionHeader(String)}
106 	 * </ul>
107 	 *
108 	 * @return The annotation value.
109 	 */
110 	String clientVersion() default "";

451 	 * Role guard.
452 	 *
453 	 * <p>
454 	 * An expression defining if a user with the specified roles are allowed to access this method.
455 	 *
456 	 * <h5 class='section'>Example:</h5>
457 	 * <p class='bjava'>
458 	 * 	<jk>public class</jk> MyResource <jk>extends</jk> BasicRestServlet {
459 	 *
460 	 * 		<ja>@RestDelete</ja>(
461 	 * 			path=<js>"/foo"</js>,
462 	 * 			roleGuard=<js>"ROLE_ADMIN || (ROLE_READ_WRITE &amp;&amp; ROLE_SPECIAL)"</js>
463 	 * 		)
464 	 * 		<jk>public</jk> Object doDelete() {
465 	 * 		}
466 	 * 	}
467 	 * </p>
468 	 *
469 	 * <h5 class='section'>Notes:</h5><ul>
470 	 * 	<li class='note'>
471 	 * 		Supports any of the following expression constructs:
472 	 * 		<ul>
473 	 * 			<li><js>"foo"</js> - Single arguments.
474 	 * 			<li><js>"foo,bar,baz"</js> - Multiple OR'ed arguments.
475 	 * 			<li><js>"foo | bar | baz"</js> - Multiple OR'ed arguments, pipe syntax.
476 	 * 			<li><js>"foo || bar || baz"</js> - Multiple OR'ed arguments, Java-OR syntax.
477 	 * 			<li><js>"fo*"</js> - Patterns including <js>'*'</js> and <js>'?'</js>.
478 	 * 			<li><js>"fo* &amp; *oo"</js> - Multiple AND'ed arguments, ampersand syntax.
479 	 * 			<li><js>"fo* &amp;&amp; *oo"</js> - Multiple AND'ed arguments, Java-AND syntax.
480 	 * 			<li><js>"fo* || (*oo || bar)"</js> - Parenthesis.
481 	 * 		</ul>
482 	 * 	<li class='note'>
483 	 * 		AND operations take precedence over OR operations (as expected).
484 	 * 	<li class='note'>
485 	 * 		Whitespace is ignored.
486 	 * 	<li class='note'>
487 	 * 		<jk>null</jk> or empty expressions always match as <jk>false</jk>.
488 	 * 	<li class='note'>
489 	 * 		If patterns are used, you must specify the list of declared roles using {@link #rolesDeclared()} or {@link org.apache.juneau.rest.RestOpContext.Builder#rolesDeclared(String...)}.
490 	 * 	<li class='note'>
491 	 * 		Supports <a class="doclink" href="https://juneau.apache.org/docs/topics/RestServerSvlVariables">SVL Variables</a>
492 	 * 		(e.g. <js>"$L{my.localized.variable}"</js>).
493 	 * 	<li class='note'>
494 	 * 		When defined on parent/child classes and methods, ALL guards within the hierarchy must pass.
495 	 * </ul>
496 	 *
497 	 * <h5 class='section'>See Also:</h5><ul>
498 	 * 	<li class='jm'>{@link org.apache.juneau.rest.RestOpContext.Builder#roleGuard(String)}
