17  package org.apache.juneau.rest.annotation;
18  
19  import static java.lang.annotation.ElementType.*;
20  import static java.lang.annotation.RetentionPolicy.*;
21  
22  import java.lang.annotation.*;
23  
24  import org.apache.juneau.rest.*;
25  
26  import jakarta.servlet.*;
27  
28  /**
29   * Identifies a method that gets called during servlet initialization.
30   *
31   * <p>
32   * This method is called from within the {@link Servlet#init(ServletConfig)} method after the {@link org.apache.juneau.rest.RestContext.Builder}
33   * object has been created and initialized with the annotations defined on the class, but before the
34   * {@link RestContext} object has been created.
35   *
36   * <p>
37   * The only valid parameter type for this method is {@link org.apache.juneau.rest.RestContext.Builder} which can be used to configure the servlet.
38   *
39   * <p>
40   * An example of this is the <c>PetStoreResource</c> class that uses an init method to perform initialization
41   * of an internal data structure.
42   *
43   * <h5 class='figure'>Example:</h5>
44   * <p class='bjava'>
45   * 	<ja>@Rest</ja>(...)
46   * 	<jk>public class</jk> PetStoreResource <jk>extends</jk> BasicRestServlet <jk>implements</jk> BasicUniversalJenaConfig {
47   *
48   * 		<jc>// Our database.</jc>
49   * 		<jk>private</jk> Map&lt;Integer,Pet&gt; <jf>petDB</jf>;
50   *
51   * 		<ja>@RestInit</ja>
52   * 		<jk>public void</jk> onInit(RestContext.Builder <jv>builder</jv>) <jk>throws</jk> Exception {
53   * 			<jc>// Load our database from a local JSON file.</jc>
54   * 			<jf>petDB</jf> = JsonParser.<jsf>DEFAULT</jsf>.parse(getClass().getResourceAsStream(<js>"PetStore.json"</js>), LinkedHashMap.<jk>class</jk>, Integer.<jk>class</jk>, Pet.<jk>class</jk>);
55   * 		}
56   * 	}
57   * </p>
58   *
59   * <h5 class='section'>Notes:</h5><ul>
60   * 	<li class='note'>
61   * 		The method should return <jk>void</jk> although if it does return any value, the value will be ignored.
62   * 	<li class='note'>
63   * 		The method should be <jk>public</jk> although other visibilities are valid if the security manager allows it.
64   * 	<li class='note'>
65   * 		Static methods can be used.
66   * 	<li class='note'>
67   * 		Multiple init methods can be defined on a class.
68   * 		<br>Init methods on parent classes are invoked before init methods on child classes.
69   * 		<br>The order of init method invocations within a class is alphabetical, then by parameter count, then by parameter types.
70   * 	<li class='note'>
71   * 		The method can throw any exception causing initialization of the servlet to fail.
72   * 	<li class='note'>
73   * 		Note that if you override a parent method, you probably need to call <code><jk>super</jk>.parentMethod(...)</code>.
74   * 		<br>The method is still considered part of the parent class for ordering purposes even though it's
75   * 		overridden by the child class.
76   * </ul>
77   *
78   * <h5 class='section'>See Also:</h5><ul>
79   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/LifecycleHooks">Lifecycle Hooks</a>
80   * </ul>
81   */
82  @Target({METHOD,TYPE})
83  @Retention(RUNTIME)
84  @Inherited
85  @Repeatable(RestInitAnnotation.Array.class)
86  public @interface RestInit {
87  
88  	/**
89  	 * Optional description for the exposed API.
90  	 *
91  	 * @return The annotation value.
92  	 * @since 9.2.0
93  	 */
94  	String[] description() default {};
95  
96  	/**
97  	 * Dynamically apply this annotation to the specified methods.
98  	 *
99  	 * <h5 class='section'>See Also:</h5><ul>
100 	 * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/DynamicallyAppliedAnnotations">Dynamically Applied Annotations</a>
101 	 * </ul>
102 	 *
103 	 * @return The annotation value.
104 	 */
105 	String[] on() default {};
106 }
