15  /**
16   * Dummy class that indicates that serializers, parsers, or transforms should be inherited from parent-class-to-class or class-to-method.
17   *
18   * <h5 class='section'>Example:</h5>
19   * <p class='bjava'>
20   * 	<ja>@RestOp</ja>(
21   *
22   * 		<jc>// Override the serializers for this method, but also inherit the serializers from the class.
23   * 		// Class-level serializers will be inserted in the location in the array.</jc>
24   * 		serializers={JsonSerializer.<jk>class</jk>, Inherit.<jk>class</jk>},
25   *
26   * 		<jc>// Override the parsers for this method, but also inherit the parsers from the class.
27   * 		// Class-level parsers will be inserted in the location in the array.</jc>
28   * 		parsers={JsonParser.<jk>class</jk>, Inherit.<jk>class</jk>},
29   *
30   * 		<jc>// Override the bean filters for this method, but also inherit the bean filters from the class.
31   * 		// Overridden bean filters only apply to NEW serializers and parsers defined on the method
32   * 		// (not those inherited from the class).</jc>
33   * 		beanFilters={MyFilter.<jk>class</jk>, Inherit.<jk>class</jk>},
34   *
35   * 		<jc>// Override the POJO swaps for this method, but also inherit the POJO swaps from the class.
36   * 		// Overridden POJO swaps only apply to NEW serializers and parsers defined on the method
37   * 		// (not those inherited from the class).</jc>
38   * 		swaps={MySwap.<jk>class</jk>, Inherit.<jk>class</jk>}
39   * 	)
40   * </p>
41   *
42   * <h5 class='section'>See Also:</h5><ul>
43   * </ul>
44   */
45  public final class Inherit {}
