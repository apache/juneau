29  /**
30   * Parent class for all bean filters.
31   *
32   * <p>
33   * Bean filters are used to control aspects of how beans are handled during serialization and parsing.
34   *
35   * <p>
36   * Bean filters are created by {@link Builder} which is the programmatic equivalent to the {@link Bean @Bean}
37   * annotation.
38   *
39   * <h5 class='section'>See Also:</h5><ul>
40  
41   * </ul>
42   */
43  public class BeanFilter {
44  	/**
45  	 * Builder class.
46  	 */
47  	public static class Builder {
48  
49  		Class<?> beanClass;
50  		String typeName, example;
51  		Set<String> properties = set(), excludeProperties = set(), readOnlyProperties = set(), writeOnlyProperties = set();
52  		Class<?> implClass, interfaceClass, stopClass;
53  		boolean sortProperties, fluentSetters;
54  		BeanCreator<PropertyNamer> propertyNamer = BeanCreator.of(PropertyNamer.class);
55  		List<Class<?>> dictionary;
56  		@SuppressWarnings("rawtypes")
57  		BeanCreator<BeanInterceptor> interceptor = BeanCreator.of(BeanInterceptor.class);
58  
59  		/**
60  		 * Constructor.
61  		 *
62  		 * @param beanClass The bean class that this filter applies to.
63  		 */
64  		protected Builder(Class<?> beanClass) {
65  			this.beanClass = beanClass;
66  		}
67  
68  		/**
69  		 * Applies the information in the specified list of {@link Bean @Bean} annotations to this filter.
70  		 *
71  		 * @param annotations The annotations to apply.
72  		 * @return This object.
73  		 */
74  		public Builder applyAnnotations(List<Bean> annotations) {
75  
76  			annotations.forEach(x -> {
77  				if (isNotEmpty(x.properties()) || isNotEmpty(x.p()))
78  					properties(x.properties(), x.p());
79  				if (x.sort())
80  					sortProperties(true);
81  				if (x.findFluentSetters())
82  					findFluentSetters();
83  				if (isNotEmpty(x.excludeProperties()) || isNotEmpty(x.xp()))
84  					excludeProperties(x.excludeProperties(), x.xp());
85  				if (isNotEmpty(x.readOnlyProperties()) || isNotEmpty(x.ro()))
86  					readOnlyProperties(x.readOnlyProperties(), x.ro());
87  				if (isNotEmpty(x.writeOnlyProperties()) || isNotEmpty(x.wo()))
88  					writeOnlyProperties(x.writeOnlyProperties(), x.wo());
89  				if (isNotEmpty(x.typeName()))
90  					typeName(x.typeName());
91  				if (ClassUtils.isNotVoid(x.propertyNamer()))
92  					propertyNamer(x.propertyNamer());
93  				if (ClassUtils.isNotVoid(x.interfaceClass()))
94  					interfaceClass(x.interfaceClass());
95  				if (ClassUtils.isNotVoid(x.stopClass()))
96  					stopClass(x.stopClass());
97  				if (ClassUtils.isNotVoid(x.interceptor()))
98  					interceptor(x.interceptor());
99  				if (ClassUtils.isNotVoid(x.implClass()))
100 					implClass(x.implClass());
101 				if (ArrayUtils.isNotEmptyArray(x.dictionary()))
102 					dictionary(x.dictionary());
103 				if (isNotEmpty(x.example()))
104 					example(x.example());
105 			});
106 			return this;
107 		}

118 		/**
119 		 * Bean dictionary.
120 		 *
121 		 * <p>
122 		 * Adds to the list of classes that make up the bean dictionary for this bean.
123 		 *
124 		 * <h5 class='section'>Example:</h5>
125 		 * <p class='bjava'>
126 		 * 	<jc>// Define our filter.</jc>
127 		 * 	<jk>public class</jk> MyFilter <jk>extends</jk> Builder&lt;MyBean&gt; {
128 		 * 		<jk>public</jk> MyFilter() {
129 		 * 			<jc>// Our bean contains generic collections of Foo and Bar objects.</jc>
130 		 * 			beanDictionary(Foo.<jk>class</jk>, Bar.<jk>class</jk>);
131 		 * 		}
132 		 * 	}
133 		 *
134 		 * 	<jc>// Register it with a parser.</jc>
135 		 * 	ReaderParser <jv>parser</jv> = JsonParser
136 		 * 		.<jsm>create</jsm>()
137 		 * 		.beanFilters(MyFilter.<jk>class</jk>)
138 		 * 		.build();
139 		 *
140 		 * 	<jc>// Instantiate our bean.</jc>
141 		 * 	MyBean <jv>bean</jv> = <jv>parser</jv>.parse(<jv>json</jv>);
142 		 * </p>
143 		 *
144 		 * <h5 class='section'>See Also:</h5><ul>
145 		 * 	<li class='ja'>{@link Bean#dictionary()}
146 		 * 	<li class='jm'>{@link BeanContext.Builder#beanDictionary(Class...)}
147 		 * </ul>
148 		 *
149 		 * @param values
150 		 * 	The values to add to this property.
151 		 * @return This object.
152 		 */
153 		public Builder dictionary(Class<?>...values) {
154 			if (dictionary == null)
155 				dictionary = list(values);
156 			else
157 				for (Class<?> cc : values)
158 					dictionary.add(cc);
159 			return this;
160 		}

306 		 *
307 		 * Identifies a class to be used as the interface class for this and all subclasses.
308 		 *
309 		 * <p>

320 		 * 	<jk>public class</jk> A1 <jk>extends</jk> A {
321 		 * 		<jk>public</jk> String <jf>f1</jf> = <js>"f1"</js>;
322 		 * 	}
323 		 *
324 		 * 	<jc>// Define our filter.</jc>
325 		 * 	<jk>public class</jk> AFilter <jk>extends</jk> Builder&lt;A&gt; {
326 		 * 		<jk>public</jk> AFilter() {
327 		 * 			interfaceClass(A.<jk>class</jk>);
328 		 * 		}
329 		 * 	}
330 		 *
331 		 * 	<jc>// Register it with a serializer.</jc>
332 		 * 	WriterSerializer <jv>serializer</jv> = JsonSerializer
333 		 * 		.<jsm>create</jsm>()
334 		 * 		.beanFilters(AFilter.<jk>class</jk>)
335 		 * 		.build();
336 		 *
337 		 * 	<jc>// Use it.</jc>
338 		 * 	A1 <jv>a1</jv> = <jk>new</jk> A1();
339 		 * 	String <jv>json</jv> = <jv>serializer</jv>.serialize(<jv>a1</jv>);
340 		 * 	<jsm>assertEquals</jsm>(<js>"{f0:'f0'}"</js>, <jv>json</jv>);  <jc>// Note f1 is not serialized</jc>

551 		/**
552 		 * Bean stop class.
553 		 *
554 		 * <p>
555 		 * Identifies a stop class for this class and all subclasses.
556 		 *
557 		 * <p>
558 		 * Identical in purpose to the stop class specified by {@link Introspector#getBeanInfo(Class, Class)}.
559 		 * <br>Any properties in the stop class or in its base classes will be ignored during analysis.
560 		 *
561 		 * <p>
562 		 * For example, in the following class hierarchy, instances of <c>C3</c> will include property <c>p3</c>,
563 		 * but not <c>p1</c> or <c>p2</c>.
564 		 *
565 		 * <h5 class='section'>Example:</h5>
566 		 * <p class='bjava'>
567 		 * 	<jk>public class</jk> C1 {
568 		 * 		<jk>public int</jk> getP1();
569 		 * 	}
570 		 *
571 		 * 	<jk>public class</jk> C2 <jk>extends</jk> C1 {

645 		/**
646 		 * Write-only bean properties.
647 		 *
648 		 * <p>
649 		 * Specifies one or more properties on a bean that are write-only despite having valid setters.
650 		 * Parsers will parse such properties as usual, but serializers will silently ignore them.
651 		 *
652 		 * <h5 class='section'>Example:</h5>
653 		 * <p class='bjava'>
654 		 * 	<jc>// Define our filter.</jc>
655 		 * 	<jk>public class</jk> MyFilter <jk>extends</jk> Builder&lt;MyBean&gt; {
656 		 * 		<jk>public</jk> MyFilter() {
657 		 * 			writeOnlyProperties(<js>"foo,bar"</js>);
658 		 * 		}
659 		 * 	}
660 		 *
661 		 * 	<jc>// Register it with a serializer.</jc>
662 		 *  WriterSerializer <jv>serializer</jv> = JsonSerializer
663 		 * 		.<jsm>create</jsm>()
664 		 * 		.beanFilters(MyFilter.<jk>class</jk>)
665 		 * 		.build();
666 		 *
667 		 * 	<jc>// Serializes all properties except for 'foo' and 'bar'.</jc>
668 		 * 	String <jv>json</jv> = <jv>serializer</jv>.serialize(<jk>new</jk> MyBean());
669 		 * </p>
670 		 *
671 		 * <h5 class='section'>See Also:</h5><ul>
672 		 * 	<li class='ja'>{@link Bean#writeOnlyProperties()}
673 		 * 	<li class='ja'>{@link Beanp#wo()}
674 		 * 	<li class='jm'>{@link BeanContext.Builder#beanPropertiesWriteOnly(Class, String)}
675 		 * 	<li class='jm'>{@link BeanContext.Builder#beanPropertiesWriteOnly(String, String)}
676 		 * 	<li class='jm'>{@link BeanContext.Builder#beanPropertiesWriteOnly(Map)}
677 		 * </ul>
678 		 *
679 		 * @param value
680 		 * 	The new value for this setting.
681 		 * 	<br>Values can contain comma-delimited list of property names.
682 		 * @return This object.
683 		 */
684 		public Builder writeOnlyProperties(String...value) {
685 			this.writeOnlyProperties = set();
686 			for (String v : value)
687 				StringUtils.split(v, x -> writeOnlyProperties.add(x));
688 			return this;
689 		}
690 	}
