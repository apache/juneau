30  /**
31   * Allows arbitrary objects to be mapped to classes and methods base on class/method name keys.
32   *
33   * <p>
34   * The valid pattern matches are:
35   * <ul class='spaced-list'>
36   *  <li>Classes:
37   * 		<ul>
38   * 			<li>Fully qualified:
39   * 				<ul>
40   * 					<li><js>"com.foo.MyClass"</js>
41   * 				</ul>
42   * 			<li>Fully qualified inner class:
43   * 				<ul>
44   * 					<li><js>"com.foo.MyClass$Inner1$Inner2"</js>
45   * 				</ul>
46   * 			<li>Simple:
47   * 				<ul>
48   * 					<li><js>"MyClass"</js>
49   * 				</ul>
50   * 			<li>Simple inner:
51   * 				<ul>
52   * 					<li><js>"MyClass$Inner1$Inner2"</js>
53   * 					<li><js>"Inner1$Inner2"</js>
54   * 					<li><js>"Inner2"</js>
55   * 				</ul>
56   * 		</ul>
57   * 	<li>Methods:
58   * 		<ul>
59   * 			<li>Fully qualified with args:
60   * 				<ul>
61   * 					<li><js>"com.foo.MyClass.myMethod(String,int)"</js>
62   * 					<li><js>"com.foo.MyClass.myMethod(java.lang.String,int)"</js>
63   * 					<li><js>"com.foo.MyClass.myMethod()"</js>
64   * 				</ul>
65   * 			<li>Fully qualified:
66   * 				<ul>
67   * 					<li><js>"com.foo.MyClass.myMethod"</js>
68   * 				</ul>
69   * 			<li>Simple with args:
70   * 				<ul>
71   * 					<li><js>"MyClass.myMethod(String,int)"</js>
72   * 					<li><js>"MyClass.myMethod(java.lang.String,int)"</js>
73   * 					<li><js>"MyClass.myMethod()"</js>
74   * 				</ul>
75   * 			<li>Simple:
76   * 				<ul>
77   * 					<li><js>"MyClass.myMethod"</js>
78   * 				</ul>
79   * 			<li>Simple inner class:
80   * 				<ul>
81   * 					<li><js>"MyClass$Inner1$Inner2.myMethod"</js>
82   * 					<li><js>"Inner1$Inner2.myMethod"</js>
83   * 					<li><js>"Inner2.myMethod"</js>
84   * 				</ul>
85   * 		</ul>
86   * 	<li>Fields:
87   * 		<ul>
88   * 			<li>Fully qualified:
89   * 				<ul>
90   * 					<li><js>"com.foo.MyClass.myField"</js>
91   * 				</ul>
92   * 			<li>Simple:
93   * 				<ul>
94   * 					<li><js>"MyClass.myField"</js>
95   * 				</ul>
96   * 			<li>Simple inner class:
97   * 				<ul>
98   * 					<li><js>"MyClass$Inner1$Inner2.myField"</js>
99   * 					<li><js>"Inner1$Inner2.myField"</js>
100  * 					<li><js>"Inner2.myField"</js>
101  * 				</ul>
102  * 		</ul>
103  * 	<li>Constructors:
104  * 		<ul>
105  * 			<li>Fully qualified with args:
106  * 				<ul>
107  * 					<li><js>"com.foo.MyClass(String,int)"</js>
108  * 					<li><js>"com.foo.MyClass(java.lang.String,int)"</js>
109  * 					<li><js>"com.foo.MyClass()"</js>
110  * 				</ul>
111  * 			<li>Simple with args:
112  * 				<ul>
113  * 					<li><js>"MyClass(String,int)"</js>
114  * 					<li><js>"MyClass(java.lang.String,int)"</js>
115  * 					<li><js>"MyClass()"</js>
116  * 				</ul>
117  * 			<li>Simple inner class:
118  * 				<ul>
119  * 					<li><js>"MyClass$Inner1$Inner2()"</js>
120  * 					<li><js>"Inner1$Inner2()"</js>
121  * 					<li><js>"Inner2()"</js>
122  * 				</ul>
123  * 		</ul>
124  * 	<li>A comma-delimited list of anything on this list.
125  * </ul>
126  *
127  * <h5 class='section'>See Also:</h5><ul>
128  * </ul>
129  *
130  * @param <V> The type of object in this map.
131  */

138 		final List<ClassEntry<V>> classEntries;
139 		final List<MethodEntry<V>> methodEntries;
140 		final List<FieldEntry<V>> fieldEntries;
141 		final List<ConstructorEntry<V>> constructorEntries;

143 		/**
144 		 * Constructor.
145 		 */
146 		protected Builder() {
147 			classEntries = list();
148 			methodEntries = list();
149 			fieldEntries = list();
150 			constructorEntries = list();
151 		}

153 		/**
154 		 * Copy constructor.
155 		 *
156 		 * @param copyFrom The builder being copied.
157 		 */
158 		protected Builder(Builder<V> copyFrom) {
159 			classEntries = CollectionUtils.copyOf(copyFrom.classEntries);
160 			methodEntries = CollectionUtils.copyOf(copyFrom.methodEntries);
161 			fieldEntries = CollectionUtils.copyOf(copyFrom.fieldEntries);
162 			constructorEntries = CollectionUtils.copyOf(copyFrom.constructorEntries);
163 		}

165 		/**
166 		 * Adds a mapping to this builder.
167 		 *
168 		 * @param key
169 		 * 	The mapping key.
170 		 * 	<br>Can be any of the following:
171 		 * 	<ul>
172 		 * 		<li>Full class name (e.g. <js>"com.foo.MyClass"</js>).
173 		 * 		<li>Simple class name (e.g. <js>"MyClass"</js>).
174 		 * 		<li>All classes (e.g. <js>"*"</js>).
175 		 * 		<li>Full method name (e.g. <js>"com.foo.MyClass.myMethod"</js>).
176 		 * 		<li>Simple method name (e.g. <js>"MyClass.myMethod"</js>).
177 		 * 		<li>A comma-delimited list of anything on this list.
178 		 * 	</ul>
179 		 * @param value The value for this mapping.
180 		 * @return This object.
181 		 */
182 		public Builder<V> append(String key, V value) {
183 			if (isEmpty(key))
184 				throw new BasicRuntimeException("Invalid reflection signature: [{0}]", key);
185 			try {
186 				splitNames(key, k -> {
187 					if (k.endsWith(")")) {
188 						int i = k.substring(0, k.indexOf('(')).lastIndexOf('.');
189 						if (i == -1 || isUpperCase(k.charAt(i + 1))) {
190 							constructorEntries.add(new ConstructorEntry<>(k, value));
191 						} else {
192 							methodEntries.add(new MethodEntry<>(k, value));
193 						}
194 					} else {
195 						int i = k.lastIndexOf('.');
196 						if (i == -1) {
197 							classEntries.add(new ClassEntry<>(k, value));
198 						} else if (isUpperCase(k.charAt(i + 1))) {
199 							classEntries.add(new ClassEntry<>(k, value));
200 							fieldEntries.add(new FieldEntry<>(k, value));
201 						} else {
202 							methodEntries.add(new MethodEntry<>(k, value));
203 							fieldEntries.add(new FieldEntry<>(k, value));
204 						}
205 					}
206 				});
207 			} catch (@SuppressWarnings("unused") IndexOutOfBoundsException e) {
208 				throw new BasicRuntimeException("Invalid reflection signature: [{0}]", key);
209 			}
210 
211 			return this;
212 		}

268 			name = name.substring(0, i).trim();

278 			return classMatches(simpleClassName, fullClassName, c) && (argsMatch(args, m.getParameterTypes()));
279 		}

328 	static class MethodEntry<V> {
329 		String simpleClassName, fullClassName, methodName, args[];
330 		V value;
331 
332 		MethodEntry(String name, V value) {
333 			int i = name.indexOf('(');
334 			this.args = i == -1 ? null : StringUtils.splitMethodArgs(name.substring(i + 1, name.length() - 1));
335 			if (nn(args)) {
336 				for (int j = 0; j < args.length; j++) {
337 
338 					// Strip off generic parameters.
339 					int k = args[j].indexOf('<');
340 					if (k > 0)
341 						args[j] = args[j].substring(0, k);
342 
343 					// Convert from xxx[][] to [[Lxxx; notation.
344 					if (args[j].endsWith("[]")) {
345 						int l = 0;
346 						while (args[j].endsWith("[]")) {
347 							l++;
348 							args[j] = args[j].substring(0, args[j].length() - 2);
349 						}
350 						var sb = new StringBuilder(args[j].length() + l + 2);
351 						for (int m = 0; m < l; m++)
352 							sb.append('[');
353 						sb.append('L').append(args[j]).append(';');
354 						args[j] = sb.toString();
355 					}
356 				}
357 			}
358 			name = i == -1 ? name : name.substring(0, i);
359 			i = name.lastIndexOf('.');
360 			var s1 = name.substring(0, i).trim();
361 			var s2 = name.substring(i + 1).trim();
362 			this.simpleClassName = simpleClassName(s1);
363 			this.fullClassName = s1;
364 			this.methodName = s2;
365 			this.value = value;
366 		}
367 
368 		public boolean matches(Method m) {
369 			if (m == null)
370 				return false;
371 			var c = m.getDeclaringClass();
372 			// @formatter:off
373 			return
374 				classMatches(simpleClassName, fullClassName, c)
375 				&& (eq(m.getName(), methodName))
376 				&& (argsMatch(args, m.getParameterTypes()));
377 			// @formatter:on
378 		}
379 
380 		@Override
381 		public String toString() {
382 			// @formatter:off
383 			return filteredMap()
384 				.append("simpleClassName", simpleClassName)
385 				.append("fullClassName", fullClassName)
386 				.append("methodName", methodName)
387 				.append("args", args)
388 				.append("value", value)
389 				.asString();
390 			// @formatter:on
391 		}
392 	}

432 	static boolean argsMatch(String[] names, Class<?>[] args) {
433 		if (names == null)
434 			return true;
435 		if (names.length != args.length)
436 			return false;
437 		for (int i = 0; i < args.length; i++) {
438 			var n = names[i];
439 			var a = args[i];
440 			if (! (eq(n, a.getSimpleName()) || eq(n, a.getName())))
441 				return false;
442 		}
443 		return true;
444 	}

446 	static boolean classMatches(String simpleName, String fullName, Class<?> c) {
447 		// For class org.apache.juneau.a.rttests.RountTripBeansWithBuilders$Ac$Builder
448 		// c.getSimpleName() == "Builder"
449 		// c.getFullName() == "org.apache.juneau.a.rttests.RountTripBeansWithBuilders$Ac$Builder"
450 		// c.getPackage() == "org.apache.juneau.a.rttests"
451 		var cSimple = c.getSimpleName();
452 		var cFull = c.getName();
453 		if (eq(simpleName, cSimple) || eq(fullName, cFull) || "*".equals(simpleName))
454 			return true;
455 		if (cFull.indexOf('$') != -1) {
456 			var p = c.getPackage();
457 			if (nn(p))
458 				cFull = cFull.substring(p.getName().length() + 1);
459 			if (eq(simpleName, cFull))
460 				return true;
461 			int i = cFull.indexOf('$');
462 			while (i != -1) {
463 				cFull = cFull.substring(i + 1);
464 				if (eq(simpleName, cFull))
465 					return true;
466 				i = cFull.indexOf('$');
467 			}
468 		}
469 		return false;
470 	}

472 	static String simpleClassName(String name) {
473 		int i = name.indexOf('.');
474 		if (i == -1)
475 			return name;
476 		return null;
477 	}

479 	static void splitNames(String key, Consumer<String> consumer) {
480 		if (key.indexOf(',') == -1) {
481 			consumer.accept(key);
482 		} else {
483 			int m = 0;
484 			boolean escaped = false;
485 			for (int i = 0; i < key.length(); i++) {
486 				char c = key.charAt(i);
487 				if (c == '(')
488 					escaped = true;
489 				else if (c == ')')
490 					escaped = false;
491 				else if (c == ',' && ! escaped) {
492 					consumer.accept(key.substring(m, i).trim());
493 					m = i + 1;
494 				}
495 			}
496 			consumer.accept(key.substring(m).trim());
497 		}
498 	}

522 	 * Finds all values in this map that matches the specified class.
523 	 *
524 	 * @param c The class to test for.
525 	 * @param ofType Only return objects of the specified type.
526 	 * @param array The array to append values to.
527 	 * @return The same list passed in or a new modifiable list if <jk>null</jk>.
528 	 */
529 	public V[] appendAll(Class<?> c, Class<? extends V> ofType, V[] array) {
530 		List<V> list = null;
531 		for (ClassEntry<V> e : classEntries)
532 			if (e.matches(c) && nn(e.value))
533 				if (ofType == null || ofType.isInstance(e.value))
534 					list = lazyAdd(array, list, e.value);
535 		return lazyArray(array, list);
536 	}
