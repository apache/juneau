40  /**
41   * Java bean store.
42   *
43   * <p>
44   * A simple storage database for beans keyed by type and name.
45   * Used to retrieve and instantiate beans using an injection-like API.
46   * It's similar in concept to the injection framework of Spring but greatly simplified in function and not intended to implement a full-fledged injection framework.
47   *
48   * <p>
49   * Beans can be stored with or without names.  Named beans are typically resolved using
50   * the <ja>@Name</ja> or <ja>@Qualified</ja> annotations on constructor or method parameters.
51   *
52   * <p>
53   * Beans are added through the following methods:
54   * <ul class='javatreec'>
55   * 	<li class='jm'>{@link #add(Class,Object) add(Class,Object)}
56   * 	<li class='jm'>{@link #add(Class,Object,String) add(Class,Object,String)}
57   * 	<li class='jm'>{@link #addBean(Class,Object) addBean(Class,Object)}
58   * 	<li class='jm'>{@link #addBean(Class,Object,String) addBean(Class,Object,String)}
59   * 	<li class='jm'>{@link #addSupplier(Class,Supplier) addSupplier(Class,Supplier)}
60   * 	<li class='jm'>{@link #addSupplier(Class,Supplier,String) addSupplier(Class,Supplier,String)}
61   * </ul>
62   *
63   * <p>
64   * Beans are retrieved through the following methods:
65   * <ul class='javatreec'>
66   * 	<li class='jm'>{@link #getBean(Class) getBean(Class)}
67   * 	<li class='jm'>{@link #getBean(Class,String) getBean(Class,String)}
68   * 	<li class='jm'>{@link #stream(Class) stream(Class)}
69   * </ul>
70   *
71   * <p>
72   * Beans are created through the following methods:
73   * <ul class='javatreec'>
74   * 	<li class='jm'>{@link #createBean(Class) createBean(Class)}
75   * 	<li class='jm'>{@link #createMethodFinder(Class) createMethodFinder(Class)}
76   * 	<li class='jm'>{@link #createMethodFinder(Class,Class) createMethodFinder(Class,Class)}
77   * 	<li class='jm'>{@link #createMethodFinder(Class,Object) createMethodFinder(Class,Object)}
78   * </ul>
79   *
80   * <h5 class='section'>Notes:</h5><ul>
81   * 	<li class='note'>Bean stores can be nested using {@link Builder#parent(BeanStore)}.
82   * 	<li class='note'>Bean stores can be made read-only using {@link Builder#readOnly()}.
83   * 	<li class='note'>Bean stores can be made thread-safe using {@link Builder#threadSafe()}.
84   * </ul>
85   *
86   * <h5 class='section'>See Also:</h5><ul>
87   * </ul>
88   */

114 			if (type == null || type == BeanStore.class)

117 		var c = info(type);

502 	 * Given an executable, returns a list of types that are missing from this factory.
503 	 *
504 	 * @param executable The constructor or method to get the params for.
505 	 * @return A comma-delimited list of types that are missing from this factory, or <jk>null</jk> if none are missing.
506 	 */
507 	public String getMissingParams(ExecutableInfo executable) {
508 		List<ParameterInfo> params = executable.getParameters();
509 		List<String> l = list();
510 		loop: for (int i = 0; i < params.size(); i++) {
511 			ParameterInfo pi = params.get(i);
512 			ClassInfo pt = pi.getParameterType();
513 			if (i == 0 && outer.isPresent() && pt.isInstance(outer.get()))
514 				continue loop;
515 			if (pt.is(Optional.class) || pt.is(BeanStore.class))
516 				continue loop;
517 			String beanName = pi.getResolvedQualifier();  // Use @Named for bean injection
518 			Class<?> ptc = pt.inner();
519 			if (beanName == null && ! hasBean(ptc))
520 				l.add(pt.getNameSimple());
521 			if (nn(beanName) && ! hasBean(ptc, beanName))
522 				l.add(pt.getNameSimple() + '@' + beanName);
523 		}
524 		return l.isEmpty() ? null : l.stream().sorted().collect(joining(","));
525 	}

528 	 * Returns the corresponding beans in this factory for the specified param types.
529 	 *
530 	 * @param executable The constructor or method to get the params for.
531 	 * @return The corresponding beans in this factory for the specified param types.
532 	 */
533 	public Object[] getParams(ExecutableInfo executable) {
534 		Object[] o = new Object[executable.getParameterCount()];
535 		for (int i = 0; i < executable.getParameterCount(); i++) {
536 			ParameterInfo pi = executable.getParameter(i);
537 			ClassInfo pt = pi.getParameterType();
538 			if (i == 0 && outer.isPresent() && pt.isInstance(outer.get())) {
539 				o[i] = outer.get();
540 			} else if (pt.is(BeanStore.class)) {
541 				o[i] = this;
542 			} else {
543 				String beanQualifier = pi.getResolvedQualifier();
544 				Class<?> ptc = pt.unwrap(Optional.class).inner();
545 				Optional<?> o2 = beanQualifier == null ? getBean(ptc) : getBean(ptc, beanQualifier);
546 				o[i] = pt.is(Optional.class) ? o2 : o2.orElse(null);
547 			}
548 		}
549 		return o;
550 	}

552 	/**
553 	 * Given the list of param types, returns <jk>true</jk> if this factory has all the parameters for the specified executable.
554 	 *
555 	 * @param executable The constructor or method to get the params for.
556 	 * @return A comma-delimited list of types that are missing from this factory.
557 	 */
558 	public boolean hasAllParams(ExecutableInfo executable) {
559 		loop: for (int i = 0; i < executable.getParameterCount(); i++) {
560 			ParameterInfo pi = executable.getParameter(i);
561 			ClassInfo pt = pi.getParameterType();
562 			if (i == 0 && outer.isPresent() && pt.isInstance(outer.get()))
563 				continue loop;
564 			if (pt.is(Optional.class) || pt.is(BeanStore.class))
565 				continue loop;
566 			String beanQualifier = pi.getResolvedQualifier();
567 			Class<?> ptc = pt.inner();
568 			if ((beanQualifier == null && ! hasBean(ptc)) || (nn(beanQualifier) && ! hasBean(ptc, beanQualifier)))
569 				return false;
570 		}
571 		return true;
572 	}

631 	 * @param <T> The bean type to return.
632 	 * @param beanType The bean type to return.
633 	 * @return The bean entries.  Never <jk>null</jk>.
634 	 */
635 	public <T> Stream<BeanStoreEntry<T>> stream(Class<T> beanType) {
636 		@SuppressWarnings("unchecked")
637 		Stream<BeanStoreEntry<T>> s = entries.stream().filter(x -> x.matches(beanType)).map(x -> (BeanStoreEntry<T>)x);
638 		if (parent.isPresent())
639 			s = Stream.concat(s, parent.get().stream(beanType));
640 		return s;
641 	}

649 		if (readOnly)
650 			throw new IllegalStateException("Method cannot be used because BeanStore is read-only.");

653 	private JsonMap properties() {
654 		Predicate<Boolean> nf = Utils::isTrue;
655 		// @formatter:off
656 		return filteredMap()
657 			.append("identity", identity(this))
658 			.append("entries", entries.stream().map(BeanStoreEntry::properties).collect(toList()))
659 			.append("outer", identity(outer.orElse(null)))
660 			.append("parent", parent.map(BeanStore::properties).orElse(null))
661 			.appendIf(nf, "readOnly", readOnly)
662 			.appendIf(nf, "threadSafe", threadSafe)
663 		;
664 		// @formatter:on
665 	}
