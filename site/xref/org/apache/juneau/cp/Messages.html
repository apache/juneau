35  /**
36   * An enhanced {@link ResourceBundle}.
37   *
38   * <p>
39   * Wraps a ResourceBundle to provide some useful additional functionality.
40   *
41   * <ul>
42   * 	<li>
43   * 		Instead of throwing {@link MissingResourceException}, the {@link ResourceBundle#getString(String)} method
44   * 		will return <js>"{!key}"</js> if the message could not be found.
45   * 	<li>
46   * 		Supported hierarchical lookup of resources from parent parent classes.
47   * 	<li>
48   * 		Support for easy retrieval of localized bundles.
49   * 	<li>
50   * 		Support for generalized resource bundles (e.g. properties files containing keys for several classes).
51   * </ul>
52   *
53   * <p>
54   * The following example shows the basic usage of this class for retrieving localized messages:
55   *
56   * <p class='bini'>
57   * 	<cc># Contents of MyClass.properties</cc>
58   * 	<ck>foo</ck> = <cv>foo {0}</cv>
59   * 	<ck>MyClass.bar</ck> = <cv>bar {0}</cv>
60   * </p>
61   * <p class='bjava'>
62   * 	<jk>public class</jk> MyClass {
63   * 		<jk>private static final</jk> Messages <jsf>MESSAGES</jsf> = Messages.<jsm>of</jsm>(MyClass.<jk>class</jk>);
64   *
65   * 		<jk>public void</jk> doFoo() {
66   *
67   *			<jc>// A normal property.</jc>
68   * 			String <jv>foo</jv> = <jsf>MESSAGES</jsf>.getString(<js>"foo"</js>,<js>"x"</js>);  <jc>// == "foo x"</jc>
69   *
70   * 			<jc>// A property prefixed by class name.</jc>
71   * 			String <jv>bar</jv> = <jsf>MESSAGES</jsf>.getString(<js>"bar"</js>,<js>"x"</js>);  <jc>// == "bar x"</jc>
72   *
73   * 			<jc>// A non-existent property.</jc>
74   * 			String <jv>baz</jv> = <jsf>MESSAGES</jsf>.getString(<js>"baz"</js>,<js>"x"</js>);  <jc>// == "{!baz}"</jc>
75   * 		}
76   * 	}
77   * </p>
78   *
79   * <p>
80   * 	The ability to resolve keys prefixed by class name allows you to place all your messages in a single file such
81   * 	as a common <js>"Messages.properties"</js> file along with those for other classes.
82   * <p>
83   * 	The following shows how to retrieve messages from a common bundle:
84   *
85   * <p class='bjava'>
86   * 	<jk>public class</jk> MyClass {
87   * 		<jk>private static final</jk> Messages <jsf>MESSAGES</jsf> = Messages.<jsm>of</jsm>(MyClass.<jk>class</jk>, <js>"Messages"</js>);
88   * 	}
89   * </p>
90   *
91   * <p>
92   * 	Resource bundles are searched using the following base name patterns:
93   * 	<ul>
94   * 		<li><js>"{package}.{name}"</js>
95   * 		<li><js>"{package}.i18n.{name}"</js>
96   * 		<li><js>"{package}.nls.{name}"</js>
97   * 		<li><js>"{package}.messages.{name}"</js>
98   * 	</ul>
99   *
100  * <p>
101  * 	These patterns can be customized using the {@link Builder#baseNames(String...)} method.
102  *
103  * <p>
104  * 	Localized messages can be retrieved in the following way:
105  *
106  * <p class='bjava'>
107  * 	<jc>// Return value from Japan locale bundle.</jc>
108  * 	String <jv>foo</jv> = <jsf>MESSAGES</jsf>.forLocale(Locale.<jsf>JAPAN</jsf>).getString(<js>"foo"</js>);
109  * </p>
110  *
111  * <h5 class='section'>See Also:</h5><ul>
112  * </ul>
113  */

251 		@SuppressWarnings("unchecked")
252 		@Override /* Overridden from BeanBuilder */
253 		protected Messages buildDefault() {
254 
255 			if (! locations.isEmpty()) {
256 				Tuple2<Class<?>,String>[] mbl = locations.toArray(new Tuple2[0]);
257 
258 				Builder x = null;
259 
260 				for (int i = mbl.length - 1; i >= 0; i--) {
261 					Class<?> c = Utils.firstNonNull(mbl[i].getA(), forClass);
262 					String value = mbl[i].getB();
263 					if (isJsonObject(value, true)) {
264 						MessagesString ms;
265 						try {
266 							ms = Json5.DEFAULT.read(value, MessagesString.class);
267 						} catch (ParseException e) {
268 							throw asRuntimeException(e);
269 						}
270 						x = Messages.create(c).name(ms.name).baseNames(StringUtils.splita(ms.baseNames, ',')).locale(ms.locale).parent(x == null ? null : x.build());
271 					} else {
272 						x = Messages.create(c).name(value).parent(x == null ? null : x.build());
273 					}
274 				}
275 
276 				return x == null ? null : x.build();  // Shouldn't be null.
277 			}
278 
279 			return new Messages(this);
280 		}

282 		ResourceBundle getBundle() {
283 			ClassLoader cl = forClass.getClassLoader();
284 			JsonMap m = JsonMap.of("name", name, "package", forClass.getPackage().getName());
285 			for (String bn : baseNames) {
286 				bn = StringUtils.replaceVars(bn, m);
287 				ResourceBundle rb = findBundle(bn, locale, cl);
288 				if (rb != null)
289 					return rb;
290 			}
291 			return null;
292 		}

354 	Messages(Class<?> forClass, ResourceBundle rb, Locale locale, Messages parent) {
355 		this.c = forClass;
356 		this.rb = rb;
357 		this.parent = parent;
358 		if (parent != null)
359 			setParent(parent);
360 		this.locale = locale == null ? Locale.getDefault() : locale;
361 
362 		Map<String,String> keyMap = new TreeMap<>();
363 
364 		String cn = c.getSimpleName() + '.';
365 		if (rb != null) {
366 			rb.keySet().forEach(x -> {
367 				keyMap.put(x, x);
368 				if (x.startsWith(cn)) {
369 					String shortKey = x.substring(cn.length());
370 					keyMap.put(shortKey, x);
371 				}
372 			});
373 		}
374 		if (parent != null) {
375 			parent.keySet().forEach(x -> {
376 				keyMap.put(x, x);
377 				if (x.startsWith(cn)) {
378 					String shortKey = x.substring(cn.length());
379 					keyMap.put(shortKey, x);
380 				}
381 			});
382 		}
383 
384 		this.keyMap = u(CollectionUtils.copyOf(keyMap));
385 		this.rbKeys = rb == null ? Collections.emptySet() : rb.keySet();
386 	}

394 	 * Looks for all the specified keys in the resource bundle and returns the first value that exists.
395 	 *
396 	 * @param keys The list of possible keys.
397 	 * @return The resolved value, or <jk>null</jk> if no value is found or the resource bundle is missing.
398 	 */
399 	public String findFirstString(String...keys) {
400 		for (String k : keys) {
401 			if (containsKey(k))
402 				return getString(k);
403 		}
404 		return null;
405 	}

407 	/**
408 	 * Returns this message bundle for the specified locale.
409 	 *
410 	 * @param locale The locale to get the messages for.
411 	 * @return A new {@link Messages} object.  Never <jk>null</jk>.
412 	 */
413 	public Messages forLocale(Locale locale) {
414 		if (locale == null)
415 			locale = Locale.getDefault();
416 		if (this.locale.equals(locale))
417 			return this;
418 		Messages mb = localizedMessages.get(locale);
419 		if (mb == null) {
420 			Messages parent = this.parent == null ? null : this.parent.forLocale(locale);
421 			ResourceBundle rb = this.rb == null ? null : findBundle(this.rb.getBaseBundleName(), locale, c.getClassLoader());
422 			mb = new Messages(c, rb, locale, parent);
423 			localizedMessages.put(locale, mb);
424 		}
425 		return mb;
426 	}

431 	/**
432 	 * Similar to {@link ResourceBundle#getString(String)} except allows you to pass in {@link MessageFormat} objects.
433 	 *
434 	 * @param key The resource bundle key.
435 	 * @param args Optional {@link MessageFormat}-style arguments.
436 	 * @return
437 	 * 	The resolved value.  Never <jk>null</jk>.
438 	 * 	<js>"{!key}"</js> if the key is missing.
439 	 */
440 	public String getString(String key, Object...args) {
441 		String s = getString(key);
442 		if (s.startsWith("{!"))
443 			return s;
444 		return format(s, args);
445 	}

452 	/**
453 	 * Returns all keys in this resource bundle with the specified prefix.
454 	 *
455 	 * <p>
456 	 * Keys are returned in alphabetical order.
457 	 *
458 	 * @param prefix The prefix.
459 	 * @return The set of all keys in the resource bundle with the prefix.
460 	 */
461 	public Set<String> keySet(String prefix) {
462 		Set<String> set = Utils.set();
463 		keySet().forEach(x -> {
464 			if (x.equals(prefix) || (x.startsWith(prefix) && x.charAt(prefix.length()) == '.'))
465 				set.add(x);
466 		});
467 		return set;
468 	}

474 			m.put(k, getString(k));
475 		return Json5.of(m);

478 	@Override /* Overridden from ResourceBundle */
479 	protected Object handleGetObject(String key) {
480 		String k = keyMap.get(key);
481 		if (k == null)
482 			return "{!" + key + "}";
483 		try {
484 			if (rbKeys.contains(k))
485 				return rb.getObject(k);
486 		} catch (@SuppressWarnings("unused") MissingResourceException e) { /* Shouldn't happen */ }
487 		return parent.handleGetObject(key);
488 	}
