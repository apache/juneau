17  package org.apache.juneau.svl.vars;
18  
19  import static org.apache.juneau.common.utils.Utils.*;
20  
21  import org.apache.juneau.collections.*;
22  import org.apache.juneau.svl.*;
23  
24  /**
25   * JVM args variable resolver.
26   *
27   * <p>
28   * The format for this var is <js>"$A{arg[,default]}"</js>.
29   *
30   * <p>
31   * This variable resolver requires that the command-line arguments be made available through any of the following:
32   * <ul class='spaced-list'>
33   * 	<li><js>"sun.java.command"</js> system property.
34   * 	<li><js>"juneau.args"</js> system property.
35   * 	<li>{@link #init(Args)} has been called.
36   * </ul>
37   *
38   * <h5 class='section'>Example:</h5>
39   * <p class='bjava'>
40   * 	<jc>// Create an args object from the main(String[]) method.</jc>
41   * 	Args <jv>args</jv> = <jk>new</jk> Args(<jv>argv</jv>);
42   *
43   * 	ArgsVar.<jsm>init</jsm>(<jv>args</jv>);
44   *
45   * 	<jc>// Create a variable resolver that resolves JVM arguments (e.g. "$A{1}")</jc>
46   * 	VarResolver <jv>varResolver</jv> = VarResolver.<jsm>create</jsm>().vars(ArgsVar.<jk>class</jk>).build();
47   *
48   * 	<jc>// Use it!</jc>
49   * 	System.<jsf>out</jsf>.println(<jv>varResolver</jv>.resolve(<js>"Arg #1 is set to $A{1}"</js>));
50   * </p>
51   *
52   * <p>
53   * Since this is a {@link SimpleVar}, any variables contained in the result will be recursively resolved.
54   * Likewise, if the arguments contain any variables, those will be resolved before they are passed to this var.
55   *
56   * <h5 class='section'>See Also:</h5><ul>
57   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SimpleVariableLanguageBasics">Simple Variable Language Basics</a>
58   * </ul>
59   */
60  public class ArgsVar extends DefaultingVar {
61  
62  	/** The name of this variable. */
63  	public static final String NAME = "A";
64  
65  	private static volatile Args ARGS;
66  
67  	/**
68  	 * Initialize the args for this variable.
69  	 *
70  	 * @param args The parsed command-line arguments.
71  	 */
72  	public static void init(Args args) {
73  		ARGS = args;
74  	}
75  
76  	private final Args args;
77  
78  	/**
79  	 * Constructor.
80  	 */
81  	public ArgsVar() {
82  		super(NAME);
83  		if (nn(ARGS))
84  			this.args = ARGS;
85  		else {
86  			String s = System.getProperty("sun.java.command");
87  			if (isNotEmpty(s)) {
88  				int i = s.indexOf(' ');
89  				args = new Args(i == -1 ? "" : s.substring(i + 1));
90  			} else {
91  				args = new Args(System.getProperty("juneau.args", ""));
92  			}
93  		}
94  	}
95  
96  	@Override /* Overridden from Var */
97  	public String resolve(VarResolverSession session, String key) {
98  		return args.getArg(key);
99  	}
100 }
