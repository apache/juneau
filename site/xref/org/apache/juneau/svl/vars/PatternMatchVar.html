17  package org.apache.juneau.svl.vars;
18  
19  import static org.apache.juneau.common.utils.AssertionUtils.*;
20  
21  import java.util.regex.*;
22  
23  import org.apache.juneau.svl.*;
24  
25  /**
26   * A logical variable resolver that resolves to <js>"true</js> if a pattern matches.
27   *
28   * <p>
29   * The format for this var is one of the following:
30   * <ul>
31   * 	<li><js>"$PM{stringArg, pattern}"</js>
32   * 	<li>...
33   * </ul>
34   *
35   * <p>
36   * The pattern can be any string optionally containing <js>'*'</js> or <js>'?'</js> representing any or one character
37   * respectively.
38   *
39   * <h5 class='section'>Example:</h5>
40   * <p class='bjava'>
41   * 	<jc>// Create a variable resolver that resolves system properties and $SW vars.</jc>
42   * 	VarResolver <jv>varResolver</jv> = VarResolver.<jsm>create</jsm>().vars(PatternMatchVar.<jk>class</jk>, SystemPropertiesVar.<jk>class</jk>).build();
43   *
44   * 	<jc>// Use it!</jc>
45   * 	System.<jsf>out</jsf>.println(<jv>varResolver</jv>.resolve(<js>"Running on Windows?  $PM{$P{os.name},*win*}!"</js>));
46   * </p>
47   *
48   * <p>
49   * Since this is a {@link MultipartVar}, any variables contained in the result will be recursively resolved.
50   * <br>Likewise, if the arguments contain any variables, those will be resolved before they are passed to this var.
51   *
52   * <h5 class='section'>See Also:</h5><ul>
53   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SimpleVariableLanguageBasics">Simple Variable Language Basics</a>
54   * </ul>
55   */
56  public class PatternMatchVar extends MultipartVar {
57  
58  	/** The name of this variable. */
59  	public static final String NAME = "PM";
60  
61  	/**
62  	 * Constructor.
63  	 */
64  	public PatternMatchVar() {
65  		super(NAME);
66  	}
67  
68  	@Override /* Overridden from MultipartVar */
69  	public String resolve(VarResolverSession session, String[] args) {
70  		assertArg(args.length >= 2, "Invalid number of arguments passed to $PM var.  Must have 2 or more arguments.");
71  
72  		String stringArg = args[0];
73  		String pattern = args[1];
74  		Pattern p = Pattern.compile(pattern.replace("*", ".*").replace("?", "."));
75  		return String.valueOf(p.matcher(stringArg).matches());
76  	}
77  }
