13  package org.apache.juneau.svl.vars;
14  
15  import org.apache.juneau.svl.*;
16  
17  /**
18   * A basic variable resolver that returns the first non-null value.
19   *
20   * <p>
21   * The format for this var is <js>"$CO{arg1[,arg2...]}"</js>.
22   *
23   * <h5 class='section'>See Also:</h5><ul>
24   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SimpleVariableLanguageBasics">Simple Variable Language</a>
25   * </ul>
26   */
27  public class CoalesceVar extends MultipartResolvingVar {
28  
29  	/** The name of this variable. */
30  	public static final String NAME = "CO";
31  
32  	/**
33  	 * Constructor.
34  	 */
35  	public CoalesceVar() {
36  		super(NAME);
37  	}
38  
39  	@Override
40  	public String resolve(VarResolverSession session, String arg) throws Exception {
41  		return arg;
42  	}
43  }
