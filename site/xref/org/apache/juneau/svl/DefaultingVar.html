13  package org.apache.juneau.svl;
14  
15  /**
16   * Interface for the resolution of vars with a default value if the <c>resolve()</c> method returns <jk>null</jk>.
17   *
18   * <p>
19   * For example, to resolve the system property <js>"myProperty"</js> but resolve to <js>"not found"</js> if the
20   * property doesn't exist: <js>"$S{myProperty,not found}"</js>
21   *
22   * <p>
23   * Subclasses must implement the following method:
24   * <ul class='javatree'>
25   * 	<li class='jm'>{@link #resolve(VarResolverSession, String)}
26   * </ul>
27   *
28   * <h5 class='section'>See Also:</h5><ul>
29   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SimpleVariableLanguageBasics">Simple Variable Language Basics</a>
30  
31   * </ul>
32   */
33  public abstract class DefaultingVar extends SimpleVar {
34  
35  	/**
36  	 * Constructor.
37  	 *
38  	 * @param name The name of this variable.
39  	 */
40  	public DefaultingVar(String name) {
41  		super(name);
42  	}
43  
44  	@Override /* Var*/
45  	public String doResolve(VarResolverSession session, String s) throws Exception {
46  		int i = s.indexOf(',');
47  		if (i == -1)
48  			return resolve(session, s.trim());
49  		String s1 = s.substring(0, i);
50  		String s2 = s.length() == i ? null : s.substring(i+1);
51  		String v = resolve(session, s1);
52  		if (v == null)
53  			v = s2;
54  		return v;
55  	}
56  }
