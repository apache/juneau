13  package org.apache.juneau.marshaller;
14  
15  import java.lang.reflect.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.parser.*;
19  import org.apache.juneau.serializer.*;
20  
21  /**
22   * A subclass of {@link Marshaller} for character-based serializers and parsers.
23   *
24   * <h5 class='section'>See Also:</h5><ul>
25   * 	<li class='link'><a class="doclink" href="../../../../index.html#jm.Marshallers">Marshallers</a>
26   * </ul>
27   */
28  public class CharMarshaller extends Marshaller {
29  
30  	//-----------------------------------------------------------------------------------------------------------------
31  	// Instance
32  	//-----------------------------------------------------------------------------------------------------------------
33  
34  	private final ReaderParser p;
35  	private final WriterSerializer s;
36  
37  	/**
38  	 * Constructor.
39  	 *
40  	 * @param s
41  	 * 	The serializer to use for serializing output.
42  	 * 	<br>Must not be <jk>null</jk>.
43  	 * @param p
44  	 * 	The parser to use for parsing input.
45  	 * 	<br>Must not be <jk>null</jk>.
46  	 */
47  	public CharMarshaller(WriterSerializer s, ReaderParser p) {
48  		super(s, p);
49  		this.s = s;
50  		this.p = p;
51  	}
52  
53  	/**
54  	 * Same as {@link #read(Object,Class)} but reads from a string and thus doesn't throw an <c>IOException</c>.
55  	 *
56  	 * <p>
57  	 * This is the preferred parse method for simple types since you don't need to cast the results.
58  	 *
59  	 * <h5 class='section'>Examples:</h5>
60  	 * <p class='bjava'>
61  	 * 	Marshaller <jv>marshaller</jv> = Json.<jsf>DEFAULT</jsf>;
62  	 *
63  	 * 	<jc>// Parse into a string.</jc>
64  	 * 	String <jv>string</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, String.<jk>class</jk>);
65  	 *
66  	 * 	<jc>// Parse into a bean.</jc>
67  	 * 	MyBean <jv>bean</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, MyBean.<jk>class</jk>);
68  	 *
69  	 * 	<jc>// Parse into a bean array.</jc>
70  	 * 	MyBean[] <jv>beanArray</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, MyBean[].<jk>class</jk>);
71  	 *
72  	 * 	<jc>// Parse into a linked-list of objects.</jc>
73  	 * 	List <jv>list</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, LinkedList.<jk>class</jk>);
74  	 *
75  	 * 	<jc>// Parse into a map of object keys/values.</jc>
76  	 * 	Map <jv>map</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, TreeMap.<jk>class</jk>);
77  	 * </p>
78  	 *
79  	 * @param <T> The class type of the object being created.
80  	 * @param input The input.
81  	 * @param type The object type to create.
82  	 * @return The parsed object.
83  	 * @throws ParseException Malformed input encountered.
84  	 */
85  	public final <T> T read(String input, Class<T> type) throws ParseException {
86  		return p.parse(input, type);
87  	}
88  
89  	/**
90  	 * Same as {@link #read(Object,Type,Type...)} but reads from a string and thus doesn't throw an <c>IOException</c>.
91  	 *
92  	 * @param <T> The class type of the object to create.
93  	 * @param input The input.
94  	 * @param type
95  	 * 	The object type to create.
96  	 * 	<br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
97  	 * @param args
98  	 * 	The type arguments of the class if it's a collection or map.
99  	 * 	<br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
100 	 * 	<br>Ignored if the main type is not a map or collection.
101 	 * @return The parsed object.
102 	 * @throws ParseException Malformed input encountered.
103 	 * @see BeanSession#getClassMeta(Type,Type...) for argument syntax for maps and collections.
104 	 */
105 	public final <T> T read(String input, Type type, Type...args) throws ParseException {
106 		return p.parse(input, type, args);
107 	}
108 
109 	/**
110 	 * Serializes a POJO directly to a <c>String</c>.
111 	 *
112 	 * @param object The object to serialize.
113 	 * @return
114 	 * 	The serialized object.
115 	 * @throws SerializeException If a problem occurred trying to convert the output.
116 	 */
117 	public final String write(Object object) throws SerializeException {
118 		return s.serializeToString(object);
119 	}
120 }
