13  package org.apache.juneau.html.annotation;
14  
15  import org.apache.juneau.*;
16  import org.apache.juneau.html.*;
17  import org.apache.juneau.reflect.*;
18  import org.apache.juneau.svl.*;
19  
20  /**
21   * Utility classes and methods for the {@link HtmlDocConfig @HtmlDocConfig} annotation.
22   *
23   * <h5 class='section'>See Also:</h5><ul>
24   * 	<li class='link'><a class="doclink" href="../../../../../index.html#jm.HtmlDetails">HTML Details</a>
25   * </ul>
26   */
27  public class HtmlDocConfigAnnotation {
28  
29  	/**
30  	 * Applies {@link HtmlDocConfig} annotations to a {@link org.apache.juneau.html.HtmlDocSerializer.Builder}.
31  	 */
32  	public static class SerializerApply extends AnnotationApplier<HtmlDocConfig,HtmlDocSerializer.Builder> {
33  
34  		/**
35  		 * Constructor.
36  		 *
37  		 * @param vr The resolver for resolving values in annotations.
38  		 */
39  		public SerializerApply(VarResolverSession vr) {
40  			super(HtmlDocConfig.class, HtmlDocSerializer.Builder.class, vr);
41  		}
42  
43  		@SuppressWarnings("unchecked")
44  		@Override
45  		public void apply(AnnotationInfo<HtmlDocConfig> ai, HtmlDocSerializer.Builder b) {
46  			HtmlDocConfig a = ai.inner();
47  
48  			strings(a.aside()).ifPresent(x -> b.aside(x));
49  			strings(a.footer()).ifPresent(x -> b.footer(x));
50  			strings(a.head()).ifPresent(x -> b.head(x));
51  			strings(a.header()).ifPresent(x -> b.header(x));
52  			strings(a.nav()).ifPresent(x -> b.nav(x));
53  			strings(a.navlinks()).ifPresent(x -> b.navlinks(x));
54  			strings(a.script()).ifPresent(x -> b.script(x));
55  			strings(a.style()).ifPresent(x -> b.style(x));
56  			strings(a.stylesheet()).ifPresent(x -> b.stylesheet(x));
57  			string(a.asideFloat()).filter(x -> ! "DEFAULT".equalsIgnoreCase(x)).map(AsideFloat::valueOf).ifPresent(x -> b.asideFloat(x));
58  			string(a.noResultsMessage()).ifPresent(x -> b.noResultsMessage(x));
59  			bool(a.nowrap()).ifPresent(x -> b.nowrap(x));
60  			bool(a.resolveBodyVars()).ifPresent(x -> b.resolveBodyVars(x));
61  			type(a.template()).ifPresent(x -> b.template(x));
62  			classes(a.widgets()).ifPresent(x -> b.widgets((Class<? extends HtmlWidget>[]) x));
63  		}
64  	}
65  }
