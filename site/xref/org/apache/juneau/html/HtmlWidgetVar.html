13  package org.apache.juneau.html;
14  
15  import org.apache.juneau.html.annotation.*;
16  import org.apache.juneau.svl.*;
17  
18  /**
19   * HTML widget variable resolver.
20   *
21   * <p>
22   * The format for this var is <js>"$W{widgetName}"</js>.
23   *
24   * <p>
25   * Widgets are simple class that produce some sort of string based on a passed-in HTTP request.
26   *
27   * <p>
28   * They're registered via the following mechanisms:
29   * <ul>
30   * 	<li>{@link HtmlDocConfig#widgets() @HtmlDocConfig(widgets)}
31   * </ul>
32   *
33   * <h5 class='section'>See Also:</h5><ul>
34   * 	<li class='link'><a class="doclink" href="../../../../index.html#jrs.HtmlWidgets">Widgets</a>
35   * 	<li class='link'><a class="doclink" href="../../../../index.html#jm.HtmlDetails">HTML Details</a>
36   * </ul>
37   */
38  public class HtmlWidgetVar extends SimpleVar {
39  
40  	/**
41  	 * The name of this variable.
42  	 */
43  	public static final String NAME = "W";
44  
45  	/**
46  	 * Constructor.
47  	 */
48  	public HtmlWidgetVar() {
49  		super(NAME);
50  	}
51  
52  	@Override /* Parameter */
53  	public String resolve(VarResolverSession session, String key) throws Exception {
54  		HtmlWidgetMap m = session.getBean(HtmlWidgetMap.class).orElseThrow(RuntimeException::new);
55  
56  		HtmlWidget w = m.get(key);
57  		if (w == null)
58  			return "unknown-widget-"+key;
59  
60  		return w.getHtml(session);
61  	}
62  
63  	@Override
64  	public boolean canResolve(VarResolverSession session) {
65  		return session.getBean(HtmlWidgetMap.class).isPresent();
66  	}
67  }
