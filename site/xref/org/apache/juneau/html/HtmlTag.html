28  /**
29   * Predefined tags that occur in the serialized output of the HTML serializer.
30   *
31   * <h5 class='section'>See Also:</h5><ul>
32   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HtmlBasics">HTML Basics</a>
33   * </ul>
34   */
35  enum HtmlTag {
36  
37  	// @formatter:off
38  	TABLE(1,"<table>"),
39  	TR(2,"<tr>"),
40  	TH(3,"<th>"),
41  	TD(4,"<td>"),
42  	UL(5,"<ul>"),
43  	LI(6,"<li>"),
44  	STRING(7,"<string>"),
45  	NUMBER(8,"<number>"),
46  	BOOLEAN(9,"<boolean>"),
47  	NULL(10,"<null>"),
48  	A(11,"<a>"),
49  	BR(12,"<br>"),		// newline
50  	FF(13,"<ff>"),		// formfeed
51  	BS(14,"<bs>"),		// backspace
52  	SP(17, "<sp>"),   // space
53  	P(18, "<p>"),
54  	HTML(19, "<html>"),
55  	xTABLE(-1,"</table>"),
56  	xTR(-2,"</tr>"),
57  	xTH(-3,"</th>"),
58  	xTD(-4,"</td>"),
59  	xUL(-5,"</ul>"),
60  	xLI(-6,"</li>"),
61  	xSTRING(-7,"</string>"),
62  	xNUMBER(-8,"</number>"),
63  	xBOOLEAN(-9,"</boolean>"),
64  	xNULL(-10,"</null>"),
65  	xA(-11,"</a>"),
66  	xBR(-12,"</br>"),
67  	xFF(-13,"</ff>"),
68  	xBS(-14,"</bs>"),
69  	xSP(-17, "</sp>"),
70  	xP(-18, "</p>"),
71  	xHTML(-19, "</html>");
72  	// @formatter:on
73  
74  	static HtmlTag forEvent(ParserSession session, XMLStreamReader r) throws ParseException {
75  		int et = r.getEventType();
76  		if (et == START_ELEMENT)
77  			return forString(r.getLocalName(), false);
78  		else if (et == END_ELEMENT)
79  			return forString(r.getLocalName(), true);
80  		throw new ParseException(session, "Invalid call to HtmlTag.forEvent on event of type ''{0}''", XmlUtils.toReadableEvent(r));
81  	}
82  
83  	static HtmlTag forString(String tag, boolean end) {
84  		char c = tag.charAt(0);
85  		HtmlTag t = null;
86  		if (c == 'u')
87  			t = (end ? xUL : UL);
88  		else if (c == 'l')
89  			t = (end ? xLI : LI);
90  		else if (c == 's') {
91  			c = tag.charAt(1);
92  			if (c == 'p')
93  				t = (end ? xSP : SP);
94  			else if (c == 't')
95  				t = (end ? xSTRING : STRING);
96  		} else if (c == 'b') {
97  			c = tag.charAt(1);
98  			if (c == 'o')
99  				t = (end ? xBOOLEAN : BOOLEAN);
100 			else if (c == 'r')
101 				t = (end ? xBR : BR);
102 			else if (c == 's')
103 				t = (end ? xBS : BS);
104 		} else if (c == 'a')
105 			t = (end ? xA : A);
106 		else if (c == 'n') {
107 			c = tag.charAt(2);
108 			if (c == 'm')
109 				t = (end ? xNUMBER : NUMBER);
110 			else if (c == 'l')
111 				t = (end ? xNULL : NULL);
112 		} else if (c == 't') {
113 			c = tag.charAt(1);
114 			if (c == 'a')
115 				t = (end ? xTABLE : TABLE);
116 			else if (c == 'r')
117 				t = (end ? xTR : TR);
118 			else if (c == 'h')
119 				t = (end ? xTH : TH);
120 			else if (c == 'd')
121 				t = (end ? xTD : TD);
122 		} else if (c == 'f')
123 			t = (end ? xFF : FF);
124 		else if (c == 'p')
125 			t = (end ? xP : P);
126 		else if (c == 'h')
127 			t = (end ? xHTML : HTML);
128 		return t;
129 	}

131 	private Map<Integer,HtmlTag> cache = new HashMap<>();
132 
133 	int id;
134 
135 	String label;
136 
137 	HtmlTag(int id, String label) {
138 		this.id = id;
139 		this.label = label;
140 		cache.put(id, this);
141 	}
142 
143 	public boolean isOneOf(HtmlTag...tags) {
144 		for (HtmlTag tag : tags)
145 			if (tag == this)
146 				return true;
147 		return false;
148 	}
149 
150 	@Override /* Overridden from Object */
151 	public String toString() {
152 		return label;
153 	}
154 }
