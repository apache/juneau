219 	/**
220 	 * Returns the anchor text to use for the specified URL object.
221 	 *
222 	 * @param pMeta
223 	 * 	The property metadata of the bean property of the object.
224 	 * 	Can be <jk>null</jk> if the object isn't from a bean property.
225 	 * @param o The URL object.
226 	 * @return The anchor text to use for the specified URL object.
227 	 */
228 	public String getAnchorText(BeanPropertyMeta pMeta, Object o) {
229 		String s = o.toString();
230 		if (isDetectLabelParameters()) {
231 			Matcher m = labelPattern.matcher(s);
232 			if (m.find())
233 				return urlDecode(m.group(1));
234 		}
235 		switch (getUriAnchorText()) {
236 			case LAST_TOKEN:
237 				s = resolveUri(s);
238 				if (s.indexOf('/') != -1)
239 					s = s.substring(s.lastIndexOf('/') + 1);
240 				if (s.indexOf('?') != -1)
241 					s = s.substring(0, s.indexOf('?'));
242 				if (s.indexOf('#') != -1)
243 					s = s.substring(0, s.indexOf('#'));
244 				if (s.isEmpty())
245 					s = "/";
246 				return urlDecode(s);
247 			case URI_ANCHOR:
248 				if (s.indexOf('#') != -1)
249 					s = s.substring(s.lastIndexOf('#') + 1);
250 				return urlDecode(s);
251 			case PROPERTY_NAME:
252 				return pMeta == null ? s : pMeta.getName();
253 			case URI:
254 				return resolveUri(s);
255 			case CONTEXT_RELATIVE:
256 				return relativizeUri("context:/", s);
257 			case SERVLET_RELATIVE:
258 				return relativizeUri("servlet:/", s);
259 			case PATH_RELATIVE:
260 				return relativizeUri("request:/", s);
261 			default /* TO_STRING */:
262 				return s;
263 		}
264 	}

270 	 * Returns <jk>true</jk> if the specified object is a URL.

272 	 * @param cm The ClassMeta of the object being serialized.
273 	 * @param pMeta
274 	 * 	The property metadata of the bean property of the object.
275 	 * 	Can be <jk>null</jk> if the object isn't from a bean property.
276 	 * @param o The object.
277 	 * @return <jk>true</jk> if the specified object is a URL.
278 	 */
279 	public boolean isUri(ClassMeta<?> cm, BeanPropertyMeta pMeta, Object o) {
280 		if (cm.isUri() || (nn(pMeta) && pMeta.isUri()))
281 			return true;
282 		if (isDetectLinksInStrings() && o instanceof CharSequence && urlPattern.matcher(o.toString()).matches())
283 			return true;
284 		return false;
285 	}

333 	/*
334 	 * Returns the table column headers for the specified collection of objects.
335 	 * Returns null if collection should not be serialized as a 2-dimensional table.
336 	 * Returns an empty array if it should be treated as a table but without headers.
337 	 * 2-dimensional tables are used for collections of objects that all have the same set of property names.
338 	 */
339 	@SuppressWarnings({ "rawtypes", "unchecked" })
340 	private Object[] getTableHeaders(Collection c, HtmlBeanPropertyMeta bpHtml) throws SerializeException {
341 
342 		if (c.isEmpty())
343 			return null;
344 
345 		c = sort(c);
346 
347 		Object o1 = null;
348 		for (var o : c)
349 			if (nn(o)) {
350 				o1 = o;
351 				break;
352 			}
353 		if (o1 == null)
354 			return null;
355 
356 		ClassMeta<?> cm1 = getClassMetaForObject(o1);
357 
358 		ObjectSwap swap = cm1.getSwap(this);
359 		o1 = swap(swap, o1);
360 		if (nn(swap))
361 			cm1 = swap.getSwapClassMeta(this);
362 
363 		if (cm1 == null || ! cm1.isMapOrBean() || cm1.hasAnnotation(HtmlLink.class))
364 			return null;
365 
366 		HtmlClassMeta cHtml = getHtmlClassMeta(cm1);
367 
368 		if (cHtml.isNoTables() || bpHtml.isNoTables() || cHtml.isXml() || bpHtml.isXml() || canIgnoreValue(cm1, null, o1))
369 			return null;
370 
371 		if (cHtml.isNoTableHeaders() || bpHtml.isNoTableHeaders())
372 			return new Object[0];
373 
374 		// If it's a non-bean map, only use table if all entries are also maps.
375 		if (cm1.isMap() && ! cm1.isBeanMap()) {
376 
377 			Set<Object> set = set();
378 			for (var o : c) {
379 				o = swap(swap, o);
380 				if (! canIgnoreValue(cm1, null, o)) {
381 					if (! cm1.isInstance(o))
382 						return null;
383 					forEachEntry((Map)o, x -> set.add(x.getKey()));
384 				}
385 			}
386 			return set.toArray(new Object[set.size()]);
387 		}
388 
389 		// Must be a bean or BeanMap.
390 		for (var o : c) {
391 			o = swap(swap, o);
392 			if (! canIgnoreValue(cm1, null, o)) {
393 				if (! cm1.isInstance(o))
394 					return null;
395 			}
396 		}
397 
398 		BeanMap<?> bm = toBeanMap(o1);
399 		return bm.keySet().toArray(new String[bm.size()]);
400 	}

402 	private void serializeBeanMap(XmlWriter out, BeanMap<?> m, ClassMeta<?> eType, BeanPropertyMeta ppMeta) throws SerializeException {
403 
404 		HtmlClassMeta cHtml = getHtmlClassMeta(m.getClassMeta());
405 		HtmlBeanPropertyMeta bpHtml = getHtmlBeanPropertyMeta(ppMeta);
406 
407 		int i = indent;
408 
409 		out.oTag(i, "table");
410 
411 		String typeName = m.getMeta().getDictionaryName();
412 		if (nn(typeName) && eType != m.getClassMeta())
413 			out.attr(getBeanTypePropertyName(m.getClassMeta()), typeName);
414 
415 		out.w('>').nl(i);
416 		if (isAddKeyValueTableHeaders() && ! (cHtml.isNoTableHeaders() || bpHtml.isNoTableHeaders())) {
417 			out.sTag(i + 1, "tr").nl(i + 1);
418 			out.sTag(i + 2, "th").append("key").eTag("th").nl(i + 2);
419 			out.sTag(i + 2, "th").append("value").eTag("th").nl(i + 2);
420 			out.ie(i + 1).eTag("tr").nl(i + 1);
421 		}
422 
423 		Predicate<Object> checkNull = x -> isKeepNullProperties() || nn(x);
424 
425 		m.forEachValue(checkNull, (pMeta, key, value, thrown) -> {
426 			ClassMeta<?> cMeta = pMeta.getClassMeta();
427 
428 			if (nn(thrown))
429 				onBeanGetterException(pMeta, thrown);
430 
431 			if (canIgnoreValue(cMeta, key, value))
432 				return;
433 
434 			String link = null, anchorText = null;
435 			if (! cMeta.isCollectionOrArray()) {
436 				link = m.resolveVars(getLink(pMeta));
437 				anchorText = m.resolveVars(getAnchorText(pMeta));
438 			}
439 
440 			if (nn(anchorText))
441 				value = anchorText;
442 
443 			out.sTag(i + 1, "tr").nl(i + 1);
444 			out.sTag(i + 2, "td").text(key).eTag("td").nl(i + 2);
445 			out.oTag(i + 2, "td");
446 			String style = getStyle(this, pMeta, value);
447 			if (nn(style))
448 				out.attr("style", style);
449 			out.cTag();
450 
451 			try {
452 				if (nn(link))
453 					out.oTag(i + 3, "a").attrUri("href", link).cTag();
454 				ContentResult cr = serializeAnything(out, value, cMeta, key, pMeta, 2, false, true);
455 				if (cr == CR_ELEMENTS)
456 					out.i(i + 2);
457 				if (nn(link))
458 					out.eTag("a");
459 			} catch (SerializeException | Error e) {
460 				throw e;
461 			} catch (Throwable e) {
462 				onBeanGetterException(pMeta, e);
463 			}
464 			out.eTag("td").nl(i + 2);
465 			out.ie(i + 1).eTag("tr").nl(i + 1);
466 		});
467 
468 		out.ie(i).eTag("table").nl(i);
469 	}

471 	@SuppressWarnings({ "rawtypes", "unchecked" })
472 	private void serializeCollection(XmlWriter out, Object in, ClassMeta<?> sType, ClassMeta<?> eType, String name, BeanPropertyMeta ppMeta) throws SerializeException {
473 
474 		HtmlClassMeta cHtml = getHtmlClassMeta(sType);
475 		HtmlBeanPropertyMeta bpHtml = getHtmlBeanPropertyMeta(ppMeta);
476 
477 		Collection c = (sType.isCollection() ? (Collection)in : toList(sType.getInnerClass(), in));
478 
479 		boolean isCdc = cHtml.isHtmlCdc() || bpHtml.isHtmlCdc();
480 		boolean isSdc = cHtml.isHtmlSdc() || bpHtml.isHtmlSdc();
481 		boolean isDc = isCdc || isSdc;
482 
483 		int i = indent;
484 		if (c.isEmpty()) {
485 			out.appendln(i, "<ul></ul>");
486 			return;
487 		}
488 
489 		String type2 = null;
490 		if (sType != eType)
491 			type2 = sType.getDictionaryName();
492 		if (type2 == null)
493 			type2 = "array";
494 
495 		c = sort(c);
496 
497 		String btpn = getBeanTypePropertyName(eType);
498 
499 		// Look at the objects to see how we're going to handle them.  Check the first object to see how we're going to
500 		// handle this.
501 		// If it's a map or bean, then we'll create a table.
502 		// Otherwise, we'll create a list.
503 		Object[] th = getTableHeaders(c, bpHtml);
504 
505 		if (nn(th)) {
506 
507 			out.oTag(i, "table").attr(btpn, type2).w('>').nl(i + 1);
508 			if (th.length > 0) {
509 				out.sTag(i + 1, "tr").nl(i + 2);
510 				for (var key : th) {
511 					out.sTag(i + 2, "th");
512 					out.text(convertToType(key, String.class));
513 					out.eTag("th").nl(i + 2);
514 				}
515 				out.ie(i + 1).eTag("tr").nl(i + 1);
516 			} else {
517 				th = null;
518 			}
519 
520 			for (var o : c) {
521 				ClassMeta<?> cm = getClassMetaForObject(o);
522 
523 				if (nn(cm) && nn(cm.getSwap(this))) {
524 					ObjectSwap swap = cm.getSwap(this);
525 					o = swap(swap, o);
526 					cm = swap.getSwapClassMeta(this);
527 				}
528 
529 				out.oTag(i + 1, "tr");
530 				String typeName = (cm == null ? null : cm.getDictionaryName());
531 				String typeProperty = getBeanTypePropertyName(cm);
532 
533 				if (nn(typeName) && eType.getElementType() != cm)
534 					out.attr(typeProperty, typeName);
535 				out.cTag().nl(i + 2);
536 
537 				if (cm == null) {
538 					out.i(i + 2);
539 					serializeAnything(out, o, null, null, null, 1, false, false);
540 					out.nl(0);
541 
542 				} else if (cm.isMap() && ! (cm.isBeanMap())) {
543 					Map m2 = sort((Map)o);
544 
545 					if (th == null)
546 						th = m2.keySet().toArray(new Object[m2.size()]);
547 
548 					for (var k : th) {
549 						out.sTag(i + 2, "td");
550 						ContentResult cr = serializeAnything(out, m2.get(k), eType.getElementType(), toString(k), null, 2, false, true);
551 						if (cr == CR_ELEMENTS)
552 							out.i(i + 2);
553 						out.eTag("td").nl(i + 2);
554 					}
555 				} else {
556 					BeanMap m2 = toBeanMap(o);
557 
558 					if (th == null)
559 						th = m2.keySet().toArray(new Object[m2.size()]);
560 
561 					for (var k : th) {
562 						BeanMapEntry p = m2.getProperty(toString(k));
563 						BeanPropertyMeta pMeta = p.getMeta();
564 						if (pMeta.canRead()) {
565 							Object value = p.getValue();
566 
567 							String link = null, anchorText = null;
568 							if (! pMeta.getClassMeta().isCollectionOrArray()) {
569 								link = m2.resolveVars(getLink(pMeta));
570 								anchorText = m2.resolveVars(getAnchorText(pMeta));
571 							}
572 
573 							if (nn(anchorText))
574 								value = anchorText;
575 
576 							String style = getStyle(this, pMeta, value);
577 							out.oTag(i + 2, "td");
578 							if (nn(style))
579 								out.attr("style", style);
580 							out.cTag();
581 							if (nn(link))
582 								out.oTag("a").attrUri("href", link).cTag();
583 							ContentResult cr = serializeAnything(out, value, pMeta.getClassMeta(), p.getKey().toString(), pMeta, 2, false, true);
584 							if (cr == CR_ELEMENTS)
585 								out.i(i + 2);
586 							if (nn(link))
587 								out.eTag("a");
588 							out.eTag("td").nl(i + 2);
589 						}
590 					}
591 				}
592 				out.ie(i + 1).eTag("tr").nl(i + 1);
593 			}
594 			out.ie(i).eTag("table").nl(i);
595 
596 		} else {
597 			out.oTag(i, isDc ? "p" : "ul");
598 			if (! type2.equals("array"))
599 				out.attr(btpn, type2);
600 			out.w('>').nl(i + 1);
601 			boolean isFirst = true;
602 			for (var o : c) {
603 				if (isDc && ! isFirst)
604 					out.append(isCdc ? ", " : " ");
605 				if (! isDc)
606 					out.oTag(i + 1, "li");
607 				String style = getStyle(this, ppMeta, o);
608 				String link = getLink(ppMeta);
609 				if (nn(style) && ! isDc)
610 					out.attr("style", style);
611 				if (! isDc)
612 					out.cTag();
613 				if (nn(link))
614 					out.oTag(i + 2, "a").attrUri("href", link.replace("{#}", s(o))).cTag();
615 				ContentResult cr = serializeAnything(out, o, eType.getElementType(), name, null, 1, false, true);
616 				if (nn(link))
617 					out.eTag("a");
618 				if (cr == CR_ELEMENTS)
619 					out.ie(i + 1);
620 				if (! isDc)
621 					out.eTag("li").nl(i + 1);
622 				isFirst = false;
623 			}
624 			out.ie(i).eTag(isDc ? "p" : "ul").nl(i);

792 	/**
793 	 * Serialize the specified object to the specified writer.
794 	 *
795 	 * @param out The writer.
796 	 * @param o The object to serialize.
797 	 * @param eType The expected type of the object if this is a bean property.
798 	 * @param name
799 	 * 	The attribute name of this object if this object was a field in a JSON object (i.e. key of a
800 	 * 	{@link java.util.Map.Entry} or property name of a bean).
801 	 * @param pMeta The bean property being serialized, or <jk>null</jk> if we're not serializing a bean property.
802 	 * @param xIndent The current indentation value.
803 	 * @param isRoot <jk>true</jk> if this is the root element of the document.
804 	 * @param nlIfElement <jk>true</jk> if we should add a newline to the output before serializing only if the object is an element and not text.
805 	 * @return The type of content encountered.  Either simple (no whitespace) or normal (elements with whitespace).
806 	 * @throws SerializeException Generic serialization error occurred.
807 	 */
808 	@SuppressWarnings({ "rawtypes", "unchecked", "null" })
809 	protected ContentResult serializeAnything(XmlWriter out, Object o, ClassMeta<?> eType, String name, BeanPropertyMeta pMeta, int xIndent, boolean isRoot, boolean nlIfElement)
810 		throws SerializeException {
811 
812 		ClassMeta<?> aType = null;       // The actual type
813 		ClassMeta<?> wType = null;     // The wrapped type (delegate)
814 		ClassMeta<?> sType = object();   // The serialized type
815 
816 		var addJsonTags = isAddJsonTags();
817 
818 		if (eType == null)
819 			eType = object();
820 
821 		aType = push2(name, o, eType);
822 
823 		// Handle recursion
824 		if (aType == null) {
825 			o = null;
826 			aType = object();
827 		}
828 
829 		// Handle Optional<X>
830 		if (isOptional(aType)) {
831 			o = getOptionalValue(o);
832 			eType = getOptionalType(eType);
833 			aType = getClassMetaForObject(o, object());
834 		}
835 
836 		indent += xIndent;
837 
838 		ContentResult cr = CR_ELEMENTS;
839 
840 		// Determine the type.
841 		if (o == null || (aType.isChar() && ((Character)o).charValue() == 0)) {
842 			if (addJsonTags)
843 				out.tag("null");
844 			cr = ContentResult.CR_MIXED;
845 
846 		} else {
847 
848 			if (aType.isDelegate()) {
849 				wType = aType;
850 				aType = ((Delegate)o).getClassMeta();
851 			}
852 
853 			sType = aType;
854 
855 			String typeName = null;
856 			if (isAddBeanTypes() && ! eType.equals(aType))
857 				typeName = aType.getDictionaryName();
858 
859 			// Swap if necessary
860 			ObjectSwap swap = aType.getSwap(this);
861 			if (nn(swap)) {
862 				o = swap(swap, o);
863 				sType = swap.getSwapClassMeta(this);
864 
865 				// If the getSwapClass() method returns Object, we need to figure out
866 				// the actual type now.
867 				if (sType.isObject())
868 					sType = getClassMetaForObject(o);
869 			}
870 
871 			// Handle the case where we're serializing a raw stream.
872 			if (sType.isReader() || sType.isInputStream()) {
873 				pop();
874 				indent -= xIndent;
875 				if (sType.isReader())
876 					pipe((Reader)o, out, SerializerSession::handleThrown);
877 				else
878 					pipe((InputStream)o, out, SerializerSession::handleThrown);
879 				return ContentResult.CR_MIXED;
880 			}
881 
882 			HtmlClassMeta cHtml = getHtmlClassMeta(sType);
883 			HtmlBeanPropertyMeta bpHtml = getHtmlBeanPropertyMeta(pMeta);
884 
885 			HtmlRender render = firstNonNull(bpHtml.getRender(), cHtml.getRender());
886 
887 			if (nn(render)) {
888 				Object o2 = render.getContent(this, o);
889 				if (o2 != o) {
890 					indent -= xIndent;
891 					pop();
892 					out.nl(indent);
893 					return serializeAnything(out, o2, null, typeName, null, xIndent, false, false);
894 				}
895 			}
896 
897 			if (cHtml.isXml() || bpHtml.isXml()) {
898 				pop();
899 				indent++;
900 				if (nlIfElement)
901 					out.nl(0);
902 				super.serializeAnything(out, o, null, null, null, null, false, XmlFormat.MIXED, false, false, null);
903 				indent -= xIndent + 1;
904 				return cr;
905 
906 			} else if (cHtml.isPlainText() || bpHtml.isPlainText()) {
907 				out.w(o == null ? "null" : o.toString());
908 				cr = CR_MIXED;
909 
910 			} else if (o == null || (sType.isChar() && ((Character)o).charValue() == 0)) {
911 				if (addJsonTags)
912 					out.tag("null");
913 				cr = CR_MIXED;
914 
915 			} else if (sType.isNumber()) {
916 				if (eType.isNumber() && ! (isRoot && addJsonTags))
917 					out.append(o);
918 				else
919 					out.sTag("number").append(o).eTag("number");
920 				cr = CR_MIXED;
921 
922 			} else if (sType.isBoolean()) {
923 				if (eType.isBoolean() && ! (isRoot && addJsonTags))
924 					out.append(o);
925 				else
926 					out.sTag("boolean").append(o).eTag("boolean");
927 				cr = CR_MIXED;
928 
929 			} else if (sType.isMap() || (nn(wType) && wType.isMap())) {
930 				out.nlIf(! isRoot, xIndent + 1);
931 				if (o instanceof BeanMap)
932 					serializeBeanMap(out, (BeanMap)o, eType, pMeta);
933 				else
934 					serializeMap(out, (Map)o, sType, eType.getKeyType(), eType.getValueType(), typeName, pMeta);
935 
936 			} else if (sType.isBean()) {
937 				BeanMap m = toBeanMap(o);
938 				if (aType.hasAnnotation(HtmlLink.class)) {
939 					Value<String> uriProperty = Value.empty(), nameProperty = Value.empty();
940 					aType.forEachAnnotation(HtmlLink.class, x -> isNotEmpty(x.uriProperty()), x -> uriProperty.set(x.uriProperty()));
941 					aType.forEachAnnotation(HtmlLink.class, x -> isNotEmpty(x.nameProperty()), x -> nameProperty.set(x.nameProperty()));
942 					Object urlProp = m.get(uriProperty.orElse(""));
943 					Object nameProp = m.get(nameProperty.orElse(""));
944 
945 					out.oTag("a").attrUri("href", urlProp).w('>').text(nameProp).eTag("a");
946 					cr = CR_MIXED;
947 				} else {
948 					out.nlIf(! isRoot, xIndent + 2);
949 					serializeBeanMap(out, m, eType, pMeta);
950 				}
951 
952 			} else if (sType.isCollection() || sType.isArray() || (nn(wType) && wType.isCollection())) {
953 				out.nlIf(! isRoot, xIndent + 1);
954 				serializeCollection(out, o, sType, eType, name, pMeta);
955 
956 			} else if (isUri(sType, pMeta, o)) {
957 				String label = getAnchorText(pMeta, o);
958 				out.oTag("a").attrUri("href", o).w('>');
959 				out.text(label);
960 				out.eTag("a");
961 				cr = CR_MIXED;
962 
963 			} else {
964 				if (isRoot && addJsonTags)
965 					out.sTag("string").text(toString(o)).eTag("string");
966 				else
967 					out.text(toString(o));
968 				cr = CR_MIXED;
969 			}
970 		}
971 		pop();
972 		indent -= xIndent;
973 		return cr;
974 	}

979 		boolean isMixed, boolean preserveWhitespace, BeanPropertyMeta pMeta) throws SerializeException {

998

1000 			return serializeAnything(out, o, eType, elementName, pMeta, 0, false, false);
1001 
1002 		return super.serializeAnything(out, o, eType, keyName, elementName, elementNamespace, addNamespaceUris, format, isMixed, preserveWhitespace, pMeta);
1003 	}
