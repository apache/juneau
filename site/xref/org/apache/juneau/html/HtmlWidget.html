17  package org.apache.juneau.html;
18  
19  import org.apache.juneau.svl.*;
20  
21  /**
22   * Defines an interface for resolvers of <js>"$W{...}"</js> string variables.
23   *
24   * <p>
25   * Widgets must provide one of the following public constructors:
26   * <ul>
27   * 	<li><code><jk>public</jk> Widget();</code>
28   * </ul>
29   *
30   * <p>
31   * Widgets can be defined as inner classes of REST resource classes.
32   *
33   * <h5 class='section'>See Also:</h5><ul>
34   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HtmlWidgets">Widgets</a>
35   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HtmlBasics">HTML Basics</a>
36   * </ul>
37   */
38  public interface HtmlWidget {
39  
40  	/**
41  	 * The name for this widget.
42  	 *
43  	 * @return A unique identifying name for this widget.
44  	 */
45  	String getName();
46  
47  	/**
48  	 * Resolves the HTML content for this widget.
49  	 *
50  	 * <p>
51  	 * A returned value of <jk>null</jk> will cause nothing to be added to the page.
52  	 *
53  	 * @param session The current serializer session.
54  	 * @return The HTML content of this widget.
55  	 */
56  	String getHtml(VarResolverSession session);
57  
58  	/**
59  	 * Resolves any Javascript that should be added to the <xt>&lt;head&gt;/&lt;script&gt;</xt> element.
60  	 *
61  	 * <p>
62  	 * A returned value of <jk>null</jk> will cause nothing to be added to the page.
63  	 *
64  	 * @param session The current serializer session.
65  	 * @return The Javascript needed by this widget.
66  	 */
67  	String getScript(VarResolverSession session);
68  
69  	/**
70  	 * Resolves any CSS styles that should be added to the <xt>&lt;head&gt;/&lt;style&gt;</xt> element.
71  	 *
72  	 * <p>
73  	 * A returned value of <jk>null</jk> will cause nothing to be added to the page.
74  	 *
75  	 * @param session The current serializer session.
76  	 * @return The CSS styles needed by this widget.
77  	 */
78  	String getStyle(VarResolverSession session);
79  }
