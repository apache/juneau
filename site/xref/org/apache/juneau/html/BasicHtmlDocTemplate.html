26  /**
27   * A basic template for the HTML doc serializer.
28   *
29   * <p>
30   * This class can be subclassed to customize page rendering.
31   *
32   * <h5 class='section'>See Also:</h5><ul>
33   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HtmlBasics">HTML Basics</a>
34   * </ul>
35   */
36  @SuppressWarnings("resource")
37  public class BasicHtmlDocTemplate implements HtmlDocTemplate {

39  	private static boolean exists(String s) {
40  		return nn(s) && ! "NONE".equals(s);
41  	}
42  
43  	@Override /* Overridden from HtmlDocTemplate */
44  	public void writeTo(HtmlDocSerializerSession session, HtmlWriter w, Object o) throws Exception {
45  		w.sTag("html").nl(0);
46  		w.sTag(1, "head").nl(1);
47  		head(session, w, o);
48  		w.eTag(1, "head").nl(1);
49  		w.sTag(1, "body").nl(1);
50  		body(session, w, o);
51  		w.eTag(1, "body").nl(1);
52  		w.eTag("html").nl(0);
53  	}

55  	/**
56  	 * Renders the contents of the <code><xt>&lt;body&gt;</xt>/<xt>&lt;article&gt;</xt></code> element.
57  	 *
58  	 * @param session The current serializer session.
59  	 * @param w The writer being written to.
60  	 * @param o The object being serialized.
61  	 * @throws Exception Any exception can be thrown.
62  	 */
63  	protected void article(HtmlDocSerializerSession session, HtmlWriter w, Object o) throws Exception {
64  		// To allow for page formatting using CSS, we encapsulate the data inside two div tags:
65  		// <div class='outerdata'><div class='data' id='data'>...</div></div>
66  		w.oTag(4, "div").attr("class", "outerdata").append('>').nl(4);
67  		w.oTag(5, "div").attr("class", "data").attr("id", "data").append('>').nl(5);
68  
69  		if (o == null) {
70  			w.append(6, "<null/>").nl(6);
71  		} else if (isEmpty(o)) {
72  			String m = session.getNoResultsMessage();
73  			if (exists(m))
74  				w.append(6, session.resolve(m)).nl(6);
75  		} else {
76  			session.indent = 6;
77  			w.flush();
78  			if (session.isResolveBodyVars()) {
79  				w = new HtmlWriter(w) {
80  					@Override
81  					public HtmlWriter text(Object value, boolean preserveWhitespace) {
82  						return super.text(session.resolve(Utils.s(value)), preserveWhitespace);
83  					}
84  				};
85  			}
86  			session.parentSerialize(w, o);
87  		}
88  
89  		w.ie(5).eTag("div").nl(5);
90  		w.ie(4).eTag("div").nl(4);
91  	}

101 	protected void aside(HtmlDocSerializerSession session, HtmlWriter w, Object o) throws Exception {
102 		String[] aside = session.getAside();
103 		for (int i = 0; i < aside.length; i++)
104 			w.sIf(i > 0).appendln(4, session.resolve(aside[i]));
105 	}

107 	/**
108 	 * Renders the contents of the <code><xt>&lt;body&gt;</xt></code> element.
109 	 *
110 	 * @param session The current serializer session.
111 	 * @param w The writer being written to.
112 	 * @param o The object being serialized.
113 	 * @throws Exception Any exception can be thrown.
114 	 */
115 	protected void body(HtmlDocSerializerSession session, HtmlWriter w, Object o) throws Exception {
116 
117 		AsideFloat asideFloat = session.getAsideFloat();
118 		boolean hasAside = hasAside(session);
119 
120 		if (hasHeader(session)) {
121 			w.sTag(2, "header").nl(2);
122 			header(session, w, o);
123 			w.ie(2).eTag("header").nl(2);
124 		}
125 
126 		if (hasNav(session)) {
127 			w.sTag(2, "nav").nl(2);
128 			nav(session, w, o);
129 			w.ie(2).eTag("nav").nl(2);
130 		}
131 
132 		if (hasAside && asideFloat.is(TOP)) {
133 			w.sTag(2, "section").nl(2);
134 			w.sTag(3, "aside").nl(3);
135 			aside(session, w, o);
136 			w.ie(3).eTag("aside").nl(3);
137 			w.ie(2).eTag("section").nl(2);
138 		}
139 
140 		w.sTag(2, "section").nl(2);
141 
142 		if (hasAside && asideFloat.is(LEFT)) {
143 			w.sTag(3, "aside").nl(3);
144 			aside(session, w, o);
145 			w.ie(3).eTag("aside").nl(3);
146 		}
147 
148 		w.sTag(3, "article").nl(3);
149 		article(session, w, o);
150 		w.ie(3).eTag("article").nl(3);
151 
152 		if (hasAside && asideFloat.isAny(RIGHT, DEFAULT)) {
153 			w.sTag(3, "aside").nl(3);
154 			aside(session, w, o);
155 			w.ie(3).eTag("aside").nl(3);
156 		}
157 
158 		w.ie(2).eTag("section").nl(2);
159 
160 		if (hasAside && asideFloat.is(BOTTOM)) {
161 			w.sTag(2, "section").nl(2);
162 			w.sTag(3, "aside").nl(3);
163 			aside(session, w, o);
164 			w.ie(3).eTag("aside").nl(3);
165 			w.ie(2).eTag("section").nl(2);
166 		}
167 
168 		if (hasFooter(session)) {
169 			w.sTag(2, "footer").nl(2);
170 			footer(session, w, o);
171 			w.ie(2).eTag("footer").nl(2);
172 		}
173 	}

253 	/**
254 	 * Renders the contents of the <code><xt>&lt;head&gt;</xt></code> element.
255 	 *
256 	 * @param session The current serializer session.
257 	 * @param w The writer being written to.
258 	 * @param o The object being serialized.
259 	 * @throws Exception Any exception can be thrown.
260 	 */
261 	protected void head(HtmlDocSerializerSession session, HtmlWriter w, Object o) throws Exception {
262 
263 		String[] head = session.getHead();
264 		for (int i = 0; i < head.length; i++)
265 			w.sIf(i > 0).appendln(2, session.resolve(head[i]));
266 
267 		if (hasStyle(session)) {
268 			w.sTag(2, "style").nl(2);
269 			style(session, w, o);
270 			w.ie(2).eTag("style").nl(2);
271 		}
272 		if (hasScript(session)) {
273 			w.sTag(2, "script").nl(2);
274 			script(session, w, o);
275 			w.ie(2).eTag("script").nl(2);
276 		}
277 	}

294 	/**
295 	 * Renders the contents of the <code><xt>&lt;body&gt;</xt>/<xt>&lt;nav&gt;</xt></code> element.
296 	 *
297 	 * @param session The current serializer session.
298 	 * @param w The writer being written to.
299 	 * @param o The object being serialized.
300 	 * @throws Exception Any exception can be thrown.
301 	 */
302 	protected void nav(HtmlDocSerializerSession session, HtmlWriter w, Object o) throws Exception {
303 		String[] links = session.getNavLinks();
304 		if (links.length > 0 && ! contains("NONE", links)) {
305 			w.sTag(3, "ol").nl(3);
306 			for (String l : links) {
307 				w.sTag(4, "li");
308 				l = session.resolve(l);
309 				if (l.matches("(?s)\\S+\\:.*")) {
310 					int i = l.indexOf(':');
311 					String key = l.substring(0, i);
312 					String val = l.substring(i + 1).trim();
313 					if (val.startsWith("<"))
314 						w.nl(4).appendln(5, val);
315 					else
316 						w.oTag("a").attr("href", session.resolveUri(val), true).cTag().text(key, true).eTag("a");
317 					w.eTag("li").nl(4);
318 				} else {
319 					w.nl(4).appendln(5, l);
320 					w.eTag(4, "li").nl(4);
321 				}
322 			}
323 			w.eTag(3, "ol").nl(3);
324 		}
325 		String[] nav = session.getNav();
326 		if (nav.length > 0) {
327 			for (int i = 0; i < nav.length; i++)
328 				w.sIf(i > 0).appendln(3, session.resolve(nav[i]));
329 		}
330 	}

332 	/**
333 	 * Renders the contents of the <code><xt>&lt;head&gt;</xt>/<xt>&lt;script&gt;</xt></code> element.
334 	 *
335 	 * @param session The current serializer session.
336 	 * @param w The writer being written to.
337 	 * @param o The object being serialized.
338 	 * @throws Exception Any exception can be thrown.
339 	 */
340 	protected void script(HtmlDocSerializerSession session, HtmlWriter w, Object o) throws Exception {
341 		Flag addSpace = Flag.create();
342 		for (String s : session.getScript())
343 			w.sIf(addSpace.getAndSet()).append(3, session.resolve(s)).append('\n'); // Must always append a newline even if whitespace disabled!
344 		session.forEachWidget(x -> {
345 			w.sIf(addSpace.getAndSet()).append(3, session.resolve(x.getScript(session.getVarResolver()))).w('\n'); // Must always append a newline even if whitespace disabled!
346 		});
347 	}

349 	/**
350 	 * Renders the contents of the <code><xt>&lt;head&gt;</xt>/<xt>&lt;style&gt;</xt></code> element.
351 	 *
352 	 * @param session The current serializer session.
353 	 * @param w The writer being written to.
354 	 * @param o The object being serialized.
355 	 * @throws Exception Any exception can be thrown.
356 	 */
357 	protected void style(HtmlDocSerializerSession session, HtmlWriter w, Object o) throws Exception {
358 		Flag addSpace = Flag.create();
359 		for (String s : session.getStylesheet())
360 			w.sIf(addSpace.getAndSet()).append(3, "@import ").q().append(session.resolveUri(session.resolve(s))).q().appendln(";");
361 		if (session.isNowrap())
362 			w.appendln(3, "div.data * {white-space:nowrap;} ");
363 		for (String s : session.getStyle())
364 			w.sIf(addSpace.getAndSet()).appendln(3, session.resolve(s));
365 		session.forEachWidget(x -> {
366 			w.sIf(addSpace.getAndSet()).appendln(3, session.resolve(x.getStyle(session.getVarResolver())));
367 		});
368 	}
