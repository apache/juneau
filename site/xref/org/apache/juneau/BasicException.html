13  package org.apache.juneau;
14  
15  import static org.apache.juneau.common.internal.StringUtils.*;
16  
17  import java.text.*;
18  
19  import org.apache.juneau.common.internal.*;
20  
21  /**
22   * Subclass of non-runtime exceptions that take in a message and zero or more arguments.
23   *
24   * <h5 class='section'>See Also:</h5><ul>
25  
26   * </ul>
27   *
28   * @serial exclude
29   */
30  public abstract class BasicException extends Exception {
31  
32  	private static final long serialVersionUID = 1L;
33  
34  	/**
35  	 * Constructor.
36  	 *
37  	 * @param message The {@link MessageFormat}-style message.
38  	 * @param args Optional {@link MessageFormat}-style arguments.
39  	 */
40  	public BasicException(String message, Object...args) {
41  		super(format(message, args));
42  	}
43  
44  	/**
45  	 * Constructor.
46  	 *
47  	 * @param causedBy The cause of this exception.
48  	 * @param message The {@link MessageFormat}-style message.
49  	 * @param args Optional {@link MessageFormat}-style arguments.
50  	 */
51  	public BasicException(Throwable causedBy, String message, Object...args) {
52  		this(message, args);
53  		initCause(causedBy);
54  	}
55  
56  	/**
57  	 * Constructor.
58  	 *
59  	 * @param causedBy The cause of this exception.
60  	 */
61  	public BasicException(Throwable causedBy) {
62  		this(causedBy, causedBy.getLocalizedMessage());
63  	}
64  
65  	/**
66  	 * Same as {@link #getCause()} but searches the throwable chain for an exception of the specified type.
67  	 *
68  	 * @param c The throwable type to search for.
69  	 * @param <T> The throwable type to search for.
70  	 * @return The exception, or <jk>null</jk> if not found.
71  	 */
72  	public <T extends Throwable> T getCause(Class<T> c) {
73  		return ThrowableUtils.getCause(c, this);
74  	}
75  
76  	/**
77  	 * Returns the caused-by exception if there is one.
78  	 *
79  	 * @return The caused-by exception if there is one, or this exception if there isn't.
80  	 */
81  	public Throwable unwrap() {
82  		Throwable t = getCause();
83  		return t == null ? this : t;
84  	}
85  }
