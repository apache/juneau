13  package org.apache.juneau;
14  
15  import static org.apache.juneau.common.internal.Utils.*;
16  import java.util.*;
17  import java.util.function.*;
18  
19  import org.apache.http.*;
20  import org.apache.http.message.*;
21  
22  
23  /**
24   * Describes a single type used in content negotiation between an HTTP client and server, as described in
25   * Section 14.1 and 14.7 of RFC2616 (the HTTP/1.1 specification).
26   *
27   * <h5 class='section'>See Also:</h5><ul>
28   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauRestCommonBasics">juneau-rest-common</a>
29   * 	<li class='extlink'><a class="doclink" href="https://www.w3.org/Protocols/rfc2616/rfc2616.html">Hypertext Transfer Protocol -- HTTP/1.1</a>
30   * </ul>
31   */
32  public class MediaRange extends MediaType {
33  
34  	private final NameValuePair[] extensions;
35  	private final Float qValue;
36  	private final String string;
37  
38  	/**
39  	 * Constructor.
40  	 *
41  	 * @param e The parsed media range element.
42  	 */
43  	public MediaRange(HeaderElement e) {
44  		super(e);
45  
46  		Float qValue = 1f;
47  
48  		// The media type consists of everything up to the q parameter.
49  		// The q parameter and stuff after is part of the range.
50  		List<NameValuePair> extensions = list();
51  		boolean foundQ = false;
52  		for (NameValuePair p : e.getParameters()) {
53  			if (p.getName().equals("q")) {
54  				qValue = Float.parseFloat(p.getValue());
55  				foundQ = true;
56  			} else if (foundQ) {
57  				extensions.add(new BasicNameValuePair(p.getName(), p.getValue()));
58  			}
59  		}
60  
61  		this.qValue = qValue;
62  		this.extensions = extensions.toArray(new NameValuePair[extensions.size()]);
63  
64  		StringBuffer sb = new StringBuffer().append(super.toString());
65  
66  		// '1' is equivalent to specifying no qValue. If there's no extensions, then we won't include a qValue.
67  		if (qValue.floatValue() == 1.0) {
68  			if (this.extensions.length > 0) {
69  				sb.append(";q=").append(qValue);
70  				extensions.forEach(x -> sb.append(';').append(x.getName()).append('=').append(x.getValue()));
71  			}
72  		} else {
73  			sb.append(";q=").append(qValue);
74  			extensions.forEach(x -> sb.append(';').append(x.getName()).append('=').append(x.getValue()));
75  		}
76  		string = sb.toString();
77  	}
78  
79  	/**
80  	 * Returns the <js>'q'</js> (quality) value for this type, as described in Section 3.9 of RFC2616.
81  	 *
82  	 * <p>
83  	 * The quality value is a float between <c>0.0</c> (unacceptable) and <c>1.0</c> (most acceptable).
84  	 *
85  	 * <p>
86  	 * If 'q' value doesn't make sense for the context (e.g. this range was extracted from a <js>"content-*"</js>
87  	 * header, as opposed to <js>"accept-*"</js> header, its value will always be <js>"1"</js>.
88  	 *
89  	 * @return The 'q' value for this type, never <jk>null</jk>.
90  	 */
91  	public Float getQValue() {
92  		return qValue;
93  	}
94  
95  	/**
96  	 * Returns the optional set of custom extensions defined for this type.
97  	 *
98  	 * <p>
99  	 * Values are lowercase and never <jk>null</jk>.
100 	 *
101 	 * @return The optional list of extensions, never <jk>null</jk>.
102 	 */
103 	public List<NameValuePair> getExtensions() {
104 		return u(alist(extensions));
105 	}
106 
107 	/**
108 	 * Performs an action on the optional set of custom extensions defined for this type.
109 	 *
110 	 * <p>
111 	 * Values are lowercase and never <jk>null</jk>.
112 	 *
113 	 * @param action The action to perform.
114 	 * @return This object.
115 	 */
116 	public MediaRange forEachExtension(Consumer<NameValuePair> action) {
117 		for (NameValuePair r : extensions)
118 			action.accept(r);
119 		return this;
120 	}
121 
122 	@Override /* Object */
123 	public String toString() {
124 		return string;
125 	}
126 }
