24  import org.apache.juneau.*;

27  /**
28   * Metadata on classes specific to the XML serializers and parsers pulled from the {@link Xml @Xml} annotation on the
29   * class.
30   *
31   * <h5 class='section'>See Also:</h5><ul>
32   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/XmlBasics">XML Basics</a>
33   * </ul>
34   */
35  public class XmlClassMeta extends ExtendedClassMeta {
36  
37  	private final Namespace namespace;
38  	private final XmlFormat format;
39  	private final String childName;
40  
41  	/**
42  	 * Constructor.
43  	 *
44  	 * @param cm The class that this annotation is defined on.
45  	 * @param mp XML metadata provider (for finding information about other artifacts).
46  	 */
47  	public XmlClassMeta(ClassMeta<?> cm, XmlMetaProvider mp) {
48  		super(cm);
49  		List<Xml> xmls = list();
50  		List<XmlSchema> schemas = list();
51  		if (nn(cm)) {
52  			cm.forEachAnnotation(Xml.class, x -> true, x -> xmls.add(x));
53  			cm.forEachAnnotation(XmlSchema.class, x -> true, x -> schemas.add(x));
54  		}
55  		this.namespace = XmlUtils.findNamespace(xmls, schemas);
56  
57  		String _childName = null;
58  		var _format = XmlFormat.DEFAULT;
59  		for (var a : xmls) {
60  			if (a.format() != XmlFormat.DEFAULT)
61  				_format = a.format();
62  			if (! a.childName().isEmpty())
63  				_childName = a.childName();
64  		}
65  		this.format = _format;
66  		this.childName = _childName;
67  	}
68  
69  	/**
70  	 * Returns the XML namespace associated with this class.
71  	 *
72  	 * <p>
73  	 * Namespace is determined in the following order of {@link Xml#prefix() @Xml(prefix)} annotation:
74  	 * <ol>
75  	 * 	<li>Class.
76  	 * 	<li>Package.
77  	 * 	<li>Superclasses.
78  	 * 	<li>Superclass packages.
79  	 * 	<li>Interfaces.
80  	 * 	<li>Interface packages.
81  	 * </ol>
82  	 *
83  	 * @return The namespace associated with this class, or <jk>null</jk> if no namespace is associated with it.
84  	 */
85  	public Namespace getNamespace() { return namespace; }
86  
87  	/**
88  	 * Returns the {@link Xml#childName() @Xml(childName)} annotation defined on the class.
89  	 *
90  	 * @return The value of the annotation, or <jk>null</jk> if not specified.
91  	 */
92  	protected String getChildName() { return childName; }
93  
94  	/**
95  	 * Returns the {@link Xml#format() @Xml(format)} annotation defined on the class.
96  	 *
97  	 * @return The value of the annotation, or {@link XmlFormat#DEFAULT} if not specified.
98  	 */
99  	protected XmlFormat getFormat() { return format; }
100 }
