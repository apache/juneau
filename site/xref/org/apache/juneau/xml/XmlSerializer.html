32  /**
33   * Serializes POJO models to XML.
34   *
35   * <h5 class='topic'>Media types</h5>
36   * <p>
37   * Handles <c>Accept</c> types:  <bc>text/xml</bc>
38   * <p>
39   * Produces <c>Content-Type</c> types:  <bc>text/xml</bc>
40   *
41   * <h5 class='topic'>Description</h5>
42   * <p>
43   * See the {@link JsonSerializer} class for details on how Java models map to JSON.
44   *
45   * <p>
46   * For example, the following JSON...
47   * <p class='bjson'>
48   * 	{
49   * 		name:<js>'John Smith'</js>,
50   * 		address: {
51   * 			streetAddress: <js>'21 2nd Street'</js>,
52   * 			city: <js>'New York'</js>,
53   * 			state: <js>'NY'</js>,
54   * 			postalCode: <js>10021</js>
55   * 		},
56   * 		phoneNumbers: [
57   * 			<js>'212 555-1111'</js>,
58   * 			<js>'212 555-2222'</js>
59   * 		],
60   * 		additionalInfo: <jk>null</jk>,
61   * 		remote: <jk>false</jk>,
62   * 		height: <js>62.4</js>,
63   * 		<js>'fico score'</js>:  <js>' &gt; 640'</js>
64   * 	}
65   * <p>
66   * 	...maps to the following XML using the default serializer...
67   * <p class='bxml'>
68   * 	<xt>&lt;object&gt;</xt>
69   * 		<xt>&lt;name&gt;</xt>John Smith<xt>&lt;/name&gt;</xt>
70   * 		<xt>&lt;address&gt;</xt>
71   * 			<xt>&lt;streetAddress&gt;</xt>21 2nd Street<xt>&lt;/streetAddress&gt;</xt>
72   * 			<xt>&lt;city&gt;</xt>New York<xt>&lt;/city&gt;</xt>
73   * 			<xt>&lt;state&gt;</xt>NY<xt>&lt;/state&gt;</xt>
74   * 			<xt>&lt;postalCode&gt;</xt>10021<xt>&lt;/postalCode&gt;</xt>
75   * 		<xt>&lt;/address&gt;</xt>
76   * 		<xt>&lt;phoneNumbers&gt;</xt>
77   * 			<xt>&lt;string&gt;</xt>212 555-1111<xt>&lt;/string&gt;</xt>
78   * 			<xt>&lt;string&gt;</xt>212 555-2222<xt>&lt;/string&gt;</xt>
79   * 		<xt>&lt;/phoneNumbers&gt;</xt>
80   * 		<xt>&lt;additionalInfo</xt> <xa>_type</xa>=<xs>'null'</xs><xt>&gt;&lt;/additionalInfo&gt;</xt>
81   * 		<xt>&lt;remote&gt;</xt>false<xt>&lt;/remote&gt;</xt>
82   * 		<xt>&lt;height&gt;</xt>62.4<xt>&lt;/height&gt;</xt>
83   * 		<xt>&lt;fico_x0020_score&gt;</xt> &amp;gt; 640<xt>&lt;/fico_x0020_score&gt;</xt>
84   * 	<xt>&lt;/object&gt;</xt>
85   *
86   * <p>
87   * An additional "add-json-properties" mode is also provided to prevent loss of JSON data types...
88   * <p class='bxml'>
89   * 	<xt>&lt;object&gt;</xt>
90   * 		<xt>&lt;name</xt> <xa>_type</xa>=<xs>'string'</xs><xt>&gt;</xt>John Smith<xt>&lt;/name&gt;</xt>
91   * 		<xt>&lt;address</xt> <xa>_type</xa>=<xs>'object'</xs><xt>&gt;</xt>
92   * 			<xt>&lt;streetAddress</xt> <xa>_type</xa>=<xs>'string'</xs><xt>&gt;</xt>21 2nd Street<xt>&lt;/streetAddress&gt;</xt>
93   * 			<xt>&lt;city</xt> <xa>_type</xa>=<xs>'string'</xs><xt>&gt;</xt>New York<xt>&lt;/city&gt;</xt>
94   * 			<xt>&lt;state</xt> <xa>_type</xa>=<xs>'string'</xs><xt>&gt;</xt>NY<xt>&lt;/state&gt;</xt>
95   * 			<xt>&lt;postalCode</xt> <xa>_type</xa>=<xs>'number'</xs><xt>&gt;</xt>10021<xt>&lt;/postalCode&gt;</xt>
96   * 		<xt>&lt;/address&gt;</xt>
97   * 		<xt>&lt;phoneNumbers</xt> <xa>_type</xa>=<xs>'array'</xs><xt>&gt;</xt>
98   * 			<xt>&lt;string&gt;</xt>212 555-1111<xt>&lt;/string&gt;</xt>
99   * 			<xt>&lt;string&gt;</xt>212 555-2222<xt>&lt;/string&gt;</xt>
100  * 		<xt>&lt;/phoneNumbers&gt;</xt>
101  * 		<xt>&lt;additionalInfo</xt> <xa>_type</xa>=<xs>'null'</xs><xt>&gt;&lt;/additionalInfo&gt;</xt>
102  * 		<xt>&lt;remote</xt> <xa>_type</xa>=<xs>'boolean'</xs><xt>&gt;</xt>false<xt>&lt;/remote&gt;</xt>
103  * 		<xt>&lt;height</xt> <xa>_type</xa>=<xs>'number'</xs><xt>&gt;</xt>62.4<xt>&lt;/height&gt;</xt>
104  * 		<xt>&lt;fico_x0020_score</xt> <xa>_type</xa>=<xs>'string'</xs><xt>&gt;</xt> &amp;gt; 640<xt>&lt;/fico_x0020_score&gt;</xt>
105  * 	<xt>&lt;/object&gt;</xt>
106  * </p>
107  *
108  * <p>
109  * This serializer provides several serialization options.
110  * Typically, one of the predefined <jsf>DEFAULT</jsf> serializers will be sufficient.
111  * However, custom serializers can be constructed to fine-tune behavior.
112  *
113  * <p>
114  * If an attribute name contains any non-valid XML element characters, they will be escaped using standard
115  * {@code _x####_} notation.
116  *
117  * <h5 class='topic'>Behavior-specific subclasses</h5>
118  * <p>
119  * The following direct subclasses are provided for convenience:
120  * <ul>
121  * 	<li>{@link Sq} - Default serializer, single quotes.
122  * 	<li>{@link SqReadable} - Default serializer, single quotes, whitespace added.
123  * </ul>
124  *
125  * <h5 class='section'>Notes:</h5><ul>
126  * 	<li class='note'>This class is thread safe and reusable.
127  * </ul>
128  *
129  * <h5 class='section'>See Also:</h5><ul>
130  * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/XmlBasics">XML Basics</a>
131 
132  * </ul>
133  */
134 public class XmlSerializer extends WriterSerializer implements XmlMetaProvider {

253 		private static final Cache<HashKey,XmlSerializer> CACHE = Cache.of(HashKey.class, XmlSerializer.class).build();
254 
255 		boolean addBeanTypesXml, addNamespaceUrisToRoot, disableAutoDetectNamespaces, enableNamespaces;
256 		Namespace defaultNamespace;
257 		List<Namespace> namespaces;
258 
259 		/**
260 		 * Constructor, default settings.
261 		 */
262 		protected Builder() {
263 			produces("text/xml");
264 			addBeanTypesXml = env("XmlSerializer.addBeanTypes", false);
265 			addNamespaceUrisToRoot = env("XmlSerializer.addNamespaceUrisToRoot", false);
266 			disableAutoDetectNamespaces = env("XmlSerializer.disableAutoDetectNamespaces", false);
267 			enableNamespaces = env("XmlSerializer.enableNamespaces", false);
268 			defaultNamespace = null;
269 			namespaces = null;
270 
271 		}
272 
273 		/**
274 		 * Copy constructor.
275 		 *
276 		 * @param copyFrom The bean to copy from.
277 		 */
278 		protected Builder(XmlSerializer copyFrom) {
279 			super(copyFrom);
280 			addBeanTypesXml = copyFrom.addBeanTypesXml;
281 			addNamespaceUrisToRoot = copyFrom.addNamespaceUrlsToRoot;
282 			disableAutoDetectNamespaces = ! copyFrom.autoDetectNamespaces;
283 			enableNamespaces = copyFrom.enableNamespaces;
284 			defaultNamespace = copyFrom.defaultNamespace;
285 			namespaces = copyFrom.namespaces.length == 0 ? null : Utils.list(copyFrom.namespaces);
286 		}
287 
288 		/**
289 		 * Copy constructor.
290 		 *
291 		 * @param copyFrom The builder to copy from.
292 		 */
293 		protected Builder(Builder copyFrom) {
294 			super(copyFrom);
295 			addBeanTypesXml = copyFrom.addBeanTypesXml;
296 			addNamespaceUrisToRoot = copyFrom.addNamespaceUrisToRoot;
297 			disableAutoDetectNamespaces = copyFrom.disableAutoDetectNamespaces;
298 			enableNamespaces = copyFrom.enableNamespaces;
299 			defaultNamespace = copyFrom.defaultNamespace;
300 			namespaces = copyOf(copyFrom.namespaces);
301 		}

392 		/**
393 		 * Don't auto-detect namespace usage.
394 		 *
395 		 * <p>
396 		 * Don't detect namespace usage before serialization.
397 		 *
398 		 * <p>
399 		 * Used in conjunction with {@link Builder#addNamespaceUrisToRoot()} to reduce the list of namespace URLs appended to the
400 		 * root element to only those that will be used in the resulting document.
401 		 *
402 		 * <p>
403 		 * If disabled, then the data structure will first be crawled looking for namespaces that will be encountered before
404 		 * the root element is serialized.
405 		 *
406 		 * <p>
407 		 * This setting is ignored if {@link Builder#enableNamespaces()} is not enabled.
408 		 *
409 		 * <h5 class='section'>Notes:</h5><ul>
410 		 * 	<li class='note'>
411 		 * 		Auto-detection of namespaces can be costly performance-wise.
412 		 * 		<br>In high-performance environments, it's recommended that namespace detection be
413 		 * 		disabled, and that namespaces be manually defined through the {@link Builder#namespaces(Namespace...)} property.
414 		 * </ul>
415 		 *
416 		 * <h5 class='section'>See Also:</h5><ul>
417 		 * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/XmlNamespaces">Namespaces</a>
418 		 * </ul>
419 		 *
420 		 * @return This object.
421 		 */

425 		}
