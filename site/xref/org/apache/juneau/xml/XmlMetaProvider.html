13  package org.apache.juneau.xml;
14  
15  import org.apache.juneau.*;
16  
17  /**
18   * Interface for providing access to {@link XmlClassMeta}, {@link XmlBeanMeta}, and {@link XmlBeanPropertyMeta} objects.
19   *
20   * <h5 class='section'>See Also:</h5><ul>
21   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/XmlBasics">XML Basics</a>
22   * </ul>
23   */
24  public interface XmlMetaProvider extends AnnotationProvider  {
25  
26  	/**
27  	 * Returns the language-specific metadata on the specified class.
28  	 *
29  	 * @param cm The class to return the metadata on.
30  	 * @return The metadata.
31  	 */
32  	XmlClassMeta getXmlClassMeta(ClassMeta<?> cm);
33  
34  	/**
35  	 * Returns the language-specific metadata on the specified bean.
36  	 *
37  	 * @param bm The bean to return the metadata on.
38  	 * @return The metadata.
39  	 */
40  	XmlBeanMeta getXmlBeanMeta(BeanMeta<?> bm);
41  
42  	/**
43  	 * Returns the language-specific metadata on the specified bean property.
44  	 *
45  	 * @param bpm The bean property to return the metadata on.
46  	 * @return The metadata.
47  	 */
48  	XmlBeanPropertyMeta getXmlBeanPropertyMeta(BeanPropertyMeta bpm);
49  }
