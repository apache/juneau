13  package org.apache.juneau.xml.annotation;
14  
15  import static java.lang.annotation.ElementType.*;
16  import static java.lang.annotation.RetentionPolicy.*;
17  
18  import java.lang.annotation.*;
19  
20  /**
21   * Identifies the default XML namespaces at the package level.
22   *
23   * <h5 class='section'>See Also:</h5><ul>
24   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/XmlBasics">XML Basics</a>
25   * </ul>
26   */
27  @Documented
28  @Target(PACKAGE)
29  @Retention(RUNTIME)
30  @Inherited
31  public @interface XmlSchema {
32  
33  	/**
34  	 * Sets the default XML namespace URL for all classes in this and child packages.
35  	 *
36  	 * <p>
37  	 * Must either be matched with a {@link #prefix()} annotation, or an {@link #xmlNs()} mapping with the same
38  	 * {@link XmlNs#namespaceURI() @XmlNs(namespaceURI)} value.
39  	 *
40  	 * @return The annotation value.
41  	 */
42  	public String namespace() default "";
43  
44  	/**
45  	 * Sets the default XML prefix for all classes in this and child packages.
46  	 *
47  	 * <p>
48  	 * Must either be matched with a {@link #namespace()} annotation, or an {@link #xmlNs()} mapping with the same
49  	 * {@link XmlNs#prefix} value.
50  	 *
51  	 * @return The annotation value.
52  	 */
53  	public String prefix() default "";
54  
55  	/**
56  	 * Lists all namespace mappings to be used on all classes within this package.
57  	 *
58  	 * <p>
59  	 * The purpose of this annotation is to allow namespace mappings to be defined in a single location and referred
60  	 * to by name through just the {@link Xml#prefix() @Xml(prefix)} annotation.
61  	 *
62  	 * <p>
63  	 * Inherited by child packages.
64  	 *
65  	 * <h5 class='section'>Example:</h5>
66  	 * <p>
67  	 * Contents of <c>package-info.java</c>...
68  	 * <p class='bjava'>
69  	 * 	<jc>// XML namespaces used within this package.</jc>
70  	 * 	<ja>@XmlSchema</ja>(prefix=<js>"ab"</js>,
71  	 * 		namespaces={
72  	 * 			<ja>@XmlNs</ja>(prefix=<js>"ab"</js>, namespaceURI=<js>"http://www.apache.org/addressBook/"</js>),
73  	 * 			<ja>@XmlNs</ja>(prefix=<js>"per"</js>, namespaceURI=<js>"http://www.apache.org/person/"</js>),
74  	 * 			<ja>@XmlNs</ja>(prefix=<js>"addr"</js>, namespaceURI=<js>"http://www.apache.org/address/"</js>),
75  	 * 			<ja>@XmlNs</ja>(prefix=<js>"mail"</js>, namespaceURI="<js>http://www.apache.org/mail/"</js>)
76  	 * 		}
77  	 * 	)
78  	 * 	<jk>package</jk> org.apache.juneau.examples.addressbook;
79  	 * 	<jk>import</jk> org.apache.juneau.xml.annotation.*;
80  	 * </p>
81  	 *
82  	 * <p>
83  	 * Class in package using defined namespaces...
84  	 * <p class='bjava'>
85  	 * 	<jk>package</jk> corg.apache.juneau.examples.addressbook;
86  	 *
87  	 * 	<jc>// Bean class, override "ab" namespace on package.</jc>
88  	 * 	<ja>@Xml</ja>(prefix=<js>"addr"</js>)
89  	 * 	<jk>public class</jk> Address {
90  	 *
91  	 * 		<jc>// Bean property, use "addr" namespace on class.</jc>
92  	 * 		<jk>public int</jk> <jf>id</jf>;
93  	 *
94  	 * 		<jc>// Bean property, override with "mail" namespace.</jc>
95  	 * 		<ja>@Xml</ja>(prefix=<js>"mail"</js>)
96  	 * 		<jk>public</jk> String <jf>street</jf>, <jf>city</jf>, <jf>state</jf>;
97  	 * 	}
98  	 * </p>
99  	 *
100 	 * @return The annotation value.
101 	 */
102 	public XmlNs[] xmlNs() default {};
103 }
