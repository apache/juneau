232 	/**
233 	 * Decodes and trims the specified string.
234 	 *
235 	 * <p>
236 	 * Any <js>'_x####_'</js> sequences in the string will be decoded.
237 	 *
238 	 * @param s The string to be decoded.
239 	 * @return The decoded string.
240 	 */
241 	protected final String decodeString(String s) {
242 		if (s == null)
243 			return null;
244 		rsb.setLength(0);
245 		s = XmlUtils.decode(s, rsb);
246 		if (isTrimStrings())
247 			s = s.trim();
248 		return s;
249 	}

285 	 *
286 	 * <p>

288 	 *

293 	 * @return The decoded text.  <jk>null</jk> if the text consists of the sequence <js>'_x0000_'</js>.

302 	/*
303 	 * Returns the content of the current CHARACTERS node.
304 	 * Any <js>'_x####_'</js> sequences in the string will be decoded.
305 	 * Leading and trailing whitespace (unencoded) will be trimmed from the result.
306 	 */
307 	private String getText(XmlReader r, boolean trim) {
308 		String s = r.getText();
309 		if (trim)
310 			s = s.trim();
311 		if (s.isEmpty())
312 			return null;
313 		return decodeString(s);
314 	}

323 	/*
324 	 * Takes the element being read from the XML stream reader and reconstructs it as XML.
325 	 * Used when reconstructing bean properties of type {@link XmlFormat#XMLTEXT}.
326 	 */
327 	private String getElementAsString(XmlReader r) {
328 		int t = r.getEventType();
329 		if (t > 2)
330 			throw new BasicRuntimeException("Invalid event type on stream reader for elementToString() method: ''{0}''", XmlUtils.toReadableEvent(r));
331 		rsb.setLength(0);
332 		rsb.append("<").append(t == 1 ? "" : "/").append(r.getLocalName());
333 		if (t == 1)
334 			for (int i = 0; i < r.getAttributeCount(); i++)
335 				rsb.append(' ').append(r.getAttributeName(i)).append('=').append('\'').append(r.getAttributeValue(i)).append('\'');
336 		rsb.append('>');
337 		return rsb.toString();
338 	}

340 	/**
341 	 * Parses the current element as text.
342 	 *
343 	 * @param r The input reader.
344 	 * @return The parsed text.
345 	 * @throws XMLStreamException Thrown by underlying reader.
346 	 * @throws IOException Thrown by underlying stream.
347 	 * @throws ParseException Malformed input encountered.
348 	 */
349 	protected String parseText(XmlReader r) throws IOException, XMLStreamException, ParseException {
350 		// Note that this is different than {@link #getText(XmlReader)} since it assumes that we're pointing to a
351 		// whitespace element.
352 
353 		StringBuilder sb2 = getStringBuilder();
354 
355 		int depth = 0;
356 		while (true) {
357 			int et = r.getEventType();
358 			if (et == START_ELEMENT) {
359 				sb2.append(getElementAsString(r));
360 				depth++;
361 			} else if (et == CHARACTERS) {
362 				sb2.append(getText(r));
363 			} else if (et == END_ELEMENT) {
364 				sb2.append(getElementAsString(r));
365 				depth--;
366 				if (depth <= 0)
367 					break;
368 			}
369 			et = r.next();
370 		}
371 		String s = sb2.toString();
372 		returnStringBuilder(sb2);
373 		return s;
374 	}

428 	/**
429 	 * Workhorse method.
430 	 *
431 	 * @param <T> The expected type of object.
432 	 * @param eType The expected type of object.
433 	 * @param currAttr The current bean property name.
434 	 * @param r The reader.
435 	 * @param outer The outer object.
436 	 * @param isRoot If <jk>true</jk>, then we're serializing a root element in the document.
437 	 * @param pMeta The bean property metadata.
438 	 * @return The parsed object.
439 	 * @throws IOException Thrown by underlying stream.
440 	 * @throws ParseException Malformed input encountered.
441 	 * @throws ExecutableException Exception occurred on invoked constructor/method/field.
442 	 * @throws XMLStreamException Malformed XML encountered.
443 	 */
444 	protected <T> T parseAnything(ClassMeta<T> eType, String currAttr, XmlReader r,
445 			Object outer, boolean isRoot, BeanPropertyMeta pMeta) throws IOException, ParseException, ExecutableException, XMLStreamException {
446 
447 		if (eType == null)
448 			eType = (ClassMeta<T>)object();
449 		ObjectSwap<T,Object> swap = (ObjectSwap<T,Object>)eType.getSwap(this);
450 		BuilderSwap<T,Object> builder = (BuilderSwap<T,Object>)eType.getBuilderSwap(this);
451 		ClassMeta<?> sType = null;
452 		if (builder != null)
453 			sType = builder.getBuilderClassMeta(this);
454 		else if (swap != null)
455 			sType = swap.getSwapClassMeta(this);
456 		else
457 			sType = eType;
458 
459 		if (sType.isOptional())
460 			return (T)Utils.opt(parseAnything(eType.getElementType(), currAttr, r, outer, isRoot, pMeta));
461 
462 		setCurrentClass(sType);
463 
464 		String wrapperAttr = (isRoot && isPreserveRootElement()) ? r.getName().getLocalPart() : null;
465 		String typeAttr = r.getAttributeValue(null, getBeanTypePropertyName(eType));
466 		boolean isNil = "true".equals(r.getAttributeValue(null, "nil"));
467 		int jsonType = getJsonType(typeAttr);
468 		String elementName = getElementName(r);
469 		if (jsonType == 0) {
470 			if (elementName == null || elementName.equals(currAttr))
471 				jsonType = UNKNOWN;
472 			else {
473 				typeAttr = elementName;
474 				jsonType = getJsonType(elementName);
475 			}
476 		}
477 
478 		ClassMeta tcm = getClassMeta(typeAttr, pMeta, eType);
479 		if (tcm == null && elementName != null && ! elementName.equals(currAttr))
480 			tcm = getClassMeta(elementName, pMeta, eType);
481 		if (tcm != null)
482 			sType = eType = tcm;
483 
484 		Object o = null;
485 
486 		if (jsonType == NULL) {
487 			r.nextTag();	// Discard end tag
488 			return null;
489 		}
490 
491 		if (sType.isObject()) {
492 			if (jsonType == OBJECT) {
493 				JsonMap m = new JsonMap(this);
494 				parseIntoMap(r, m, string(), object(), pMeta);
495 				if (wrapperAttr != null)
496 					m = new JsonMap(this).append(wrapperAttr, m);
497 				o = cast(m, pMeta, eType);
498 			} else if (jsonType == ARRAY)
499 				o = parseIntoCollection(r, new JsonList(this), null, pMeta);
500 			else if (jsonType == STRING) {
501 				o = getElementText(r);
502 				if (sType.isChar())
503 					o = parseCharacter(o);
504 			}
505 			else if (jsonType == NUMBER)
506 				o = parseNumber(getElementText(r), null);
507 			else if (jsonType == BOOLEAN)
508 				o = Boolean.parseBoolean(getElementText(r));
509 			else if (jsonType == UNKNOWN)
510 				o = getUnknown(r);
511 		} else if (sType.isBoolean()) {
512 			o = Boolean.parseBoolean(getElementText(r));
513 		} else if (sType.isCharSequence()) {
514 			o = getElementText(r);
515 		} else if (sType.isChar()) {
516 			o = parseCharacter(getElementText(r));
517 		} else if (sType.isMap()) {
518 			Map m = (sType.canCreateNewInstance(outer) ? (Map)sType.newInstance(outer) : newGenericMap(sType));
519 			o = parseIntoMap(r, m, sType.getKeyType(), sType.getValueType(), pMeta);
520 			if (wrapperAttr != null)
521 				o = new JsonMap(this).append(wrapperAttr, m);
522 		} else if (sType.isCollection()) {
523 			Collection l = (sType.canCreateNewInstance(outer) ? (Collection)sType.newInstance(outer) : new JsonList(this));
524 			o = parseIntoCollection(r, l, sType, pMeta);
525 		} else if (sType.isNumber()) {
526 			o = parseNumber(getElementText(r), (Class<? extends Number>)sType.getInnerClass());
527 		} else if (builder != null || sType.canCreateNewBean(outer)) {
528 			if (getXmlClassMeta(sType).getFormat() == COLLAPSED) {
529 				String fieldName = r.getLocalName();
530 				BeanMap<?> m = builder != null ? toBeanMap(builder.create(this, eType)) : newBeanMap(outer, sType.getInnerClass());
531 				BeanPropertyMeta bpm = getXmlBeanMeta(m.getMeta()).getPropertyMeta(fieldName);
532 				ClassMeta<?> cm = m.getMeta().getClassMeta();
533 				Object value = parseAnything(cm, currAttr, r, m.getBean(false), false, null);
534 				setName(cm, value, currAttr);
535 				bpm.set(m, currAttr, value);
536 				o = builder != null ? builder.build(this, m.getBean(), eType) : m.getBean();
537 			} else {
538 				BeanMap m = builder != null ? toBeanMap(builder.create(this, eType)) : newBeanMap(outer, sType.getInnerClass());
539 				m = parseIntoBean(r, m, isNil);
540 				o = builder != null ? builder.build(this, m.getBean(), eType) : m.getBean();
541 			}
542 		} else if (sType.isArray() || sType.isArgs()) {
543 			ArrayList l = (ArrayList)parseIntoCollection(r, list(), sType, pMeta);
544 			o = toArray(sType, l);
545 		} else if (sType.canCreateNewInstanceFromString(outer)) {
546 			o = sType.newInstanceFromString(outer, getElementText(r));
547 		} else if (sType.getProxyInvocationHandler() != null) {
548 			JsonMap m = new JsonMap(this);
549 			parseIntoMap(r, m, string(), object(), pMeta);
550 			if (wrapperAttr != null)
551 				m = new JsonMap(this).append(wrapperAttr, m);
552 			o = newBeanMap(outer, sType.getInnerClass()).load(m).getBean();
553 		} else {
554 			throw new ParseException(this,
555 				"Class ''{0}'' could not be instantiated.  Reason: ''{1}'', property: ''{2}''",
556 				sType.getInnerClass().getName(), sType.getNotABeanReason(), pMeta == null ? null : pMeta.getName());
557 		}
558 
559 		if (swap != null && o != null)
560 			o = unswap(swap, o, eType);
561 
562 		if (outer != null)
563 			setParent(eType, o, outer);
564 
565 		return (T)o;
566 	}

568 	private <K,V> Map<K,V> parseIntoMap(XmlReader r, Map<K,V> m, ClassMeta<K> keyType,
569 			ClassMeta<V> valueType, BeanPropertyMeta pMeta) throws IOException, ParseException, ExecutableException, XMLStreamException {
570 		int depth = 0;
571 		for (int i = 0; i < r.getAttributeCount(); i++) {
572 			String a = r.getAttributeLocalName(i);
573 			// TODO - Need better handling of namespaces here.
574 			if (! isSpecialAttr(a)) {
575 				K key = trim(convertAttrToType(m, a, keyType));
576 				V value = trim(convertAttrToType(m, r.getAttributeValue(i), valueType));
577 				setName(valueType, value, key);
578 				m.put(key, value);
579 			}
580 		}
581 		do {
582 			int event = r.nextTag();
583 			String currAttr;
584 			if (event == START_ELEMENT) {
585 				depth++;
586 				currAttr = getNameProperty(r);
587 				if (currAttr == null)
588 					currAttr = getElementName(r);
589 				K key = convertAttrToType(m, currAttr, keyType);
590 				V value = parseAnything(valueType, currAttr, r, m, false, pMeta);
591 				setName(valueType, value, currAttr);
592 				if (valueType.isObject() && m.containsKey(key)) {
593 					Object o = m.get(key);
594 					if (o instanceof List)
595 						((List)o).add(value);
596 					else
597 						m.put(key, (V)new JsonList(o, value).setBeanSession(this));
598 				} else {
599 					m.put(key, value);
600 				}
601 			} else if (event == END_ELEMENT) {
602 				depth--;
603 				return m;
604 			}
605 		} while (depth > 0);
606 		return m;
607 	}

609 	private <E> Collection<E> parseIntoCollection(XmlReader r, Collection<E> l,
610 			ClassMeta<?> type, BeanPropertyMeta pMeta) throws IOException, ParseException, ExecutableException, XMLStreamException {
611 		int depth = 0;
612 		int argIndex = 0;
613 		do {
614 			int event = r.nextTag();
615 			if (event == START_ELEMENT) {
616 				depth++;
617 				ClassMeta<?> elementType = type == null ? object() : type.isArgs() ? type.getArg(argIndex++) : type.getElementType();
618 				E value = (E)parseAnything(elementType, null, r, l, false, pMeta);
619 				l.add(value);
620 			} else if (event == END_ELEMENT) {
621 				depth--;
622 				return l;
623 			}
624 		} while (depth > 0);
625 		return l;
626 	}

628 	private static int getJsonType(String s) {
629 		if (s == null)
630 			return UNKNOWN;
631 		char c = s.charAt(0);
632 		switch(c) {
633 			case 'o': return (s.equals("object") ? OBJECT : UNKNOWN);
634 			case 'a': return (s.equals("array") ? ARRAY : UNKNOWN);
635 			case 's': return (s.equals("string") ? STRING : UNKNOWN);
636 			case 'b': return (s.equals("boolean") ? BOOLEAN : UNKNOWN);
637 			case 'n': {
638 				c = s.charAt(2);
639 				switch(c) {
640 					case 'm': return (s.equals("number") ? NUMBER : UNKNOWN);
641 					case 'l': return (s.equals("null") ? NULL : UNKNOWN);
642 				}
643 				//return NUMBER;
644 			}
645 		}
646 		return UNKNOWN;
647 	}

649 	private <T> BeanMap<T> parseIntoBean(XmlReader r, BeanMap<T> m, boolean isNil) throws IOException, ParseException, ExecutableException, XMLStreamException {
650 		BeanMeta<?> bMeta = m.getMeta();
651 		XmlBeanMeta xmlMeta = getXmlBeanMeta(bMeta);
652 
653 		for (int i = 0; i < r.getAttributeCount(); i++) {
654 			String key = getAttributeName(r, i);
655 			if (! ("nil".equals(key) || isSpecialAttr(key))) {
656 				String val = r.getAttributeValue(i);
657 				String ns = r.getAttributeNamespace(i);
658 				BeanPropertyMeta bpm = xmlMeta.getPropertyMeta(key);
659 				if (bpm == null) {
660 					if (xmlMeta.getAttrsProperty() != null) {
661 						xmlMeta.getAttrsProperty().add(m, key, key, val);
662 					} else if (ns == null) {
663 						onUnknownProperty(key, m, val);
664 					}
665 				} else {
666 					try {
667 						bpm.set(m, key, val);
668 					} catch (BeanRuntimeException e) {
669 						onBeanSetterException(bpm, e);
670 						throw e;
671 					}
672 				}
673 			}
674 		}
675 
676 		BeanPropertyMeta cp = xmlMeta.getContentProperty();
677 		XmlFormat cpf = xmlMeta.getContentFormat();
678 		boolean trim = cp == null || ! cpf.isOneOf(MIXED_PWS, TEXT_PWS);
679 		ClassMeta<?> cpcm = (cp == null ? object() : cp.getClassMeta());
680 		StringBuilder sb = null;
681 		BeanRegistry breg = cp == null ? null : cp.getBeanRegistry();
682 		LinkedList<Object> l = null;
683 
684 		int depth = 0;
685 		do {
686 			int event = r.next();
687 			String currAttr;
688 			// We only care about text in MIXED mode.
689 			// Ignore if in ELEMENTS mode.
690 			if (event == CHARACTERS) {
691 				if (cp != null && cpf.isOneOf(MIXED, MIXED_PWS)) {
692 					if (cpcm.isCollectionOrArray()) {
693 						if (l == null)
694 							l = new LinkedList<>();
695 						l.add(getText(r, false));
696 					} else {
697 						cp.set(m, null, getText(r, trim));
698 					}
699 				} else if (cpf != ELEMENTS) {
700 					String s = getText(r, trim);
701 					if (s != null) {
702 						if (sb == null)
703 							sb = getStringBuilder();
704 						sb.append(s);
705 					}
706 				} else {
707 					// Do nothing...we're in ELEMENTS mode.
708 				}
709 			} else if (event == START_ELEMENT) {
710 				if (cp != null && cpf.isOneOf(TEXT, TEXT_PWS)) {
711 					String s = parseText(r);
712 					if (s != null) {
713 						if (sb == null)
714 							sb = getStringBuilder();
715 						sb.append(s);
716 					}
717 					depth--;
718 				} else if (cpf == XMLTEXT) {
719 					if (sb == null)
720 						sb = getStringBuilder();
721 					sb.append(getElementAsString(r));
722 					depth++;
723 				} else if (cp != null && cpf.isOneOf(MIXED, MIXED_PWS)) {
724 					if (isWhitespaceElement(r) && (breg == null || ! breg.hasName(r.getLocalName()))) {
725 						if (cpcm.isCollectionOrArray()) {
726 							if (l == null)
727 								l = new LinkedList<>();
728 							l.add(parseWhitespaceElement(r));
729 						} else {
730 							cp.set(m, null, parseWhitespaceElement(r));
731 						}
732 					} else {
733 						if (cpcm.isCollectionOrArray()) {
734 							if (l == null)
735 								l = new LinkedList<>();
736 							l.add(parseAnything(cpcm.getElementType(), cp.getName(), r, m.getBean(false), false, cp));
737 						} else {
738 							cp.set(m, null, parseAnything(cpcm, cp.getName(), r, m.getBean(false), false, cp));
739 						}
740 					}
741 				} else if (cp != null && cpf == ELEMENTS) {
742 					cp.add(m, null, parseAnything(cpcm.getElementType(), cp.getName(), r, m.getBean(false), false, cp));
743 				} else {
744 					currAttr = getNameProperty(r);
745 					if (currAttr == null)
746 						currAttr = getElementName(r);
747 					BeanPropertyMeta pMeta = xmlMeta.getPropertyMeta(currAttr);
748 					if (pMeta == null) {
749 						Object value = parseAnything(object(), currAttr, r, m.getBean(false), false, null);
750 						onUnknownProperty(currAttr, m, value);
751 					} else {
752 						setCurrentProperty(pMeta);
753 						XmlFormat xf = getXmlBeanPropertyMeta(pMeta).getXmlFormat();
754 						if (xf == COLLAPSED) {
755 							ClassMeta<?> et = pMeta.getClassMeta().getElementType();
756 							Object value = parseAnything(et, currAttr, r, m.getBean(false), false, pMeta);
757 							setName(et, value, currAttr);
758 							pMeta.add(m, currAttr, value);
759 						} else if (xf == ATTR)  {
760 							pMeta.set(m, currAttr, getAttributeValue(r, 0));
761 							r.nextTag();
762 						} else {
763 							ClassMeta<?> cm = pMeta.getClassMeta();
764 							Object value = parseAnything(cm, currAttr, r, m.getBean(false), false, pMeta);
765 							setName(cm, value, currAttr);
766 							pMeta.set(m, currAttr, value);
767 						}
768 						setCurrentProperty(null);
769 					}
770 				}
771 			} else if (event == END_ELEMENT) {
772 				if (depth > 0) {
773 					if (cpf == XMLTEXT) {
774 						if (sb == null)
775 							sb = getStringBuilder();
776 						sb.append(getElementAsString(r));
777 					}
778 					else
779 						throw new ParseException("End element found where one was not expected.  {0}", XmlUtils.toReadableEvent(r));
780 				}
781 				depth--;
782 			} else if (event == COMMENT) {
783 				// Ignore comments.
784 			} else {
785 				throw new ParseException("Unexpected event type: {0}", XmlUtils.toReadableEvent(r));
786 			}
787 		} while (depth >= 0);
788 
789 		if (cp != null && ! isNil) {
790 			if (sb != null)
791 				cp.set(m, null, sb.toString());
792 			else if (l != null)
793 				cp.set(m, null, XmlUtils.collapseTextNodes(l));
794 			else if (cpcm.isCollectionOrArray()) {
795 				Object o = cp.get(m, null);
796 				if (o == null)
797 					cp.set(m, cp.getName(), list());
798 			}
799 		}
800 
801 		returnStringBuilder(sb);
802 		return m;
803 	}

805 	private boolean isSpecialAttr(String key) {
806 		return key.equals(getBeanTypePropertyName(null)) || key.equals(getNamePropertyName());
807 	}

809 	private Object getUnknown(XmlReader r) throws IOException, ParseException, ExecutableException, XMLStreamException {
810 		if (r.getEventType() != START_ELEMENT) {
811 			throw new ParseException(this, "Parser must be on START_ELEMENT to read next text.");
812 		}
813 		JsonMap m = null;
814 
815 		// If this element has attributes, then it's always a JsonMap.
816 		if (r.getAttributeCount() > 0) {
817 			m = new JsonMap(this);
818 			for (int i = 0; i < r.getAttributeCount(); i++) {
819 				String key = getAttributeName(r, i);
820 				String val = r.getAttributeValue(i);
821 				if (! isSpecialAttr(key))
822 					m.put(key, val);
823 			}
824 		}
825 		int eventType = r.next();
826 		StringBuilder sb = getStringBuilder();
827 		while (eventType != END_ELEMENT) {
828 			if (eventType == CHARACTERS || eventType == CDATA || eventType == SPACE || eventType == ENTITY_REFERENCE) {
829 				sb.append(r.getText());
830 			} else if (eventType == PROCESSING_INSTRUCTION || eventType == COMMENT) {
831 				// skipping
832 			} else if (eventType == END_DOCUMENT) {
833 				throw new ParseException(this, "Unexpected end of document when reading element text content");
834 			} else if (eventType == START_ELEMENT) {
835 				// Oops...this has an element in it.
836 				// Parse it as a map.
837 				if (m == null)
838 					m = new JsonMap(this);
839 				int depth = 0;
840 				do {
841 					int event = (eventType == -1 ? r.nextTag() : eventType);
842 					String currAttr;
843 					if (event == START_ELEMENT) {
844 						depth++;
845 						currAttr = getNameProperty(r);
846 						if (currAttr == null)
847 							currAttr = getElementName(r);
848 						String key = convertAttrToType(null, currAttr, string());
849 						Object value = parseAnything(object(), currAttr, r, null, false, null);
850 						if (m.containsKey(key)) {
851 							Object o = m.get(key);
852 							if (o instanceof JsonList)
853 								((JsonList)o).add(value);
854 							else
855 								m.put(key, new JsonList(o, value).setBeanSession(this));
856 						} else {
857 							m.put(key, value);
858 						}
859 
860 					} else if (event == END_ELEMENT) {
861 						depth--;
862 						break;
863 					}
864 					eventType = -1;
865 				} while (depth > 0);
866 				break;
867 			} else {
868 				throw new ParseException(this, "Unexpected event type ''{0}''", eventType);
869 			}
870 			eventType = r.next();
871 		}
872 		String s = sb.toString().trim();
873 		returnStringBuilder(sb);
874 		s = decodeString(s);
875 		if (m != null) {
876 			if (! s.isEmpty())
877 				m.put("contents", s);
878 			return m;
879 		}
880 		return s;
881 	}
