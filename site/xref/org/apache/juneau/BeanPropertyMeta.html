137 			if (! p.format().isEmpty())

142 		private static ObjectSwap getPropertySwap(Swap s) throws RuntimeException {
143 			var c = s.value();
144 			if (isVoid(c))
145 				c = s.impl();
146 			if (isVoid(c))
147 				return null;
148 			var ci = info(c);
149 			if (ci.isChildOf(ObjectSwap.class)) {
150 				var ps = BeanCreator.of(ObjectSwap.class).type(c).run();
151 				if (nn(ps.forMediaTypes()))
152 					throw unsupportedOp("TODO - Media types on swaps not yet supported on bean properties.");
153 				if (nn(ps.withTemplate()))
154 					throw unsupportedOp("TODO - Templates on swaps not yet supported on bean properties.");
155 				return ps;
156 			}
157 			if (ci.isChildOf(Surrogate.class))
158 				throw unsupportedOp("TODO - Surrogate swaps not yet supported on bean properties.");
159 			throw rex("Invalid class used in @Swap annotation.  Must be a subclass of ObjectSwap or Surrogate. {0}", cn(c));
160 		}

207 		boolean validate(BeanContext bc, BeanRegistry parentBeanRegistry, Map<Class<?>,Class<?>[]> typeVarImpls, Set<String> bpro, Set<String> bpwo) throws Exception {
208 
209 			var bdClasses = list();
210 			var ap = bc.getAnnotationProvider();
211 
212 			if (field == null && getter == null && setter == null)
213 				return false;
214 
215 			if (field == null && setter == null && bc.isBeansRequireSettersForGetters() && ! isConstructorArg)
216 				return false;
217 
218 			canRead |= (nn(field) || nn(getter));
219 			canWrite |= (nn(field) || nn(setter));
220 
221 			var ifi = innerField == null ? null : info(innerField);
222 			var gi = getter == null ? null : info(getter);
223 			var si = setter == null ? null : info(setter);
224 
225 			if (nn(innerField)) {
226 				List<Beanp> lp = list();
227 				ap.find(Beanp.class, ifi).forEach(x -> lp.add(x.inner()));
228 				if (nn(field) || isNotEmpty(lp)) {
229 					// Only use field type if it's a bean property or has @Beanp annotation.
230 					// Otherwise, we want to infer the type from the getter or setter.
231 					rawTypeMeta = bc.resolveClassMeta(last(lp), innerField.getGenericType(), typeVarImpls);
232 					isUri |= (rawTypeMeta.isUri());
233 				}
234 				lp.forEach(x -> {
235 					if (swap == null)
236 						swap = getPropertySwap(x);
237 					if (! x.properties().isEmpty())
238 						properties = splita(x.properties());
239 					addAll(bdClasses, (Object[])x.dictionary());
240 					if (! x.ro().isEmpty())
241 						readOnly = Boolean.valueOf(x.ro());
242 					if (! x.wo().isEmpty())
243 						writeOnly = Boolean.valueOf(x.wo());
244 				});
245 				ap.find(Swap.class, ifi).stream().findFirst().ifPresent(x -> swap = getPropertySwap(x.inner()));
246 				isUri |= ap.has(Uri.class, ifi);
247 			}
248 
249 			if (nn(getter)) {
250 				var lp = ap.find(Beanp.class, gi).stream().map(AnnotationInfo::inner).toList();
251 				if (rawTypeMeta == null)
252 					rawTypeMeta = bc.resolveClassMeta(last(lp), getter.getGenericReturnType(), typeVarImpls);
253 				isUri |= (rawTypeMeta.isUri() || ap.has(Uri.class, gi));
254 				lp.forEach(x -> {
255 					if (swap == null)
256 						swap = getPropertySwap(x);
257 					if (nn(properties) && ! x.properties().isEmpty())
258 						properties = splita(x.properties());
259 					addAll(bdClasses, (Object[])x.dictionary());
260 					if (! x.ro().isEmpty())
261 						readOnly = Boolean.valueOf(x.ro());
262 					if (! x.wo().isEmpty())
263 						writeOnly = Boolean.valueOf(x.wo());
264 				});
265 				ap.find(Swap.class, gi).stream().map(AnnotationInfo::inner).forEach(x -> swap = getPropertySwap(x));
266 			}
267 
268 			if (nn(setter)) {
269 				var lp = ap.find(Beanp.class, si).stream().map(AnnotationInfo::inner).toList();
270 				if (rawTypeMeta == null)
271 					rawTypeMeta = bc.resolveClassMeta(last(lp), setter.getGenericParameterTypes()[0], typeVarImpls);
272 				isUri |= (rawTypeMeta.isUri() || ap.has(Uri.class, si));
273 				lp.forEach(x -> {
274 					if (swap == null)
275 						swap = getPropertySwap(x);
276 					if (nn(properties) && ! x.properties().isEmpty())
277 						properties = splita(x.properties());
278 					addAll(bdClasses, (Object[])x.dictionary());
279 					if (! x.ro().isEmpty())
280 						readOnly = Boolean.valueOf(x.ro());
281 					if (! x.wo().isEmpty())
282 						writeOnly = Boolean.valueOf(x.wo());
283 				});
284 				ap.find(Swap.class, si).stream().map(AnnotationInfo::inner).forEach(x -> swap = getPropertySwap(x));
285 			}
286 
287 			if (rawTypeMeta == null)
288 				return false;
289 
290 			this.beanRegistry = new BeanRegistry(bc, parentBeanRegistry, bdClasses.toArray(new Class<?>[0]));
291 
292 			isDyna = "*".equals(name);
293 
294 			// Do some annotation validation.
295 			var ci = rawTypeMeta;
296 			if (nn(getter)) {
297 				var pt = getter.getParameterTypes();
298 				if (isDyna) {
299 					if (ci.isChildOf(Map.class) && pt.length == 0) {
300 						isDynaGetterMap = true;
301 					} else if (pt.length == 1 && pt[0] == String.class) {
302 						// OK.
303 					} else {
304 						return false;
305 					}
306 				} else {
307 					if (! ci.isChildOf(getter.getReturnType()))
308 						return false;
309 				}
310 			}
311 			if (nn(setter)) {
312 				var pt = setter.getParameterTypes();
313 				if (isDyna) {
314 					if (pt.length == 2 && pt[0] == String.class) {
315 						// OK.
316 					} else {
317 						return false;
318 					}
319 				} else {
320 					if (pt.length != 1 || ! ci.isChildOf(pt[0]))
321 						return false;
322 				}
323 			}
324 			if (nn(field)) {
325 				if (isDyna) {
326 					if (! info(field.getType()).isChildOf(Map.class))
327 						return false;
328 				} else {
329 					if (! ci.isChildOf(field.getType()))
330 						return false;
331 				}
332 			}
333 
334 			if (isDyna) {
335 				rawTypeMeta = rawTypeMeta.getValueType();
336 				if (rawTypeMeta == null)
337 					rawTypeMeta = bc.object();
338 			}
339 			if (rawTypeMeta == null)
340 				return false;
341 
342 			if (typeMeta == null)
343 				typeMeta = (nn(swap) ? bc.getClassMeta(swap.getSwapClass().innerType()) : rawTypeMeta == null ? bc.object() : rawTypeMeta);
344 			if (typeMeta == null)
345 				typeMeta = rawTypeMeta;
346 
347 			if (bpro.contains(name) || bpro.contains("*"))
348 				readOnly = true;
349 			if (bpwo.contains(name) || bpwo.contains("*"))
350 				writeOnly = true;
351 
352 			return true;
353 		}

434 	/**
435 	 * Adds a value to a {@link Collection} or array property.
436 	 *
437 	 * <p>
438 	 * Note that adding values to an array property is inefficient for large arrays since it must copy the array into a
439 	 * larger array on each operation.
440 	 *
441 	 * @param m The bean of the field being set.
442 	 * @param pName
443 	 * 	The property name if this is a dyna property (i.e. <js>"*"</js>).
444 	 * 	<br>Otherwise can be <jk>null</jk>.
445 	 * @param value The value to add to the field.
446 	 * @throws BeanRuntimeException If field is not a collection or array.
447 	 */
448 	public void add(BeanMap<?> m, String pName, Object value) throws BeanRuntimeException {
449 
450 		// Read-only beans get their properties stored in a cache.
451 		if (m.bean == null) {
452 			if (! m.propertyCache.containsKey(name))
453 				m.propertyCache.put(name, new JsonList(m.getBeanSession()));
454 			((JsonList)m.propertyCache.get(name)).add(value);
455 			return;
456 		}
457 
458 		var session = m.getBeanSession();
459 
460 		var isCollection = rawTypeMeta.isCollection();
461 		var isArray = rawTypeMeta.isArray();
462 
463 		if (! (isCollection || isArray))
464 			throw bex(beanMeta.c, "Attempt to add element to property ''{0}'' which is not a collection or array", name);
465 
466 		var bean = m.getBean(true);
467 
468 		var elementType = rawTypeMeta.getElementType();
469 
470 		try {
471 			var v = session.convertToType(value, elementType);
472 
473 			if (isCollection) {
474 				var c = (Collection)invokeGetter(bean, pName);
475 
476 				var c2 = (Collection)null;
477 				if (nn(c)) {
478 					if (canAddTo(c)) {
479 						c.add(v);
480 						return;
481 					}
482 					c2 = c;
483 				}
484 
485 				if (rawTypeMeta.canCreateNewInstance())
486 					c = (Collection)rawTypeMeta.newInstance();
487 				else
488 					c = new JsonList(session);
489 
490 				if (c2 != null)
491 					c.addAll(c2);
492 
493 				c.add(v);
494 
495 				invokeSetter(bean, pName, c);
496 
497 			} else /* isArray() */ {
498 
499 				if (m.arrayPropertyCache == null)
500 					m.arrayPropertyCache = new TreeMap<>();
501 
502 				List l = m.arrayPropertyCache.get(name);
503 				if (l == null) {
504 					l = new LinkedList();  // ArrayLists and LinkLists appear to perform equally.
505 					m.arrayPropertyCache.put(name, l);
506 
507 					// Copy any existing array values into the temporary list.
508 					var oldArray = invokeGetter(bean, pName);
509 					copyArrayToList(oldArray, l);
510 				}
511 
512 				// Add new entry to our array.
513 				l.add(v);
514 			}
515 
516 		} catch (BeanRuntimeException e) {
517 			throw e;
518 		} catch (Exception e) {
519 			throw bex(e);
520 		}
521 	}

523 	/**
524 	 * Adds a value to a {@link Map} or bean property.
525 	 *
526 	 * @param m The bean of the field being set.
527 	 * @param pName
528 	 * 	The property name if this is a dyna property (i.e. <js>"*"</js>).
529 	 * 	<br>Otherwise can be <jk>null</jk>.
530 	 * @param key The key to add to the field.
531 	 * @param value The value to add to the field.
532 	 * @throws BeanRuntimeException If field is not a map or array.
533 	 */
534 	public void add(BeanMap<?> m, String pName, String key, Object value) throws BeanRuntimeException {
535 
536 		// Read-only beans get their properties stored in a cache.
537 		if (m.bean == null) {
538 			if (! m.propertyCache.containsKey(name))
539 				m.propertyCache.put(name, new JsonMap(m.getBeanSession()));
540 			((JsonMap)m.propertyCache.get(name)).append(key.toString(), value);
541 			return;
542 		}
543 
544 		var session = m.getBeanSession();
545 
546 		var isMap = rawTypeMeta.isMap();
547 		var isBean = rawTypeMeta.isBean();
548 
549 		if (! (isBean || isMap))
550 			throw bex(beanMeta.c, "Attempt to add key/value to property ''{0}'' which is not a map or bean", name);
551 
552 		var bean = m.getBean(true);
553 
554 		var elementType = rawTypeMeta.getElementType();
555 
556 		try {
557 			var v = session.convertToType(value, elementType);
558 
559 			if (isMap) {
560 				var map = (Map)invokeGetter(bean, pName);
561 
562 				if (nn(map)) {
563 					map.put(key, v);
564 					return;
565 				}
566 
567 				if (rawTypeMeta.canCreateNewInstance())
568 					map = (Map)rawTypeMeta.newInstance();
569 				else
570 					map = new JsonMap(session);
571 
572 				map.put(key, v);
573 
574 				invokeSetter(bean, pName, map);
575 
576 			} else /* isBean() */ {
577 
578 				var b = invokeGetter(bean, pName);
579 
580 				if (nn(b)) {
581 					var bm = session.toBeanMap(b);
582 					bm.put(key, v);
583 					return;
584 				}
585 
586 				if (rawTypeMeta.canCreateNewInstance(m.getBean(false))) {
587 					b = rawTypeMeta.newInstance();
588 					var bm = session.toBeanMap(b);
589 					bm.put(key, v);
590 				}
591 
592 				invokeSetter(bean, pName, b);
593 			}
594 
595 		} catch (BeanRuntimeException e) {
596 			throw e;
597 		} catch (Exception e) {
598 			throw bex(e);
599 		}
600 	}

945 	/**
946 	 * Walks up the class hierarchy to find parent methods that match the signature of the given method.
947 	 * Executes the consumer for each parent method found, starting from the topmost parent down to
948 	 * the immediate parent (but not including the method itself).
949 	 *
950 	 * @param method The method to find parents for.
951 	 * @param consumer The action to perform for each parent method.
952 	 */
953 	private static void forEachParentMethod(Method method, Consumer<Method> consumer) {
954 		// TODO - Convert to use MethodInfo parameters
955 		if (method == null)
956 			return;
957 
958 		var methodName = method.getName();
959 		var paramTypes = method.getParameterTypes();
960 		var declaringClass = method.getDeclaringClass();
961 
962 		// Collect parent methods in a list (we'll reverse it to get parent-to-child order)
963 		var parentMethods = new LinkedList<Method>();
964 
965 		// Walk up the class hierarchy
966 		var currentClass = declaringClass.getSuperclass();
967 		while (nn(currentClass) && currentClass != Object.class) {
968 			try {
969 				var parentMethod = currentClass.getDeclaredMethod(methodName, paramTypes);
970 				parentMethods.add(parentMethod);
971 			} catch (@SuppressWarnings("unused") NoSuchMethodException e) {
972 				// No matching method in this parent class, continue up the hierarchy
973 			}
974 			currentClass = currentClass.getSuperclass();
975 		}
976 
977 		// Also check interfaces
978 		for (var iface : declaringClass.getInterfaces()) {
979 			try {
980 				var ifaceMethod = iface.getDeclaredMethod(methodName, paramTypes);
981 				parentMethods.add(ifaceMethod);
982 			} catch (@SuppressWarnings("unused") NoSuchMethodException e) {
983 				// No matching method in this interface
984 			}
985 		}
986 
987 		// Process in reverse order (parent-to-child) to match the "ParentFirst" semantics
988 		for (var i = parentMethods.size() - 1; i >= 0; i--) {
989 			consumer.accept(parentMethods.get(i));
990 		}
991 	}

1266 		} catch (Exception e) {
