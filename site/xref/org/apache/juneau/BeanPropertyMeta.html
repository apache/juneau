38  /**
39   * Contains metadata about a bean property.
40   *
41   * <p>
42   * Contains information such as type of property (e.g. field/getter/setter), class type of property value, and whether
43   * any transforms are associated with this property.
44   *
45   * <p>
46   * Developers will typically not need access to this class.  The information provided by it is already exposed through
47   * several methods on the {@link BeanMap} API.
48   *
49   * <h5 class='section'>See Also:</h5><ul>
50  
51   * </ul>
52   */
53  @SuppressWarnings({ "rawtypes", "unchecked" })
54  public class BeanPropertyMeta implements Comparable<BeanPropertyMeta> {

56  	final BeanMeta<?> beanMeta;                               // The bean that this property belongs to.
57  	private final BeanContext beanContext;                    // The context that created this meta.

60  	private final Field field;                                // The bean property field (if it has one).

63  	private final boolean isUri;                              // True if this is a URL/URI or annotated with @URI.

170 		boolean validate(BeanContext bc, BeanRegistry parentBeanRegistry, Map<Class<?>,Class<?>[]> typeVarImpls, Set<String> bpro, Set<String> bpwo) throws Exception {
171 
172 			List<Class<?>> bdClasses = list();
173 
174 			if (field == null && getter == null && setter == null)
175 				return false;
176 
177 			if (field == null && setter == null && bc.isBeansRequireSettersForGetters() && ! isConstructorArg)
178 				return false;
179 
180 			canRead |= (field != null || getter != null);
181 			canWrite |= (field != null || setter != null);
182 
183 			if (innerField != null) {
184 				List<Beanp> lp = list();
185 				bc.forEachAnnotation(Beanp.class, innerField, x -> true, x -> lp.add(x));
186 				if (field != null || Utils.isNotEmpty(lp)) {
187 					// Only use field type if it's a bean property or has @Beanp annotation.
188 					// Otherwise, we want to infer the type from the getter or setter.
189 					rawTypeMeta = bc.resolveClassMeta(last(lp), innerField.getGenericType(), typeVarImpls);
190 					isUri |= (rawTypeMeta.isUri());
191 				}
192 				lp.forEach(x -> {
193 					if (! x.properties().isEmpty())
194 						properties = splita(x.properties());
195 					addAll(bdClasses, x.dictionary());
196 					if (! x.ro().isEmpty())
197 						readOnly = Boolean.valueOf(x.ro());
198 					if (! x.wo().isEmpty())
199 						writeOnly = Boolean.valueOf(x.wo());
200 				});
201 				bc.forEachAnnotation(Swap.class, innerField, x -> true, x -> swap = getPropertySwap(x));
202 				isUri |= bc.firstAnnotation(Uri.class, innerField, x->true) != null;
203 			}
204 
205 			if (getter != null) {
206 				List<Beanp> lp = list();
207 				bc.forEachAnnotation(Beanp.class, getter, x -> true, x -> lp.add(x));
208 				if (rawTypeMeta == null)
209 					rawTypeMeta = bc.resolveClassMeta(last(lp), getter.getGenericReturnType(), typeVarImpls);
210 				isUri |= (rawTypeMeta.isUri() || bc.hasAnnotation(Uri.class, getter));
211 				lp.forEach(x -> {
212 					if (properties != null && ! x.properties().isEmpty())
213 						properties = splita(x.properties());
214 					addAll(bdClasses, x.dictionary());
215 					if (! x.ro().isEmpty())
216 						readOnly = Boolean.valueOf(x.ro());
217 					if (! x.wo().isEmpty())
218 						writeOnly = Boolean.valueOf(x.wo());
219 				});
220 				bc.forEachAnnotation(Swap.class, getter, x -> true, x -> swap = getPropertySwap(x));
221 			}
222 
223 			if (setter != null) {
224 				List<Beanp> lp = list ();
225 				bc.forEachAnnotation(Beanp.class, setter, x -> true, x -> lp.add(x));
226 				if (rawTypeMeta == null)
227 					rawTypeMeta = bc.resolveClassMeta(last(lp), setter.getGenericParameterTypes()[0], typeVarImpls);
228 				isUri |= (rawTypeMeta.isUri() || bc.hasAnnotation(Uri.class, setter));
229 				lp.forEach(x -> {
230 					if (swap == null)
231 						swap = getPropertySwap(x);
232 					if (properties != null && ! x.properties().isEmpty())
233 						properties = splita(x.properties());
234 					addAll(bdClasses, x.dictionary());
235 					if (! x.ro().isEmpty())
236 						readOnly = Boolean.valueOf(x.ro());
237 					if (! x.wo().isEmpty())
238 						writeOnly = Boolean.valueOf(x.wo());
239 				});
240 				bc.forEachAnnotation(Swap.class, setter, x -> true, x -> swap = getPropertySwap(x));
241 			}
242 
243 			if (rawTypeMeta == null)
244 				return false;
245 
246 			this.beanRegistry = new BeanRegistry(beanContext, parentBeanRegistry, bdClasses.toArray(new Class<?>[0]));
247 
248 			isDyna = "*".equals(name);
249 
250 			// Do some annotation validation.
251 			ClassInfo ci = rawTypeMeta.getInfo();
252 			if (getter != null) {
253 				Class<?>[] pt = getter.getParameterTypes();
254 				if (isDyna) {
255 					if (ci.isChildOf(Map.class) && pt.length == 0) {
256 						isDynaGetterMap = true;
257 					} else if (pt.length == 1 && pt[0] == String.class) {
258 						// OK.
259 					} else {
260 						return false;
261 					}
262 				} else {
263 					if (! ci.isChildOf(getter.getReturnType()))
264 						return false;
265 				}
266 			}
267 			if (setter != null) {
268 				Class<?>[] pt = setter.getParameterTypes();
269 				if (isDyna) {
270 					if (pt.length == 2 && pt[0] == String.class) {
271 						// OK.
272 					} else {
273 						return false;
274 					}
275 				} else {
276 					if (pt.length != 1 || ! ci.isChildOf(pt[0]))
277 						return false;
278 				}
279 			}
280 			if (field != null) {
281 				if (isDyna) {
282 					if (! ClassInfo.of(field.getType()).isChildOf(Map.class))
283 						return false;
284 				} else {
285 					if (! ci.isChildOf(field.getType()))
286 						return false;
287 				}
288 			}
289 
290 			if (isDyna) {
291 				rawTypeMeta = rawTypeMeta.getValueType();
292 				if (rawTypeMeta == null)
293 					rawTypeMeta = beanContext.object();
294 			}
295 			if (rawTypeMeta == null)
296 				return false;
297 
298 			if (typeMeta == null)
299 				typeMeta = (swap != null ? beanContext.getClassMeta(swap.getSwapClass().innerType()) : rawTypeMeta == null ? beanContext.object() : rawTypeMeta);
300 			if (typeMeta == null)
301 				typeMeta = rawTypeMeta;
302 
303 			if (bpro.contains(name) || bpro.contains("*"))
304 				readOnly = true;
305 			if (bpwo.contains(name) || bpwo.contains("*"))
306 				writeOnly = true;
307 
308 			return true;
309 		}

318 		private ObjectSwap getPropertySwap(Beanp p) {
319 			if (! p.format().isEmpty())
320 				return BeanCreator.of(ObjectSwap.class).type(StringFormatSwap.class).arg(String.class, p.format()).run();
321 			return null;
322 		}

324 		private ObjectSwap getPropertySwap(Swap s) throws RuntimeException {
325 			Class<?> c = s.value();
326 			if (isVoid(c))
327 				c = s.impl();
328 			if (isVoid(c))
329 				return null;
330 			ClassInfo ci = ClassInfo.of(c);
331 			if (ci.isChildOf(ObjectSwap.class)) {
332 				ObjectSwap ps = BeanCreator.of(ObjectSwap.class).type(c).run();
333 				if (ps.forMediaTypes() != null)
334 					throw new UnsupportedOperationException("TODO - Media types on swaps not yet supported on bean properties.");
335 				if (ps.withTemplate() != null)
336 					throw new UnsupportedOperationException("TODO - Templates on swaps not yet supported on bean properties.");
337 				return ps;
338 			}
339 			if (ci.isChildOf(Surrogate.class))
340 				throw new UnsupportedOperationException("TODO - Surrogate swaps not yet supported on bean properties.");
341 			throw new BasicRuntimeException("Invalid class used in @Swap annotation.  Must be a subclass of ObjectSwap or Surrogate. {0}", c);
342 		}

543 	 * Equivalent to calling {@link BeanMap#get(Object)}, but is faster since it avoids looking up the property meta.
544 	 *
545 	 * @param m The bean map to get the transformed value from.
546 	 * @param pName
547 	 * 	The property name if this is a dyna property (i.e. <js>"*"</js>).
548 	 * 	<br>Otherwise can be <jk>null</jk>.
549 	 * @return
550 	 * 	The property value.
551 	 * 	<br>Returns <jk>null</jk> if this is a write-only property.
552 	 */
553 	public Object get(BeanMap<?> m, String pName) {
554 		return m.meta.onReadProperty(m.bean, pName, getInner(m, pName));
555 	}

557 	private Object getInner(BeanMap<?> m, String pName) {
558 		try {
559 
560 			if (writeOnly)
561 				return null;
562 
563 			if (overrideValue != null)
564 				return overrideValue;
565 
566 			// Read-only beans have their properties stored in a cache until getBean() is called.
567 			Object bean = m.bean;
568 			if (bean == null)
569 				return m.propertyCache.get(name);
570 
571 			return toSerializedForm(m.getBeanSession(), getRaw(m, pName));
572 
573 		} catch (Throwable e) {
574 			if (beanContext.isIgnoreInvocationExceptionsOnGetters()) {
575 				if (rawTypeMeta.isPrimitive())
576 					return rawTypeMeta.getPrimitiveDefault();
577 				return null;
578 			}
579 			throw new BeanRuntimeException(e, beanMeta.c, "Exception occurred while getting property ''{0}''", name);
580 		}
581 	}

583 	/**
584 	 * Equivalent to calling {@link BeanMap#getRaw(Object)}, but is faster since it avoids looking up the property meta.
585 	 *
586 	 * @param m The bean map to get the transformed value from.
587 	 * @param pName
588 	 * 	The property name if this is a dyna property (i.e. <js>"*"</js>).
589 	 * 	<br>Otherwise can be <jk>null</jk>.
590 	 * @return The raw property value.
591 	 */
592 	public Object getRaw(BeanMap<?> m, String pName) {
593 		try {
594 			// Read-only beans have their properties stored in a cache until getBean() is called.
595 			Object bean = m.bean;
596 			if (bean == null)
597 				return m.propertyCache.get(name);
598 
599 			return invokeGetter(bean, pName);
600 
601 		} catch (Throwable e) {
602 			if (beanContext.isIgnoreInvocationExceptionsOnGetters()) {
603 				if (rawTypeMeta.isPrimitive())
604 					return rawTypeMeta.getPrimitiveDefault();
605 				return null;
606 			}
607 			throw new BeanRuntimeException(e, beanMeta.c, "Exception occurred while getting property ''{0}''", name);
608 		}
609 	}

611 	/**
612 	 * Converts a raw bean property value to serialized form.
613 	 * Applies transforms and child property filters.
614 	 */
615 	Object toSerializedForm(BeanSession session, Object o) {
616 		try {
617 			o = transform(session, o);
618 			if (o == null)
619 				return null;
620 			if (properties != null) {
621 				if (rawTypeMeta.isArray()) {
622 					Object[] a = (Object[])o;
623 					List l = new DelegateList(rawTypeMeta);
624 					ClassMeta childType = rawTypeMeta.getElementType();
625 					for (Object c : a)
626 						l.add(applyChildPropertiesFilter(session, childType, c));
