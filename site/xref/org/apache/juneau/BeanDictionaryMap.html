17  package org.apache.juneau;
18  
19  import static org.apache.juneau.common.utils.Utils.*;
20  
21  import java.lang.reflect.*;
22  import java.util.*;
23  
24  import org.apache.juneau.annotation.*;
25  
26  /**
27   * Represents a map of dictionary type names to bean classes that make up a bean dictionary.
28   *
29   * <p>
30   * In general, this approach for defining dictionary names for classes is used when it's not possible to use the
31   * {@link Bean#typeName() @Bean(typeName)} annotation.
32   *
33   * <h5 class='section'>Example:</h5>
34   * <p class='bjava'>
35   * 	<jc>// A bean dictionary map consisting of classes without @Bean(typeName) annotations</jc>
36   * 	<jc>// that require type names to be explicitly specified.</jc>
37   * 	<jk>public class</jk> MyBeanDictionaryMap <jk>extends</jk> BeanDictionaryMap {
38   *
39   * 		<jc>// Must provide a no-arg constructor!</jc>
40   * 		<jk>public</jk> MyBeanDictionaryMap() {
41   * 			append(<js>"MyBean"</js>, MyBean.<jk>class</jk>);
42   * 			append(<js>"MyBeanArray"</js>, MyBean[].<jk>class</jk>);
43   * 			append(<js>"StringArray"</js>, String[].<jk>class</jk>);
44   * 			append(<js>"String2dArray"</js>, String[][].<jk>class</jk>);
45   * 			append(<js>"IntArray"</js>, <jk>int</jk>[].<jk>class</jk>);
46   * 			append(<js>"Int2dArray"</js>, <jk>int</jk>[][].<jk>class</jk>);
47   * 			append(<js>"LinkedList"</js>, LinkedList.<jk>class</jk>);
48   * 			append(<js>"TreeMap"</js>, TreeMap.<jk>class</jk>);
49   * 			append(<js>"LinkedListOfInts"</js>, LinkedList.<jk>class</jk>, Integer.<jk>class</jk>);
50   * 			append(<js>"LinkedListOfR1"</js>, LinkedList.<jk>class</jk>, R1.<jk>class</jk>);
51   * 			append(<js>"LinkedListOfCalendar"</js>, LinkedList.<jk>class</jk>, Calendar.<jk>class</jk>);
52   * 		}
53   * 	}
54   *
55   * 	<jc>// Use it in a parser.</jc>
56   * 	ReaderParser <jv>parser</jv> = JsonParser
57   * 		.<jsm>create</jsm>()
58   * 		.dictionary(MyBeanDictionaryMap.<jk>class</jk>)
59   * 		.build();
60   * </p>
61   *
62   * <p>
63   * Subclasses must implement a public no-arg constructor so that it can be instantiated by the bean context code.
64   *
65   * <h5 class='section'>See Also:</h5><ul>
66   * </ul>
67   *
68   * @serial exclude
69   */
70  @SuppressWarnings("rawtypes")
71  public class BeanDictionaryMap extends LinkedHashMap<String,Object> {
72  	private static final long serialVersionUID = 1L;
73  
74  	/**
75  	 * Constructor.
76  	 */
77  	protected BeanDictionaryMap() {}
78  
79  	private void assertValidParameter(Object o) {
80  		if (nn(o)) {
81  			if (o instanceof Class)
82  				return;
83  			if (isArray(o)) {
84  				for (int i = 0; i < Array.getLength(o); i++)
85  					assertValidParameter(Array.get(o, i));
86  				return;
87  			}
88  		}
89  		throw new BeanRuntimeException("Invalid object type passed to BeanDictionaryMap: ''{0}''.  Only objects of type Class or Object[] containing Class or Object[] objects can be used.");
90  	}
91  
92  	/**
93  	 * Add a dictionary name mapping for the specified class.
94  	 *
95  	 * @param typeName The dictionary name of the class.
96  	 * @param c The class represented by the dictionary name.
97  	 * @return This object.
98  	 */
99  	protected BeanDictionaryMap append(String typeName, Class<?> c) {
100 		put(typeName, c);
101 		return this;
102 	}
103 
104 	/**
105 	 * Add a dictionary name mapping for the specified collection class with the specified entry class.
106 	 *
107 	 * @param typeName The dictionary name of the class.
108 	 * @param collectionClass The collection implementation class.
109 	 * @param entryClass The entry class.
110 	 * @return This object.
111 	 */
112 	protected BeanDictionaryMap append(String typeName, Class<? extends Collection> collectionClass, Object entryClass) {
113 		assertValidParameter(entryClass);
114 		put(typeName, a(collectionClass, entryClass));
115 		return this;
116 	}
117 
118 	/**
119 	 * Add a dictionary name mapping for the specified map class with the specified key and value classes.
120 	 *
121 	 * @param typeName The dictionary name of the class.
122 	 * @param mapClass The map implementation class.
123 	 * @param keyClass The key class.
124 	 * @param valueClass The value class.
125 	 * @return This object.
126 	 */
127 	protected BeanDictionaryMap append(String typeName, Class<? extends Map> mapClass, Object keyClass, Object valueClass) {
128 		assertValidParameter(keyClass);
129 		assertValidParameter(valueClass);
130 		put(typeName, a(mapClass, keyClass, valueClass));
131 		return this;
132 	}
133 }
