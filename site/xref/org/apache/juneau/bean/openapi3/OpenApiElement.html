13  package org.apache.juneau.bean.openapi3;
14  
15  import static org.apache.juneau.common.internal.Utils.*;
16  import static org.apache.juneau.internal.CollectionUtils.*;
17  import static org.apache.juneau.internal.ConverterUtils.*;
18  
19  import java.lang.reflect.*;
20  import java.util.*;
21  
22  import org.apache.juneau.annotation.*;
23  import org.apache.juneau.collections.*;
24  import org.apache.juneau.internal.*;
25  import org.apache.juneau.json.*;
26  
27  /**
28   * Root class for all Swagger beans.
29   */
30  @FluentSetters
31  public abstract class OpenApiElement {
32  
33  	private boolean strict;
34  	private Map<String,Object> extra;
35  
36  	OpenApiElement() {}
37  
38  	OpenApiElement(OpenApiElement copyFrom) {
39  		this.strict = copyFrom.strict;
40  		this.extra = copyOf(copyFrom.extra);
41  	}
42  
43  	/**
44  	 * Returns <jk>true</jk> if contents should be validated per the Swagger spec.
45  	 *
46  	 * @return <jk>true</jk> if contents should be validated per the Swagger spec.
47  	 */
48  	protected boolean isStrict() {
49  		return strict;
50  	}
51  
52  	/**
53  	 * Sets strict mode on this bean.
54  	 *
55  	 * @return This object
56  	 */
57  	protected OpenApiElement strict() {
58  		strict = true;
59  		return this;
60  	}
61  
62  	/**
63  	 * Sets strict mode on this bean.
64  	 *
65  	 * @param value
66  	 * 	The new value for this property.
67  	 * 	<br>Non-boolean values will be converted to boolean using <code>Boolean.<jsm>valueOf</jsm>(value.toString())</code>.
68  	 * 	<br>Can be <jk>null</jk> (interpreted as <jk>false</jk>).
69  	 * @return This object
70  	 */
71  	protected OpenApiElement strict(Object value) {
72  		strict = value != null && toBoolean(value);
73  		return this;
74  	}
75  
76  	/**
77  	 * Generic property getter.
78  	 *
79  	 * <p>
80  	 * Can be used to retrieve non-standard Swagger fields such as <js>"$ref"</js>.
81  	 *
82  	 * @param property The property name to retrieve.
83  	 * @param type The datatype to cast the value to.
84  	 * @param <T> The datatype to cast the value to.
85  	 * @return The property value, or <jk>null</jk> if the property does not exist or is not set.
86  	 */
87  	public <T> T get(String property, Class<T> type) {
88  		assertArgNotNull("property", property);
89  		return toType(get(property), type);
90  	}
91  
92  	/**
93  	 * Generic property getter.
94  	 *
95  	 * <p>
96  	 * Can be used to retrieve non-standard Swagger fields such as <js>"$ref"</js>.
97  	 *
98  	 * @param property The property name to retrieve.  Must not be <jk>null</jk>.
99  	 * @return The property value, or <jk>null</jk> if the property does not exist or is not set.
100 	 */
101 	@Beanp("*")
102 	public Object get(String property) {
103 		assertArgNotNull("property", property);
104 		if (extra == null)
105 			return null;
106 		return extra.get(property);
107 	}
108 
109 	/**
110 	 * Generic property setter.
111 	 *
112 	 * <p>
113 	 * Can be used to set non-standard Swagger fields such as <js>"$ref"</js>.
114 	 *
115 	 * @param property The property name to set.  Must not be <jk>null</jk>.
116 	 * @param value The new value for the property.
117 	 * @return This object
118 	 * @throws RuntimeException if strict mode is enabled.
119 	 */
120 	@Beanp("*")
121 	public OpenApiElement set(String property, Object value) {
122 		assertArgNotNull("property", property);
123 		if (strict)
124 			throw new RuntimeException("Cannot set property '" + property + "' in strict mode.");
125 		if (extra == null)
126 			extra = map();
127 		extra.put(property, value);
128 		return this;
129 	}
130 
131 	/**
132 	 * Generic property keyset.
133 	 *
134 	 * @return
135 	 * 	All the non-standard keys on this element.
136 	 * 	<br>Never <jk>null</jk>.
137 	 */
138 	@Beanp("*")
139 	public Set<String> extraKeys() {
140 		return extra == null ? Collections.emptySet() : extra.keySet();
141 	}
142 
143 	/**
144 	 * Returns all the keys on this element.
145 	 *
146 	 * @return
147 	 * 	All the keys on this element.
148 	 * 	<br>Never <jk>null</jk>.
149 	 */
150 	public Set<String> keySet() {
151 		return extraKeys();
152 	}
153 
154 	/**
155 	 * Returns a copy of this swagger element as a modifiable map.
156 	 *
157 	 * <p>
158 	 * Each call produces a new map.
159 	 *
160 	 * @return A map containing all the values in this swagger element.
161 	 */
162 	public JsonMap asMap() {
163 		var m = new JsonMap();
164 		for (var s : keySet())
165 			m.put(s, get(s, Object.class));
166 		return m;
167 	}
168 
169 	// <FluentSetters>
170 
171 	// </FluentSetters>
172 
173 	@Override /* Object */
174 	public String toString() {
175 		if (Modifier.isAbstract(getClass().getModifiers())) {
176 			return JsonSerializer.DEFAULT_SORTED.toString(extra);
177 		}
178 		return JsonSerializer.DEFAULT_SORTED.toString(this);
179 	}
180 }
