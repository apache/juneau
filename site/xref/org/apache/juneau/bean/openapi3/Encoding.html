26  /**
27   * A single encoding definition applied to a single schema property.
28   *
29   * <p>
30   * The Encoding Object is a single encoding definition applied to a single schema property. It allows you to define 
31   * how a property should be serialized when it's part of a request or response body with a specific media type.
32   *
33   * <h5 class='section'>OpenAPI Specification:</h5>
34   * <p>
35   * The Encoding Object is composed of the following fields:
36   * <ul class='spaced-list'>
37   * 	<li><c>contentType</c> (string) - The Content-Type for encoding a specific property. Default value depends on the property type
38   * 	<li><c>headers</c> (map of {@link HeaderInfo}) - A map allowing additional information to be provided as headers
39   * 	<li><c>style</c> (string) - Describes how a specific property value will be serialized depending on its type
40   * 	<li><c>explode</c> (boolean) - When this is true, property values of type array or object generate separate parameters for each value
41   * 	<li><c>allowReserved</c> (boolean) - Determines whether the parameter value should allow reserved characters
42   * </ul>
43   *
44   * <h5 class='section'>Example:</h5>
45   * <p class='bcode'>
46   * 	<jc>// Construct using SwaggerBuilder.</jc>
47   * 	Encoding <jv>x</jv> = <jsm>encoding</jsm>()
48   * 		.setContentType(<js>"application/x-www-form-urlencoded"</js>)
49   * 		.setStyle(<js>"form"</js>)
50   * 		.setExplode(<jk>true</jk>);
51   *
52   * 	<jc>// Serialize using JsonSerializer.</jc>
53   * 	String <jv>json</jv> = Json.<jsm>from</jsm>(<jv>x</jv>);
54   *
55   * 	<jc>// Or just use toString() which does the same as above.</jc>
56   * 	<jv>json</jv> = <jv>x</jv>.toString();
57   * </p>
58   * <p class='bcode'>
59   * 	<jc>// Output</jc>
60   * 	{
61   * 		<js>"contentType"</js>: <js>"application/x-www-form-urlencoded"</js>,
62   * 		<js>"style"</js>: <js>"form"</js>,
63   * 		<js>"explode"</js>: <jk>true</jk>
64   * 	}
65   * </p>
66   *
67   * <h5 class='section'>See Also:</h5><ul>
68   * 	<li class='link'><a class="doclink" href="https://spec.openapis.org/oas/v3.0.0#encoding-object">OpenAPI Specification &gt; Encoding Object</a>
69   * 	<li class='link'><a class="doclink" href="https://swagger.io/docs/specification/describing-request-body/">OpenAPI Describing Request Body</a>
70   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauBeanOpenApi3">juneau-bean-openapi-v3</a>
71   * </ul>
72   */

278 	public <T> T get(String property, Class<T> type) {
279 		assertArgNotNull("property", property);
280 		return switch (property) {
281 			case "contentType" -> toType(getContentType(), type);
282 			case "style" -> toType(getStyle(), type);
283 			case "headers" -> toType(getHeaders(), type);
284 			case "explode" -> toType(getExplode(), type);
285 			case "allowReserved" -> toType(getAllowReserved(), type);
286 			default -> super.get(property, type);
287 		};
288 	}

290 	@Override /* OpenApiElement */
291 	public Encoding set(String property, Object value) {
292 		assertArgNotNull("property", property);
293 		return switch (property) {
294 			case "allowReserved" -> setAllowReserved(toBoolean(value));
295 			case "contentType" -> setContentType(Utils.s(value));
296 			case "explode" -> setExplode(toBoolean(value));
297 			case "headers" -> setHeaders(mapBuilder(String.class, HeaderInfo.class).sparse().addAny(value).build());
298 			case "style" -> setStyle(Utils.s(value));
299 			default -> {
300 				super.set(property, value);
301 				yield this;
302 			}
303 		};
304 	}
