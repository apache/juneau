25  /**
26   * Describes a single HTTP header.
27   *
28   * <p>
29   * The Header Object follows the structure of the Parameter Object with the following changes: it does not have a 
30   * <c>name</c> field since the header name is specified in the key, and it does not have a <c>required</c> field 
31   * since headers are always optional in HTTP.
32   *
33   * <h5 class='section'>OpenAPI Specification:</h5>
34   * <p>
35   * The Header Object is composed of the following fields:
36   * <ul class='spaced-list'>
37   * 	<li><c>description</c> (string) - A brief description of the header (CommonMark syntax may be used)
38   * 	<li><c>required</c> (boolean) - Determines whether this header is mandatory (default is <jk>false</jk>)
39   * 	<li><c>deprecated</c> (boolean) - Specifies that a header is deprecated
40   * 	<li><c>allowEmptyValue</c> (boolean) - Sets the ability to pass empty-valued headers
41   * 	<li><c>style</c> (string) - Describes how the header value will be serialized
42   * 	<li><c>explode</c> (boolean) - When true, header values of type array or object generate separate headers for each value
43   * 	<li><c>allowReserved</c> (boolean) - Determines whether the header value should allow reserved characters
44   * 	<li><c>schema</c> ({@link SchemaInfo}) - The schema defining the type used for the header
45   * 	<li><c>example</c> (any) - Example of the header's potential value
46   * 	<li><c>examples</c> (map of {@link Example}) - Examples of the header's potential value
47   * </ul>
48   *
49   * <h5 class='section'>Example:</h5>
50   * <p class='bcode'>
51   * 	<jc>// Construct using SwaggerBuilder.</jc>
52   * 	HeaderInfo <jv>x</jv> = <jsm>headerInfo</jsm>(<js>"integer"</js>).description(<js>"The number of allowed requests in the current period"</js>);
53   *
54   * 	<jc>// Serialize using JsonSerializer.</jc>
55   * 	String <jv>json</jv> = Json.<jsm>from</jsm>(<jv>x</jv>);
56   *
57   * 	<jc>// Or just use toString() which does the same as above.</jc>
58   * 	String <jv>json</jv> = <jv>x</jv>.toString();
59   * </p>
60   * <p class='bcode'>
61   * 	<jc>// Output</jc>
62   * 	{
63   * 		<js>"description"</js>: <js>"The number of allowed requests in the current period"</js>,
64   * 		<js>"type"</js>: <js>"integer"</js>
65   * 	}
66   * </p>
67   *
68   * <h5 class='section'>See Also:</h5><ul>
69   * 	<li class='link'><a class="doclink" href="https://spec.openapis.org/oas/v3.0.0#header-object">OpenAPI Specification &gt; Header Object</a>
70   * 	<li class='link'><a class="doclink" href="https://swagger.io/docs/specification/describing-parameters/">OpenAPI Describing Parameters</a>
71   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauBeanOpenApi3">juneau-bean-openapi-v3</a>
72   * </ul>
73   */
