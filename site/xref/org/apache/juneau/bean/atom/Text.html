17  package org.apache.juneau.bean.atom;

19  import static org.apache.juneau.xml.annotation.XmlFormat.*;
20  
21  import org.apache.juneau.xml.annotation.*;

23  /**
24   * Represents human-readable text in an Atom document.
25   *
26   * <p>
27   * Text constructs are used throughout Atom for elements that contain human-readable text
28   * such as titles, summaries, rights statements, and subtitles. They support three content types:
29   *
30   * <ul class='spaced-list'>
31   * 	<li><c>"text"</c> - Plain text with no markup (default)
32   * 	<li><c>"html"</c> - HTML markup, entity-escaped
33   * 	<li><c>"xhtml"</c> - Well-formed XHTML in a div container
34   * </ul>
35   *
36   * <p>
37   * Text constructs are the base class for {@link Content} and are used directly for:
38   * <ul class='spaced-list'>
39   * 	<li><c>atom:title</c> - Entry and feed titles
40   * 	<li><c>atom:subtitle</c> - Feed subtitles
41   * 	<li><c>atom:summary</c> - Entry summaries
42   * 	<li><c>atom:rights</c> - Copyright and rights statements
43   * </ul>
44   *
45   * <h5 class='figure'>Schema</h5>
46   * <p class='bschema'>
47   * 	atomTextConstruct = atomPlainTextConstruct | atomXHTMLTextConstruct
48   *
49   * 	atomPlainTextConstruct =
50   * 		atomCommonAttributes,
51   * 		attribute type { "text" | "html" }?,
52   * 		text
53   *
54   * 	atomXHTMLTextConstruct =
55   * 		atomCommonAttributes,
56   * 		attribute type { "xhtml" },
57   * 		xhtmlDiv
58   *
59   * 	xhtmlDiv = element xhtml:div {
60   * 		(attribute * { text }
61   * 		| text
62   * 		| anyXHTML)*
63   * 	}
64   * </p>
65   *
66   * <h5 class='section'>Examples:</h5>
67   * <p class='bjava'>
68   * 	<jc>// Plain text</jc>
69   * 	Text <jv>t1</jv> = <jk>new</jk> Text(<js>"text"</js>)
70   * 		.setText(<js>"Plain text title"</js>);
71   *
72   * 	<jc>// HTML (entity-escaped)</jc>
73   * 	Text <jv>t2</jv> = <jk>new</jk> Text(<js>"html"</js>)
74   * 		.setText(<js>"Title with &lt;em&gt;emphasis&lt;/em&gt;"</js>);
75   *
76   * 	<jc>// XHTML</jc>
77   * 	Text <jv>t3</jv> = <jk>new</jk> Text(<js>"xhtml"</js>)
78   * 		.setText(<js>"&lt;div xmlns='http://www.w3.org/1999/xhtml'&gt;&lt;p&gt;XHTML title&lt;/p&gt;&lt;/div&gt;"</js>);
79   * </p>
80   *
81   * <h5 class='section'>Specification:</h5>
82   * <p>
83   * Represents an <c>atomTextConstruct</c> in the
84   * <a class="doclink" href="https://tools.ietf.org/html/rfc4287#section-3.1">RFC 4287 - Section 3.1</a> specification.
85   *
86   * <h5 class='section'>See Also:</h5><ul>
87   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauBeanAtom">juneau-bean-atom</a>
88   * 	<li class='extlink'><a class="doclink" href="https://tools.ietf.org/html/rfc4287">RFC 4287 - The Atom Syndication Format</a>
89   * </ul>
90   */
91  public class Text extends Common {
92  
93  	private String type;
94  	private String text;  // NOSONAR - Intentional naming.
95  
96  	/**
97  	 * Normal content.
98  	 *
99  	 * @param type The content type of this content.
100 	 */
101 	public Text(String type) {
102 		setType(type);
103 	}
104 
105 	/** Bean constructor. */
106 	public Text() {}
107 
108 	//-----------------------------------------------------------------------------------------------------------------
109 	// Bean properties
110 	//-----------------------------------------------------------------------------------------------------------------
111 
112 	/**
113 	 * Bean property getter:  <property>type</property>.
114 	 *
115 	 * <p>
116 	 * The content type of this content.
117 	 *
118 	 * @return The property value, or <jk>null</jk> if it is not set.
119 	 */
120 	@Xml(format=ATTR)
121 	public String getType() {
122 		return type;
123 	}
124 
125 	/**
126 	 * Bean property setter:  <property>type</property>.
127 	 *
128 	 * <p>
129 	 * The content type of this content.
130 	 *
131 	 * <p>
132 	 * Must be one of the following:
133 	 * <ul>
134 	 * 	<li><js>"text"</js>
135 	 * 	<li><js>"html"</js>
136 	 * 	<li><js>"xhtml"</js>
137 	 * 	<li><jk>null</jk> (defaults to <js>"text"</js>)
138 	 * </ul>
139 	 *
140 	 * @param value
141 	 * 	The new value for this property.
142 	 * 	<br>Can be <jk>null</jk> to unset the property.
143 	 * @return This object
144 	 */
145 	public Text setType(String value) {
146 		this.type = value;
147 		return this;
148 	}
149 
150 	/**
151 	 * Bean property getter:  <property>text</property>.
152 	 *
153 	 * <p>
154 	 * The content of this content.
155 	 *
156 	 * @return The property value, or <jk>null</jk> if it is not set.
157 	 */
158 	@Xml(format=XMLTEXT)
159 	public String getText() {
160 		return text;
161 	}
162 
163 	/**
164 	 * Bean property setter:  <property>text</property>.
165 	 *
166 	 * <p>
167 	 * The content of this content.
168 	 *
169 	 * @param value
170 	 * 	The new value for this property.
171 	 * 	<br>Can be <jk>null</jk> to unset the property.
172 	 * @return This object
173 	 */
174 	public Text setText(String value) {
175 		this.text = value;
176 		return this;
177 	}
178 
179 	//-----------------------------------------------------------------------------------------------------------------
180 	// Overridden setters (to simplify method chaining)
181 	//-----------------------------------------------------------------------------------------------------------------
182 
183 	@Override /* Overridden from Common */
184 	public Text setBase(Object value) {
185 		super.setBase(value);
186 		return this;
187 	}
188 
189 	@Override /* Overridden from Common */
190 	public Text setLang(String value) {
191 		super.setLang(value);
192 		return this;
193 	}
194 }
