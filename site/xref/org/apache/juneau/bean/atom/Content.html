17  package org.apache.juneau.bean.atom;
18  
19  import static org.apache.juneau.commons.utils.StringUtils.*;
20  import static org.apache.juneau.xml.annotation.XmlFormat.*;
21  
22  import java.net.*;
23  
24  import org.apache.juneau.xml.annotation.*;
25  
26  /**
27   * Represents the content of an Atom entry.
28   *
29   * <p>
30   * The content element contains or links to the complete content of an entry. It supports multiple
31   * content types and delivery methods:
32   *
33   * <ul class='spaced-list'>
34   * 	<li><b>Inline text content</b> - Plain text or HTML content embedded in the feed
35   * 	<li><b>Inline XHTML content</b> - Well-formed XHTML embedded in the feed
36   * 	<li><b>Inline other content</b> - Other media types (XML, base64-encoded binary, etc.)
37   * 	<li><b>Out-of-line content</b> - Link to external content via the <c>src</c> attribute
38   * </ul>
39   *
40   * <p>
41   * The <c>type</c> attribute indicates the media type of the content. Common values:
42   * <ul class='spaced-list'>
43   * 	<li><c>"text"</c> - Plain text (default)
44   * 	<li><c>"html"</c> - HTML, entity-escaped
45   * 	<li><c>"xhtml"</c> - XHTML wrapped in a div element
46   * 	<li>Other MIME types - For multimedia content or base64-encoded data
47   * </ul>
48   *
49   * <h5 class='figure'>Schema</h5>
50   * <p class='bschema'>
51   * 	atomContent = atomInlineTextContent
52   * 		| atomInlineXHTMLContent
53   * 		| atomInlineOtherContent
54   * 		| atomOutOfLineContent
55   *
56   * 	atomInlineTextContent =
57   * 		element atom:content {
58   * 			atomCommonAttributes,
59   * 			attribute type { "text" | "html" }?,
60   * 			(text)*
61   * 		}
62   *
63   * 	atomInlineXHTMLContent =
64   * 		element atom:content {
65   * 			atomCommonAttributes,
66   * 			attribute type { "xhtml" },
67   * 			xhtmlDiv
68   * 		}
69   *
70   * 	atomInlineOtherContent =
71   * 		element atom:content {
72   * 			atomCommonAttributes,
73   * 			attribute type { atomMediaType }?,
74   * 			(text|anyElement)*
75   * 	}
76   *
77   * 	atomOutOfLineContent =
78   * 		element atom:content {
79   * 			atomCommonAttributes,
80   * 			attribute type { atomMediaType }?,
81   * 			attribute src { atomUri },
82   * 			empty
83   * 	}
84   * </p>
85   *
86   * <h5 class='section'>Examples:</h5>
87   * <p class='bjava'>
88   * 	<jc>// Plain text content</jc>
89   * 	Content <jv>c1</jv> = <jk>new</jk> Content(<js>"text"</js>)
90   * 		.setText(<js>"This is plain text content"</js>);
91   *
92   * 	<jc>// HTML content</jc>
93   * 	Content <jv>c2</jv> = <jk>new</jk> Content(<js>"html"</js>)
94   * 		.setText(<js>"&lt;p&gt;This is &lt;strong&gt;HTML&lt;/strong&gt; content&lt;/p&gt;"</js>);
95   *
96   * 	<jc>// XHTML content</jc>
97   * 	Content <jv>c3</jv> = <jk>new</jk> Content(<js>"xhtml"</js>)
98   * 		.setText(<js>"&lt;div xmlns='http://www.w3.org/1999/xhtml'&gt;&lt;p&gt;XHTML content&lt;/p&gt;&lt;/div&gt;"</js>);
99   *
100  * 	<jc>// External content (out-of-line)</jc>
101  * 	Content <jv>c4</jv> = <jk>new</jk> Content()
102  * 		.setType(<js>"video/mp4"</js>)
103  * 		.setSrc(<js>"http://example.org/movie.mp4"</js>);
104  * </p>
105  *
106  * <h5 class='section'>Specification:</h5>
107  * <p>
108  * Represents an <c>atomContent</c> construct in the
109  * <a class="doclink" href="https://tools.ietf.org/html/rfc4287#section-4.1.3">RFC 4287 - Section 4.1.3</a> specification.
110  *
111  * <h5 class='section'>See Also:</h5><ul>
112  * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauBeanAtom">juneau-bean-atom</a>
113  * 	<li class='extlink'><a class="doclink" href="https://tools.ietf.org/html/rfc4287">RFC 4287 - The Atom Syndication Format</a>
114  * </ul>
115  */
116 public class Content extends Text {
117 
118 	private URI src;
119 
120 	/**
121 	 * Normal content.
122 	 */
123 	public Content() {}
124 
125 	/**
126 	 * Normal content.
127 	 *
128 	 * @param type The content type of this content.
129 	 */
130 	public Content(String type) {
131 		super(type);
132 	}
133 
134 	/**
135 	 * Bean property getter:  <property>src</property>.
136 	 *
137 	 * <p>
138 	 * Returns the URI of externally-hosted content (out-of-line content).
139 	 *
140 	 * <p>
141 	 * When <c>src</c> is present, the content is not embedded in the feed but is instead
142 	 * referenced by URI. This is useful for large media files or content hosted elsewhere.
143 	 *
144 	 * @return The property value, or <jk>null</jk> if it is not set.
145 	 */
146 	@Xml(format = ATTR)
147 	public URI getSrc() { return src; }
148 
149 	@Override /* Overridden from Common */
150 	public Content setBase(Object value) {
151 		super.setBase(value);
152 		return this;
153 	}
154 
155 	@Override /* Overridden from Common */
156 	public Content setLang(String value) {
157 		super.setLang(value);
158 		return this;
159 	}
160 
161 	/**
162 	 * Bean property setter:  <property>src</property>.
163 	 *
164 	 * <p>
165 	 * Sets the URI of externally-hosted content (out-of-line content).
166 	 *
167 	 * <p>
168 	 * The value can be of any of the following types: {@link URI}, {@link URL}, {@link String}.
169 	 * Strings must be valid URIs.
170 	 *
171 	 * <h5 class='section'>Example:</h5>
172 	 * <p class='bjava'>
173 	 * 	<jc>// Link to external video</jc>
174 	 * 	Content <jv>content</jv> = <jk>new</jk> Content()
175 	 * 		.setType(<js>"video/mp4"</js>)
176 	 * 		.setSrc(<js>"http://example.org/videos/intro.mp4"</js>);
177 	 * </p>
178 	 *
179 	 * @param value
180 	 * 	The new value for this property.
181 	 * 	<br>Can be <jk>null</jk> to unset the property.
182 	 * @return This object.
183 	 */
184 	public Content setSrc(Object value) {
185 		this.src = toURI(value);
186 		return this;
187 	}
188 
189 	@Override /* Overridden from Text */
190 	public Content setText(String value) {
191 		super.setText(value);
192 		return this;
193 	}
194 
195 	@Override /* Overridden from Text */
196 	public Content setType(String value) {
197 		super.setType(value);
198 		return this;
199 	}
200 }
