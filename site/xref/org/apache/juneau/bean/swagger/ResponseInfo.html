24  /**
25   * Describes a single response from an API Operation.
26   *
27   * <h5 class='section'>Example:</h5>
28   * <p class='bjava'>
29   * 	<jc>// Construct using SwaggerBuilder.</jc>
30   * 	ResponseInfo <jv>info</jv> = <jsm>responseInfo</jsm>(<js>"A complex object array response"</js>)
31   * 		.schema(
32   * 			<jsm>schemaInfo</jsm>
33   * 				.type(<js>"array"</js>)
34   * 				.items(
35   * 					<jsm>items</jsm>()
36   * 						.set(<js>"$ref"</js>, <js>"#/definitions/VeryComplexType"</js>)
37   * 				)
38   * 		);
39   *
40   * 	<jc>// Serialize using JsonSerializer.</jc>
41   * 	String <jv>json</jv> = JsonSerializer.<jsf>DEFAULT</jsf>.toString(<jv>info</jv>);
42   *
43   * 	<jc>// Or just use toString() which does the same as above.</jc>
44   * 	<jv>json</jv> = <jv>info</jv>.toString();
45   * </p>
46   * <p class='bjson'>
47   * 	<jc>// Output</jc>
48   * 	{
49   * 		<js>"description"</js>: <js>"A complex object array response"</js>,
50   * 		<js>"schema"</js>: {
51   * 			<js>"type"</js>: <js>"array"</js>,
52   * 			<js>"items"</js>: {
53   * 				<js>"$ref"</js>: <js>"#/definitions/VeryComplexType"</js>
54   * 			}
55   * 		}
56   * 	}
57   * </p>
58   *
59   * <h5 class='section'>See Also:</h5><ul>
60   * 	<li class='link'><a class="doclink" href="../../../../../index.html#jrs.Swagger">Overview &gt; juneau-rest-server &gt; Swagger</a>
61   * </ul>
62   */
63  @Bean(properties="description,schema,headers,examples,*")
64  @FluentSetters
65  public class ResponseInfo extends SwaggerElement {
66  
67  	private String description;
68  	private SchemaInfo schema;
69  	private Map<String,HeaderInfo> headers;
70  	private Map<String,Object> examples;

75  	public ResponseInfo() {}

77  	/**
78  	 * Copy constructor.
79  	 *
80  	 * @param copyFrom The object to copy.
81  	 */
82  	public ResponseInfo(ResponseInfo copyFrom) {
83  		super(copyFrom);
84  
85  		this.description = copyFrom.description;
86  		this.schema = copyFrom.schema == null ? null : copyFrom.schema.copy();
87  		this.examples = copyOf(copyFrom.examples);
88  		this.headers = copyOf(copyFrom.headers, HeaderInfo::copy);
89  	}

97  		return new ResponseInfo(this);
98  	}

110 			if (r.description != null)

119 		return this;
120 	}

282 	@Override /* SwaggerElement */
283 	public <T> T get(String property, Class<T> type) {
284 		if (property == null)
285 			return null;
286 		return switch (property) {
287 			case "description" -> toType(getDescription(), type);
288 			case "examples" -> toType(getExamples(), type);
289 			case "headers" -> toType(getHeaders(), type);
290 			case "schema" -> toType(getSchema(), type);
291 			default -> super.get(property, type);
292 		};
293 	}
294 
295 	@Override /* SwaggerElement */
296 	public ResponseInfo set(String property, Object value) {
297 		if (property == null)
298 			return this;
299 		return switch (property) {
300 			case "description" -> setDescription(Utils.s(value));
301 			case "examples" -> setExamples(mapBuilder(String.class,Object.class).sparse().addAny(value).build());
302 			case "headers" -> setHeaders(mapBuilder(String.class,HeaderInfo.class).sparse().addAny(value).build());
303 			case "schema" -> setSchema(toType(value, SchemaInfo.class));
304 			default -> {
305 				super.set(property, value);
306 				yield this;
307 			}
308 		};
309 	}

311 	@Override /* SwaggerElement */
312 	public Set<String> keySet() {
313 		var s = setBuilder(String.class)
314 			.addIf(description != null, "description")
315 			.addIf(examples != null, "examples")
316 			.addIf(headers != null, "headers")
317 			.addIf(schema != null, "schema")
318 			.build();
319 		return new MultiSet<>(s, super.keySet());
320 	}

322 	/**
323 	 * Resolves any <js>"$ref"</js> attributes in this element.
324 	 *
325 	 * @param swagger The swagger document containing the definitions.
326 	 * @param refStack Keeps track of previously-visited references so that we don't cause recursive loops.
327 	 * @param maxDepth
328 	 * 	The maximum depth to resolve references.
329 	 * 	<br>After that level is reached, <c>$ref</c> references will be left alone.
330 	 * 	<br>Useful if you have very complex models and you don't want your swagger page to be overly-complex.
331 	 * @return
332 	 * 	This object with references resolved.
333 	 * 	<br>May or may not be the same object.
334 	 */
335 	public ResponseInfo resolveRefs(Swagger swagger, Deque<String> refStack, int maxDepth) {
336 
337 		if (schema != null)
338 			schema = schema.resolveRefs(swagger, refStack, maxDepth);
339 
340 		if (headers != null)
341 			headers.entrySet().forEach(x -> x.setValue(x.getValue().resolveRefs(swagger, refStack, maxDepth)));
342 
343 		return this;
344 	}
