33  /**
34   * This is the root document object for the Swagger 2.0 API specification.
35   *
36   * <p>
37   * The Swagger Object is the root document that describes an entire API. It contains metadata about the API,
38   * available paths and operations, parameters, responses, security definitions, and other information. This is
39   * the Swagger 2.0 specification (predecessor to OpenAPI 3.0).
40   *
41   * <h5 class='section'>Swagger Specification:</h5>
42   * <p>
43   * The Swagger Object is composed of the following fields:
44   * <ul class='spaced-list'>
45   * 	<li><c>swagger</c> (string, REQUIRED) - The Swagger Specification version (must be <js>"2.0"</js>)
46   * 	<li><c>info</c> ({@link Info}, REQUIRED) - Provides metadata about the API
47   * 	<li><c>host</c> (string) - The host (name or IP) serving the API
48   * 	<li><c>basePath</c> (string) - The base path on which the API is served (relative to host)
49   * 	<li><c>schemes</c> (array of string) - The transfer protocols of the API (e.g., <js>"http"</js>, <js>"https"</js>)
50   * 	<li><c>consumes</c> (array of string) - A list of MIME types the APIs can consume
51   * 	<li><c>produces</c> (array of string) - A list of MIME types the APIs can produce
52   * 	<li><c>paths</c> (map of {@link OperationMap}, REQUIRED) - The available paths and operations for the API
53   * 	<li><c>definitions</c> (map of {@link SchemaInfo}) - Schema definitions that can be referenced
54   * 	<li><c>parameters</c> (map of {@link ParameterInfo}) - Parameters definitions that can be referenced
55   * 	<li><c>responses</c> (map of {@link ResponseInfo}) - Response definitions that can be referenced
56   * 	<li><c>securityDefinitions</c> (map of {@link SecurityScheme}) - Security scheme definitions
57   * 	<li><c>security</c> (array of map) - Security requirements applied to all operations
58   * 	<li><c>tags</c> (array of {@link Tag}) - A list of tags used by the specification with additional metadata
59   * 	<li><c>externalDocs</c> ({@link ExternalDocumentation}) - Additional external documentation
60   * </ul>
61   *
62   * <h5 class='section'>Example:</h5>
63   * <p class='bjava'>
64   * 	<jc>// Create a Swagger document</jc>
65   * 	Swagger <jv>doc</jv> = <jk>new</jk> Swagger()
66   * 		.setSwagger(<js>"2.0"</js>)
67   * 		.setInfo(
68   * 			<jk>new</jk> Info()
69   * 				.setTitle(<js>"Pet Store API"</js>)
70   * 				.setVersion(<js>"1.0.0"</js>)
71   * 		)
72   * 		.setHost(<js>"petstore.swagger.io"</js>)
73   * 		.setBasePath(<js>"/v2"</js>)
74   * 		.setSchemes(<js>"https"</js>)
75   * 		.addPath(<js>"/pets"</js>, <js>"get"</js>,
76   * 			<jk>new</jk> Operation()
77   * 				.setSummary(<js>"List all pets"</js>)
78   * 				.addResponse(<js>"200"</js>, <jk>new</jk> ResponseInfo(<js>"Success"</js>))
79   * 		);
80   * </p>
81   *
82   * <h5 class='section'>See Also:</h5><ul>
83   * 	<li class='link'><a class="doclink" href="https://swagger.io/specification/v2/">Swagger 2.0 Specification</a>
84   * 	<li class='link'><a class="doclink" href="https://swagger.io/docs/specification/2-0/basic-structure/">Swagger Basic Structure</a>
85   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauBeanSwagger2">juneau-bean-swagger-v2</a>
86   * </ul>
87   */

120 		super(copyFrom);
121 
122 		this.basePath = copyFrom.basePath;
123 		this.consumes = CollectionUtils.copyOf(copyFrom.consumes);
124 		this.externalDocs = copyFrom.externalDocs == null ? null : copyFrom.externalDocs.copy();
125 		this.host = copyFrom.host;
126 		this.info = copyFrom.info == null ? null : copyFrom.info.copy();
127 		this.produces = CollectionUtils.copyOf(copyFrom.produces);
128 		this.schemes = CollectionUtils.copyOf(copyFrom.schemes);
129 		this.swagger = copyFrom.swagger;
130 
131 		// TODO - Definitions are not deep copied, so they should not contain references.
132 		this.definitions = CollectionUtils.copyOf(copyFrom.definitions, JsonMap::new);
133 
134 		this.paths = CollectionUtils.copyOf(copyFrom.paths, v -> {
135 			var m = new OperationMap();
136 			v.forEach((k2, v2) -> m.put(k2, v2.copy()));
137 			return m;
138 		});
139 
140 		this.parameters = CollectionUtils.copyOf(copyFrom.parameters, ParameterInfo::copy);
141 		this.responses = CollectionUtils.copyOf(copyFrom.responses, ResponseInfo::copy);
142 		this.securityDefinitions = CollectionUtils.copyOf(copyFrom.securityDefinitions, SecurityScheme::copy);
143 
144 		this.security = CollectionUtils.copyOf(copyFrom.security, x -> {
145 			Map<String,List<String>> m2 = CollectionUtils.map();
146 			x.forEach((k, v) -> m2.put(k, CollectionUtils.copyOf(v)));
147 			return m2;
148 		});
149 
150 		this.tags = CollectionUtils.copyOf(copyFrom.tags, x -> x.copy());
151 	}

449 	/**
450 	 * Resolves a <js>"$ref"</js> tags to nodes in this swagger document.
451 	 *
452 	 * @param <T> The class to convert the reference to.
453 	 * @param ref The ref tag value.  Must not be <jk>null</jk> or blank.
454 	 * @param c The class to convert the reference to.  Must not be <jk>null</jk>.
455 	 * @return The referenced node, or <jk>null</jk> if not found.
456 	 */
457 	public <T> T findRef(String ref, Class<T> c) {
458 		assertArgNotNullOrBlank("ref", ref);
459 		assertArgNotNull("c", c);
460 		if (! ref.startsWith("#/"))
461 			throw new BasicRuntimeException("Unsupported reference:  ''{0}''", ref);
462 		try {
463 			return new ObjectRest(this).get(ref.substring(1), c);
464 		} catch (Exception e) {
465 			throw new BeanRuntimeException(e, c, "Reference ''{0}'' could not be converted to type ''{1}''.", ref, ClassUtils.className(c));
466 		}
467 	}

470 	public <T> T get(String property, Class<T> type) {
471 		assertArgNotNull("property", property);
472 		return switch (property) {
473 			case "basePath" -> toType(getBasePath(), type);
474 			case "consumes" -> toType(getConsumes(), type);
475 			case "definitions" -> toType(getDefinitions(), type);
476 			case "externalDocs" -> toType(getExternalDocs(), type);
477 			case "host" -> toType(getHost(), type);
478 			case "info" -> toType(getInfo(), type);
479 			case "parameters" -> toType(getParameters(), type);
480 			case "paths" -> toType(getPaths(), type);
481 			case "produces" -> toType(getProduces(), type);
482 			case "responses" -> toType(getResponses(), type);
483 			case "schemes" -> toType(getSchemes(), type);
484 			case "security" -> toType(getSecurity(), type);
485 			case "securityDefinitions" -> toType(getSecurityDefinitions(), type);
486 			case "swagger" -> toType(getSwagger(), type);
487 			case "tags" -> toType(getTags(), type);
488 			default -> super.get(property, type);
489 		};
490 	}
