19  
20  import org.apache.juneau.common.internal.*;

22  /**
23   * Map meant for method-name/operation mappings.
24   *
25   * <p>
26   * Forces entries to be sorted in the following order:
27   * <ul>
28   * 	<li><c>GET</c>
29   * 	<li><c>PUT</c>
30   * 	<li><c>POST</c>
31   * 	<li><c>DELETE</c>
32   * 	<li><c>OPTIONS</c>
33   * 	<li><c>HEAD</c>
34   * 	<li><c>PATCH</c>
35   * 	<li>Everything else.
36   * </ul>
37   *
38   * <h5 class='section'>See Also:</h5><ul>
39   * 	<li class='link'><a class="doclink" href="../../../../../index.html#jrs.Swagger">Overview &gt; juneau-rest-server &gt; Swagger</a>
40   * </ul>
41   *
42   * @serial exclude
43   */
44  public class OperationMap extends TreeMap<String,Operation> {
45  	private static final long serialVersionUID = 1L;
46  
47  	private static final Comparator<String> OP_SORTER = new Comparator<>() {
48  		private final Map<String,String> methods = mapBuilder(String.class,String.class)
49  			.add("get","0").add("put","1").add("post","2").add("delete","3").add("options","4").add("head","5").add("patch","6")
50  			.build();
51  
52  		@Override
53  		public int compare(String o1, String o2) {
54  			var s1 = methods.get(o1);
55  			var s2 = methods.get(o2);
56  			if (s1 == null)
57  				s1 = o1;
58  			if (s2 == null)
59  				s2 = o2;
60  			return StringUtils.compare(s1, s2);
61  		}
62  	};
63  
64  	/**
65  	 * Constructor.
66  	 */
67  	public OperationMap() {
68  		super(OP_SORTER);
69  	}
70  
71  	/**
72  	 * Fluent-style put method.
73  	 *
74  	 * @param httpMethodName The HTTP method name.
75  	 * @param operation The operation.
76  	 * @return This object.
77  	 */
78  	public OperationMap append(String httpMethodName, Operation operation) {
79  		put(emptyIfNull(httpMethodName).toLowerCase(), operation);
80  		return this;
81  	}
82  }
