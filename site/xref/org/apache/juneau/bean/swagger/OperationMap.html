17  package org.apache.juneau.bean.swagger;
18  
19  import static org.apache.juneau.common.utils.Utils.*;
20  import static org.apache.juneau.internal.CollectionUtils.*;
21  
22  import java.util.*;
23  
24  import org.apache.juneau.common.utils.*;
25  
26  /**
27   * Map meant for method-name/operation mappings.
28   *
29   * <p>
30   * The OperationMap is a specialized TreeMap that represents the operations available on a single path in Swagger 2.0. 
31   * It forces entries to be sorted in a specific order to ensure consistent output. This map is used within PathItem 
32   * objects to define the HTTP methods and their corresponding operations.
33   *
34   * <h5 class='section'>Swagger Specification:</h5>
35   * <p>
36   * The OperationMap represents the operations field in a Path Item Object, where each key is an HTTP method and each 
37   * value is an Operation object. The supported HTTP methods are:
38   * <ul class='spaced-list'>
39   * 	<li><c>get</c> ({@link Operation}) - A definition of a GET operation
40   * 	<li><c>put</c> ({@link Operation}) - A definition of a PUT operation
41   * 	<li><c>post</c> ({@link Operation}) - A definition of a POST operation
42   * 	<li><c>delete</c> ({@link Operation}) - A definition of a DELETE operation
43   * 	<li><c>options</c> ({@link Operation}) - A definition of an OPTIONS operation
44   * 	<li><c>head</c> ({@link Operation}) - A definition of a HEAD operation
45   * 	<li><c>patch</c> ({@link Operation}) - A definition of a PATCH operation
46   * </ul>
47   *
48   * <p>
49   * Forces entries to be sorted in the following order:
50   * <ul>
51   * 	<li><c>GET</c>
52   * 	<li><c>PUT</c>
53   * 	<li><c>POST</c>
54   * 	<li><c>DELETE</c>
55   * 	<li><c>OPTIONS</c>
56   * 	<li><c>HEAD</c>
57   * 	<li><c>PATCH</c>
58   * 	<li>Everything else.
59   * </ul>
60   *
61   * <h5 class='section'>Example:</h5>
62   * <p class='bjava'>
63   * 	<jc>// Construct using SwaggerBuilder.</jc>
64   * 	OperationMap <jv>operations</jv> = <jsm>operationMap</jsm>()
65   * 		.append(<js>"get"</js>, <jsm>operation</jsm>().setSummary(<js>"Get users"</js>))
66   * 		.append(<js>"post"</jsm>, <jsm>operation</jsm>().setSummary(<js>"Create user"</js>));
67   *
68   * 	<jc>// Serialize using JsonSerializer.</jc>
69   * 	String <jv>json</jv> = Json.<jsm>from</jsm>(<jv>operations</jv>);
70   *
71   * 	<jc>// Or just use toString() which does the same as above.</jc>
72   * 	<jv>json</jv> = <jv>operations</jv>.toString();
73   * </p>
74   * <p class='bjson'>
75   * 	<jc>// Output</jc>
76   * 	{
77   * 		<js>"get"</js>: { <js>"summary"</js>: <js>"Get users"</js> },
78   * 		<js>"post"</js>: { <js>"summary"</js>: <js>"Create user"</js> }
79   * 	}
80   * </p>
81   *
82   * <h5 class='section'>See Also:</h5><ul>
83   * 	<li class='link'><a class="doclink" href="https://swagger.io/specification/v2/#path-item-object">Swagger 2.0 Specification &gt; Path Item Object</a>
84   * 	<li class='link'><a class="doclink" href="https://swagger.io/docs/specification/2-0/paths-and-operations/">Swagger Paths and Operations</a>
85   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauBeanSwagger2">juneau-bean-swagger-v2</a>
86   * </ul>
87   *
88   * @serial exclude
89   */
90  public class OperationMap extends TreeMap<String,Operation> {
91  	private static final long serialVersionUID = 1L;
92  
93  	private static final Comparator<String> OP_SORTER = new Comparator<>() {
94  		private final Map<String,String> methods = mapBuilder(String.class,String.class)
95  			.add("get","0").add("put","1").add("post","2").add("delete","3").add("options","4").add("head","5").add("patch","6")
96  			.build();
97  
98  		@Override
99  		public int compare(String o1, String o2) {
100 			// Since keys are now stored in lowercase, we need to normalize them for comparison
101 			var s1 = methods.get(emptyIfNull(o1).toLowerCase());
102 			var s2 = methods.get(emptyIfNull(o2).toLowerCase());
103 			if (s1 == null)
104 				s1 = emptyIfNull(o1).toLowerCase();
105 			if (s2 == null)
106 				s2 = emptyIfNull(o2).toLowerCase();
107 			return StringUtils.compare(s1, s2);
108 		}
109 	};
110 
111 	/**
112 	 * Constructor.
113 	 */
114 	public OperationMap() {
115 		super(OP_SORTER);
116 	}
117 
118 	/**
119 	 * Override put to normalize keys to lowercase.
120 	 *
121 	 * @param key The key.
122 	 * @param value The value.
123 	 * @return The previous value associated with key, or null if there was no mapping for key.
124 	 */
125 	@Override
126 	public Operation put(String key, Operation value) {
127 		return super.put(emptyIfNull(key).toLowerCase(), value);
128 	}
129 
130 	/**
131 	 * Fluent-style put method.
132 	 *
133 	 * @param httpMethodName The HTTP method name.
134 	 * @param operation The operation.
135 	 * @return This object.
136 	 */
137 	public OperationMap append(String httpMethodName, Operation operation) {
138 		put(httpMethodName, operation);
139 		return this;
140 	}
141 }
