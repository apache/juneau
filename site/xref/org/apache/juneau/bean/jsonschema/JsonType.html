17  package org.apache.juneau.bean.jsonschema;
18  
19  /**
20   * Represents possible JSON types in the JSON-Schema core specification.
21   *
22   * <p>
23   * Implements custom <c>toString()</c> and <c>fromString(String)</c> methods that override the default
24   * serialization/parsing behavior of <c>Enum</c> types so that they are represented in lowercase form (as per the
25   * specification).
26   *
27   * <h5 class='section'>Example:</h5>
28   * <p class='bjava'>
29   * 	<jc>// Produces 'number', not 'NUMBER'.</jc>
30   * 	String <jv>json</jv> = Json.<jsm>from</jsm>(JsonType.<jsf>NUMBER</jsf>);
31   * </p>
32   */
33  public enum JsonType {
34  
35  	/** JSON type {@code array}. */
36  	ARRAY("array"),
37  
38  	/** JSON type {@code boolean}. */
39  	BOOLEAN("boolean"),
40  
41  	/** JSON type {@code integer}. */
42  	INTEGER("integer"),
43  
44  	/** JSON type {@code null}. */
45  	NULL("null"),
46  
47  	/** JSON type {@code number}. */
48  	NUMBER("number"),
49  
50  	/** JSON type {@code object}. */
51  	OBJECT("object"),
52  
53  	/** JSON type {@code string}. */
54  	STRING("string"),
55  
56  	/** JSON type {@code any}. */
57  	ANY("any");
58  
59  	private final String value;	// The serialized format of the enum.
60  
61  	JsonType(String value) {
62  		this.value = value;
63  	}
64  
65  	/**
66  	 * Returns the lowercase form of this enum that's compatible with the JSON-Schema specification.
67  	 */
68  	@Override /* Overridden from Object */
69  	public String toString() {
70  		return value;
71  	}
72  
73  	/**
74  	 * Converts the specified lowercase form of the enum back into an <c>Enum</c>.
75  	 *
76  	 * @param value The lowercase form of the enum (e.g. <js>"array"</js>).
77  	 * @return The matching <c>Enum</c>, or <jk>null</jk> if no match found.
78  	 */
79  	public static JsonType fromString(String value) {
80  		if (value == null || value.length() < 4)
81  			return null;
82  		var c = value.charAt(0);
83  		if (c == 'a') {
84  			if (value.equals("array"))
85  				return ARRAY;
86  			if (value.equals("any"))
87  				return ANY;
88  		}
89  		if (c == 'b' && value.equals("boolean"))
90  			return BOOLEAN;
91  		if (c == 'i' && value.equals("integer"))
92  			return INTEGER;
93  		if (c == 'n') {
94  			c = value.charAt(2);
95  			if (c == 'l' && value.equals("null"))
96  				return NULL;
97  			if (c == 'm' && value.equals("number"))
98  				return NUMBER;
99  			return null;
100 		}
101 		if (c == 'o' && value.equals("object"))
102 			return OBJECT;
103 		if (c == 's' && value.equals("string"))
104 			return STRING;
105 		return null;
106 	}
107 }
