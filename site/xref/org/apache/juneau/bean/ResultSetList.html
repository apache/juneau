15  import static org.apache.juneau.common.internal.IOUtils.*;
16  
17  import java.sql.*;
18  import java.util.*;
19  
20  import org.apache.juneau.internal.*;
21  
22  /**
23   * Transforms an SQL {@link ResultSet ResultSet} into a list of maps.
24   * <p>
25   * Loads the entire result set into an in-memory data structure, and then closes the result set object.
26   *
27   * <h5 class='section'>See Also:</h5><ul>
28  
29   * </ul>
30   *
31   * @serial exclude
32   */
33  public class ResultSetList extends LinkedList<Map<String,Object>> {
34  
35  	private static final long serialVersionUID = 1L;
36  
37  	/**
38  	 * Constructor.
39  	 *
40  	 * @param rs The result set to load into this DTO.
41  	 * @param pos The start position (zero-indexed).
42  	 * @param limit The maximum number of rows to retrieve.
43  	 * @param includeRowNums Make the first column be the row number.
44  	 * @throws SQLException Database error.
45  	 */
46  	public ResultSetList(ResultSet rs, int pos, int limit, boolean includeRowNums) throws SQLException {
47  		try {
48  			var rowNum = pos;
49  
50  			// Get the column names.
51  			var rsmd = rs.getMetaData();
52  			var offset = (includeRowNums ? 1 : 0);
53  			var cc = rsmd.getColumnCount();
54  			var columns = new String[cc + offset];
55  			if (includeRowNums)
56  				columns[0] = "ROW";
57  			var colTypes = new int[cc];
58  
59  			for (var i = 0; i < cc; i++) {
60  				columns[i+offset] = rsmd.getColumnName(i+1);
61  				colTypes[i] = rsmd.getColumnType(i+1);
62  			}
63  
64  			while (--pos > 0 && rs.next()) { /* Skip to the specified position. */ }
65  
66  			// Get the rows.
67  			while (limit-- > 0 && rs.next()) {
68  				var row = new Object[cc + offset];
69  				if (includeRowNums)
70  					row[0] = rowNum++;
71  				for (var i = 0; i < cc; i++) {
72  					var o = readEntry(rs, i+1, colTypes[i]);
73  					row[i+offset] = o;
74  				}
75  				add(new SimpleMap<>(columns, row));
76  			}
77  		} finally {
78  			rs.close();
79  		}
80  	}
81  
82  	/**
83  	 * Reads the specified column from the current row in the result set.
84  	 *
85  	 * <p>
86  	 * Subclasses can override this method to handle specific data types in special ways.
87  	 *
88  	 * @param rs The result set to read from.
89  	 * @param col The column number (indexed by 1).
90  	 * @param dataType The {@link Types type} of the entry.
91  	 * @return The entry as an Object.
92  	 */
93  	static Object readEntry(ResultSet rs, int col, int dataType) {
94  		try {
95  			return switch (dataType) {
96  				case Types.BLOB -> {
97  					var b = rs.getBlob(col);
98  					yield "blob[" + b.length() + "]";
99  				}
100 				case Types.CLOB -> {
101 					var c = rs.getClob(col);
102 					yield "clob[" + c.length() + "]";
103 				}
104 				case Types.LONGVARBINARY -> "longvarbinary[" + count(rs.getBinaryStream(col)) + "]";
105 				case Types.LONGVARCHAR -> "longvarchar[" + count(rs.getAsciiStream(col)) + "]";
106 				case Types.LONGNVARCHAR -> "longnvarchar[" + count(rs.getCharacterStream(col)) + "]";
107 				case Types.TIMESTAMP -> rs.getTimestamp(col); // Oracle returns com.oracle.TIMESTAMP objects from getObject()
108 				default -> rs.getObject(col); };
109 		} catch (Exception e) {
110 			return e.getLocalizedMessage();
111 		}
112 	}
113 }
