29  /**
30   * Used to swap out non-serializable objects with serializable replacements during serialization, and vis-versa during
31   * parsing.
32   *
33   * <h5 class='topic'>Description</h5>
34   *
35   * <p>
36   * <c>ObjectSwaps</c> are used to extend the functionality of the serializers and parsers to be able to handle
37   * objects that aren't automatically handled by the serializers or parsers.
38   * <br>For example, JSON does not have a standard representation for rendering dates.
39   * By defining a special {@code Date} swap and associating it with a serializer and parser, you can convert a
40   * {@code Date} object to a {@code String} during serialization, and convert that {@code String} object back into a
41   * {@code Date} object during parsing.
42   *
43   * <p>
44   * Swaps MUST declare a public no-arg constructor so that the bean context can instantiate them.
45   *
46   * <p>
47   * <c>ObjectSwaps</c> are associated with serializers and parsers through the following:
48   * <ul class='javatree'>
49   * 	<li class='ja'>{@link Swap @Swap}
50   * 	<li class='jm'>{@link org.apache.juneau.BeanContext.Builder#swaps(Class...)}
51   * </ul>
52   *
53   * <p>
54   * <c>ObjectSwaps</c> have two parameters:
55   * <ol>
56   * 	<li>{@code <T>} - The normal representation of an object.
57   * 	<li>{@code <S>} - The swapped representation of an object.
58   * </ol>
59   * <br>{@link Serializer Serializers} use swaps to convert objects of type T into objects of type S, and on calls to
60   * {@link BeanMap#get(Object)}.
61   * <br>{@link Parser Parsers} use swaps to convert objects of type S into objects of type T, and on calls to
62   * {@link BeanMap#put(String,Object)}.
63   *
64   * <h5 class='topic'>Swap Class Type {@code <S>}</h5>
65   *
66   * <p>
67   * For normal serialization, the swapped object representation of an object must be an object type that the serializers can natively convert to
68   * JSON (or language-specific equivalent).
69   * <br>The list of valid transformed types are as follows...
70   * <ul>
71   * 	<li>
72   * 		{@link String}
73   * 	<li>
74   * 		{@link Number}
75   * 	<li>
76   * 		{@link Boolean}
77   * 	<li>
78   * 		{@link Collection} containing anything on this list.
79   * 	<li>
80   * 		{@link Map} containing anything on this list.
81   * 	<li>
82   * 		A java bean with properties of anything on this list.
83   * 	<li>
84   * 		An array of anything on this list.
85   * </ul>
86   *
87   * <p>
88   * For OpenAPI serialization, the valid swapped types also include <code><jk>byte</jk>[]</code> and <c>Calendar</c>.
89   *
90   * <h5 class='topic'>Normal Class Type {@code <T>}</h5>
91   *
92   * <p>
93   * The normal object representation of an object.
94   *
95   * <h5 class='section'>See Also:</h5><ul>
96   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SwapBasics">Swap Basics</a>
97  
98   * </ul>
99   *
100  * @param <T> The normal form of the class.
101  * @param <S> The swapped form of the class.
102  */
103 @SuppressWarnings({ "unchecked", "rawtypes" })
104 public abstract class ObjectSwap<T,S> {
105 
106 	/**
107 	 * Represents a non-existent object swap.
108 	 */
109 	public static final ObjectSwap NULL = new ObjectSwap((Class)null, (Class)null) {};
110 
111 	private final Class<T> normalClass;
112 	private final Class<?> swapClass;
113 	private final ClassInfo normalClassInfo, swapClassInfo;
114 	private ClassMeta<?> swapClassMeta;
115 
116 	// Unfortunately these cannot be made final because we want to allow for ObjectSwaps with no-arg constructors
117 	// which simplifies declarations.
118 	private MediaType[] forMediaTypes;
119 	private String template;
120 
121 	/**
122 	 * Constructor.
123 	 */
124 	protected ObjectSwap() {
125 		var ci = ClassInfo.of(this.getClass());
126 		normalClass = (Class<T>)ci.getParameterType(0, ObjectSwap.class);
127 		swapClass = ci.getParameterType(1, ObjectSwap.class);
128 		normalClassInfo = ClassInfo.of(normalClass);
129 		swapClassInfo = ClassInfo.of(swapClass);
130 		forMediaTypes = forMediaTypes();
131 		template = withTemplate();
132 	}
133 
134 	/**
135 	 * Constructor for when the normal and transformed classes are already known.
136 	 *
137 	 * @param normalClass The normal class (cannot be serialized).
138 	 * @param swapClass The transformed class (serializable).
139 	 */
140 	protected ObjectSwap(Class<T> normalClass, Class<?> swapClass) {
141 		this.normalClass = normalClass;
142 		this.swapClass = swapClass;
143 		normalClassInfo = opt(normalClass).map(x -> ClassInfo.of(x)).orElse(null);
144 		swapClassInfo = opt(swapClass).map(x -> ClassInfo.of(x)).orElse(null);
145 		this.forMediaTypes = forMediaTypes();
146 		this.template = withTemplate();
147 	}
148 
149 	/**
150 	 * Returns the media types that this swap is applicable to.
151 	 *
152 	 * <p>
153 	 * This method can be overridden to programmatically specify what media types it applies to.
154 	 *
155 	 * <p>
156 	 * This method is the programmatic equivalent to the {@link Swap#mediaTypes() @Swap(mediaTypes)} annotation.
157 	 *
158 	 * <h5 class='section'>See Also:</h5><ul>
159 	 * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/PerMediaTypeSwaps">Per-media-type Swaps</a>
160 	 * </ul>
161 	 * @return The media types that this swap is applicable to, or <jk>null</jk> if it's applicable for all media types.
162 	 */
163 	public MediaType[] forMediaTypes() {
164 		return null;
165 	}
166 
167 	/**
168 	 * Sets the media types that this swap is associated with.
169 	 *
170 	 * <h5 class='section'>See Also:</h5><ul>
171 	 * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/PerMediaTypeSwaps">Per-media-type Swaps</a>
172 	 * </ul>
173 	 *
174 	 * @param mediaTypes The media types that this swap is associated with.
175 	 * @return This object.
176 	 */
177 	public ObjectSwap<T,?> forMediaTypes(MediaType[] mediaTypes) {
178 		this.forMediaTypes = mediaTypes;
179 		return this;
180 	}

182 	/**
183 	 * Returns the T class, the normalized form of the class.
184 	 *
185 	 * @return The normal form of this class.
186 	 */
187 	public ClassInfo getNormalClass() { return normalClassInfo; }
188 
189 	/**
190 	 * Returns the G class, the generalized form of the class.
191 	 *
192 	 * <p>
193 	 * Subclasses must override this method if the generalized class is {@code Object}, meaning it can produce multiple
194 	 * generalized forms.
195 	 *
196 	 * @return The transformed form of this class.
197 	 */
198 	public ClassInfo getSwapClass() { return swapClassInfo; }
199 
200 	/**
201 	 * Returns the {@link ClassMeta} of the transformed class type.
202 	 *
203 	 * <p>
204 	 * This value is cached for quick lookup.
205 	 *
206 	 * @param session
207 	 * 	The bean context to use to get the class meta.
208 	 * 	This is always going to be the same bean context that created this swap.
209 	 * @return The {@link ClassMeta} of the transformed class type.
210 	 */
211 	public ClassMeta<?> getSwapClassMeta(BeanSession session) {
212 		if (swapClassMeta == null)
213 			swapClassMeta = session.getClassMeta(swapClass);
214 		return swapClassMeta;
215 	}
216 
217 	/**
218 	 * Checks if the specified object is an instance of the normal class defined on this swap.
219 	 *
220 	 * @param o The object to check.
221 	 * @return
222 	 * 	<jk>true</jk> if the specified object is a subclass of the normal class defined on this transform.
223 	 * 	<jk>null</jk> always return <jk>false</jk>.
224 	 */
225 	public boolean isNormalObject(Object o) {
226 		if (o == null)
227 			return false;
228 		return normalClassInfo.isParentOf(o.getClass());
229 	}
230 
231 	/**
232 	 * Checks if the specified object is an instance of the swap class defined on this swap.
233 	 *
234 	 * @param o The object to check.
235 	 * @return
236 	 * 	<jk>true</jk> if the specified object is a subclass of the transformed class defined on this transform.
237 	 * 	<jk>null</jk> always return <jk>false</jk>.
238 	 */
239 	public boolean isSwappedObject(Object o) {
240 		if (o == null)
241 			return false;
242 		return swapClassInfo.isParentOf(o.getClass());
243 	}
244 
245 	/**
246 	 * Returns a number indicating how well this swap matches the specified session.
247 	 *
248 	 * <p>
249 	 * Uses the {@link MediaType#match(MediaType, boolean)} method algorithm to produce a number whereby a
250 	 * larger value indicates a "better match".
251 	 * The idea being that if multiple swaps are associated with a given object, we want to pick the best one.
252 	 *
253 	 * <p>
254 	 * For example, if the session media type is <js>"text/json"</js>, then the match values are shown below:
255 	 *
256 	 * <ul>
257 	 * 	<li><js>"text/json"</js> = <c>100,000</c>
258 	 * 	<li><js>"&#42;/json"</js> = <c>5,100</c>
259 	 * 	<li><js>"&#42;/&#42;"</js> = <c>5,000</c>
260 	 * 	<li>No media types specified on swap = <c>1</c>
261 	 * 	<li><js>"text/xml"</js> = <c>0</c>
262 	 * </ul>
263 	 *
264 	 * @param session The bean session.
265 	 * @return Zero if swap doesn't match the session, or a positive number if it does.
266 	 */
267 	public int match(BeanSession session) {
268 		if (forMediaTypes == null)
269 			return 1;
270 		int i = 0;
271 		var mt = session.getMediaType();
272 		if (mt == null)
273 			return 0;
274 		if (nn(forMediaTypes))
275 			for (var mt2 : forMediaTypes)
276 				i = Math.max(i, mt2.match(mt, false));
277 		return i;
278 	}

280 	/**
281 	 * If this transform is to be used to serialize non-serializable objects, it must implement this method.
282 	 *
283 	 * <p>
284 	 * The object must be converted into one of the following serializable types:
285 	 * <ul class='spaced-list'>
286 	 * 	<li>
287 	 * 		{@link String}
288 	 * 	<li>
289 	 * 		{@link Number}
290 	 * 	<li>
291 	 * 		{@link Boolean}
292 	 * 	<li>
293 	 * 		{@link Collection} containing anything on this list.
294 	 * 	<li>
295 	 * 		{@link Map} containing anything on this list.
296 	 * 	<li>
297 	 * 		A java bean with properties of anything on this list.
298 	 * 	<li>
299 	 * 		An array of anything on this list.
300 	 * </ul>
301 	 *
302 	 * @param session
303 	 * 	The bean session to use to get the class meta.
304 	 * 	This is always going to be the same bean context that created this swap.
305 	 * @param o The object to be transformed.
306 	 * @return The transformed object.
307 	 * @throws Exception If a problem occurred trying to convert the output.
308 	 */
309 	public S swap(BeanSession session, T o) throws Exception {
310 		return swap(session, o, template);
311 	}

326 		throw new SerializeException("Swap method not implemented on ObjectSwap ''{0}''", cn(this));

329 	@Override /* Overridden from Object */
330 	public String toString() {
331 		return getClass().getSimpleName() + '<' + getNormalClass().getNameSimple() + "," + getSwapClass().getNameSimple() + '>';
332 	}
