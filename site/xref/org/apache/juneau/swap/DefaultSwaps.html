27  /**
28   * Maintain the list of default swaps used by all serializers and parsers.
29   *
30   * <h5 class='section'>See Also:</h5><ul>
31   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SwapBasics">Swaps</a>
32   * </ul>
33   */
34  public class DefaultSwaps {
35  
36  	private static final Map<Class<?>,ObjectSwap<?,?>> SWAPS = new ConcurrentHashMap<>();
37  	static {
38  		SWAPS.put(Enumeration.class, new EnumerationSwap());
39  		SWAPS.put(Iterator.class, new IteratorSwap());
40  		SWAPS.put(Locale.class, new LocaleSwap());
41  		SWAPS.put(Class.class, new ClassSwap());
42  		SWAPS.put(Calendar.class, new TemporalCalendarSwap.IsoOffsetDateTime());
43  		SWAPS.put(Date.class, new TemporalDateSwap.IsoLocalDateTime());
44  		SWAPS.put(Instant.class, new TemporalSwap.IsoInstant());
45  		SWAPS.put(ZonedDateTime.class, new TemporalSwap.IsoOffsetDateTime());
46  		SWAPS.put(LocalDate.class, new TemporalSwap.IsoLocalDate());
47  		SWAPS.put(LocalDateTime.class, new TemporalSwap.IsoLocalDateTime());
48  		SWAPS.put(LocalTime.class, new TemporalSwap.IsoLocalTime());
49  		SWAPS.put(OffsetDateTime.class, new TemporalSwap.IsoOffsetDateTime());
50  		SWAPS.put(OffsetTime.class, new TemporalSwap.IsoOffsetTime());
51  		SWAPS.put(StackTraceElement.class, new StackTraceElementSwap());
52  		SWAPS.put(Year.class, new TemporalSwap.IsoYear());
53  		SWAPS.put(YearMonth.class, new TemporalSwap.IsoYearMonth());
54  		SWAPS.put(Temporal.class, new TemporalSwap.IsoInstant());
55  		SWAPS.put(TimeZone.class, new TimeZoneSwap());
56  		SWAPS.put(XMLGregorianCalendar.class, new XMLGregorianCalendarSwap());
57  		SWAPS.put(ZoneId.class, new ZoneIdSwap());
58  		SWAPS.put(MatchResult.class, new MatchResultSwap());
59  		SWAPS.put(URL.class, new UrlSwap());
60  	}
61  
62  	/**
63  	 * Find the default ObjectSwap for the specified class.
64  	 *
65  	 * @param ci The class to find the swap for.
66  	 * @return The matched swap, or <jk>null</jk> if it couldn't be found.
67  	 */
68  	public static ObjectSwap<?,?> find(ClassInfo ci) {
69  		ClassInfo ci2 = ci.getAnyParent(x -> SWAPS.get(x.inner()) != null);
70  		return ci2 == null ? null : SWAPS.get(ci2.inner());
71  	}
72  }
