33  /**
34   * Maintain the list of default swaps used by all serializers and parsers.
35   *
36   * <h5 class='section'>See Also:</h5><ul>
37   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SwapBasics">Swap Basics</a>
38   * </ul>
39   */
40  public class DefaultSwaps {
41  
42  	private static final Map<Class<?>,ObjectSwap<?,?>> SWAPS = new ConcurrentHashMap<>();
43  	static {
44  		SWAPS.put(Enumeration.class, new EnumerationSwap());
45  		SWAPS.put(Iterator.class, new IteratorSwap());
46  		SWAPS.put(Locale.class, new LocaleSwap());
47  		SWAPS.put(Class.class, new ClassSwap());
48  		SWAPS.put(Calendar.class, new TemporalCalendarSwap.IsoOffsetDateTime());
49  		SWAPS.put(Date.class, new TemporalDateSwap.IsoLocalDateTime());
50  		SWAPS.put(Instant.class, new TemporalSwap.IsoInstant());
51  		SWAPS.put(ZonedDateTime.class, new TemporalSwap.IsoOffsetDateTime());
52  		SWAPS.put(LocalDate.class, new TemporalSwap.IsoLocalDate());
53  		SWAPS.put(LocalDateTime.class, new TemporalSwap.IsoLocalDateTime());
54  		SWAPS.put(LocalTime.class, new TemporalSwap.IsoLocalTime());
55  		SWAPS.put(OffsetDateTime.class, new TemporalSwap.IsoOffsetDateTime());
56  		SWAPS.put(OffsetTime.class, new TemporalSwap.IsoOffsetTime());
57  		SWAPS.put(StackTraceElement.class, new StackTraceElementSwap());
58  		SWAPS.put(Year.class, new TemporalSwap.IsoYear());
59  		SWAPS.put(YearMonth.class, new TemporalSwap.IsoYearMonth());
60  		SWAPS.put(Temporal.class, new TemporalSwap.IsoInstant());
61  		SWAPS.put(TimeZone.class, new TimeZoneSwap());
62  		SWAPS.put(XMLGregorianCalendar.class, new XMLGregorianCalendarSwap());
63  		SWAPS.put(ZoneId.class, new ZoneIdSwap());
64  		SWAPS.put(MatchResult.class, new MatchResultSwap());
65  		SWAPS.put(URL.class, new UrlSwap());
66  	}
67  
68  	/**
69  	 * Find the default ObjectSwap for the specified class.
70  	 *
71  	 * @param ci The class to find the swap for.
72  	 * @return The matched swap, or <jk>null</jk> if it couldn't be found.
73  	 */
74  	public static ObjectSwap<?,?> find(ClassInfo ci) {
75  		var ci2 = ci.getAllParents().stream().filter(x -> nn(SWAPS.get(x.inner()))).findFirst().orElse(null);
76  		return ci2 == null ? null : SWAPS.get(ci2.inner());
77  	}
78  }
