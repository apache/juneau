13  package org.apache.juneau.swap;
14  
15  import org.apache.juneau.*;
16  import org.apache.juneau.collections.*;
17  
18  /**
19   * Abstract subclass for object swaps that swap objects for object maps.
20   *
21   * <h5 class='section'>Example:</h5>
22   * <p class='bjava'>
23   * 	<jc>// A swap that converts beans into generic maps.</jc>
24   * 	<jk>public class</jk> MyBeanSwap <jk>extends</jk> MapSwap&lt;<jk>byte</jk>[]&gt; {
25   *
26   * 		<ja>@Override</ja>
27   * 		<jk>public</jk> JsonMap swap(BeanSession <jv>session</jv>, MyBean <jv>bean</jv>) <jk>throws</jk> Exception {
28   * 			<jk>return</jk> JsonMap.<jsm>of</jsm>(<js>"foo"</js>, <jv>bean</jv>.getFoo());
29   * 		}
30   *
31   * 		<ja>@Override</ja>
32   * 		<jk>public</jk> MyBean unswap(BeanSession <jv>session</jv>, JsonMap <jv>map</jv>, ClassMeta&lt;?&gt; <jv>hint</jv>) <jk>throws</jk> Exception {
33   * 			<jk>return</jk> new</jk> MyBean(<jv>map</jv>.get(<js>"foo"</js>));
34   * 		}
35   * 	}
36   *
37   * 	<jc>// Use it to serialize a byte array.</jc>
38   * 	WriterSerializer <jv>serializer</jv> = JsonSerializer.<jsm>create</jsm>().simple().swaps(MyBeanSwap.<jk>class</jk>).build();
39   * 	String <jv>json</jv> = <jv>serializer</jv>.serialize(<jk>new</jk> MyBean(<js>"bar"</js>));  <jc>// Produces "{foo:'bar'}"</jc>
40   * </p>
41   *
42   * <h5 class='section'>See Also:</h5><ul>
43   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SwapBasics">Swaps</a>
44   * </ul>
45   *
46   * @param <T> The normal form of the class.
47   */
48  public abstract class MapSwap<T> extends ObjectSwap<T,JsonMap> {
49  
50  	@Override /* ObjectSwap */
51  	public JsonMap swap(BeanSession session, T o) throws Exception {
52  		return super.swap(session, o);
53  	}
54  
55  	@Override /* ObjectSwap */
56  	public T unswap(BeanSession session, JsonMap f, ClassMeta<?> hint) throws Exception {
57  		return super.unswap(session, f, hint);
58  	}
59  }
