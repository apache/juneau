13  package org.apache.juneau.swap;
14  
15  import org.apache.juneau.annotation.*;
16  
17  /**
18   * Identifies a class as being a surrogate class.
19   *
20   * <p>
21   * Surrogate classes are used in place of other classes during serialization.
22   * <br>For example, you may want to use a surrogate class to change the names or order of bean properties on a bean.
23   *
24   * <p>
25   * This interface has no methods to implement.
26   * <br>It's simply used by the framework to identify the class as a surrogate class when specified as a swap.
27   *
28   * <p>
29   * The following is an example of a surrogate class change changes a property name:
30   * <p class='bjava'>
31   * 	<jk>public class</jk> MySurrogate <jk>implements</jk> Surrogate {
32   *
33   * 		<jc>// Public constructor that wraps the normal object during serialization.</jc>
34   * 		<jk>public</jk> MySurrogate(NormalClass <jv>object</jv>) {...}
35   *
36   * 		<jc>// Public no-arg constructor using during parsing.</jc>
37   * 		<jc>// Not required if only used during serializing.</jc>
38   * 		<jk>public</jk> MySurrogate() {...}
39   *
40   * 		<jc>// Public method that converts surrogate back into normal object during parsing.</jc>
41   * 		<jc>// The method name can be anything (e.g. "build", "create", etc...).</jc>
42   * 		<jc>// Not required if only used during serializing.</jc>
43   * 		<jk>public</jk> NormalClass unswap() {...}
44   * 	}
45   * </p>
46   *
47   * <p>
48   * Surrogate classes are associated with serializers and parsers using the {@link org.apache.juneau.BeanContext.Builder#swaps(Class...)}
49   * method.
50   * <p class='bjava'>
51   * 	JsonSerializer <jv>serializer</jv> = JsonSerializer
52   * 		.<jsm>create</jsm>()
53   * 		.swaps(MySurrogate.<jk>class</jk>)
54   * 		.build();
55   *
56   * 	JsonParser <jv>parser</jv> = JsonParser
57   * 		.<jsm>create</jsm>()
58   * 		.swaps(MySurrogate.<jk>class</jk>)
59   * 		.build();
60   * </p>
61   *
62   * <p>
63   * Surrogates can also be associated using the {@link Swap @Swap} annotation.
64   * <p class='bjava'>
65   * 	<ja>@Swap</ja>(MySurrogate.<jk>class</jk>)
66   * 	<jk>public class</jk> NormalClass {...}
67   * </p>
68   *
69   * <p>
70   * On a side note, a surrogate class is functionally equivalent to the following {@link ObjectSwap}
71   * implementation:
72   * <p class='bjava'>
73   * 	<jk>public class</jk> MySurrogate <jk>extends</jk> ObjectSwap&lt;NormalClass,MySurrogate&gt; {
74   * 		<jk>public</jk> MySurrogate swap(NormalClass <jv>object</jv>) <jk>throws</jk> SerializeException {
75   * 			<jk>return new</jk> MySurrogate(<jv>object</jv>);
76   * 		}
77   * 		<jk>public</jk> NormalClass unswap(MySurrogate <jv>surrogate</jv>, ClassMeta&lt;?&gt; <jv>hint</jv>) <jk>throws</jk> ParseException {
78   * 			<jk>return</jk> <jv>surrogate</jv>.unswap();
79   * 		}
80   * 	}
81   * </p>
82   *
83   * <h5 class='section'>See Also:</h5><ul>
84   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SurrogateClasses">Surrogate Classes</a>
85   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SwapBasics">Swaps</a>
86   * </ul>
87   */
88  public interface Surrogate {}
