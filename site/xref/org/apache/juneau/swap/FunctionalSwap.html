13  package org.apache.juneau.swap;
14  
15  import org.apache.juneau.*;
16  import org.apache.juneau.utils.*;
17  
18  /**
19   * A subclass of {@link ObjectSwap} that allows swap and unswap methods to be defined as functions.
20   *
21   * <p class='bjava'>
22   * 	<jc>// Example</jc>
23   * 	<jk>public class</jk> MyBeanSwap <jk>extends</jk> FunctionalSwap&lt;MyBean,String&gt; {
24   * 		<jk>public</jk> MyBeanSwap() {
25   * 			<jk>super</jk>(MyBean.<jk>class</jk>, String.<jk>class</jk>, <jv>x</jv> -&gt; <jsm>myStringifyier</jsm>(<jv>x</jv>), <jv>x</jv> -&gt; <jsm>myDeStringifier</jsm>(<jv>x</jv>));
26   * 		}
27   * 	}
28   * </p>
29   *
30   * <h5 class='section'>See Also:</h5><ul>
31   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SwapBasics">Swaps</a>
32   * </ul>
33   *
34   * @param <T> The normal form of the class.
35   * @param <S> The swapped form of the class.
36   */
37  public class FunctionalSwap<T,S> extends ObjectSwap<T,S> {
38  
39  	private final ThrowingFunction<T,S> swapFunction;
40  	private final ThrowingFunction<S,T> unswapFunction;
41  
42  	/**
43  	 * Constructor.
44  	 *
45  	 * @param normalClass The normal class.
46  	 * @param swappedClass The swapped class.
47  	 * @param swapFunction The function for converting from normal to swapped.
48  	 */
49  	public FunctionalSwap(Class<T> normalClass, Class<S> swappedClass, ThrowingFunction<T,S> swapFunction) {
50  		this(normalClass, swappedClass, swapFunction, null);
51  	}
52  
53  	/**
54  	 * Constructor.
55  	 *
56  	 * @param normalClass The normal class.
57  	 * @param swappedClass The swapped class.
58  	 * @param swapFunction The function for converting from normal to swapped.
59  	 * @param unswapFunction The function for converting swapped to normal.
60  	 */
61  	public FunctionalSwap(Class<T> normalClass, Class<S> swappedClass, ThrowingFunction<T,S> swapFunction, ThrowingFunction<S,T> unswapFunction) {
62  		super(normalClass, swappedClass);
63  		this.swapFunction = swapFunction;
64  		this.unswapFunction = unswapFunction;
65  	}
66  
67  	@Override
68  	public S swap(BeanSession session, T o, String template) throws Exception {
69  		if (swapFunction == null)
70  			return super.swap(session, o, template);
71  		return swapFunction.applyThrows(o);
72  	}
73  
74  	@Override
75  	public T unswap(BeanSession session, S f, ClassMeta<?> hint, String template) throws Exception {
76  		if (unswapFunction == null)
77  			return super.unswap(session, f, hint, template);
78  		return unswapFunction.applyThrows(f);
79  	}
80  }
