17  package org.apache.juneau.swap;
18  
19  import org.apache.juneau.*;
20  import org.apache.juneau.commons.function.*;
21  
22  /**
23   * A subclass of {@link ObjectSwap} that allows swap and unswap methods to be defined as functions.
24   *
25   * <p class='bjava'>
26   * 	<jc>// Example</jc>
27   * 	<jk>public class</jk> MyBeanSwap <jk>extends</jk> FunctionalSwap&lt;MyBean,String&gt; {
28   * 		<jk>public</jk> MyBeanSwap() {
29   * 			<jk>super</jk>(MyBean.<jk>class</jk>, String.<jk>class</jk>, <jv>x</jv> -&gt; <jsm>myStringifyier</jsm>(<jv>x</jv>), <jv>x</jv> -&gt; <jsm>myDeStringifier</jsm>(<jv>x</jv>));
30   * 		}
31   * 	}
32   * </p>
33   *
34   * <h5 class='section'>See Also:</h5><ul>
35   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SwapBasics">Swap Basics</a>
36   * </ul>
37   *
38   * @param <T> The normal form of the class.
39   * @param <S> The swapped form of the class.
40   */
41  public class FunctionalSwap<T,S> extends ObjectSwap<T,S> {
42  
43  	private final ThrowingFunction<T,S> swapFunction;
44  	private final ThrowingFunction<S,T> unswapFunction;
45  
46  	/**
47  	 * Constructor.
48  	 *
49  	 * @param normalClass The normal class.
50  	 * @param swappedClass The swapped class.
51  	 * @param swapFunction The function for converting from normal to swapped.
52  	 */
53  	public FunctionalSwap(Class<T> normalClass, Class<S> swappedClass, ThrowingFunction<T,S> swapFunction) {
54  		this(normalClass, swappedClass, swapFunction, null);
55  	}
56  
57  	/**
58  	 * Constructor.
59  	 *
60  	 * @param normalClass The normal class.
61  	 * @param swappedClass The swapped class.
62  	 * @param swapFunction The function for converting from normal to swapped.
63  	 * @param unswapFunction The function for converting swapped to normal.
64  	 */
65  	public FunctionalSwap(Class<T> normalClass, Class<S> swappedClass, ThrowingFunction<T,S> swapFunction, ThrowingFunction<S,T> unswapFunction) {
66  		super(normalClass, swappedClass);
67  		this.swapFunction = swapFunction;
68  		this.unswapFunction = unswapFunction;
69  	}
70  
71  	@Override
72  	public S swap(BeanSession session, T o, String template) throws Exception {
73  		if (swapFunction == null)
74  			return super.swap(session, o, template);
75  		return swapFunction.applyThrows(o);
76  	}
77  
78  	@Override
79  	public T unswap(BeanSession session, S f, ClassMeta<?> hint, String template) throws Exception {
80  		if (unswapFunction == null)
81  			return super.unswap(session, f, hint, template);
82  		return unswapFunction.applyThrows(f);
83  	}
84  }
