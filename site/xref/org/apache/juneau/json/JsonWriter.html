26  /**
27   * Specialized writer for serializing JSON.
28   *
29   * <h5 class='section'>Notes:</h5><ul>
30   * 	<li class='note'>
31   * 		This class is not intended for external use.
32   * </ul>
33   *
34   * <h5 class='section'>See Also:</h5><ul>
35   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JsonBasics">JSON Basics</a>
36  
37   * </ul>
38   */
39  public class JsonWriter extends SerializerWriter {
40  
41  	private final boolean simpleAttrs, escapeSolidus;
42  
43  	// Characters that trigger special handling of serializing attribute values.
44  	private static final AsciiSet
45  		encodedChars = AsciiSet.of("\n\t\b\f\r'\"\\"),
46  		encodedChars2 = AsciiSet.of("\n\t\b\f\r'\"\\/");
47  
48  	private static final KeywordSet reservedWords = new KeywordSet(
49  		"arguments","break","case","catch","class","const","continue","debugger","default","delete",
50  		"do","else","enum","eval","export","extends","false","finally","for","function","if",
51  		"implements","import","in","instanceof","interface","let","new","null","package",
52  		"private","protected","public","return","static","super","switch","this","throw",
53  		"true","try","typeof","var","void","while","with","undefined","yield"
54  	);
55  
56  	// Characters that represent attribute name characters that don't trigger quoting.
57  	// These are actually more strict than the actual Javascript specification, but
58  	// can be narrowed in the future if necessary.
59  	// For example, we quote attributes that start with $ even though we don't need to.
60  	private static final AsciiSet validAttrChars = AsciiSet.create().ranges("a-z","A-Z","0-9").chars("_").build();
61  	private static final AsciiSet validFirstAttrChars = AsciiSet.create().ranges("a-z","A-Z").chars("_").build();
62  
63  	private final AsciiSet ec;
64  
65  	/**
66  	 * Constructor.
67  	 *
68  	 * @param out The writer being wrapped.
69  	 * @param useWhitespace If <jk>true</jk>, tabs and spaces will be used in output.
70  	 * @param maxIndent The maximum indentation level.
71  	 * @param escapeSolidus If <jk>true</jk>, forward slashes should be escaped in the output.
72  	 * @param quoteChar The quote character to use (i.e. <js>'\''</js> or <js>'"'</js>)
73  	 * @param simpleAttrs If <jk>true</jk>, JSON attributes will only be quoted when necessary.
74  	 * @param trimStrings If <jk>true</jk>, strings will be trimmed before being serialized.
75  	 * @param uriResolver The URI resolver for resolving URIs to absolute or root-relative form.
76  	 */
77  	protected JsonWriter(Writer out, boolean useWhitespace, int maxIndent, boolean escapeSolidus, char quoteChar,
78  			boolean simpleAttrs, boolean trimStrings, UriResolver uriResolver) {
79  		super(out, useWhitespace, maxIndent, trimStrings, quoteChar, uriResolver);
80  		this.simpleAttrs = simpleAttrs;
81  		this.escapeSolidus = escapeSolidus;
82  		this.ec = escapeSolidus ? encodedChars2 : encodedChars;
83  	}
84  
85  	/**
86  	 * Serializes the specified object as a JSON string value.
87  	 *
88  	 * @param s The object being serialized.
89  	 * @return This object.
90  	 */
91  	public JsonWriter stringValue(String s) {
92  		if (s == null)
93  			return this;
94  		boolean doConvert = false;
95  		for (int i = 0; i < s.length() && ! doConvert; i++) {
96  			char c = s.charAt(i);
97  			doConvert |= ec.contains(c);
98  		}
99  		q();
100 		if (! doConvert) {
101 			w(s);
102 		} else {
103 			for (int i = 0; i < s.length(); i++) {
104 				char c = s.charAt(i);
105 				if (ec.contains(c)) {
106 					if (c == '\n')
107 						w('\\').w('n');
108 					else if (c == '\t')
109 						w('\\').w('t');
110 					else if (c == '\b')
111 						w('\\').w('b');
112 					else if (c == '\f')
113 						w('\\').w('f');
114 					else if (c == quoteChar)
115 						w('\\').w(quoteChar);
116 					else if (c == '\\')
117 						w('\\').w('\\');
118 					else if (c == '/' && escapeSolidus)
119 						w('\\').w('/');
120 					else if (c != '\r')
121 						w(c);
122 				} else {
123 					w(c);
124 				}
125 			}
126 		}
127 		q();
128 		return this;
129 	}
130 
131 	/**
132 	 * Serializes the specified object as a JSON attribute name.
133 	 *
134 	 * @param s The object being serialized.
135 	 * @return This object.
136 	 */
137 	public JsonWriter attr(String s) {
138 
139 		if (trimStrings)
140 			s = StringUtils.trim(s);
141 
142 		/*
143 		 * Converts a Java string to an acceptable JSON attribute name. If
144 		 * simpleAttrs is true, then quotes will only be used if the attribute
145 		 * name consists of only alphanumeric characters.
146 		 */
147 		boolean doConvert = ! simpleAttrs;		// Always convert when not in lax mode.
148 
149 		// If the attribute is null, it must always be printed as null without quotes.
150 		// Technically, this isn't part of the JSON spec, but it does allow for null key values.
151 		if (s == null) {
152 			s = "null";
153 			doConvert = false;
154 
155 		} else {
156 
157 			// Look for characters that would require the attribute to be quoted.
158 			// All possible numbers should be caught here.
159 			if (! doConvert) {
160 				for (int i = 0; i < s.length() && ! doConvert; i++) {
161 					char c = s.charAt(i);
162 					doConvert |= ! (i == 0 ? validFirstAttrChars.contains(c) : validAttrChars.contains(c));
163 				}
164 			}
165 
166 			// Reserved words and blanks must be quoted.
167 			if (! doConvert) {
168 				if (s.isEmpty() || reservedWords.contains(s))
169 					doConvert = true;
170 			}
171 		}
172 
173 		// If no conversion necessary, just print the attribute as-is.
174 		if (doConvert)
175 			stringValue(s);
176 		else
177 			w(s);
178 
179 		return this;
180 	}
181 
182 	/**
183 	 * Appends a URI to the output.
184 	 *
185 	 * @param uri The URI to append to the output.
186 	 * @return This object.
187 	 */
188 	public SerializerWriter uriValue(Object uri) {
189 		return stringValue(uriResolver.resolve(uri));
190 	}
191 
192 	/**
193 	 * Performs an indentation only if we're currently past max indentation.
194 	 *
195 	 * @param depth The current indentation depth.
196 	 * @return This object.
197 	 */
198 	public JsonWriter smi(int depth) {
199 		if (depth > maxIndent)
200 			super.s();
201 		return this;
202 	}
203 
204 	/**
205 	 * Adds a space only if the current indentation level is below maxIndent.
206 	 *
207 	 * @param indent The number of spaces to indent.
208 	 * @return This object.
209 	 */
210 	public JsonWriter s(int indent) {
211 		if (indent <= maxIndent)
212 			super.s();
213 		return this;
214 	}
