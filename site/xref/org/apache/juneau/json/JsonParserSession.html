200 	private <T> T parseAnything(ClassMeta<?> eType, ParserReader r, Object outer, BeanPropertyMeta pMeta) throws IOException, ParseException, ExecutableException {
201 
202 		if (eType == null)
203 			eType = object();
204 		ObjectSwap<T,Object> swap = (ObjectSwap<T,Object>)eType.getSwap(this);
205 		BuilderSwap<T,Object> builder = (BuilderSwap<T,Object>)eType.getBuilderSwap(this);
206 		ClassMeta<?> sType = null;
207 		if (builder != null)
208 			sType = builder.getBuilderClassMeta(this);
209 		else if (swap != null)
210 			sType = swap.getSwapClassMeta(this);
211 		else
212 			sType = eType;
213 
214 		if (sType.isOptional())
215 			return (T)opt(parseAnything(eType.getElementType(), r, outer, pMeta));
216 
217 		setCurrentClass(sType);
218 		String wrapperAttr = getJsonClassMeta(sType).getWrapperAttr();
219 
220 		Object o = null;
221 
222 		skipCommentsAndSpace(r);
223 		if (wrapperAttr != null)
224 			skipWrapperAttrStart(r, wrapperAttr);
225 		int c = r.peek();
226 		if (c == -1) {
227 			if (isStrict())
228 				throw new ParseException(this, "Empty input.");
229 			// Let o be null.
230 		} else if ((c == ',' || c == '}' || c == ']')) {
231 			if (isStrict())
232 				throw new ParseException(this, "Missing value detected.");
233 			// Handle bug in Cognos 10.2.1 that can product non-existent values.
234 			// Let o be null;
235 		} else if (c == 'n') {
236 			parseKeyword("null", r);
237 		} else if (sType.isObject()) {
238 			if (c == '{') {
239 				JsonMap m2 = new JsonMap(this);
240 				parseIntoMap2(r, m2, string(), object(), pMeta);
241 				o = cast(m2, pMeta, eType);
242 			} else if (c == '[') {
243 				o = parseIntoCollection2(r, new JsonList(this), object(), pMeta);
244 			} else if (c == '\'' || c == '"') {
245 				o = parseString(r);
246 				if (sType.isChar())
247 					o = parseCharacter(o);
248 			} else if (c >= '0' && c <= '9' || c == '-' || c == '.') {
249 				o = parseNumber(r, null);
250 			} else if (c == 't') {
251 				parseKeyword("true", r);
252 				o = Boolean.TRUE;
253 			} else {
254 				parseKeyword("false", r);
255 				o = Boolean.FALSE;
256 			}
257 		} else if (sType.isBoolean()) {
258 			o = parseBoolean(r);
259 		} else if (sType.isCharSequence()) {
260 			o = parseString(r);
261 		} else if (sType.isChar()) {
262 			o = parseCharacter(parseString(r));
263 		} else if (sType.isNumber()) {
264 			o = parseNumber(r, (Class<? extends Number>)sType.getInnerClass());
265 		} else if (sType.isMap()) {
266 			Map m = (sType.canCreateNewInstance(outer) ? (Map)sType.newInstance(outer) : newGenericMap(sType));
267 			o = parseIntoMap2(r, m, sType.getKeyType(), sType.getValueType(), pMeta);
268 		} else if (sType.isCollection()) {
269 			if (c == '{') {
270 				JsonMap m = new JsonMap(this);
271 				parseIntoMap2(r, m, string(), object(), pMeta);
272 				o = cast(m, pMeta, eType);
273 			} else {
274 				Collection l = (sType.canCreateNewInstance(outer) ? (Collection)sType.newInstance() : new JsonList(this));
275 				o = parseIntoCollection2(r, l, sType, pMeta);
276 			}
277 		} else if (builder != null) {
278 			BeanMap m = toBeanMap(builder.create(this, eType));
279 			o = builder.build(this, parseIntoBeanMap2(r, m).getBean(), eType);
280 		} else if (sType.canCreateNewBean(outer)) {
281 			BeanMap m = newBeanMap(outer, sType.getInnerClass());
282 			o = parseIntoBeanMap2(r, m).getBean();
283 		} else if (sType.canCreateNewInstanceFromString(outer) && (c == '\'' || c == '"')) {
284 			o = sType.newInstanceFromString(outer, parseString(r));
285 		} else if (sType.isArray() || sType.isArgs()) {
286 			if (c == '{') {
287 				JsonMap m = new JsonMap(this);
288 				parseIntoMap2(r, m, string(), object(), pMeta);
289 				o = cast(m, pMeta, eType);
290 			} else {
291 				ArrayList l = (ArrayList)parseIntoCollection2(r, list(), sType, pMeta);
292 				o = toArray(sType, l);
293 			}
294 		} else if (c == '{') {
295 			Map m = new JsonMap(this);
296 			parseIntoMap2(r, m, sType.getKeyType(), sType.getValueType(), pMeta);
297 			if (m.containsKey(getBeanTypePropertyName(eType)))
298 				o = cast((JsonMap)m, pMeta, eType);
299 			else if (sType.getProxyInvocationHandler() != null)
300 				o = newBeanMap(outer, sType.getInnerClass()).load(m).getBean();
301 			else
302 				throw new ParseException(this, "Class ''{0}'' could not be instantiated.  Reason: ''{1}''", sType.getInnerClass().getName(), sType.getNotABeanReason());
303 		} else if (sType.canCreateNewInstanceFromString(outer) && ! isStrict()) {
304 			o = sType.newInstanceFromString(outer, parseString(r));
305 		} else {
306 			throw new ParseException(this, "Unrecognized syntax for class type ''{0}'', starting character ''{1}''", sType, (char)c);
307 		}
308 
309 		if (wrapperAttr != null)
310 			skipWrapperAttrEnd(r);
311 
312 		if (swap != null && o != null)
313 			o = unswap(swap, o, eType);
314 
315 		if (outer != null)
316 			setParent(eType, o, outer);
317 
318 		return (T)o;
319 	}

321 	private Boolean parseBoolean(ParserReader r) throws IOException, ParseException {
322 		int c = r.peek();
323 		if (c == '\'' || c == '"')
324 			return Boolean.valueOf(parseString(r));
325 		if (c == 't') {
326 			parseKeyword("true", r);
327 			return Boolean.TRUE;
328 		} else if (c == 'f') {
329 			parseKeyword("false", r);
330 			return Boolean.FALSE;
331 		} else {
332 			throw new ParseException(this, "Unrecognized syntax.  Expected boolean value, actual=''{0}''", r.read(100));
333 		}
334 	}

336 	/*
337 	 * Parse a JSON attribute from the character array at the specified position, then
338 	 * set the position marker to the last character in the field name.
339 	 */
340 	private String parseFieldName(ParserReader r) throws IOException, ParseException {
341 		int c = r.peek();
342 		if (c == '\'' || c == '"')
343 			return parseString(r);
344 		if (isStrict())
345 			throw new ParseException(this, "Unquoted attribute detected.");
346 		if (! VALID_BARE_CHARS.contains(c))
347 			throw new ParseException(this, "Could not find the start of the field name.");
348 		r.mark();
349 		// Look for whitespace.
350 		while (c != -1) {
351 			c = r.read();
352 			if (! VALID_BARE_CHARS.contains(c)) {
353 				r.unread();
354 				String s = r.getMarked().intern();
355 				return s.equals("null") ? null : s;
356 			}
357 		}
358 		throw new ParseException(this, "Could not find the end of the field name.");
359 	}

361 	private <T> BeanMap<T> parseIntoBeanMap2(ParserReader r, BeanMap<T> m) throws IOException, ParseException, ExecutableException {
362 
363 		// S1: Looking for outer {
364 		// S2: Looking for attrName start.
365 		// S3: Found attrName end, looking for :.
366 		// S4: Found :, looking for valStart: { [ " ' LITERAL.
367 		// S5: Looking for , or }
368 
369 		var state = S1;
370 		String currAttr = "";
371 		int c = 0;
372 		mark();
373 		try {
374 			while (c != -1) {
375 				c = r.read();
376 				if (state == S1) {
377 					if (c == '{') {
378 						state = S2;
379 					} else if (isCommentOrWhitespace(c)) {
380 						skipCommentsAndSpace(r.unread());
381 					} else {
382 						break;
383 					}
384 				} else if (state == S2) {
385 					if (c == '}') {
386 						return m;
387 					} else if (isCommentOrWhitespace(c)) {
388 						skipCommentsAndSpace(r.unread());
389 					} else {
390 						r.unread();
391 						mark();
392 						currAttr = parseFieldName(r);
393 						state = S3;
394 					}
395 				} else if (state == S3) {
396 					if (c == ':')
397 						state = S4;
398 				} else if (state == S4) {
399 					if (isCommentOrWhitespace(c)) {
400 						skipCommentsAndSpace(r.unread());
401 					} else {
402 						if (! currAttr.equals(getBeanTypePropertyName(m.getClassMeta()))) {
403 							BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
404 							setCurrentProperty(pMeta);
405 							if (pMeta == null) {
406 								onUnknownProperty(currAttr, m, parseAnything(object(), r.unread(), m.getBean(false), null));
407 								unmark();
408 							} else {
409 								unmark();
410 								ClassMeta<?> cm = pMeta.getClassMeta();
411 								Object value = parseAnything(cm, r.unread(), m.getBean(false), pMeta);
412 								setName(cm, value, currAttr);
413 								try {
414 									pMeta.set(m, currAttr, value);
415 								} catch (BeanRuntimeException e) {
416 									onBeanSetterException(pMeta, e);
417 									throw e;
418 								}
419 							}
420 							setCurrentProperty(null);
421 						}
422 						state = S5;
423 					}
424 				} else if (state == S5) {
425 					if (c == ',')
426 						state = S2;
427 					else if (isCommentOrWhitespace(c))
428 						skipCommentsAndSpace(r.unread());
429 					else if (c == '}') {
430 						return m;
431 					}
432 				}
433 			}
434 			if (state == S1)
435 				throw new ParseException(this, "Expected '{' at beginning of JSON object.");
436 			if (state == S2)
437 				throw new ParseException(this, "Could not find attribute name on JSON object.");
438 			if (state == S3)
439 				throw new ParseException(this, "Could not find ':' following attribute name on JSON object.");
440 			if (state == S4)
441 				throw new ParseException(this, "Expected one of the following characters: {,[,',\",LITERAL.");
442 			if (state == S5)
443 				throw new ParseException(this, "Could not find '}' marking end of JSON object.");
444 		} finally {
445 			unmark();
446 		}
447 
448 		return null; // Unreachable.
449 	}

451 	private <E> Collection<E> parseIntoCollection2(ParserReader r, Collection<E> l, ClassMeta<?> type, BeanPropertyMeta pMeta) throws IOException, ParseException, ExecutableException {
452 
453 		// S1: Looking for outermost [
454 		// S2: Looking for starting [ or { or " or ' or LITERAL or ]
455 		// S3: Looking for , or ]
456 		// S4: Looking for starting [ or { or " or ' or LITERAL
457 
458 		int argIndex = 0;
459 
460 		var state = S1;
461 		int c = 0;
462 		while (c != -1) {
463 			c = r.read();
464 			if (state == S1) {
465 				if (c == '[')
466 					state = S2;
467 				else if (isCommentOrWhitespace(c))
468 					skipCommentsAndSpace(r.unread());
469 				else
470 					break;  // Invalid character found.
471 			} else if (state == S2) {
472 				if (c == ']') {
473 					return l;
474 				} else if (isCommentOrWhitespace(c)) {
475 					skipCommentsAndSpace(r.unread());
476 				} else if (c != -1) {
477 					l.add((E)parseAnything(type.isArgs() ? type.getArg(argIndex++) : type.getElementType(), r.unread(), l, pMeta));
478 					state = S3;
479 				}
480 			} else if (state == S3) {
481 				if (c == ',') {
482 					state = S4;
483 				} else if (isCommentOrWhitespace(c)) {
484 					skipCommentsAndSpace(r.unread());
485 				} else if (c == ']') {
486 					return l;
487 				} else {
488 					break;  // Invalid character found.
489 				}
490 			} else if (state == S4) {
491 				if (isCommentOrWhitespace(c)) {
492 					skipCommentsAndSpace(r.unread());
493 				} else if (c == ']') {
494 					break;
495 				} else if (c != -1) {
496 					l.add((E)parseAnything(type.isArgs() ? type.getArg(argIndex++) : type.getElementType(), r.unread(), l, pMeta));
497 					state = S3;
498 				}
499 			}
500 		}
501 		if (state == S1)
502 			throw new ParseException(this, "Expected '[' at beginning of JSON array.");
503 		if (state == S2)
504 			throw new ParseException(this, "Expected one of the following characters: {,[,',\",LITERAL.");
505 		if (state == S3)
506 			throw new ParseException(this, "Expected ',' or ']'.");
507 		if (state == S4)
508 			throw new ParseException(this, "Unexpected trailing comma in array.");
509 
510 		return null;  // Unreachable.
511 	}

513 	private <K,V> Map<K,V> parseIntoMap2(ParserReader r, Map<K,V> m, ClassMeta<K> keyType, ClassMeta<V> valueType, BeanPropertyMeta pMeta) throws IOException, ParseException, ExecutableException {
514 
515 		if (keyType == null)
516 			keyType = (ClassMeta<K>)string();
517 
518 		// S1: Looking for outer {
519 		// S2: Looking for attrName start.
520 		// S3: Found attrName end, looking for :.
521 		// S4: Found :, looking for valStart: { [ " ' LITERAL.
522 		// S5: Looking for , or }
523 		// S6: Found , looking for attr start.
524 
525 		skipCommentsAndSpace(r);
526 		var state = S1;
527 		String currAttr = null;
528 		int c = 0;
529 		while (c != -1) {
530 			c = r.read();
531 			if (state == S1) {
532 				if (c == '{')
533 					state = S2;
534 				else
535 					break;
536 			} else if (state == S2) {
537 				if (c == '}') {
538 					return m;
539 				} else if (isCommentOrWhitespace(c)) {
540 					skipCommentsAndSpace(r.unread());
541 				} else {
542 					currAttr = parseFieldName(r.unread());
543 					state = S3;
544 				}
545 			} else if (state == S3) {
546 				if (c == ':')
547 					state = S4;
548 			} else if (state == S4) {
549 				if (isCommentOrWhitespace(c)) {
550 					skipCommentsAndSpace(r.unread());
551 				} else {
552 					K key = convertAttrToType(m, currAttr, keyType);
553 					V value = parseAnything(valueType, r.unread(), m, pMeta);
554 					setName(valueType, value, key);
555 					m.put(key, value);
556 					state = S5;
557 				}
558 			} else if (state == S5) {
559 				if (c == ',') {
560 					state = S6;
561 				} else if (isCommentOrWhitespace(c)) {
562 					skipCommentsAndSpace(r.unread());
563 				} else if (c == '}') {
564 					return m;
565 				} else {
566 					break;
567 				}
568 			} else if (state == S6) {
569 				if (c == '}') {
570 					break;
571 				} else if (isCommentOrWhitespace(c)) {
572 					skipCommentsAndSpace(r.unread());
573 				} else {
574 					currAttr = parseFieldName(r.unread());
575 					state = S3;
576 				}
577 			}
578 		}
579 		if (state == S1)
580 			throw new ParseException(this, "Expected '{' at beginning of JSON object.");
581 		if (state == S2)
582 			throw new ParseException(this, "Could not find attribute name on JSON object.");
583 		if (state == S3)
584 			throw new ParseException(this, "Could not find ':' following attribute name on JSON object.");
585 		if (state == S4)
586 			throw new ParseException(this, "Expected one of the following characters: {,[,',\",LITERAL.");
587 		if (state == S5)
588 			throw new ParseException(this, "Could not find '}' marking end of JSON object.");
589 		if (state == S6)
590 			throw new ParseException(this, "Unexpected '}' found in JSON object.");
591 
592 		return null; // Unreachable.
593 	}

595 	/*
596 	 * Looks for the keywords true, false, or null.
597 	 * Throws an exception if any of these keywords are not found at the specified position.
598 	 */
599 	private void parseKeyword(String keyword, ParserReader r) throws IOException, ParseException {
600 		try {
601 			String s = r.read(keyword.length());
602 			if (s.equals(keyword))
603 				return;
604 			throw new ParseException(this, "Unrecognized syntax.  Expected=''{0}'', Actual=''{1}''", keyword, s);
605 		} catch (@SuppressWarnings("unused") IndexOutOfBoundsException e) {
606 			throw new ParseException(this, "Unrecognized syntax.  Expected=''{0}'', found end-of-file.", keyword);
607 		}
608 	}

610 	private Number parseNumber(ParserReader r, Class<? extends Number> type) throws IOException, ParseException {
611 		int c = r.peek();
612 		if (c == '\'' || c == '"')
613 			return parseNumber(r, parseString(r), type);
614 		return parseNumber(r, r.parseNumberString(), type);
615 	}

617 	private Number parseNumber(ParserReader r, String s, Class<? extends Number> type) throws ParseException {
618 
619 		// JSON has slightly different number rules from Java.
620 		// Strict mode enforces these different rules, lax does not.
621 		if (isStrict()) {
622 
623 			// Lax allows blank strings to represent 0.
624 			// Strict does not allow blank strings.
625 			if (s.isEmpty())
626 				throw new ParseException(this, "Invalid JSON number: ''{0}''", s);
627 
628 			// Need to weed out octal and hexadecimal formats:  0123,-0123,0x123,-0x123.
629 			// Don't weed out 0 or -0.
630 			boolean isNegative = false;
631 			char c = s.charAt(0);
632 			if (c == '-') {
633 				isNegative = true;
634 				c = (s.length() == 1 ? 'x' : s.charAt(1));
635 			}
636 
637 			// JSON doesn't allow '.123' and '-.123'.
638 			if (c == '.')
639 				throw new ParseException(this, "Invalid JSON number: ''{0}''", s);
640 
641 			// '01' is not a valid number, but '0.1', '0e1', '0e+1' are valid.
642 			if (c == '0' && s.length() > (isNegative ? 2 : 1)) {
643 				char c2 = s.charAt((isNegative ? 2 : 1));
644 				if (c2 != '.' && c2 != 'e' && c2 != 'E')
645 					throw new ParseException(this, "Invalid JSON number: ''{0}''", s);
646 			}
647 
648 			// JSON doesn't allow '1.' or '0.e1'.
649 			int i = s.indexOf('.');
650 			if (i != -1 && (s.length() == (i + 1) || ! decChars.contains(s.charAt(i + 1))))
651 				throw new ParseException(this, "Invalid JSON number: ''{0}''", s);
652 
653 		}
654 		return StringUtils.parseNumber(s, type);
655 	}

657 	/*
658 	 * Starting from the specified position in the character array, returns the
659 	 * position of the character " or '.
660 	 * If the string consists of a concatenation of strings (e.g. 'AAA' + "BBB"), this method
661 	 * will automatically concatenate the strings and return the result.
662 	 */
663 	private String parseString(ParserReader r) throws IOException, ParseException {
664 		r.mark();
665 		int qc = r.read();		// The quote character being used (" or ')
666 		if (qc != '"' && isStrict()) {
667 			String msg = (qc == '\'' ? "Invalid quote character \"{0}\" being used." : "Did not find quote character marking beginning of string.  Character=\"{0}\"");
668 			throw new ParseException(this, msg, (char)qc);
669 		}
670 		final boolean isQuoted = (qc == '\'' || qc == '"');
671 		String s = null;
672 		boolean isInEscape = false;
673 		int c = 0;
674 		while (c != -1) {
675 			c = r.read();
676 			// Strict syntax requires that all control characters be escaped.
677 			if (isStrict() && c <= 0x1F)
678 				throw new ParseException(this, "Unescaped control character encountered: ''0x{0}''", String.format("%04X", c));
679 			if (isInEscape) {
680 				// @formatter:off
681 				switch (c) {
682 					case 'n': r.replace('\n'); break;
683 					case 'r': r.replace('\r'); break;
684 					case 't': r.replace('\t'); break;
685 					case 'f': r.replace('\f'); break;
686 					case 'b': r.replace('\b'); break;
687 					case '\\': r.replace('\\'); break;
688 					case '/': r.replace('/'); break;
689 					case '\'': r.replace('\''); break;
690 					case '"': r.replace('"'); break;
691 					case 'u': {
692 						String n = r.read(4);
693 						try {
694 							r.replace(Integer.parseInt(n, 16), 6);
695 						} catch (@SuppressWarnings("unused") NumberFormatException e) {
696 							throw new ParseException(this, "Invalid Unicode escape sequence in string.");
697 						}
698 						break;
699 					}
700 					default:
701 						throw new ParseException(this, "Invalid escape sequence in string.");
702 				}
703 				// @formatter:on
704 				isInEscape = false;
705 			} else {
706 				if (c == '\\') {
707 					isInEscape = true;
708 					r.delete();
709 				} else if (isQuoted) {
710 					if (c == qc) {
711 						s = r.getMarked(1, -1);
712 						break;
713 					}
714 				} else {
715 					if (c == ',' || c == '}' || c == ']' || isWhitespace(c)) {
716 						s = r.getMarked(0, -1);
717 						r.unread();
718 						break;
719 					} else if (c == -1) {
720 						s = r.getMarked(0, 0);
721 						break;
722 					}
723 				}
724 			}
725 		}
726 		if (s == null)
727 			throw new ParseException(this, "Could not find expected end character ''{0}''.", (char)qc);
728 
729 		// Look for concatenated string (i.e. whitespace followed by +).
730 		skipCommentsAndSpace(r);
731 		if (r.peek() == '+') {
732 			if (isStrict())
733 				throw new ParseException(this, "String concatenation detected.");
734 			r.read();  // Skip past '+', NOSONAR - Intentional.
735 			skipCommentsAndSpace(r);
736 			s += parseString(r);
737 		}
738 		return trim(s); // End of input reached.
739 	}

741 	/*
742 	 * Doesn't actually parse anything, but when positioned at the beginning of comment,
743 	 * it will move the pointer to the last character in the comment.
744 	 */
745 	private void skipComments(ParserReader r) throws ParseException, IOException {
746 		int c = r.read();
747 		//  "/* */" style comments
748 		if (c == '*') {
749 			while (c != -1)
750 				if ((c = r.read()) == '*')
751 					if ((c = r.read()) == '/')
752 						return;
753 			//  "//" style comments
754 		} else if (c == '/') {
755 			while (c != -1) {
756 				c = r.read();
757 				if (c == -1 || c == '\n')
758 					return;
759 			}
760 		}
761 		throw new ParseException(this, "Open ended comment.");
762 	}
763 
764 	/*
765 	 * Doesn't actually parse anything, but moves the position beyond any whitespace or comments.
766 	 * If positionOnNext is 'true', then the cursor will be set to the point immediately after
767 	 * the comments and whitespace.  Otherwise, the cursor will be set to the last position of
768 	 * the comments and whitespace.
769 	 */
770 	private void skipCommentsAndSpace(ParserReader r) throws IOException, ParseException {
771 		int c = 0;
772 		while ((c = r.read()) != -1) {
773 			if (! isWhitespace(c)) {
774 				if (c == '/') {
775 					if (isStrict())
776 						throw new ParseException(this, "Javascript comment detected.");
777 					skipComments(r);
778 				} else {
779 					r.unread();
780 					return;
781 				}
782 			}
783 		}
784 	}

786 	/*
787 	 * Doesn't actually parse anything, but moves the position beyond the construct "}" when
788 	 * the @Json(wrapperAttr) annotation is used on a class.
789 	 */
790 	private void skipWrapperAttrEnd(ParserReader r) throws ParseException, IOException {
791 		int c = 0;
792 		while ((c = r.read()) != -1) {
793 			if (! isWhitespace(c)) {
794 				if (c == '/') {
795 					if (isStrict())
796 						throw new ParseException(this, "Javascript comment detected.");
797 					skipComments(r);
798 				} else if (c == '}') {
799 					return;
800 				} else {
801 					throw new ParseException(this, "Could not find '}' at the end of JSON wrapper object.");
802 				}
803 			}
804 		}
805 	}

807 	/*
808 	 * Doesn't actually parse anything, but moves the position beyond the construct "{wrapperAttr:" when
809 	 * the @Json(wrapperAttr) annotation is used on a class.
810 	 */
811 	private void skipWrapperAttrStart(ParserReader r, String wrapperAttr) throws IOException, ParseException {
812 
813 		// S1: Looking for outer '{'
814 		// S2: Looking for attrName start.
815 		// S3: Found attrName end, looking for :.
816 		// S4: Found :, looking for valStart: { [ " ' LITERAL.
817 
818 		var state = S1;
819 		String currAttr = null;
820 		int c = 0;
821 		while (c != -1) {
822 			c = r.read();
823 			if (state == S1) {
824 				if (c == '{')
825 					state = S2;
826 			} else if (state == S2) {
827 				if (isCommentOrWhitespace(c)) {
828 					skipCommentsAndSpace(r.unread());
829 				} else {
830 					currAttr = parseFieldName(r.unread());
831 					if (! currAttr.equals(wrapperAttr))
832 						throw new ParseException(this, "Expected to find wrapper attribute ''{0}'' but found attribute ''{1}''", wrapperAttr, currAttr);
833 					state = S3;
834 				}
835 			} else if (state == S3) {
836 				if (c == ':')
837 					state = S4;
838 			} else if (state == S4) {
839 				if (isCommentOrWhitespace(c)) {
840 					skipCommentsAndSpace(r.unread());
841 				} else {
842 					r.unread();
843 					return;
844 				}
845 			}
846 		}
847 		if (state == S1)
848 			throw new ParseException(this, "Expected '{' at beginning of JSON object.");
849 		if (state == S2)
850 			throw new ParseException(this, "Could not find attribute name on JSON object.");
851 		if (state == S3)
852 			throw new ParseException(this, "Could not find ':' following attribute name on JSON object.");
853 		if (state == S4)
854 			throw new ParseException(this, "Expected one of the following characters: {,[,',\",LITERAL.");
855 	}

858 	 * Call this method after you've finished a parsing a string to make sure that if there's any
859 	 * remainder in the input, that it consists only of whitespace and comments.
860 	 */
861 	private void validateEnd(ParserReader r) throws IOException, ParseException {
862 		if (! isValidateEnd())
863 			return;
864 		skipCommentsAndSpace(r);
865 		int c = r.read();
866 		if (c != -1 && c != ';')  // var x = {...}; expressions can end with a semicolon.
867 			throw new ParseException(this, "Remainder after parse: ''{0}''.", (char)c);
868 	}

933 	/**
934 	 * Returns <jk>true</jk> if the specified character is whitespace.
935 	 *
936 	 * <p>
937 	 * The definition of whitespace is different for strict vs lax mode.
938 	 * Strict mode only interprets 0x20 (space), 0x09 (tab), 0x0A (line feed) and 0x0D (carriage return) as whitespace.
939 	 * Lax mode uses {@link Character#isWhitespace(int)} to make the determination.
940 	 *
941 	 * @param cp The codepoint.
942 	 * @return <jk>true</jk> if the specified character is whitespace.
943 	 */

945 		if (isStrict())
946 			return cp <= 0x20 && (cp == 0x09 || cp == 0x0A || cp == 0x0D || cp == 0x20);
947 		return Character.isWhitespace(cp);
