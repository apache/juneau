18  /**
19   * Metadata on bean properties specific to the JSON serializers and parsers pulled from the {@link Json @Json}
20   * annotation on the bean property.
21   *
22   * <h5 class='section'>See Also:</h5><ul>
23   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JsonBasics">JSON Details</a>
24  
25   * </ul>
26   */
27  public final class JsonBeanPropertyMeta extends ExtendedBeanPropertyMeta {
28  
29  	/**
30  	 * Default instance.
31  	 */
32  	public static final JsonBeanPropertyMeta DEFAULT = new JsonBeanPropertyMeta();
33  
34  	/**
35  	 * Constructor.
36  	 *
37  	 * @param bpm The metadata of the bean property of this additional metadata.
38  	 * @param mp JSON metadata provider (for finding information about other artifacts).
39  	 */
40  	public JsonBeanPropertyMeta(BeanPropertyMeta bpm, JsonMetaProvider mp) {
41  		super(bpm);
42  	}
43  
44  	private JsonBeanPropertyMeta() {
45  		super(null);
46  	}
47  }
