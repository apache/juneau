13  package org.apache.juneau.json;
14  
15  /**
16   * Parses any valid JSON text into a POJO model.
17   *
18   * <h5 class='topic'>Media types</h5>
19   *
20   * Handles <c>Content-Type</c> types:  <bc>application/json5, text/json5</bc>
21   *
22   * <h5 class='topic'>Description</h5>
23   *
24   * Identical to {@link JsonParser} but with the media type <bc>application/json5</bc>.
25   *
26   * <h5 class='section'>Notes:</h5><ul>
27   * 	<li class='note'>This class is thread safe and reusable.
28   * </ul>
29   *
30   * <h5 class='section'>See Also:</h5><ul>
31   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JsonBasics">JSON Details</a>
32   * </ul>
33   */
34  public class Json5Parser extends JsonParser {
35  
36  	//-------------------------------------------------------------------------------------------------------------------
37  	// Static
38  	//-------------------------------------------------------------------------------------------------------------------
39  
40  	/** Default parser, Accept=application/json5. */
41  	public static final Json5Parser DEFAULT = new Json5Parser(create());
42  
43  	/**
44  	 * Creates a new builder for this object.
45  	 *
46  	 * @return A new builder.
47  	 */
48  	public static JsonParser.Builder create() {
49  		return JsonParser.create().consumes("application/json5,text/json5,application/json,text/json");
50  	}
51  
52  	//-------------------------------------------------------------------------------------------------------------------
53  	// Instance
54  	//-------------------------------------------------------------------------------------------------------------------
55  
56  	/**
57  	 * Constructor.
58  	 *
59  	 * @param builder The builder for this object.
60  	 */
61  	public Json5Parser(JsonParser.Builder builder) {
62  		super(builder);
63  	}
64  
65  	@Override /* Context */
66  	public JsonParser.Builder copy() {
67  		return new JsonParser.Builder(this);
68  	}
69  }
