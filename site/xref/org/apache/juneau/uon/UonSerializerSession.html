224 	@Override /* Overridden from HttpPartSerializer */
225 	public String serialize(HttpPartType type, HttpPartSchema schema, Object value) throws SerializeException, SchemaValidationException {
226 		try {
227 			// Shortcut for simple types.
228 			ClassMeta<?> cm = getClassMetaForObject(value);
229 			if (cm != null && (schema == null || schema.getType() == HttpPartDataType.NO_TYPE)) {
230 				if (cm.isNumber() || cm.isBoolean())
231 					return Mutaters.toString(value);
232 				if (cm.isString()) {
233 					String s = Mutaters.toString(value);
234 					if (s.isEmpty() || ! UonUtils.needsQuotes(s))
235 						return s;
236 				}
237 			}
238 			StringWriter w = new StringWriter();
239 			serializeAnything(getUonWriter(w).i(getInitialDepth()), value, getExpectedRootType(value), "root", null);
240 			return w.toString();
241 		} catch (Exception e) {
242 			throw asRuntimeException(e);
243 		}
244 	}

250 	private SerializerWriter serializeBeanMap(UonWriter out, BeanMap<?> m, String typeName) throws SerializeException {
251 
252 		if (! plainTextParams)
253 			out.append('(');
254 
255 		Flag addComma = Flag.create();
256 
257 		if (typeName != null) {
258 			BeanPropertyMeta pm = m.getMeta().getTypeProperty();
259 			out.cr(indent).appendObject(pm.getName(), false).append('=').appendObject(typeName, false);
260 			addComma.set();
261 		}
262 
263 		Predicate<Object> checkNull = x -> isKeepNullProperties() || x != null;
264 		m.forEachValue(checkNull, (pMeta, key, value, thrown) -> {
265 			ClassMeta<?> cMeta = pMeta.getClassMeta();
266 
267 			if (thrown != null)
268 				onBeanGetterException(pMeta, thrown);
269 
270 			if (canIgnoreValue(cMeta, key, value))
271 				return;
272 
273 			addComma.ifSet(() -> out.append(',')).set();
274 
275 			out.cr(indent).appendObject(key, false).append('=');
276 
277 			serializeAnything(out, value, cMeta, key, pMeta);
278 		});
279 
280 		if (m.size() > 0)
281 			out.cre(indent - 1);
282 		if (! plainTextParams)
283 			out.append(')');
284 
285 		return out;
286 	}

288 	@SuppressWarnings({ "rawtypes", "unchecked" })
289 	private SerializerWriter serializeCollection(UonWriter out, Collection c, ClassMeta<?> type) throws SerializeException {
290 
291 		ClassMeta<?> elementType = type.getElementType();
292 
293 		if (! plainTextParams)
294 			out.append('@').append('(');
295 
296 		Flag addComma = Flag.create();
297 		forEachEntry(c, x -> {
298 			addComma.ifSet(() -> out.append(',')).set();
299 			out.cr(indent);
300 			serializeAnything(out, x, elementType, "<iterator>", null);
301 		});
302 
303 		addComma.ifSet(() -> out.cre(indent - 1));
304 		if (! plainTextParams)
305 			out.append(')');
306 
307 		return out;
308 	}

310 	@SuppressWarnings({ "rawtypes", "unchecked" })
311 	private SerializerWriter serializeMap(UonWriter out, Map m, ClassMeta<?> type) throws SerializeException {
312 
313 		ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
314 
315 		if (! plainTextParams)
316 			out.append('(');
317 
318 		Flag addComma = Flag.create();
319 		forEachEntry(m, x -> {
320 			addComma.ifSet(() -> out.append(',')).set();
321 			Object value = x.getValue();
322 			Object key = generalize(x.getKey(), keyType);
323 			out.cr(indent).appendObject(key, false).append('=');
324 			serializeAnything(out, value, valueType, toString(key), null);
325 		});
326 
327 		addComma.ifSet(() -> out.cre(indent - 1));
328 
329 		if (! plainTextParams)
330 			out.append(')');
331 
332 		return out;
333 	}

367 		Object output = out.getRawOutput();

395 	/**
396 	 * Workhorse method.
397 	 *
398 	 * <p>
399 	 * Determines the type of object, and then calls the appropriate type-specific serialization method.
400 	 *
401 	 * @param out The writer to serialize to.
402 	 * @param o The object being serialized.
403 	 * @param eType The expected type of the object if this is a bean property.
404 	 * @param attrName
405 	 * 	The bean property name if this is a bean property.
406 	 * 	<jk>null</jk> if this isn't a bean property being serialized.
407 	 * @param pMeta The bean property metadata.
408 	 * @return The same writer passed in.
409 	 * @throws SerializeException Generic serialization error occurred.
410 	 */
411 	@SuppressWarnings({ "rawtypes" })
412 	protected SerializerWriter serializeAnything(UonWriter out, Object o, ClassMeta<?> eType, String attrName, BeanPropertyMeta pMeta) throws SerializeException {
413 
414 		if (o == null) {
415 			out.appendObject(null, false);
416 			return out;
417 		}
418 
419 		if (eType == null)
420 			eType = object();
421 
422 		ClassMeta<?> aType;			// The actual type
423 		ClassMeta<?> sType;			// The serialized type
424 
425 		aType = push2(attrName, o, eType);
426 		boolean isRecursion = aType == null;
427 
428 		// Handle recursion
429 		if (aType == null) {
430 			o = null;
431 			aType = object();
432 		}
433 
434 		// Handle Optional<X>
435 		if (isOptional(aType)) {
436 			o = getOptionalValue(o);
437 			eType = getOptionalType(eType);
438 			aType = getClassMetaForObject(o, object());
439 		}
440 
441 		sType = aType;
442 		String typeName = getBeanTypeName(this, eType, aType, pMeta);
443 
444 		// Swap if necessary
445 		ObjectSwap swap = aType.getSwap(this);
446 		if (swap != null) {
447 			o = swap(swap, o);
448 			sType = swap.getSwapClassMeta(this);
449 
450 			// If the getSwapClass() method returns Object, we need to figure out
451 			// the actual type now.
452 			if (sType.isObject())
453 				sType = getClassMetaForObject(o);
454 		}
455 
456 		// '\0' characters are considered null.
457 		if (o == null || (sType.isChar() && ((Character)o).charValue() == 0))
458 			out.appendObject(null, false);
459 		else if (sType.isBoolean())
460 			out.appendBoolean(o);
461 		else if (sType.isNumber())
462 			out.appendNumber(o);
463 		else if (sType.isBean())
464 			serializeBeanMap(out, toBeanMap(o), typeName);
465 		else if (sType.isUri() || (pMeta != null && pMeta.isUri()))
466 			out.appendUri(o);
467 		else if (sType.isMap()) {
468 			if (o instanceof BeanMap)
469 				serializeBeanMap(out, (BeanMap)o, typeName);
470 			else
471 				serializeMap(out, (Map)o, eType);
472 		} else if (sType.isCollection()) {
473 			serializeCollection(out, (Collection)o, eType);
474 		} else if (sType.isArray()) {
475 			serializeCollection(out, toList(sType.getInnerClass(), o), eType);
476 		} else if (sType.isReader()) {
477 			pipe((Reader)o, out, SerializerSession::handleThrown);
478 		} else if (sType.isInputStream()) {
479 			pipe((InputStream)o, out, SerializerSession::handleThrown);
480 		} else {
481 			out.appendObject(o, false);
482 		}
483 
484 		if (! isRecursion)
485 			pop();
486 		return out;
487 	}
