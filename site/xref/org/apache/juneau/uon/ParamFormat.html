13  package org.apache.juneau.uon;
14  
15  /**
16   * Identifies the possible values for the {@link UonSerializer.Builder#paramFormat(ParamFormat)} setting.
17   *
18   * <h5 class='section'>See Also:</h5><ul>
19   * 	<li class='link'><a class="doclink" href="../../../../index.html#jm.UonDetails">UON Details</a>
20   * </ul>
21   */
22  public enum ParamFormat {
23  
24  	/**
25  	 * Use UON notation for values.
26  	 *
27  	 * <p>
28  	 * String values such as <js>"(foo='bar')"</js> will end up being quoted and escaped to <js>"'(foo=bar~'baz~')'"</js>.
29  	 *
30  	 * <p>
31  	 * Boolean strings (<js>"true"</js>/<js>"false"</js>) and numeric values (<js>"123"</js>) will also end up quoted
32  	 * (<js>"'true'"</js>, <js>"'false'"</js>, <js>"'123'"</js>.
33  	 */
34  	UON,
35  
36  	/**
37  	 * Serialize as plain text.
38  	 *
39  	 * <p>
40  	 * Strings will never be quoted or escaped.
41  	 * <br>Maps and array constructs (<js>"(...)"</js>, <js>"@(...)"</js>) will never be used.
42  	 *
43  	 * <p>
44  	 * Note that this can cause errors during parsing if you're using the URL-encoding parser to parse the results since
45  	 * UON constructs won't be differentiable.
46  	 * <br>However, this is not an issue if you're simply creating queries or form posts against 3rd-party interfaces.
47  	 */
48  	PLAINTEXT;
49  }
