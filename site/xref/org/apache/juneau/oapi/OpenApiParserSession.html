229 	@SuppressWarnings("unchecked")
230 	@Override /* HttpPartParser */
231 	public <T> T parse(HttpPartType partType, HttpPartSchema schema, String in, ClassMeta<T> type) throws ParseException, SchemaValidationException {
232 		if (partType == null)
233 			partType = HttpPartType.OTHER;
234 
235 		boolean isOptional = type.isOptional();
236 
237 		while (type != null && type.isOptional())
238 			type = (ClassMeta<T>)type.getElementType();
239 
240 		if (type == null)
241 			type = (ClassMeta<T>)object();
242 
243 		schema = Utils.firstNonNull(schema, getSchema(), DEFAULT_SCHEMA);
244 
245 		T t = parseInner(partType, schema, in, type);
246 		if (t == null && type.isPrimitive())
247 			t = type.getPrimitiveDefault();
248 		schema.validateOutput(t, ctx.getBeanContext());
249 
250 		if (isOptional)
251 			t = (T)Utils.opt(t);
252 
253 		return t;
254 	}
255 
256 	@Override /* ParserSession */
257 	protected <T> T doParse(ParserPipe pipe, ClassMeta<T> type) throws IOException, ParseException, ExecutableException {
258 		return parseInner(null, HttpPartSchema.DEFAULT, pipe.asString(), type);
259 	}

261 	@SuppressWarnings({ "unchecked" })
262 	private<T> T parseInner(HttpPartType partType, HttpPartSchema schema, String in, ClassMeta<T> type) throws SchemaValidationException, ParseException {
263 		schema.validateInput(in);
264 		if (in == null || "null".equals(in)) {
265 			if (schema.getDefault() == null)
266 				return null;
267 			in = schema.getDefault();
268 		} else {
269 
270 			ObjectSwap<T,Object> swap = (ObjectSwap<T,Object>)type.getSwap(this);
271 			BuilderSwap<T,Object> builder = (BuilderSwap<T,Object>)type.getBuilderSwap(this);
272 			ClassMeta<?> sType = null;
273 			if (builder != null)
274 				sType = builder.getBuilderClassMeta(this);
275 			else if (swap != null)
276 				sType = swap.getSwapClassMeta(this);
277 			else
278 				sType = type;
279 
280 			if (sType.isOptional())
281 				return (T)Utils.opt(parseInner(partType, schema, in, sType.getElementType()));
282 
283 			HttpPartDataType t = schema.getType(sType);
284 			if (partType == null)
285 				partType = HttpPartType.OTHER;
286 
287 			HttpPartFormat f = schema.getFormat(sType);
288 			if (f == HttpPartFormat.NO_FORMAT)
289 				f = ctx.getFormat();
290 
291 			if (t == STRING) {
292 				if (sType.isObject()) {
293 					if (f == BYTE)
294 						return toType(base64Decode(in), type);
295 					if (f == DATE || f == DATE_TIME)
296 						return toType(parseIsoCalendar(in), type);
297 					if (f == BINARY)
298 						return toType(fromHex(in), type);
299 					if (f == BINARY_SPACED)
300 						return toType(fromSpacedHex(in), type);
301 					if (f == HttpPartFormat.UON)
302 						return super.parse(partType, schema, in, type);
303 					return toType(in, type);
304 				}
305 				if (f == BYTE)
306 					return toType(base64Decode(in), type);
307 				if (f == DATE) {
308 					try {
309 						if (type.isCalendar())
310 							return toType(TemporalCalendarSwap.IsoDate.DEFAULT.unswap(this, in, type), type);
311 						if (type.isDate())
312 							return toType(TemporalDateSwap.IsoDate.DEFAULT.unswap(this, in, type), type);
313 						if (type.isTemporal())
314 							return toType(TemporalSwap.IsoDate.DEFAULT.unswap(this, in, type), type);
315 						return toType(in, type);
316 					} catch (Exception e) {
317 						throw new ParseException(e);
318 					}
319 				}
320 				if (f == DATE_TIME) {
321 					try {
322 						if (type.isCalendar())
323 							return toType(TemporalCalendarSwap.IsoDateTime.DEFAULT.unswap(this, in, type), type);
324 						if (type.isDate())
325 							return toType(TemporalDateSwap.IsoDateTime.DEFAULT.unswap(this, in, type), type);
326 						if (type.isTemporal())
327 							return toType(TemporalSwap.IsoDateTime.DEFAULT.unswap(this, in, type), type);
328 						return toType(in, type);
329 					} catch (Exception e) {
330 						throw new ParseException(e);
331 					}
332 				}
333 				if (f == BINARY)
334 					return toType(fromHex(in), type);
335 				if (f == BINARY_SPACED)
336 					return toType(fromSpacedHex(in), type);
337 				if (f == HttpPartFormat.UON)
338 					return super.parse(partType, schema, in, type);
339 				return toType(in, type);
340 
341 			} else if (t == BOOLEAN) {
342 				if (type.isObject())
343 					type = (ClassMeta<T>)CM_Boolean;
344 				if (type.isBoolean())
345 					return super.parse(partType, schema, in, type);
346 				return toType(super.parse(partType, schema, in, CM_Boolean), type);
347 
348 			} else if (t == INTEGER) {
349 				if (type.isObject()) {
350 					if (f == INT64)
351 						type = (ClassMeta<T>)CM_Long;
352 					else
353 						type = (ClassMeta<T>)CM_Integer;
354 				}
355 				if (type.isNumber())
356 					return super.parse(partType, schema, in, type);
357 				return toType(super.parse(partType, schema, in, CM_Integer), type);
358 
359 			} else if (t == NUMBER) {
360 				if (type.isObject()) {
361 					if (f == DOUBLE)
362 						type = (ClassMeta<T>)CM_Double;
363 					else
364 						type = (ClassMeta<T>)CM_Float;
365 				}
366 				if (type.isNumber())
367 					return super.parse(partType, schema, in, type);
368 				return toType(super.parse(partType, schema, in, CM_Double), type);
369 
370 			} else if (t == ARRAY) {
371 
372 				HttpPartCollectionFormat cf = schema.getCollectionFormat();
373 				if (cf == HttpPartCollectionFormat.NO_COLLECTION_FORMAT)
374 					cf = ctx.getCollectionFormat();
375 
376 				if (cf == HttpPartCollectionFormat.UONC)
377 					return super.parse(partType, schema, in, type);
378 
379 				if (type.isObject())
380 					type = (ClassMeta<T>)CM_JsonList;
381 
382 				ClassMeta<?> eType = type.isObject() ? string() : type.getElementType();
383 				if (eType == null)
384 					eType = schema.getParsedType().getElementType();
385 				if (eType == null)
386 					eType = string();
387 
388 				String[] ss = {};
389 
390 				if (cf == MULTI)
391 					ss = new String[]{in};
392 				else if (cf == CSV)
393 					ss = Utils.splita(in, ',');
394 				else if (cf == PIPES)
395 					ss = Utils.splita(in, '|');
396 				else if (cf == SSV)
397 					ss = Utils.splitQuoted(in);
398 				else if (cf == TSV)
399 					ss = Utils.splita(in, '\t');
400 				else if (cf == HttpPartCollectionFormat.UONC)
401 					return super.parse(partType, null, in, type);
402 				else if (cf == NO_COLLECTION_FORMAT) {
403 					if (firstNonWhitespaceChar(in) == '@' && lastNonWhitespaceChar(in) == ')')
404 						return super.parse(partType, null, in, type);
405 					ss = Utils.splita(in, ',');
406 				}
407 
408 				HttpPartSchema items = schema.getItems();
409 				if (items == null)
410 					items = HttpPartSchema.DEFAULT;
411 				Object o = Array.newInstance(eType.getInnerClass(), ss.length);
412 				for (int i = 0; i < ss.length; i++)
413 					Array.set(o, i, parse(partType, items, ss[i], eType));
414 				if (type.hasMutaterFrom(schema.getParsedType()) || schema.getParsedType().hasMutaterTo(type))
415 					return toType(toType(o, schema.getParsedType()), type);
416 				return toType(o, type);
417 
418 			} else if (t == OBJECT) {
419 
420 				HttpPartCollectionFormat cf = schema.getCollectionFormat();
421 				if (cf == HttpPartCollectionFormat.NO_COLLECTION_FORMAT)
422 					cf = ctx.getCollectionFormat();
423 
424 				if (cf == HttpPartCollectionFormat.UONC)
425 					return super.parse(partType, schema, in, type);
426 
427 				if (type.isObject())
428 					type = (ClassMeta<T>)CM_JsonMap;
429 
430 				if (! type.isMapOrBean())
431 					throw new ParseException("Invalid type {0} for part type OBJECT.", type);
432 
433 				String[] ss = {};
434 
435 				if (cf == MULTI)
436 					ss = new String[]{in};
437 				else if (cf == CSV)
438 					ss = Utils.splita(in, ',');
439 				else if (cf == PIPES)
440 					ss = Utils.splita(in, '|');
441 				else if (cf == SSV)
442 					ss = Utils.splitQuoted(in);
443 				else if (cf == TSV)
444 					ss = Utils.splita(in, '\t');
445 				else if (cf == HttpPartCollectionFormat.UONC)
446 					return super.parse(partType, null, in, type);
447 				else if (cf == NO_COLLECTION_FORMAT) {
448 					if (firstNonWhitespaceChar(in) == '@' && lastNonWhitespaceChar(in) == ')')
449 						return super.parse(partType, null, in, type);
450 					ss = Utils.splita(in, ',');
451 				}
452 
453 				if (type.isBean()) {
454 					BeanMap<T> m = ctx.getBeanContext().newBeanMap(type.getInnerClass());
455 					for (String s : ss) {
456 						String[] kv = Utils.splita(s, '=', 2);
457 						if (kv.length != 2)
458 							throw new ParseException("Invalid input {0} for part type OBJECT.", in);
459 						String key = kv[0], value = kv[1];
460 						BeanPropertyMeta bpm = m.getPropertyMeta(key);
461 						if (bpm == null && ! isIgnoreUnknownBeanProperties())
462 							throw new ParseException("Invalid input {0} for part type OBJECT.  Cannot find property {1}", in, key);
463 						m.put(key, parse(partType, schema.getProperty(key), value, ((ClassMeta<T>)(bpm == null ? object() : bpm.getClassMeta()))));
464 					}
465 					return m.getBean();
466 				}
467 
468 				ClassMeta<?> eType = type.isObject() ? string() : type.getValueType();
469 				if (eType == null)
470 					eType = schema.getParsedType().getValueType();
471 				if (eType == null)
472 					eType = string();
473 
474 				try {
475 					Map<String,Object> m = (Map<String,Object>)type.newInstance();
476 					if (m == null)
477 						m = JsonMap.create();
478 
479 					for (String s : ss) {
480 						String[] kv = Utils.splita(s, '=', 2);
481 						if (kv.length != 2)
482 							throw new ParseException("Invalid input {0} for part type OBJECT.", in);
483 						String key = kv[0], value = kv[1];
484 						m.put(key, parse(partType, schema.getProperty(key), value, eType));
485 					}
486 					return (T)m;
487 				} catch (ExecutableException e) {
488 					throw new ParseException(e);
489 				}
490 
491 			} else if (t == FILE) {
492 				throw new ParseException("File part not supported.");
493 
494 			} else if (t == NO_TYPE) {
495 				// This should never be returned by HttpPartSchema.getType(ClassMeta).
496 				throw new ParseException("Invalid type.");
497 			}
498 		}
499 
500 		return super.parse(partType, schema, in, type);
501 	}
502 
503 	private <T> T toType(Object in, ClassMeta<T> type) throws ParseException {
504 		try {
505 			return convertToType(in, type);
506 		} catch (InvalidDataConversionException e) {
507 			throw new ParseException(e.getMessage());
508 		}
509 	}
