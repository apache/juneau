245 	protected void doSerialize(SerializerPipe out, Object o) throws IOException, SerializeException {
246 		try {
247 			out.getWriter().write(serialize(HttpPartType.BODY, getSchema(), o));
248 		} catch (SchemaValidationException e) {

253 	@SuppressWarnings("rawtypes")
254 	@Override /* Overridden from PartSerializer */
255 	public String serialize(HttpPartType partType, HttpPartSchema schema, Object value) throws SerializeException, SchemaValidationException {
256 
257 		ClassMeta<?> type = getClassMetaForObject(value);
258 		if (type == null)
259 			type = object();
260 
261 		// Swap if necessary
262 		ObjectSwap swap = type.getSwap(this);
263 		if (swap != null && ! type.isDateOrCalendarOrTemporal()) {
264 			value = swap(swap, value);
265 			type = swap.getSwapClassMeta(this);
266 
267 			// If the getSwapClass() method returns Object, we need to figure out
268 			// the actual type now.
269 			if (type.isObject())
270 				type = getClassMetaForObject(value);
271 		}
272 
273 		schema = Utils.firstNonNull(schema, DEFAULT_SCHEMA);
274 
275 		HttpPartDataType t = schema.getType(type);
276 
277 		HttpPartFormat f = schema.getFormat(type);
278 		if (f == HttpPartFormat.NO_FORMAT)
279 			f = ctx.getFormat();
280 
281 		HttpPartCollectionFormat cf = schema.getCollectionFormat();
282 		if (cf == HttpPartCollectionFormat.NO_COLLECTION_FORMAT)
283 			cf = ctx.getCollectionFormat();
284 
285 		String out = null;
286 
287 		schema.validateOutput(value, ctx.getBeanContext());
288 
289 		if (type.hasMutaterTo(schema.getParsedType()) || schema.getParsedType().hasMutaterFrom(type)) {
290 			value = toType(value, schema.getParsedType());
291 			type = schema.getParsedType();
292 		}
293 
294 		if (type.isUri()) {
295 			value = getUriResolver().resolve(value);
296 			type = string();
297 		}
298 
299 		if (value != null) {
300 
301 			if (t == STRING) {
302 
303 				if (f == BYTE) {
304 					out = base64Encode(toType(value, CM_ByteArray));
305 				} else if (f == BINARY) {
306 					out = toHex(toType(value, CM_ByteArray));
307 				} else if (f == BINARY_SPACED) {
308 					out = toSpacedHex(toType(value, CM_ByteArray));
309 				} else if (f == DATE) {
310 					try {
311 						if (value instanceof Calendar)
312 							out = TemporalCalendarSwap.IsoDate.DEFAULT.swap(this, (Calendar)value);
313 						else if (value instanceof Date)
314 							out = TemporalDateSwap.IsoDate.DEFAULT.swap(this, (Date)value);
315 						else if (value instanceof Temporal)
316 							out = TemporalSwap.IsoDate.DEFAULT.swap(this, (Temporal)value);
317 						else
318 							out = value.toString();
319 					} catch (Exception e) {
320 						throw new SerializeException(e);
321 					}
322 				} else if (f == DATE_TIME) {
323 					try {
324 						if (value instanceof Calendar)
325 							out = TemporalCalendarSwap.IsoInstant.DEFAULT.swap(this, (Calendar)value);
326 						else if (value instanceof Date)
327 							out = TemporalDateSwap.IsoInstant.DEFAULT.swap(this, (Date)value);
328 						else if (value instanceof Temporal)
329 							out = TemporalSwap.IsoInstant.DEFAULT.swap(this, (Temporal)value);
330 						else
331 							out = value.toString();
332 					} catch (Exception e) {
333 						throw new SerializeException(e);
334 					}
335 				} else if (f == HttpPartFormat.UON) {
336 					out = super.serialize(partType, schema, value);
337 				} else {
338 					out = toType(value, string());
339 				}
340 
341 			} else if (t == BOOLEAN) {
342 
343 				out = Utils.s(toType(value, CM_Boolean));
344 
345 			} else if (t == INTEGER) {
346 
347 				if (f == INT64)
348 					out = Utils.s(toType(value, CM_Long));
349 				else
350 					out = Utils.s(toType(value, CM_Integer));
351 
352 			} else if (t == NUMBER) {
353 
354 				if (f == DOUBLE)
355 					out = Utils.s(toType(value, CM_Double));
356 				else
357 					out = Utils.s(toType(value, CM_Float));
358 
359 			} else if (t == ARRAY) {
360 
361 				if (cf == HttpPartCollectionFormat.UONC)
362 					out = super.serialize(partType, null, toList(partType, type, value, schema));
363 				else {
364 
365 					HttpPartSchema items = schema.getItems();
366 					ClassMeta<?> vt = getClassMetaForObject(value);
367 					OapiStringBuilder sb = new OapiStringBuilder(cf);
368 
369 					if (type.isArray()) {
370 						for (int i = 0; i < Array.getLength(value); i++)
371 							sb.append(serialize(partType, items, Array.get(value, i)));
372 					} else if (type.isCollection()) {
373 						((Collection<?>)value).forEach(x -> sb.append(serialize(partType, items, x)));
374 					} else if (vt.hasMutaterTo(String[].class)) {
375 						String[] ss = toType(value, CM_StringArray);
376 						for (String element : ss)
377 							sb.append(serialize(partType, items, element));
378 					} else {
379 						throw new SerializeException("Input is not a valid array type: " + type);
380 					}
381 
382 					out = sb.toString();
383 				}
384 
385 			} else if (t == OBJECT) {
386 
387 				if (cf == HttpPartCollectionFormat.UONC) {
388 					if (schema.hasProperties() && type.isMapOrBean())
389 						value = toMap(partType, type, value, schema);
390 					out = super.serialize(partType, null, value);
391 
392 				} else if (type.isBean()) {
393 					OapiStringBuilder sb = new OapiStringBuilder(cf);
394 					Predicate<Object> checkNull = x -> isKeepNullProperties() || x != null;
395 					HttpPartSchema schema2 = schema;
396 
397 					toBeanMap(value).forEachValue(checkNull, (pMeta,key,val,thrown) -> {
398 						if (thrown == null)
399 							sb.append(key, serialize(partType, schema2.getProperty(key), val));
400 					});
401 					out = sb.toString();
402 
403 				} else if (type.isMap()) {
404 					OapiStringBuilder sb = new OapiStringBuilder(cf);
405 					HttpPartSchema schema2 = schema;
406 					((Map<?,?>)value).forEach((k,v) -> sb.append(k, serialize(partType, schema2.getProperty(Utils.s(k)), v)));
407 					out = sb.toString();
408 
409 				} else {
410 					throw new SerializeException("Input is not a valid object type: " + type);
411 				}
412 
413 			} else if (t == FILE) {
414 				throw new SerializeException("File part not supported.");
415 
416 			} else if (t == NO_TYPE) {
417 				// This should never be returned by HttpPartSchema.getType(ClassMeta).
418 				throw new SerializeException("Invalid type.");
419 			}
420 		}
421 
422 		schema.validateInput(out);
423 		if (out == null)
424 			out = schema.getDefault();
425 		if (out == null)
426 			out = "null";
427 		return out;
428 	}

430 	private static class OapiStringBuilder {
431 		static final AsciiSet EQ = AsciiSet.of("=\\");
432 		static final AsciiSet PIPE = AsciiSet.of("|\\");
433 		static final AsciiSet PIPE_OR_EQ = AsciiSet.of("|=\\");
434 		static final AsciiSet COMMA = AsciiSet.of(",\\");
435 		static final AsciiSet COMMA_OR_EQ = AsciiSet.of(",=\\");
436 
437 		private final StringBuilder sb = new StringBuilder();
438 		private final HttpPartCollectionFormat cf;
439 		private boolean first = true;
440 
441 		OapiStringBuilder(HttpPartCollectionFormat cf) {
442 			this.cf = cf;
443 		}
444 
445 		private void delim(HttpPartCollectionFormat cf) {
446 			if (cf == PIPES)
447 				sb.append('|');
448 			else if (cf == SSV)
449 				sb.append(' ');
450 			else if (cf == TSV)
451 				sb.append('\t');
452 			else
453 				sb.append(',');
454 		}
455 
456 		OapiStringBuilder append(Object o) {
457 			if (! first)
458 				delim(cf);
459 			first = false;
460 			if (cf == PIPES)
461 				sb.append(escapeChars(Utils.s(o), PIPE));
462 			else if (cf == SSV || cf == TSV)
463 				sb.append(Utils.s(o));
464 			else
465 				sb.append(escapeChars(Utils.s(o), COMMA));
466 			return this;
467 		}
468 
469 		OapiStringBuilder append(Object key, Object val) {
470 			if (! first)
471 				delim(cf);
472 			first = false;
473 			if (cf == PIPES)
474 				sb.append(escapeChars(Utils.s(key), PIPE_OR_EQ)).append('=').append(escapeChars(Utils.s(val), PIPE_OR_EQ));
475 			else if (cf == SSV || cf == TSV)
476 				sb.append(escapeChars(Utils.s(key), EQ)).append('=').append(escapeChars(Utils.s(val), EQ));
477 			else
478 				sb.append(escapeChars(Utils.s(key), COMMA_OR_EQ)).append('=').append(escapeChars(Utils.s(val), COMMA_OR_EQ));
479 			return this;
480 		}
481 
482 		@Override
483 		public String toString() {
484 			return sb.toString();
485 		}
486 	}

488 	private Map<String,Object> toMap(HttpPartType partType, ClassMeta<?> type, Object o, HttpPartSchema s) throws SerializeException, SchemaValidationException {
489 		if (s == null)
490 			s = DEFAULT_SCHEMA;
491 		JsonMap m = new JsonMap();
492 		if (type.isBean()) {
493 			Predicate<Object> checkNull = x -> isKeepNullProperties() || x != null;
494 			HttpPartSchema s2 = s;
495 			toBeanMap(o).forEachValue(checkNull, (pMeta,key,val,thrown) -> {
496 				if (thrown == null)
497 					m.put(key, toObject(partType, val, s2.getProperty(key)));
498 			});
499 		} else {
500 			HttpPartSchema s2 = s;
501 			((Map<?,?>)o).forEach((k,v) -> m.put(Utils.s(k), toObject(partType, v, s2.getProperty(Utils.s(k)))));
502 		}
503 		if (isSortMaps())
504 			return sort(m);
505 		return m;
506 	}

508 	@SuppressWarnings("rawtypes")
509 	private List toList(HttpPartType partType, ClassMeta<?> type, Object o, HttpPartSchema s) throws SerializeException, SchemaValidationException {
510 		if (s == null)
511 			s = DEFAULT_SCHEMA;
512 		JsonList l = new JsonList();
513 		HttpPartSchema items = s.getItems();
514 		if (type.isArray()) {
515 			for (int i = 0; i < Array.getLength(o); i++)
516 				l.add(toObject(partType, Array.get(o, i), items));
517 		} else if (type.isCollection()) {
518 			((Collection<?>)o).forEach(x -> l.add(toObject(partType, x, items)));
519 		} else {
520 			l.add(toObject(partType, o, items));
521 		}
522 		if (isSortCollections())
523 			return sort(l);
524 		return l;
525 	}

527 	@SuppressWarnings("rawtypes")
528 	private Object toObject(HttpPartType partType, Object o, HttpPartSchema s) throws SerializeException, SchemaValidationException {
529 		if (o == null)
530 			return null;
531 		if (s == null)
532 			s = DEFAULT_SCHEMA;
533 		ClassMeta cm = getClassMetaForObject(o);
534 		HttpPartDataType t = s.getType(cm);
535 		HttpPartFormat f = s.getFormat(cm);
536 		HttpPartCollectionFormat cf = s.getCollectionFormat();
537 
538 		if (t == STRING) {
539 			if (f == BYTE)
540 				return base64Encode(toType(o, CM_ByteArray));
541 			if (f == BINARY)
542 				return toHex(toType(o, CM_ByteArray));
543 			if (f == BINARY_SPACED)
544 				return toSpacedHex(toType(o, CM_ByteArray));
545 			if (f == DATE)
546 				return toIsoDate(toType(o, CM_Calendar));
547 			if (f == DATE_TIME)
548 				return toIsoDateTime(toType(o, CM_Calendar));
549 			return o;
550 		} else if (t == ARRAY) {
551 			List l = toList(partType, getClassMetaForObject(o), o, s);
552 			if (cf == CSV)
553 				return StringUtils.joine(l, ',');
554 			if (cf == PIPES)
555 				return StringUtils.joine(l, '|');
556 			if (cf == SSV)
557 				return Utils.join(l, ' ');
558 			if (cf == TSV)
559 				return Utils.join(l, '\t');
560 			return l;
561 		} else if (t == OBJECT) {
562 			return toMap(partType, getClassMetaForObject(o), o, s);
563 		}
564 
565 		return o;
566 	}

568 	private <T> T toType(Object in, ClassMeta<T> type) throws SerializeException {
569 		try {
570 			return convertToType(in, type);
571 		} catch (InvalidDataConversionException e) {
572 			throw new SerializeException(e);
573 		}
