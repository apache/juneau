13  package org.apache.juneau.examples.rest;
14  
15  import org.apache.juneau.annotation.*;
16  import org.apache.juneau.html.annotation.*;
17  import org.apache.juneau.rest.annotation.*;
18  import org.apache.juneau.rest.beans.*;
19  import org.apache.juneau.rest.servlet.*;
20  import org.apache.juneau.rest.widget.*;
21  
22  /**
23   * Sample resource that allows images to be uploaded and retrieved.
24   *
25   * <h5 class='section'>See Also:</h5><ul>
26   * 	<li class='link'><a class="doclink" href="../../../../../index.html#jrs.Marshalling">REST Marshalling</a>
27   * 	<li class='link'><a class="doclink" href="../../../../../index.html#jrs.UtilityBeans">Utility Beans</a>
28   * </ul>
29   */
30  @Rest(
31  	path="/utilitybeans",
32  	title="Utility beans examples",
33  	description="Examples of utility bean usage."
34  )
35  @HtmlDocConfig(
36  	widgets={
37  		ContentTypeMenuItem.class
38  	},
39  	navlinks={
40  		"up: request:/..",
41  		"api: servlet:/api",
42  		"stats: servlet:/stats",
43  		"$W{ContentTypeMenuItem}",
44  		"source: $C{Source/gitHub}/org/apache/juneau/examples/rest/UtilityBeansResource.java"
45  	},
46  	aside={
47  		"<div class='text'>",
48  		"	<p>Examples of serialized beans in the org.apache.juneau.rest.utilitybeans package.</p>",
49  		"</div>"
50  	},
51  	asideFloat="RIGHT"
52  )
53  public class UtilityBeansResource extends BasicRestObject {
54  
55  	@SuppressWarnings("unused")
56  	private static final long serialVersionUID = 1L;
57  
58  	/**
59  	 * [HTTP GET /utilitybeans]
60  	 * @return Descriptive links to the child endpoints.
61  	 */
62  	@RestGet("/")
63  	public ResourceDescriptions getChildDescriptions() {
64  		return ResourceDescriptions
65  			.create()
66  			.append("BeanDescription", "Example of BeanDescription bean")
67  			.append("Hyperlink", "Example of Hyperlink bean")
68  			.append("SeeOtherRoot", "Example of SeeOtherRoot bean");
69  	}
70  
71  	/**
72  	 * [HTTP GET /utilitybeans/BeanDescription]
73  	 * @return Example of serialized org.apache.juneau.rest.utilitybeans.ResourceDescriptions bean.
74  	 */
75  	@RestGet("/BeanDescription")
76  	@HtmlDocConfig(
77  		aside={
78  			"<div class='text'>",
79  			"	<p>Example of serialized ResourceDescriptions bean.</p>",
80  			"</div>"
81  		}
82  	)
83  	public BeanDescription aBeanDescription() {
84  		return BeanDescription.of(Address.class);
85  	}
86  
87  	/**
88  	 * Sample address bean used for demonstrating utility bean functionality.
89  	 */
90  	@Bean(p="street,city,state,zip,isCurrent")
91  	public static class Address {
92  
93  		/** Street address. */
94  		public String street;
95  
96  		/** City name. */
97  		public String city;
98  
99  		/** State abbreviation. */
100 		public String state;
101 
102 		/** ZIP code. */
103 		public int zip;
104 
105 		/** Whether this is the current address. */
106 		public boolean isCurrent;
107 
108 		/** Default constructor. */
109 		public Address() {}
110 	}
111 
112 	/**
113 	 * [HTTP GET /utilitybeans/Hyperlink]
114 	 * @return Example of serialized org.apache.juneau.rest.utilitybeans.Hyperlink bean.
115 	 */
116 	@RestGet("/Hyperlink")
117 	@HtmlDocConfig(
118 		aside={
119 			"<div class='text'>",
120 			"	<p>Example of serialized Hyperlink bean.</p>",
121 			"</div>"
122 		}
123 	)
124 	public Hyperlink aHyperlink() {
125 		return Hyperlink.create("/utilitybeans", "Back to /utilitybeans");
126 	}
127 
128 	/**
129 	 * [HTTP GET /utilitybeans/SeeOtherRoot]
130 	 * @return Example of serialized SeeOtherRoot bean.
131 	 * This just redirects back to the servlet root.
132 	 */
133 	@RestGet("/SeeOtherRoot")
134 	@HtmlDocConfig(
135 		aside={
136 			"<div class='text'>",
137 			"	<p>Example of serialized org.apache.juneau.rest.utilitybeans.SeeOtherRoot bean.</p>",
138 			"</div>"
139 		}
140 	)
141 	public SeeOtherRoot aSeeOtherRoot() {
142 		return SeeOtherRoot.INSTANCE;
143 	}
144 }
