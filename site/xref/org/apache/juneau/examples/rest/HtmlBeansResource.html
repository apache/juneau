17  package org.apache.juneau.examples.rest;
18  
19  import static org.apache.juneau.bean.html5.HtmlBuilder.*;
20  
21  import org.apache.juneau.bean.html5.*;
22  import org.apache.juneau.html.annotation.*;
23  import org.apache.juneau.rest.annotation.*;
24  import org.apache.juneau.rest.beans.*;
25  import org.apache.juneau.rest.servlet.*;
26  import org.apache.juneau.rest.widget.*;
27  
28  /**
29   * Sample resource that allows images to be uploaded and retrieved.
30   *
31   * <h5 class='section'>See Also:</h5><ul>
32   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/Marshalling">Marshalling</a>
33   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HtmlBeans">Using with HTML Beans</a>
34   * </ul>
35   */
36  @Rest(
37  	path="/htmlbeans",
38  	title="HTML bean examples",
39  	description="Examples of serialized HTML beans."
40  )
41  @HtmlDocConfig(
42  	widgets={
43  		ContentTypeMenuItem.class
44  	},
45  	navlinks={
46  		"up: request:/..",
47  		"api: servlet:/api",
48  		"stats: servlet:/stats",
49  		"$W{ContentTypeMenuItem}",
50  		"source: $C{Source/gitHub}/org/apache/juneau/examples/rest/HtmlBeansResource.java"
51  	},
52  	aside={
53  		"<div class='text'>",
54  		"	<p>Examples of serialized HTML beans.</p>",
55  		"</div>"
56  	},
57  	asideFloat="RIGHT"
58  )
59  public class HtmlBeansResource extends BasicRestObject {
60  
61  	@SuppressWarnings("unused")
62  	private static final long serialVersionUID = 1L;
63  
64  	/**
65  	 * [HTTP GET /htmlbeans/div]
66  	 * @return An example div tag.
67  	 */
68  	@RestGet("/div")
69  	@HtmlDocConfig(
70  		aside={
71  			"<div class='text'>",
72  			"	<p>Example of serialized div tag.</p>",
73  			"</div>"
74  		}
75  	)
76  	public HtmlElement aDiv() {
77  		return div()
78  			.children(
79  				p("Juneau supports ", b(i("mixed")), " content!")
80  			)
81  			.onmouseover("alert(\"boo!\");");
82  	}
83  
84  	/**
85  	 * [HTTP GET /htmlbeans/form]
86  	 * @return An example form tag.
87  	 */
88  	@RestGet("/form")
89  	@HtmlDocConfig(
90  		aside={
91  			"<div class='text'>",
92  			"	<p>Example of serialized HTML form.</p>",
93  			"</div>"
94  		}
95  	)
96  	public Form aForm() {
97  		return form().action("/submit").method("POST")
98  			.children(
99  				"Position (1-10000): ", input("number").name("pos").value(1), br(),
100 				"Limit (1-10000): ", input("number").name("limit").value(100), br(),
101 				button("submit", "Submit"),
102 				button("reset", "Reset")
103 			);
104 	}
105 
106 	/**
107 	 * [HTTP GET /htmlbeans/table]
108 	 * @return An example table.
109 	 */
110 	@RestGet("/table")
111 	@HtmlDocConfig(
112 		aside={
113 			"<div class='text'>",
114 			"	<p>Example of serialized table.</p>",
115 			"</div>"
116 		}
117 	)
118 	public Table aTable() {
119 		return table(
120 			tr(
121 				th("c1"),
122 				th("c2")
123 			),
124 			tr(
125 				td("v1"),
126 				td("v2")
127 			)
128 		);
129 	}
130 
131 	/**
132 	 * [HTTP GET /htmlbeans]
133 	 * @return Descriptive links to the child endpoints.
134 	 */
135 	@RestGet("/")
136 	public ResourceDescriptions getChildDescriptions() {
137 		return ResourceDescriptions
138 			.create()
139 			.append("table", "Example of a serialized table")
140 			.append("div", "Example of a serialized div tag")
141 			.append("form", "Example of a serialized form");
142 	}
143 }
