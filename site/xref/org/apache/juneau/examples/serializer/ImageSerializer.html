17  package org.apache.juneau.examples.serializer;
18  
19  import java.awt.image.*;
20  import java.io.*;
21  
22  import javax.imageio.*;
23  
24  import org.apache.juneau.*;
25  import org.apache.juneau.serializer.*;
26  
27  /**
28   * Example serializer that converts {@link BufferedImage} objects to byte streams.
29   *
30   * <h5 class='section'>See Also:</h5><ul>
31   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/Marshalling">Marshalling</a>
32   * </ul>
33   */
34  @SuppressWarnings("javadoc")
35  public class ImageSerializer extends OutputStreamSerializer {
36  
37  	public ImageSerializer() {
38  		super(create().produces("image/png,image/jpeg"));
39  	}
40  
41  	@Override
42  	public void doSerialize(SerializerSession session, SerializerPipe pipe, Object o) throws IOException, SerializeException {
43  		RenderedImage image = (RenderedImage)o;
44  		MediaType mediaType = session.getMediaType();
45  		try (OutputStream os = pipe.getOutputStream()) {
46  			ImageIO.write(image, mediaType.getType(), os);
47  		}
48  	}
49  }
