13  package org.apache.juneau.examples.bean;
14  
15  import static org.apache.juneau.bean.atom.AtomBuilder.*;
16  import static org.apache.juneau.bean.atom.AtomBuilder.link;
17  import static org.apache.juneau.bean.html5.HtmlBuilder.*;
18  import static org.apache.juneau.bean.swagger.SwaggerBuilder.*;
19  
20  import java.net.*;
21  
22  import org.apache.juneau.*;
23  import org.apache.juneau.html.*;
24  import org.apache.juneau.json.*;
25  
26  /**
27   * Sample class which shows the usage of DTO module which is a
28   * Sub module of the core.
29   *
30   * <h5 class='section'>See Also:</h5><ul>
31   * </ul>
32   */
33  public class BeanExample {
34  
35  	/**
36  	 * DTO Samples
37  	 *
38  	 * @param args Unused.
39  	 * @throws Exception Unused.
40  	 */
41  	@SuppressWarnings("unused")
42  	public static void main(String[] args) throws Exception {
43  
44  		//Produces
45  		/**
46  		 * <table>
47  		 * <tr>
48  		 * <th>c1</th>
49  		 * <th>c2</th>
50  		 * </tr>
51  		 * <tr>
52  		 * <td>v1</td>
53  		 * <td>v2</td>
54  		 * </tr>
55  		 * </table>
56  		 */
57  		var mytable =
58  			table(
59  				tr(
60  					th("c1"),
61  					th("c2")
62  				),
63  				tr(
64  					td("v1"),
65  					td("v2")
66  				)
67  			);
68  
69  		var html = HtmlSerializer.DEFAULT.serialize(mytable);
70  
71  		var mainJsp =
72  			form().action("main.jsp").method("GET")
73  			.children(
74  				input("text").name("first_name").value("apache"), br(),
75  				input("text").name("last_name").value("juneau"), br(),
76  				button("submit", "Submit"),
77  				button("reset", "Reset")
78  			);
79  
80  		/**
81  		 * <form action='main.jsp' method='POST'>
82  		 * Position (1-10000): <input name='pos' type='number'
83  		 * value='1'/><br/>
84  		 * Limit (1-10000): <input name='pos' type='number'
85  		 * value='100'/><br/>
86  		 * <button type='submit'>Submit</button>
87  		 * <button type='reset'>Reset</button>
88  		 * </form>
89  		 */
90  		html = HtmlSerializer.DEFAULT.serialize(mainJsp);
91  
92  		/**
93  		 * Produces
94  		 * {
95  		 *    a:{action:'main.jsp',method:'GET'},
96  		 *    c:[
97  		 *    {a:{type:'text',name:'first_name',value:'apache'}},{},
98  		 *    {a:{type:'text',name:'last_name',value:'juneau'}},{},
99  		 *    {a:{type:'submit'},c:['Submit']},
100 		 *    {a:{type:'reset'},c:['Reset']}
101 		 *    ]
102 		 * }
103 		 */
104 		html =  Json5Serializer.DEFAULT.serialize(mainJsp);
105 
106 		var feed =
107 			feed("tag:foo.org", "Title", "2016-12-31T05:02:03Z")
108 			.setSubtitle(text("html").setText("Subtitle"))
109 			.setLinks(
110 				link("alternate", "text/html", "http://foo.org/").setHreflang("en"),
111 				link("self", "application/atom+xml", "http://foo.org/feed.atom")
112 			)
113 			.setGenerator(
114 				generator("Example Toolkit").setUri("http://www.foo.org/").setVersion("1.0")
115 			)
116 			.setEntries(
117 				entry("tag:foo.org", "Title", "2016-12-31T05:02:03Z")
118 				.setLinks(
119 					link("alternate", "text/html", "http://foo.org/2005/04/02/atom"),
120 					link("enclosure", "audio/mpeg", "http://foo.org/audio/foobar.mp3").setLength(1337)
121 				)
122 				.setPublished("2016-12-31T05:02:03Z")
123 				.setAuthors(
124 					person("John Smith").setUri(new URI("http://foo.org/")).setEmail("foo@foo.org")
125 				)
126 				.setContributors(
127 					person("John Smith"),
128 					person("Jane Smith")
129 				)
130 				.setContent(
131 					content("xhtml")
132 					.setLang("en")
133 					.setBase("http://foo.org/")
134 					.setText("<div><p><i>[Sample content]</i></p></div>")
135 				)
136 			);
137 
138 		var swagger = swagger()
139 			.setSwagger("2.0")
140 			.setInfo(
141 				info("Swagger Petstore", "1.0.0")
142 				.setDescription("This is a sample server Petstore server.")
143 				.setTermsOfService("http://swagger.io/terms/")
144 				.setContact(
145 					contact().setEmail("apiteam@swagger.io")
146 				)
147 				.setLicense(
148 					license("Apache 2.0").setUrl(URI.create("http://www.apache.org/licenses/LICENSE-2.0.html"))
149 				)
150 			)
151 			.addPath("/pet", "post",
152 				operation()
153 				.setTags("pet")
154 				.setSummary("Add a new pet to the store")
155 				.setDescription("")
156 				.setOperationId("addPet")
157 				.setConsumes(MediaType.JSON, MediaType.XML)
158 				.setProduces(MediaType.JSON, MediaType.XML)
159 				.setParameters(
160 					parameterInfo("body", "body")
161 					.setDescription("Pet object that needs to be added to the store")
162 					.setRequired(true)
163 				)
164 				.addResponse("405", responseInfo("Invalid input"))
165 			);
166 
167 		// Serialize to Swagger/JSON
168 		/**
169 		 * Produces
170 		 * {
171 		 *  "swagger": "2.0",
172 		 *  "info": {
173 		 *      "title": "Swagger Petstore",
174 		 *      "description": "This is a sample server Petstore server.",
175 		 *      "version": "1.0.0",
176 		 *      "termsOfService": "http://swagger.io/terms/",
177 		 *      "contact": {
178 		 *          "email": "apiteam@swagger.io"
179 		 *      },
180 		 *      "license": {
181 		 *          "name": "Apache 2.0",
182 		 *          "url": "http://www.apache.org/licenses/LICENSE-2.0.html"
183 		 *      }
184 		 *  },
185 		 * "paths": {
186 		 *      "/pet": {
187 		 *          "post": {
188 		 *              "tags": [
189 		 *                  "pet"
190 		 *               ],
191 		 *              "summary": "Add a new pet to the store",
192 		 *              "description": "",
193 		 *              "operationId": "addPet",
194 		 *              "consumes": [
195 		 *                  "application/json",
196 		 *                  "text/xml"
197 		 *              ],
198 		 *              "produces": [
199 		 *                  "application/json",
200 		 *                  "text/xml"
201 		 *              ],
202 		 *              "parameters": [
203 		 *                  {
204 		 *                      "in": "body",
205 		 *                      "name": "body",
206 		 *                      "description": "Pet object that needs to be added to the store",
207 		 *                      "required": true
208 		 *                  }
209 		 *              ],
210 		 *              "responses": {
211 		 *                  "405": {
212 		 *                      "description": "Invalid input"
213 		 *                  }
214 		 *              }
215 		 *         }
216 		 *      }
217 		 *  },
218 		 *  }
219 		 */
220 		var swaggerJson = JsonSerializer.DEFAULT_READABLE.serialize(swagger);
221 	}
