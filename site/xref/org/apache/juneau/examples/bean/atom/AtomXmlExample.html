17  package org.apache.juneau.examples.bean.atom;
18  
19  import org.apache.juneau.xml.*;
20  
21  /**
22   * Atom feed XML example.
23   *
24   */
25  public class AtomXmlExample {
26  
27  	/**
28  	 * XML Atom feed example.
29  	 *
30  	 * @param args Unused.
31  	 * @throws Exception Unused.
32  	 */
33  	@SuppressWarnings("unused")
34  	public static void main(String[] args) throws Exception {
35  
36  		var feed = AtomFeed.getAtomFeed();
37  
38  		// Example with no namespaces
39  		// Create a serializer with readable output, no namespaces yet.
40  		var s = XmlSerializer.create().sq().ws().build();
41  
42  		//Produces
43  		/**
44  		 *<feed>
45  		 *<id> tag:juneau.apache.org</id>
46  		 *<link href='http://juneau.apache.org/' rel='alternate' type='text/html' hreflang='en'/>
47  		 *<link href='http://juneau.apache.org/feed.atom' rel='self' type='application/atom+xml'/>
48  		 *<title type='text'>Juneau ATOM specification</title>
49  		 *<updated>2016-01-02T03:04:05Z</updated>
50  		 *<generator uri='http://juneau.apache.org/' version='1.0'>Juneau</generator>
51  		 *<subtitle type='html'>Describes <em>stuff</em> about Juneau</subtitle>
52  		 *<entry>
53  		 *  <author>
54  		 *      <name>Jane Smith</name>
55  		 *      <uri>http://juneau.apache.org/</uri>
56  		 *      <email>janesmith@apache.org</email>
57  		 *  </author>
58  		 *  <contributor>
59  		 *      <name>John Smith</name>
60  		 *  </contributor>
61  		 *  <id>tag:juneau.apache.org</id>
62  		 *  <link href='http://juneau.apache.org/juneau.atom' rel='alternate' type='text/html'/>
63  		 *  <link href='http://juneau.apache.org/audio/juneau_podcast.mp3' rel='enclosure' type='audio/mpeg' length='12345'/>
64  		 *  <title>Juneau ATOM specification snapshot</title>
65  		 *  <updated>2016-01-02T03:04:05Z</updated>
66  		 *  <content base='http://www.apache.org/' lang='en' type='xhtml'>
67  		 *      <div xmlns="http://www.w3.org/1999/xhtml"><p><i>[Update: Juneau supports ATOM.]</i></p></div>
68  		 *  </content>
69  		 *  <published>2016-01-02T03:04:05Z</published>
70  		 *</entry>
71  		 *</feed>
72  		 */
73  		//Serialize to ATOM/XML
74  		var atomXml = s.serialize(feed);
75  
76  		/**
77  		 * Produces
78  		 * <feed>
79  		 * <link hreflang='en' rel='alternate' href='http://juneau.apache.org' type='text/html'/>
80  		 * <link rel='self' href='http://juneau.apache.org/feed.atom' type='application/atom+xml'/>
81  		 * <title>Juneau ATOM specification</title>
82  		 * <updated>2016-01-02T03:04:05Z</updated>
83  		 * <id>tag:juneau.apache.org</id>
84  		 * <subtitle type='html'>Describes <em>stuff</em> about Juneau</subtitle>
85  		 * <generator version='1.0' uri='http://juneau.apache.org'>Juneau</generator>
86  		 * <entry>
87  		 *  <link rel='alternate' href='http://juneau.apache.org/juneau.atom' type='text/html'/>
88  		 *  <link rel='enclosure' href='http://juneau.apache.org/audio/juneau_podcast.mp3' type='audio/mpeg' length='1337'/>
89  		 *  <author>
90  		 *      <uri>http://juneau.apache.org</uri>
91  		 *      <email>janesmith@apache.org</email>
92  		 *      <name>Jane Smith</name>
93  		 *  </author>
94  		 *  <contributor>
95  		 *      <name>John Smith</name>
96  		 *  </contributor>
97  		 *  <title>Juneau ATOM specification snapshot</title>
98  		 *  <updated>2016-01-02T03:04:05Z</updated>
99  		 *  <id>tag:juneau.sample.com,2013:1.2345</id>
100 		 *  <published>2016-01-02T03:04:05Z</published>
101 		 *  <content lang='en' base='http://www.apache.org/' type='xhtml'><div><p><i>[Update: Juneau supports ATOM.]</i></p></div></content>
102 		 * </entry>
103 		 * </feed>
104 		 */
105 		// Create a serializer with readable output, no namespaces yet.
106 		var ns = XmlSerializer.create().sq().ws().build();
107 
108 		// Serialize to ATOM/XML
109 		atomXml = ns.serialize(feed);
110 	}
111 }
