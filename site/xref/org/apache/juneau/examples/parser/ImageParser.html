13  package org.apache.juneau.examples.parser;
14  
15  import java.awt.image.*;
16  import java.io.*;
17  
18  import javax.imageio.*;
19  
20  import org.apache.juneau.*;
21  import org.apache.juneau.parser.*;
22  
23  /**
24   * Example parser that converts byte streams to {@link BufferedImage} objects.
25   *
26   * <h5 class='section'>See Also:</h5><ul>
27   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/Marshalling">Marshalling</a>
28   * </ul>
29   */
30  @SuppressWarnings("javadoc")
31  public class ImageParser extends InputStreamParser {
32  
33  	public ImageParser() {
34  		super(create().consumes("image/png,image/jpg"));
35  	}
36  
37  	@Override /* Parser */
38  	@SuppressWarnings("unchecked")
39  	public <T> T doParse(ParserSession session, ParserPipe pipe, ClassMeta<T> type) throws IOException, ParseException {
40  		try (InputStream is = pipe.getInputStream()) {
41  			BufferedImage image = ImageIO.read(is);
42  			return (T)image;
43  		}
44  	}
45  }
