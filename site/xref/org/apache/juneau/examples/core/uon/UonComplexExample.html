13  package org.apache.juneau.examples.core.uon;
14  
15  import org.apache.juneau.examples.core.pojo.Pojo;
16  import org.apache.juneau.examples.core.pojo.PojoComplex;
17  import org.apache.juneau.uon.UonParser;
18  import org.apache.juneau.uon.UonSerializer;
19  
20  import java.util.ArrayList;
21  import java.util.HashMap;
22  import java.util.List;
23  
24  /**
25   * UON complex example.
26   *
27   * <h5 class='section'>See Also:</h5><ul>
28   * </ul>
29   */
30  public class UonComplexExample {
31  	/**
32  	 * Serializing PojoComplex bean into UON format.
33  	 *
34  	 * @param args Unused.
35  	 * @throws Exception Unused.
36  	 */
37  	public static void main(String[] args) throws Exception {
38  
39  		// Fill some data to a PojoComplex bean
40  		HashMap<String, List<Pojo>> values = new HashMap<>();
41  		ArrayList<Pojo> setOne = new ArrayList<>();
42  		setOne.add(new Pojo("1.1", "name1"));
43  		setOne.add(new Pojo("1.1", "name2"));
44  		ArrayList<Pojo> setTwo = new ArrayList<>();
45  		setTwo.add(new Pojo("1.2", "name1"));
46  		setTwo.add(new Pojo("1.2", "name2"));
47  		values.put("setOne", setOne);
48  		values.put("setTwo", setTwo);
49  		PojoComplex pojoc = new PojoComplex("pojo", new Pojo("1.0", "name0"), values);
50  
51  		// this creates an RDF serializer with the default XML structure
52  		/**Produces
53  		 * (innerPojo=(name=name0,id='1.0'),
54  		 * values=(setOne=@((name=name1,id='1.1'),(name=name2,id='1.1')),
55  		 * setTwo=@((name=name1,id='1.2'),(name=name2,id='1.2'))),id=pojo)
56  		 */
57  		UonSerializer uonSerializer = UonSerializer.DEFAULT;
58  		// This will show the final output from the bean
59  		System.out.println(uonSerializer.serialize(pojoc));
60  
61  		PojoComplex obj = UonParser.DEFAULT.parse(uonSerializer.serialize(pojoc), PojoComplex.class);
62  
63  		assert obj.getId().equals(pojoc.getId());
64  
65  	}
66  }
