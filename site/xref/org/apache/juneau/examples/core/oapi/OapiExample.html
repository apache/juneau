17  package org.apache.juneau.examples.core.oapi;
18  
19  import static org.apache.juneau.httppart.HttpPartType.*;
20  
21  import org.apache.juneau.examples.core.pojo.*;
22  import org.apache.juneau.httppart.*;
23  import org.apache.juneau.oapi.*;
24  
25  /**
26   * Sample class which shows the simple usage of OpenApiSerializer.
27   *
28   * <h5 class='section'>See Also:</h5><ul>
29   * </ul>
30   */
31  public class OapiExample {
32  
33  	/**
34  	 * Get a reference to a parser and usage of oapiserializer.
35  	 *
36  	 * @param args Unused.
37  	 * @throws Exception Unused.
38  	 */
39  	@SuppressWarnings("unused")
40  	public static void main(String[] args) throws Exception {
41  
42  		OpenApiSerializer oapiSerializer = OpenApiSerializer.DEFAULT;
43  
44  		OpenApiParser oapiParser = OpenApiParser.DEFAULT;
45  
46  		Pojo pojo = new Pojo("id", "name");
47  
48  		String flat = oapiSerializer.serialize(pojo);
49  		// Print out the created POJO in OpenAPI format.
50  
51  		Pojo parse = oapiParser.parse(flat, Pojo.class);
52  
53  		assert parse.getId().equals(pojo.getId());
54  		assert parse.getName().equals(pojo.getName());
55  
56  		//Http part schmea
57  		HttpPartSchema schema = HttpPartSchema
58  			.create("array")
59  			.collectionFormat("pipes")
60  			.items(
61  				HttpPartSchema
62  				.create("array")
63  				.collectionFormat("csv")
64  				.items(
65  					HttpPartSchema.create("integer","int64")
66  				)
67  			)
68  			.build();
69  		Object value = new long[][] { { 1, 2, 3 }, { 4, 5, 6 }, { 7, 8, 9 } };
70  		String output = OpenApiSerializer.DEFAULT.serialize(HEADER, schema, value);
71  
72  		HttpPartSchema schemab = HttpPartSchema.create().type("string").build();
73  		// Convert POJO to BASE64-encoded string.
74  		HttpPartSerializer s = OpenApiSerializer.DEFAULT;
75  		String httpPart = s.getPartSession().serialize(HEADER, schemab, pojo);
76  		System.out.println(httpPart);
77  
78  		// Convert BASE64-encoded string back into a POJO.
79  		HttpPartParser p = OpenApiParser.DEFAULT;
80  		pojo = p.getPartSession().parse(HEADER, schemab, httpPart, oapiParser.getClassMeta(Pojo.class));
81  
82  		// The object above can be parsed thanks to the @Beanc(properties = id,name) annotation on Pojo
83  		// Using this approach, you can keep your POJOs immutable, and still serialize and deserialize them.
84  	}
85  }
