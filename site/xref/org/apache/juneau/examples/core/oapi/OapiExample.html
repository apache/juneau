17  package org.apache.juneau.examples.core.oapi;
18  
19  import static org.apache.juneau.httppart.HttpPartType.*;
20  
21  import org.apache.juneau.examples.core.pojo.*;
22  import org.apache.juneau.httppart.*;
23  import org.apache.juneau.oapi.*;
24  
25  /**
26   * Sample class which shows the simple usage of OpenApiSerializer.
27   *
28   */
29  public class OapiExample {
30  
31  	/**
32  	 * Get a reference to a parser and usage of oapiserializer.
33  	 *
34  	 * @param args Unused.
35  	 * @throws Exception Unused.
36  	 */
37  	@SuppressWarnings("unused")
38  	public static void main(String[] args) throws Exception {
39  
40  		var oapiSerializer = OpenApiSerializer.DEFAULT;
41  
42  		var oapiParser = OpenApiParser.DEFAULT;
43  
44  		var pojo = new Pojo("id", "name");
45  
46  		var flat = oapiSerializer.serialize(pojo);
47  		// Print out the created POJO in OpenAPI format.
48  
49  		var parse = oapiParser.parse(flat, Pojo.class);
50  
51  		assert parse.getId().equals(pojo.getId());
52  		assert parse.getName().equals(pojo.getName());
53  
54  		//Http part schmea
55  		var schema = HttpPartSchema
56  			.create("array")
57  			.collectionFormat("pipes")
58  			.items(
59  				HttpPartSchema
60  				.create("array")
61  				.collectionFormat("csv")
62  				.items(
63  					HttpPartSchema.create("integer","int64")
64  				)
65  			)
66  			.build();
67  		var value = new long[][] { { 1, 2, 3 }, { 4, 5, 6 }, { 7, 8, 9 } };
68  		var output = OpenApiSerializer.DEFAULT.serialize(HEADER, schema, value);
69  
70  		var schemab = HttpPartSchema.create().type("string").build();
71  		// Convert POJO to BASE64-encoded string.
72  		var s = OpenApiSerializer.DEFAULT;
73  		var httpPart = s.getPartSession().serialize(HEADER, schemab, pojo);
74  		System.out.println(httpPart);
75  
76  		// Convert BASE64-encoded string back into a POJO.
77  		var p = OpenApiParser.DEFAULT;
78  		pojo = p.getPartSession().parse(HEADER, schemab, httpPart, oapiParser.getClassMeta(Pojo.class));
79  
80  		// The object above can be parsed thanks to the @Beanc(properties = id,name) annotation on Pojo
81  		// Using this approach, you can keep your POJOs immutable, and still serialize and deserialize them.
82  	}
83  }
