140 	private final String jdbcUrl;
141 	private final String tableName, nameColumn, valueColumn;
142 	private final Timer watcher;
143 	private final ConcurrentHashMap<String,String> cache = new ConcurrentHashMap<>();
144 
145 	/**
146 	 * Constructor.
147 	 *
148 	 * @param builder The builder for this object.
149 	 */
150 	protected SqlStore(Builder builder) {
151 		super(builder);
152 		this.jdbcUrl = builder.jdbcUrl;
153 		this.tableName = builder.tableName;
154 		this.nameColumn = builder.nameColumn;
155 		this.valueColumn = builder.valueColumn;
156 
157 		var pollInterval = builder.pollInterval;
158 
159 		var timerTask = new TimerTask() {
160 			@Override
161 			public void run() {
162 				SqlStore.this.poll();
163 			}
164 		};
165 
166 		this.watcher = new Timer("MyTimer");
167 		watcher.scheduleAtFixedRate(timerTask, 0, pollInterval * 1000);
168 	}
169 
170 	@Override /* Closeable */
171 	public synchronized void close() {
172 		if (watcher != null)
173 			watcher.cancel();
174 	}
175 
176 	@Override /* ConfigStore */
177 	public boolean exists(String name) {
178 		// Implement me!
179 		return false;
180 	}
181 
182 	@Override /* ConfigStore */
183 	public synchronized String read(String name) {
184 		var contents = cache.get(name);
185 		if (contents == null) {
186 			contents = getDatabaseValue(name);
187 			update(name, contents);
188 		}
189 		return contents;
190 	}
191 
192 	@Override /* ConfigStore */
193 	public synchronized SqlStore update(String name, String newContents) {
194 		cache.put(name, newContents);
195 		super.update(name, newContents);  // Trigger any listeners.
196 		return this;
197 	}
198 
199 	@Override /* ConfigStore */
200 	public synchronized String write(String name, String expectedContents, String newContents) {
201 
202 		// This is a no-op.
203 		if (Utils.eq(expectedContents, newContents))
204 			return null;
205 
206 		var currentContents = read(name);
207 
208 		if (expectedContents != null && Utils.ne(currentContents, expectedContents))
209 			return currentContents;
210 
211 		update(name, newContents);
212 
213 		// Success!
214 		return null;
215 	}
216 
217 	// Reads the value from the database.
218 	protected String getDatabaseValue(String name) {
219 		// Implement me!
220 		return null;
221 	}
222 
223 	synchronized void poll() {
224 
225 		// Loop through all our entries and find the latest values.
226 		cache.forEach((name, cacheContents) -> {
227 			var newContents = getDatabaseValue(name);
228 
229 			// Change detected!
230 			if (! cacheContents.equals(newContents))
231 				update(name, newContents);
232 		});
233 	}
