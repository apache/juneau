17  package org.apache.juneau.examples.core.html;
18  
19  import java.util.*;
20  
21  import org.apache.juneau.examples.core.pojo.*;
22  import org.apache.juneau.html.*;
23  
24  /**
25   * Sample class which shows the complex usage of HtmlSerializer and HtmlParser.
26   *
27   * <h5 class='section'>See Also:</h5><ul>
28   * </ul>
29   */
30  public class HtmlComplexExample {
31  
32  	/**
33  	 * Serializing PojoComplex bean into Html type
34  	 * and Deserialize back to PojoComplex instance type.
35  	 *
36  	 * @param args Unused.
37  	 * @throws Exception Unused.
38  	 */
39  	public static void main(String[] args) throws Exception {
40  		// Juneau provides static constants with the most commonly used configurations
41  		// Get a reference to a serializer - converting POJO to flat format
42  		/**
43  		 * Produces
44  		 * <table><tr><td>innerPojo</td><td><table><tr><td>name</td><td>name0</td></tr>
45  		 * <tr><td>id</td><td>1.0</td></tr></table></td></tr><tr><td>values</td><td><table>
46  		 * <tr><td>setOne</td><td><table _type="array"><tr><th>name</th><th>id</th></tr>
47  		 * <tr><td>name1</td><td>1.1</td></tr><tr><td>name2</td><td>1.1</td></tr>
48  		 * </table></td></tr><tr><td>setTwo</td><td><table _type="array"><tr><th>name
49  		 * </th><th>id</th></tr><tr><td>name1</td><td>1.2</td></tr><tr><td>name2</td><td>1.2
50  		 * </td></tr></table></td></tr></table></td></tr><tr><td>id</td><td>pojo</td></tr></table>
51  		 */
52  		var htmlSerializer = HtmlSerializer.DEFAULT;
53  		// Get a reference to a parser - converts that flat format back into the POJO
54  		var htmlParser = HtmlParser.DEFAULT;
55  
56  		// Fill some data to a PojoComplex bean
57  		var values = new HashMap<String,List<Pojo>>();
58  		var setOne = new ArrayList<Pojo>();
59  		setOne.add(new Pojo("1.1", "name1"));
60  		setOne.add(new Pojo("1.1", "name2"));
61  		var setTwo = new ArrayList<Pojo>();
62  		setTwo.add(new Pojo("1.2", "name1"));
63  		setTwo.add(new Pojo("1.2", "name2"));
64  		values.put("setOne", setOne);
65  		values.put("setTwo", setTwo);
66  		var pojoc = new PojoComplex("pojo", new Pojo("1.0", "name0"), values);
67  
68  		var flat = htmlSerializer.serialize(pojoc);
69  
70  		// Print out the created POJO in JSON format.
71  		System.out.println(flat);
72  
73  		var parse = htmlParser.parse(flat, PojoComplex.class);
74  
75  		assert parse.getId().equals(pojoc.getId());
76  		assert parse.getInnerPojo().getName().equals(pojoc.getInnerPojo().getName());
77  		assert parse.getInnerPojo().getId().equals(pojoc.getInnerPojo().getId());
78  
79  		// The object above can be parsed thanks to the @Beanc(properties = id,name) annotation on Pojo
80  		// Using this approach, you can keep your POJOs immutable, and still serialize and deserialize them.
81  	}
82  }
