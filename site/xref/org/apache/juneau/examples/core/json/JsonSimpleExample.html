20  package org.apache.juneau.examples.core.json;
21  
22  import org.apache.juneau.examples.core.pojo.Pojo;
23  import org.apache.juneau.json.JsonParser;
24  import org.apache.juneau.json.JsonSerializer;
25  import org.apache.juneau.json.Json5Serializer;
26  
27  import java.util.Map;
28  
29  /**
30   * Sample class which shows the simple usage of JsonSerializer and JsonParser.
31   *
32   * <h5 class='section'>See Also:</h5><ul>
33   * </ul>
34   */
35  public class JsonSimpleExample {
36  
37  	/**
38  	 * Serializing Pojo bean into Json format and Deserialize back to Pojo instance type.
39  	 *
40  	 * @param args Unused.
41  	 * @throws Exception Unused.
42  	 */
43  	@SuppressWarnings({ "unused", "rawtypes" })
44  	public static void main(String[] args) throws Exception{
45  		// Juneau provides static constants with the most commonly used configurations
46  		// Get a reference to a serializer - converting POJO to flat format
47  		// Produces
48  		// {"name":"name","id":"id"}
49  		JsonSerializer jsonSerializer = JsonSerializer.DEFAULT;
50  		// Get a reference to a parser - converts that flat format back into the POJO
51  		JsonParser jsonParser = JsonParser.DEFAULT;
52  
53  		Pojo pojo = new Pojo("id","name");
54  
55  		String flat = jsonSerializer.serialize(pojo);
56  		// Print out the created POJO in JSON format.
57  		System.out.println(flat);
58  
59  		Pojo parse = jsonParser.parse(flat, Pojo.class);
60  
61  		assert parse.getId().equals(pojo.getId());
62  		assert parse.getName().equals(pojo.getName());
63  
64  		// Produces
65  		// {name:'name',id:'id'}
66  		String json5 = Json5Serializer.DEFAULT.serialize(pojo);
67  		System.out.println(json5);
68  
69  		// Parse a JSON object (creates a generic JsonMap).
70  		String json = "{name:'John Smith',age:21}";
71  		Map m1 = jsonParser.parse(json, Map.class);
72  
73  		// Parse a JSON string.
74  		json = "'foobar'";
75  		String s2 = jsonParser.parse(json, String.class);
76  
77  		// Parse a JSON number as a Long or Float.
78  		json = "123";
79  		Long l3 = jsonParser.parse(json, Long.class);
80  		Float f3 = jsonParser.parse(json, Float.class);
81  
82  		// The object above can be parsed thanks to the @Beanc(properties = id,name) annotation on Pojo
83  		// Using this approach, you can keep your POJOs immutable, and still serialize and deserialize them.
84  	}
85  }
