17  package org.apache.juneau.examples.core.xml;
18  
19  import java.util.*;
20  
21  import org.apache.juneau.examples.core.pojo.*;
22  import org.apache.juneau.xml.*;
23  
24  /**
25   * Xml configuration example.
26   *
27   * <h5 class='section'>See Also:</h5><ul>
28   * </ul>
29   */
30  public class XmlConfigurationExample {
31  
32  	/**
33  	 * Examples on XML Serializers configured using properties
34  	 * defined in XmlSerializer class.
35  	 *
36  	 * @param args Unused.
37  	 * @throws Exception Unused.
38  	 */
39  	public static void main(String[] args) throws Exception {
40  
41  		Pojo aPojo = new Pojo("a","<pojo>");
42  
43  		/**
44  		 * Xml Serializers can be configured using properties defined in XmlSerializer.
45  		 * Produces
46  		 * <object>
47  		 * <name>&lt;pojo&gt;</name>
48  		 * <id>a</id>
49  		 * </object>
50  		 */
51  		String withWhitespace = XmlSerializer.create().ws().build().serialize(aPojo);
52  		// the output will be padded with spaces after format characters.
53  		System.out.println(withWhitespace);
54  
55  		HashMap<String, List<Pojo>> values = new HashMap<>();
56  		PojoComplex pojoc = new PojoComplex("pojo", new Pojo("1.0", "name0"), values);
57  
58  		//Produces
59  		//<object><innerPojo><name>name0</name><id>1.0</id></innerPojo><id>pojo</id></object>
60  		String mapescaped = XmlSerializer.create().trimEmptyMaps().build().serialize(pojoc);
61  		// the output will have trimmed Empty maps.
62  		System.out.println(mapescaped);
63  
64  		//Produces
65  		//<object xmlns="http://www.apache.org/2013/Juneau"><name>&lt;pojo&gt;</name><id>a</id></object>
66  		String nspaceToRoot = XmlSerializer.create().ns().addNamespaceUrisToRoot().build().serialize(aPojo);
67  		// the output will add default name space to the xml document root.
68  		System.out.println(nspaceToRoot);
69  
70  		Pojo nPojo = new Pojo("a",null);
71  
72  		//Produces
73  		//<object><id>a</id></object>
74  		String nullescaped = XmlSerializer.create().build().serialize(nPojo);
75  		// the output will have trimmed null properties.
76  		System.out.println(nullescaped);
77  
78  		//Produces
79  		//<object xmlns="http://www.pierobon.org/iis/review1.htm.html#one"><name>&lt;pojo&gt;</name><id>a</id></object>
80  		String dNamsSpace = XmlSerializer.create().enableNamespaces().defaultNamespace(Namespace.create("http://www.pierobon.org" +
81  			"/iis/review1.htm.html#one")).addNamespaceUrisToRoot().build()
82  			.serialize(aPojo);
83  		// the output will have new default namespace added.
84  		System.out.println(dNamsSpace);
85  
86  	}
87  }
