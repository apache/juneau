17  package org.apache.juneau.examples.core.xml;
18  
19  import java.util.*;
20  
21  import org.apache.juneau.examples.core.pojo.*;
22  import org.apache.juneau.xml.*;
23  
24  /**
25   * Xml configuration example.
26   *
27   */
28  public class XmlConfigurationExample {
29  
30  	/**
31  	 * Examples on XML Serializers configured using properties
32  	 * defined in XmlSerializer class.
33  	 *
34  	 * @param args Unused.
35  	 * @throws Exception Unused.
36  	 */
37  	public static void main(String[] args) throws Exception {
38  
39  		var aPojo = new Pojo("a", "<pojo>");
40  
41  		/**
42  		 * Xml Serializers can be configured using properties defined in XmlSerializer.
43  		 * Produces
44  		 * <object>
45  		 * <name>&lt;pojo&gt;</name>
46  		 * <id>a</id>
47  		 * </object>
48  		 */
49  		var withWhitespace = XmlSerializer.create().ws().build().serialize(aPojo);
50  		// the output will be padded with spaces after format characters.
51  		System.out.println(withWhitespace);
52  
53  		var values = new HashMap<String,List<Pojo>>();
54  		var pojoc = new PojoComplex("pojo", new Pojo("1.0", "name0"), values);
55  
56  		//Produces
57  		//<object><innerPojo><name>name0</name><id>1.0</id></innerPojo><id>pojo</id></object>
58  		var mapescaped = XmlSerializer.create().trimEmptyMaps().build().serialize(pojoc);
59  		// the output will have trimmed Empty maps.
60  		System.out.println(mapescaped);
61  
62  		//Produces
63  		//<object xmlns="http://www.apache.org/2013/Juneau"><name>&lt;pojo&gt;</name><id>a</id></object>
64  		var nspaceToRoot = XmlSerializer.create().ns().addNamespaceUrisToRoot().build().serialize(aPojo);
65  		// the output will add default name space to the xml document root.
66  		System.out.println(nspaceToRoot);
67  
68  		var nPojo = new Pojo("a", null);
69  
70  		//Produces
71  		//<object><id>a</id></object>
72  		var nullescaped = XmlSerializer.create().build().serialize(nPojo);
73  		// the output will have trimmed null properties.
74  		System.out.println(nullescaped);
75  
76  		//Produces
77  		//<object xmlns="http://www.pierobon.org/iis/review1.htm.html#one"><name>&lt;pojo&gt;</name><id>a</id></object>
78  		var dNamsSpace = XmlSerializer.create().enableNamespaces().defaultNamespace(Namespace.create("http://www.pierobon.org" + "/iis/review1.htm.html#one")).addNamespaceUrisToRoot().build()
79  			.serialize(aPojo);
80  		// the output will have new default namespace added.
81  		System.out.println(dNamsSpace);
82  
83  	}
84  }
