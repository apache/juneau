13  package org.apache.juneau.examples.core.xml;
14  
15  import org.apache.juneau.examples.core.pojo.*;
16  import org.apache.juneau.xml.*;
17  
18  /**
19   * Sample class which shows the simple usage of XmlSerializer.
20   *
21   * <h5 class='section'>See Also:</h5><ul>
22   * </ul>
23   */
24  public class XmlSimpleExample {
25  
26  	/**
27  	 * Serializing SimplePojo bean into human readable XML
28  	 * and Deserialize back to Pojo instance type.
29  	 *
30  	 * @param args Unused.
31  	 * @throws Exception Unused.
32  	 */
33  	public static void main(String[] args) throws Exception {
34  
35  		// Fill some data to a Pojo bean
36  		Pojo pojo = new Pojo("id","name");
37  
38  		// Serialize to human readable XML and print
39  		/**
40  		 * <object>
41  		 * <name>name</name>
42  		 * <id>id</id>
43  		 * </object>
44  		 */
45  		String serial = XmlSerializer.DEFAULT_SQ_READABLE.serialize(pojo);
46  		System.out.println(serial);
47  
48  		// Deserialize back to Pojo instance
49  		Pojo obj = XmlParser.DEFAULT.parse(serial, Pojo.class);
50  
51  		assert obj.getId().equals(pojo.getId());
52  		assert obj.getName().equals(pojo.getName());
53  
54  		// The object above can be parsed thanks to the @Beanc annotation on PojoComplex
55  		// Using this approach, you can keep your POJOs immutable, and still serialize and deserialize them.
56  
57  	}
58  }
