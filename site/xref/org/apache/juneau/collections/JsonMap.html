33  /**
34   * Java implementation of a JSON object.
35   *
36   * <p>
37   * An extension of {@link LinkedHashMap}, so all methods available in that class are also available to this class.
38   *
39   * <p>
40   * Note that the use of this class is optional for generating JSON. The serializers will accept any objects that implement the
41   * {@link java.util.Map} interface.  But this class provides some useful additional functionality when working with
42   * JSON models constructed from Java Collections Framework objects.  For example, a constructor is provided for
43   * converting a JSON object string directly into a {@link Map}.  It also contains accessor methods for to avoid common
44   * typecasting when accessing elements in a list.
45   *
46   * <h5 class='section'>Example:</h5>
47   * <p class='bjava'>
48   * 	<jc>// Construct an empty Map</jc>
49   * 	JsonMap <jv>map</jv> = JsonMap.<jsm>of</jsm>();
50   *
51   * 	<jc>// Construct a Map from JSON</jc>
52   * 	<jv>map</jv> = JsonMap.<jsm>ofJson</jsm>(<js>"{a:'A',b:{c:'C',d:123}}"</js>);
53   *
54   * 	<jc>// Construct a Map using the append method</jc>
55   * 	<jv>map</jv> = JsonMap.<jsm>of</jsm>().a(<js>"foo"</js>,<js>"x"</js>).a(<js>"bar"</js>,123).a(<js>"baz"</js>,<jk>true</jk>);
56   *
57   * 	<jc>// Construct a Map from XML generated by XmlSerializer</jc>
58   * 	String <jv>xml</jv> = <js>"&lt;object&gt;&lt;a type='string'&gt;A&lt;/a&gt;&lt;b type='object'&gt;&lt;c type='string'&gt;C&lt;/c&gt;&lt;d type='number'&gt;123&lt;/d&gt;&lt;/b&gt;&lt;/object&gt;"</js>;
59   * 	<jv>map</jv> = JsonMap.<jsm>of</jsm>(<jv>xml</jv>, XmlParser.<jsf>DEFAULT</jsf>);
60   *
61   * 	<jc>// Construct a Map from a URL GET parameter string generated by UrlEncodingParser</jc>
62   * 	String <jv>urlParams</jv> = <js>"?a='A'&amp;b={c:'C',d:123}"</js>;
63   * 	<jv>map</jv> = JsonMap.<jsm>of</jsm>(<jv>urlParams</jv>, UrlEncodingParser.<jsf>DEFAULT</jsf>);
64   *
65   * 	<jc>// Construct JSON from JsonMap</jc>
66   * 	<jv>map</jv> = JsonMap.<jsm>ofJson</jsm>(<js>"{foo:'bar'},{baz:[123,true]}"</js>);
67   * 	String <jv>json</jv> = <jv>map</jv>.toString();  <jc>// Produces "{foo:'bar'},{baz:[123,true]}"</jc>
68   * 	<jv>json</jv> = <jv>map</jv>.toString(JsonSerializer.<jsf>DEFAULT</jsf>);  <jc>// Equivalent</jc>
69   * 	<jv>json</jv> = JsonSerializer.<jsf>DEFAULT</jsf>.serialize(<jv>map</jv>);  <jc>// Equivalent</jc>
70   *
71   * 	<jc>// Get a map entry as an Integer</jc>
72   * 	<jv>map</jv> = JsonMap.<jsm>ofJson</jsm>(<js>"{foo:123}"</js>);
73   * 	Integer <jv>integer</jv> = <jv>map</jv>.getInt(<js>"foo"</js>);
74   * 	<jv>integer</jv> = <jv>map</jv>.get(Integer.<jk>class</jk>, <js>"foo"</js>);  <jc>// Equivalent</jc>
75   *
76   * 	<jc>// Get a map entry as a Float</jc>
77   * 	<jv>map</jv> = JsonMap.<jsm>ofJson</jsm>(<js>"{foo:123}"</js>);
78   * 	Float <jv>_float</jv> = <jv>map</jv>.getFloat(<js>"foo"</js>);
79   * 	<jv>_float</jv> = <jv>map</jv>.get(Float.<jk>class</jk>, <js>"foo"</js>);  <jc>// Equivalent</jc>
80   *
81   * 	<jc>// Same as above, except converted to a String</jc>
82   * 	<jv>map</jv> = JsonMap.<jsm>ofJson</jsm>(<js>"{foo:123}"</js>);
83   * 	String <jv>string</jv> = <jv>map</jv>.getString(<js>"foo"</js>); <jc>// Returns "123"</jc>
84   * 	<jv>string</jv> = <jv>map</jv>.get(String.<jk>class</jk>, <js>"foo"</js>);  <jc>// Equivalent</jc>
85   *
86   * 	<jc>// Get one of the entries in the list as a bean (converted to a bean if it isn't already one)</jc>
87   * 	<jv>map</jv> = JsonMap.<jsm>ofJson</jsm>(<js>"{person:{name:'John Smith',age:45}}"</js>);
88   * 	Person <jv>person</jv> = <jv>map</jv>.get(Person.<jk>class</jk>, <js>"person"</js>);
89   *
90   * 	<jc>// Add an inner map</jc>
91   * 	JsonMap <jv>map1</jv> = JsonMap.<jsm>ofJson</jsm>(<js>"{a:1}"</js>);
92   * 	JsonMap <jv>map2</jv> = JsonMap.<jsm>ofJson</jsm>(<js>"{b:2}"</js>).setInner(<jv>map1</jv>);
93   * 	<jk>int</jk> <jv>_int</jv> = <jv>map2</jv>.getInt(<js>"a"</js>);  <jc>// a == 1 </jc>
94   * </p>
95   *
96   * <h5 class='section'>Notes:</h5><ul>
97   * 	<li class='warn'>This class is not thread safe.
98   * </ul>
99   */

383 	 * Set an inner map in this map to allow for chained get calls.
384 	 *
385 	 * <p>
386 	 * If {@link #get(Object)} returns <jk>null</jk>, then {@link #get(Object)} will be called on the inner map.
387 	 *
388 	 * <p>
389 	 * In addition to providing the ability to chain maps, this method also provides the ability to wrap an existing map
390 	 * inside another map so that you can add entries to the outer map without affecting the values on the inner map.
391 	 *
392 	 * <p class='bjava'>
393 	 * 	JsonMap <jv>map1</jv> = JsonMap.<jsm>ofJson</jsm>(<js>"{foo:1}"</js>);
394 	 * 	JsonMap <jv>map2</jv> = JsonMap.<jsm>of</jsm>().setInner(<jv>map1</jv>);
395 	 * 	<jv>map2</jv>.put(<js>"foo"</js>, 2);                      <jc>// Overwrite the entry</jc>
396 	 * 	<jk>int</jk> <jv>foo1</jv> = <jv>map1</jv>.getInt(<js>"foo"</js>);           <jc>// foo1 == 1 </jc>
397 	 * 	<jk>int</jk> <jv>foo2</jv> = <jv>map2</jv>.getInt(<js>"foo"</js>);           <jc>// foo2 == 2 </jc>
398 	 * </p>

403 	 * @return This object.
404 	 */

523 		return this;

526 	/**
527 	 * Enables filtering based on default values.
528 	 *
529 	 * <p>
530 	 * Any of the following types will be ignored when set as values in this map:
531 	 * <ul>
532 	 * 	<li><jk>null</jk>
533 	 * 	<li><jk>false</jk>
534 	 * 	<li><c>-1</c> (any Number type)
535 	 * 	<li>Empty arrays/collections/maps.
536 	 * </ul>
537 	 * @return This object.
538 	 */
539 	public JsonMap filtered() {
540 		return filtered(x -> ! (
541 			x == null
542 			|| (x instanceof Boolean && x.equals(false))
543 			|| (x instanceof Number && ((Number)x).intValue() == -1)
544 			|| (isArray(x) && Array.getLength(x) == 0)
545 			|| (x instanceof Map && ((Map<?,?>)x).isEmpty())
546 			|| (x instanceof Collection && ((Collection<?>)x).isEmpty())
547 		));
548 	}

569 	 * Same as {@link Map#get(Object) get()}, but casts or converts the value to the specified class type.
570 	 *
571 	 * <p>
572 	 * This is the preferred get method for simple types.
573 	 *
574 	 * <h5 class='section'>Examples:</h5>
575 	 * <p class='bjava'>
576 	 * 	JsonMap <jv>map</jv> = JsonMap.<jsm>ofJson</jsm>(<js>"..."</js>);
577 	 *

579 	 * 	String <jv>string</jv> = <jv>map</jv>.get(<js>"key1"</js>, String.<jk>class</jk>);
