14  
15  import static java.util.Arrays.*;
16  
17  import java.lang.reflect.*;
18  import java.util.*;
19  
20  import org.apache.juneau.*;
21  
22  /**
23   * POJO model paginator.
24   *
25   * <p>
26   * 	This class is designed to extract sublists from arrays/collections of maps or beans.
27   * </p>
28   *
29   * <h5 class='section'>Example:</h5>
30   * <p class='bjava'>
31   * 	MyBean[] <jv>arrayOfBeans</jv> = ...;
32   * 	ObjectPaginator <jv>paginator</jv> = ObjectPaginator.<jsm>create</jsm>();
33   *
34   * 	<jc>// Returns all rows from 100 to 110.</jc>
35   * 	List&lt;MyBean&gt; <jv>result</jv> = <jv>paginator</jv>.run(<jv>arrayOfBeans</jv>, 100, 10);
36   * </p>
37   * <p>
38   * 	The tool can be used against the following data types:
39   * </p>
40   * <ul>
41   * 	<li>Arrays/collections of maps or beans.
42   * </ul>
43   *
44   * <h5 class='section'>See Also:</h5><ul>
45   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/ObjectTools">Object Tools</a>
46  
47   * </ul>
48   */
49  public final class ObjectPaginator implements ObjectTool<PageArgs> {
50  
51  	//-----------------------------------------------------------------------------------------------------------------
52  	// Static
53  	//-----------------------------------------------------------------------------------------------------------------
54  
55  	/**
56  	 * Static creator.
57  	 * @return A new {@link ObjectPaginator} object.
58  	 */
59  	public static ObjectPaginator create() {
60  		return new ObjectPaginator();
61  	}
62  
63  	//-----------------------------------------------------------------------------------------------------------------
64  	// Instance
65  	//-----------------------------------------------------------------------------------------------------------------
66  
67  	/**
68  	 * Convenience method for executing the paginator.
69  	 *
70  	 * @param <R> The collection element type.
71  	 * @param input The input.  Must be a collection or array of objects.
72  	 * @param pos The zero-index position to start from.
73  	 * @param limit The max number of entries to retrieve.
74  	 * @return A sublist of representing the entries from the position with the specified limit.
75  	 */
76  	@SuppressWarnings("unchecked")
77  	public <R> List<R> run(Object input, int pos, int limit) {
78  		BeanSession bs = BeanContext.DEFAULT_SESSION;
79  		Object r = run(BeanContext.DEFAULT_SESSION, input, PageArgs.create(pos, limit));
80  		if (r instanceof List)
81  			return (List<R>)r;
82  		return bs.convertToType(r, List.class);
83  	}
84  
85  	@Override /* ObjectTool */
86  	@SuppressWarnings({ "rawtypes", "unchecked" })
87  	public Object run(BeanSession session, Object input, PageArgs args) {
88  
89  		if (input == null)
90  			return null;
91  
92  		ClassMeta type = session.getClassMetaForObject(input);
93  
94  		if (! type.isCollectionOrArray())
95  			return input;
96  
97  		int pos = args.getPosition();
98  		int limit = args.getLimit();
99  
100 		if (type.isArray()) {
101 			int size = Array.getLength(input);
102 			int end = (limit+pos >= size) ? size : limit + pos;
103 			pos = Math.min(pos, size);
104 			ClassMeta<?> et = type.getElementType();
105 			if (! et.isPrimitive())
106 				return copyOfRange((Object[])input, pos, end);
107 			if (et.is(boolean.class))
108 				return copyOfRange((boolean[])input, pos, end);
109 			if (et.is(byte.class))
110 				return copyOfRange((byte[])input, pos, end);
111 			if (et.is(char.class))
112 				return copyOfRange((char[])input, pos, end);
113 			if (et.is(double.class))
114 				return copyOfRange((double[])input, pos, end);
115 			if (et.is(float.class))
116 				return copyOfRange((float[])input, pos, end);
117 			if (et.is(int.class))
118 				return copyOfRange((int[])input, pos, end);
119 			if (et.is(long.class))
120 				return copyOfRange((long[])input, pos, end);
121 			return copyOfRange((short[])input, pos, end);
122 		}
123 
124 		List l = type.isList() ? (List)input : new ArrayList((Collection)input);
125 		int end = (limit+pos >= l.size()) ? l.size() : limit + pos;
126 		pos = Math.min(pos, l.size());
127 		return l.subList(pos, end);
128 	}
129 }
