15  import static org.apache.juneau.common.internal.StringUtils.*;
16  
17  import java.util.*;
18  import java.util.regex.*;
19  
20  import org.apache.juneau.*;
21  import org.apache.juneau.common.internal.*;

23  /**
24   * String matcher factory for the {@link ObjectSearcher} class.
25   *
26   * <p>
27   * 	The class provides searching based on the following patterns:
28   * </p>
29   * <ul>
30   * 	<li><js>"property=foo"</js> - Simple full word match
31   * 	<li><js>"property=fo*"</js>, <js>"property=?ar"</js> - Meta-character matching
32   * 	<li><js>"property=foo bar"</js>(implicit), <js>"property=^foo ^bar"</js>(explicit) - Multiple OR'ed patterns
33   * 	<li><js>"property=+fo* +*ar"</js> - Multiple AND'ed patterns
34   * 	<li><js>"property=fo* -bar"</js> - Negative patterns
35   * 	<li><js>"property='foo bar'"</js> - Patterns with whitespace
36   * 	<li><js>"property=foo\\'bar"</js> - Patterns with single-quotes
37   * 	<li><js>"property=/foo\\s+bar"</js> - Regular expression match
38   * </ul>
39   *
40   * <h5 class='section'>See Also:</h5><ul>
41   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/ObjectTools">Object Tools</a>
42   * </ul>
43   */
44  public class StringMatcherFactory extends MatcherFactory {
45  
46  	/**
47  	 * Default reusable matcher.
48  	 */
49  	public static final StringMatcherFactory DEFAULT = new StringMatcherFactory();
50  
51  	@Override
52  	public boolean canMatch(ClassMeta<?> cm) {
53  		return true;
54  	}
55  
56  	@Override
57  	public AbstractMatcher create(String pattern) {
58  		return new StringMatcher(pattern);
59  	}
60  
61  	/**
62  	 * A construct representing a single search pattern.
63  	 */
64  	private static class StringMatcher extends AbstractMatcher {
65  		private String pattern;
66  		private static final AsciiSet
67  			META_CHARS = AsciiSet.of("*?'\""),
68  			SQ_CHAR = AsciiSet.of("'"),
69  			DQ_CHAR = AsciiSet.of("\""),
70  			REGEX_CHARS = AsciiSet.of("+\\[]{}()^$.");
71  
72  		Pattern[] orPatterns, andPatterns, notPatterns;
73  
74  		public StringMatcher(String searchPattern) {
75  
76  			this.pattern = searchPattern.trim();
77  			List<Pattern> ors = new LinkedList<>();
78  			List<Pattern> ands = new LinkedList<>();
79  			List<Pattern> nots = new LinkedList<>();
80  
81  			for (String s : Utils.splitQuoted(pattern, true)) {
82  				char c0 = s.charAt(0), c9 = s.charAt(s.length()-1);
83  
84  				if (c0 == '/' && c9 == '/' && s.length() > 1) {
85  					ands.add(Pattern.compile(strip(s)));
86  				} else {
87  					char prefix = '^';
88  					boolean ignoreCase = false;
89  					if (s.length() > 1 && (c0 == '^' || c0 == '+' || c0 == '-')) {
90  						prefix = c0;
91  						s = s.substring(1);
92  						c0 = s.charAt(0);
93  					}
94  
95  					if (c0 == '\'') {
96  						s = unEscapeChars(strip(s), SQ_CHAR);
97  						ignoreCase = true;
98  					} else if (c0 == '"') {
99  						s = unEscapeChars(strip(s), DQ_CHAR);
100 					}
101 
102 					if (REGEX_CHARS.contains(s) || META_CHARS.contains(s)) {
103 						StringBuilder sb = new StringBuilder();
104 						boolean isInEscape = false;
105 						for (int i = 0; i < s.length(); i++) {
106 							char c = s.charAt(i);
107 							if (isInEscape) {
108 								if (c == '?' || c == '*' || c == '\\')
109 									sb.append('\\').append(c);
110 								else
111 									sb.append(c);
112 								isInEscape = false;
113 							} else {
114 								if (c == '\\')
115 									isInEscape = true;
116 								else if (c == '?')
117 									sb.append(".?");
118 								else if (c == '*')
119 									sb.append(".*");
120 								else if (REGEX_CHARS.contains(c))
121 									sb.append("\\").append(c);
122 								else
123 									sb.append(c);
124 							}
125 						}
126 						s = sb.toString();
127 					}
128 
129 
130 					int flags = Pattern.DOTALL;
131 					if (ignoreCase)
132 						flags |= Pattern.CASE_INSENSITIVE;
133 
134 					Pattern p = Pattern.compile(s, flags);
135 
136 					if (prefix == '-')
137 						nots.add(p);
138 					else if (prefix == '+')
139 						ands.add(p);
140 					else
141 						ors.add(p);
142 				}
143 			}
144 			orPatterns = ors.toArray(new Pattern[ors.size()]);
145 			andPatterns = ands.toArray(new Pattern[ands.size()]);
146 			notPatterns = nots.toArray(new Pattern[nots.size()]);
147 		}
148 
149 		@Override
150 		public boolean matches(ClassMeta<?> cm, Object o) {
151 			String s = (String)o;
152 			for (Pattern andPattern : andPatterns)
153                 if (! andPattern.matcher(s).matches())
154 					return false;
155 			for (Pattern notPattern : notPatterns)
156                 if (notPattern.matcher(s).matches())
157 					return false;
158 			for (Pattern orPattern : orPatterns)
159                 if (orPattern.matcher(s).matches())
160 					return true;
161 			return orPatterns.length == 0;
162 		}
163 
164 		@Override
165 		public String toString() {
166 			return pattern;
167 		}
168 	}
169 }
