28  /**
29   * Date/time matcher factory for the {@link ObjectSearcher} class.
30   *
31   * <p>
32   * 	The class provides searching based on the following patterns:
33   * </p>
34   * <ul>
35   * 	<li><js>"property=2011"</js> - A single year
36   * 	<li><js>"property=2011 2013 2015"</js> - Multiple years
37   * 	<li><js>"property=2011-01"</js> - A single month
38   * 	<li><js>"property=2011-01-01"</js> - A single day
39   * 	<li><js>"property=2011-01-01T12"</js> - A single hour
40   * 	<li><js>"property=2011-01-01T12:30"</js> - A single minute
41   * 	<li><js>"property=2011-01-01T12:30:45"</js> - A single second
42   * 	<li><js>"property=&gt;2011"</js>,<js>"property=&gt;=2011"</js>,<js>"property=&lt;2011"</js>,<js>"property=&lt;=2011"</js> - Open-ended ranges
43   * 	<li><js>"property=&gt;2011"</js>,<js>"property=&gt;=2011"</js>,<js>"property=&lt;2011"</js>,<js>"property=&lt;=2011"</js> - Open-ended ranges
44   * 	<li><js>"property=2011 - 2013-06-30"</js> - Closed ranges
45   * </ul>
46   *
47   * <h5 class='section'>See Also:</h5><ul>
48   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/ObjectTools">Object Tools</a>
49   * </ul>
50   */
51  public class TimeMatcherFactory extends MatcherFactory {
52  
53  	/**
54  	 * A construct representing a single search pattern.
55  	 */
56  	private static class TimeMatcher extends AbstractMatcher {
57  
58  		// @formatter:off
59  		private static final AsciiSet
60  			DT = AsciiSet.of("0123456789-:T./"),
61  			WS = AsciiSet.of(" \t");
62  		// @formatter:on
63  
64  		TimestampRange[] ranges;
65  		List<TimestampRange> l = new LinkedList<>();
66  
67  		public TimeMatcher(String s) {
68  
69  			// Possible patterns:
70  			// >2000, <2000, >=2000, <=2000, > 2000, 2000 - 2001, '2000', >'2000', '2000'-'2001', '2000' - '2001'
71  
72  			// Possible states:
73  			// S1 = Looking for [<]/[>]/quote/NUM ([>]=S2, [<]=S3, [']=S5, ["]=S6, NUM=S8)
74  			// S2 = Found [>], looking for [=]/quote/NUM ([=]=S4, [']=S5, ["]=S6, NUM=S8)
75  			// S3 = Found [<], looking for [=]/quote/NUM ([=]=S4, [']=S5, ["]=S6, NUM=S8)
76  			// S4 = Found [>=] or [<=], looking for quote/NUM ([']=S5, ["]=S6, NUM=S8)
77  			// S5 = Found ['], looking for ['] ([']=S1)
78  			// S6 = Found ["], looking for ["] (["]=S1)
79  			// S7 = Found [123"] or [123'], looking for WS (WS=S9)
80  			// S8 = Found [2], looking for WS (WS=S9)
81  			// S9 = Found [2000 ], looking for [-]/quote/NUM ([-]=S10, [']=S11, ["]=S12, NUM=S13)
82  			// S10 = Found [2000 -], looking for quote/NUM ([']=S11, ["]=S12, NUM=S13)
83  			// S11 = Found [2000 - '], looking for ['] ([']=S1)
84  			// S12 = Found [2000 - "], looking for ["] (["]=S1)
85  			// S13 = Found [2000 - 2], looking for WS (WS=S1)
86  
87  			var state = S1;
88  			var mark = 0;
89  			var eq = Equality.NONE;
90  			var s1 = (String)null;
91  			var s2 = (String)null;
92  
93  			int i;
94  			char c = 0;
95  			for (i = 0; i < s.trim().length(); i++) {
96  				c = s.charAt(i);
97  				if (state == S1) {
98  					// S1 = Looking for [>]/[<]/quote/NUM ([>]=S2, [<]=S3, [']=S5, ["]=S6, NUM=S8)
99  					if (WS.contains(c)) {
100 						state = S1;
101 					} else if (c == '>') {
102 						state = S2;
103 						eq = Equality.GT;
104 					} else if (c == '<') {
105 						state = S3;
106 						eq = Equality.LT;
107 					} else if (c == '\'') {
108 						state = S5;
109 						mark = i + 1;
110 					} else if (c == '"') {
111 						state = S6;
112 						mark = i + 1;
113 					} else if (DT.contains(c)) {
114 						state = S8;
115 						mark = i;
116 					} else {
117 						break;
118 					}
119 				} else if (state == S2) {
120 					// S2 = Found [>], looking for [=]/quote/NUM ([=]=S4, [']=S5, ["]=S6, NUM=S8)
121 					if (WS.contains(c)) {
122 						state = S2;
123 					} else if (c == '=') {
124 						state = S4;
125 						eq = Equality.GTE;
126 					} else if (c == '\'') {
127 						state = S5;
128 						mark = i + 1;
129 					} else if (c == '"') {
130 						state = S6;
131 						mark = i + 1;
132 					} else if (DT.contains(c)) {
133 						state = S8;
134 						mark = i;
135 					} else {
136 						break;
137 					}
138 				} else if (state == S3) {
139 					// S3 = Found [<], looking for [=]/quote/NUM ([=]=S4, [']=S5, ["]=S6, NUM=S8)
140 					if (WS.contains(c)) {
141 						state = S3;
142 					} else if (c == '=') {
143 						state = S4;
144 						eq = Equality.LTE;
145 					} else if (c == '\'') {
146 						state = S5;
147 						mark = i + 1;
148 					} else if (c == '"') {
149 						state = S6;
150 						mark = i + 1;
151 					} else if (DT.contains(c)) {
152 						state = S8;
153 						mark = i;
154 					} else {
155 						break;
156 					}
157 				} else if (state == S4) {
158 					// S4 = Found [>=] or [<=], looking for quote/NUM ([']=S5, ["]=S6, NUM=S8)
159 					if (WS.contains(c)) {
160 						state = S4;
161 					} else if (c == '\'') {
162 						state = S5;
163 						mark = i + 1;
164 					} else if (c == '"') {
165 						state = S6;
166 						mark = i + 1;
167 					} else if (DT.contains(c)) {
168 						state = S8;
169 						mark = i;
170 					} else {
171 						break;
172 					}
173 				} else if (state == S5) {
174 					// S5 = Found ['], looking for ['] ([']=S7)
175 					if (c == '\'') {
176 						state = S7;
177 						s1 = s.substring(mark, i);
178 					}
179 				} else if (state == S6) {
180 					// S6 = Found ["], looking for ["] (["]=S7)
181 					if (c == '"') {
182 						state = S7;
183 						s1 = s.substring(mark, i);
184 					}
185 				} else if (state == S7) {
186 					// S7 = Found [123"] or [123'], looking for WS (WS=S9)
187 					if (WS.contains(c)) {
188 						state = S9;
189 					} else if (c == '-') {
190 						state = S10;
191 					} else {
192 						break;
193 					}
194 				} else if (state == S8) {
195 					// S8 = Found [1], looking for WS (WS=S9)
196 					if (WS.contains(c)) {
197 						state = S9;
198 						s1 = s.substring(mark, i);
199 					}
200 				} else if (state == S9) {
201 					// S9 = Found [2000 ], looking for [-]/[>]/[<]/quote/NUM ([-]=S10, [>]=S2, [<]=S3, [']=S5, ["]=S6, NUM=S8)
202 					if (WS.contains(c)) {
203 						state = S9;
204 					} else if (c == '-') {
205 						state = S10;
206 					} else if (c == '>') {
207 						state = S2;
208 						l.add(new TimestampRange(eq, s1));
209 						eq = Equality.GT;
210 						s1 = null;
211 					} else if (c == '<') {
212 						state = S3;
213 						l.add(new TimestampRange(eq, s1));
214 						eq = Equality.LT;
215 						s1 = null;
216 					} else if (c == '\'') {
217 						state = S5;
218 						l.add(new TimestampRange(eq, s1));
219 						mark = i + 1;
220 						eq = null;
221 						s1 = null;
222 					} else if (c == '"') {
223 						state = S6;
224 						l.add(new TimestampRange(eq, s1));
225 						mark = i + 1;
226 						eq = null;
227 						s1 = null;
228 					} else if (DT.contains(c)) {
229 						state = S8;
230 						l.add(new TimestampRange(eq, s1));
231 						eq = null;
232 						s1 = null;
233 						mark = i;
234 					} else {
235 						break;
236 					}
237 				} else if (state == S10) {
238 					// S10 = Found [2000 -], looking for quote/NUM ([']=S11, ["]=S12, NUM=S13)
239 					if (WS.contains(c)) {
240 						state = S10;
241 					} else if (c == '\'') {
242 						state = S11;
243 						mark = i + 1;
244 					} else if (c == '"') {
245 						state = S12;
246 						mark = i + 1;
247 					} else if (DT.contains(c)) {
248 						state = S13;
249 						mark = i;
250 					} else {
251 						break;
252 					}
253 				} else if (state == S11) {
254 					// S11 = Found [2000 - '], looking for ['] ([']=S1)
255 					if (c == '\'') {
256 						state = S1;
257 						s2 = s.substring(mark, i);
258 						l.add(new TimestampRange(s1, s2));
259 						s1 = null;
260 						s2 = null;
261 					}
262 				} else if (state == S12) {
263 					// S12 = Found [2000 - "], looking for ["] (["]=S1)
264 					if (c == '"') {
265 						state = S1;
266 						s2 = s.substring(mark, i);
267 						l.add(new TimestampRange(s1, s2));
268 						s1 = null;
269 						s2 = null;
270 					}
271 				} else /* (state == S13) */ {
272 					// S13 = Found [2000 - 2], looking for WS (WS=S1)
273 					if (WS.contains(c)) {
274 						state = S1;
275 						s2 = s.substring(mark, i);
276 						l.add(new TimestampRange(s1, s2));
277 						s1 = null;
278 						s2 = null;
279 					}
280 				}
281 			}
282 
283 			if (i != s.length())
284 				throw new PatternException("Invalid range pattern ({0}): pattern=[{1}], pos=[{2}], char=[{3}]", state, s, i, c);
285 
286 			if (state == S1) {
287 				// No tokens found.
288 			} else if (state == S2 || state == S3 || state == S4 || state == S5 || state == S6 || state == S10 || state == S11 || state == S12) {
289 				throw new PatternException("Invalid range pattern (E{0}): {1}", state, s);
290 			} else if (state == S7) {
291 				l.add(new TimestampRange(eq, s1));
292 			} else if (state == S8) {
293 				s1 = s.substring(mark).trim();
294 				l.add(new TimestampRange(eq, s1));
295 			} else /* (state == S13) */ {
296 				s2 = s.substring(mark).trim();
297 				l.add(new TimestampRange(s1, s2));
298 			}
299 
300 			ranges = l.toArray(new TimestampRange[l.size()]);
301 		}
302 
303 		@Override
304 		public boolean matches(ClassMeta<?> cm, Object o) {
305 			if (ranges.length == 0)
306 				return true;
307 
308 			var zdt = (ZonedDateTime)null;
309 			if (cm.isCalendar()) {
310 				var c = (Calendar)o;
311 				zdt = c.toInstant().atZone(c.getTimeZone().toZoneId());
312 			} else {
313 				var date = (Date)o;
314 				zdt = date.toInstant().atZone(ZoneId.systemDefault());
315 			}
316 			for (var range : ranges)
317 				if (range.matches(zdt))
318 					return true;
319 			return false;
320 		}
321 	}

323 	/**
324 	 * A construct representing a single search range in a single search pattern.
325 	 * All possible forms of search patterns are boiled down to these timestamp ranges.
326 	 */
327 	private static class TimestampRange {
328 		ZonedDateTime start;
329 		ZonedDateTime end;
330 
331 		public TimestampRange(Equality eq, String singleDate) {
332 			var singleDate1 = GranularZonedDateTime.of(singleDate);
333 			if (eq == Equality.GT) {
334 				this.start = singleDate1.roll(1).roll(MILLI_OF_SECOND, -1).getZonedDateTime();
335 				this.end = Instant.ofEpochMilli(Long.MAX_VALUE).atZone(ZoneId.systemDefault());
336 			} else if (eq == Equality.LT) {
337 				this.start = Instant.ofEpochMilli(0).atZone(ZoneId.systemDefault());
338 				this.end = singleDate1.getZonedDateTime();
339 			} else if (eq == Equality.GTE) {
340 				this.start = singleDate1.roll(MILLI_OF_SECOND, -1).getZonedDateTime();
341 				this.end = Instant.ofEpochMilli(Long.MAX_VALUE).atZone(ZoneId.systemDefault());
342 			} else if (eq == Equality.LTE) {
343 				this.start = Instant.ofEpochMilli(0).atZone(ZoneId.systemDefault());
344 				this.end = singleDate1.roll(1).getZonedDateTime();
345 			} else {
346 				this.start = singleDate1.copy().roll(MILLI_OF_SECOND, -1).getZonedDateTime();
347 				this.end = singleDate1.roll(1).getZonedDateTime();
348 			}
349 		}
350 
351 		public TimestampRange(String start, String end) {
352 			var start1 = GranularZonedDateTime.of(start);
353 			var end1 = GranularZonedDateTime.of(end);
354 			this.start = start1.copy().roll(MILLI_OF_SECOND, -1).getZonedDateTime();
355 			this.end = end1.roll(1).getZonedDateTime();
356 		}
357 
358 		public boolean matches(ZonedDateTime zdt) {
359 			return zdt.isAfter(start) && zdt.isBefore(end);
360 		}
361 	}

373 	@Override
374 	public boolean canMatch(ClassMeta<?> cm) {
375 		return cm.isDateOrCalendar();
376 	}

378 	@Override

380 		return new TimeMatcher(pattern);
381 	}
