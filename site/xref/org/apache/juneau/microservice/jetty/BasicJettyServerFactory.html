17  package org.apache.juneau.microservice.jetty;
18  
19  import java.io.*;
20  import java.util.*;
21  
22  import org.apache.juneau.common.utils.*;
23  import org.eclipse.jetty.server.*;
24  import org.eclipse.jetty.util.resource.*;
25  import org.eclipse.jetty.xml.*;
26  
27  /**
28   * Basic implementation of a Jetty server factory.
29   *
30   * <h5 class='section'>See Also:</h5><ul>
31   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauMicroserviceJettyBasics">juneau-microservice-jetty Basics</a>
32   * </ul>
33   */
34  public class BasicJettyServerFactory implements JettyServerFactory {
35  
36  	@Override
37  	public Server create(String jettyXml) throws Exception {
38  		Objects.requireNonNull(jettyXml,
39  			"jetty.xml file location was not specified in the configuration file (Jetty/config) or manifest file (Jetty-Config) or found on the file system or classpath.");
40  		File f = FileUtils.createTempFile("jetty.xml");
41  		try (Reader r = new StringReader(jettyXml); Writer w = new FileWriter(f)) {
42  			IOUtils.pipe(r, w);
43  			w.flush();
44  		}
45  		XmlConfiguration xmlConfiguration = new XmlConfiguration(new PathResourceFactory().newResource(f.toPath()));
46  		return (Server)xmlConfiguration.configure();
47  	}
48  }
