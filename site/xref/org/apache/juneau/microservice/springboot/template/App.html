17  package org.apache.juneau.microservice.springboot.template;
18  
19  import jakarta.servlet.*;
20  
21  import org.apache.juneau.rest.annotation.*;
22  import org.apache.juneau.rest.springboot.*;
23  import org.springframework.boot.autoconfigure.SpringBootApplication;
24  import org.springframework.boot.builder.*;
25  import org.springframework.boot.web.servlet.*;
26  import org.springframework.context.annotation.*;
27  import org.springframework.stereotype.Controller;
28  import org.springframework.web.filter.*;
29  
30  /**
31   * Entry point for Examples REST application when deployed as a Spring Boot application.
32   *
33   * <h5 class='section'>See Also:</h5><ul>
34   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/MySpringBootMicroserviceBasics">My SpringBoot Microservice Basics</a>
35   * </ul>
36   */
37  @SpringBootApplication
38  @Controller
39  public class App {
40  
41  	/**
42  	 * Entry point method.
43  	 * @param args Command-line arguments.
44  	 */
45  	@SuppressWarnings("resource")
46  	public static void main(String[] args) {
47  		new SpringApplicationBuilder(App.class).run(args);
48  	}
49  
50  	/**
51  	 * Optionally return an injectable message provider for the {@link HelloWorldResource} class.
52  	 *
53  	 * @return The message provider for the hello-world REST bean.
54  	 */
55  	@Bean
56  	public HelloWorldMessageProvider getHelloWorldMessageProvider() {
57  		return new HelloWorldMessageProvider("Hello Spring injection user!");
58  	}
59  
60  	/**
61  	 * Optionally return the {@link HelloWorldResource} object as an injectable bean.
62  	 *
63  	 * @return The hello-world REST bean.
64  	 */
65  	@Bean
66  	public HelloWorldResource getHelloWorldResource() {
67  		return new HelloWorldResource("Hello Spring user!");
68  	}
69  
70  	/**
71  	 * Our root REST bean.
72  	 * <p>
73  	 * Note that this must extend from {@link SpringRestServlet} so that child resources can be resolved as Spring
74  	 * beans.
75  	 * <p>
76  	 * All REST objects are attached to this bean using the {@link Rest#children()} annotation.
77  	 *
78  	 * @return The root resources REST bean.
79  	 */
80  	@Bean
81  	public RootResources getRootResources() {
82  		return new RootResources();
83  	}
84  
85  	/**
86  	 * @param rootResources The root REST resource servlet
87  	 * @return The servlet registration mapped to "/*".
88  	 */
89  	@Bean
90  	public ServletRegistrationBean<Servlet> getRootServlet(RootResources rootResources) {
91  		return new ServletRegistrationBean<>(rootResources, "/*");
92  	}
93  
94  	/**
95  	 * We want to be able to consume url-encoded-form-post bodies, but HiddenHttpMethodFilter triggers the HTTP
96  	 * body to be consumed.  So disable it.
97  	 *
98  	 * @param filter The filter.
99  	 * @return Filter registration bean.
100 	 */
101 	@Bean
102 	public FilterRegistrationBean<HiddenHttpMethodFilter> registration(HiddenHttpMethodFilter filter) {
103 		FilterRegistrationBean<HiddenHttpMethodFilter> registration = new FilterRegistrationBean<>(filter);
104 		registration.setEnabled(false);
105 		return registration;
106 	}
107 }
