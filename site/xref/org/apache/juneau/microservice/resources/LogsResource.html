78  	@Response(schema=@Schema(description="File or directory details"))
79  	@Bean(properties="type,name,size,lastModified,actions,files")
80  	public static class FileResource {
81  		static final FileFilter FILE_FILTER = f -> f.isDirectory() || f.getName().endsWith(".log");
82  		static final Comparator<FileResource> FILE_COMPARATOR = (o1, o2) -> {
83  			int c = o1.getType().compareTo(o2.getType());
84  			return c != 0 ? c : o1.getName().compareTo(o2.getName());
85  		};
86  		private final File f;
87  		private final String path;
88  
89  		private final String uri;
90  
91  		private final boolean includeChildren, allowDeletes;
92  
93  		public FileResource(File f, String path, boolean allowDeletes, boolean includeChildren) {
94  			this.f = f;
95  			this.path = path;
96  			this.uri = "servlet:/"+(path == null ? "" : path);
97  			this.includeChildren = includeChildren;
98  			this.allowDeletes = allowDeletes;
99  		}
100 
101 		@Html(format=HtmlFormat.HTML_CDC)
102 		public List<Action> getActions() throws Exception {
103 			List<Action> l = new ArrayList<>();
104 			if (f.canRead() && ! f.isDirectory()) {
105 				l.add(new Action("view", uri + "?method=VIEW"));
106 				l.add(new Action("highlighted", uri + "?method=VIEW&highlight=true"));
107 				l.add(new Action("parsed", uri + "?method=PARSE"));
108 				l.add(new Action("download", uri + "?method=DOWNLOAD"));
109 				if (allowDeletes)
110 					l.add(new Action("delete", uri + "?method=DELETE"));
111 			}
112 			return l;
113 		}
114 
115 		public Set<FileResource> getFiles() {
116 			if (f.isFile() || ! includeChildren)
117 				return null;
118 			Set<FileResource> s = new TreeSet<>(FILE_COMPARATOR);
119 			for (File fc : f.listFiles(FILE_FILTER))
120 				s.add(new FileResource(fc, (path != null ? (path + '/') : "") + urlEncode(fc.getName()), allowDeletes, false));
121 			return s;
122 		}
123 
124 		public Date getLastModified() {
125 			return new Date(f.lastModified());
126 		}
127 
128 		public LinkString getName() {
129 			return new LinkString(f.getName(), uri);
130 		}
131 
132 		public long getSize() {
133 			return f.isDirectory() ? f.listFiles().length : f.length();
134 		}
135 
136 		public String getType() {
137 			return (f.isDirectory() ? "dir" : "file");
138 		}
139 	}

213 	@RestOp(
214 		method="VIEW",
215 		path="/*",
216 		summary="View contents of log file",
217 		description="View the contents of a log file."
218 	)
219 	public void viewFile(
220 			RestResponse res,
221 			@Path("/*") String path,
222 			@Query(name="highlight", schema=@Schema(d="Add severity color highlighting.")) boolean highlight,
223 			@Query(name="start", schema=@Schema(d="Start timestamp (ISO8601, full or partial).\nDon't print lines logged before the specified timestamp.\nUse any of the following formats: yyyy, yyyy-MM, yyyy-MM-dd, yyyy-MM-ddThh, yyyy-MM-ddThh:mm, yyyy-MM-ddThh:mm:ss, yyyy-MM-ddThh:mm:ss.SSS")) String start,
224 			@Query(name="end", schema=@Schema(d="End timestamp (ISO8601, full or partial).\nDon't print lines logged after the specified timestamp.\nUse any of the following formats: yyyy, yyyy-MM, yyyy-MM-dd, yyyy-MM-ddThh, yyyy-MM-ddThh:mm, yyyy-MM-ddThh:mm:ss, yyyy-MM-ddThh:mm:ss.SSS")) String end,
225 			@Query(name="thread", schema=@Schema(d="Thread name filter.\nOnly show log entries with the specified thread name.")) String thread,
226 			@Query(name="loggers", schema=@Schema(d="Logger filter (simple class name).\nOnly show log entries if they were produced by one of the specified loggers.")) String[] loggers,
227 			@Query(name="severity",schema=@Schema( d="Severity filter.\nOnly show log entries with the specified severity.")) String[] severity
228 		) throws NotFound, MethodNotAllowed, IOException {
229 
230 		File f = getFile(path);
231 
232 		Date startDate = parseIsoDate(start), endDate = parseIsoDate(end);
233 
234 		if (! highlight) {
235 			Object o = getReader(f, startDate, endDate, thread, loggers, severity);
236 			res.setContentType("text/plain");
237 			if (o instanceof Reader)
238 				res.setContent(o);
239 			else {
240 				try (LogParser p = (LogParser)o; Writer w = res.getNegotiatedWriter()) {
241 					p.writeTo(w);
242 				}
243 			}
244 			return;
245 		}
246 
247 		res.setContentType("text/html");
248 		try (PrintWriter w = res.getNegotiatedWriter()) {
249 			w.println("<html><body style='font-family:monospace;font-size:8pt;white-space:pre;'>");
250 			try (LogParser lp = getLogParser(f, startDate, endDate, thread, loggers, severity)) {
251 				if (! lp.hasNext())
252 					w.append("<span style='color:gray'>[EMPTY]</span>");
253 				else for (LogParser.Entry le : lp) {
254 					char s = le.severity.charAt(0);
255 					String color = "black";
256 					//SEVERE|WARNING|INFO|CONFIG|FINE|FINER|FINEST
257 					if (s == 'I')
258 						color = "#006400";
259 					else if (s == 'W')
260 						color = "#CC8400";
261 					else if (s == 'E' || s == 'S')
262 						color = "#DD0000";
263 					else if (s == 'D' || s == 'F' || s == 'T')
264 						color = "#000064";
265 					w.append("<span style='color:").append(color).append("'>");
266 					le.appendHtml(w).append("</span>");
267 				}
268 				w.append("</body></html>");
269 			}
270 		}
271 	}

305 	private void deleteFile(File f) {
306 		if (! allowDeletes)
307 			throw new MethodNotAllowed("DELETE not enabled");
308 		if (f.isDirectory()) {
309 			File[] files = f.listFiles();
310 			if (files != null) {
311 				for (File fc : files)
312 					deleteFile(fc);
313 			}
314 		}
315 		if (! f.delete())
316 			throw new Forbidden("Could not delete file {0}", f.getAbsolutePath()) ;
317 	}
318 
319 	private File getFile(String path) throws NotFound {
320 		if (path == null)
321 			return logDir;
322 		File f = new File(logDir.getAbsolutePath() + '/' + path);
323 		if (f.exists())
324 			return f;
325 		throw new NotFound("File not found.");
326 	}
327 
328 	private LogParser getLogParser(File f, final Date start, final Date end, final String thread, final String[] loggers, final String[] severity) throws IOException {
329 		return new LogParser(leFormatter, f, start, end, thread, loggers, severity);
330 	}
331 
332 	private Object getReader(File f, final Date start, final Date end, final String thread, final String[] loggers, final String[] severity) throws IOException {
333 		if (start == null && end == null && thread == null && loggers == null)
334 			return getReader(f);
335 		return getLogParser(f, start, end, thread, loggers, severity);
336 	}
