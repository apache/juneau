13  package org.apache.juneau;
14  
15  import static org.apache.juneau.common.internal.StringUtils.*;
16  
17  import java.text.*;
18  
19  /**
20   * An extension of {@link AssertionError} with helper constructors for messages with message-style arguments.
21   *
22   * <h5 class='section'>See Also:</h5><ul>
23   * </ul>
24   *
25   * @serial exclude
26   */
27  public class BasicAssertionError extends AssertionError {
28  
29  	private static final long serialVersionUID = 1L;
30  
31  	/**
32  	 * Constructor.
33  	 *
34  	 * @param message The {@link MessageFormat}-style message.
35  	 * @param args Optional {@link MessageFormat}-style arguments.
36  	 */
37  	public BasicAssertionError(String message, Object...args) {
38  		super(format(message, args));
39  	}
40  
41  	/**
42  	 * Constructor.
43  	 *
44  	 * @param cause The cause of this exception.
45  	 * @param message The {@link MessageFormat}-style message.
46  	 * @param args Optional {@link MessageFormat}-style arguments.
47  	 */
48  	public BasicAssertionError(Throwable cause, String message, Object...args) {
49  		this(getMessage(cause, message, null), args);
50  		initCause(cause);
51  	}
52  
53  	/**
54  	 * Finds the message.
55  	 *
56  	 * @param cause The cause.
57  	 * @param msg The message.
58  	 * @param def The default value if both above are <jk>null</jk>.
59  	 * @return The resolved message.
60  	 */
61  	protected static final String getMessage(Throwable cause, String msg, String def) {
62  		if (msg != null)
63  			return msg;
64  		if (cause != null)
65  			return cause.getMessage();
66  		return def;
67  	}
68  }
