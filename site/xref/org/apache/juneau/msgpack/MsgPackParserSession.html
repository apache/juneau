180 	/*
181 	 * Workhorse method.
182 	 */
183 	private <T> T parseAnything(ClassMeta<?> eType, MsgPackInputStream is, Object outer, BeanPropertyMeta pMeta) throws IOException, ParseException, ExecutableException {
184 
185 		if (eType == null)
186 			eType = object();
187 		ObjectSwap<T,Object> swap = (ObjectSwap<T,Object>)eType.getSwap(this);
188 		BuilderSwap<T,Object> builder = (BuilderSwap<T,Object>)eType.getBuilderSwap(this);
189 		ClassMeta<?> sType = null;
190 		if (nn(builder))
191 			sType = builder.getBuilderClassMeta(this);
192 		else if (nn(swap))
193 			sType = swap.getSwapClassMeta(this);
194 		else
195 			sType = eType;
196 
197 		if (sType.isOptional())
198 			return (T)opt(parseAnything(eType.getElementType(), is, outer, pMeta));
199 
200 		setCurrentClass(sType);
201 
202 		Object o = null;
203 		DataType dt = is.readDataType();
204 		int length = (int)is.readLength();
205 
206 		if (dt != DataType.NULL) {
207 			if (dt == BOOLEAN)
208 				o = is.readBoolean();
209 			else if (dt == INT)
210 				o = is.readInt();
211 			else if (dt == LONG)
212 				o = is.readLong();
213 			else if (dt == FLOAT)
214 				o = is.readFloat();
215 			else if (dt == DOUBLE)
216 				o = is.readDouble();
217 			else if (dt == STRING)
218 				o = trim(is.readString());
219 			else if (dt == BIN)
220 				o = is.readBinary();
221 			else if (dt == ARRAY && sType.isObject()) {
222 				JsonList jl = new JsonList(this);
223 				for (int i = 0; i < length; i++)
224 					jl.add(parseAnything(object(), is, outer, pMeta));
225 				o = jl;
226 			} else if (dt == MAP && sType.isObject()) {
227 				JsonMap jm = new JsonMap(this);
228 				for (int i = 0; i < length; i++)
229 					jm.put((String)parseAnything(string(), is, outer, pMeta), parseAnything(object(), is, jm, pMeta));
230 				o = cast(jm, pMeta, eType);
231 			}
232 
233 			if (sType.isObject()) {
234 				// Do nothing.
235 			} else if (sType.isBoolean() || sType.isCharSequence() || sType.isChar() || sType.isNumber() || sType.isByteArray()) {
236 				o = convertToType(o, sType);
237 			} else if (sType.isMap()) {
238 				if (dt == MAP) {
239 					Map m = (sType.canCreateNewInstance(outer) ? (Map)sType.newInstance(outer) : newGenericMap(sType));
240 					for (int i = 0; i < length; i++) {
241 						Object key = parseAnything(sType.getKeyType(), is, outer, pMeta);
242 						ClassMeta<?> vt = sType.getValueType();
243 						Object value = parseAnything(vt, is, m, pMeta);
244 						setName(vt, value, key);
245 						m.put(key, value);
246 					}
247 					o = m;
248 				} else {
249 					throw new ParseException(this, "Invalid data type {0} encountered for parse type {1}", dt, sType);
250 				}
251 			} else if (nn(builder) || sType.canCreateNewBean(outer)) {
252 				if (dt == MAP) {
253 					BeanMap m = builder == null ? newBeanMap(outer, sType.getInnerClass()) : toBeanMap(builder.create(this, eType));
254 					for (int i = 0; i < length; i++) {
255 						String pName = parseAnything(string(), is, m.getBean(false), null);
256 						BeanPropertyMeta bpm = m.getPropertyMeta(pName);
257 						if (bpm == null) {
258 							if (pName.equals(getBeanTypePropertyName(eType)))
259 								parseAnything(string(), is, null, null);
260 							else
261 								onUnknownProperty(pName, m, parseAnything(string(), is, null, null));
262 						} else {
263 							ClassMeta<?> cm = bpm.getClassMeta();
264 							Object value = parseAnything(cm, is, m.getBean(false), bpm);
265 							setName(cm, value, pName);
266 							try {
267 								bpm.set(m, pName, value);
268 							} catch (BeanRuntimeException e) {
269 								onBeanSetterException(pMeta, e);
270 								throw e;
271 							}
272 						}
273 					}
274 					o = builder == null ? m.getBean() : builder.build(this, m.getBean(), eType);
275 				} else {
276 					throw new ParseException(this, "Invalid data type {0} encountered for parse type {1}", dt, sType);
277 				}
278 			} else if (sType.canCreateNewInstanceFromString(outer) && dt == STRING) {
279 				o = sType.newInstanceFromString(outer, o == null ? "" : o.toString());
280 			} else if (sType.isCollection()) {
281 				if (dt == MAP) {
282 					JsonMap m = new JsonMap(this);
283 					for (int i = 0; i < length; i++)
284 						m.put((String)parseAnything(string(), is, outer, pMeta), parseAnything(object(), is, m, pMeta));
285 					o = cast(m, pMeta, eType);
286 				} else if (dt == ARRAY) {
287 					Collection l = (sType.canCreateNewInstance(outer) ? (Collection)sType.newInstance() : new JsonList(this));
288 					for (int i = 0; i < length; i++)
289 						l.add(parseAnything(sType.getElementType(), is, l, pMeta));
290 					o = l;
291 				} else {
292 					throw new ParseException(this, "Invalid data type {0} encountered for parse type {1}", dt, sType);
293 				}
294 			} else if (sType.isArray() || sType.isArgs()) {
295 				if (dt == MAP) {
296 					JsonMap m = new JsonMap(this);
297 					for (int i = 0; i < length; i++)
298 						m.put((String)parseAnything(string(), is, outer, pMeta), parseAnything(object(), is, m, pMeta));
299 					o = cast(m, pMeta, eType);
300 				} else if (dt == ARRAY) {
301 					Collection l = (sType.isCollection() && sType.canCreateNewInstance(outer) ? (Collection)sType.newInstance() : new JsonList(this));
302 					for (int i = 0; i < length; i++)
303 						l.add(parseAnything(sType.isArgs() ? sType.getArg(i) : sType.getElementType(), is, l, pMeta));
304 					o = toArray(sType, l);
305 				} else {
306 					throw new ParseException(this, "Invalid data type {0} encountered for parse type {1}", dt, sType);
307 				}
308 			} else if (dt == MAP) {
309 				JsonMap m = new JsonMap(this);
310 				for (int i = 0; i < length; i++)
311 					m.put((String)parseAnything(string(), is, outer, pMeta), parseAnything(object(), is, m, pMeta));
312 				if (m.containsKey(getBeanTypePropertyName(eType)))
313 					o = cast(m, pMeta, eType);
314 				else if (nn(sType.getProxyInvocationHandler()))
315 					o = newBeanMap(outer, sType.getInnerClass()).load(m).getBean();
316 				else
317 					throw new ParseException(this, "Class ''{0}'' could not be instantiated.  Reason: ''{1}''", sType.getInnerClass().getName(), sType.getNotABeanReason());
318 			} else {
319 				throw new ParseException(this, "Invalid data type {0} encountered for parse type {1}", dt, sType);
320 			}
321 		}
322 
323 		if (nn(swap) && nn(o))
324 			o = unswap(swap, o, eType);
325 
326 		if (nn(outer))
327 			setParent(eType, o, outer);
328 
329 		return (T)o;
330 	}
331 
332 	@Override /* Overridden from ParserSession */
333 	protected <T> T doParse(ParserPipe pipe, ClassMeta<T> type) throws IOException, ParseException, ExecutableException {
334 		try (MsgPackInputStream is = new MsgPackInputStream(pipe)) {
335 			return parseAnything(type, is, getOuter(), null);
336 		}
337 	}
