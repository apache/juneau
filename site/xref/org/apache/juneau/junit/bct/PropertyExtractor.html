17  package org.apache.juneau.junit.bct;
18  
19  /**
20   * Interface for custom property extraction strategies in the Bean-Centric Testing framework.
21   *
22   * <p>Property extractors define how the converter accesses object properties during nested
23   * field navigation (e.g., {@code "user.address.city"}). The framework uses a chain-of-responsibility
24   * pattern, trying each registered extractor until one can handle the property access.</p>
25   *
26   * <h5 class='section'>Extraction Strategy:</h5>
27   * <p>The two-phase approach ensures efficient and flexible property access:</p>
28   * <ol>
29   *    <li><b>{@link #canExtract(BeanConverter, Object, String)}:</b> Quick check if this extractor can handle the property</li>
30   *    <li><b>{@link #extract(BeanConverter, Object, String)}:</b> Perform the actual property extraction</li>
31   * </ol>
32   *
33   * <h5 class='section'>Common Use Cases:</h5>
34   * <ul>
35   *    <li><b>JavaBean properties:</b> Standard getter methods and public fields</li>
36   *    <li><b>Map-style access:</b> Key-based property retrieval from Map objects</li>
37   *    <li><b>Collection indices:</b> Numeric access for arrays and lists</li>
38   *    <li><b>Custom data structures:</b> Domain-specific property access patterns</li>
39   *    <li><b>Dynamic properties:</b> Computed or cached property values</li>
40   * </ul>
41   *
42   * <h5 class='section'>Implementation Example:</h5>
43   * <p class='bjava'>
44   *    <jc>// Custom extractor for database entities</jc>
45   *    <jk>public class</jk> DatabaseEntityExtractor <jk>implements</jk> PropertyExtractor {
46   *
47   *       <ja>@Override</ja>
48   *       <jk>public boolean</jk> canExtract(BeanConverter <jv>converter</jv>, Object <jv>obj</jv>, String <jv>property</jv>) {
49   *          <jk>return</jk> <jv>obj</jv> <jk>instanceof</jk> DatabaseEntity;
50   *       }
51   *
52   *       <ja>@Override</ja>
53   *       <jk>public</jk> Object extract(BeanConverter <jv>converter</jv>, Object <jv>obj</jv>, String <jv>property</jv>) {
54   *          DatabaseEntity <jv>entity</jv> = (DatabaseEntity) <jv>obj</jv>;
55   *          <jk>switch</jk> (<jv>property</jv>) {
56   *             <jk>case</jk> <js>"id"</js>: <jk>return</jk> <jv>entity</jv>.getPrimaryKey();
57   *             <jk>case</jk> <js>"lastModified"</js>: <jk>return</jk> <jv>entity</jv>.getTimestamp();
58   *             <jk>case</jk> <js>"metadata"</js>: <jk>return</jk> <jv>entity</jv>.getMetadata().asMap();
59   *             <jk>default</jk>: <jk>return</jk> <jv>entity</jv>.getAttribute(<jv>property</jv>);
60   *          }
61   *       }
62   *    }
63   * </p>
64   *
65   * <h5 class='section'>Registration and Usage:</h5>
66   * <p class='bjava'>
67   *    <jk>var</jk> <jv>converter</jv> = BasicBeanConverter.<jsm>builder</jsm>()
68   *       .defaultSettings() <jc>// Adds standard extractors</jc>
69   *       .addPropertyExtractor(<jk>new</jk> DatabaseEntityExtractor())
70   *       .addPropertyExtractor((<jp>conv</jp>, <jp>obj</jp>, <jp>prop</jp>) -&gt; {
71   *          <jc>// Lambda-based extractor for simple cases</jc>
72   *          <jk>if</jk> (<jp>obj</jp> <jk>instanceof</jk> MyConfig <jv>config</jv> &amp;&amp; <js>"timeout"</js>.equals(<jp>prop</jp>)) {
73   *             <jk>return</jk> <jv>config</jv>.getTimeoutMillis();
74   *          }
75   *          <jk>return</jk> <jk>null</jk>; <jc>// Let next extractor try</jc>
76   *       })
77   *       .build();
78   * </p>
79   *
80   * <h5 class='section'>Best Practices:</h5>
81   * <ul>
82   *    <li><b>Fast canExtract() checks:</b> Use efficient type checking and avoid expensive operations</li>
83   *    <li><b>Handle edge cases:</b> Gracefully handle null objects and invalid property names</li>
84   *    <li><b>Consider caching:</b> Cache reflection results for better performance</li>
85   * </ul>
86   *
87   * @see PropertyExtractors
88   * @see BasicBeanConverter.Builder#addPropertyExtractor(PropertyExtractor)
89   * @see BeanConverter#getProperty(Object, String)
90   */
91  public interface PropertyExtractor {
92  
93  	/**
94  	 * Determines if this extractor can handle property access for the given object and property name.
95  	 *
96  	 * <p>This method should perform a quick check to determine compatibility without doing
97  	 * expensive operations. It's called frequently during property navigation.</p>
98  	 *
99  	 * @param converter The bean converter instance (for recursive operations if needed)
100 	 * @param o The object to extract the property from
101 	 * @param key The property name to extract
102 	 * @return {@code true} if this extractor can handle the property access, {@code false} otherwise
103 	 */
104 	boolean canExtract(BeanConverter converter, Object o, String key);
105 
106 	/**
107 	 * Extracts the specified property value from the given object.
108 	 *
109 	 * <p>This method is only called after {@link #canExtract(BeanConverter, Object, String)}
110 	 * returns {@code true}. It should perform the actual property extraction and return
111 	 * the property value.</p>
112 	 *
113 	 * @param converter The bean converter instance (for recursive operations if needed)
114 	 * @param o The object to extract the property from
115 	 * @param key The property name to extract
116 	 * @return The property value
117 	 * @throws PropertyNotFoundException if the property cannot be found on the object
118 	 */
119 	Object extract(BeanConverter converter, Object o, String key);
120 }
