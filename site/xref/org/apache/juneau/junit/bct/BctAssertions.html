33  /**
34   * Comprehensive utility class for Bean-Centric Tests (BCT) and general testing operations.
35   *
36   * <p>This class extends the functionality provided by the JUnit Assertions class, with particular emphasis
37   * on the Bean-Centric Testing (BCT) framework. BCT enables sophisticated assertion patterns for
38   * testing object properties, collections, maps, and complex nested structures with minimal code.</p>
39   *
40   * <h5 class='section'>Bean-Centric Testing (BCT) Framework:</h5>
41   * <p>The BCT framework consists of several key components:</p>
42   * <ul>
43   *    <li><b>{@link BeanConverter}:</b> Core interface for object conversion and property access</li>
44   *    <li><b>{@link BasicBeanConverter}:</b> Default implementation with extensible type handlers</li>
45   *    <li><b>Assertion Methods:</b> High-level testing methods that leverage the converter framework</li>
46   * </ul>
47   *
48   * <h5 class='section'>Primary BCT Assertion Methods:</h5>
49   * <dl>
50   *    <dt><b>{@link #assertBean(Object, String, String)}</b></dt>
51   *    <dd>Tests object properties with nested syntax support and collection iteration</dd>
52   *
53   *    <dt><b>{@link #assertBeans(Collection, String, String...)}</b></dt>
54   *    <dd>Tests collections of objects by extracting and comparing specific fields</dd>
55   *
56   *    <dt><b>{@link #assertMapped(Object, java.util.function.BiFunction, String, String)}</b></dt>
57   *    <dd>Tests custom property access using BiFunction for non-standard objects</dd>
58   *
59   *    <dt><b>{@link #assertList(List, Object...)}</b></dt>
60   *    <dd>Tests list/collection elements with varargs for expected values</dd>
61   * </dl>
62   *
63   * <h5 class='section'>BCT Advanced Features:</h5>
64   * <ul>
65   *    <li><b>Nested Property Syntax:</b> "address{street,city}" for testing nested objects</li>
66   *    <li><b>Collection Iteration:</b> "#{address{street,city}}" syntax for testing all elements</li>
67   *    <li><b>Universal Size Properties:</b> "length" and "size" work on all collection types</li>
68   *    <li><b>Array/List Access:</b> Numeric indices for element-specific testing</li>
69   *    <li><b>Method Chaining:</b> Fluent setters can be tested directly</li>
70   *    <li><b>Direct Field Access:</b> Public fields accessed without getters</li>
71   *    <li><b>Map Key Access:</b> Including special <js>"&lt;null&gt;"</js> syntax for null keys</li>
72   * </ul>
73   *
74   * <h5 class='section'>Converter Extensibility:</h5>
75   * <p>The BCT framework is built on the extensible {@link BasicBeanConverter} which allows:</p>
76   * <ul>
77   *    <li><b>Custom Stringifiers:</b> Type-specific string conversion logic</li>
78   *    <li><b>Custom Listifiers:</b> Collection-type conversion for iteration</li>
79   *    <li><b>Custom Swappers:</b> Object transformation before conversion</li>
80   *    <li><b>Custom PropertyExtractors:</b> Property extraction</li>
81   *    <li><b>Configurable Settings:</b> Formatting, delimiters, and display options</li>
82   * </ul>
83   *
84   * <h5 class='section'>Usage Examples:</h5>
85   *
86   * <p><b>Basic Property Testing:</b></p>
87   * <p class='bjava'>
88   *    <jc>// Test multiple properties</jc>
89   *    <jsm>assertBean</jsm>(<jv>user</jv>, <js>"name,age,active"</js>, <js>"John,30,true"</js>);
90   *
91   *    <jc>// Test nested properties - user has getAddress() returning Address with getStreet() and getCity()</jc>
92   *    <jsm>assertBean</jsm>(<jv>user</jv>, <js>"name,address{street,city}"</js>, <js>"John,{123 Main St,Springfield}"</js>);
93   * </p>
94   *
95   * <p><b>Collection and Array Testing:</b></p>
96   * <p class='bjava'>
97   *    <jc>// Test collection size and iterate over all elements - order has getItems() returning List&lt;Product&gt; where Product has getName()</jc>
98   *    <jsm>assertBean</jsm>(<jv>order</jv>, <js>"items{length,#{name}}"</js>, <js>"{3,[{Laptop},{Phone},{Tablet}]}"</js>);
99   *
100  *    <jc>// Test specific array elements - listOfData is a List&lt;DataObject&gt; where DataObject has getData()</jc>
101  *    <jsm>assertBean</jsm>(<jv>listOfData</jv>, <js>"0{data},1{data}"</js>, <js>"{100},{200}"</js>);
102  * </p>
103  *
104  * <p><b>Collection Testing:</b></p>
105  * <p class='bjava'>
106  *    <jc>// Test list elements</jc>
107  *    <jsm>assertList</jsm>(tags, <js>"red"</js>, <js>"green"</js>, <js>"blue"</js>);
108  *
109  *    <jc>// Test map entries using assertBean</jc>
110  *    <jsm>assertBean</jsm>(<jv>config</jv>, <js>"timeout,retries"</js>, <js>"30000,3"</js>);
111  * </p>
112  *
113  * <p><b>Custom Property Access:</b></p>
114  * <p class='bjava'>
115  *    <jc>// Test with custom accessor function</jc>
116  *    <jsm>assertMapped</jsm>(<jv>myObject</jv>, (<jp>obj</jp>, <jp>prop</jp>) -> <jp>obj</jp>.getProperty(<jp>prop</jp>),
117  *       <js>"prop1,prop2"</js>, <js>"value1,value2"</js>);
118  * </p>
119  *
120  * <h5 class='section'>Performance and Thread Safety:</h5>
121  * <p>The BCT framework is designed for high performance with:</p>
122  * <ul>
123  *    <li><b>Caching:</b> Type-to-handler mappings cached for fast lookup</li>
124  *    <li><b>Thread Safety:</b> All operations are thread-safe for concurrent testing</li>
125  *    <li><b>Minimal Allocation:</b> Efficient object reuse and minimal temporary objects</li>
126  * </ul>
127  *
128  * @see BeanConverter
129  * @see BasicBeanConverter
130  */
131 public class BctAssertions {

177 	 *
178 	 * @param args Assertion configuration. See {@link #args()} for usage examples.
179 	 * @param actual The bean to test. Must not be null.

181 	 * @param expected The expected property values as a comma-delimited string.
182 	 * @see #assertBean(Object, String, String)
183 	 * @see #args()

185 	public static void assertBean(AssertionArgs args, Object actual, String fields, String expected) {
186 		assertNotNull(actual, "Actual was null.");
187 		assertArgNotNull("args", args);
188 		assertArgNotNull("fields", fields);
189 		assertArgNotNull("expected", expected);
190 		assertEquals(expected, tokenize(fields).stream().map(x -> args.getBeanConverter().orElse(DEFAULT_CONVERTER).getNested(actual, x)).collect(joining(",")),
191 			args.getMessage("Bean assertion failed."));
192 	}

194 	/**
195 	 * Asserts that the fields/properties on the specified bean are the specified values after being converted to strings.
196 	 *
197 	 * <p>This is the primary method for Bean-Centric Tests (BCT), supporting extensive property validation
198 	 * patterns including nested objects, collections, arrays, method chaining, direct field access, collection iteration
199 	 * with <js>"#{property}"</js> syntax, and universal <js>"length"</js>/<js>"size"</js> properties for all collection types.</p>
200 	 *
201 	 * <p>The method uses the {@link BasicBeanConverter#DEFAULT} converter internally for object introspection
202 	 * and value extraction. The converter provides sophisticated property access through the {@link BeanConverter}
203 	 * interface, supporting multiple fallback mechanisms for accessing object properties and values.</p>
204 	 *
205 	 * <h5 class='section'>Basic Usage:</h5>
206 	 * <p class='bjava'>
207 	 *    <jc>// Test multiple properties</jc>
208 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"prop1,prop2,prop3"</js>, <js>"val1,val2,val3"</js>);
209 	 *
210 	 *    <jc>// Test single property</jc>
211 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"name"</js>, <js>"John"</js>);
212 	 * </p>
213 	 *
214 	 * <h5 class='section'>Nested Property Testing:</h5>
215 	 * <p class='bjava'>
216 	 *    <jc>// Test nested bean properties</jc>
217 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"name,address{street,city,state}"</js>, <js>"John,{123 Main St,Springfield,IL}"</js>);
218 	 *
219 	 *    <jc>// Test arbitrarily deep nesting</jc>
220 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"name,person{address{geo{lat,lon}}}"</js>, <js>"John,{{{40.7,-74.0}}}"</js>);
221 	 * </p>
222 	 *
223 	 * <h5 class='section'>Array, List, and Stream Testing:</h5>
224 	 * <p class='bjava'>
225 	 *    <jc>// Test array/list elements by index - items is a String[] or List&lt;String&gt;</jc>
226 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"items{0,1,2}"</js>, <js>"{item1,item2,item3}"</js>);
227 	 *
228 	 *    <jc>// Test nested properties within array elements - orders is a List&lt;Order&gt; where Order has getId() and getTotal()</jc>
229 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"orders{0{id,total}}"</js>, <js>"{{123,99.95}}"</js>);
230 	 *
231 	 *    <jc>// Test array length property - items can be any array or collection type</jc>
232 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"items{length}"</js>, <js>"{5}"</js>);
233 	 *
234 	 *    <jc>// Works with any iterable type including Streams - userStream returns a Stream&lt;User&gt; where User has getName()</jc>
235 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"userStream{#{name}}"</js>, <js>"[{Alice},{Bob}]"</js>);
236 	 * </p>
237 	 *
238 	 * <h5 class='section'>Collection Iteration Syntax:</h5>
239 	 * <p class='bjava'>
240 	 *    <jc>// Test properties across ALL elements in a collection using #{...} syntax - userList is a List&lt;User&gt; where User has getName()</jc>
241 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"userList{#{name}}"</js>, <js>"[{John},{Jane},{Bob}]"</js>);
242 	 *
243 	 *    <jc>// Test multiple properties from each element - orderList is a List&lt;Order&gt; where Order has getId() and getStatus()</jc>
244 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"orderList{#{id,status}}"</js>, <js>"[{123,ACTIVE},{124,PENDING}]"</js>);
245 	 *
246 	 *    <jc>// Works with nested properties within each element - customers is a List&lt;Customer&gt; where Customer has getAddress() returning Address with getCity()</jc>
247 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"customers{#{address{city}}}"</js>, <js>"[{{New York}},{{Los Angeles}}]"</js>);
248 	 *
249 	 *    <jc>// Works with arrays and any iterable collection type (including Streams)</jc>
250 	 *    <jsm>assertBean</jsm>(<jv>config</jv>, <js>"itemArray{#{type}}"</js>, <js>"[{String},{Integer},{Boolean}]"</js>);
251 	 *    <jsm>assertBean</jsm>(<jv>data</jv>, <js>"statusSet{#{name}}"</js>, <js>"[{ACTIVE},{PENDING},{CANCELLED}]"</js>);
252 	 *    <jsm>assertBean</jsm>(<jv>processor</jv>, <js>"dataStream{#{value}}"</js>, <js>"[{A},{B},{C}]"</js>);
253 	 * </p>
254 	 *
255 	 * <h5 class='section'>Universal Collection Size Properties:</h5>
256 	 * <p class='bjava'>
257 	 *    <jc>// Both 'length' and 'size' work universally across all collection types</jc>
258 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"myArray{length}"</js>, <js>"{5}"</js>);        <jc>// Arrays</jc>
259 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"myArray{size}"</js>, <js>"{5}"</js>);          <jc>// Also works for arrays</jc>
260 	 *
261 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"myList{size}"</js>, <js>"{3}"</js>);           <jc>// Collections</jc>
262 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"myList{length}"</js>, <js>"{3}"</js>);         <jc>// Also works for collections</jc>
263 	 *
264 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"myMap{size}"</js>, <js>"{7}"</js>);            <jc>// Maps</jc>
265 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"myMap{length}"</js>, <js>"{7}"</js>);          <jc>// Also works for maps</jc>
266 	 * </p>
267 	 *
268 	 * <h5 class='section'>Class Name Testing:</h5>
269 	 * <p class='bjava'>
270 	 *    <jc>// Test class properties (prefer simple names for maintainability)</jc>
271 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"obj{class{simpleName}}"</js>, <js>"{{MyClass}}"</js>);
272 	 *
273 	 *    <jc>// Test full class names when needed</jc>
274 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"obj{class{name}}"</js>, <js>"{{com.example.MyClass}}"</js>);
275 	 * </p>
276 	 *
277 	 * <h5 class='section'>Method Chaining Support:</h5>
278 	 * <p class='bjava'>
279 	 *    <jc>// Test fluent setter chains (returns same object)</jc>
280 	 *    <jsm>assertBean</jsm>(
281 	 *       <jv>item</jv>.setType(<js>"foo"</js>).setFormat(<js>"bar"</js>).setDefault(<js>"baz"</js>),
282 	 *       <js>"type,format,default"</js>,
283 	 *       <js>"foo,bar,baz"</js>
284 	 *    );
285 	 * </p>
286 	 *
287 	 * <h5 class='section'>Advanced Collection Analysis:</h5>
288 	 * <p class='bjava'>
289 	 *    <jc>// Combine size/length, metadata, and content iteration in single assertions - users is a List&lt;User&gt;</jc>
290 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"users{length,class{simpleName},#{name}}"</js>,
291 	 *       <js>"{3,{ArrayList},[{John},{Jane},{Bob}]}"</js>);
292 	 *
293 	 *    <jc>// Comprehensive collection validation with multiple iteration patterns - items is a List&lt;Product&gt; where Product has getName() and getPrice()</jc>
294 	 *    <jsm>assertBean</jsm>(<jv>order</jv>, <js>"items{size,#{name},#{price}}"</js>,
295 	 *       <js>"{3,[{Laptop},{Phone},{Tablet}],[{999.99},{599.99},{399.99}]}"</js>);
296 	 *
297 	 *    <jc>// Perfect for validation testing - verify error count and details; errors is a List&lt;ValidationError&gt; where ValidationError has getField() and getCode()</jc>
298 	 *    <jsm>assertBean</jsm>(<jv>result</jv>, <js>"errors{length,#{field},#{code}}"</js>,
299 	 *       <js>"{2,[{email},{password}],[{E001},{E002}]}"</js>);
300 	 *
301 	 *    <jc>// Mixed collection types with consistent syntax - results and metadata are different collection types</jc>
302 	 *    <jsm>assertBean</jsm>(<jv>response</jv>, <js>"results{size},metadata{length}"</js>, <js>"{25},{4}"</js>);
303 	 * </p>
304 	 *
305 	 * <h5 class='section'>Direct Field Access:</h5>
306 	 * <p class='bjava'>
307 	 *    <jc>// Test public fields directly (no getters required)</jc>
308 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"f1,f2,f3"</js>, <js>"val1,val2,val3"</js>);
309 	 *
310 	 *    <jc>// Test field properties with chaining</jc>
311 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"f1{length},f2{class{simpleName}}"</js>, <js>"{5},{{String}}"</js>);
312 	 * </p>
313 	 *
314 	 * <h5 class='section'>Map Testing:</h5>
315 	 * <p class='bjava'>
316 	 *    <jc>// Test map values by key</jc>
317 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"configMap{timeout,retries}"</js>, <js>"{30000,3}"</js>);
318 	 *
319 	 *    <jc>// Test map size</jc>
320 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"settings{size}"</js>, <js>"{5}"</js>);
321 	 *
322 	 *    <jc>// Test null keys using special &lt;null&gt; syntax</jc>
323 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"mapWithNullKey{&lt;null&gt;}"</js>, <js>"{nullKeyValue}"</js>);
324 	 * </p>
325 	 *
326 	 * <h5 class='section'>Collection and Boolean Values:</h5>
327 	 * <p class='bjava'>
328 	 *    <jc>// Test boolean values</jc>
329 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"enabled,visible"</js>, <js>"true,false"</js>);
330 	 *
331 	 *    <jc>// Test enum collections</jc>
332 	 *    <jsm>assertBean</jsm>(<jv>myBean</jv>, <js>"statuses"</js>, <js>"[ACTIVE,PENDING]"</js>);
333 	 * </p>
334 	 *
335 	 * <h5 class='section'>Value Syntax Rules:</h5>
336 	 * <ul>
337 	 *    <li><b>Simple values:</b> <js>"value"</js> for direct property values</li>
338 	 *    <li><b>Nested values:</b> <js>"{value}"</js> for single-level nested properties</li>
339 	 *    <li><b>Deep nested values:</b> <js>"{{value}}"</js>, <js>"{{{value}}}"</js> for multiple nesting levels</li>
340 	 *    <li><b>Array/Collection values:</b> <js>"[item1,item2]"</js> for collections</li>
341 	 *    <li><b>Collection iteration:</b> <js>"#{property}"</js> iterates over ALL collection elements, returns <js>"[{val1},{val2}]"</js></li>
342 	 *    <li><b>Universal size properties:</b> <js>"length"</js> and <js>"size"</js> work on arrays, collections, and maps</li>
343 	 *    <li><b>Boolean values:</b> <js>"true"</js>, <js>"false"</js></li>
344 	 *    <li><b>Null values:</b> <js>"null"</js></li>
345 	 * </ul>
346 	 *
347 	 * <h5 class='section'>Property Access Priority:</h5>
348 	 * <ol>
349 	 *    <li><b>Collection/Array access:</b> Numeric indices for arrays/lists (e.g., <js>"0"</js>, <js>"1"</js>)</li>
350 	 *    <li><b>Universal size properties:</b> <js>"length"</js> and <js>"size"</js> for arrays, collections, and maps</li>
351 	 *    <li><b>Map key access:</b> Direct key lookup for Map objects (including <js>"&lt;null&gt;"</js> for null keys)</li>
352 	 *    <li><b>is{Property}()</b> methods (for boolean properties)</li>
353 	 *    <li><b>get{Property}()</b> methods</li>
354 	 *    <li><b>Public fields</b> (direct field access)</li>
355 	 * </ol>
356 	 *
357 	 * @param actual The bean object to test. Must not be null.
358 	 * @param fields Comma-delimited list of property names to test. Supports nested syntax with {}.
359 	 * @param expected Comma-delimited list of expected values. Must match the order of fields.
360 	 * @throws NullPointerException if the bean is null
361 	 * @throws AssertionError if any property values don't match expected values
362 	 * @see BeanConverter
363 	 * @see BasicBeanConverter
364 	 */
365 	public static void assertBean(Object actual, String fields, String expected) {
366 		assertBean(args(), actual, fields, expected);
367 	}

369 	/**
370 	 * Same as {@link #assertBeans(Object, String, String...)} but with configurable assertion behavior.
371 	 *
372 	 * @param args Assertion configuration. See {@link #args()} for usage examples.
373 	 * @param actual The collection of beans to test. Must not be null.
374 	 * @param fields A comma-delimited list of bean property names (supports nested syntax).
375 	 * @param expected Array of expected value strings, one per bean.
376 	 * @see #assertBeans(Object, String, String...)
377 	 * @see #args()
378 	 */
379 	public static void assertBeans(AssertionArgs args, Object actual, String fields, String...expected) {
380 		assertNotNull(actual, "Value was null.");
381 		assertArgNotNull("args", args);
382 		assertArgNotNull("fields", fields);
383 		assertArgNotNull("expected", expected);
384 
385 		var converter = args.getBeanConverter().orElse(DEFAULT_CONVERTER);
386 		var tokens = tokenize(fields);
387 		var errors = new ArrayList<AssertionFailedError>();
388 		var actualList = converter.listify(actual);
389 
390 		if (ne(expected.length, actualList.size())) {
391 			errors.add(assertEqualsFailed(expected.length, actualList.size(), args.getMessage("Wrong number of beans.")));
392 		} else {
393 			for (var i = 0; i < actualList.size(); i++) {
394 				var i2 = i;
