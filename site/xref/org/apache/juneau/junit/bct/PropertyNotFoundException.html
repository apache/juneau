17  package org.apache.juneau.junit.bct;
18  
19  import static org.apache.juneau.junit.bct.Utils.*;
20  
21  /**
22   * Exception thrown when a requested property cannot be found on an object.
23   *
24   * <p>This exception is typically thrown by {@link PropertyExtractor} implementations
25   * when attempting to access a property that does not exist on the target object.</p>
26   *
27   * <h5 class='section'>Examples:</h5>
28   * <p class='bjava'>
29   *   <jc>// Property extraction that may throw PropertyNotFoundException</jc>
30   *   <jk>try</jk> {
31   *      String <jv>value</jv> = <jv>converter</jv>.getNested(<jv>object</jv>, <jsm>tokenize</jsm>(<js>"nonExistentProperty"</js>).get(0));
32   *   } <jk>catch</jk> (PropertyNotFoundException <jv>e</jv>) {
33   *      <jc>// Handle missing property gracefully</jc>
34   *      System.<jsm>err</jsm>.println(<js>"Property not found: "</js> + <jv>e</jv>.getMessage());
35   *   }
36   * </p>
37   *
38   * <p class='bjava'>
39   *   <jc>// Testing for PropertyNotFoundException</jc>
40   *   <jsm>assertThrows</jsm>(PropertyNotFoundException.<jk>class</jk>, () -> {
41   *      <jv>converter</jv>.getNested(<jv>bean</jv>, <jsm>tokenize</jsm>(<js>"invalidField"</js>).get(0));
42   *   });
43   * </p>
44   */
45  public class PropertyNotFoundException extends RuntimeException {
46  
47  	private static final long serialVersionUID = 1L;
48  
49  	/**
50  	 * Constructs a new PropertyNotFoundException with the specified detail message.
51  	 *
52  	 * @param message The detail message describing the missing property
53  	 */
54  	public PropertyNotFoundException(String message) {
55  		super(message);
56  	}
57  
58  	/**
59  	 * Constructs a new PropertyNotFoundException for a specific property name and object type.
60  	 *
61  	 * @param propertyName The name of the property that could not be found
62  	 * @param objectType The type of object on which the property was sought
63  	 */
64  	public PropertyNotFoundException(String propertyName, Class<?> objectType) {
65  		super(f("Property ''{0}'' not found on object of type {1}", propertyName, objectType.getSimpleName()));
66  	}
67  
68  	/**
69  	 * Constructs a new PropertyNotFoundException for a specific property name and object type with a cause.
70  	 *
71  	 * @param propertyName The name of the property that could not be found
72  	 * @param objectType The type of object on which the property was sought
73  	 * @param cause The underlying cause of the exception
74  	 */
75  	public PropertyNotFoundException(String propertyName, Class<?> objectType, Throwable cause) {
76  		super(f("Property ''{0}'' not found on object of type {1}", propertyName, objectType.getSimpleName()), cause);
77  	}
78  
79  	/**
80  	 * Constructs a new PropertyNotFoundException with the specified detail message and cause.
81  	 *
82  	 * @param message The detail message describing the missing property
83  	 * @param cause The underlying cause of the exception
84  	 */
85  	public PropertyNotFoundException(String message, Throwable cause) {
86  		super(message, cause);
87  	}
88  }
