17  package org.apache.juneau.junit.bct;
18  
19  import java.util.function.*;
20  
21  /**
22   * Functional interface for pre-processing and transforming objects before conversion.
23   *
24   * <p>Swappers provide object transformation logic that runs before stringification or
25   * listification. They enable unwrapping, preprocessing, and value extraction from
26   * wrapper objects, containers, and lazy evaluation constructs.</p>
27   *
28   * <h5 class='section'>Key Features:</h5>
29   * <ul>
30   *    <li><b>Pre-processing:</b> Transform objects before string/list conversion</li>
31   *    <li><b>Wrapper unwrapping:</b> Extract values from Optional, Supplier, Future, etc.</li>
32   *    <li><b>Lazy evaluation:</b> Trigger computation of lazy values</li>
33   *    <li><b>Type transformation:</b> Convert between related types for better testing</li>
34   * </ul>
35   *
36   * <h5 class='section'>Common Use Cases:</h5>
37   * <ul>
38   *    <li><b>Optional unwrapping:</b> Extract values from Optional containers</li>
39   *    <li><b>Supplier evaluation:</b> Call get() on Supplier objects</li>
40   *    <li><b>Future resolution:</b> Extract completed values from Future objects</li>
41   *    <li><b>Proxy unwrapping:</b> Extract underlying objects from proxies</li>
42   *    <li><b>Value extraction:</b> Pull relevant data from complex wrapper objects</li>
43   * </ul>
44   *
45   * <h5 class='section'>Usage Examples:</h5>
46   * <p class='bjava'>
47   *    <jc>// Future value extraction</jc>
48   *    Swapper&lt;CompletableFuture&gt; <jv>futureSwapper</jv> = (<jp>conv</jp>, <jp>future</jp>) -&gt; {
49   *       <jk>try</jk> {
50   *          <jk>return</jk> <jp>future</jp>.isDone() ? <jp>future</jp>.get() : <js>"&lt;pending&gt;"</js>;
51   *       } <jk>catch</jk> (Exception <jv>e</jv>) {
52   *          <jk>return</jk> <js>"&lt;error: "</js> + <jv>e</jv>.getMessage() + <js>"&gt;"</js>;
53   *       }
54   *    };
55   *
56   *    <jc>// Custom wrapper unwrapping</jc>
57   *    Swapper&lt;LazyValue&gt; <jv>lazySwapper</jv> = (<jp>conv</jp>, <jp>lazy</jp>) -&gt;
58   *       <jp>lazy</jp>.isEvaluated() ? <jp>lazy</jp>.getValue() : <js>"&lt;unevaluated&gt;"</js>;
59   *
60   *    <jc>// Entity to DTO conversion</jc>
61   *    Swapper&lt;UserEntity&gt; <jv>entitySwapper</jv> = (<jp>conv</jp>, <jp>entity</jp>) -&gt;
62   *       <jk>new</jk> UserDTO(<jp>entity</jp>.getId(), <jp>entity</jp>.getName(), <jp>entity</jp>.getEmail());
63   * </p>
64   *
65   * <h5 class='section'>Registration:</h5>
66   * <p class='bjava'>
67   *    <jk>var</jk> <jv>converter</jv> = BasicBeanConverter.<jsm>builder</jsm>()
68   *       .defaultSettings()
69   *       .addSwapper(CompletableFuture.<jk>class</jk>, <jv>futureSwapper</jv>)
70   *       .addSwapper(LazyValue.<jk>class</jk>, <jv>lazySwapper</jv>)
71   *       .addSwapper(UserEntity.<jk>class</jk>, <jv>entitySwapper</jv>)
72   *       .build();
73   * </p>
74   *
75   * <h5 class='section'>Execution Flow:</h5>
76   * <p>Swappers are applied early in the conversion process:</p>
77   * <ol>
78   *    <li><b>Object received</b> for conversion</li>
79   *    <li><b>Swapper applied</b> if one is registered for the object's type</li>
80   *    <li><b>Result processed</b> through normal stringification/listification</li>
81   *    <li><b>Chain continues</b> recursively for nested objects</li>
82   * </ol>
83   *
84   * <h5 class='section'>Best Practices:</h5>
85   * <ul>
86   *    <li><b>Handle exceptions</b> gracefully, returning error indicators rather than throwing</li>
87   *    <li><b>Preserve semantics</b> - the swapped object should represent the same logical value</li>
88   *    <li><b>Consider performance</b> - swappers are called frequently during conversion</li>
89   *    <li><b>Return meaningful values</b> for null or invalid states</li>
90   *    <li><b>Avoid recursion</b> - ensure swappers don't create circular transformations that could 
91   *        lead to {@link StackOverflowError}. The framework does not detect recursion, so developers
92   *        must ensure swapped objects don't trigger the same or related swappers in an endless cycle</li>
93   * </ul>
94   *
95   * @param <T> The type of object this swapper handles
96   * @see BasicBeanConverter.Builder#addSwapper(Class, Swapper)
97   * @see BeanConverter#swap(Object)
98   */
99  @FunctionalInterface
100 public interface Swapper<T> extends BiFunction<BeanConverter,T,Object> {}
