27  /**
28   * Configuration and context object for advanced assertion operations.
29   *
30   * <p>This class encapsulates additional arguments and configuration options for assertion methods
31   * in the Bean-Centric Testing (BCT) framework. It provides a fluent API for customizing assertion
32   * behavior including custom converters and enhanced error messaging.</p>
33   *
34   * <p>The primary purposes of this class are:</p>
35   * <ul>
36   *     <li><b>Custom Bean Conversion:</b> Override the default {@link BeanConverter} for specialized object introspection</li>
37   *     <li><b>Enhanced Error Messages:</b> Add context-specific error messages with parameter substitution</li>
38   *     <li><b>Fluent Configuration:</b> Chain configuration calls for readable test setup</li>
39   *     <li><b>Assertion Context:</b> Provide additional context for complex assertion scenarios</li>
40   * </ul>
41   *
42   * <h5 class='section'>Basic Usage:</h5>
43   * <p class='bjava'>
44   *     <jc>// Simple usage with default settings</jc>
45   *     <jsm>assertBean</jsm>(<jsm>args</jsm>(), <jv>myBean</jv>, <js>"name,age"</js>, <js>"John,30"</js>);
46   *
47   *     <jc>// Custom error message</jc>
48   *     <jsm>assertBean</jsm>(<jsm>args</jsm>().setMessage(<js>"User validation failed"</js>),
49   *         <jv>user</jv>, <js>"email,active"</js>, <js>"john@example.com,true"</js>);
50   * </p>
51   *
52   * <h5 class='section'>Custom Bean Converter:</h5>
53   * <p class='bjava'>
54   *     <jc>// Use custom converter for specialized object handling</jc>
55   *     <jk>var</jk> <jv>customConverter</jv> = BasicBeanConverter.<jsm>builder</jsm>()
56   *         .defaultSettings()
57   *         .addStringifier(MyClass.<jk>class</jk>, <jp>obj</jp> -> <jp>obj</jp>.getDisplayName())
58   *         .build();
59   *
60   *     <jsm>assertBean</jsm>(<jsm>args</jsm>().setBeanConverter(<jv>customConverter</jv>),
61   *         <jv>myCustomObject</jv>, <js>"property"</js>, <js>"expectedValue"</js>);
62   * </p>
63   *
64   * <h5 class='section'>Advanced Error Messages:</h5>
65   * <p class='bjava'>
66   *     <jc>// Parameterized error messages</jc>
67   *     <jsm>assertBean</jsm>(<jsm>args</jsm>().setMessage(<js>"Validation failed for user {0}"</js>, <jv>userId</jv>),
68   *         <jv>user</jv>, <js>"status"</js>, <js>"ACTIVE"</js>);
69   *
70   *     <jc>// Dynamic error message with supplier</jc>
71   *     <jsm>assertBean</jsm>(<jsm>args</jsm>().setMessage(() -> <js>"Test failed at "</js> + Instant.<jsm>now</jsm>()),
72   *         <jv>result</jv>, <js>"success"</js>, <js>"true"</js>);
73   * </p>
74   *
75   * <h5 class='section'>Fluent Configuration:</h5>
76   * <p class='bjava'>
77   *     <jc>// Chain multiple configuration options</jc>
78   *     <jk>var</jk> <jv>testArgs</jv> = args()
79   *         .setBeanConverter(<jv>customConverter</jv>)
80   *         .setMessage(<js>"Integration test failed for module {0}"</js>, <jv>moduleName</jv>);
81   *
82   *     <jsm>assertBean</jsm>(<jv>testArgs</jv>, <jv>moduleConfig</jv>, <js>"enabled,version"</js>, <js>"true,2.1.0"</js>);
83   *     <jsm>assertBeans</jsm>(<jv>testArgs</jv>, <jv>moduleList</jv>, <js>"name,status"</js>,
84   *         <js>"ModuleA,ACTIVE"</js>, <js>"ModuleB,ACTIVE"</js>);
85   * </p>
86   *
87   * <h5 class='section'>Error Message Composition:</h5>
88   * <p>When assertion failures occur, error messages are intelligently composed:</p>
89   * <ul>
90   *     <li><b>Base Message:</b> Custom message set via {@link #setMessage(String, Object...)} or {@link #setMessage(Supplier)}</li>
91   *     <li><b>Assertion Context:</b> Specific context provided by individual assertion methods</li>
92   *     <li><b>Composite Format:</b> <js>"{base message}, Caused by: {assertion context}"</js></li>
93   * </ul>
94   *
95   * <p class='bjava'>
96   *     <jc>// Example error message composition:</jc>
97   *     <jc>// Base: "User validation failed for user 123"</jc>
98   *     <jc>// Context: "Bean assertion failed."</jc>
99   *     <jc>// Result: "User validation failed for user 123, Caused by: Bean assertion failed."</jc>
100  * </p>
101  *
102  * <h5 class='section'>Thread Safety:</h5>
103  * <p>This class is <b>not thread-safe</b> and is intended for single-threaded test execution.
104  * Each test method should create its own instance using {@link BctAssertions#args()} or create
105  * a new instance directly with {@code new AssertionArgs()}.</p>
106  *
107  * <h5 class='section'>Immutability Considerations:</h5>
108  * <p>While this class uses fluent setters that return {@code this} for chaining, the instance
109  * is mutable. For reusable configurations across multiple tests, consider creating a factory
110  * method that returns pre-configured instances.</p>
111  *
112  * @see BctAssertions#args()
113  * @see BeanConverter
114  * @see BasicBeanConverter
115  */
116 public class AssertionArgs {
117 
118 	private BeanConverter beanConverter;
119 	private Supplier<String> messageSupplier;
120 
121 	/**
122 	 * Creates a new instance with default settings.
123 	 *
124 	 * <p>Instances start with no custom bean converter and no custom error message.
125 	 * All assertion methods will use default behavior until configured otherwise.</p>
126 	 */
127 	public AssertionArgs() { /* no-op */ }
128 
129 	/**
130 	 * Sets a custom {@link BeanConverter} for object introspection and property access.
131 	 *
132 	 * <p>The custom converter allows fine-tuned control over how objects are converted to strings,
133 	 * how collections are listified, and how nested properties are accessed. This is particularly
134 	 * useful for:</p>
135 	 * <ul>
136 	 *     <li><b>Custom Object Types:</b> Objects that don't follow standard JavaBean patterns</li>
137 	 *     <li><b>Specialized Formatting:</b> Custom string representations for assertion comparisons</li>
138 	 *     <li><b>Performance Optimization:</b> Cached or optimized property access strategies</li>
139 	 *     <li><b>Domain-Specific Logic:</b> Business-specific property resolution rules</li>
140 	 * </ul>
141 	 *
142 	 * <h5 class='section'>Example:</h5>
143 	 * <p class='bjava'>
144 	 *     <jc>// Create converter with custom stringifiers</jc>
145 	 *     <jk>var</jk> <jv>converter</jv> = BasicBeanConverter.<jsm>builder</jsm>()
146 	 *         .defaultSettings()
147 	 *         .addStringifier(LocalDate.<jk>class</jk>, <jp>date</jp> -> <jp>date</jp>.format(DateTimeFormatter.<jsf>ISO_LOCAL_DATE</jsf>))
148 	 *         .addStringifier(Money.<jk>class</jk>, <jp>money</jp> -> <jp>money</jp>.getAmount().toPlainString())
149 	 *         .build();
150 	 *
151 	 *     <jc>// Use in assertions</jc>
152 	 *     <jsm>assertBean</jsm>(<jsm>args</jsm>().setBeanConverter(<jv>converter</jv>),
153 	 *         <jv>order</jv>, <js>"date,total"</js>, <js>"2023-12-01,99.99"</js>);
154 	 * </p>
155 	 *
156 	 * @param value The custom bean converter to use. If null, assertions will fall back to the default converter.
157 	 * @return This instance for method chaining.
158 	 */
159 	public AssertionArgs setBeanConverter(BeanConverter value) {
160 		beanConverter = value;
161 		return this;
162 	}
163 
164 	/**
165 	 * Sets a parameterized error message for assertion failures.
166 	 *
167 	 * <p>This method uses {@link MessageFormat} to substitute parameters into the message template.
168 	 * The formatting occurs immediately when this method is called, not when the assertion fails.</p>
169 	 *
170 	 * <h5 class='section'>Parameter Substitution:</h5>
171 	 * <p>Uses standard MessageFormat patterns:</p>
172 	 * <ul>
173 	 *     <li><code>{0}</code> - First parameter</li>
174 	 *     <li><code>{1}</code> - Second parameter</li>
175 	 *     <li><code>{0,number,#}</code> - Formatted number</li>
176 	 *     <li><code>{0,date,short}</code> - Formatted date</li>
177 	 * </ul>
178 	 *
179 	 * <h5 class='section'>Examples:</h5>
180 	 * <p class='bjava'>
181 	 *     <jc>// Simple parameter substitution</jc>
182 	 *     <jsm>assertBean</jsm>(<jsm>args</jsm>().setMessage(<js>"User {0} validation failed"</js>, <jv>userId</jv>),
183 	 *         <jv>user</jv>, <js>"active"</js>, <js>"true"</js>);
184 	 *
185 	 *     <jc>// Multiple parameters</jc>
186 	 *     <jsm>assertBean</jsm>(<jsm>args</jsm>().setMessage(<js>"Test {0} failed on iteration {1}"</js>, <jv>testName</jv>, <jv>iteration</jv>),
187 	 *         <jv>result</jv>, <js>"success"</js>, <js>"true"</js>);
188 	 *
189 	 *     <jc>// Number formatting</jc>
190 	 *     <jsm>assertBean</jsm>(<jsm>args</jsm>().setMessage(<js>"Expected {0,number,#.##} but got different value"</js>, <jv>expectedValue</jv>),
191 	 *         <jv>actual</jv>, <js>"value"</js>, <js>"123.45"</js>);
192 	 * </p>
193 	 *
194 	 * @param message The message template with MessageFormat placeholders.
195 	 * @param args The parameters to substitute into the message template.
196 	 * @return This instance for method chaining.
197 	 */
198 	public AssertionArgs setMessage(String message, Object...args) {
199 		messageSupplier = fs(message, args);
200 		return this;
201 	}
202 
203 	/**
204 	 * Sets a custom error message supplier for assertion failures.
205 	 *
206 	 * <p>The supplier allows for dynamic message generation, including context that may only
207 	 * be available at the time of assertion failure. This is useful for:</p>
208 	 * <ul>
209 	 *     <li><b>Timestamps:</b> Including the exact time of failure</li>
210 	 *     <li><b>Test State:</b> Including runtime state information</li>
211 	 *     <li><b>Expensive Operations:</b> Deferring costly string operations until needed</li>
212 	 *     <li><b>Conditional Messages:</b> Different messages based on runtime conditions</li>
213 	 * </ul>
214 	 *
215 	 * <h5 class='section'>Example:</h5>
216 	 * <p class='bjava'>
217 	 *     <jc>// Dynamic message with timestamp</jc>
218 	 *     <jsm>assertBean</jsm>(<jsm>args</jsm>().setMessage(() -> <js>"Test failed at "</js> + Instant.<jsm>now</jsm>()),
219 	 *         <jv>result</jv>, <js>"status"</js>, <js>"SUCCESS"</js>);
220 	 *
221 	 *     <jc>// Message with expensive computation</jc>
222 	 *     <jsm>assertBean</jsm>(<jsm>args</jsm>().setMessage(() -> <js>"Failed after "</js> + computeTestDuration() + <js>" ms"</js>),
223 	 *         <jv>response</jv>, <js>"error"</js>, <js>"null"</js>);
224 	 * </p>
225 	 *
226 	 * @param value The message supplier. Called only when an assertion fails.
227 	 * @return This instance for method chaining.
228 	 */
229 	public AssertionArgs setMessage(Supplier<String> value) {
230 		messageSupplier = value;
231 		return this;
232 	}
233 
234 	/**
235 	 * Gets the configured bean converter, if any.
236 	 *
237 	 * @return An Optional containing the custom converter, or empty if using default behavior.
238 	 */
239 	protected Optional<BeanConverter> getBeanConverter() { return ofNullable(beanConverter); }
240 
241 	/**
242 	 * Gets the base message supplier for composition with assertion-specific messages.
243 	 *
244 	 * @return The configured message supplier, or null if no custom message was set.
245 	 */
246 	protected Supplier<String> getMessage() { return messageSupplier; }
247 
248 	/**
249 	 * Composes the final error message by combining custom and assertion-specific messages.
250 	 *
251 	 * <p>This method implements the message composition strategy used throughout the assertion framework:</p>
252 	 * <ul>
253 	 *     <li><b>No Custom Message:</b> Returns the assertion-specific message as-is</li>
254 	 *     <li><b>With Custom Message:</b> Returns <code>"{custom}, Caused by: {assertion}"</code></li>
255 	 * </ul>
256 	 *
257 	 * <p>This allows tests to provide high-level context while preserving the specific
258 	 * technical details about what assertion failed.</p>
259 	 *
260 	 * @param msg The assertion-specific message template.
261 	 * @param args Parameters for the assertion-specific message.
262 	 * @return A supplier that produces the composed error message.
263 	 */
264 	protected Supplier<String> getMessage(String msg, Object...args) {
265 		return messageSupplier == null ? fs(msg, args) : fs("{0}, Caused by: {1}", messageSupplier.get(), f(msg, args));
266 	}
267 }
