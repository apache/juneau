17  package org.apache.juneau.junit.bct;
18  
19  import java.util.function.*;
20  
21  /**
22   * Functional interface for converting objects of a specific type to their string representation.
23   *
24   * <p>Stringifiers provide custom string conversion logic for specific object types within the
25   * Bean-Centric Testing framework. They enable precise control over how objects are displayed
26   * in test assertions and error messages, going beyond simple {@code toString()} calls.</p>
27   *
28   * <h5 class='section'>Usage Examples:</h5>
29   * <p class='bjava'>
30   *    <jc>// Simple custom format</jc>
31   *    Stringifier&lt;LocalDate&gt; <jv>dateStringifier</jv> = (<jp>conv</jp>, <jp>date</jp>) -&gt;
32   *       <jp>date</jp>.format(DateTimeFormatter.<jsf>ISO_LOCAL_DATE</jsf>);
33   *
34   *    <jc>// Complex nested object formatting</jc>
35   *    Stringifier&lt;Person&gt; <jv>personStringifier</jv> = (<jp>conv</jp>, <jp>person</jp>) -&gt;
36   *       <jp>person</jp>.getName() + <js>" ("</js> + <jp>conv</jp>.stringify(<jp>person</jp>.getAddress()) + <js>")"</js>;
37   *
38   *    <jc>// Conditional formatting</jc>
39   *    Stringifier&lt;Order&gt; <jv>orderStringifier</jv> = (<jp>conv</jp>, <jp>order</jp>) -&gt; {
40   *       <jk>var</jk> <jv>status</jv> = <jp>order</jp>.getStatus();
41   *       <jk>return</jk> <jv>status</jv> == OrderStatus.<jsf>COMPLETED</jsf> ?
42   *          <js>"Order #"</js> + <jp>order</jp>.getId() + <js>" [DONE]"</js> :
43   *          <js>"Order #"</js> + <jp>order</jp>.getId() + <js>" ["</js> + <jv>status</jv> + <js>"]"</js>;
44   *    };
45   * </p>
46   *
47   * <h5 class='section'>Registration:</h5>
48   * <p class='bjava'>
49   *    <jk>var</jk> <jv>converter</jv> = BasicBeanConverter.<jsm>builder</jsm>()
50   *       .defaultSettings()
51   *       .addStringifier(LocalDate.<jk>class</jk>, <jv>dateStringifier</jv>)
52   *       .addStringifier(Person.<jk>class</jk>, <jv>personStringifier</jv>)
53   *       .addStringifier(Order.<jk>class</jk>, <jv>orderStringifier</jv>)
54   *       .build();
55   * </p>
56   *
57   * <h5 class='section'>Best Practices:</h5>
58   * <ul>
59   *    <li><b>Use the converter parameter</b> for recursive conversion of nested objects</li>
60   *    <li><b>Keep output concise</b> but informative for test readability</li>
61   *    <li><b>Handle edge cases</b> gracefully (empty collections, special values)</li>
62   *    <li><b>Consider test context</b> - focus on properties relevant to testing</li>
63   * </ul>
64   *
65   * @param <T> The type of object this stringifier handles
66   * @see BasicBeanConverter.Builder#addStringifier(Class, Stringifier)
67   * @see BeanConverter#stringify(Object)
68   */
69  @FunctionalInterface
70  public interface Stringifier<T> extends BiFunction<BeanConverter,T,String> {}
