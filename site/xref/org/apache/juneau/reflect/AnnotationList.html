17  package org.apache.juneau.reflect;
18  
19  import static org.apache.juneau.internal.ConsumerUtils.*;
20  
21  import java.lang.annotation.*;
22  import java.util.*;
23  import java.util.function.*;
24  
25  /**
26   * An ordered list of annotations and the classes/methods/packages they were found on.
27   *
28   * <h5 class='section'>See Also:</h5><ul>
29   * </ul>
30   *
31   * @serial exclude
32   */
33  public class AnnotationList extends ArrayList<AnnotationInfo<?>> {
34  	private static final long serialVersionUID = 1L;
35  
36  	private static final Comparator<AnnotationInfo<?>> RANK_COMPARATOR = (o1, o2) -> o1.rank - o2.rank;
37  
38  	/**
39  	 * Performs an action on all matching annotations in this list.
40  	 *
41  	 * @param <A> The annotation type.
42  	 * @param type The annotation type.
43  	 * @param filter A predicate to apply to the entries to determine if action should be performed.  Can be <jk>null</jk>.
44  	 * @param action An action to perform on the entry.
45  	 * @return This object.
46  	 */
47  	@SuppressWarnings("unchecked")
48  	public <A extends Annotation> AnnotationList forEach(Class<A> type, Predicate<AnnotationInfo<A>> filter, Consumer<AnnotationInfo<A>> action) {
49  		forEach(x -> {
50  			if (x.isType(type))
51  				consume(filter, action, (AnnotationInfo<A>)x);
52  		});
53  		return this;
54  	}
55  
56  	/**
57  	 * Performs an action on all matching annotations in this list.
58  	 *
59  	 * @param filter A predicate to apply to the entries to determine if action should be performed.  Can be <jk>null</jk>.
60  	 * @param action An action to perform on the entry.
61  	 * @return This object.
62  	 */
63  	public AnnotationList forEach(Predicate<AnnotationInfo<?>> filter, Consumer<AnnotationInfo<?>> action) {
64  		forEach(x -> consume(filter, action, x));
65  		return this;
66  	}
67  
68  	/**
69  	 * Performs an action on the specified matching values from all annotations in this list.
70  	 *
71  	 * @param <T> The annotation value type.
72  	 * @param type The annotation value type.
73  	 * @param name The annotation value name.
74  	 * @param filter A predicate to apply to the value to determine if action should be performed.  Can be <jk>null</jk>.
75  	 * @param action An action to perform on the value.
76  	 * @return This object.
77  	 */
78  	public <T> AnnotationList forEachValue(Class<T> type, String name, Predicate<T> filter, Consumer<T> action) {
79  		forEach(x -> x.forEachValue(type, name, filter, action));
80  		return this;
81  	}
82  
83  	/**
84  	 * Sort the annotations in this list based on rank.
85  	 *
86  	 * @return This object.
87  	 */
88  	public AnnotationList sort() {
89  		Collections.sort(this, RANK_COMPARATOR);
90  		return this;
91  	}
92  }
