75  	private static List<MethodInfo> findMatching(List<MethodInfo> l, MethodInfo m, ClassInfo c) {
76  		for (MethodInfo m2 : c._getDeclaredMethods())
77  			if (m.hasName(m2.getName()) && Arrays.equals(m._getParameterTypes(), m2._getParameterTypes()))
78  				l.add(m2);
79  		ClassInfo pc = c.getSuperclass();
80  		if (pc != null)
81  			findMatching(l, m, pc);
82  		for (ClassInfo ic : c._getDeclaredInterfaces())
83  			findMatching(l, m, ic);
84  		return l;
85  	}

121 	 * Returns <jk>true</jk> if the parameters on the method only consist of the types specified in the list.
122 	 *
123 	 * <h5 class='figure'>Example:</h5>
124 	 * <p class='bjava'>
125 	 *
126 	 * 	<jc>// Example method:</jc>
127 	 * 	<jk>public void</jk> foo(String <jv>bar</jv>, Integer <jv>baz</jv>);
128 	 *
129 	 * 	argsOnlyOfType(<jv>fooMethod</jv>, String.<jk>class</jk>, Integer.<jk>class</jk>);  <jc>// True.</jc>
130 	 * 	argsOnlyOfType(<jv>fooMethod</jv>, String.<jk>class</jk>, Integer.<jk>class</jk>, Map.<jk>class</jk>);  <jc>// True.</jc>
131 	 * 	argsOnlyOfType(<jv>fooMethod</jv>, String.<jk>class</jk>);  <jc>// False.</jc>
132 	 * </p>
133 	 *
134 	 * @param args The valid class types (exact) for the arguments.
135 	 * @return <jk>true</jk> if the method parameters only consist of the types specified in the list.
136 	 */
137 	public boolean argsOnlyOfType(Class<?>...args) {
138 		for (Class<?> c1 : _getRawParamTypes()) {
139 			boolean foundMatch = false;
140 			for (Class<?> c2 : args)
141 				if (c1 == c2)
142 					foundMatch = true;
143 			if (! foundMatch)
144 				return false;
145 		}
146 		return true;
147 	}

150 	 * Returns <jk>true</jk> if this constructor can accept the specified arguments in the specified order.
151 	 *
152 	 * @param args The arguments to check.
153 	 * @return <jk>true</jk> if this constructor can accept the specified arguments in the specified order.
154 	 */
155 	public boolean canAccept(Object...args) {
156 		Class<?>[] pt = m.getParameterTypes();
157 		if (pt.length != args.length)
158 			return false;
159 		for (int i = 0; i < pt.length; i++)
160 			if (! pt[i].isInstance(args[i]))
161 				return false;
162 		return true;
163 	}

166 	 * Returns the number of matching arguments for this method.
167 	 *
168 	 * @param args The arguments to check.
169 	 * @return the number of matching arguments for this method.
170 	 */
171 	public int canAcceptFuzzy(Object...args) {
172 		int matches = 0;
173 		outer: for (ClassInfo pi : _getParameterTypes()) {
174 			for (Object a : args) {
175 				if (pi.canAcceptArg(a)) {
176 					matches++;
177 					continue outer;
178 				}
179 			}
180 			return -1;
181 		}
182 		return matches;
183 	}

186 	public int compareTo(MethodInfo o) {
187 		int i = getSimpleName().compareTo(o.getSimpleName());
188 		if (i == 0) {
189 			i = _getRawParamTypes().length - o._getRawParamTypes().length;
190 			if (i == 0) {
191 				for (int j = 0; j < _getRawParamTypes().length && i == 0; j++) {
192 					i = _getRawParamTypes()[j].getName().compareTo(o._getRawParamTypes()[j].getName());
193 				}
194 			}
195 		}
196 		return i;
197 	}

204 	 * and the return type on the method.
205 	 * <br>Results are parent-to-child ordered.
206 	 *
207 	 * @param <A> The annotation type to look for.
208 	 * @param annotationProvider The annotation provider.
209 	 * @param type The annotation type.

211 	 * @param action An action to perform on the entry.
212 	 * @return This object.
213 	 */
214 	public <A extends Annotation> MethodInfo forEachAnnotation(AnnotationProvider annotationProvider, Class<A> type, Predicate<A> filter, Consumer<A> action) {
215 		declaringClass.forEachAnnotation(annotationProvider, type, filter, action);
216 		MethodInfo[] m = _getMatching();
217 		for (int i = m.length-1; i >= 0; i--)
218 			for (Annotation a2 : m[i]._getDeclaredAnnotations())
219 				consume(type, filter, action, a2);
220 		getReturnType().unwrap(Value.class,Optional.class).forEachAnnotation(annotationProvider, type, filter, action);
221 		return this;
222 	}

243 	 *
244 	 * @param filter A predicate to apply to the entries to determine if action should be performed.  Can be <jk>null</jk>.
245 	 * @param action An action to perform on the entry.
246 	 * @return This object.

248 	public MethodInfo forEachAnnotationInfo(Predicate<AnnotationInfo<?>> filter, Consumer<AnnotationInfo<?>> action) {
249 		ClassInfo c = this.declaringClass;
250 		forEachDeclaredAnnotationInfo(c.getPackage(), filter, action);
251 		ClassInfo[] interfaces = c._getInterfaces();
252 		for (int i = interfaces.length-1; i >= 0; i--) {
253 			forEachDeclaredAnnotationInfo(interfaces[i], filter, action);
254 			forEachDeclaredMethodAnnotationInfo(interfaces[i], filter, action);
255 		}
256 		ClassInfo[] parents = c._getParents();
257 		for (int i = parents.length-1; i >= 0; i--) {
258 			forEachDeclaredAnnotationInfo(parents[i], filter, action);
259 			forEachDeclaredMethodAnnotationInfo(parents[i], filter, action);
260 		}
261 		return this;
262 	}

315 	public <A extends Annotation> A getAnnotation(AnnotationProvider annotationProvider, Class<A> type) {
316 		if (type == null)
317 			return null;
318 		Value<A> t = Value.empty();
319 		for (MethodInfo m2 : _getMatching()) {
320 			annotationProvider.forEachAnnotation(type, m2.inner(), x -> true, x -> t.set(x));
321 			if (t.isPresent())
322 				return t.get();
323 		}
324 		return null;

472 		}

619 	 * Shortcut for calling the invoke method on the underlying method.
620 	 *
621 	 * @param <T> The method return type.
622 	 * @param obj the object the underlying method is invoked from.
623 	 * @param args the arguments used for the method call
624 	 * @return The object returned from the method.
625 	 * @throws ExecutableException Exception occurred on invoked constructor/method/field.
626 	 */
627 	@SuppressWarnings("unchecked")
628 	public <T> T invoke(Object obj, Object...args) throws ExecutableException {
629 		try {
630 			return (T)m.invoke(obj, args);
631 		} catch (IllegalAccessException e) {
632 			throw new ExecutableException(e);
633 		} catch (InvocationTargetException e) {
634 			throw new ExecutableException(e.getTargetException());
635 		}
636 	}

638 	/**
639 	 * Invokes the specified method using fuzzy-arg matching.
640 	 *
641 	 * <p>
642 	 * Arguments will be matched to the parameters based on the parameter types.
643 	 * <br>Arguments can be in any order.
644 	 * <br>Extra arguments will be ignored.
645 	 * <br>Missing arguments will be left <jk>null</jk>.
646 	 *
647 	 * <p>
648 	 * Note that this only works for methods that have distinguishable argument types.
649 	 * <br>It's not going to work on methods with generic argument types like <c>Object</c>
650 	 *
651 	 * @param pojo
652 	 * 	The POJO the method is being called on.
653 	 * 	<br>Can be <jk>null</jk> for static methods.
654 	 * @param args
655 	 * 	The arguments to pass to the method.
656 	 * @return
657 	 * 	The results of the method invocation.
658 	 * @throws ExecutableException Exception occurred on invoked constructor/method/field.
659 	 */
660 	public Object invokeFuzzy(Object pojo, Object...args) throws ExecutableException {
661 		try {
662 			return m.invoke(pojo, ClassUtils.getMatchingArgs(m.getParameterTypes(), args));
663 		} catch (IllegalAccessException | InvocationTargetException e) {
664 			throw new ExecutableException(e);
665 		}
666 	}

687 	private MethodInfo findMatchingOnClass(ClassInfo c) {
688 		for (MethodInfo m2 : c._getDeclaredMethods())
689 			if (hasName(m2.getName()) && Arrays.equals(_getParameterTypes(), m2._getParameterTypes()))
690 				return m2;
691 		return null;
692 	}

694 	private void forEachAnnotationInfoMethodOnly(Predicate<AnnotationInfo<?>> filter, Consumer<AnnotationInfo<?>> action) {
695 		ClassInfo c = this.declaringClass;
696 		ClassInfo[] interfaces = c._getInterfaces();
697 		for (int i = interfaces.length-1; i >= 0; i--)
698 			forEachDeclaredMethodAnnotationInfo(interfaces[i], filter, action);
699 		ClassInfo[] parents = c._getParents();
700 		for (int i = parents.length-1; i >= 0; i--)
701 			forEachDeclaredMethodAnnotationInfo(parents[i], filter, action);
702 	}

704 	private void forEachDeclaredAnnotationInfo(ClassInfo ci, Predicate<AnnotationInfo<?>> filter, Consumer<AnnotationInfo<?>> action) {
705 		if (ci != null)
706 			for (Annotation a : ci._getDeclaredAnnotations())
707 				AnnotationInfo.of(ci, a).accept(filter, action);
708 	}

710 	private void forEachDeclaredAnnotationInfo(Package p, Predicate<AnnotationInfo<?>> filter, Consumer<AnnotationInfo<?>> action) {
711 		if (p != null)
712 			for (Annotation a : p.getDeclaredAnnotations())
713 				AnnotationInfo.of(p, a).accept(filter, action);
714 	}
715 	private void forEachDeclaredMethodAnnotationInfo(ClassInfo ci, Predicate<AnnotationInfo<?>> filter, Consumer<AnnotationInfo<?>> action) {
716 		MethodInfo m = findMatchingOnClass(ci);
717 		if (m != null)
718 			for (Annotation a : m._getDeclaredAnnotations())
719 				AnnotationInfo.of(m, a).accept(filter, action);
720 	}

722 	MethodInfo[] _getMatching() {
723 		if (matching == null) {
724 			synchronized(this) {
725 				List<MethodInfo> l = findMatching(list(), this, getDeclaringClass());
726 				matching = l.toArray(new MethodInfo[l.size()]);
727 			}
728 		}
729 		return matching;
