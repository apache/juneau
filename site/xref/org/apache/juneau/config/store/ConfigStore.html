28  /**
29   * Represents a storage location for configuration files.
30   *
31   * <p>
32   * Content stores require two methods to be implemented:
33   * <ul class='javatree'>
34   * 	<li class='jm'>{@link #read(String)} - Retrieve a config file.
35   * 	<li class='jm'>{@link #write(String,String,String)} - ConfigStore a config file.
36   * </ul>
37   *
38   * <h5 class='section'>Notes:</h5><ul>
39   * 	<li class='note'>This class is thread safe and reusable.
40   * </ul>
41  */
42  public abstract class ConfigStore extends Context implements Closeable {

148 	private final ConcurrentHashMap<String,Set<ConfigStoreListener>> listeners = new ConcurrentHashMap<>();
149 	private final ConcurrentHashMap<String,ConfigMap> configMaps = new ConcurrentHashMap<>();
150 
151 	/**
152 	 * Constructor.
153 	 *
154 	 * @param builder The builder for this object.
155 	 */
156 	protected ConfigStore(Builder builder) {
157 		super(builder);
158 	}
159 
160 	/**
161 	 * Returns the contents of the configuration file.
162 	 *
163 	 * @param name The config file name.
164 	 * @return
165 	 * 	The contents of the configuration file.
166 	 * 	<br>A blank string if the config does not exist.
167 	 * 	<br>Never <jk>null</jk>.
168 	 * @throws IOException Thrown by underlying stream.
169 	 */
170 	public abstract String read(String name) throws IOException;
171 
172 	/**
173 	 * Saves the contents of the configuration file if the underlying storage hasn't been modified.
174 	 *
175 	 * @param name The config file name.
176 	 * @param expectedContents The expected contents of the file.
177 	 * @param newContents The new contents.
178 	 * @return
179 	 * 	If <jk>null</jk>, then we successfully stored the contents of the file.
180 	 * 	<br>Otherwise the contents of the file have changed and we return the new contents of the file.
181 	 * @throws IOException Thrown by underlying stream.
182 	 */
183 	public abstract String write(String name, String expectedContents, String newContents) throws IOException;
184 
185 	/**
186 	 * Checks whether the configuration with the specified name exists in this store.
187 	 *
188 	 * @param name The config name.
189 	 * @return <jk>true</jk> if the configuration with the specified name exists in this store.
190 	 */
191 	public abstract boolean exists(String name);
192 
193 	/**
194 	 * Registers a new listener on this store.
195 	 *
196 	 * @param name The configuration name to listen for.
197 	 * @param l The new listener.
198 	 * @return This object.
199 	 */
200 	public synchronized ConfigStore register(String name, ConfigStoreListener l) {
201 		name = resolveName(name);
202 		var s = listeners.computeIfAbsent(
203 			name,
204 			k -> synced(newSetFromMap(new IdentityHashMap<>()))
205 		);
206 		s.add(l);
207 		return this;
208 	}
209 
210 	/**
211 	 * Unregisters a listener from this store.
212 	 *
213 	 * @param name The configuration name to listen for.
214 	 * @param l The listener to unregister.
215 	 * @return This object.
216 	 */
217 	public synchronized ConfigStore unregister(String name, ConfigStoreListener l) {
218 		name = resolveName(name);
219 		var s = listeners.get(name);
220 		if (s != null)
221 			s.remove(l);
222 		return this;
223 	}
224 
225 	/**
226 	 * Returns a map file containing the parsed contents of a configuration.
227 	 *
228 	 * @param name The configuration name.
229 	 * @return
230 	 * 	The parsed configuration.
231 	 * 	<br>Never <jk>null</jk>.
232 	 * @throws IOException Thrown by underlying stream.
233 	 */
234 	public synchronized ConfigMap getMap(String name) throws IOException {
235 		name = resolveName(name);
236 		var cm = configMaps.get(name);
237 		if (cm != null)
238 			return cm;
239 		cm = new ConfigMap(this, name);
240 		var cm2 = configMaps.putIfAbsent(name, cm);
241 		if (cm2 != null)
242 			return cm2;
243 		register(name, cm);
244 		return cm;
245 	}
246 
247 	/**
248 	 * Called when the physical contents of a config file have changed.
249 	 *
250 	 * <p>
251 	 * Triggers calls to {@link ConfigStoreListener#onChange(String)} on all registered listeners.
252 	 *
253 	 * @param name The config name (e.g. the filename without the extension).
254 	 * @param contents The new contents.
255 	 * @return This object.
256 	 */
257 	public synchronized ConfigStore update(String name, String contents) {
258 		name = resolveName(name);
259 		var s = listeners.get(name);
260 		if (s != null)
261 			listeners.get(name).forEach(x -> x.onChange(contents));
262 		return this;
263 	}
264 
265 	/**
266 	 * Convenience method for updating the contents of a file with lines.
267 	 *
268 	 * @param name The config name (e.g. the filename without the extension).
269 	 * @param contentLines The new contents.
270 	 * @return This object.
271 	 */
272 	public synchronized ConfigStore update(String name, String...contentLines) {
273 		name = resolveName(name);
274 		var sb = new StringBuilder();
275 		for (var l : contentLines)
276 			sb.append(l).append('\n');
277 		return update(name, sb.toString());
278 	}
279 
280 	/**
281 	 * Subclasses can override this method to convert config names to internal forms.
282 	 *
283 	 * <p>
284 	 * For example, the {@link FileStore} class can take in both <js>"MyConfig"</js> and <js>"MyConfig.cfg"</js>
285 	 * as names that both resolve to <js>"MyConfig.cfg"</js>.
286 	 *
287 	 * @param name The name to resolve.
288 	 * @return The resolved name.
289 	 */
290 	protected String resolveName(String name) {
291 		return name;
292 	}
