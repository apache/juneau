367 	private final File dir;
368 	private final WatcherThread watcher;
369 	private final ConcurrentHashMap<String,String> cache = new ConcurrentHashMap<>();
370 	private final ConcurrentHashMap<String,String> nameCache = new ConcurrentHashMap<>();
371 	private final String[] exts;

373 	/**
374 	 * Constructor.
375 	 *
376 	 * @param builder The builder for this object.
377 	 */
378 	public FileStore(Builder builder) {
379 		super(builder);
380 		directory = builder.directory;
381 		extensions = builder.extensions;
382 		charset = builder.charset;
383 		enableWatcher = builder.enableWatcher;
384 		updateOnWrite = builder.updateOnWrite;
385 		watcherSensitivity = builder.watcherSensitivity;
386 		try {
387 			dir = new File(directory).getCanonicalFile();
388 			dir.mkdirs();
389 			exts = Utils.split(extensions).toArray(String[]::new);
390 			watcher = enableWatcher ? new WatcherThread(dir, watcherSensitivity) : null;
391 			if (watcher != null)
392 				watcher.start();
393 		} catch (Exception e) {
394 			throw asRuntimeException(e);
395 		}
396 	}

398 	@Override /* ConfigStore */
399 	public synchronized String read(String name) throws IOException {
400 		name = resolveName(name);
401 
402 		var p = resolveFile(name);
403 		name = p.getFileName().toString();
404 
405 		var s = cache.get(name);
406 		if (s != null)
407 			return s;
408 
409 		dir.mkdirs();
410 
411 		// If file doesn't exist, don't trigger creation.
412 		if (! Files.exists(p))
413 			return "";
414 
415 		var isWritable = isWritable(p);
416 		var oo = isWritable ? new OpenOption[]{READ,WRITE,CREATE} : new OpenOption[]{READ};
417 
418 		try (var fc = FileChannel.open(p, oo)) {
419 			try (var lock = isWritable ? fc.lock() : null) {
420 				var buf = ByteBuffer.allocate(1024);
421 				var sb = new StringBuilder();
422 				while (fc.read(buf) != -1) {
423 					sb.append(charset.decode((buf.flip()))); // Fixes Java 11 issue involving overridden flip method.
424 					buf.clear();
425 				}
426 				s = sb.toString();
427 				cache.put(name, s);
428 			}
429 		}
430 
431 		return cache.get(name);
432 	}

434 	@Override /* ConfigStore */
435 	public synchronized String write(String name, String expectedContents, String newContents) throws IOException {
436 		name = resolveName(name);
437 
438 		// This is a no-op.
439 		if (Utils.eq(expectedContents, newContents))
440 			return null;
441 
442 		dir.mkdirs();
443 
444 		var p = resolveFile(name);
445 		name = p.getFileName().toString();
446 
447 		var exists = Files.exists(p);
448 
449 		// Don't create the file if we're not going to match.
450 		if ((!exists) && Utils.isNotEmpty(expectedContents))
451 			return "";
452 
453 		if (isWritable(p)) {
454 			if (newContents == null)
455 				Files.delete(p);
456 			else {
457 				try (var fc = FileChannel.open(p, READ, WRITE, CREATE)) {
458 					try (var lock = fc.lock()) {
459 						var currentContents = "";
460 						if (exists) {
461 							var buf = ByteBuffer.allocate(1024);
462 							var sb = new StringBuilder();
463 							while (fc.read(buf) != -1) {
464 								sb.append(charset.decode(buf.flip()));
465 								buf.clear();
466 							}
467 							currentContents = sb.toString();
468 						}
469 						if (expectedContents != null && ! Utils.eq(currentContents, expectedContents)) {
470 							if (currentContents == null)
471 								cache.remove(name);
472 							else
473 								cache.put(name, currentContents);
474 							return currentContents;
475 						}
476 						fc.position(0);
477 						fc.write(charset.encode(newContents));
478 					}
479 				}
480 			}
481 		}
482 
483 		if (updateOnWrite)
484 			update(name, newContents);
485 		else
486 			cache.remove(name);  // Invalidate the cache.
487 
488 		return null;
489 	}

496 	private Path resolveFile(String name) {
497 		return dir.toPath().resolve(resolveName(name));
498 	}

500 	@Override
501 	protected String resolveName(String name) {
502 		if (! nameCache.containsKey(name)) {
503 			String n = null;
504 
505 			// Does file exist as-is?
506 			if (FileUtils.exists(dir, name))
507 				n = name;
508 
509 			// Does name already have an extension?
510 			if (n == null) {
511 				for (var ext : exts) {
512 					if (FileUtils.hasExtension(name, ext)) {
513 						n = name;
514 						break;
515 					}
516 				}
517 			}
518 
519 			// Find file with the correct extension.
520 			if (n == null) {
521 				for (var ext : exts) {
522 					if (FileUtils.exists(dir, name + '.' + ext)) {
523 						n = name + '.' + ext;
524 						break;
525 					}
526 				}
527 			}
528 
529 			// If file not found, use the default which is the name with the first extension.
530 			if (n == null)
531 				n = exts.length == 0 ? name : (name + "." + exts[0]);
532 
533 			nameCache.put(name, n);
534 		}
535 		return nameCache.get(name);
536 	}

538 	private synchronized boolean isWritable(Path p) {
539 		try {
540 			if (! Files.exists(p)) {
541 				Files.createDirectories(p.getParent());
542 				if (! Files.exists(p) && ! p.toFile().createNewFile()) {
543 					throw new IOException("Could not create file: " + p);
544 				}
545 			}
546 		} catch (IOException e) {
547 			return false;
548 		}
549 		return Files.isWritable(p);
550 	}

569 	class WatcherThread extends Thread {
570 		private final WatchService watchService;
571 
572 		WatcherThread(File dir, WatcherSensitivity s) throws Exception {
573 			watchService = FileSystems.getDefault().newWatchService();
574 			var kinds = new WatchEvent.Kind[]{ENTRY_CREATE, ENTRY_DELETE, ENTRY_MODIFY};
575 			var modifier = lookupModifier(s);
576 			dir.toPath().register(watchService, kinds, modifier);
577 		}
578 
579 		private WatchEvent.Modifier lookupModifier(WatcherSensitivity s) {
580 			try {
581 				return switch (s) {
582 					case LOW    -> com.sun.nio.file.SensitivityWatchEventModifier.LOW;
583 					case MEDIUM -> com.sun.nio.file.SensitivityWatchEventModifier.MEDIUM;
584 					case HIGH   -> com.sun.nio.file.SensitivityWatchEventModifier.HIGH;
585 				};
586 			} catch (Exception e) {
587 				/* Ignore */
588 			}
589 			return null;
590 		}
591 
592 		@SuppressWarnings("unchecked")
593 		@Override /* Thread */
594 		public void run() {
595 			try {
596 				WatchKey key;
597 				while ((key = watchService.take()) != null) {
598 					for (var event : key.pollEvents()) {
599 						var kind = event.kind();
600 						if (kind != OVERFLOW)
601 							FileStore.this.onFileEvent(((WatchEvent<Path>)event));
602 					}
603 					if (! key.reset())
604 						break;
605 				}
606 			} catch (Exception e) {
607 				throw asRuntimeException(e);
608 			}
609 		}
610 
611 		@Override /* Thread */
612 		public void interrupt() {
613 			try {
614 				watchService.close();
615 			} catch (IOException e) {
616 				throw asRuntimeException(e);
617 			} finally {
618 				super.interrupt();
619 			}
620 		}
621 	}

623 	/**
624 	 * Gets called when the watcher service on this store is triggered with a file system change.
625 	 *
626 	 * @param e The file system event.
627 	 * @throws IOException Thrown by underlying stream.
628 	 */
629 	protected synchronized void onFileEvent(WatchEvent<Path> e) throws IOException {
630 		var fn = e.context().getFileName().toString();
631 
632 		var oldContents = cache.get(fn);
633 		cache.remove(fn);
634 		var newContents = read(fn);
635 
636 		if (! Utils.eq(oldContents, newContents)) {
637 			update(fn, newContents);
638 		}
639 	}
