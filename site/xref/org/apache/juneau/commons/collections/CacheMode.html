17  package org.apache.juneau.commons.collections;
18  
19  /**
20   * Cache modes for {@link Cache} and related cache classes.
21   *
22   * <p>
23   * Defines how cache storage is implemented and when cached entries may be evicted.
24   *
25   * <h5 class='section'>See Also:</h5>
26   * <ul>
27   * 	<li class='jc'>{@link Cache}
28   * 	<li class='jc'>{@link Cache2}
29   * 	<li class='jc'>{@link Cache3}
30   * 	<li class='jc'>{@link Cache4}
31   * 	<li class='jc'>{@link Cache5}
32   * </ul>
33   */
34  public enum CacheMode {
35  
36  	/**
37  	 * No caching - all lookups invoke the supplier.
38  	 *
39  	 * <p>
40  	 * When this mode is used, the cache will not store any values. Every call to
41  	 * {@link Cache#get(Object)} or {@link Cache#get(Object, java.util.function.Supplier)}
42  	 * will invoke the supplier to compute the value.
43  	 *
44  	 * <p>
45  	 * This mode is useful for:
46  	 * <ul>
47  	 * 	<li>Testing scenarios where you want fresh computation each time
48  	 * 	<li>Development environments where caching might hide issues
49  	 * 	<li>Temporarily disabling caching without code changes
50  	 * </ul>
51  	 */
52  	NONE,
53  
54  	/**
55  	 * Weak caching - uses {@link java.util.WeakHashMap} for storage.
56  	 *
57  	 * <p>
58  	 * Cache entries can be garbage collected when keys are no longer strongly referenced elsewhere.
59  	 * The WeakHashMap is wrapped with {@link java.util.Collections#synchronizedMap(java.util.Map)}
60  	 * for thread safety.
61  	 *
62  	 * <p>
63  	 * This mode is useful for:
64  	 * <ul>
65  	 * 	<li>Caching metadata about objects that may be unloaded (e.g., {@link Class} objects)
66  	 * 	<li>Memory-sensitive applications where cache entries should not prevent garbage collection
67  	 * 	<li>Scenarios where keys have limited lifetimes
68  	 * </ul>
69  	 *
70  	 * <p>
71  	 * <b>Note:</b> Weak caching comes with performance trade-offs:
72  	 * <ul>
73  	 * 	<li>Slightly slower access due to synchronization overhead
74  	 * 	<li>Entries may be removed unpredictably by the garbage collector
75  	 * 	<li>Not suitable for high-concurrency scenarios
76  	 * </ul>
77  	 */
78  	WEAK,
79  
80  	/**
81  	 * Full caching - uses {@link java.util.concurrent.ConcurrentHashMap} for storage.
82  	 *
83  	 * <p>
84  	 * Provides the best performance with lock-free reads and writes. Cached entries
85  	 * will remain in memory until explicitly removed or the cache is cleared due to
86  	 * exceeding the maximum size.
87  	 *
88  	 * <p>
89  	 * This is the default and recommended mode for most use cases, offering:
90  	 * <ul>
91  	 * 	<li>Excellent performance with no synchronization overhead for reads
92  	 * 	<li>Thread-safe concurrent access
93  	 * 	<li>Predictable memory usage (entries stay until evicted)
94  	 * 	<li>Suitable for high-concurrency scenarios
95  	 * </ul>
96  	 */
97  	FULL;
98  
99  	/**
100 	 * Parses a string value into a {@link CacheMode}.
101 	 *
102 	 * <p>
103 	 * Performs case-insensitive matching against the enum constant names.
104 	 *
105 	 * <h5 class='section'>Examples:</h5>
106 	 * <p class='bjava'>
107 	 * 	CacheMode.<jsm>parse</jsm>(<js>"none"</js>);  <jc>// Returns NONE</jc>
108 	 * 	CacheMode.<jsm>parse</jsm>(<js>"WEAK"</js>);  <jc>// Returns WEAK</jc>
109 	 * 	CacheMode.<jsm>parse</jsm>(<js>"Full"</js>);  <jc>// Returns FULL</jc>
110 	 * 	CacheMode.<jsm>parse</jsm>(<jk>null</jk>);    <jc>// Returns FULL (default)</jc>
111 	 * 	CacheMode.<jsm>parse</jsm>(<js>"invalid"</js>); <jc>// Returns FULL (default)</jc>
112 	 * </p>
113 	 *
114 	 * @param value The string value to parse. Can be <jk>null</jk>.
115 	 * @return The corresponding {@link CacheMode}, or {@link #FULL} if the value is <jk>null</jk> or invalid.
116 	 */
117 	public static CacheMode parse(String value) {
118 		if (value == null)
119 			return FULL;
120 		return switch (value.toUpperCase()) {
121 			case "NONE" -> NONE;
122 			case "WEAK" -> WEAK;
123 			case "FULL" -> FULL;
124 			default -> FULL;
125 		};
126 	}
127 }
