17  package org.apache.juneau.commons.collections;

19  import static org.apache.juneau.commons.utils.AssertionUtils.*;

21  import java.util.*;

23  /**
24   * A fluent wrapper around an arbitrary list that provides convenient methods for adding elements.
25   *
26   * <p>
27   * This class wraps an underlying list and provides a fluent API for adding elements. All methods return
28   * <c>this</c> to allow method chaining. The underlying list can be any {@link List} implementation.
29   *
30   * <h5 class='section'>Features:</h5>
31   * <ul class='spaced-list'>
32   * 	<li><b>Fluent API:</b> All methods return <c>this</c> for method chaining
33   * 	<li><b>Arbitrary List Support:</b> Works with any list implementation
34   * 	<li><b>Conditional Adding:</b> Add elements conditionally based on boolean expressions
35   * 	<li><b>Transparent Interface:</b> Implements the full {@link List} interface, so it can be used anywhere a list is expected
36   * </ul>
37   *
38   * <h5 class='section'>Usage:</h5>
39   * <p class='bjava'>
40   * 	<jc>// Create a FluentList wrapping an ArrayList</jc>
41   * 	FluentList&lt;String&gt; <jv>list</jv> = <jk>new</jk> FluentList&lt;&gt;(<jk>new</jk> ArrayList&lt;&gt;());
42   *
43   * 	<jv>list</jv>
44   * 		.a(<js>"item1"</js>)
45   * 		.a(<js>"item2"</js>)
46   * 		.ai(<jk>true</jk>, <js>"item3"</js>)   <jc>// Added</jc>
47   * 		.ai(<jk>false</jk>, <js>"item4"</js>); <jc>// Not added</jc>
48   *
49   * 	<jc>// Add all elements from another collection</jc>
50   * 	List&lt;String&gt; <jv>other</jv> = List.of(<js>"item5"</js>, <js>"item6"</js>);
51   * 	<jv>list</jv>.aa(<jv>other</jv>);
52   * </p>
53   *
54   * <h5 class='section'>Example - Conditional Building:</h5>
55   * <p class='bjava'>
56   * 	<jk>boolean</jk> <jv>includeDebug</jv> = <jk>true</jk>;
57   * 	<jk>boolean</jk> <jv>includeTest</jv> = <jk>false</jk>;
58   *
59   * 	FluentList&lt;String&gt; <jv>config</jv> = <jk>new</jk> FluentList&lt;&gt;(<jk>new</jk> ArrayList&lt;&gt;())
60   * 		.a(<js>"setting1"</js>)
61   * 		.a(<js>"setting2"</js>)
62   * 		.ai(<jv>includeDebug</jv>, <js>"debug"</js>)   <jc>// Added</jc>
63   * 		.ai(<jv>includeTest</jv>, <js>"test"</js>);    <jc>// Not added</jc>
64   * </p>
65   *
66   * <h5 class='section'>Behavior Notes:</h5>
67   * <ul class='spaced-list'>
68   * 	<li>All list operations are delegated to the underlying list
69   * 	<li>The fluent methods ({@link #a(Object)}, {@link #aa(Collection)}, {@link #ai(boolean, Object)}) return <c>this</c> for chaining
70   * 	<li>If a <jk>null</jk> collection is passed to {@link #aa(Collection)}, it is treated as a no-op
71   * 	<li>The underlying list is stored by reference (not copied), so modifications affect the original list
72   * </ul>
73   *
74   * <h5 class='section'>Thread Safety:</h5>
75   * <p>
76   * This class is not thread-safe unless the underlying list is thread-safe. If thread safety is required,
77   * use a thread-safe list type (e.g., {@link java.util.concurrent.CopyOnWriteArrayList}).
78   *
79   * <h5 class='section'>See Also:</h5>
80   * <ul>
81   * 	<li class='link'><a class="doclink" href="../../../../../index.html#juneau-commons">Overview &gt; juneau-commons</a>
82   * </ul>
83   *
84   * @param <E> The element type.
85   */
86  public class FluentList<E> extends AbstractList<E> {
87  
88  	private final List<E> list;
89  
90  	/**
91  	 * Constructor.
92  	 *
93  	 * @param inner The underlying list to wrap. Must not be <jk>null</jk>.
94  	 */
95  	public FluentList(List<E> inner) {
96  		this.list = assertArgNotNull("inner", inner);
97  	}
98  
99  	/**
100 	 * Adds a single element to this list.
101 	 *
102 	 * <p>
103 	 * This is a convenience method that calls {@link #add(Object)} and returns <c>this</c>
104 	 * for method chaining.
105 	 *
106 	 * <h5 class='section'>Example:</h5>
107 	 * <p class='bjava'>
108 	 * 	FluentList&lt;String&gt; <jv>list</jv> = <jk>new</jk> FluentList&lt;&gt;(<jk>new</jk> ArrayList&lt;&gt;());
109 	 * 	<jv>list</jv>.a(<js>"item1"</js>).a(<js>"item2"</js>);
110 	 * </p>
111 	 *
112 	 * @param element The element to add.
113 	 * @return This object for method chaining.
114 	 */
115 	public FluentList<E> a(E element) {
116 		list.add(element);
117 		return this;
118 	}
119 
120 	/**
121 	 * Adds all elements from the specified collection to this list.
122 	 *
123 	 * <p>
124 	 * This is a convenience method that calls {@link #addAll(Collection)} and returns <c>this</c>
125 	 * for method chaining. If the specified collection is <jk>null</jk>, this is a no-op.
126 	 *
127 	 * <h5 class='section'>Example:</h5>
128 	 * <p class='bjava'>
129 	 * 	FluentList&lt;String&gt; <jv>list</jv> = <jk>new</jk> FluentList&lt;&gt;(<jk>new</jk> ArrayList&lt;&gt;());
130 	 * 	List&lt;String&gt; <jv>other</jv> = List.of(<js>"item1"</js>, <js>"item2"</js>);
131 	 * 	<jv>list</jv>.aa(<jv>other</jv>).a(<js>"item3"</js>);
132 	 * </p>
133 	 *
134 	 * @param c The collection whose elements are to be added. Can be <jk>null</jk> (no-op).
135 	 * @return This object for method chaining.
136 	 */
137 	public FluentList<E> aa(Collection<? extends E> c) {
138 		if (c != null)
139 			list.addAll(c);
140 		return this;
141 	}
142 
143 	/**
144 	 * Adds an element to this list if the specified boolean condition is <jk>true</jk>.
145 	 *
146 	 * <p>
147 	 * This method is useful for conditionally adding elements based on runtime conditions.
148 	 * If the condition is <jk>false</jk>, the element is not added and this method returns <c>this</c>
149 	 * without modifying the list.
150 	 *
151 	 * <h5 class='section'>Example:</h5>
152 	 * <p class='bjava'>
153 	 * 	<jk>boolean</jk> <jv>includeDebug</jv> = <jk>true</jk>;
154 	 * 	<jk>boolean</jk> <jv>includeTest</jv> = <jk>false</jk>;
155 	 *
156 	 * 	FluentList&lt;String&gt; <jv>list</jv> = <jk>new</jk> FluentList&lt;&gt;(<jk>new</jk> ArrayList&lt;&gt;())
157 	 * 		.a(<js>"basic"</js>)
158 	 * 		.ai(<jv>includeDebug</jv>, <js>"debug"</js>)   <jc>// Added</jc>
159 	 * 		.ai(<jv>includeTest</jv>, <js>"test"</js>);    <jc>// Not added</jc>
160 	 * </p>
161 	 *
162 	 * @param condition The condition to evaluate. If <jk>true</jk>, the element is added; if <jk>false</jk>, it is not.
163 	 * @param element The element to add if the condition is <jk>true</jk>.
164 	 * @return This object for method chaining.
165 	 */
166 	public FluentList<E> ai(boolean condition, E element) {
167 		if (condition)
168 			list.add(element);
169 		return this;
170 	}
