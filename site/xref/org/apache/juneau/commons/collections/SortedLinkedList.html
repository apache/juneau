17  package org.apache.juneau.commons.collections;
18  
19  import static org.apache.juneau.commons.utils.AssertionUtils.*;
20  
21  import java.util.*;
22  
23  /**
24   * A sorted list implementation backed by a {@link LinkedList}.
25   *
26   * <p>
27   * Unlike {@link TreeSet}, this class allows duplicate elements and maintains them in sorted order
28   * according to a specified {@link Comparator} or natural ordering. All modification operations
29   * (add, set, etc.) automatically maintain the sorted order.
30   *
31   * <p>
32   * This implementation uses a {@link LinkedList} internally, providing:
33   * <ul>
34   * 	<li>Efficient insertions (O(1) once insertion point is found)
35   * 	<li>No array shifting overhead
36   * 	<li>Linear search for finding insertion points (O(n))
37   * </ul>
38   *
39   * <p>
40   * For better random access performance, consider using {@link SortedArrayList}.
41   *
42   * <h5 class='section'>Features:</h5>
43   * <ul class='spaced-list'>
44   * 	<li>Maintains elements in sorted order automatically
45   * 	<li>Allows duplicate elements (unlike {@link TreeSet})
46   * 	<li>Efficient insertion (O(1) once position is found)
47   * 	<li>No array shifting overhead
48   * 	<li>Supports custom comparators or natural ordering
49   * </ul>
50   *
51   * <h5 class='section'>Performance Characteristics:</h5>
52   * <ul class='spaced-list'>
53   * 	<li><b>add(E)</b>: O(n) - linear search to find position + O(1) insertion
54   * 	<li><b>get(int)</b>: O(n) - must traverse to index
55   * 	<li><b>contains(Object)</b>: O(n) - linear search
56   * 	<li><b>remove(Object)</b>: O(n) - linear search + O(1) removal
57   * 	<li><b>set(int, E)</b>: O(n) - may need to re-sort if element order changes
58   * </ul>
59   *
60   * <h5 class='section'>Usage:</h5>
61   * <p class='bjava'>
62   * 	<jc>// Natural ordering</jc>
63   * 	SortedLinkedList&lt;String&gt; <jv>list</jv> = <jk>new</jk> SortedLinkedList&lt;&gt;();
64   * 	<jv>list</jv>.add(<js>"c"</js>);
65   * 	<jv>list</jv>.add(<js>"a"</js>);
66   * 	<jv>list</jv>.add(<js>"b"</js>);
67   * 	<jc>// list contains: ["a", "b", "c"]</jc>
68   *
69   * 	<jc>// Custom comparator</jc>
70   * 	SortedLinkedList&lt;String&gt; <jv>list2</jv> = <jk>new</jk> SortedLinkedList&lt;&gt;(Comparator.comparing(String::length));
71   * 	<jv>list2</jv>.add(<js>"ccc"</js>);
72   * 	<jv>list2</jv>.add(<js>"a"</js>);
73   * 	<jv>list2</jv>.add(<js>"bb"</js>);
74   * 	<jc>// list2 contains: ["a", "bb", "ccc"] (sorted by length)</jc>
75   * </p>
76   *
77   * <h5 class='section'>Notes:</h5>
78   * <ul class='spaced-list'>
79   * 	<li>All elements must be mutually comparable (or comparable via the comparator)
80   * 	<li>Null elements are allowed only if the comparator supports them
81   * 	<li>The list maintains sorted order at all times - you cannot insert at a specific index
82   * 	<li>Calling {@link #set(int, E)} may cause the element to move to maintain sorted order
83   * 	<li>This implementation is not thread-safe
84   * 	<li>Better suited for frequent insertions and removals, less suited for random access
85   * </ul>
86   *
87   * @param <E> The element type.
88   * @see SortedArrayList
89   */
90  public class SortedLinkedList<E> extends AbstractList<E> {
91  
92  	private final List<E> list;
93  	private final Comparator<? super E> comparator;
94  
95  	/**
96  	 * Creates a new sorted list using natural ordering.
97  	 *
98  	 * <p>
99  	 * Elements must implement {@link Comparable}.
100 	 *
101 	 * @throws ClassCastException if elements are not comparable.
102 	 */
103 	@SuppressWarnings("unchecked")
104 	public SortedLinkedList() {
105 		this((Comparator<? super E>)Comparator.naturalOrder());
106 	}
107 
108 	/**
109 	 * Creates a new sorted list using the specified comparator.
110 	 *
111 	 * @param comparator The comparator to use for sorting. Must not be <jk>null</jk>.
112 	 */
113 	public SortedLinkedList(Comparator<? super E> comparator) {
114 		this.comparator = assertArgNotNull("comparator", comparator);
115 		this.list = new LinkedList<>();
116 	}
117 
118 	/**
119 	 * Creates a new sorted list containing the elements of the specified collection.
120 	 *
121 	 * <p>
122 	 * The elements are sorted according to natural ordering.
123 	 *
124 	 * @param c The collection whose elements are to be placed into this list.
125 	 */
126 	@SuppressWarnings("unchecked")
127 	public SortedLinkedList(Collection<? extends E> c) {
128 		this((Comparator<? super E>)Comparator.naturalOrder(), c);
129 	}
130 
131 	/**
132 	 * Creates a new sorted list containing the elements of the specified collection.
133 	 *
134 	 * @param comparator The comparator to use for sorting. Must not be <jk>null</jk>.
135 	 * @param c The collection whose elements are to be placed into this list.
136 	 */
137 	public SortedLinkedList(Comparator<? super E> comparator, Collection<? extends E> c) {
138 		this.comparator = assertArgNotNull("comparator", comparator);
139 		this.list = new LinkedList<>(c);
140 		Collections.sort(this.list, this.comparator);
141 	}
142 
143 	/**
144 	 * Adds the specified element to this list in sorted order.
145 	 *
146 	 * <p>
147 	 * The element is inserted at the appropriate position to maintain sorted order.
148 	 * If there are equal elements, the new element is inserted after existing equal elements
149 	 * (maintains insertion order for equal elements).
150 	 *
151 	 * @param e The element to be added.
152 	 * @return <jk>true</jk> (as specified by {@link Collection#add(Object)}).
153 	 */
154 	@Override
155 	public boolean add(E e) {
156 		int index = findInsertionPoint(e);
157 		list.add(index, e);
158 		return true;
159 	}
160 
161 	/**
162 	 * Inserts the specified element at the specified position.
163 	 *
164 	 * <p>
165 	 * <b>Note:</b> The index parameter is ignored - the element is always inserted in sorted order.
166 	 * This method exists to satisfy the {@link List} interface contract.
167 	 *
168 	 * @param index Ignored - element is inserted in sorted order.
169 	 * @param element The element to be inserted.
170 	 */
171 	@Override
172 	public void add(int index, E element) {
173 		// Ignore index - always insert in sorted order
174 		add(element);
175 	}
176 
177 	/**
178 	 * Adds all elements from the specified collection to this list in sorted order.
179 	 *
180 	 * @param c The collection containing elements to be added.
181 	 * @return <jk>true</jk> if this list changed as a result of the call.
182 	 */
183 	@Override
184 	public boolean addAll(Collection<? extends E> c) {
185 		boolean modified = false;
186 		for (E e : c) {
187 			add(e);
188 			modified = true;
189 		}
190 		return modified;
191 	}
192 
193 	/**
194 	 * Adds all elements from the specified collection at the specified position.
195 	 *
196 	 * <p>
197 	 * <b>Note:</b> The index parameter is ignored - elements are always inserted in sorted order.
198 	 *
199 	 * @param index Ignored - elements are inserted in sorted order.
200 	 * @param c The collection containing elements to be added.
201 	 * @return <jk>true</jk> if this list changed as a result of the call.
202 	 */
203 	@Override
204 	public boolean addAll(int index, Collection<? extends E> c) {
205 		// Ignore index - always insert in sorted order
206 		return addAll(c);
207 	}
208 
209 	/**
210 	 * Replaces the element at the specified position with the specified element.
211 	 *
212 	 * <p>
213 	 * <b>Important:</b> After setting the element, the list is re-sorted to maintain sorted order.
214 	 * The element may move to a different position if its sort order changed.
215 	 *
216 	 * @param index The index of the element to replace.
217 	 * @param element The element to be stored at the specified position.
218 	 * @return The element previously at the specified position.
219 	 * @throws IndexOutOfBoundsException if the index is out of range.
220 	 */
221 	@Override
222 	public E set(int index, E element) {
223 		E oldValue = list.remove(index);
224 		add(element);  // Re-insert in sorted order
225 		return oldValue;
226 	}
227 
228 	/**
229 	 * Returns the element at the specified position.
230 	 *
231 	 * @param index The index of the element to return.
232 	 * @return The element at the specified position.
233 	 * @throws IndexOutOfBoundsException if the index is out of range.
234 	 */
235 	@Override
236 	public E get(int index) {
237 		return list.get(index);
238 	}
239 
240 	/**
241 	 * Removes the element at the specified position.
242 	 *
243 	 * @param index The index of the element to be removed.
244 	 * @return The element that was removed.
245 	 * @throws IndexOutOfBoundsException if the index is out of range.
246 	 */
247 	@Override
248 	public E remove(int index) {
249 		return list.remove(index);
250 	}
251 
252 	/**
253 	 * Returns the number of elements in this list.
254 	 *
255 	 * @return The number of elements in this list.
256 	 */
257 	@Override
258 	public int size() {
259 		return list.size();
260 	}
261 
262 	/**
263 	 * Returns the comparator used to order the elements in this list.
264 	 *
265 	 * @return The comparator, or <jk>null</jk> if natural ordering is used.
266 	 */
267 	public Comparator<? super E> comparator() {
268 		return comparator == Comparator.naturalOrder() ? null : comparator;
269 	}
270 
271 	/**
272 	 * Finds the insertion point for the specified element using linear search.
273 	 *
274 	 * <p>
275 	 * Since {@link LinkedList} doesn't implement {@link RandomAccess}, binary search would be
276 	 * inefficient. This method uses linear search to find the insertion point.
277 	 *
278 	 * <p>
279 	 * If there are equal elements, the new element is inserted after existing equal elements
280 	 * to maintain insertion order for equal elements.
281 	 *
282 	 * @param e The element to find the insertion point for.
283 	 * @return The index at which the element should be inserted.
284 	 */
285 	private int findInsertionPoint(E e) {
286 		int index = 0;
287 		int size = list.size();
288 		while (index < size && comparator.compare(list.get(index), e) < 0) {
289 			index++;
290 		}
291 		// If we found an equal element, find the last occurrence and insert after it
292 		if (index < size && comparator.compare(list.get(index), e) == 0) {
293 			while (index < size && comparator.compare(list.get(index), e) == 0) {
294 				index++;
295 			}
296 		}
297 		return index;
298 	}
299 
300 	/**
301 	 * Compares the specified object with this list for equality.
302 	 *
303 	 * <p>
304 	 * Returns <jk>true</jk> if and only if the specified object is also a list, both lists have the same size,
305 	 * and all corresponding pairs of elements in the two lists are <i>equal</i>. Two elements <c>e1</c> and
306 	 * <c>e2</c> are considered equal if <c>(e1==null ? e2==null : e1.equals(e2))</c>.
307 	 *
308 	 * @param obj The object to be compared for equality with this list.
309 	 * @return <jk>true</jk> if the specified object is equal to this list.
310 	 */
311 	@Override
312 	public boolean equals(Object obj) {
313 		if (obj == this)
314 			return true;
315 		if (!(obj instanceof List))
316 			return false;
317 		return list.equals(obj);
318 	}
319 
320 	/**
321 	 * Returns the hash code value for this list.
322 	 *
323 	 * <p>
324 	 * The hash code of a list is defined to be the result of the following calculation:
325 	 * <p class='bcode'>
326 	 * 	<jk>int</jk> <jv>hashCode</jv> = 1;
327 	 * 	<jk>for</jk> (<jk>E</jk> <jv>e</jv> : <jv>list</jv>)
328 	 * 		<jv>hashCode</jv> = 31 * <jv>hashCode</jv> + (<jv>e</jv> == <jk>null</jk> ? 0 : <jv>e</jv>.hashCode());
329 	 * </p>
330 	 *
331 	 * @return The hash code value for this list.
332 	 */
333 	@Override
334 	public int hashCode() {
335 		return list.hashCode();
336 	}
337 
338 	/**
339 	 * Returns a string representation of this list.
340 	 *
341 	 * <p>
342 	 * The string representation consists of a list of the list's elements in the order they are returned
343 	 * by its iterator, enclosed in square brackets (<c>"[]"</c>). Adjacent elements are separated by the
344 	 * characters <c>", "</c> (comma and space). Elements are converted to strings as by
345 	 * <c>String.valueOf(Object)</c>.
346 	 *
347 	 * @return A string representation of this list.
348 	 */
349 	@Override
350 	public String toString() {
351 		return list.toString();
352 	}
353 }
