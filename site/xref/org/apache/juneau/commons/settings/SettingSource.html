17  package org.apache.juneau.commons.settings;
18  
19  import java.util.*;
20  
21  /**
22   * Interface for pluggable property sources used by {@link Settings}.
23   *
24   * <p>
25   * A setting source provides a way to retrieve property values.
26   * Sources are checked in reverse order (last added is checked first) when looking up properties.
27   *
28   * <p>
29   * For writable sources that support modifying property values, see {@link SettingStore}.
30   *
31   * <h5 class='section'>Return Value Semantics:</h5>
32   * <ul class='spaced-list'>
33   * 	<li><c>null</c> - The setting does not exist in this source. The lookup will continue to the next source.
34   * 	<li><c>Optional.empty()</c> - The setting exists but has an explicitly null value. This will be returned
35   * 		immediately, overriding any values from lower-priority sources.
36   * 	<li><c>Optional.of(value)</c> - The setting exists and has a non-null value. This will be returned immediately.
37   * </ul>
38   *
39   * <h5 class='section'>Examples:</h5>
40   * <p class='bjava'>
41   * 	<jc>// Create a read-only functional source directly from a lambda</jc>
42   * 	FunctionalSource <jv>readOnly</jv> = name -&gt; opt(System.getProperty(name));
43   *
44   * 	<jc>// Or use the factory method</jc>
45   * 	FunctionalSource <jv>readOnly2</jv> = FunctionalSource.<jsf>of</jsf>(System::getProperty);
46   *
47   * 	<jc>// Stores can be used as sources (they extend SettingSource)</jc>
48   * 	MapStore <jv>store</jv> = <jk>new</jk> MapStore();
49   * 	<jv>store</jv>.set(<js>"my.property"</js>, <js>"value"</js>);
50   * 	Settings.<jsf>get</jsf>().addSource(<jv>store</jv>);  <jc>// Stores can be added as sources</jc>
51   * </p>
52   */
53  public interface SettingSource {
54  
55  	/**
56  	 * Returns a setting in this setting source.
57  	 *
58  	 * <p>
59  	 * Return value semantics:
60  	 * <ul>
61  	 * 	<li><c>null</c> - The setting does not exist in this source. The lookup will continue to the next source.
62  	 * 	<li><c>Optional.empty()</c> - The setting exists but has an explicitly null value. This will be returned
63  	 * 		immediately, overriding any values from lower-priority sources.
64  	 * 	<li><c>Optional.of(value)</c> - The setting exists and has a non-null value. This will be returned immediately.
65  	 * </ul>
66  	 *
67  	 * @param name The property name.
68  	 * @return The property value, <c>null</c> if the property doesn't exist in this source, or <c>Optional.empty()</c>
69  	 * 	if the property exists but has a null value.
70  	 */
71  	Optional<String> get(String name);
72  }
