17  package org.apache.juneau.commons.settings;
18  
19  import static org.apache.juneau.commons.utils.Utils.*;
20  
21  import java.util.*;
22  import java.util.concurrent.*;
23  import java.util.concurrent.atomic.*;
24  
25  /**
26   * A writable {@link SettingSource} implementation backed by a thread-safe map.
27   *
28   * <p>
29   * This class provides a mutable source for settings that can be modified at runtime. It's particularly useful
30   * for creating custom property sources (e.g., Spring properties, configuration files) that can be added to
31   * {@link Settings}.
32   *
33   * <h5 class='section'>Thread Safety:</h5>
34   * <p>
35   * This class is thread-safe. The internal map is lazily initialized using {@link AtomicReference} and
36   * {@link ConcurrentHashMap} for thread-safe operations.
37   *
38   * <h5 class='section'>Null Value Handling:</h5>
39   * <p>
40   * Setting a value to <c>null</c> stores <c>Optional.empty()</c> in the map, which means {@link #get(String)}
41   * will return <c>Optional.empty()</c> (not <c>null</c>). This allows you to explicitly override system properties
42   * with null values. Use {@link #unset(String)} if you want to remove a key entirely (so {@link #get(String)} returns <c>null</c>).
43   *
44   * <h5 class='section'>Example:</h5>
45   * <p class='bjava'>
46   * 	<jc>// Create a source and add properties</jc>
47   * 	MapSource <jv>source</jv> = <jk>new</jk> MapSource();
48   * 	<jv>source</jv>.set(<js>"my.property"</js>, <js>"value"</js>);
49   * 	<jv>source</jv>.set(<js>"another.property"</js>, <js>"another-value"</js>);
50   *
51   * 	<jc>// Add to Settings</jc>
52   * 	Settings.<jsf>get</jsf>().addSource(<jv>source</jv>);
53   *
54   * 	<jc>// Override a system property with null</jc>
55   * 	<jv>source</jv>.set(<js>"system.property"</js>, <jk>null</jk>);
56   * 	<jc>// get() will now return Optional.empty() for "system.property"</jc>
57   *
58   * 	<jc>// Remove a property entirely</jc>
59   * 	<jv>source</jv>.unset(<js>"my.property"</js>);
60   * 	<jc>// get() will now return null for "my.property"</jc>
61   * </p>
62   */
63  public class MapSource implements SettingSource {
64  
65  	private final AtomicReference<Map<String,Optional<String>>> map = new AtomicReference<>();
66  
67  	/**
68  	 * Returns a setting from this source.
69  	 *
70  	 * <p>
71  	 * Returns <c>null</c> if the key doesn't exist in the map, or the stored value (which may be
72  	 * <c>Optional.empty()</c> if the value was explicitly set to <c>null</c>).
73  	 *
74  	 * @param key The property name.
75  	 * @return The property value, <c>null</c> if the key doesn't exist, or <c>Optional.empty()</c> if the key
76  	 * 	exists but has a null value.
77  	 */
78  	@Override
79  	public Optional<String> get(String key) {
80  		var m = map.get();
81  		if (m == null)
82  			return null; // Key not in source (map doesn't exist)
83  		if (! m.containsKey(key))
84  			return null; // Key not in source (key doesn't exist in map)
85  		// Key exists in map - return the value (which may be Optional.empty() if value was set to null)
86  		return m.get(key);
87  	}
88  
89  	/**
90  	 * Sets a setting in this source.
91  	 *
92  	 * <p>
93  	 * The internal map is lazily initialized on the first call to this method. Setting a value to <c>null</c>
94  	 * stores <c>Optional.empty()</c> in the map, which means {@link #get(String)} will return <c>Optional.empty()</c>
95  	 * (not <c>null</c>). This allows you to explicitly override system properties with null values.
96  	 *
97  	 * @param key The property name.
98  	 * @param value The property value, or <c>null</c> to set an empty override.
99  	 */
100 	@Override
101 	public void set(String key, String value) {
102 		if (! canWrite())
103 			return;
104 		var m = map.get();
105 		if (m == null) {
106 			var newMap = new ConcurrentHashMap<String,Optional<String>>();
107 			m = map.compareAndSet(null, newMap) ? newMap : map.get();
108 		}
109 		m.put(key, opt(value));
110 	}
111 
112 	/**
113 	 * Clears all entries from this source.
114 	 *
115 	 * <p>
116 	 * After calling this method, all keys will be removed from the map, and {@link #get(String)} will return
117 	 * <c>null</c> for all keys.
118 	 */
119 	@Override
120 	public void clear() {
121 		if (! canWrite())
122 			return;
123 		var m = map.get();
124 		if (m != null)
125 			m.clear();
126 	}
127 
128 	/**
129 	 * Returns <c>true</c> since this source is writable.
130 	 *
131 	 * @return <c>true</c>
132 	 */
133 	@Override
134 	public boolean canWrite() {
135 		return true;
136 	}
137 
138 	/**
139 	 * Removes a setting from this source.
140 	 *
141 	 * <p>
142 	 * After calling this method, {@link #get(String)} will return <c>null</c> for the specified key,
143 	 * indicating that the key doesn't exist in this source (as opposed to returning <c>Optional.empty()</c>,
144 	 * which would indicate the key exists but has a null value).
145 	 *
146 	 * @param name The property name to remove.
147 	 */
148 	@Override
149 	public void unset(String name) {
150 		if (! canWrite())
151 			return;
152 		var m = map.get();
153 		if (m != null)
154 			m.remove(name);
155 	}
156 }
