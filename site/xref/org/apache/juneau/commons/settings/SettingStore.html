17  package org.apache.juneau.commons.settings;
18  
19  /**
20   * A writable extension of {@link SettingSource} that supports modifying property values.
21   *
22   * <p>
23   * This interface extends {@link SettingSource} with methods for setting, unsetting, and clearing properties.
24   * All stores that implement this interface provide read/write access and can be modified at runtime.
25   *
26   * <p>
27   * <b>Sources vs Stores:</b>
28   * <ul>
29   * 	<li><b>Sources</b> ({@link SettingSource}) - Provide read-only access to property values
30   * 	<li><b>Stores</b> ({@link SettingStore}) - Provide read/write access to property values
31   * </ul>
32   *
33   * <h5 class='section'>Example:</h5>
34   * <p class='bjava'>
35   * 	<jc>// Create a writable store</jc>
36   * 	MapStore <jv>store</jv> = <jk>new</jk> MapStore();
37   * 	<jv>store</jv>.set(<js>"my.property"</js>, <js>"value"</js>);
38   * 	<jv>store</jv>.unset(<js>"my.property"</js>);
39   * 	<jv>store</jv>.clear();
40   * </p>
41   */
42  public interface SettingStore extends SettingSource {
43  
44  	/**
45  	 * Sets a setting in this store.
46  	 *
47  	 * <p>
48  	 * Setting a value to <c>null</c> means that {@link #get(String)} will return <c>Optional.empty()</c> for that key,
49  	 * effectively overriding any values from lower-priority sources. Use {@link #unset(String)} if you want
50  	 * {@link #get(String)} to return <c>null</c> (indicating the key doesn't exist in this store).
51  	 *
52  	 * @param name The property name.
53  	 * @param value The property value, or <c>null</c> to set an empty override.
54  	 */
55  	void set(String name, String value);
56  
57  	/**
58  	 * Removes a setting from this store.
59  	 *
60  	 * <p>
61  	 * After calling this method, {@link #get(String)} will return <c>null</c> for the specified key,
62  	 * indicating that the key doesn't exist in this store (as opposed to returning <c>Optional.empty()</c>,
63  	 * which would indicate the key exists but has a null value).
64  	 *
65  	 * @param name The property name to remove.
66  	 */
67  	void unset(String name);
68  
69  	/**
70  	 * Clears all settings from this store.
71  	 *
72  	 * <p>
73  	 * After calling this method, all keys will be removed from this store, and {@link #get(String)} will
74  	 * return <c>null</c> for all keys.
75  	 */
76  	void clear();
77  }
