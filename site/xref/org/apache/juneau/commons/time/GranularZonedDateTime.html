30  /**
31   * A ZonedDateTime with precision information for granular time operations.
32   *
33   * <p>
34   * This class combines a {@link ZonedDateTime} with a {@link ChronoField} precision identifier,
35   * allowing for granular time operations such as rolling by specific time units.
36   *
37   * <p>
38   * The precision field indicates the granularity of the time value, which determines how the
39   * {@link #roll(int)} method behaves. For example, a precision of {@link ChronoField#YEAR} means
40   * rolling by 1 will advance the year, while a precision of {@link ChronoField#HOUR_OF_DAY} means
41   * rolling by 1 will advance the hour.
42   *
43   * <h5 class='section'>ISO8601 Parsing:</h5>
44   * <p>
45   * The {@link #of(String)} method can parse various ISO8601 timestamp formats:
46   * <ul>
47   * 	<li>Date formats: <js>"2011"</js>, <js>"2011-01"</js>, <js>"2011-01-15"</js>
48   * 	<li>DateTime formats: <js>"2011-01-15T12"</js>, <js>"2011-01-15T12:30"</js>, <js>"2011-01-15T12:30:45"</js>
49   * 	<li>With fractional seconds: <js>"2011-01-15T12:30:45.123"</js>, <js>"2011-01-15T12:30:45,123"</js>
50   * 	<li>Time-only formats: <js>"T12"</js>, <js>"T12:30"</js>, <js>"T12:30:45"</js>
51   * 	<li>With timezone: <js>"2011-01-15T12:30:45Z"</js>, <js>"2011-01-15T12:30:45+05:30"</js>, <js>"2011-01-15T12:30:45-05:30"</js>
52   * </ul>
53   *
54   * <p>
55   * The precision is automatically determined from the input format. For example:
56   * <ul>
57   * 	<li><js>"2011"</js> → {@link ChronoField#YEAR}
58   * 	<li><js>"2011-01"</js> → {@link ChronoField#MONTH_OF_YEAR}
59   * 	<li><js>"2011-01-15"</js> → {@link ChronoField#DAY_OF_MONTH}
60   * 	<li><js>"2011-01-15T12"</js> → {@link ChronoField#HOUR_OF_DAY}
61   * 	<li><js>"2011-01-15T12:30"</js> → {@link ChronoField#MINUTE_OF_HOUR}
62   * 	<li><js>"2011-01-15T12:30:45"</js> → {@link ChronoField#SECOND_OF_MINUTE}
63   * 	<li><js>"2011-01-15T12:30:45.123"</js> → {@link ChronoField#MILLI_OF_SECOND} (1-3 digits)
64   * 	<li><js>"2011-01-15T12:30:45.123456789"</js> → {@link ChronoField#NANO_OF_SECOND} (4-9 digits)
65   * </ul>
66   *
67   * <h5 class='section'>Example:</h5>
68   * <p class='bjava'>
69   * 	<jc>// Parse an ISO8601 timestamp with year precision</jc>
70   * 	GranularZonedDateTime <jv>gdt</jv> = GranularZonedDateTime.<jsm>of</jsm>(<js>"2011"</js>);
71   * 	<jc>// Roll forward by one year</jc>
72   * 	<jv>gdt</jv> = <jv>gdt</jv>.<jsm>roll</jsm>(1);
73   * 	<jc>// Get the ZonedDateTime</jc>
74   * 	ZonedDateTime <jv>zdt</jv> = <jv>gdt</jv>.<jsm>getZonedDateTime</jsm>();
75   * 	<jc>// Result: 2012-01-01T00:00:00 with system default timezone</jc>
76   * </p>
77   *
78   * <p class='bjava'>
79   * 	<jc>// Parse a datetime with hour precision</jc>
80   * 	GranularZonedDateTime <jv>gdt2</jv> = GranularZonedDateTime.<jsm>of</jsm>(<js>"2011-01-15T12Z"</js>);
81   * 	<jc>// Roll forward by 2 hours</jc>
82   * 	<jv>gdt2</jv> = <jv>gdt2</jv>.<jsm>roll</jsm>(<jv>ChronoField</jv>.<jf>HOUR_OF_DAY</jf>, 2);
83   * 	<jc>// Result: 2011-01-15T14:00:00Z</jc>
84   * </p>
85   *
86   * <h5 class='section'>Thread Safety:</h5>
87   * <p>
88   * This class is immutable and thread-safe.
89   *
90   * <h5 class='section'>See Also:</h5><ul>
91   * 	<li class='jm'>{@link ZonedDateTime}
92   * 	<li class='jm'>{@link ChronoField}
93   * 	<li class='jm'>{@link DateUtils}
94   * </ul>
95   */
96  public class GranularZonedDateTime {

98  	/**
99  	 * Creates a GranularZonedDateTime from a Date with the specified precision.
100 	 *
101 	 * <p>
102 	 * The date is converted to a ZonedDateTime using the system default timezone.
103 	 *
104 	 * @param date The date to convert.
105 	 * @param precision The precision of the time value.
106 	 * @return A new GranularZonedDateTime instance.
107 	 * @throws IllegalArgumentException if date or precision is null.
108 	 */
109 	public static GranularZonedDateTime of(Date date, ChronoField precision) {
110 		return of(date, precision, ZoneId.systemDefault());
111 	}

113 	/**
114 	 * Creates a GranularZonedDateTime from a Date with the specified precision and timezone.
115 	 *
116 	 * <p>
117 	 * The date is converted to a ZonedDateTime using the specified timezone.
118 	 *
119 	 * @param date The date to convert.
120 	 * @param precision The precision of the time value.
121 	 * @param zoneId The timezone to use.
122 	 * @return A new GranularZonedDateTime instance.
123 	 * @throws IllegalArgumentException if date, precision, or zoneId is null.
124 	 */

127 	}

129 	/**
130 	 * Parses an ISO8601 timestamp string into a GranularZonedDateTime.
131 	 *
132 	 * <p>
133 	 * This method parses various ISO8601 formats and automatically determines the precision
134 	 * based on the format. If no timezone is specified in the string, the system default
135 	 * timezone is used.
136 	 *
137 	 * <h5 class='section'>Supported Formats:</h5>
138 	 * <ul>
139 	 * 	<li>Date: <js>"2011"</js>, <js>"2011-01"</js>, <js>"2011-01-15"</js>
140 	 * 	<li>DateTime: <js>"2011-01-15T12"</js>, <js>"2011-01-15T12:30"</js>, <js>"2011-01-15T12:30:45"</js>
141 	 * 	<li>With fractional seconds: <js>"2011-01-15T12:30:45.123"</js>, <js>"2011-01-15T12:30:45,123"</js>
142 	 * 	<li>Time-only: <js>"T12"</js>, <js>"T12:30"</js>, <js>"T12:30:45"</js> (uses current date)
143 	 * 	<li>With timezone: <js>"2011-01-15T12:30:45Z"</js>, <js>"2011-01-15T12:30:45+05:30"</js>, <js>"2011-01-15T12:30:45-05:30"</js>
144 	 * </ul>
145 	 *
146 	 * <h5 class='section'>Timezone Handling:</h5>
147 	 * <ul>
148 	 * 	<li>If the string contains a timezone (Z, +HH:mm, -HH:mm, etc.), that timezone is used.
149 	 * 	<li>If no timezone is specified, the system default timezone is used.
150 	 * 	<li>For time-only formats (starting with "T"), the current date is used with the specified or default timezone.
151 	 * </ul>
152 	 *
153 	 * <h5 class='section'>Precision Detection:</h5>
154 	 * <p>
155 	 * The precision is automatically determined from the format:
156 	 * <ul>
157 	 * 	<li>Year only → {@link ChronoField#YEAR}
158 	 * 	<li>Year-Month → {@link ChronoField#MONTH_OF_YEAR}
159 	 * 	<li>Year-Month-Day → {@link ChronoField#DAY_OF_MONTH}
160 	 * 	<li>With hour → {@link ChronoField#HOUR_OF_DAY}
161 	 * 	<li>With minute → {@link ChronoField#MINUTE_OF_HOUR}
162 	 * 	<li>With second → {@link ChronoField#SECOND_OF_MINUTE}
163 	 * 	<li>With 1-3 fractional digits → {@link ChronoField#MILLI_OF_SECOND}
164 	 * 	<li>With 4-9 fractional digits → {@link ChronoField#NANO_OF_SECOND}
165 	 * </ul>
166 	 *
167 	 * @param timestamp The ISO8601 timestamp string to parse.
168 	 * @return A new GranularZonedDateTime instance.
169 	 * @throws IllegalArgumentException if timestamp is null.
170 	 * @throws DateTimeParseException if the timestamp format is invalid.
171 	 */
172 	public static GranularZonedDateTime of(String timestamp) {
173 		return of(timestamp, null);
174 	}

176 	/**
177 	 * Parses an ISO8601 timestamp string into a GranularZonedDateTime with a default timezone.
178 	 *
179 	 * <p>
180 	 * This method is similar to {@link #of(String)}, but allows you to specify a default
181 	 * timezone to use when no timezone is present in the timestamp string.
182 	 *
183 	 * <p>
184 	 * If the timestamp string contains a timezone (Z, +HH:mm, -HH:mm, etc.), that timezone
185 	 * takes precedence over the defaultZoneId parameter. The defaultZoneId is only used when
186 	 * no timezone is specified in the string.
187 	 *
188 	 * <h5 class='section'>Example:</h5>
189 	 * <p class='bjava'>
190 	 * 	<jc>// Parse with default timezone</jc>
191 	 * 	GranularZonedDateTime <jv>gdt1</jv> = GranularZonedDateTime.<jsm>of</jsm>(
192 	 * 		<js>"2011-01-15T12:30:45"</js>,
193 	 * 		<jv>ZoneId</jv>.<jsm>of</jsm>(<js>"America/New_York"</js>)
194 	 * 	);
195 	 * 	<jc>// Result uses America/New_York timezone</jc>
196 	 *
197 	 * 	<jc>// Parse with timezone in string (defaultZoneId is ignored)</jc>
198 	 * 	GranularZonedDateTime <jv>gdt2</jv> = GranularZonedDateTime.<jsm>of</jsm>(
199 	 * 		<js>"2011-01-15T12:30:45Z"</js>,
200 	 * 		<jv>ZoneId</jv>.<jsm>of</jsm>(<js>"America/New_York"</js>)
201 	 * 	);
202 	 * 	<jc>// Result uses UTC (Z), not America/New_York</jc>
203 	 * </p>
204 	 *
205 	 * @param seg The ISO8601 timestamp string to parse.
206 	 * @param defaultZoneId The default timezone to use if no timezone is specified in the string.
207 	 * 	If null, {@link ZoneId#systemDefault()} is used.
208 	 * @return A new GranularZonedDateTime instance.
209 	 * @throws IllegalArgumentException if seg is null.
210 	 * @throws DateTimeParseException if the timestamp format is invalid.
211 	 */
212 	public static GranularZonedDateTime of(String seg, ZoneId defaultZoneId) {
213 		assertArgNotNull("seg", seg);
214 		var digit = StringUtils.DIGIT;
215 
216 		// States:
217 		// S01: Looking for Y(S02) or T(S07).
218 		// S02: Found Y, looking for Y(S02)/-(S03)/T(S07).
219 		// S03: Found -, looking for M(S04).
220 		// S04: Found M, looking for M(S04)/-(S05)/T(S07).
221 		// S05: Found -, looking for D(S10).
222 		// S06  Found D, looking for D(S06)/T(S07).
223 		// S07: Found T, looking for h(S08)/Z(S15)/+(S16)/-(S17).
224 		// S08: Found h, looking for h(S08)/:(S09)/Z(S15)/+(S16)/-(S17).
225 		// S09: Found :, looking for m(S10).
226 		// S10: Found m, looking for m(S10)/:(S11)/Z(S15)/+(S16)/-(S17).
227 		// S11: Found :, looking for s(S12).
228 		// S12: Found s, looking for s(S12)/.(S13)/Z(S15)/+(S16)/-(S17).
229 		// S13: Found ., looking for S(S14)/Z(S15)/+(S16)/-(S17).
230 		// S14: Found S, looking for S(S14)/Z(S15)/+(S16)/-(S17).
231 		// S15: Found Z.
232 		// S16: Found +, looking for oh(S18).
233 		// S17: Found -, looking for oh(S18).
234 		// S18: Found oh, looking for oh(S18)/:(S19).
235 		// S19: Found :, looking for om(S20).
236 		// S20: Found om, looking for om(S20).
237 
238 
239 		int year = 1, month = 1, day = 1, hour = 0, minute = 0, second = 0, nanos = 0, ohour = -1, ominute = -1;
240 		boolean nego = false; // negative offset
241 		boolean timeOnly = false; // Track if format started with "T" (time-only)
242 		ZoneId zoneId = null;
243 		var state = S1;
244 		var mark = 0;
245 		ChronoField precision = ChronoField.YEAR; // Track precision as we go
246 
247 		for (var i = 0; i < seg.length(); i++) {
248 			var c = seg.charAt(i);
249 
250 			if (state == S1) {
251 				// S01: Looking for Y(S02) or T(S07)
252 				if (digit.contains(c)) {
253 					mark = i;
254 					state = S2;
255 				} else if (c == 'T') {
256 					timeOnly = true; // Mark as time-only format
257 					state = S7;
258 				} else {
259 					throw bad(seg, i);
260 				}
261 			} else if (state == S2) {
262 				// S02: Found Y, looking for Y(S02)/-(S03)/T(S07)
263 				if (digit.contains(c)) {
264 					// Stay in S2
265 				} else if (c == '-') {
266 					year = parse(seg, 4, mark, i, 0, 9999);
267 					state = S3;
268 				} else if (c == 'T') {
269 					year = parse(seg, 4, mark, i, 0, 9999);
270 					state = S7;
271 				} else if (c == 'Z') {
272 					zoneId = ZoneId.of("Z");
273 					year = parse(seg, 4, mark, i, 0, 9999);
274 					state = S15;
275 				} else if (c == '+') {
276 					year = parse(seg, 4, mark, i, 0, 9999);
277 					nego = false;
278 					state = S16;
279 				} else {
280 					throw bad(seg, i);
281 				}
282 			} else if (state == S3) {
283 				// S03: Found -, looking for M(S04)
284 				if (digit.contains(c)) {
285 					mark = i;
286 					state = S4;
287 					precision = ChronoField.MONTH_OF_YEAR;
288 				} else {
289 					throw bad(seg, i);
290 				}
291 			} else if (state == S4) {
292 				// S04: Found M, looking for M(S04)/-(S05)/T(S07)
293 				if (digit.contains(c)) {
294 					// Stay in S4
295 				} else if (c == '-') {
296 					month = parse(seg, 2, mark, i, 1, 12);
297 					state = S5;
298 				} else if (c == 'T') {
299 					month = parse(seg, 2, mark, i, 1, 12);
300 					state = S7;
301 				} else if (c == 'Z') {
302 					month = parse(seg, 2, mark, i, 1, 12);
303 					zoneId = ZoneId.of("Z");
304 					state = S15;
305 				} else if (c == '+') {
306 					month = parse(seg, 2, mark, i, 1, 12);
307 					nego = false;
308 					state = S16;
309 				} else {
310 					throw bad(seg, i);
311 				}
312 			} else if (state == S5) {
313 				// S05: Found -, looking for D(S06)
314 				if (digit.contains(c)) {
315 					mark = i;
316 					state = S6;
317 					precision = ChronoField.DAY_OF_MONTH;
318 				} else {
319 					throw bad(seg, i);
320 				}
321 			} else if (state == S6) {
322 				// S06: Found D, looking for D(S06)/T(S07)
323 				if (digit.contains(c)) {
324 					// Stay in S6
325 				} else if (c == 'T') {
326 					day = parse(seg, 2, mark, i, 1, 31);
327 					state = S7;
328 				} else if (c == 'Z') {
329 					day = parse(seg, 2, mark, i, 1, 31);
330 					zoneId = ZoneId.of("Z");
331 					state = S15;
332 				} else if (c == '+') {
333 					day = parse(seg, 2, mark, i, 1, 31);
334 					nego = false;
335 					state = S16;
336 				} else if (c == '-') {
337 					day = parse(seg, 2, mark, i, 1, 31);
338 					nego = true;
339 					state = S17;
340 				} else {
341 					throw bad(seg, i);
342 				}
343 			} else if (state == S7) {
344 				// S07: Found T, looking for h(S08)/Z(S15)/+(S16)/-(S17)
345 				if (digit.contains(c)) {
346 					mark = i;
347 					state = S8;
348 					precision = ChronoField.HOUR_OF_DAY;
349 				} else if (c == 'Z') {
350 					zoneId = ZoneId.of("Z");
351 					state = S15;
352 				} else if (c == '+') {
353 					nego = false;
354 					state = S16;
355 				} else if (c == '-') {
356 					nego = true;
357 					state = S17;
358 				} else {
359 					throw bad(seg, i);
360 				}
361 			} else if (state == S8) {
362 				// S08: Found h, looking for h(S08)/:(S09)/Z(S15)/+(S16)/-(S17)
363 				if (digit.contains(c)) {
364 					// Stay in S8
365 				} else if (c == ':') {
366 					hour = parse(seg, 2, mark, i, 0, 23);
367 					state = S9;
368 				} else if (c == 'Z') {
369 					hour = parse(seg, 2, mark, i, 0, 23);
370 					zoneId = ZoneId.of("Z");
371 					state = S15;
372 				} else if (c == '+') {
373 					hour = parse(seg, 2, mark, i, 0, 23);
374 					nego = false;
375 					state = S16;
376 				} else if (c == '-') {
377 					hour = parse(seg, 2, mark, i, 0, 23);
378 					nego = true;
379 					state = S17;
380 				} else {
381 					throw bad(seg, i);
382 				}
383 			} else if (state == S9) {
384 				// S09: Found :, looking for m(S10)
385 				if (digit.contains(c)) {
386 					mark = i;
387 					state = S10;
388 					precision = ChronoField.MINUTE_OF_HOUR;
389 				} else {
390 					throw bad(seg, i);
391 				}
392 			} else if (state == S10) {
393 				// S10: Found m, looking for m(S10)/:(S11)/Z(S15)/+(S16)/-(S17)
394 				if (digit.contains(c)) {
395 					// Stay in S10
396 				} else if (c == ':') {
397 					minute = parse(seg, 2, mark, i, 0, 59);
398 					state = S11;
399 				} else if (c == 'Z') {
400 					minute = parse(seg, 2, mark, i, 0, 59);
401 					zoneId = ZoneId.of("Z");
402 					state = S15;
403 				} else if (c == '+') {
404 					minute = parse(seg, 2, mark, i, 0, 59);
405 					nego = false;
406 					state = S16;
407 				} else if (c == '-') {
408 					minute = parse(seg, 2, mark, i, 0, 59);
409 					nego = true;
410 					state = S17;
411 				} else {
412 					throw bad(seg, i);
413 				}
414 			} else if (state == S11) {
415 				// S11: Found :, looking for s(S12)
416 				if (digit.contains(c)) {
417 					mark = i;
418 					state = S12;
419 					precision = ChronoField.SECOND_OF_MINUTE;
420 				} else {
421 					throw bad(seg, i);
422 				}
423 			} else if (state == S12) {
424 				// S12: Found s, looking for s(S12)/.(S13)/Z(S15)/+(S16)/-(S17)
425 				if (digit.contains(c)) {
426 					// Stay in S12
427 			} else if (c == '.' || c == ',') {
428 				second = parse(seg, 2, mark, i, 0, 59);
429 				state = S13;
430 				// Precision will be set based on number of fractional digits
431 				} else if (c == 'Z') {
432 					second = parse(seg, 2, mark, i, 0, 59);
433 					zoneId = ZoneId.of("Z");
434 					state = S15;
435 				} else if (c == '+') {
436 					second = parse(seg, 2, mark, i, 0, 59);
437 					nego = false;
438 					state = S16;
439 				} else if (c == '-') {
440 					second = parse(seg, 2, mark, i, 0, 59);
441 					nego = true;
442 					state = S17;
443 				} else {
444 					throw bad(seg, i);
445 				}
446 			} else if (state == S13) {
447 				// S13: Found . or ,, looking for S(S14)/Z(S15)/+(S16)/-(S17)
448 				if (digit.contains(c)) {
449 					mark = i;
450 					state = S14;
451 				} else if (c == 'Z') {
452 					zoneId = ZoneId.of("Z");
453 					state = S15;
454 				} else if (c == '+') {
455 					nego = false;
456 					state = S16;
457 				} else if (c == '-') {
458 					nego = true;
459 					state = S17;
460 				} else {
461 					throw bad(seg, i);
462 				}
463 			} else if (state == S14) {
464 				// S14: Found S, looking for S(S14)/Z(S15)/+(S16)/-(S17)
465 				if (digit.contains(c)) {
466 					// Stay in S14
467 				} else if (c == 'Z') {
468 					nanos = parseNanos(seg, mark, i);
469 					zoneId = ZoneId.of("Z");
470 					// Set precision based on number of fractional digits: 1-3 = milliseconds, 4-9 = nanoseconds
471 					var digitCount = i - mark;
472 					precision = (digitCount <= 3) ? ChronoField.MILLI_OF_SECOND : ChronoField.NANO_OF_SECOND;
473 					state = S15;
474 				} else if (c == '+') {
475 					nanos = parseNanos(seg, mark, i);
476 					// Set precision based on number of fractional digits: 1-3 = milliseconds, 4-9 = nanoseconds
477 					var digitCount = i - mark;
478 					precision = (digitCount <= 3) ? ChronoField.MILLI_OF_SECOND : ChronoField.NANO_OF_SECOND;
479 					nego = false;
480 					state = S16;
481 				} else if (c == '-') {
482 					nanos = parseNanos(seg, mark, i);
483 					// Set precision based on number of fractional digits: 1-3 = milliseconds, 4-9 = nanoseconds
484 					var digitCount = i - mark;
485 					precision = (digitCount <= 3) ? ChronoField.MILLI_OF_SECOND : ChronoField.NANO_OF_SECOND;
486 					nego = true;
487 					state = S17;
488 				} else {
489 					throw bad(seg, i);
490 				}
491 			} else if (state == S15) {
492 				// Shouldn't find anything after Z
493 				throw bad(seg, i);
494 			} else if (state == S16) {
495 				// S16: Found +, looking for oh(S18)
496 				if (digit.contains(c)) {
497 					mark = i;
498 					state = S18;
499 				} else {
500 					throw bad(seg, i);
501 				}
502 			} else if (state == S17) {
503 				// S17: Found -, looking for oh(S18)
504 				if (digit.contains(c)) {
505 					mark = i;
506 					state = S18;
507 				} else {
508 					throw bad(seg, i);
509 				}
510 			} else if (state == S18) {
511 				// S18: Found oh, looking for oh(S18)/:(S19)/end
512 				if (digit.contains(c)) {
513 					// Stay in S18
514 				} else if (c == ':') {
515 					ohour = parse(seg, 2, mark, i, 0, 18);
516 					state = S19;
517 				} else {
518 					throw bad(seg, i);
519 				}
520 				// If we reach end of string, ohour is complete (2 digits)
521 			} else if (state == S19) {
522 				// S19: Found :, looking for om(S20)
523 				if (digit.contains(c)) {
524 					mark = i;
525 					state = S20;
526 				} else {
527 					throw bad(seg, i);
528 				}
529 			} else /* (state == S20) */ {
530 				// S20: Found om, looking for om(S20)
531 				if (digit.contains(c)) {
532 					// Stay in S20
533 				} else {
534 					throw bad(seg, i);
535 				}
536 			}
537 		}
538 
539 		var end = seg.length(); // end is exclusive (one past last character)
540 		if (state.isAny(S1, S3, S5, S7, S9, S11, S13, S16, S17, S19)) {
541 			throw bad(seg, end - 1);
542 		} else if (state == S2) {
543 			// S02: Found Y, looking for Y(S02)/-(S03)/T(S07).
544 			year = parse(seg, 4, mark, end, 0, 9999);
545 			precision = ChronoField.YEAR;
546 		} else if (state == S4) {
547 			// S04: Found M, looking for M(S04)/-(S05)/T(S07).
548 			month = parse(seg, 2, mark, end, 1, 12);
549 			precision = ChronoField.MONTH_OF_YEAR;
550 		} else if (state == S6) {
551 			// S06  Found D, looking for D(S06)/T(S07).
552 			day = parse(seg, 2, mark, end, 1, 31);
553 			precision = ChronoField.DAY_OF_MONTH;
554 		} else if (state == S8) {
555 			// S08: Found h, looking for h(S08)/:(S09)/Z(S15)/+(S16)/-(S17).
556 			hour = parse(seg, 2, mark, end, 0, 23);
557 			precision = ChronoField.HOUR_OF_DAY;
558 		} else if (state == S10) {
559 			// S10: Found m, looking for m(S10)/:(S11)/Z(S15)/+(S16)/-(S17).
560 			minute = parse(seg, 2, mark, end, 0, 59);
561 			precision = ChronoField.MINUTE_OF_HOUR;
562 		} else if (state == S12) {
563 			// S12: Found s, looking for s(S12)/.(S13)/Z(S15)/+(S16)/-(S17).
564 			second = parse(seg, 2, mark, end, 0, 59);
565 			precision = ChronoField.SECOND_OF_MINUTE;
566 		} else if (state == S14) {
567 			// S14: Found S, looking for S(S14)/Z(S15)/+(S16)/-(S17).
568 			nanos = parseNanos(seg, mark, end);
569 			// Set precision based on number of digits: 1-3 = milliseconds, 4-9 = nanoseconds
570 			var digitCount = end - mark;
571 			precision = (digitCount <= 3) ? ChronoField.MILLI_OF_SECOND : ChronoField.NANO_OF_SECOND;
572 		} else if (state == S15) {
573 			// S15: Found Z.
574 		} else if (state == S18) {
575 			// S18: Found oh, looking for oh(S18)/:(S19).
576 			// Check if we have 2 digits (+hh) or 4 digits (+hhmm)
577 			if (end - mark == 2) {
578 				ohour = parse(seg, 2, mark, end, 0, 18);
579 			} else if (end - mark == 4) {
580 				// +hhmm format: parse hours from mark to mark+2, minutes from mark+2 to end
581 				ohour = parse(seg, 2, mark, mark + 2, 0, 18);
582 				ominute = parse(seg, 2, mark + 2, end, 0, 59);
583 			} else {
584 				throw bad(seg, mark);
585 			}
586 		} else /* (state == S20) */ {
587 			// S20: Found om, looking for om(S20).
588 			ominute = parse(seg, 2, mark, end, 0, 59);
589 		}
590 
591 		// Build ZoneId if we have offset information
592 		if (zoneId == null) {
593 			if (ohour >= 0) {
594 				if (ominute >= 0) {
595 					// If negative offset, both hours and minutes must be negative
596 					var offset = ZoneOffset.ofHoursMinutes(nego ? -ohour : ohour, nego ? -ominute : ominute);
597 					zoneId = offset;
598 				} else {
599 					var offset = ZoneOffset.ofHours(nego ? -ohour : ohour);
600 					zoneId = offset;
601 				}
602 			}
603 		}
604 
605 		// Use provided default zone if no zone specified, otherwise use system default
606 		if (zoneId == null) {
607 			zoneId = defaultZoneId != null ? defaultZoneId : ZoneId.systemDefault();
608 		}
609 
610 		// Construct ZonedDateTime from parsed values
611 		// Default values for missing components
612 		// For time-only formats (started with "T"), use current date
613 		// For date formats, default to 1/1/1
614 		if (timeOnly) {
615 			// Time-only format: use current year/month/day
616 			var now = ZonedDateTime.now(zoneId);
617 			year = now.getYear();
618 			month = now.getMonthValue();
619 			day = now.getDayOfMonth();
620 		}
621 
622 		var localDateTime = LocalDateTime.of(year, month, day, hour, minute, second);
623 		if (nanos > 0) {
624 			localDateTime = localDateTime.plusNanos(nanos);
625 		}
626 
627 		var zdt = ZonedDateTime.of(localDateTime, zoneId);
628 
629 		// Return GranularZonedDateTime with the determined precision
630 		return new GranularZonedDateTime(zdt, precision);
631 	}

650 	private static DateTimeParseException bad(String s, int pos) {
651 		return new DateTimeParseException("Invalid ISO8601 timestamp", s, pos);
652 	}
653 
654 	private static int parse(String s, int chars, int pos, int end, int min, int max) {
655 		if (end-pos != chars) throw bad(s, pos);
656 		var i = Integer.parseInt(s, pos, end, 10);
657 		if (i < min || i > max) throw bad(s, pos);
658 		return i;
659 	}
660 
661 	private static int parseNanos(String s, int pos, int end) {
662 		var len = end - pos; // Length of the substring being parsed
663 		if (len > 9) {
664 			throw bad(s, pos);
665 		}
666 		var n = Integer.parseInt(s, pos, end, 10);
667 		// Convert to nanoseconds based on number of digits
668 		// 1 digit = hundreds of milliseconds, 2 = tens, 3 = milliseconds, etc.
669 		if (len == 1) return n * 100000000;
670 		if (len == 2) return n * 10000000;
671 		if (len == 3) return n * 1000000;
672 		if (len == 4) return n * 100000;
673 		if (len == 5) return n * 10000;
674 		if (len == 6) return n * 1000;
675 		if (len == 7) return n * 100;
676 		if (len == 8) return n * 10;
677 		return n;
678 	}

680 	/**
681 	 * Converts a ChronoField to its corresponding ChronoUnit.
682 	 *
683 	 * <p>
684 	 * This method provides a mapping from date/time fields to time units.
685 	 * Not all ChronoField values have direct ChronoUnit equivalents.
686 	 *
687 	 * @param field The ChronoField to convert
688 	 * @return The corresponding ChronoUnit, or null if no direct mapping exists
689 	 */
690 	private static ChronoUnit toChronoUnit(ChronoField field) {
691 		return switch (field) {
692 			case YEAR -> ChronoUnit.YEARS;
693 			case MONTH_OF_YEAR -> ChronoUnit.MONTHS;
694 			case DAY_OF_MONTH -> ChronoUnit.DAYS;
695 			case HOUR_OF_DAY -> ChronoUnit.HOURS;
696 			case MINUTE_OF_HOUR -> ChronoUnit.MINUTES;
697 			case SECOND_OF_MINUTE -> ChronoUnit.SECONDS;
698 			case MILLI_OF_SECOND -> ChronoUnit.MILLIS;
699 			default -> null;
700 		};
701 	}

736 	/**
737 	 * Rolls this time value by the specified amount using the specified field.
738 	 *
739 	 * <p>
740 	 * This method creates a new GranularZonedDateTime by adding the specified amount to the
741 	 * specified field. The precision of the returned object remains the same as this object.
742 	 *
743 	 * <h5 class='section'>Example:</h5>
744 	 * <p class='bjava'>
745 	 * 	<jc>// Create a datetime with hour precision</jc>
746 	 * 	GranularZonedDateTime <jv>gdt</jv> = GranularZonedDateTime.<jsm>of</jsm>(<js>"2011-01-15T12Z"</js>);
747 	 * 	<jc>// Roll forward by 2 hours</jc>
748 	 * 	<jv>gdt</jv> = <jv>gdt</jv>.<jsm>roll</jsm>(<jv>ChronoField</jv>.<jf>HOUR_OF_DAY</jf>, 2);
749 	 * 	<jc>// Result: 2011-01-15T14:00:00Z (precision still HOUR_OF_DAY)</jc>
750 	 * </p>
751 	 *
752 	 * <p>
753 	 * If the field cannot be converted to a ChronoUnit (e.g., unsupported field), this method
754 	 * returns this object unchanged.
755 	 *
756 	 * @param field The field to roll by (e.g., {@link ChronoField#YEAR}, {@link ChronoField#HOUR_OF_DAY}).
757 	 * @param amount The amount to roll by. Positive values roll forward, negative values roll backward.
758 	 * @return A new GranularZonedDateTime with the rolled value, or this object if the field is unsupported.
759 	 */
760 	public GranularZonedDateTime roll(ChronoField field, int amount) {
761 		ChronoUnit unit = toChronoUnit(field);
762 		if (nn(unit)) {
763 			ZonedDateTime newZdt = zdt.plus(amount, unit);
764 			return new GranularZonedDateTime(newZdt, precision);
765 		}
766 		return this;
767 	}
