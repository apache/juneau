22  /**
23   * Utility methods for argument validation and assertion.
24   *
25   * <p>
26   * This class provides static utility methods for validating method arguments and throwing
27   * {@link IllegalArgumentException} when validation fails. These methods are designed to
28   * simplify common parameter validation patterns and provide consistent error messages.
29   *
30   * <h5 class='section'>Features:</h5>
31   * <ul>
32   *   <li><b>Null checking:</b> Validate single or multiple arguments are not null
33   *   <li><b>Boolean expressions:</b> Assert arbitrary boolean conditions with custom messages
34   *   <li><b>String validation:</b> Check for null or blank strings
35   *   <li><b>Type checking:</b> Validate class array types
36   * </ul>
37   *
38   * <h5 class='section'>Usage:</h5>
39   * <p class='bjava'>
40   *   <jk>import static</jk> org.apache.juneau.commons.utils.AssertionUtils.*;
41   *
42   *   <jk>public</jk> <jk>void</jk> setFooBar(String <jv>foo</jv>, String <jv>bar</jv>) {
43   *       <jc>// Validate multiple arguments at once</jc>
44   *       <jsm>assertArgsNotNull</jsm>(<js>"foo"</js>, <jv>foo</jv>, <js>"bar"</js>, <jv>bar</jv>);
45   *
46   *       <jc>// Validate custom condition</jc>
47   *       <jsm>assertArg</jsm>(<jv>foo</jv>.length() &gt; 0, <js>"Foo cannot be empty"</js>);
48   *   }
49   * </p>
50   *
51   * <h5 class='section'>See Also:</h5>
52   * <ul>
53   *   <li class='link'><a class="doclink" href='../../../../../index.html#juneau-commons.utils'>Overview &gt; juneau-commons.utils</a>
54   * </ul>
55   */

58  	/**
59  	 * Throws an {@link IllegalArgumentException} if the specified expression is <jk>false</jk>.
60  	 *
61  	 * <h5 class='section'>Example:</h5>
62  	 * <p class='bjava'>
63  	 * 	<jk>import static</jk> org.apache.juneau.commons.utils.AssertionUtils.*;
64  	 *
65  	 *	<jk>public</jk> String setFoo(List&lt;String&gt; <jv>foo</jv>) {
66  	 *		<jsm>assertArg</jsm>(<jv>foo</jv> != <jk>null</jk> &amp;&amp; ! <jv>foo</jv>.isEmpty(), <js>"'foo' cannot be null or empty."</js>);
67  	 *		...
68  	 *	}
69  	 * </p>
70  	 *
71  	 * @param expression The boolean expression to check.
72  	 * @param msg The exception message.
73  	 * @param args The exception message args.
74  	 * @throws IllegalArgumentException Constructed exception.
75  	 */
76  	public static final void assertArg(boolean expression, String msg, Object...args) throws IllegalArgumentException {
77  		if (! expression)
78  			throw illegalArg(msg, args);
79  	}

82  	 * Throws an {@link IllegalArgumentException} if the specified argument is <jk>null</jk>.
83  	 *
84  	 * <h5 class='section'>Example:</h5>
85  	 * <p class='bjava'>
86  	 * 	<jk>import static</jk> org.apache.juneau.commons.utils.AssertionUtils.*;
87  	 *
88  	 *	<jk>public</jk> String setFoo(String <jv>foo</jv>) {
89  	 *		<jsm>assertArgNotNull</jsm>(<js>"foo"</js>, <jv>foo</jv>);
90  	 *		...
91  	 *	}
92  	 * </p>
93  	 *
94  	 * @param <T> The argument data type.
95  	 * @param name The argument name.
96  	 * @param o The object to check.
97  	 * @return The same argument.
98  	 * @throws IllegalArgumentException Constructed exception.
99  	 */
100 	public static final <T> T assertArgNotNull(String name, T o) throws IllegalArgumentException {
101 		assertArg(o != null, "Argument ''{0}'' cannot be null.", name);
102 		return o;
103 	}

108 	 * @param name The argument name.
109 	 * @param o The object to check.
110 	 * @return The same object.
111 	 * @throws IllegalArgumentException Thrown if the specified string is <jk>null</jk> or blank.

113 	public static final String assertArgNotNullOrBlank(String name, String o) throws IllegalArgumentException {
114 		assertArgNotNull(name, o);
115 		assertArg(! o.isBlank(), "Argument ''{0}'' cannot be blank.", name);
116 		return o;

214 	/**
215 	 * Throws an {@link IllegalArgumentException} if the specified object is not an instance of the specified type.
216 	 *
217 	 * <h5 class='section'>Example:</h5>
218 	 * <p class='bjava'>
219 	 * 	<jk>import static</jk> org.apache.juneau.commons.utils.AssertionUtils.*;
220 	 *
221 	 *	<jk>public</jk> <jk>void</jk> setValue(Object <jv>value</jv>) {
222 	 *		<jc>// Validate that value is a String</jc>
223 	 *		<jsm>assertType</jsm>(String.<jk>class</jk>, <jv>value</jv>);
224 	 *		...
225 	 *	}
226 	 * </p>
227 	 *
228 	 * @param <T> The expected type.
229 	 * @param type The expected class type.
230 	 * @param o The object to check.
231 	 * @return The object cast to the specified type.
232 	 * @throws IllegalArgumentException Thrown if the object is not an instance of the specified type.
233 	 */
234 	@SuppressWarnings({ "unchecked" })
235 	public static final <T> T assertType(Class<T> type, Object o) throws IllegalArgumentException {
236 		assertArgNotNull("type", type);
237 		assertArgNotNull("o", o);
238 		if (! type.isInstance(o))
239 			throw illegalArg("Object is not an instance of {0}: {1}", cn(type), cn(o));
240 		return (T)o;
241 	}

243 	/**
244 	 * Throws the exception provided by the supplier if the specified object is not an instance of the specified type.
245 	 *
246 	 * <h5 class='section'>Example:</h5>
247 	 * <p class='bjava'>
248 	 * 	<jk>import static</jk> org.apache.juneau.commons.utils.AssertionUtils.*;
249 	 *
250 	 *	<jk>public</jk> <jk>void</jk> setValue(Object <jv>value</jv>) {
251 	 *		<jc>// Validate that value is a String, throw custom exception</jc>
252 	 *		<jsm>assertType</jsm>(String.<jk>class</jk>, <jv>value</jv>, () -&gt; <jk>new</jk> IllegalStateException(<js>"Invalid value type"</js>));
253 	 *		...
254 	 *	}
255 	 * </p>
256 	 *
257 	 * @param <T> The expected type.
258 	 * @param type The expected class type.
259 	 * @param o The object to check.
260 	 * @param exceptionSupplier The supplier that provides the exception to throw if validation fails.
261 	 * @return The object cast to the specified type.
262 	 * @throws RuntimeException Thrown if the object is not an instance of the specified type (the exception is provided by the supplier).
263 	 */
264 	@SuppressWarnings({ "unchecked" })
265 	public static final <T> T assertType(Class<T> type, Object o, java.util.function.Supplier<? extends RuntimeException> exceptionSupplier) throws RuntimeException {
266 		assertArgNotNull("type", type);
267 		assertArgNotNull("o", o);
268 		if (! type.isInstance(o))
269 			throw exceptionSupplier.get();
270 		return (T)o;
271 	}

273 	/**
274 	 * Throws an {@link IllegalArgumentException} if the specified value doesn't have all subclasses of the specified type.
275 	 *
276 	 * @param <E> The element type.
277 	 * @param name The argument name.
278 	 * @param type The expected parent class.
279 	 * @param value The array value being checked.
280 	 * @return The value cast to the specified array type.
281 	 * @throws IllegalArgumentException Constructed exception.
282 	 */
283 	@SuppressWarnings("unchecked")
284 	public static final <E> Class<E>[] assertClassArrayArgIsType(String name, Class<E> type, Class<?>[] value) throws IllegalArgumentException {
285 		for (var i = 0; i < value.length; i++)
286 			if (! type.isAssignableFrom(value[i]))
287 				throw illegalArg("Arg {0} did not have arg of type {1} at index {2}: {3}", name, cn(type), i, cn(value[i]));
288 		return (Class<E>[])value;
289 	}

291 	/**
292 	 * Throws an {@link AssertionError} if the specified actual value is not one of the expected values.
293 	 *
294 	 * @param <T> The value type.
295 	 * @param actual The actual value.
296 	 * @param expected The expected values.
297 	 * @return The actual value if it matches one of the expected values.
298 	 * @throws AssertionError if the value is not one of the expected values.
299 	 */
300 	@SafeVarargs
301 	public static final <T> T assertOneOf(T actual, T...expected) {
302 		for (var e : expected) {
303 			if (eq(actual, e))
304 				return actual;
305 		}
306 		throw new AssertionError("Invalid value specified: " + actual);
307 	}

310 	 * Throws an {@link IllegalArgumentException} if the specified varargs array or any of its elements are <jk>null</jk>.
311 	 *
312 	 * @param <T> The element type.
313 	 * @param name The argument name.
314 	 * @param o The object to check.
315 	 * @return The same object.
316 	 * @throws IllegalArgumentException Thrown if the specified varargs array or any of its elements are <jk>null</jk>.
317 	 */
318 	public static final <T> T[] assertVarargsNotNull(String name, T[] o) throws IllegalArgumentException {
319 		assertArgNotNull(name, o);
320 		for (var i = 0; i < o.length; i++)
321 			assertArg(nn(o[i]), "Argument ''{0}'' parameter {1} cannot be null.", name, i);
322 		return o;
323 	}
