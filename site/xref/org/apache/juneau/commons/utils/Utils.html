79  	 *
80  	 * <p>
81  	 * This is a null-safe and type-safe casting operation. Returns <jk>null</jk> if:
82  	 * <ul>

84  	 *   <li>The object is not an instance of the specified type</li>
85  	 * </ul>
86  	 *
87  	 * <h5 class='section'>Example:</h5>
88  	 * <p class='bjava'>
89  	 * 	Object <jv>obj</jv> = <js>"Hello"</js>;

92  	 * 	String <jv>str2</jv> = cast(String.<jk>class</jk>, <jk>null</jk>);   <jc>// null</jc>
93  	 * </p>
94  	 *
95  	 * @param <T> The type to cast to.
96  	 * @param c The type to cast to.
97  	 * @param o The object to cast.
98  	 * @return The cast object, or <jk>null</jk> if the object wasn't the specified type or was <jk>null</jk>.
99  	 * @see #castOrNull(Object, Class)
100 	 */

102 		return nn(o) && c.isInstance(o) ? c.cast(o) : null;
103 	}

106 	 * Casts an object to a specific type if it's an instance of that type.
107 	 *
108 	 * <p>
109 	 * This method is similar to {@link #cast(Class, Object)} but with a different parameter order.
110 	 * Returns <jk>null</jk> if the object is not an instance of the specified class.

119 	 * 	String <jv>str2</jv> = castOrNull(<jk>null</jk>, String.<jk>class</jk>);   <jc>// null</jc>
120 	 * </p>
121 	 *
122 	 * @param <T> The class to cast to.
123 	 * @param o The object to cast.
124 	 * @param c The class to cast to.
125 	 * @return The cast object, or <jk>null</jk> if the object wasn't an instance of the specified class.
126 	 * @see #cast(Class, Object)
127 	 */
128 	public static <T> T castOrNull(Object o, Class<T> c) {
129 		if (c.isInstance(o))
130 			return c.cast(o);
131 		return null;

248 	 * Compares two objects for ordering.
249 	 *
250 	 * <p>
251 	 * This method attempts to compare two objects using their natural ordering if they implement
252 	 * {@link Comparable} and are of the same type. Null handling:
253 	 * <ul>
254 	 *   <li>Both <jk>null</jk> → returns <c>0</c> (equal)</li>
255 	 *   <li>First <jk>null</jk> → returns <c>-1</c> (null is less-than)</li>
256 	 *   <li>Second <jk>null</jk> → returns <c>1</c> (null is less-than)</li>
257 	 *   <li>Different types or not Comparable → returns <c>0</c> (cannot compare)</li>
258 	 * </ul>
259 	 *
260 	 * <h5 class='section'>Example:</h5>
261 	 * <p class='bjava'>
262 	 * 	compare(<js>"apple"</js>, <js>"banana"</js>);   <jc>// negative (apple &lt; banana)</jc>
263 	 * 	compare(5, 10);                                 <jc>// negative (5 &lt; 10)</jc>
264 	 * 	compare(<js>"apple"</js>, <js>"apple"</js>);    <jc>// 0 (equal)</jc>
265 	 * 	compare(<jk>null</jk>, <jk>null</jk>);          <jc>// 0 (equal)</jc>
266 	 * 	compare(<jk>null</jk>, <js>"apple"</js>);       <jc>// -1 (null &lt; non-null)</jc>
267 	 * 	compare(<js>"apple"</js>, 5);                   <jc>// 0 (different types, cannot compare)</jc>
268 	 * </p>
269 	 *
270 	 * @param o1 Object 1.
271 	 * @param o2 Object 2.
272 	 * @return <c>-1</c>, <c>0</c>, or <c>1</c> if <c>o1</c> is less-than, equal, or greater-than <c>o2</c>.
273 	 *         Returns <c>0</c> if objects are not of the same type or do not implement the {@link Comparable} interface.
274 	 */
275 	@SuppressWarnings("unchecked")
276 	public static int compare(Object o1, Object o2) {
277 		if (o1 == null) {
278 			if (o2 == null)
279 				return 0;
280 			return -1;
281 		} else if (o2 == null) {
282 			return 1;
283 		}
284 
285 		if (eq(o1.getClass(), o2.getClass()) && o1 instanceof Comparable o1a)
286 			return o1a.compareTo(o2);
287 
288 		return 0;
289 	}

439 	 * Tests two objects for equality, gracefully handling nulls and arrays.
440 	 *
441 	 * <p>
442 	 * This method handles annotations specially by delegating to {@link AnnotationUtils#equals(Annotation, Annotation)}
443 	 * to ensure proper annotation comparison according to the annotation equality contract.
444 	 *
445 	 * @param <T> The value types.
446 	 * @param o1 Object 1.
447 	 * @param o2 Object 2.
448 	 * @return <jk>true</jk> if both objects are equal based on the {@link Object#equals(Object)} method.
449 	 * @see AnnotationUtils#equals(Annotation, Annotation)
450 	 */
451 	public static <T> boolean eq(T o1, T o2) {
452 		// Handle annotations specially
453 		if (o1 instanceof java.lang.annotation.Annotation o1a && o2 instanceof java.lang.annotation.Annotation o2a)
454 			return AnnotationUtils.equals(o1a, o2a);
455 
456 		if (isArray(o1) && isArray(o2)) {
457 			int l1 = Array.getLength(o1), l2 = Array.getLength(o2);
458 			if (l1 != l2)
459 				return false;
460 			for (var i = 0; i < l1; i++)
461 				if (ne(Array.get(o1, i), Array.get(o2, i)))
462 					return false;
463 			return true;
464 		}
465 		return Objects.equals(o1, o2);
466 	}

483 	 * 			eq(x.createdBy, y.createdBy)
484 	 * 		);
485 	 * 	}

487 	 * 	<jc>// Usage</jc>
