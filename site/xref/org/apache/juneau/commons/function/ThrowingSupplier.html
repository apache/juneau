17  package org.apache.juneau.commons.function;
18  
19  /**
20   * A functional interface representing a supplier of results that may throw a checked exception.
21   *
22   * <p>
23   * This interface extends the standard Java {@link java.util.function.Supplier} pattern to allow
24   * the {@link #get()} method to throw checked exceptions. This is useful when you need to pass
25   * suppliers that may throw exceptions (such as I/O operations) to methods that expect functional
26   * interfaces.
27   *
28   * <h5 class='section'>Features:</h5>
29   * <ul class='spaced-list'>
30   * 	<li>Functional interface - can be used with lambda expressions and method references
31   * 	<li>Exception support - allows checked exceptions to be thrown
32   * 	<li>Compatible with Supplier - can be used where Supplier is expected (exceptions are wrapped)
33   * </ul>
34   *
35   * <h5 class='section'>Use Cases:</h5>
36   * <ul class='spaced-list'>
37   * 	<li>Suppliers that perform I/O operations (file reading, network calls)
38   * 	<li>Suppliers that may throw business logic exceptions
39   * 	<li>Lazy initialization that may fail
40   * 	<li>Resource creation that may throw exceptions
41   * </ul>
42   *
43   * <h5 class='section'>Usage:</h5>
44   * <p class='bjava'>
45   * 	<jc>// Lambda with exception</jc>
46   * 	ThrowingSupplier&lt;String&gt; <jv>fileReader</jv> = () -&gt; {
47   * 		<jk>return new</jk> String(Files.readAllBytes(Paths.get(<js>"data.txt"</js>)));
48   * 	};
49   *
50   * 	<jc>// Method reference</jc>
51   * 	ThrowingSupplier&lt;Connection&gt; <jv>dbConnection</jv> = <jv>dataSource</jv>::getConnection;
52   *
53   * 	<jc>// Usage in try-catch</jc>
54   * 	<jk>try</jk> {
55   * 		String <jv>content</jv> = <jv>fileReader</jv>.get();
56   * 	} <jk>catch</jk> (Exception <jv>e</jv>) {
57   * 		<jc>// Handle exception</jc>
58   * 	}
59   * </p>
60   *
61   * <h5 class='section'>Comparison with Supplier:</h5>
62   * <ul class='spaced-list'>
63   * 	<li><b>Supplier:</b> Cannot throw checked exceptions (must catch and wrap)
64   * 	<li><b>ThrowingSupplier:</b> Can throw checked exceptions directly
65   * 	<li><b>Supplier:</b> Used for standard functional programming patterns
66   * 	<li><b>ThrowingSupplier:</b> Used when operations may fail with checked exceptions
67   * </ul>
68   *
69   * <h5 class='section'>See Also:</h5><ul>
70   * 	<li class='jc'>{@link ThrowingFunction} - Function that throws exceptions
71   * 	<li class='jc'>{@link ThrowingConsumer} - Consumer that throws exceptions
72   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauCommonFunction">juneau-common-function</a>
73   * </ul>
74   *
75   * @param <T> The type of results supplied by this supplier.
76   */
77  @FunctionalInterface
78  public interface ThrowingSupplier<T> {
79  
80  	/**
81  	 * Gets a result, potentially throwing a checked exception.
82  	 *
83  	 * <h5 class='section'>Example:</h5>
84  	 * <p class='bjava'>
85  	 * 	ThrowingSupplier&lt;String&gt; <jv>supplier</jv> = () -&gt; {
86  	 * 		<jk>if</jk> (<jv>fileNotFound</jv>) {
87  	 * 			<jk>throw new</jk> FileNotFoundException(<js>"File not found"</js>);
88  	 * 		}
89  	 * 		<jk>return</jk> <js>"content"</js>;
90  	 * 	};
91  	 *
92  	 * 	<jk>try</jk> {
93  	 * 		String <jv>result</jv> = <jv>supplier</jv>.get();
94  	 * 	} <jk>catch</jk> (FileNotFoundException <jv>e</jv>) {
95  	 * 		<jc>// Handle exception</jc>
96  	 * 	}
97  	 * </p>
98  	 *
99  	 * @return A result.
100 	 * @throws Exception If an error occurs during result computation.
101 	 */
102 	T get() throws Exception;
103 }
