17  package org.apache.juneau.commons.function;
18  
19  import static org.apache.juneau.commons.utils.AssertionUtils.*;
20  
21  /**
22   * A functional interface representing an operation that accepts three arguments and returns no result.
23   *
24   * <p>
25   * This interface extends the standard Java {@link java.util.function.Consumer} pattern to support
26   * three-argument consumers. It's useful when you need to pass operations with three parameters to methods
27   * that expect functional interfaces, such as in iteration, builders, or callback patterns.
28   *
29   * <h5 class='section'>Features:</h5>
30   * <ul class='spaced-list'>
31   * 	<li>Functional interface - can be used with lambda expressions and method references
32   * 	<li>Composition support - provides {@link #andThen(Consumer3)} for operation chaining
33   * 	<li>Side-effect operations - designed for operations that perform side effects rather than return values
34   * </ul>
35   *
36   * <h5 class='section'>Use Cases:</h5>
37   * <ul class='spaced-list'>
38   * 	<li>Three-argument operations in iteration patterns
39   * 	<li>Builder methods that accept three-parameter operations
40   * 	<li>Callback patterns with three parameters
41   * 	<li>Event handlers that process three values
42   * </ul>
43   *
44   * <h5 class='section'>Usage:</h5>
45   * <p class='bjava'>
46   * 	<jc>// Lambda expression</jc>
47   * 	Consumer3&lt;String,Integer,Boolean&gt; <jv>printer</jv> = (<jv>name</jv>, <jv>age</jv>, <jv>active</jv>) -&gt;
48   * 		System.<jsf>out</jsf>.println(<jv>name</jv> + <js>" is "</js> + <jv>age</jv> + <js>" ("</js> + (<jv>active</jv> ? <js>"active"</js> : <js>"inactive"</js>) + <js>")"</js>);
49   * 	<jv>printer</jv>.apply(<js>"Alice"</js>, 30, <jk>true</jk>);  <jc>// Prints "Alice is 30 (active)"</jc>
50   *
51   * 	<jc>// Operation composition</jc>
52   * 	Consumer3&lt;String,Integer,Boolean&gt; <jv>validate</jv> = (<jv>s</jv>, <jv>i</jv>, <jv>b</jv>) -&gt; {
53   * 		<jk>if</jk> (<jv>i</jv> &lt; 0) <jk>throw new</jk> IllegalArgumentException();
54   * 	};
55   * 	Consumer3&lt;String,Integer,Boolean&gt; <jv>process</jv> = (<jv>s</jv>, <jv>i</jv>, <jv>b</jv>) -&gt; <jsm>process</jsm>(<jv>s</jv>, <jv>i</jv>, <jv>b</jv>);
56   * 	Consumer3&lt;String,Integer,Boolean&gt; <jv>validateAndProcess</jv> = <jv>validate</jv>.andThen(<jv>process</jv>);
57   * 	<jv>validateAndProcess</jv>.apply(<js>"data"</js>, 42, <jk>true</jk>);  <jc>// Validates then processes</jc>
58   * </p>
59   *
60   * <h5 class='section'>See Also:</h5><ul>
61   * 	<li class='jc'>{@link Consumer2} - Two-argument consumer
62   * 	<li class='jc'>{@link Consumer4} - Four-argument consumer
63   * 	<li class='jc'>{@link Consumer5} - Five-argument consumer
64   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauCommonsFunction">juneau-commons-function</a>
65   * </ul>
66   *
67   * @param <A> The type of the first argument to the operation.
68   * @param <B> The type of the second argument to the operation.
69   * @param <C> The type of the third argument to the operation.
70   */
71  @FunctionalInterface
72  public interface Consumer3<A,B,C> {
73  
74  	/**
75  	 * Returns a composed {@link Consumer3} that performs, in sequence, this operation followed by the {@code after} operation.
76  	 *
77  	 * <p>
78  	 * This method enables operation composition, allowing you to chain multiple operations together.
79  	 * If performing either operation throws an exception, it is relayed to the caller of the composed operation.
80  	 * If performing this operation throws an exception, the {@code after} operation will not be performed.
81  	 *
82  	 * <h5 class='section'>Example:</h5>
83  	 * <p class='bjava'>
84  	 * 	Consumer3&lt;String,Integer,Boolean&gt; <jv>log</jv> = (<jv>s</jv>, <jv>i</jv>, <jv>b</jv>) -&gt; <jsm>logger</jsm>.info(<jv>s</jv> + <js>": "</js> + <jv>i</jv> + <js>", "</js> + <jv>b</jv>);
85  	 * 	Consumer3&lt;String,Integer,Boolean&gt; <jv>store</jv> = (<jv>s</jv>, <jv>i</jv>, <jv>b</jv>) -&gt; <jv>cache</jv>.put(<jv>s</jv>, <jv>i</jv>, <jv>b</jv>);
86  	 * 	Consumer3&lt;String,Integer,Boolean&gt; <jv>logAndStore</jv> = <jv>log</jv>.andThen(<jv>store</jv>);
87  	 * 	<jv>logAndStore</jv>.apply(<js>"data"</js>, 100, <jk>true</jk>);  <jc>// Logs and stores</jc>
88  	 * </p>
89  	 *
90  	 * @param after The operation to perform after this operation. Must not be <jk>null</jk>.
91  	 * @return A composed {@link Consumer3} that performs in sequence this operation followed by the {@code after} operation.
92  	 * @throws NullPointerException if {@code after} is <jk>null</jk>.
93  	 */
94  	default Consumer3<A,B,C> andThen(Consumer3<? super A,? super B,? super C> after) {  // NOSONAR - false positive on generics
95  		assertArgNotNull("after", after);
96  		return (A a, B b, C c) -> {
97  			apply(a, b, c);
98  			after.apply(a, b, c);
99  		};
100 	}
101 
102 	/**
103 	 * Performs this operation on the given arguments.
104 	 *
105 	 * <h5 class='section'>Example:</h5>
106 	 * <p class='bjava'>
107 	 * 	Consumer3&lt;String,Integer,Boolean&gt; <jv>handler</jv> = (<jv>key</jv>, <jv>value</jv>, <jv>flag</jv>) -&gt; {
108 	 * 		<jv>map</jv>.put(<jv>key</jv>, <jv>value</jv>);
109 	 * 		<jk>if</jk> (<jv>flag</jv>) <jv>cache</jv>.invalidate(<jv>key</jv>);
110 	 * 	};
111 	 * 	<jv>handler</jv>.apply(<js>"user"</js>, 12345, <jk>true</jk>);
112 	 * </p>
113 	 *
114 	 * @param a The first operation argument.
115 	 * @param b The second operation argument.
116 	 * @param c The third operation argument.
117 	 */
118 	void apply(A a, B b, C c);
119 }
