17  package org.apache.juneau.commons.function;
18  
19  import static org.apache.juneau.commons.utils.ThrowableUtils.*;
20  
21  import java.util.function.*;
22  
23  /**
24   * A functional interface representing a function that accepts one argument, produces a result, and may throw a checked exception.
25   *
26   * <p>
27   * This interface extends the standard Java {@link java.util.function.Function} to allow the functional method
28   * to throw checked exceptions. The default {@link #apply(Object)} method wraps any checked exceptions in a
29   * {@link RuntimeException}, making it compatible with standard {@link Function} usage while still allowing
30   * the implementation to throw checked exceptions via {@link #applyThrows(Object)}.
31   *
32   * <h5 class='section'>Features:</h5>
33   * <ul class='spaced-list'>
34   * 	<li>Functional interface - can be used with lambda expressions and method references
35   * 	<li>Exception support - allows checked exceptions to be thrown via {@link #applyThrows(Object)}
36   * 	<li>Compatible with Function - implements {@link Function#apply(Object)} by wrapping exceptions
37   * 	<li>Dual interface - can be used as both Function and ThrowingFunction
38   * </ul>
39   *
40   * <h5 class='section'>Use Cases:</h5>
41   * <ul class='spaced-list'>
42   * 	<li>Transformations that may throw I/O exceptions
43   * 	<li>Parsing operations that may throw validation exceptions
44   * 	<li>Data conversion that may fail with checked exceptions
45   * 	<li>Operations that need to propagate checked exceptions
46   * </ul>
47   *
48   * <h5 class='section'>Usage:</h5>
49   * <p class='bjava'>
50   * 	<jc>// Lambda with exception</jc>
51   * 	ThrowingFunction&lt;String,File&gt; <jv>fileParser</jv> = (<jv>path</jv>) -&gt; {
52   * 		<jk>if</jk> (! Files.exists(Paths.get(<jv>path</jv>))) {
53   * 			<jk>throw new</jk> FileNotFoundException(<js>"File not found: "</js> + <jv>path</jv>);
54   * 		}
55   * 		<jk>return new</jk> File(<jv>path</jv>);
56   * 	};
57   *
58   * 	<jc>// Using applyThrows to get checked exception</jc>
59   * 	<jk>try</jk> {
60   * 		File <jv>file</jv> = <jv>fileParser</jv>.applyThrows(<js>"/path/to/file"</js>);
61   * 	} <jk>catch</jk> (FileNotFoundException <jv>e</jv>) {
62   * 		<jc>// Handle checked exception</jc>
63   * 	}
64   *
65   * 	<jc>// Using apply wraps exception in RuntimeException</jc>
66   * 	File <jv>file</jv> = <jv>fileParser</jv>.apply(<js>"/path/to/file"</js>);  <jc>// May throw RuntimeException</jc>
67   * </p>
68   *
69   * <h5 class='section'>Exception Handling:</h5>
70   * <ul class='spaced-list'>
71   * 	<li>{@link #applyThrows(Object)} - Throws checked exceptions directly
72   * 	<li>{@link #apply(Object)} - Wraps checked exceptions in {@link RuntimeException}
73   * 	<li>Use {@code applyThrows} when you need to handle specific checked exceptions
74   * 	<li>Use {@code apply} when you want standard Function behavior
75   * </ul>
76   *
77   * <h5 class='section'>See Also:</h5><ul>
78   * 	<li class='jc'>{@link ThrowingSupplier} - Supplier that throws exceptions
79   * 	<li class='jc'>{@link ThrowingConsumer} - Consumer that throws exceptions
80   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauCommonFunction">juneau-common-function</a>
81   * </ul>
82   *
83   * @param <T> The type of the input to the function.
84   * @param <R> The type of the result of the function.
85   */
86  @FunctionalInterface
87  public interface ThrowingFunction<T,R> extends Function<T,R> {
88  
89  	/**
90  	 * Applies this function to the given argument, wrapping any checked exceptions in a {@link RuntimeException}.
91  	 *
92  	 * <p>
93  	 * This is the default implementation that makes this interface compatible with {@link Function}.
94  	 * It calls {@link #applyThrows(Object)} and wraps any checked exceptions.
95  	 *
96  	 * @param t The function argument.
97  	 * @return The function result.
98  	 * @throws RuntimeException if {@link #applyThrows(Object)} throws a checked exception.
99  	 */
100 	@Override
101 	default R apply(T t) {
102 		try {
103 			return applyThrows(t);
104 		} catch (Exception e) {
105 			throw toRex(e);
106 		}
107 	}
108 
109 	/**
110 	 * Applies this function to the given argument, potentially throwing a checked exception.
111 	 *
112 	 * <p>
113 	 * This is the functional method that implementations must provide. It allows checked exceptions
114 	 * to be thrown directly, unlike the standard {@link Function#apply(Object)} method.
115 	 *
116 	 * <h5 class='section'>Example:</h5>
117 	 * <p class='bjava'>
118 	 * 	ThrowingFunction&lt;String,Integer&gt; <jv>parser</jv> = (<jv>s</jv>) -&gt; {
119 	 * 		<jk>try</jk> {
120 	 * 			<jk>return</jk> Integer.parseInt(<jv>s</jv>);
121 	 * 		} <jk>catch</jk> (NumberFormatException <jv>e</jv>) {
122 	 * 			<jk>throw new</jk> ParseException(<js>"Invalid number: "</js> + <jv>s</jv>, 0);
123 	 * 		}
124 	 * 	};
125 	 *
126 	 * 	<jk>try</jk> {
127 	 * 		<jk>int</jk> <jv>value</jv> = <jv>parser</jv>.applyThrows(<js>"123"</js>);
128 	 * 	} <jk>catch</jk> (ParseException <jv>e</jv>) {
129 	 * 		<jc>// Handle checked exception</jc>
130 	 * 	}
131 	 * </p>
132 	 *
133 	 * @param t The function argument.
134 	 * @return The function result.
135 	 * @throws Exception If an error occurs during function execution.
136 	 */
137 	R applyThrows(T t) throws Exception;
138 }
