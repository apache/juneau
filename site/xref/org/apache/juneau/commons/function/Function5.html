17  package org.apache.juneau.commons.function;
18  
19  import static org.apache.juneau.commons.utils.AssertionUtils.*;
20  
21  import java.util.function.*;
22  
23  /**
24   * A functional interface representing a function that accepts five arguments and produces a result.
25   *
26   * <p>
27   * This interface extends the standard Java {@link java.util.function.Function} pattern to support
28   * five-argument functions. It's useful when you need to pass functions with five parameters to methods
29   * that expect functional interfaces, such as in stream operations, builders, or callback patterns.
30   *
31   * <h5 class='section'>Features:</h5>
32   * <ul class='spaced-list'>
33   * 	<li>Functional interface - can be used with lambda expressions and method references
34   * 	<li>Composition support - provides {@link #andThen(Function)} for function chaining
35   * 	<li>Type-safe - generic type parameters ensure compile-time type safety
36   * </ul>
37   *
38   * <h5 class='section'>Use Cases:</h5>
39   * <ul class='spaced-list'>
40   * 	<li>Five-argument transformations in functional programming patterns
41   * 	<li>Builder methods that accept five-parameter functions
42   * 	<li>Stream operations requiring five-argument functions
43   * 	<li>Callback patterns with five parameters
44   * </ul>
45   *
46   * <h5 class='section'>Usage:</h5>
47   * <p class='bjava'>
48   * 	<jc>// Lambda expression</jc>
49   * 	Function5&lt;String,Integer,Boolean,Double,Long,String&gt; <jv>format</jv> = (<jv>s</jv>, <jv>i</jv>, <jv>b</jv>, <jv>d</jv>, <jv>l</jv>) -&gt;
50   * 		<jv>s</jv> + <js>"-"</js> + <jv>i</jv> + <js>"-"</js> + (<jv>b</jv> ? <js>"Y"</js> : <js>"N"</js>) + <js>"-"</js> + <jv>d</jv> + <js>"-"</js> + <jv>l</jv>;
51   * 	String <jv>result</jv> = <jv>format</jv>.apply(<js>"prefix"</js>, 42, <jk>true</jk>, 95.5, 1234567890L);
52   *
53   * 	<jc>// Function composition</jc>
54   * 	Function5&lt;Integer,Integer,Integer,Integer,Integer,Integer&gt; <jv>add</jv> = (<jv>a</jv>, <jv>b</jv>, <jv>c</jv>, <jv>d</jv>, <jv>e</jv>) -&gt; <jv>a</jv> + <jv>b</jv> + <jv>c</jv> + <jv>d</jv> + <jv>e</jv>;
55   * 	Function5&lt;Integer,Integer,Integer,Integer,Integer,String&gt; <jv>addAndFormat</jv> = <jv>add</jv>.andThen(Object::toString);
56   * 	String <jv>sum</jv> = <jv>addAndFormat</jv>.apply(5, 3, 2, 1, 1);  <jc>// Returns "12"</jc>
57   * </p>
58   *
59   * <h5 class='section'>See Also:</h5><ul>
60   * 	<li class='jc'>{@link Function2} - Two-argument function
61   * 	<li class='jc'>{@link Function3} - Three-argument function
62   * 	<li class='jc'>{@link Function4} - Four-argument function
63   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/JuneauCommonFunction">juneau-common-function</a>
64   * </ul>
65   *
66   * @param <A> The type of the first argument to the function.
67   * @param <B> The type of the second argument to the function.
68   * @param <C> The type of the third argument to the function.
69   * @param <D> The type of the fourth argument to the function.
70   * @param <E> The type of the fifth argument to the function.
71   * @param <R> The type of the result of the function.
72   */
73  @FunctionalInterface
74  public interface Function5<A,B,C,D,E,R> {
75  
76  	/**
77  	 * Returns a composed function that first applies this function to its input, and then applies
78  	 * the {@code after} function to the result.
79  	 *
80  	 * <p>
81  	 * This method enables function composition, allowing you to chain multiple transformations together.
82  	 *
83  	 * <h5 class='section'>Example:</h5>
84  	 * <p class='bjava'>
85  	 * 	Function5&lt;Integer,Integer,Integer,Integer,Integer,Integer&gt; <jv>multiply</jv> = (<jv>a</jv>, <jv>b</jv>, <jv>c</jv>, <jv>d</jv>, <jv>e</jv>) -&gt; <jv>a</jv> * <jv>b</jv> * <jv>c</jv> * <jv>d</jv> * <jv>e</jv>;
86  	 * 	Function5&lt;Integer,Integer,Integer,Integer,Integer,String&gt; <jv>multiplyAndFormat</jv> = <jv>multiply</jv>.andThen(<jv>n</jv> -&gt; <js>"Result: "</js> + <jv>n</jv>);
87  	 * 	String <jv>result</jv> = <jv>multiplyAndFormat</jv>.apply(2, 2, 2, 2, 2);  <jc>// Returns "Result: 32"</jc>
88  	 * </p>
89  	 *
90  	 * @param <V> The type of output of the {@code after} function, and of the composed function.
91  	 * @param after The function to apply after this function is applied. Must not be <jk>null</jk>.
92  	 * @return A composed function that first applies this function and then applies the {@code after} function.
93  	 * @throws NullPointerException if {@code after} is <jk>null</jk>.
94  	 */
95  	default <V> Function5<A,B,C,D,E,V> andThen(Function<? super R,? extends V> after) {
96  		assertArgNotNull("after", after);
97  		return (A a, B b, C c, D d, E e) -> after.apply(apply(a, b, c, d, e));
98  	}
99  
100 	/**
101 	 * Applies this function to the given arguments.
102 	 *
103 	 * <h5 class='section'>Example:</h5>
104 	 * <p class='bjava'>
105 	 * 	Function5&lt;String,Integer,Boolean,Double,Long,String&gt; <jv>format</jv> = (<jv>s</jv>, <jv>n</jv>, <jv>upper</jv>, <jv>mult</jv>, <jv>time</jv>) -&gt; {
106 	 * 		String <jv>result</jv> = <jv>s</jv>.repeat(<jv>n</jv>);
107 	 * 		<jv>result</jv> = <jv>upper</jv> ? <jv>result</jv>.toUpperCase() : <jv>result</jv>;
108 	 * 		<jk>return</jk> <jv>result</jv> + <js>" x"</js> + <jv>mult</jv> + <js>" @"</js> + <jv>time</jv>;
109 	 * 	};
110 	 * 	String <jv>result</jv> = <jv>format</jv>.apply(<js>"ha"</js>, 2, <jk>true</jk>, 1.5, 1234567890L);
111 	 * </p>
112 	 *
113 	 * @param a The first function argument.
114 	 * @param b The second function argument.
115 	 * @param c The third function argument.
116 	 * @param d The fourth function argument.
117 	 * @param e The fifth function argument.
118 	 * @return The function result.
119 	 */
120 	R apply(A a, B b, C c, D d, E e);
121 }
