13  package org.apache.juneau.parser;
14  
15  import org.apache.juneau.*;
16  
17  /**
18   * Represents a parser and media type that matches an HTTP <c>Content-Type</c> header value.
19   *
20   * <h5 class='section'>See Also:</h5><ul>
21   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SerializersAndParsers">Serializers and Parsers</a>
22   * </ul>
23   */
24  public final class ParserMatch {
25  
26  	private final MediaType mediaType;
27  	private final Parser parser;
28  
29  	/**
30  	 * Constructor.
31  	 *
32  	 * @param mediaType The media type of the match.
33  	 * @param parser The parser that matched.
34  	 */
35  	public ParserMatch(MediaType mediaType, Parser parser) {
36  		this.mediaType = mediaType;
37  		this.parser = parser;
38  	}
39  
40  	/**
41  	 * Returns the media type of the parser that matched the HTTP <c>Content-Type</c> header value.
42  	 *
43  	 * @return The media type of the match.
44  	 */
45  	public MediaType getMediaType() {
46  		return mediaType;
47  	}
48  
49  	/**
50  	 * Returns the parser that matched the HTTP <c>Content-Type</c> header value.
51  	 *
52  	 * @return The parser of the match.
53  	 */
54  	public Parser getParser() {
55  		return parser;
56  	}
57  }
