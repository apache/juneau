33  /**
34   * Represents a group of {@link Parser Parsers} that can be looked up by media type.
35   *
36   * <h5 class='topic'>Description</h5>
37   *
38   * Provides the following features:
39   * <ul class='spaced-list'>
40   * 	<li>
41   * 		Finds parsers based on HTTP <c>Content-Type</c> header values.
42   * 	<li>
43   * 		Sets common properties on all parsers in a single method call.
44   * 	<li>
45   * 		Locks all parsers in a single method call.
46   * 	<li>
47   * 		Clones existing groups and all parsers within the group in a single method call.
48   * </ul>
49   *
50   * <h5 class='topic'>Match ordering</h5>
51   *
52   * Parsers are matched against <c>Content-Type</c> strings in the order they exist in this group.
53   *
54   * <p>
55   * Adding new entries will cause the entries to be prepended to the group.
56   * This allows for previous parsers to be overridden through subsequent calls.
57   *
58   * <p>
59   * For example, calling <code>g.append(P1.<jk>class</jk>,P2.<jk>class</jk>).append(P3.<jk>class</jk>,P4.<jk>class</jk>)</code>
60   * will result in the order <c>P3, P4, P1, P2</c>.
61   *
62   * <h5 class='section'>Example:</h5>
63   * <p class='bjava'>
64   * 	<jc>// Construct a new parser group builder</jc>
65   * 	ParserSet <jv>parsers</jv> = ParserSet.<jsm>create</jsm>();
66   * 		.add(JsonParser.<jk>class</jk>, XmlParser.<jk>class</jk>); <jc>// Add some parsers to it</jc>
67   *		.forEach(<jv>x</jv> -&gt; <jv>x</jv>.swaps(CalendarSwap.IsoLocalDateTime.<jk>class</jk>))
68   *		.forEach(<jv>x</jv> -&gt; <jv>x</jv>.beansRequireSerializable())
69   * 		.build();
70   *
71   * 	<jc>// Find the appropriate parser by Content-Type</jc>
72   * 	ReaderParser <jv>parser</jv> = (ReaderParser)<jv>parsers</jv>.getParser(<js>"text/json"</js>);
73   *
74   * 	<jc>// Parse a bean from JSON</jc>
75   * 	String <jv>json</jv> = <js>"{...}"</js>;
76   * 	AddressBook <jv>addressBook</jv> = <jv>parser</jv>.parse(<jv>json</jv>, AddressBook.<jk>class</jk>);
77   * </p>
78   *
79   * <h5 class='section'>Notes:</h5><ul>
80   * 	<li class='note'>This class is thread safe and reusable.
81   * </ul>
82   *
83   * <h5 class='section'>See Also:</h5><ul>
84   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SerializersAndParsers">Serializers and Parsers</a>
85  
86   * </ul>
87   */

160 		 * Clone an existing parser group.

162 		 * @param copyFrom The parser group that we're copying settings and parsers from.

164 		protected Builder(ParserSet copyFrom) {
165 			super(copyFrom.getClass(), BeanStore.INSTANCE);
166 			this.entries = list((Object[])copyFrom.entries);
167 		}
168 
169 		/**
170 		 * Clone an existing parser group builder.
171 		 *
172 		 * <p>
173 		 * Parser builders will be cloned during this process.
174 		 *
175 		 * @param copyFrom The parser group that we're copying settings and parsers from.
176 		 */
177 		protected Builder(Builder copyFrom) {
178 			super(copyFrom);
179 			bcBuilder = copyFrom.bcBuilder == null ? null : copyFrom.bcBuilder.copy();
180 			entries = list();
181 			copyFrom.entries.stream().map(this::copyBuilder).forEach(x -> entries.add(x));
182 		}
183 
184 		private Object copyBuilder(Object o) {
185 			if (o instanceof Parser.Builder) {
186 				Parser.Builder x = (Parser.Builder)o;
187 				Parser.Builder x2 = x.copy();
188 				if (Utils.ne(x.getClass(), x2.getClass()))
189 					throw new BasicRuntimeException("Copy method not implemented on class {0}", x.getClass().getName());
190 				x = x2;
191 				if (bcBuilder != null)
192 					x.beanContext(bcBuilder);
193 				return x;
194 			}
195 			return o;
196 		}

240 		/**
241 		 * Adds the specified parsers to this group.
242 		 *
243 		 * <p>
244 		 * Entries are added in-order to the beginning of the list in the group.
245 		 *
246 		 * <p>
247 		 * The {@link NoInherit} class can be used to clear out the existing list of parsers before adding the new entries.
248 		 *
249 		 * <h5 class='section'>Example:</h5>
250 		 * <p class='bjava'>
251 		 * 	ParserSet.Builder <jv>builder</jv> = ParserSet.<jsm>create</jsm>();  <jc>// Create an empty builder.</jc>
252 		 *
253 		 * 	<jv>builder</jv>.add(FooParser.<jk>class</jk>);  <jc>// Now contains:  [FooParser]</jc>
254 		 *
255 		 * 	<jv>builder</jv>.add(BarParser.<jk>class</jk>, BazParser.<jk>class</jk>);  <jc>// Now contains:  [BarParser,BazParser,FooParser]</jc>
256 		 *
257 		 * 	<jv>builder</jv>.add(NoInherit.<jk>class</jk>, QuxParser.<jk>class</jk>);  <jc>// Now contains:  [QuxParser]</jc>
258 		 * </p>
259 		 *
260 		 * @param values The parsers to add to this group.
261 		 * @return This object.
262 		 * @throws IllegalArgumentException If one or more values do not extend from {@link Parser}.
263 		 */
264 		public Builder add(Class<?>...values) {
265 			List<Object> l = list();
266 			for (Class<?> v : values)
267 				if (v.getSimpleName().equals("NoInherit"))
268 					clear();
269 			for (Class<?> v : values) {
270 				if (Parser.class.isAssignableFrom(v)) {
271 					l.add(createBuilder(v));
272 				} else if (! v.getSimpleName().equals("NoInherit")) {
273 					throw new BasicRuntimeException("Invalid type passed to ParserSet.Builder.add(): {0}", v.getName());
274 				}
275 			}
276 			entries.addAll(0, l);
277 			return this;
278 		}
279 
280 		/**
281 		 * Sets the specified parsers for this group.
282 		 *
283 		 * <p>
284 		 * Existing values are overwritten.
285 		 *
286 		 * <p>
287 		 * The {@link Inherit} class can be used to insert existing entries in this group into the position specified.
288 		 *
289 		 * <h5 class='section'>Example:</h5>
290 		 * <p class='bjava'>
291 		 * 	ParserSet.Builder <jv>builder</jv> = ParserSet.<jsm>create</jsm>();  <jc>// Create an empty builder.</jc>
292 		 *
293 		 * 	<jv>builder</jv>.set(FooParser.<jk>class</jk>);  <jc>// Now contains:  [FooParser]</jc>
294 		 *
295 		 * 	<jv>builder</jv>.set(BarParser.<jk>class</jk>, BazParser.<jk>class</jk>);  <jc>// Now contains:  [BarParser,BazParser]</jc>
296 		 *
297 		 * 	<jv>builder</jv>.set(Inherit.<jk>class</jk>, QuxParser.<jk>class</jk>);  <jc>// Now contains:  [BarParser,BazParser,QuxParser]</jc>
298 		 * </p>
299 		 *
300 		 * @param values The parsers to set in this group.
301 		 * @return This object.
302 		 * @throws IllegalArgumentException If one or more values do not extend from {@link Parser} or named <js>"Inherit"</js>.
303 		 */
304 		public Builder set(Class<?>...values) {
305 			List<Object> l = list();
306 			for (Class<?> v : values) {
307 				if (v.getSimpleName().equals("Inherit")) {
308 					l.addAll(entries);
309 				} else if (Parser.class.isAssignableFrom(v)) {
310 					l.add(createBuilder(v));
311 				} else {
312 					throw new BasicRuntimeException("Invalid type passed to ParserGrouup.Builder.set(): {0}", v.getName());
313 				}
314 			}
315 			entries = l;
316 			return this;
317 		}
318 
319 		private Object createBuilder(Object o) {
320 			if (o instanceof Class) {
321 
322 				// Check for no-arg constructor.
323 				ConstructorInfo ci = ClassInfo.of((Class<?>)o).getPublicConstructor(ConstructorInfo::hasNoParams);
324 				if (ci != null)
325 					return ci.invoke();
326 
327 				// Check for builder.
328 				@SuppressWarnings("unchecked")
329 				Parser.Builder b = Parser.createParserBuilder((Class<? extends Parser>)o);
330 				if (bcBuilder != null)
331 					b.beanContext(bcBuilder);
332 				o = b;
333 			}
334 			return o;
335 		}

478 	// Maps Content-Type headers to matches.
479 	private final ConcurrentHashMap<String,ParserMatch> cache = new ConcurrentHashMap<>();
480 
481 	private final MediaType[] mediaTypes;
482 	private final Parser[] mediaTypeParsers;
483 
484 	final Parser[] entries;
485 
486 	/**
487 	 * Constructor.
488 	 *
489 	 * @param builder The builder for this bean.
490 	 */
491 	public ParserSet(Builder builder) {
492 
493 		this.entries = builder.entries.stream().map(this::build).toArray(Parser[]::new);
494 
495 		List<MediaType> lmt = list();
496 		List<Parser> l = list();
497 		for (Parser e : entries) {
498 			e.getMediaTypes().forEach(x -> {
499 				lmt.add(x);
500 				l.add(e);
501 			});
502 		}
503 
504 		this.mediaTypes = Utils.array(lmt, MediaType.class);
505 		this.mediaTypeParsers = Utils.array(l, Parser.class);
506 	}
507 
508 	private Parser build(Object o) {
509 		if (o instanceof Parser)
510 			return (Parser)o;
511 		return ((Parser.Builder)o).build();
512 	}

523 	/**
524 	 * Searches the group for a parser that can handle the specified <l>Content-Type</l> header value.
525 	 *
526 	 * <p>
527 	 * The returned object includes both the parser and media type that matched.
528 	 *
529 	 * @param contentTypeHeader The HTTP <l>Content-Type</l> header value.
530 	 * @return The parser and media type that matched the content type header, or <jk>null</jk> if no match was made.
531 	 */
532 	public ParserMatch getParserMatch(String contentTypeHeader) {
533 		ParserMatch pm = cache.get(contentTypeHeader);
534 		if (pm != null)
535 			return pm;
536 
537 		MediaType ct = MediaType.of(contentTypeHeader);
538 		int match = ct.match(alist(mediaTypes));
539 
540 		if (match >= 0) {
541 			pm = new ParserMatch(mediaTypes[match], mediaTypeParsers[match]);
542 			cache.putIfAbsent(contentTypeHeader, pm);
543 		}
544 
545 		return cache.get(contentTypeHeader);
546 	}
547 
548 	/**
549 	 * Same as {@link #getParserMatch(String)} but matches using a {@link MediaType} instance.
550 	 *

552 	 * @return The parser and media type that matched the media type, or <jk>null</jk> if no match was made.
553 	 */
554 	public ParserMatch getParserMatch(MediaType mediaType) {
555 		return getParserMatch(mediaType.toString());
556 	}
