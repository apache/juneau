17  package org.apache.juneau.parser;
18  
19  import static org.apache.juneau.common.utils.StringUtils.*;
20  
21  import org.apache.juneau.*;
22  
23  /**
24   * Class for listening for certain parse events during a document parse.
25   *
26   * <h5 class='section'>See Also:</h5><ul>
27   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SerializersAndParsers">Serializers and Parsers</a>
28  
29   * </ul>
30   */
31  public class ParserListener {
32  
33  	/**
34  	 * Represents no parser listener.
35  	 */
36  	public static final class Void extends ParserListener {}
37  
38  	/**
39  	 * Called when an exception is thrown when trying to call a bean setter method.
40  	 *
41  	 * @param session The serializer session.
42  	 * @param t The throwable that was thrown by the setter method.
43  	 * @param p The bean property we had an issue on.
44  	 */
45  	public void onBeanSetterException(ParserSession session, Throwable t, BeanPropertyMeta p) {
46  		onError(session, t, format("Could not call setValue() on property ''{0}'' of class ''{1}'', exception = {2}",
47  			p.getName(), p.getBeanMeta().getClassMeta(), t.getLocalizedMessage()));
48  	}
49  
50  	/**
51  	 * Called when an error occurs during parsing but is ignored.
52  	 *
53  	 * @param session The parser session.
54  	 * @param t The throwable that was thrown by the getter method.
55  	 * @param msg The error message.
56  	 */
57  	public void onError(ParserSession session, Throwable t, String msg) {
58  		// Do something with this information.
59  	}
60  
61  	/**
62  	 * Gets called when an unknown bean property is detected in a document.
63  	 *
64  	 * <p>
65  	 * This method only gets called if {@link org.apache.juneau.BeanContext.Builder#ignoreUnknownBeanProperties()} setting is <jk>true</jk>.
66  	 * Otherwise, the parser will throw a {@link ParseException}.
67  	 *
68  	 * @param <T> The class type of the bean.
69  	 * @param session The parser session.
70  	 * @param propertyName The property name encountered in the document.
71  	 * @param beanClass The bean class.
72  	 * @param bean The bean.
73  	 */
74  	public <T> void onUnknownBeanProperty(ParserSession session, String propertyName, Class<T> beanClass, T bean) {
75  		onError(session, null,
76  			format("Unknown property ''{0}'' encountered while trying to parse into class ''{1}'' at location {2}",
77  				propertyName, beanClass, session.getPosition())
78  		);
79  	}
80  }
