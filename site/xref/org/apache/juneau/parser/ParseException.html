29  /**
30   * Exception that indicates invalid syntax encountered during parsing.
31   *
32   * <h5 class='section'>See Also:</h5><ul>
33   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/SerializersAndParsers">Serializers and Parsers</a>
34  
35   * </ul>
36   *
37   * @serial exclude
38   */
39  public class ParseException extends BasicRuntimeException {
40  
41  	private static final long serialVersionUID = 1L;
42  
43  	/**
44  	 * Creator method.
45  	 *
46  	 * <p>
47  	 * If the throwable is already a {@link ParseException}, we simply return that exception as-is.
48  	 * If the throwable is an {@link InvocationTargetException}, we unwrap the thrown exception.
49  	 * Otherwise we create a new {@link ParseException}.
50  	 *
51  	 * @param e The exception being wrapped or unwrapped.
52  	 * @return A new {@link SerializeException}.
53  	 */
54  	public static ParseException create(Throwable e) {
55  		if (e instanceof InvocationTargetException)
56  			e = ((InvocationTargetException)e).getCause();
57  		if (e instanceof ParseException)
58  			return (ParseException)e;
59  		return new ParseException(e);
60  	}
61  
62  	private static String getMessage(ParserSession session, String msg, Object...args) {
63  		if (args.length != 0)
64  			msg = format(msg, args);
65  
66  		if (nn(session)) {
67  			Position p = session.getPosition();
68  			var sb = new StringBuilder(msg);
69  
70  			sb.append("\n\tAt: ").append(p);
71  
72  			JsonMap lastLocation = session.getLastLocation();
73  			if (nn(lastLocation)) {
74  				sb.append("\n\tWhile parsing into: ");
75  				lastLocation.forEach((k, v) -> sb.append("\n\t\t").append(k).append(": ").append(v));
76  			}
77  
78  			String lines = session.getInputAsString();
79  			if (lines == null)
80  				sb.append("\n\tUse BEAN_debug setting to display content.");
81  			else {
82  				int numLines = session.getDebugOutputLines();
83  				int start = p.line - numLines, end = p.line + numLines;
84  				sb.append("\n---start--\n").append(getNumberedLines(lines, start, end)).append("---end---");
85  			}
86  
87  			msg = sb.toString();
88  		}
89  		return msg;
90  	}

92  	/**
93  	 * Constructor.
94  	 *
95  	 * @param session The parser session.
96  	 * @param causedBy The inner exception.
97  	 */
98  	public ParseException(ParserSession session, Exception causedBy) {
99  		super(causedBy, getMessage(session, causedBy.getMessage()));
100 	}

103 	 * Constructor.

105 	 * @param session The parser session.
106 	 * @param message The exception message containing {@link MessageFormat}-style arguments.
107 	 * @param args Optional {@link MessageFormat}-style arguments.
108 	 */
109 	public ParseException(ParserSession session, String message, Object...args) {
110 		super(getMessage(session, message, args));
111 	}

113 	/**
114 	 * Constructor.

116 	 * @param session The parser session.

119 	 * @param args Optional {@link MessageFormat}-style arguments.
120 	 */
121 	public ParseException(ParserSession session, Throwable causedBy, String message, Object...args) {
122 		super(causedBy, getMessage(session, message, args));
123 	}
124

128 	 * @param message The {@link MessageFormat}-style message.
129 	 * @param args Optional {@link MessageFormat}-style arguments.

131 	public ParseException(String message, Object...args) {
132 		super(message, args);
133 	}
134 
135 	/**
136 	 * Constructor.
137 	 *
138 	 * @param causedBy The cause of this exception.
139 	 */
140 	public ParseException(Throwable causedBy) {
141 		super(causedBy);
142 	}
143 
144 	/**
145 	 * Constructor.
146 	 *
147 	 * @param causedBy The cause of this exception.
148 	 * @param message The {@link MessageFormat}-style message.
149 	 * @param args Optional {@link MessageFormat}-style arguments.
150 	 */
151 	public ParseException(Throwable causedBy, String message, Object...args) {
152 		super(causedBy, message, args);
153 	}
154 
155 	/**
156 	 * Returns the highest-level <c>ParseException</c> in the stack trace.
157 	 *
158 	 * <p>
159 	 * Useful for JUnit testing of error conditions.
160 	 *
161 	 * @return The root parse exception, or this exception if there isn't one.
162 	 */
163 	public ParseException getRootCause() {
164 		ParseException t = this;
165 		while (! (t.getCause() == null || ! (t.getCause() instanceof ParseException)))
166 			t = (ParseException)t.getCause();
167 		return t;
168 	}
169 
170 	@Override /* Overridden from BasicRuntimeException */
171 	public ParseException setMessage(String message, Object...args) {
172 		super.setMessage(message, args);
173 		return this;
174 	}
