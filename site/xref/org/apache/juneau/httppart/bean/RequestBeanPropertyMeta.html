32  /**
33   * Represents the metadata gathered from a getter method of a class annotated with {@link Request}.
34   *
35   * <h5 class='section'>See Also:</h5><ul>
36   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HttpPartSerializersParsers">HTTP Part Serializers and Parsers</a>
37   * </ul>
38   */
39  public class RequestBeanPropertyMeta {
40  
41  	static class Builder {
42  		HttpPartType partType;
43  		HttpPartSchema schema;
44  		Method getter;
45  
46  		RequestBeanPropertyMeta build(HttpPartSerializer serializer, HttpPartParser parser) {
47  			return new RequestBeanPropertyMeta(this, serializer, parser);
48  		}
49  
50  		Builder getter(Method value) {
51  			getter = value;
52  			return this;
53  		}
54  
55  		Builder partType(HttpPartType value) {
56  			partType = value;
57  			return this;
58  		}
59  
60  		Builder schema(HttpPartSchema value) {
61  			schema = value;
62  			return this;
63  		}
64  	}
65  
66  	static RequestBeanPropertyMeta.Builder create(HttpPartType partType, Class<? extends Annotation> c, MethodInfo m) {
67  		HttpPartSchema.Builder sb = HttpPartSchema.create().name(m.getPropertyName());
68  		AnnotationProvider ap = AnnotationProvider.INSTANCE;
69  		ap.findTopDown(m).map(x -> x.cast(Schema.class)).filter(Objects::nonNull).map(AnnotationInfo::inner).forEach(x -> sb.apply(x));
70  		ap.findTopDown(m).map(x -> x.cast(c)).filter(Objects::nonNull).map(AnnotationInfo::inner).forEach(x -> sb.apply(x));
71  		return new Builder().partType(partType).schema(sb.build()).getter(m.inner());
72  	}
73  
74  	private final Method getter;
75  	private final HttpPartType partType;
76  	private final Optional<HttpPartSerializer> serializer;
77  	private final HttpPartParser parser;
78  
79  	private final HttpPartSchema schema;
80  
81  	RequestBeanPropertyMeta(Builder b, HttpPartSerializer serializer, HttpPartParser parser) {
82  		this.partType = b.partType;
83  		this.schema = b.schema;
84  		this.getter = b.getter;
85  		this.serializer = opt(schema.getSerializer() == null ? serializer : BeanCreator.of(HttpPartSerializer.class).type(schema.getSerializer()).run());
86  		this.parser = schema.getParser() == null ? parser : BeanCreator.of(HttpPartParser.class).type(schema.getParser()).run();
87  	}
88  
89  	/**
90  	 * Returns the name of the Java method getter that defines this property.
91  	 *
92  	 * @return
93  	 * 	The name of the Java method getter that defines this property.
94  	 * 	<br>Never <jk>null</jk>.
95  	 */
96  	public Method getGetter() { return getter; }
97  
98  	/**
99  	 * Returns the parser to use for parsing the bean property value.
100 	 *
101 	 * @param _default The default parsing to use if not defined on the annotation.
102 	 * @return The parsing to use for serializing the bean property value.
103 	 */
104 	public HttpPartParserSession getParser(HttpPartParserSession _default) {
105 		return parser == null ? _default : parser.getPartSession();
106 	}
107 
108 	/**
109 	 * Returns the HTTP part name for this property (query parameter name for example).
110 	 *
111 	 * @return The HTTP part name, or <jk>null</jk> if it doesn't have a part name.
112 	 */
113 	public String getPartName() { return schema == null ? null : schema.getName(); }
114 
115 	/**
116 	 * Returns the HTTP part type for this property (query parameter for example).
117 	 *
118 	 * @return
119 	 * 	The HTTP part type for this property.
120 	 * 	<br>Never <jk>null</jk>.
121 	 */
122 	public HttpPartType getPartType() { return partType; }
123 
124 	/**
125 	 * Returns the schema information gathered from annotations on the method and return type.
126 	 *
127 	 * @return
128 	 * 	The schema information gathered from annotations on the method and return type.
129 	 * 	<br>Never <jk>null</jk>.
130 	 */
131 	public HttpPartSchema getSchema() { return schema; }
132 
133 	/**
134 	 * Returns the serializer to use for serializing the bean property value.
135 	 *
136 	 * @return The serializer to use for serializing the bean property value.
137 	 */
138 	public Optional<HttpPartSerializer> getSerializer() { return serializer; }
139 }
