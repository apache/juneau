36  /**
37   * Represents the metadata gathered from a parameter or class annotated with {@link Response}.
38   *
39   * <h5 class='section'>See Also:</h5><ul>
40   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HttpPartSerializersParsers">HTTP Part Serializers and Parsers</a>
41   * </ul>
42   */
43  public class ResponseBeanMeta {
44  
45  	static class Builder {
46  		ClassMeta<?> cm;
47  		int code;
48  		AnnotationWorkList annotations;
49  		Class<? extends HttpPartSerializer> partSerializer;
50  		Class<? extends HttpPartParser> partParser;
51  		HttpPartSchema.Builder schema = HttpPartSchema.create();
52  
53  		Map<String,ResponseBeanPropertyMeta.Builder> headerMethods = CollectionUtils.map();
54  		ResponseBeanPropertyMeta.Builder contentMethod;
55  		ResponseBeanPropertyMeta.Builder statusMethod;
56  
57  		Builder(AnnotationWorkList annotations) {
58  			this.annotations = annotations;
59  		}
60  
61  		Builder apply(Response a) {
62  			if (nn(a)) {
63  				if (isNotVoid(a.serializer()))
64  					partSerializer = a.serializer();
65  				if (isNotVoid(a.parser()))
66  					partParser = a.parser();
67  				schema.apply(a.schema());
68  			}
69  			return this;
70  		}
71  
72  		Builder apply(StatusCode a) {
73  			if (nn(a)) {
74  				if (a.value().length > 0)
75  					code = a.value()[0];
76  			}
77  			return this;
78  		}
79  
80  		Builder apply(Type t) {
81  			Class<?> c = toClass(t);
82  			this.cm = BeanContext.DEFAULT.getClassMeta(c);
83  			ClassInfo ci = cm.getInfo();
84  			ci.forEachPublicMethod(x -> true, x -> {
85  				assertNoInvalidAnnotations(x, Query.class, FormData.class);
86  				if (x.hasAnnotation(Header.class)) {
87  					assertNoArgs(x, Header.class);
88  					assertReturnNotVoid(x, Header.class);
89  					HttpPartSchema s = HttpPartSchema.create(x.getAnnotation(Header.class), x.getPropertyName());
90  					headerMethods.put(s.getName(), ResponseBeanPropertyMeta.create(RESPONSE_HEADER, s, x));
91  				} else if (x.hasAnnotation(StatusCode.class)) {
92  					assertNoArgs(x, Header.class);
93  					assertReturnType(x, Header.class, int.class, Integer.class);
94  					statusMethod = ResponseBeanPropertyMeta.create(RESPONSE_STATUS, x);
95  				} else if (x.hasAnnotation(Content.class)) {
96  					if (x.hasNoParams())
97  						assertReturnNotVoid(x, Header.class);
98  					else
99  						assertArgType(x, Header.class, OutputStream.class, Writer.class);
100 					contentMethod = ResponseBeanPropertyMeta.create(RESPONSE_BODY, x);
101 				}
102 			});
103 			return this;
104 		}
105 
106 		ResponseBeanMeta build() {
107 			return new ResponseBeanMeta(this);
108 		}
109 	}

116 	/**
117 	 * Create metadata from specified method return.
118 	 *
119 	 * @param m The method annotated with {@link Response}.
120 	 * @param annotations The annotations to apply to any new part serializers or parsers.
121 	 * @return Metadata about the class, or <jk>null</jk> if class not annotated with {@link Response}.
122 	 */
123 	public static ResponseBeanMeta create(MethodInfo m, AnnotationWorkList annotations) {
124 		if (! (m.hasAnnotation(Response.class) || m.getReturnType().unwrap(Value.class, Optional.class).hasAnnotation(Response.class)))
125 			return null;
126 		Builder b = new Builder(annotations);
127 		b.apply(m.getReturnType().unwrap(Value.class, Optional.class).innerType());
128 		m.forEachAnnotation(Response.class, x -> true, x -> b.apply(x));
129 		m.forEachAnnotation(StatusCode.class, x -> true, x -> b.apply(x));
130 		return b.build();
131 	}
132 
133 	/**
134 	 * Create metadata from specified method parameter.
135 	 *
136 	 * @param mpi The method parameter.
137 	 * @param annotations The annotations to apply to any new part serializers or parsers.
138 	 * @return Metadata about the class, or <jk>null</jk> if class not annotated with {@link Response}.
139 	 */
140 	public static ResponseBeanMeta create(ParamInfo mpi, AnnotationWorkList annotations) {
141 		if (mpi.hasNoAnnotation(Response.class))
142 			return null;
143 		Builder b = new Builder(annotations);
144 		b.apply(mpi.getParameterType().unwrap(Value.class, Optional.class).innerType());
145 		mpi.forEachAnnotation(Response.class, x -> true, x -> b.apply(x));
146 		mpi.forEachAnnotation(StatusCode.class, x -> true, x -> b.apply(x));
147 		return b.build();
148 	}
149 
150 	/**
151 	 * Create metadata from specified class.
152 	 *
153 	 * @param t The class annotated with {@link Response}.
154 	 * @param annotations The annotations to apply to any new part serializers or parsers.
155 	 * @return Metadata about the class, or <jk>null</jk> if class not annotated with {@link Response}.
156 	 */
157 	public static ResponseBeanMeta create(Type t, AnnotationWorkList annotations) {
158 		ClassInfo ci = ClassInfo.of(t).unwrap(Value.class, Optional.class);
159 		if (ci.hasNoAnnotation(Response.class))
160 			return null;
161 		Builder b = new Builder(annotations);
162 		b.apply(ci.innerType());
163 		ci.forEachAnnotation(Response.class, x -> true, x -> b.apply(x));
164 		ci.forEachAnnotation(StatusCode.class, x -> true, x -> b.apply(x));
165 		return b.build();
166 	}

168 	private final ClassMeta<?> cm;
169 	private final Map<String,ResponseBeanPropertyMeta> properties;
170 	private final int code;
171 	private final Map<String,ResponseBeanPropertyMeta> headerMethods;
172 	private final ResponseBeanPropertyMeta statusMethod, contentMethod;
173 	private final Optional<HttpPartSerializer> partSerializer;
174 	private final Optional<HttpPartParser> partParser;
175 
176 	private final HttpPartSchema schema;
177 
178 	ResponseBeanMeta(Builder b) {
179 		this.cm = b.cm;
180 		this.code = b.code;
181 		this.partSerializer = opt(b.partSerializer).map(x -> HttpPartSerializer.creator().type(x).apply(b.annotations).create());
182 		this.partParser = opt(b.partParser).map(x -> HttpPartParser.creator().type(x).apply(b.annotations).create());
183 		this.schema = b.schema.build();
184 
185 		Map<String,ResponseBeanPropertyMeta> properties = CollectionUtils.map();
186 
187 		Map<String,ResponseBeanPropertyMeta> hm = CollectionUtils.map();
188 		b.headerMethods.forEach((k, v) -> {
189 			ResponseBeanPropertyMeta pm = v.build(partSerializer, partParser);
190 			hm.put(k, pm);
191 			properties.put(pm.getGetter().getName(), pm);
192 		});
193 		this.headerMethods = u(hm);
194 
195 		this.contentMethod = b.contentMethod == null ? null : b.contentMethod.schema(schema).build(partSerializer, partParser);
196 		this.statusMethod = b.statusMethod == null ? null : b.statusMethod.build(opte(), opte());
197 
198 		if (nn(contentMethod))
199 			properties.put(contentMethod.getGetter().getName(), contentMethod);
200 		if (nn(statusMethod))
201 			properties.put(statusMethod.getGetter().getName(), statusMethod);
202 
203 		this.properties = u(properties);
204 	}
