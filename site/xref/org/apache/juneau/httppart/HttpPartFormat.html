13  package org.apache.juneau.httppart;
14  
15  /**
16   * Valid values for the <c>format</c> field.
17   *
18   * <h5 class='section'>See Also:</h5><ul>
19   * 	<li class='link'><a class="doclink" href="https://juneau.apache.org/docs/topics/HttpPartSerializersParsers">HTTP Part Serializers and Parsers</a>
20   * </ul>
21   */
22  public enum HttpPartFormat {
23  
24  	/**
25  	 * Signed 32 bits.
26  	 */
27  	INT32,
28  
29  	/**
30  	 * Signed 64 bits.
31  	 */
32  	INT64,
33  
34  	/**
35  	 * 32-bit floating point number.
36  	 */
37  	FLOAT,
38  
39  	/**
40  	 * 64-bit floating point number.
41  	 */
42  	DOUBLE,
43  
44  	/**
45  	 * BASE-64 encoded characters.
46  	 */
47  	BYTE,
48  
49  	/**
50  	 * Hexadecimal encoded octets (e.g. <js>"00FF"</js>).
51  	 */
52  	BINARY,
53  
54  	/**
55  	 * Spaced-separated hexadecimal encoded octets (e.g. <js>"00 FF"</js>).
56  	 */
57  	BINARY_SPACED,
58  
59  	/**
60  	 * An <a href='http://xml2rfc.ietf.org/public/rfc/html/rfc3339.html#anchor14'>RFC3339 full-date</a>.
61  	 */
62  	DATE,
63  
64  	/**
65  	 *  An <a href='http://xml2rfc.ietf.org/public/rfc/html/rfc3339.html#anchor14'>RFC3339 date-time</a>.
66  	 */
67  	DATE_TIME,
68  
69  	/**
70  	 * Used to hint UIs the input needs to be obscured.
71  	 */
72  	PASSWORD,
73  
74  	/**
75  	 * UON notation (e.g. <js>"(foo=bar,baz=@(qux,123))"</js>).
76  	 */
77  	UON,
78  
79  	/**
80  	 * Not specified.
81  	 */
82  	NO_FORMAT;
83  
84  	/**
85  	 * Create from lowercase dashed name.
86  	 *
87  	 * @param value The enum name.
88  	 * @return The enum.
89  	 */
90  	public static HttpPartFormat fromString(String value) {
91  		value = value.toUpperCase().replace('-','_');
92  		return valueOf(value);
93  	}
94  
95  	/**
96  	 * Returns <jk>true</jk> if this format is in the provided list.
97  	 *
98  	 * @param list The list of formats to check against.
99  	 * @return <jk>true</jk> if this format is in the provided list.
100 	 */
101 	public boolean isOneOf(HttpPartFormat...list) {
102 		for (HttpPartFormat ff : list)
103 			if (this == ff)
104 				return true;
105 		return false;
106 	}
107 
108 	@Override /* Object */
109 	public String toString() {
110 		String s = name().toLowerCase().replace('_','-');
111 		return s;
112 	}
113 }
