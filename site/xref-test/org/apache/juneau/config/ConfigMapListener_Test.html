541 	public static class LatchedListener implements ConfigEventListener {
542 		private final CountDownLatch latch;
543 		private volatile String error = null;
544 		public LatchedListener(CountDownLatch latch) {
545 			this.latch = latch;
546 		}
547 
548 		@Override
549 		public void onConfigChange(ConfigEvents events) {
550 			try {
551 				check(events);
552 			} catch (Exception e) {
553 				error = e.getLocalizedMessage();
554 			}
555 			for (var i = 0; i < events.size(); i++)
556 				latch.countDown();
557 		}
558 
559 		public void check(ConfigEvents events) throws Exception {}  // NOSONAR
560 	}
561 
562 	private static void wait(CountDownLatch latch) throws InterruptedException {
563 		if (! latch.await(10, TimeUnit.SECONDS))
564 			fail("Latch failed.");
565 	}
