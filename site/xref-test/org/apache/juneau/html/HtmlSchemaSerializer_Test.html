27  class HtmlSchemaSerializer_Test extends TestBase {
28  
29  	//====================================================================================================
30  	// Simple objects
31  	//====================================================================================================
32  	@Test void simpleObjects() throws Exception {
33  		var s = HtmlSchemaSerializer.DEFAULT_SIMPLE;
34  
35  		assertEquals("<table><tr><td>type</td><td>integer</td></tr><tr><td>format</td><td>int16</td></tr></table>", s.serialize((short)1));
36  		assertEquals("<table><tr><td>type</td><td>integer</td></tr><tr><td>format</td><td>int32</td></tr></table>", s.serialize(1));
37  		assertEquals("<table><tr><td>type</td><td>integer</td></tr><tr><td>format</td><td>int64</td></tr></table>", s.serialize(1L));
38  		assertEquals("<table><tr><td>type</td><td>number</td></tr><tr><td>format</td><td>float</td></tr></table>", s.serialize(1f));
39  		assertEquals("<table><tr><td>type</td><td>number</td></tr><tr><td>format</td><td>double</td></tr></table>", s.serialize(1d));
40  		assertEquals("<table><tr><td>type</td><td>boolean</td></tr></table>", s.serialize(true));
41  		assertEquals("<table><tr><td>type</td><td>string</td></tr></table>", s.serialize("foo"));
42  		assertEquals("<table><tr><td>type</td><td>string</td></tr></table>", s.serialize(new StringBuilder("foo")));
43  		assertEquals("<table><tr><td>type</td><td>string</td></tr></table>", s.serialize('c'));
44  		assertEquals("<table><tr><td>type</td><td>string</td></tr><tr><td>enum</td><td><ul><li>one</li><li>two</li><li>three</li></ul></td></tr></table>", s.serialize(TestEnumToString.ONE));
45  		assertEquals("<table><tr><td>type</td><td>object</td></tr><tr><td>properties</td><td><table><tr><td>f1</td><td><table><tr><td>type</td><td>string</td></tr></table></td></tr></table></td></tr></table>", s.serialize(new SimpleBean()));
46  	}
47  
48  	public static class SimpleBean {
49  		public String f1;
50  	}
