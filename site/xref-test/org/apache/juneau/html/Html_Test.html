204 	private static String strip(String html) {
205 		return html
206 			.replace("<table><tr><th>key</th><th>value</th></tr>", "")
207 			.replace("</table>", "")
208 			.replace("<tr><td>", "\n[")
209 			.replace("</td><td>", "]=")
210 			.replace("</td></tr>", "");
211 	}
