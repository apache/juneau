28  	@Test void a01_constructors() {
29  		var x = new ControlledArrayList<>(false);
30  		assertTrue(x.isModifiable());
31  
32  		x = new ControlledArrayList<>(true);
33  		assertFalse(x.isModifiable());
34  
35  		x = new ControlledArrayList<>(false, Arrays.asList(1));
36  		assertTrue(x.isModifiable());
37  
38  		x = new ControlledArrayList<>(true, Arrays.asList(1));
39  		assertFalse(x.isModifiable());
40  	}

42  	@Test void a02_basicMethods() {
43  		var x1 = new ControlledArrayList<>(false, Arrays.asList(1));
44  		var x2 = new ControlledArrayList<>(true, Arrays.asList(1));
45  
46  		x1.set(0, 2);
47  		assertThrows(UnsupportedOperationException.class, () -> x2.set(0, 2));
48  		x2.overrideSet(0, 2);
49  		assertEquals(x2, x1);
50  
51  		x1.add(0, 2);
52  		assertThrows(UnsupportedOperationException.class, () -> x2.add(0, 2));
53  		x2.overrideAdd(0, 2);
54  		assertEquals(x2, x1);
55  
56  		x1.remove(0);
57  		assertThrows(UnsupportedOperationException.class, () -> x2.remove(0));
58  		x2.overrideRemove(0);
59  		assertEquals(x2, x1);
60  
61  		x1.addAll(0, Arrays.asList(3));
62  		assertThrows(UnsupportedOperationException.class, () -> x2.addAll(0, Arrays.asList(3)));
63  		x2.overrideAddAll(0, Arrays.asList(3));
64  		assertEquals(x2, x1);
65  
66  		x1.replaceAll(x -> x);
67  		assertThrows(UnsupportedOperationException.class, () -> x2.replaceAll(x -> x));
68  		x2.overrideReplaceAll(x -> x);
69  		assertEquals(x2, x1);
70  
71  		x1.sort(null);
72  		assertThrows(UnsupportedOperationException.class, () -> x2.sort(null));
73  		x2.overrideSort(null);
74  		assertEquals(x2, x1);
75  
76  		x1.add(1);
77  		assertThrows(UnsupportedOperationException.class, () -> x2.add(1));
78  		x2.overrideAdd(1);
79  		assertEquals(x2, x1);
80  
81  		x1.remove((Integer)1);
82  		assertThrows(UnsupportedOperationException.class, () -> x2.remove((Integer)1));
83  		x2.overrideRemove((Integer)1);
84  		assertEquals(x2, x1);
85  
86  		x1.addAll(Arrays.asList(3));
87  		assertThrows(UnsupportedOperationException.class, () -> x2.addAll(Arrays.asList(3)));
88  		x2.overrideAddAll(Arrays.asList(3));
89  		assertEquals(x2, x1);
90  
91  		x1.removeAll(Arrays.asList(3));
92  		assertThrows(UnsupportedOperationException.class, () -> x2.removeAll(Arrays.asList(3)));
93  		x2.overrideRemoveAll(Arrays.asList(3));
94  		assertEquals(x2, x1);
95  
96  		x1.retainAll(Arrays.asList(2));
97  		assertThrows(UnsupportedOperationException.class, () -> x2.retainAll(Arrays.asList(2)));
98  		x2.overrideRetainAll(Arrays.asList(2));
99  		assertEquals(x2, x1);
100 
101 		x1.clear();
102 		assertThrows(UnsupportedOperationException.class, x2::clear);
103 		x2.overrideClear();
104 		assertEquals(x2, x1);
105 
106 		x1.add(1);
107 		x2.overrideAdd(1);
108 
109 		x1.removeIf(x -> x == 1);
110 		assertThrows(UnsupportedOperationException.class, () -> x2.removeIf(x -> x == 1));
111 		x2.overrideRemoveIf(x -> x == 1);
112 		assertEquals(x2, x1);
113 
114 		x1.add(1);
115 		x2.overrideAdd(1);
116 
117 		var x1a = (ControlledArrayList<Integer>) x1.subList(0, 0);
118 		var x2a = (ControlledArrayList<Integer>) x2.subList(0, 0);
119 		assertTrue(x1a.isModifiable());
120 		assertFalse(x2a.isModifiable());
121 	}

123 	@Test void a03_iterator() {
124 		var x1 = new ControlledArrayList<>(false, Arrays.asList(1));
125 		var x2 = new ControlledArrayList<>(true, Arrays.asList(1));
126 
127 		var i1 = x1.iterator();
128 		var i2 = x2.iterator();
129 
130 		assertTrue(i1.hasNext());
131 		assertTrue(i2.hasNext());
132 
133 		assertEquals(1, i1.next().intValue());
134 		assertEquals(1, i2.next().intValue());
135 
136 		i1.remove();
137 		assertThrows(UnsupportedOperationException.class, i2::remove);
138 
139 		i1.forEachRemaining(x -> {});
140 		i2.forEachRemaining(x -> {});
141 	}

143 	@Test void a04_listIterator() {
144 		var x1 = new ControlledArrayList<>(false, Arrays.asList(1));
145 		var x2 = new ControlledArrayList<>(true, Arrays.asList(1));
146 
147 		var i1a = x1.listIterator();
148 		var i2a = x2.listIterator();
149

152

154 		assertEquals(1, i2a.next().intValue());
155 
156 		assertTrue(i1a.hasPrevious());
157 		assertTrue(i2a.hasPrevious());
158 
159 		assertEquals(1, i1a.nextIndex());

161 
162 		assertEquals(0, i1a.previousIndex());
163 		assertEquals(0, i2a.previousIndex());
164 
165 		i1a.previous();
166 		i2a.previous();
167 
168 		i1a.set(1);
169 		assertThrows(UnsupportedOperationException.class, () -> i2a.set(1));
170 
171 		i1a.add(1);
172 		assertThrows(UnsupportedOperationException.class, () -> i2a.add(1));
173 
174 		i1a.next();
175 		i2a.next();
176 
177 		i1a.remove();
178 		assertThrows(UnsupportedOperationException.class, i2a::remove);
179 
180 		i1a.forEachRemaining(x -> {});
181 		i2a.forEachRemaining(x -> {});
182 	}
