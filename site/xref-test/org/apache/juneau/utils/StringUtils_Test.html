149 	@Test void a02_numberRanges() {
150 		// An integer range is -2,147,483,648 to 2,147,483,647
151 
152 		assertFalse(isNumeric(null));
153 		assertFalse(isNumeric(""));
154 		assertFalse(isNumeric("x"));
155 
156 		var s = "-2147483648";
157 		assertTrue(isNumeric(s));
158 		assertTrue(parseNumber(s, null) instanceof Integer);
159 		assertEquals(-2147483648, parseNumber(s, null));
160 
161 		s = "2147483647";
162 		assertTrue(isNumeric(s));
163 		assertTrue(parseNumber(s, null) instanceof Integer);
164 		assertEquals(2147483647, parseNumber(s, null));
165 
166 		s = "-2147483649";
167 		assertTrue(isNumeric(s));
168 		assertTrue(parseNumber(s, null) instanceof Long);
169 		assertEquals(-2147483649L, parseNumber(s, null));
170 
171 		s = "2147483648";
172 		assertTrue(isNumeric(s));
173 		assertTrue(parseNumber(s, null) instanceof Long);
174 		assertEquals(2147483648L, parseNumber(s, null));
175 
176 		// An long range is -9,223,372,036,854,775,808 to +9,223,372,036,854,775,807
177 
178 		s = "-9223372036854775808";
179 		assertTrue(isNumeric(s));
180 		assertTrue(parseNumber(s, null) instanceof Long);
181 		assertEquals(-9223372036854775808L, parseNumber(s, null));
182 
183 		s = "9223372036854775807";
184 		assertTrue(isNumeric(s));
185 		assertTrue(parseNumber(s, null) instanceof Long);
186 		assertEquals(9223372036854775807L, parseNumber(s, null));
187 
188 		// Anything that falls outside this range should be a double.
189 
190 		s = "-9223372036854775809";
191 		assertTrue(isNumeric(s));
192 		assertTrue(parseNumber(s, null) instanceof Double);
193 		assertEquals(-9223372036854775808L, parseNumber(s, null).longValue());
194 		assertEquals(-9.223372036854776E18, parseNumber(s, null));
195 
196 		s = "9223372036854775808";
197 		assertTrue(isNumeric(s));
198 		assertTrue(parseNumber(s, null) instanceof Double);
199 		assertEquals(9223372036854775807L, parseNumber(s, null).longValue());
200 		assertEquals(9.223372036854776E18, parseNumber(s, null));
201 
202 		// Check case where string is longer than 20 characters since it's a different code path.
203 
204 		s = "-123456789012345678901";
205 		assertTrue(isNumeric(s));
206 		assertTrue(parseNumber(s, null) instanceof Double);
207 		assertEquals(-9223372036854775808L, parseNumber(s, null).longValue());
208 		assertEquals(-1.2345678901234568E20, parseNumber(s, null));
209 
210 		s = "123456789012345678901";
211 		assertTrue(isNumeric(s));
212 		assertTrue(parseNumber(s, null) instanceof Double);
213 		assertEquals(9223372036854775807L, parseNumber(s, null).longValue());
214 		assertEquals(1.2345678901234568E20, parseNumber(s, null));
215 
216 		// Autodetected floating point numbers smaller than Float.MAX_VALUE
217 		s = String.valueOf(Float.MAX_VALUE / 2);
218 		assertTrue(isNumeric(s));
219 		assertTrue(parseNumber(s, null) instanceof Float);
220 		assertEquals(1.7014117E38f, parseNumber(s, null));
221 
222 		s = String.valueOf((-Float.MAX_VALUE) / 2);
223 		assertTrue(isNumeric(s));
224 		assertTrue(parseNumber(s, null) instanceof Float);
225 		assertEquals(-1.7014117E38f, parseNumber(s, null));
226 
227 		// Autodetected floating point numbers larger than Float.MAX_VALUE
228 		s = String.valueOf((double)Float.MAX_VALUE * 2);
229 		assertTrue(isNumeric(s));
230 		assertTrue(parseNumber(s, null) instanceof Double);
231 		assertEquals("6.805646932770577E38", parseNumber(s, null).toString());
232 
233 		s = String.valueOf((double)Float.MAX_VALUE * -2);
234 		assertTrue(isNumeric(s));
235 		assertTrue(parseNumber(s, null) instanceof Double);
236 		assertEquals("-6.805646932770577E38", parseNumber(s, null).toString());
237 
238 		s = String.valueOf("214748364x");
239 		assertFalse(isNumeric(s));
240 		assertThrows(NumberFormatException.class, ()->parseNumber("214748364x", Number.class));
241 
242 		s = String.valueOf("2147483640x");
243 		assertFalse(isNumeric(s));
244 		assertThrows(NumberFormatException.class, ()->parseNumber("2147483640x", Long.class));
245 	}

267 		assertEquals("xEFEx", replaceVars("x{d}x", m));
268 		assertEquals("xEFExEFEx", replaceVars("x{d}x{d}x", m));
