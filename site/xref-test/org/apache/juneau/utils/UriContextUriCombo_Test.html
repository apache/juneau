182 	private static class Data {
183 		final String label;
184 		final Input in;
185 		final Results r;
186 
187 		public Data(String label, Input in, Results r) {
188 			this.label = label;
189 			this.in = in;
190 			this.r = r;
191 		}
192 	}
193 
194 	private static class Input {
195 		final UriContext uriContext;
196 
197 		public Input(String authority, String context, String resource, String path) {
198 			this.uriContext = UriContext.of(authority, context, resource, path);
199 		}
200 	}
201 
202 	private static class Results {
203 		final String eAbsoluteAuthority, eAbsoluteContext, eAbsoluteResource, eAbsolutePath, eRootRelativeContext, eRootRelativeResource, eRootRelativePath;
204 
205 		public Results(String eAbsoluteAuthority, String eAbsoluteContext, String eAbsoluteResource, String eAbsolutePath,
206 					String eRootRelativeContext, String eRootRelativeResource, String eRootRelativePath) {
207 			this.eAbsoluteAuthority = eAbsoluteAuthority;
208 			this.eAbsoluteContext = eAbsoluteContext;
209 			this.eAbsoluteResource = eAbsoluteResource;
210 			this.eAbsolutePath = eAbsolutePath;
211 			this.eRootRelativeContext = eRootRelativeContext;
212 			this.eRootRelativeResource = eRootRelativeResource;
213 			this.eRootRelativePath = eRootRelativePath;
214 		}
215 	}
216 
217 	public static Data data(String label, Input input, Results results) {
218 		return new Data(label, input, results);
219 	}
220 
221 	public static Input input(String authority, String context, String resource, String path) {
222 		return new Input(authority, context, resource, path);
223 	}
224 
225 	public static Results results(String eAbsoluteAuthority, String eAbsoluteContext, String eAbsoluteResource, String eAbsolutePath,
226 			String eRootRelativeContext, String eRootRelativeResource, String eRootRelativePath) {
227 		return new Results(eAbsoluteAuthority, eAbsoluteContext, eAbsoluteResource, eAbsolutePath, eRootRelativeContext, eRootRelativeResource, eRootRelativePath);
228 	}
