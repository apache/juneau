28  	@Test void a06_test() {
29  		var vr = VarResolver.create().vars(XVar.class).build();
30  
31  		var t = (String)null;
32  		assertEquals(null, vr.resolve(t));
33  
34  		t = "";
35  		assertEquals("", vr.resolve(t));
36  
37  		t = "foo";
38  		assertEquals("foo", vr.resolve(t));
39  
40  		t = "$X{y}";
41  		assertEquals("xyx", vr.resolve(t));
42  
43  		t = "$X{y}x";
44  		assertEquals("xyxx", vr.resolve(t));
45  
46  		t = "x$X{y}";
47  		assertEquals("xxyx", vr.resolve(t));
48  
49  		t = "$X{y}$X{y}";
50  		assertEquals("xyxxyx", vr.resolve(t));
51  
52  		t = "z$X{y}z$X{y}z";
53  		assertEquals("zxyxzxyxz", vr.resolve(t));
54  
55  		t = "$X{$X{y}}";
56  		assertEquals("xxyxx", vr.resolve(t));
57  
58  		t = "$X{z$X{y}z}";
59  		assertEquals("xzxyxzx", vr.resolve(t));
60  
61  		t = "$X.{y}";
62  		assertEquals("$X.{y}", vr.resolve(t));
63  
64  		t = "z$X.{y}z";
65  		assertEquals("z$X.{y}z", vr.resolve(t));
66  
67  		t = "z$X.{$X.{z}}z";
68  		assertEquals("z$X.{$X.{z}}z", vr.resolve(t));
69  
70  		// Non-existent vars
71  		t = "$Y{y}";
72  		assertEquals("$Y{y}", vr.resolve(t));
73  
74  		t = "$Y{y}x";
75  		assertEquals("$Y{y}x", vr.resolve(t));
76  
77  		t = "x$Y{y}";
78  		assertEquals("x$Y{y}", vr.resolve(t));
79  	}

139 		var vr = VarResolver.create().vars(XVar.class).build();
140 		var b = '\\';
141 		var t = "A|A".replace('|',b);
142 
143 		assertEquals("A|A".replace('|',b), vr.resolve(t));
144 		t = "A||A".replace('|',b);
145 		assertEquals("A|A".replace('|',b), vr.resolve(t));
146 		t = "A|A$X{B}".replace('|',b);
147 		assertEquals("A|AxBx".replace('|',b), vr.resolve(t));
148 		t = "A||A$X{B}".replace('|',b);
149 		assertEquals("A|AxBx".replace('|',b), vr.resolve(t));
150 		t = "A|$X{B}".replace('|',b);
151 		assertEquals("A$X{B}".replace('|',b), vr.resolve(t));
152 		t = "A||$X{B}".replace('|',b);
153 		assertEquals("A|xBx".replace('|',b), vr.resolve(t));
154 		t = "A$X|{B}".replace('|',b);
155 		assertEquals("A$X{B}".replace('|',b), vr.resolve(t));
156 		t = "A$X{B|}".replace('|',b);
157 		assertEquals("A$X{B}".replace('|',b), vr.resolve(t));
158 		t = "A$X{B}|".replace('|',b);
159 		assertEquals("AxBx|".replace('|',b), vr.resolve(t));
160 	}

197 	//====================================================================================================
198 	@Test void a05_falseTriggers() {
199 		var vrb = VarResolver.create().defaultVars();
200 
201 		// Should reject names with characters outside A-Za-z
202 		for (Class<? extends Var> c : a(InvalidVar1.class, InvalidVar2.class, InvalidVar3.class, InvalidVar4.class, InvalidVar5.class)) {
203 			assertThrows(Exception.class, ()->VarResolver.create().vars(c).build());
204 		}
205 
206 		var vr = vrb.build();
207 
208 		// These should all be unchanged.
209 		var in = "$@{foobar}";
210 		assertEquals(in, vr.resolve(in));
211 		in = "$[{foobar}";
212 		assertEquals(in, vr.resolve(in));
213 		in = "$`{foobar}";
214 		assertEquals(in, vr.resolve(in));
215 		in = "$|{foobar}";
216 		assertEquals(in, vr.resolve(in));
217 		in = "${{foobar}";
218 		assertEquals(in, vr.resolve(in));
219 		in = "${$foobar}";
220 		assertEquals(in, vr.resolve(in));
221 
222 		System.setProperty("foobar", "baz");
223 
224 		in = "$";
225 		assertEquals(in, vr.resolve(in));
226 
227 		in = "$S";
228 		assertEquals(in, vr.resolve(in));
229 
230 		in = "$S{";
231 		assertEquals(in, vr.resolve(in));
232 
233 		in = "$S{foobar";
234 
235 		assertEquals(in, vr.resolve(in));
236 		in = "$S{foobar}$";
237 		assertEquals("baz$", vr.resolve(in));
238 
239 		in = "$S{foobar}$S";
240 		assertEquals("baz$S", vr.resolve(in));
241 
242 		in = "$S{foobar}$S{";
243 		assertEquals("baz$S{", vr.resolve(in));
244 
245 		in = "$S{foobar}$S{foobar";
246 		assertEquals("baz$S{foobar", vr.resolve(in));
247 
248 		System.clearProperty("foobar");
249 
250 		// Test nulls returned by StringVar.
251 		// Should be converted to blanks.
252 		vrb.vars(AlwaysNullVar.class);
253 
254 		vr = vrb.build();
255 
256 		in = "$A{xxx}";
257 		assertEquals("", vr.resolve(in));
258 		in = "x$A{xxx}";
259 		assertEquals("x", vr.resolve(in));
260 		in = "$A{xxx}x";
261 		assertEquals("x", vr.resolve(in));
262 	}
