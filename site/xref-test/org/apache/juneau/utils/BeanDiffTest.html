13  package org.apache.juneau.utils;
14  
15  import static org.apache.juneau.common.internal.Utils.*;
16  import static org.junit.jupiter.api.Assertions.*;
17  
18  import org.apache.juneau.*;
19  import org.junit.jupiter.api.*;
20  
21  class BeanDiffTest extends SimpleTestBase {
22  
23  	public static class A {
24  		public int f1;
25  		public String f2;
26  
27  		static A create(int f1, String f2) {
28  			var a = new A();
29  			a.f1 = f1;
30  			a.f2 = f2;
31  			return a;
32  		}
33  	}
34  
35  	@Test void a01_same() {
36  		var bd = BeanDiff.create(A.create(1, "a"), A.create(1, "a")).build();
37  		assertFalse(bd.hasDiffs());
38  		assertEquals("{v1:{},v2:{}}", bd.toString());
39  	}
40  
41  	@Test void a02_different() {
42  		var bd = BeanDiff.create(A.create(1, "a"), A.create(2, "b")).build();
43  		assertTrue(bd.hasDiffs());
44  		assertEquals("{v1:{f1:1,f2:'a'},v2:{f1:2,f2:'b'}}", bd.toString());
45  	}
46  
47  	@Test void a03_firstNull() {
48  		var bd = BeanDiff.create(null, A.create(2, "b")).build();
49  		assertTrue(bd.hasDiffs());
50  		assertEquals("{v1:{},v2:{f1:2,f2:'b'}}", bd.toString());
51  	}
52  
53  	@Test void a04_secondNull() {
54  		var bd = BeanDiff.create(A.create(1, "a"), null).build();
55  		assertTrue(bd.hasDiffs());
56  		assertEquals("{v1:{f1:1,f2:'a'},v2:{}}", bd.toString());
57  	}
58  
59  	@Test void a05_bothNull() {
60  		var bd = BeanDiff.create(null, null).build();
61  		assertFalse(bd.hasDiffs());
62  		assertEquals("{v1:{},v2:{}}", bd.toString());
63  	}
64  
65  	@Test void a06_nullFields() {
66  		var bd = BeanDiff.create(A.create(1, null), A.create(2, "b")).build();
67  		assertTrue(bd.hasDiffs());
68  		assertEquals("{v1:{f1:1},v2:{f1:2,f2:'b'}}", bd.toString());
69  	}
70  
71  	@Test void a07_includes() {
72  		var bd = BeanDiff.create(A.create(1, null), A.create(2, "b")).include("f1").build();
73  		assertTrue(bd.hasDiffs());
74  		assertEquals("{v1:{f1:1},v2:{f1:2}}", bd.toString());
75  	}
76  
77  	@Test void a08_includesSet() {
78  		var bd = BeanDiff.create(A.create(1, null), A.create(2, "b")).include(set("f1")).build();
79  		assertTrue(bd.hasDiffs());
80  		assertEquals("{v1:{f1:1},v2:{f1:2}}", bd.toString());
81  	}
82  
83  	@Test void a09_excludes() {
84  		var bd = BeanDiff.create(A.create(1, null), A.create(2, "b")).exclude("f2").build();
85  		assertTrue(bd.hasDiffs());
86  		assertEquals("{v1:{f1:1},v2:{f1:2}}", bd.toString());
87  	}
88  
89  	@Test void a10_excludesSet() {
90  		var bd = BeanDiff.create(A.create(1, null), A.create(2, "b")).exclude(set("f2")).build();
91  		assertTrue(bd.hasDiffs());
92  		assertEquals("{v1:{f1:1},v2:{f1:2}}", bd.toString());
93  	}
94  
95  	@Test void a11_differentBeanContext() {
96  		var bd = BeanDiff.create(A.create(1, null), A.create(2, "b")).beanContext(BeanContext.DEFAULT_SORTED).build();
97  		assertTrue(bd.hasDiffs());
98  		assertEquals("{v1:{f1:1},v2:{f1:2,f2:'b'}}", bd.toString());
99  	}
100 }
