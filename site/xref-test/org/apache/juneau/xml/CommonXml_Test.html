19  import java.net.*;
20  
21  import org.apache.juneau.*;
22  import org.apache.juneau.annotation.*;
23  import org.apache.juneau.xml.annotation.*;
24  import org.junit.jupiter.api.*;
25  
26  class CommonXml_Test extends TestBase {
27  
28  	//====================================================================================================
29  	// Test 18a - @Bean.uri annotation
30  	//====================================================================================================
31  	@Test void a01_beanUriAnnotation() throws Exception {
32  		var p = XmlParser.DEFAULT;
33  		var s = XmlSerializer.DEFAULT_SQ;
34  
35  		var t = new A("http://foo", 123, "bar");
36  		var xml = s.serialize(t);
37  		assertEquals("<object url='http://foo' id='123'><name>bar</name></object>", xml);
38  
39  		t = p.parse(xml, A.class);
40  		assertEquals("http://foo", t.url.toString());
41  		assertEquals(123, t.id);
42  		assertEquals("bar", t.name);
43  
44  		validateXml(t, s);
45  	}
46  
47  	@Bean(p="url,id,name")
48  	public static class A {
49  		@Xml(format=XmlFormat.ATTR) public URL url;
50  		@Xml(format=ATTR) public int id;
51  		public String name;
52  		public A() {}
53  		public A(String url, int id, String name) {
54  			this.url = url(url);
55  			this.id = id;
56  			this.name = name;
57  		}
58  	}
59  
60  	//====================================================================================================
61  	// Bean.uri annotation, only uri property
62  	//====================================================================================================
63  	@Test void a02_beanUriAnnotationOnlyUriProperty() throws Exception {
64  		var s = XmlSerializer.create().sq().build();
65  
66  		var t = new B("http://foo");
67  		var xml = s.serialize(t);
68  		assertEquals("<object url='http://foo'><url2>http://foo/2</url2></object>", xml);
69  	}
70  
71  	public static class B {
72  		@Xml(format=XmlFormat.ATTR) public URL url;
73  		public URL url2;
74  		public B() {}
75  		public B(String url) {
76  			this.url = url(url);
77  			this.url2 = url(url+"/2");
78  		}
79  	}
80  }
