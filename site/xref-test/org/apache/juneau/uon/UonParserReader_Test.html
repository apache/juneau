182 	private String escape(String s) throws UnsupportedEncodingException {
183 		var sb = new StringBuilder();
184 		byte[] b = s.getBytes("UTF-8");
185 		for (byte element : b)
186 			sb.append('%').append(StringUtils.toHex(element));
187 		return sb.toString();
188 	}
189 
190 	private UonReader r(String in, boolean decodeChars) throws Exception {
191 		return new UonReader(new ParserPipe(in), decodeChars);
192 	}
193 
194 	private static class SlowStringReader extends Reader {
195 
196 		String s;
197 		int i = 0;
198 
199 		SlowStringReader(String s) {
200 			this.s = s;
201 		}
202 
203 		@Override /* Reader */
204 		public int read(char[] cbuf, int off, int len) throws IOException {
205 			if (i >= s.length())
206 				return -1;
207 			cbuf[off] = s.charAt(i++);
208 			return 1;
209 		}
210 
211 		@Override /* Reader */
212 		public void close() throws IOException { /* no-op */ }
213 	}
