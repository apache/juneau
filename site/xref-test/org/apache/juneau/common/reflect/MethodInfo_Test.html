64  		protected AConfigApply(VarResolverSession vr) {
65  			super(AConfig.class, Context.Builder.class, vr);

71  	private static void check(String expected, Object o) {
72  		assertEquals(expected, TO_STRING.apply(o));
73  	}

75  	private static final Function<Object,String> TO_STRING = new Function<>() {
76  		@Override
77  		public String apply(Object t) {
78  			if (t == null)
79  				return null;
80  			if (t instanceof MethodInfo t2)
81  				return t2.getDeclaringClass().getNameSimple() + '.' + ((MethodInfo)t).getShortName();
82  			if (t instanceof Method t2)
83  				return t2.getDeclaringClass().getSimpleName() + '.' + MethodInfo.of((Method)t).getShortName();
84  			if (t instanceof List<?> t2)
85  				return (t2.stream().map(this).collect(Collectors.joining(",")));
86  			if (t instanceof AnnotationInfo t2)
87  				return apply(t2.inner());
88  			if (t instanceof A t2)
89  				return "@A(" + t2.value() + ")";
90  			if (t instanceof PA t2)
91  				return "@PA(" + t2.value() + ")";
92  			if (t instanceof AConfig t2)
93  				return "@AConfig(" + t2.value() + ")";
94  			if (t instanceof AnnotationList t2)
95  				return t2.toString();
96  			if (t instanceof ClassInfo t2)
97  				return t2.getNameSimple();
98  			return t.toString();
99  		}
100 	};

102 	private static MethodInfo ofm(Class<?> c, String name, Class<?>...pt) {
103 		try {
104 			return MethodInfo.of(c.getDeclaredMethod(name, pt));
105 		} catch (NoSuchMethodException | SecurityException e) {
106 			fail(e.getLocalizedMessage());
107 		}
108 		return null;
109 	}
