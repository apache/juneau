17  package org.apache.juneau.common.reflect;
18  
19  import static org.junit.jupiter.api.Assertions.*;
20  
21  import java.lang.annotation.*;
22  
23  import org.junit.jupiter.api.*;
24  
25  /**
26   * Tests for AnnotationInfo value retrieval methods.
27   */
28  public class AnnotationInfo_ValueMethods_Test {
29  
30  	@Target(ElementType.TYPE)
31  	@Retention(RetentionPolicy.RUNTIME)
32  	public @interface TestAnnotation {
33  		String stringValue() default "default";
34  		int intValue() default 42;
35  		boolean boolValue() default true;
36  		long longValue() default 100L;
37  		double doubleValue() default 3.14;
38  		float floatValue() default 2.5f;
39  		Class<?> classValue() default String.class;
40  		String[] stringArray() default {"a", "b"};
41  		Class<?>[] classArray() default {String.class, Integer.class};
42  	}
43  
44  	@TestAnnotation(
45  		stringValue = "test",
46  		intValue = 123,
47  		boolValue = false,
48  		longValue = 999L,
49  		doubleValue = 1.23,
50  		floatValue = 4.56f,
51  		classValue = Integer.class,
52  		stringArray = {"x", "y", "z"},
53  		classArray = {Long.class, Double.class}
54  	)
55  	public static class AnnotatedClass {}
56  
57  	private AnnotationInfo<TestAnnotation> getTestAnnotationInfo() {
58  		var ci = ClassInfo.of(AnnotatedClass.class);
59  		return ci.getAnnotationInfos(TestAnnotation.class).findFirst().orElse(null);
60  	}
61  
62  	@Test
63  	public void testHasName() {
64  		var ai = getTestAnnotationInfo();
65  		
66  		assertTrue(ai.hasName("org.apache.juneau.common.reflect.AnnotationInfo_ValueMethods_Test$TestAnnotation"));
67  		assertFalse(ai.hasName("TestAnnotation"));
68  	}
69  
70  	@Test
71  	public void testHasSimpleName() {
72  		var ai = getTestAnnotationInfo();
73  		
74  		assertTrue(ai.hasSimpleName("TestAnnotation"));
75  		assertFalse(ai.hasSimpleName("org.apache.juneau.common.reflect.AnnotationInfo_ValueMethods_Test$TestAnnotation"));
76  	}
77  
78  	@Test
79  	public void testGetString() {
80  		var ai = getTestAnnotationInfo();
81  		
82  		assertTrue(ai.getString("stringValue").isPresent());
83  		assertEquals("test", ai.getString("stringValue").get());
84  		assertTrue(ai.getString("nonexistent").isEmpty());
85  	}

168 	@Test
169 	public void testGetReturnType() {
170 		var ai = getTestAnnotationInfo();
171 		
172 		// Test various return types
173 		assertTrue(ai.getReturnType("stringValue").isPresent());
174 		assertEquals(String.class, ai.getReturnType("stringValue").get().inner());
175 		assertEquals(int.class, ai.getReturnType("intValue").get().inner());
176 		assertEquals(boolean.class, ai.getReturnType("boolValue").get().inner());
177 		assertEquals(long.class, ai.getReturnType("longValue").get().inner());
178 		assertEquals(double.class, ai.getReturnType("doubleValue").get().inner());
179 		assertEquals(float.class, ai.getReturnType("floatValue").get().inner());
180 		assertEquals(Class.class, ai.getReturnType("classValue").get().inner());
181 		assertEquals(String[].class, ai.getReturnType("stringArray").get().inner());
182 		assertEquals(Class[].class, ai.getReturnType("classArray").get().inner());
183 		
184 		// Nonexistent method
185 		assertTrue(ai.getReturnType("nonexistent").isEmpty());
186 	}
