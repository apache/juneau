33  	@Target({ElementType.TYPE, ElementType.METHOD})
34  	@Retention(RetentionPolicy.RUNTIME)
35  	public @interface TestAnnotation {
36  		String value() default "";
37  		int number() default 0;
38  		boolean flag() default false;
39  		String[] array() default {};
40  	}
41  
42  	/**
43  	 * Implementation of TestAnnotation using AnnotationObject
44  	 */
45  	public static class TestAnnotationObject extends AnnotationObject implements TestAnnotation {
46  
47  		private final String value;
48  		private final int number;
49  		private final boolean flag;
50  		private final String[] array;
51  
52  		public static class Builder extends AnnotationObject.Builder {
53  			String value = "";
54  			int number = 0;
55  			boolean flag = false;
56  			String[] array = {};
57  
58  			public Builder() {
59  				super(TestAnnotation.class);
60  			}
61  
62  			public Builder value(String value) {
63  				this.value = value;
64  				return this;
65  			}
66  
67  			public Builder number(int number) {
68  				this.number = number;
69  				return this;
70  			}
71  
72  			public Builder flag(boolean flag) {
73  				this.flag = flag;
74  				return this;
75  			}
76  
77  			public Builder array(String...array) {
78  				this.array = array;
79  				return this;
80  			}
81  
82  			public TestAnnotation build() {
83  				return new TestAnnotationObject(this);
84  			}
85  		}
86  
87  		public static Builder create() {
88  			return new Builder();
89  		}
90  
91  		public TestAnnotationObject(Builder b) {
92  			super(b);
93  			value = b.value;
94  			number = b.number;
95  			flag = b.flag;
96  			this.array = Arrays.copyOf(b.array, b.array.length);
97  		}
98  
99  		@Override
100 		public String value() {
101 			return value;
102 		}
103 
104 		@Override
105 		public int number() {
106 			return number;
107 		}
108 
109 		@Override
110 		public boolean flag() {
111 			return flag;
112 		}
113 
114 		@Override
115 		public String[] array() {
116 			return array;
117 		}
118 	}
