33  	//====================================================================================================
34  	@Test
35  	void a01_findCause() {
36  		IOException rootCause = new IOException("root");
37  		RuntimeException middleCause = new RuntimeException("middle", rootCause);
38  		Exception topException = new Exception("top", middleCause);
39  
40  		// Find IOException
41  		assertTrue(findCause(topException, IOException.class).isPresent());
42  		assertEquals("root", findCause(topException, IOException.class).get().getMessage());
43  
44  		// Find RuntimeException
45  		assertTrue(findCause(topException, RuntimeException.class).isPresent());
46  		assertEquals("middle", findCause(topException, RuntimeException.class).get().getMessage());
47  
48  		// Find Exception (returns itself)
49  		assertTrue(findCause(topException, Exception.class).isPresent());
50  		assertEquals("top", findCause(topException, Exception.class).get().getMessage());
51  
52  		// Not found
53  		assertFalse(findCause(topException, IllegalArgumentException.class).isPresent());
54  
55  		// Null exception
56  		assertFalse(findCause(null, IOException.class).isPresent());
57  	}

71  	@Test
72  	void a03_findCause_longChain() {
73  		Throwable cause = new IllegalStateException("root");
74  		for (int i = 0; i < 10; i++) {
75  			cause = new RuntimeException("level" + i, cause);
76  		}
77  
78  		// Should find the root cause
79  		assertTrue(findCause(cause, IllegalStateException.class).isPresent());
80  		assertEquals("root", findCause(cause, IllegalStateException.class).get().getMessage());
81  	}
