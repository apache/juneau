17  package org.apache.juneau.common.utils;
18  
19  import static org.apache.juneau.common.utils.CollectionUtils.*;
20  import static org.junit.jupiter.api.Assertions.*;
21  
22  import java.util.*;
23  
24  import org.junit.jupiter.api.*;
25  
26  public class SetBuilder_Test {
27  
28      @Test
29      void build_basic_add_and_addAll() {
30          Set<String> s = setb(String.class)
31              .add("a")
32              .add("b", "c")
33              .addAll(new LinkedHashSet<>(l("d", "e")))
34              .build();
35          assertEquals(new LinkedHashSet<>(l("a","b","c","d","e")), s);
36      }
37  
38      @Test
39      void build_sparse_and_unmodifiable() {
40          assertNull(setb(String.class).sparse().build());
41          Set<String> s = setb(String.class).add("a").unmodifiable().build();
42          assertThrows(UnsupportedOperationException.class, () -> s.add("b"));
43      }
44  
45      @Test
46      void sorted_natural_and_custom() {
47          Set<Integer> s1 = setb(Integer.class).add(3,1,2).sorted().build();
48          assertEquals(new TreeSet<>(l(1,2,3)), s1);
49  
50          Set<Integer> s2 = setb(Integer.class).add(3,1,2).sorted(Comparator.reverseOrder()).build();
51          assertEquals(new TreeSet<>(l(3,2,1)), s2);
52      }
53  }
