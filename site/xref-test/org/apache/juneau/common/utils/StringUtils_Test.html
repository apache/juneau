183 		assertEquals(2147483648L, parseNumber(s, null));

225 		// Autodetected floating point numbers smaller than Float.MAX_VALUE
226 		s = String.valueOf(Float.MAX_VALUE / 2);
227 		assertTrue(isNumeric(s));
228 		assertTrue(parseNumber(s, null) instanceof Float);

230 
231 		s = String.valueOf((-Float.MAX_VALUE) / 2);
