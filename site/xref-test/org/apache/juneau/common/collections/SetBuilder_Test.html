233 	@Test
234 	void g01_copy() {
235 		var original = new LinkedHashSet<String>();
236 		original.add("a");
237 
238 		var set = SetBuilder.create(String.class)
239 			.to(original)
240 			.add("b")
241 			.copy()
242 			.add("c")
243 			.build();
244 
245 		assertSize(3, set);
246 		assertSize(2, original);  // Original has "a" and "b" added before copy()
247 		assertNotSame(original, set);  // After copy(), they're different sets
248 	}
