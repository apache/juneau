37  	@Test void a01_gettersAndSetters() {
38  		var t = new Swagger();
39  
40  		// Basic property setters
41  		assertBean(
42  			t.setBasePath("a").setHost("b").setSwagger("c"),
43  			"basePath,host,swagger",
44  			"a,b,c"
45  		);
46  
47  		// Complex objects
48  		assertBean(
49  			t.setInfo(info("a", "b")).setExternalDocs(externalDocumentation("c"))
50  				.setSchemes(set("d","e")).setConsumes(set(MediaType.of("text/f")))
51  				.setProduces(set(MediaType.of("text/g")))
52  				.setPaths(map("h", new OperationMap().append("i",operation().setSummary("j"))))
53  				.setDefinitions(map("k",JsonMap.of("type","l")))
54  				.setParameters(map("m",parameterInfo().setName("n")))
55  				.setResponses(map("1",responseInfo("o")))
56  				.setSecurityDefinitions(map("p",securityScheme("q")))
57  				.setSecurity(set(map("r",alist("s"))))
58  				.setTags(set(tag("t"))),
59  			"consumes,definitions{k{type}},externalDocs{url},info{title,version},parameters{m{name}},paths{h{i{summary}}},produces,responses{1{description}},schemes,security,securityDefinitions{p{type}},tags{#{name}}",
60  			"[text/f],{{l}},{c},{a,b},{{n}},{{{j}}},[text/g],{{o}},[d,e],[{r=[s]}],{{q}},{[{t}]}"
61  		);
62  
63  		// Null values
64  		assertBean(
65  			t.setSwagger(null).setHost(null).setBasePath(null).setSchemes((Collection<String>)null)
66  				.setConsumes((Collection<MediaType>)null).setProduces((Collection<MediaType>)null)
67  				.setPaths((Map<String,OperationMap>)null).setDefinitions((Map<String,JsonMap>)null)
68  				.setParameters((Map<String,ParameterInfo>)null).setResponses((Map<String,ResponseInfo>)null)
69  				.setSecurityDefinitions((Map<String,SecurityScheme>)null).setSecurity((Collection<Map<String, List<String>>>)null)
70  				.setTags((Collection<Tag>)null),
71  			"basePath,consumes,definitions,host,parameters,paths,produces,responses,schemes,security,securityDefinitions,swagger,tags",
72  			"<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>"
73  		);
74  
75  		// Other methods - empty collections/maps
76  		assertBean(
77  			t.setSchemes(set()).setConsumes(set()).setProduces(set()).setDefinitions(map())
78  				.setParameters(map()).setResponses(map()).setSecurityDefinitions(map())
79  				.setSecurity(set()).setTags(set()),
80  			"consumes,definitions,parameters,produces,responses,schemes,security,securityDefinitions,tags",
81  			"[],{},{},[],{},[],[],{},[]"
82  		);
83  
84  		assertNull(t.setPaths(map()).getPaths());  // BUG?
85  
86  		// addX methods.
87  		assertBean(t.setPaths(null).addPath("a", "a1", operation().setDescription("a2")).addPath("b", null, null),
88  			"paths", "{a={a1={\"description\":\"a2\"}},b={<null>=<null>}}");
89  		assertBean(t.setDefinitions(null).addDefinition("a", JsonMap.of("type","a1")).addDefinition("b", (JsonMap)null).addDefinition(null, JsonMap.of("type", "c1")),
90  			"definitions", "{a={type=a1},b=<null>,<null>={type=c1}}");
91  		assertBean(t.setParameters(null).addParameter("a", parameterInfo().setIn("a1")).addParameter("b", null).addParameter(null, parameterInfo().setIn("c1")),
92  			"parameters", "{a={\"in\":\"a1\"},b=<null>,<null>={\"in\":\"c1\"}}");
93  		assertBean(t.setResponses(null).addResponse("a", responseInfo("a1")).addResponse(null, responseInfo("b1")).addResponse("c", null),
94  			"responses", "{a={\"description\":\"a1\"},<null>={\"description\":\"b1\"},c=<null>}");
95  		assertBean(t.setSecurityDefinitions(null).addSecurityDefinition("a", securityScheme("a1")).addSecurityDefinition("b", null).addSecurityDefinition(null, securityScheme("c1")),
96  			"securityDefinitions", "{a={\"type\":\"a1\"},b=<null>,<null>={\"type\":\"c1\"}}");
97  		assertBean(t.setSecurity(null).addSecurity("a", "a1", "a2").addSecurity("b", (String)null).addSecurity(null, "d1", "d2"),
98  			"security", "[{a=[a1,a2]},{b=[<null>]},{<null>=[d1,d2]}]");
99  	}

104 	@Test void b01_set() {
105 		var t = new Swagger();
106 
107 		t
108 			.set("basePath", "a")
109 			.set("consumes", set(MediaType.of("text/b")))
110 			.set("definitions", map("c",schemaInfo().setType("c1")))
111 			.set("externalDocs", externalDocumentation("d"))
112 			.set("host", "e")
113 			.set("info", info("f1", "f2"))
114 			.set("parameters", map("g",parameterInfo("g1", "g2")))
115 			.set("paths", map("h",map("h1",operation().setOperationId("h2"))))
116 			.set("produces", set(MediaType.of("text/i")))
117 			.set("responses", map("j",responseInfo("j1")))
118 			.set("schemes", set("k1"))
119 			.set("security", set(map("l1",alist("l2"))))
120 			.set("securityDefinitions", map("m",securityScheme("m1")))
121 			.set("swagger", "n")
122 			.set("tags", set(tag("o")))
123 			.set("$ref", "p");
124 
125 		assertBean(t,
126 			"basePath,consumes,definitions{c{type}},externalDocs{url},host,info{title,version},parameters{g{in,name}},paths{h{h1{operationId}}},produces,$ref,responses{j{description}},schemes,security,securityDefinitions{m{type}},swagger,tags{#{name}}",
127 			"a,[text/b],{{c1}},{d},e,{f1,f2},{{g1,g2}},{{{h2}}},[text/i],p,{{j1}},[k1],[{l1=[l2]}],{{m1}},n,{[{o}]}");
128 
129 		t
130 			.set("basePath", "a")
131 			.set("consumes", "['text/b']")
132 			.set("definitions", "{c:{type:'c1'}}")
133 			.set("externalDocs", "{url:'d'}")
134 			.set("host", "e")
135 			.set("info", "{title:'f1',version:'f2'}")
136 			.set("parameters", "{g:{'in':'g1',name:'g2'}}")
137 			.set("paths", "{h:{h1:{operationId:'h2'}}}")
138 			.set("produces", "['text/i']")
139 			.set("responses", "{j:{description:'j1'}}")
140 			.set("schemes", "['k1']")
141 			.set("security", "[{l1:['l2']}]")
142 			.set("securityDefinitions", "{m:{type:'m1'}}")
143 			.set("swagger", "n")
144 			.set("tags", "[{name:'o'}]")
145 			.set("$ref", "ref");
146 
147 		assertBean(t,
148 			"swagger,info{title,version},tags{#{name}},externalDocs{url},basePath,schemes,consumes,produces,paths{h{h1{operationId}}},definitions{c{type}},parameters{g{in,name}},responses{j{description}},securityDefinitions{m{type}},security,$ref",
149 			"n,{f1,f2},{[{o}]},{d},a,[k1],[text/b],[text/i],{{{h2}}},{{c1}},{{g1,g2}},{{j1}},{{m1}},[{l1=[l2]}],ref");
150 
151 		t
152 			.set("basePath", Utils.sb("a"))
153 			.set("consumes", Utils.sb("['text/b']"))
154 			.set("definitions", Utils.sb("{c:{type:'c1'}}"))
155 			.set("externalDocs", Utils.sb("{url:'d'}"))
156 			.set("host", Utils.sb("e"))
157 			.set("info", Utils.sb("{title:'f1',version:'f2'}"))
158 			.set("parameters", Utils.sb("{g:{'in':'g1',name:'g2'}}"))
159 			.set("paths", Utils.sb("{h:{h1:{operationId:'h2'}}}"))
160 			.set("produces", Utils.sb("['text/i']"))
161 			.set("responses", Utils.sb("{j:{description:'j1'}}"))
162 			.set("schemes", Utils.sb("['k1']"))
163 			.set("security", Utils.sb("[{l1:['l2']}]"))
164 			.set("securityDefinitions", Utils.sb("{m:{type:'m1'}}"))
165 			.set("swagger", Utils.sb("n"))
166 			.set("tags", Utils.sb("[{name:'o'}]"))
167 			.set("$ref", Utils.sb("ref"));
168 
169 		assertBean(t,
170 			"swagger,info{title,version},tags{#{name}},externalDocs{url},basePath,schemes,consumes,produces,paths{h{h1{operationId}}},definitions{c{type}},parameters{g{in,name}},responses{j{description}},securityDefinitions{m{type}},security,$ref",
171 			"n,{f1,f2},{[{o}]},{d},a,[k1],[text/b],[text/i],{{{h2}}},{{c1}},{{g1,g2}},{{j1}},{{m1}},[{l1=[l2]}],ref");
172 
173 		assertMapped(t, (obj,prop) -> obj.get(prop, String.class),
174 			"basePath,consumes,definitions,externalDocs,host,info,parameters,paths,produces,responses,schemes,security,securityDefinitions,swagger,tags,$ref",
175 			"a,['text/b'],{c:{type:'c1'}},{url:'d'},e,{title:'f1',version:'f2'},{g:{'in':'g1',name:'g2'}},{h:{h1:{operationId:'h2'}}},['text/i'],{j:{description:'j1'}},['k1'],[{l1:['l2']}],{m:{type:'m1'}},n,[{name:'o'}],ref");
176 
177 		assertMapped(t, (obj,prop) -> obj.get(prop, Object.class).getClass().getSimpleName(),
178 			"basePath,consumes,definitions,externalDocs,host,info,parameters,paths,produces,responses,schemes,security,securityDefinitions,swagger,tags,$ref",
179 			"String,LinkedHashSet,LinkedHashMap,ExternalDocumentation,String,Info,LinkedHashMap,TreeMap,LinkedHashSet,LinkedHashMap,LinkedHashSet,ArrayList,LinkedHashMap,String,LinkedHashSet,StringBuilder");
180 
181 		assertMapped(t, (o,k) -> o.get(k, List.class).get(0).getClass().getSimpleName(), "consumes,tags", "MediaType,Tag");
182 		assertMapped(t, (o,k) -> o.get(k, Map.class).values().iterator().next().getClass().getSimpleName(), "definitions,parameters,responses,securityDefinitions", "JsonMap,ParameterInfo,ResponseInfo,SecurityScheme");
183 
184 		t.set("null", null).set(null, "null");
185 		assertNull(t.get("null", Object.class));
186 		assertNull(t.get(null, Object.class));
187 		assertNull(t.get("foo", Object.class));
188 	}
