13  package org.apache.juneau;
14  
15  import static org.apache.juneau.TestUtils.*;
16  import static org.junit.jupiter.api.Assertions.*;
17  
18  import org.apache.juneau.json.*;
19  import org.apache.juneau.testutils.pojos.*;
20  import org.junit.jupiter.api.*;
21  
22  class BeanContext_Test extends SimpleTestBase {
23  
24  	BeanContext bc = BeanContext.DEFAULT;
25  	BeanSession bs = BeanContext.DEFAULT_SESSION;
26  
27  	public interface A1 {
28  		int getF1();
29  		void setF1(int f1);
30  	}
31  
32  	@Test void a01_normalCachableBean() throws ExecutableException {
33  		var cm1 = bc.getClassMeta(A1.class);
34  		var cm2 = bc.getClassMeta(A1.class);
35  		assertSame(cm1, cm2);
36  	}
37  
38  	interface A2 {
39  		void foo(int x);
40  	}
41  
42  	@Test void a02_lambdaExpressionsNotCached() throws ExecutableException {
43  		var bc2 = BeanContext.DEFAULT;
44  		var fi = (A2)System.out::println;
45  		var cm1 = bc2.getClassMeta(fi.getClass());
46  		var cm2 = bc2.getClassMeta(fi.getClass());
47  		assertNotSame(cm1, cm2);
48  	}
49  
50  	@Test void a03_proxiesNotCached() throws ExecutableException {
51  		var a1 = bs.getBeanMeta(A1.class).newBean(null);
52  		var cm1 = bc.getClassMeta(a1.getClass());
53  		var cm2 = bc.getClassMeta(a1.getClass());
54  		assertNotSame(cm1, cm2);
55  	}
56  
57  	@Test void b01_ignoreUnknownEnumValues() {
58  		var p1 = JsonParser.DEFAULT;
59  		assertThrowsWithMessage(Exception.class, "Could not resolve enum value 'UNKNOWN' on class 'org.apache.juneau.testutils.pojos.TestEnum'", () -> p1.parse("'UNKNOWN'", TestEnum.class));
60  
61  		var p2 = JsonParser.create().ignoreUnknownEnumValues().build();
62  		assertNull(p2.parse("'UNKNOWN'", TestEnum.class));
63  	}
64  }
