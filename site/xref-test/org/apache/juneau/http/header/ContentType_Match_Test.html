24  /**
25   * Verifies that the Content-Type class handles matching correctly.
26   */
27  class ContentType_Match_Test extends TestBase {
28  
29  	private static final Input[] INPUT = {
30  		// Simple matches
31  		input("SimpleMatch-1", "text/json", "['text/json']", 0),
32  		input("SimpleMatch-2", "text/json", "['text/json','text/foo']", 0),
33  		input("SimpleMatch-3", "text/json", "['text/foo','text/json']", 1),
34  
35  		// Simple no-matches
36  		input("SimpleNoMatch-1", "text/jsonx", "['text/json']", -1),
37  		input("SimpleNoMatch-2", "text/jso", "['text/json']", -1),
38  		input("SimpleNoMatch-3", "text/json", "['application/json']", -1),
39  		input("SimpleNoMatch-4", "text/json", "[]", -1),
40  
41  		input("XmlAndRdf-1", "text/xml+rdf", "['text/xml','text/xml+rdf']", 1),
42  		input("XmlAndRdf-2", "text/xml+rdf", "['text/xml+rdf','text/xml']", 0),
43  
44  		// Fuzzy matches
45  		input("Fuzzy-1", "text/1+2", "['text/1+2']", 0),
46  		// Order of subtype parts shouldn't matter.
47  		input("Fuzzy-2", "text/2+1", "['text/1+2']", 0),
48  		// Should match if Accept has 'extra' subtypes.
49  		// For example, "Accept: text/json+activity" should match against the "text/json" serializer.
50  		input("Fuzzy-3", "text/1+2", "['text/1']", 0),
51  		// Shouldn't match because the accept media type must be at least a subset of the real media type
52  		// For example, "Accept: text/json" should not match against the "text/json+lax" serializer.
53  		input("Fuzzy-4", "text/1", "['text/1+2']", -1),
54  		input("Fuzzy-5", "text/1+2", "['text/1','text/1+3']", 0),
55  		// "text/1+2" should be a better match than just "text/1"
56  		input("Fuzzy-6", "text/1+2", "['text/1','text/1+2','text/1+2+3']", 1),
57  		// Same as last, but mix up the order a bit.
58  		input("Fuzzy-7", "text/1+2", "['text/1+2+3','text/1','text/1+2']", 2),
59  		// Same as last, but mix up the order of the subtypes as well.
60  		input("Fuzzy-8", "text/1+2", "['text/3+2+1','text/1','text/2+1']", 2),
61  		input("Fuzzy-9", "text/1+2+3+4", "['text/1+2','text/1+2+3']", 1),
62  		input("Fuzzy-10", "text/1+2+3+4", "['text/1+2+3','text/1+2']", 0),
63  		input("Fuzzy-11", "text/4+2+3+1", "['text/1+2+3','text/1+2']", 0),
64  		input("Fuzzy-12", "text/4+2+3+1", "['text/1+2','text/1+2+3']", 1)
65  	};
66  
67  	static Input[] input() {
68  		return INPUT;
69  	}
70  
71  	protected static Input input(String label, String contentType, String mediaTypes, int expected) {
72  		return new Input(label, contentType, mediaTypes, expected);
73  	}
74  
75  	private static class Input {
76  		String label, contentType, mediaTypes;
77  		int expected;
78  
79  		Input(String label, String contentType, String mediaTypes, int expected) {
80  			this.label = label;
81  			this.contentType = contentType;
82  			this.mediaTypes = mediaTypes;
83  			this.expected = expected;
84  		}
85  	}
86  
87  	@ParameterizedTest
88  	@MethodSource("input")
89  	void a01_basic(Input input) throws Exception {
90  		var ct = contentType(input.contentType);
91  		var mt = JsonParser.DEFAULT.parse(input.mediaTypes, MediaType[].class);
92  		assertEquals(input.expected, ct.match(alist(mt)), fms("{0} failed", input.label));
93  	}
94  }
