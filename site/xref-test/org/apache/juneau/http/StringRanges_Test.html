13  package org.apache.juneau.http;
14  
15  import static org.apache.juneau.StringRanges.*;
16  import static org.apache.juneau.TestUtils.*;
17  import static org.junit.jupiter.api.Assertions.*;
18  
19  import org.apache.juneau.*;
20  import org.junit.jupiter.api.*;
21  
22  class StringRanges_Test extends TestBase {
23  
24  	@Test void a01_match() {
25  		var x = alist("foo","bar","baz");
26  
27  		assertEquals(-1, of((String)null).match(x));
28  
29  		assertEquals(1, of("foo;q=0.5,bar").match(x));
30  		assertEquals(1, of("foo;q=0.5,bar").match(x));
31  		assertEquals(0, of("foo,*").match(x));
32  		assertEquals(1, of("*,bar").match(x));
33  		assertEquals(0, of("foo;q=0.6,bar;q=0.5").match(x));
34  		assertEquals(0, of("foo;q=0.6,bar;q=0.5,qux").match(x));
35  		assertEquals(-1, of("qux").match(x));
36  		assertEquals(-1, of("qux,q2x;q=0").match(x));
37  		assertEquals(-1, of("foo;q=0").match(x));
38  	}
39  
40  	@Test void a02_getRange() {
41  		assertNotNull(of("foo").getRange(0));
42  		assertNull(of((String)null).getRange(0));
43  		assertNull(of("").getRange(0));
44  		assertNull(of((String)null).getRange(-1));
45  		assertNull(of((String)null).getRange(1));
46  	}
47  
48  	@Test void a03_getRanges() {
49  		assertList(of("foo").toList(), "foo");
50  		assertEmpty(of((String)null).toList());
51  	}
52  
53  	@Test void a04_toString() {
54  		assertString("*", new StringRange("*"));
55  		assertString("qux, bar;q=0.9, foo;q=0.6", of("foo;q=0.6,bar;q=0.9,qux"));
56  	}
57  }
