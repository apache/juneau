31  	// Verifies that media type parameters are distinguished from media range extensions.
32  	//-----------------------------------------------------------------------------------------------------------------
33  	@Test void a01_extensions() {
34  		var x1 = of("text/json");
35  		var x2 = x1.getRange(0);
36  		assertBean(x2, "toString,parameters,qValue,extensions", "text/json,[],1.0,[]");
37  
38  		assertNull(x1.getRange(-1));
39  		assertNull(x1.getRange(1));
40  
41  		x1 = of("foo,bar");
42  		x2 = x1.getRange(0);
43  		assertString("foo", x2);
44  		assertJson("[]", x2.getParameters());
45  		assertString("1.0", x2.getQValue());
46  		assertJson("[]", x2.getExtensions());
47  
48  		x1 = of(" foo , bar ");
49  		x2 = x1.getRange(0);
50  		assertString("foo", x2);
51  		assertJson("[]", x2.getParameters());
52  		assertString("1.0", x2.getQValue());
53  		assertJson("[]", x2.getExtensions());
54  
55  		x1 = of("text/json;a=1;q=0.9;b=2");
56  		x2 = x1.getRange(0);
57  		assertString("text/json;a=1;q=0.9;b=2", x2);
58  		assertJson("[{name:'a',value:'1'}]", x2.getParameters());
59  		assertString("0.9", x2.getQValue());
60  		assertJson("[{name:'b',value:'2'}]", x2.getExtensions());
61  
62  		x1 = of("text/json;a=1;a=2;q=0.9;b=3;b=4");
63  		x2 = x1.getRange(0);
64  		assertString("text/json;a=1;a=2;q=0.9;b=3;b=4", x2);
65  		assertJson("[{name:'a',value:'1'},{name:'a',value:'2'}]", x2.getParameters());
66  		assertString("0.9", x2.getQValue());
67  		assertJson("[{name:'b',value:'3'},{name:'b',value:'4'}]", x2.getExtensions());
68  
69  		x1 = of("text/json;a=1;a=2;q=1.0;b=3;b=4");
70  		x2 = x1.getRange(0);
71  		assertString("text/json;a=1;a=2;q=1.0;b=3;b=4", x2);
72  		assertJson("[{name:'a',value:'1'},{name:'a',value:'2'}]", x2.getParameters());
73  		assertString("1.0", x2.getQValue());
74  		assertJson("[{name:'b',value:'3'},{name:'b',value:'4'}]", x2.getExtensions());
75  
76  		x1 = of("text/json;a=1");
77  		x2 = x1.getRange(0);
78  		assertString("text/json;a=1", x2);
79  		assertJson("[{name:'a',value:'1'}]", x2.getParameters());
80  		assertString("1.0", x2.getQValue());
81  		assertJson("[]", x2.getExtensions());
82  
83  		x1 = of("text/json;a=1;");
84  		x2 = x1.getRange(0);
85  		assertString("text/json;a=1", x2);
86  		assertJson("[{name:'a',value:'1'}]", x2.getParameters());
87  		assertString("1.0", x2.getQValue());
88  		assertJson("[]", x2.getExtensions());
89  
90  		x1 = of("text/json;q=0.9");
91  		x2 = x1.getRange(0);
92  		assertString("text/json;q=0.9", x2);
93  		assertJson("[]", x2.getParameters());
94  		assertString("0.9", x2.getQValue());
95  		assertJson("[]", x2.getExtensions());
96  
97  		x1 = of("text/json;q=0.9;");
98  		x2 = x1.getRange(0);
99  		assertString("text/json;q=0.9", x2);
100 		assertJson("[]", x2.getParameters());
101 		assertString("0.9", x2.getQValue());
102 		assertJson("[]", x2.getExtensions());
103 	}

105 	@Test void a02_hasSubtypePart() {
106 		var mr = of("text/json+x,text/foo+y;q=0.0");
107 		assertTrue(mr.hasSubtypePart("json"));
108 		assertTrue(mr.hasSubtypePart("x"));
109 		assertFalse(mr.hasSubtypePart("foo"));
110 		assertFalse(mr.hasSubtypePart("y"));
111 	}

113 	@Test void a03_ordering() {
114 		assertString("text/json", of("text/json"));
115 		assertString("text/json,text/*", of("text/json,text/*"));
116 		assertString("text/json,text/*", of("text/*,text/json"));
117 		assertString("text/*,text/*", of("text/*,text/*"));
118 		assertString("text/*,*/text", of("*/text,text/*"));
119 		assertString("text/*,*/text", of("text/*,*/text"));
120 		assertString("a;q=0.9,b;q=0.1", of("a;q=0.9,b;q=0.1"));
121 		assertString("b;q=0.9,a;q=0.1", of("b;q=0.9,a;q=0.1"));
122 		assertString("a,b;q=0.9,c;q=0.1,d;q=0.0", of("a,b;q=0.9,c;q=0.1,d;q=0"));
123 		assertString("a,b;q=0.9,c;q=0.1,d;q=0.0", of("d;q=0,c;q=0.1,b;q=0.9,a"));
124 		assertString("a,b;q=0.9,c;q=0.1,d;q=0.0", of("a;q=1,b;q=0.9,c;q=0.1,d;q=0"));
125 		assertString("a,b;q=0.9,c;q=0.1,d;q=0.0", of("d;q=0,c;q=0.1,b;q=0.9,a;q=1"));
126 		assertString("d,c;q=0.9,b;q=0.1,a;q=0.0", of("a;q=0,b;q=0.1,c;q=0.9,d;q=1"));
127 		assertString("*", of("*"));
128 		assertString("", of(""));
129 		assertString("", of(null));
130 		assertString("foo/bar/baz", of("foo/bar/baz"));
131 	}
