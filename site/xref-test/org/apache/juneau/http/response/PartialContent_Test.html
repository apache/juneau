13  package org.apache.juneau.http.response;
14  
15  import static org.apache.juneau.http.HttpResponses.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.rest.annotation.*;
19  import org.apache.juneau.rest.mock.*;
20  import org.junit.jupiter.api.*;
21  
22  class PartialContent_Test extends SimpleTestBase {
23  
24  	@Rest
25  	public static class A {
26  		@RestGet public PartialContent a1() { return PARTIAL_CONTENT; }
27  		@RestGet public PartialContent a2() { return partialContent().setContent("foo"); }
28  		@RestGet public PartialContent a3() { return partialContent().setHeader2("Foo","bar"); }
29  	}
30  
31  	@Test void a01_basic() throws Exception {
32  		var client = MockRestClient.createLax(A.class).build();
33  
34  		client.get("/a1")
35  			.run()
36  			.assertStatus(206)
37  			.assertContent("Partial Content");
38  		client.get("/a2")
39  			.run()
40  			.assertStatus(206)
41  			.assertContent("foo");
42  		client.get("/a3")
43  			.run()
44  			.assertStatus(206)
45  			.assertHeader("Foo").is("bar");
46  	}
47  }
