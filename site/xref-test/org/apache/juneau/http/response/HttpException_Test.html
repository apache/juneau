33  	@Rest
34  	public static class A {
35  		@RestGet
36  		public void f1() throws BasicHttpException {
37  			throw new BasicHttpException(225, new RuntimeException("foo"), "bar {0}", "baz");
38  		}
39  		@RestGet
40  		public void f2() throws BasicHttpException {
41  			throw new BasicHttpException(225, "foo");
42  		}
43  		@RestGet
44  		public void f3() throws BasicHttpException {
45  			throw new BasicHttpException(225, new RuntimeException("baz"));
46  		}
47  		@RestGet
48  		public void f4() throws BasicHttpException {
49  			throw new BasicHttpException(225, "bar {0}", "baz");
50  		}
51  		@RestGet
52  		public void f5() throws BasicHttpException {
53  			throw httpException().setStatusCode2(225).setHeader2("Foo", "bar");
54  		}
55  		@RestGet
56  		public void f6() throws BasicHttpException {
57  			throw httpException().setStatusCode2(225).setHeaders(l(
58  				BasicHeader.of("X-Custom", "value1"),
59  				BasicHeader.of("X-Test", "value2")
60  			));
61  		}
62  		@RestGet
63  		public void f7() throws BasicHttpException {
64  			throw httpException().setStatusCode2(225).setContent("Custom exception content");
65  		}
66  	}

68  	@Test void a01_basic() throws Exception {
69  		var c = MockRestClient.create(A.class).ignoreErrors().build();
70  
71  		c.get("/f1").run()
72  			.assertStatus().asCode().is(225)
73  			.assertContent("bar baz");
74  		c.get("/f2").run()
75  			.assertStatus().asCode().is(225)
76  			.assertContent("foo");
77  		c.get("/f3").run()
78  			.assertStatus().asCode().is(225)
79  			.assertContent("java.lang.RuntimeException: baz");
80  		c.get("/f4").run()
81  			.assertStatus().asCode().is(225)
82  			.assertContent("bar baz");
83  		c.get("/f5").run()
84  			.assertStatus().asCode().is(225)
85  			.assertContent("")
86  			.assertHeader("Foo").is("bar");
87  		c.get("/f6").run()
88  			.assertStatus().asCode().is(225)
89  			.assertHeader("X-Custom").is("value1")
90  			.assertHeader("X-Test").is("value2");
91  		c.get("/f7").run()
92  			.assertStatus().asCode().is(225)
93  			.assertContent("Custom exception content");
94  	}

96  	@Test void a02_getRootCause() {
97  		var x = new BasicHttpException(100, null);
98  		assertNull(x.getRootCause());
99  
100 		x = new BasicHttpException(100, new BasicHttpException(100,"foo"));
101 		assertNull(x.getRootCause());
102 
103 		x = new BasicHttpException(100, new RuntimeException("foo"));
104 		assertInstanceOf(RuntimeException.class, x.getRootCause());
105 
106 		x = new BasicHttpException(100, new BasicHttpException(100, new RuntimeException("foo")));
107 		assertInstanceOf(RuntimeException.class, x.getRootCause());
108 
109 		x = new BasicHttpException(100, new InvocationTargetException(new RuntimeException("foo")));
110 		assertInstanceOf(RuntimeException.class, x.getRootCause());
111 	}
112 
113 	@Test void a03_getFullStackMessage() {
114 		var x = new BasicHttpException(100, null);
115 		assertEquals("Continue", x.getFullStackMessage(false));
116 		assertEquals("Continue", x.getFullStackMessage(true));
117 
118 		x = new BasicHttpException(100, "foo<bar>&baz");
119 		assertEquals("foo<bar>&baz", x.getFullStackMessage(false));
120 		assertEquals("foo bar  baz", x.getFullStackMessage(true));
121 
122 		x = new BasicHttpException(100, new RuntimeException("foo<bar>&qux"), "foo{0}","<bar>&baz");
123 		assertEquals("foo<bar>&baz\nCaused by (RuntimeException): foo<bar>&qux", x.getFullStackMessage(false));
124 		assertEquals("foo bar  baz\nCaused by (RuntimeException): foo bar  qux", x.getFullStackMessage(true));
125 
126 		x = new BasicHttpException(100, new RuntimeException(), "foo{0}","<bar>&baz");
127 		assertEquals("foo<bar>&baz\nCaused by (RuntimeException)", x.getFullStackMessage(false));
128 		assertEquals("foo bar  baz\nCaused by (RuntimeException)", x.getFullStackMessage(true));
129 	}

131 	@Test void a04_fluentSetters() {
132 		var x = httpException().setStatusCode2(500);
133 
134 		// Test setHeaders(List<Header>) returns same instance for fluent chaining
135 		assertSame(x, x.setHeaders(l(
136 			BasicHeader.of("X-Fluent-Test", "fluent-value")
137 		)));
138 		assertEquals("fluent-value", x.getFirstHeader("X-Fluent-Test").getValue());
139 
140 		// Test setContent(String) returns same instance for fluent chaining
141 		assertSame(x, x.setContent("test error content"));
142 
143 		// Test setContent(HttpEntity) returns same instance for fluent chaining
144 		var x2 = httpException().setStatusCode2(500);
145 		HttpEntity entity = x2.getEntity();
146 		assertSame(x2, x2.setContent(entity));
147 	}
