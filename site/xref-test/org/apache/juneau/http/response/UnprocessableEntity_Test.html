13  package org.apache.juneau.http.response;
14  
15  import static org.apache.juneau.http.HttpResponses.*;
16  import static org.apache.juneau.http.response.UnprocessableEntity.*;
17  
18  import org.apache.juneau.*;
19  import org.apache.juneau.rest.annotation.*;
20  import org.apache.juneau.rest.mock.*;
21  import org.junit.jupiter.api.*;
22  
23  class UnprocessableEntity_Test extends SimpleTestBase {
24  
25  	@Rest
26  	public static class A {
27  		@RestGet
28  		public void f1() throws UnprocessableEntity {
29  			throw new UnprocessableEntity();
30  		}
31  		@RestGet
32  		public void f2() throws UnprocessableEntity {
33  			throw new UnprocessableEntity("foo {0}", "bar");
34  		}
35  		@RestGet
36  		public void f3() throws UnprocessableEntity {
37  			throw new UnprocessableEntity(new RuntimeException("baz"));
38  		}
39  		@RestGet
40  		public void f4() throws UnprocessableEntity {
41  			throw new UnprocessableEntity(new RuntimeException("baz"), "foo {0}", "bar");
42  		}
43  		@RestGet
44  		public void f5() throws UnprocessableEntity {
45  			throw unprocessableEntity().setHeader2("Foo", "bar");
46  		}
47  		@RestGet
48  		public void f6() throws UnprocessableEntity {
49  			throw new UnprocessableEntity("foo");
50  		}
51  	}
52  
53  	@Test void a01_basic() throws Exception {
54  		var c = MockRestClient.create(A.class).ignoreErrors().noTrace().build();
55  
56  		c.get("/f1").run()
57  			.assertStatus().asCode().is(STATUS_CODE)
58  			.assertContent(REASON_PHRASE);
59  		c.get("/f2").run()
60  			.assertStatus().asCode().is(STATUS_CODE)
61  			.assertContent("foo bar");
62  		c.get("/f3").run()
63  			.assertStatus().asCode().is(STATUS_CODE)
64  			.assertContent("baz");
65  		c.get("/f4").run()
66  			.assertStatus().asCode().is(STATUS_CODE)
67  			.assertContent("foo bar");
68  		c.get("/f5").run()
69  			.assertStatus().asCode().is(STATUS_CODE)
70  			.assertContent(REASON_PHRASE)
71  			.assertHeader("Foo").is("bar");
72  		c.get("/f6").run()
73  			.assertStatus().asCode().is(STATUS_CODE)
74  			.assertContent("foo");
75  	}
76  }
