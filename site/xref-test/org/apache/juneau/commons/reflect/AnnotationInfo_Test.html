732 		var annotationType = ToMapTestAnnotation.class;
733 		var handler = new java.lang.reflect.InvocationHandler() {
734 			@Override
735 			public Object invoke(Object proxy, java.lang.reflect.Method method, Object[] args) throws Throwable {
736 				if (method.getName().equals("value")) {
737 					throw new RuntimeException("Test exception");
738 				}
739 				if (method.getName().equals("annotationType")) {
740 					return annotationType;
741 				}
742 				if (method.getName().equals("toString")) {
743 					return "@ToMapTestAnnotation";
744 				}
745 				if (method.getName().equals("hashCode")) {
746 					return 0;
747 				}
748 				if (method.getName().equals("equals")) {
749 					return false;
750 				}
751 				return method.getDefaultValue();
752 			}
753 		};
754 		
755 		var proxyAnnotation = (ToMapTestAnnotation)java.lang.reflect.Proxy.newProxyInstance(
756 			annotationType.getClassLoader(),

759 		);
760

763 		
764 		var map = ai.toMap();
