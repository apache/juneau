33  	private static void check(String expected, Object o) {
34  		assertEquals(expected, TO_STRING.apply(o));
35  	}
36  
37  	private static final Function<Object,String> TO_STRING = new Function<>() {
38  		@Override
39  		public String apply(Object t) {
40  			if (t == null)
41  				return null;
42  			if (t instanceof Iterable)
43  				return StreamSupport.stream(((Iterable<?>)t).spliterator(), false).map(this).collect(Collectors.joining(","));
44  			if (t instanceof ClassInfo)
45  				return ((ClassInfo)t).getNameSimple();
46  			if (t instanceof ConstructorInfo)
47  				return ((ConstructorInfo)t).getShortName();
48  			if (t instanceof Constructor)
49  				return ConstructorInfo.of((Constructor<?>)t).getShortName();
50  			return t.toString();
51  		}
52  	};
53  
54  	private static ConstructorInfo ofc(Class<?> c, Class<?>...pt) {
55  		try {
56  			return of(c.getConstructor(pt));
57  		} catch (NoSuchMethodException | SecurityException e) {
58  			fail(e.getLocalizedMessage());
59  		}
60  		return null;
61  	}

125 	@Test void isAccessible() {
126 		// Test isAccessible() before and after setAccessible()
127 		// Note: isAccessible() was added in Java 9, so behavior may vary
128 		
129 		// Before setAccessible(), private/protected constructors should not be accessible
130 		// (unless they're already accessible due to module system)
131 		var privateBefore = b_c3.isAccessible();
132 		
133 		// Make it accessible
134 		b_c3.setAccessible();
135 		
136 		// After setAccessible(), it should be accessible (if Java 9+)
137 		// If Java 8 or earlier, isAccessible() will return false
138 		var privateAfter = b_c3.isAccessible();
139 		
140 		// Verify the method doesn't throw and returns a boolean
141 		// The actual value depends on Java version, but it should be consistent
142 		assertTrue(privateAfter || !privateBefore, "After setAccessible(), isAccessible() should return true (Java 9+) or false (Java 8)");
143 		
144 		// Public constructors might already be accessible
145 		var publicAccessible = b_c1.isAccessible();
146 		// Should return a boolean (either true or false depending on Java version)
147 		assertNotNull(Boolean.valueOf(publicAccessible));
148 	}
