609 	@Target({ElementType.TYPE, ElementType.METHOD})
610 	@Retention(RetentionPolicy.RUNTIME)
611 	@interface FilterTestAnnotation {
612 		String value() default "";
613 	}
614 
615 	@FilterTestAnnotation("test")
616 	static class TestClass33 {}
617 
618 	@Test
619 	void c01_getAnnotationMethods_filter() throws Exception {
620 		// Test line 151: getAnnotationMethods filter
621 		// The filter checks: x.getParameterCount() == 0
622 		// 
623 		// Branch analysis:
624 		// 1. getParameterCount() == 0  passes filter (covered by normal annotation methods)
625 		// 2. getParameterCount() != 0  fails filter (not covered - impossible to test)
626 		//
627 		// The false branch (getParameterCount() != 0) cannot be tested because:
628 		// - Annotation member methods must have 0 parameters by Java language specification
629 		// - getDeclaredMethods() on an annotation class only returns annotation member methods
630 		// - Even synthetic methods on annotation classes would have 0 parameters
631 		//
632 		// To attempt to cover the false branch, we can inspect what methods are actually returned
