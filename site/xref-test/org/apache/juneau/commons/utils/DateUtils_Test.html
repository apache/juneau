99  	private static String toIso8601(Calendar c) {
100 		var sdf = new java.text.SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ssXXX");
101 		sdf.setTimeZone(c.getTimeZone());
102 		return sdf.format(c.getTime());
103 	}
