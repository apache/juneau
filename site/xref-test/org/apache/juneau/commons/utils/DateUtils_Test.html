99  	private static String toIso8601(Calendar c) {
100 		var sdf = new java.text.SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ssXXX");
101 		sdf.setTimeZone(c.getTimeZone());
102 		return sdf.format(c.getTime());
103 	}

388 		// Various ISO8601 formats
389 		var cal1 = parseIsoCalendar("2023");
390 		assertNotNull(cal1);
391 		assertEquals(2023, cal1.get(Calendar.YEAR));
392 
393 		var cal2 = parseIsoCalendar("2023-12");
394 		assertNotNull(cal2);
395 		assertEquals(2023, cal2.get(Calendar.YEAR));
396 		assertEquals(Calendar.DECEMBER, cal2.get(Calendar.MONTH));
397 
398 		var cal3 = parseIsoCalendar("2023-12-25");
399 		assertNotNull(cal3);
400 		assertEquals(2023, cal3.get(Calendar.YEAR));
401 		assertEquals(Calendar.DECEMBER, cal3.get(Calendar.MONTH));
402 		assertEquals(25, cal3.get(Calendar.DAY_OF_MONTH));
403 
404 		var cal4 = parseIsoCalendar("2023-12-25T14:30:00");
405 		assertNotNull(cal4);
406 		assertEquals(14, cal4.get(Calendar.HOUR_OF_DAY));
407 		assertEquals(30, cal4.get(Calendar.MINUTE));
408 		assertEquals(0, cal4.get(Calendar.SECOND));
409 
410 		// Should throw for invalid dates (DateTimeParseException is thrown by DateUtils, not IllegalArgumentException)
411 		assertThrows(Exception.class, () -> parseIsoCalendar("invalid"));
412 		assertThrows(Exception.class, () -> parseIsoCalendar("2023-13-25")); // Invalid month
413 
414 		// Test empty input - triggers code path
415 		assertNull(parseIsoCalendar(null));
416 		assertNull(parseIsoCalendar(""));
417 		assertNull(parseIsoCalendar("   "));
418 
419 		// Test with milliseconds (comma) - triggers code path
420 		var cal5 = parseIsoCalendar("2023-12-25T14:30:00,123");
421 		assertNotNull(cal5);
422 		assertEquals(14, cal5.get(Calendar.HOUR_OF_DAY));
423 		assertEquals(30, cal5.get(Calendar.MINUTE));
424 		assertEquals(0, cal5.get(Calendar.SECOND)); // Milliseconds trimmed
