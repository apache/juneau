30  	@Rest(defaultRequestAttributes={"p1:v1","p2:v2"})
31  	public static class A {}
32  
33  	@Rest(defaultRequestAttributes={"p2:v2a","p3:v3","p4:v4"})
34  	public static class A1 extends A {}

36  	@Rest
37  	public static class A2 extends A1 {
38  		@RestOp
39  		public JsonMap a1(RequestAttributes attrs) {
40  			// Should show {p1:'v1',p2:'v2a',p3:'v3',p4:'v4'} - NOSONAR
41  			return transform(attrs);
42  		}
43  		@RestOp(defaultRequestAttributes={"p4:v4a","p5:v5"})
44  		public JsonMap a2(RequestAttributes attrs, @HasQuery("override") boolean override) {
45  			// Should show {p1:'v1',p2:'v2a',p3:'v3',p4:'v4a',p5:'v5'} when override is false.
46  			// Should show {p1:'x',p2:'x',p3:'x',p4:'x',p5:'x'} when override is true.
47  			if (override) {
48  				attrs.set("p1", "x");
49  				attrs.set("p2", "x");
50  				attrs.set("p3", "x");
51  				attrs.set("p4", "x");
52  				attrs.set("p5", "x");
53  			}
54  			return transform(attrs);
55  		}

58  		public JsonMap b1(RequestAttributes attrs) {
59  			// Should show {p1:'v1',p2:'v2a',p3:'v3',p4:'v4'} - NOSONAR
60  			return transform(attrs);
61  		}
62  		@RestGet(defaultRequestAttributes={"p4:v4a","p5:v5"})
63  		public JsonMap b2(RequestAttributes attrs, @HasQuery("override") boolean override) {
64  			// Should show {p1:'v1',p2:'v2a',p3:'v3',p4:'v4a',p5:'v5'} when override is false.

74  		}

79  			return transform(attrs);
80  		}

132 
133 		private JsonMap transform(RequestAttributes attrs) {
134 			var m = new JsonMap();
135 			for (Map.Entry<String,Object> e : attrs.asMap().entrySet()) {
136 				if (e.getKey().startsWith("p"))
137 					m.put(e.getKey(), e.getValue());
138 			}
139 			return m;
140 		}
141 	}
