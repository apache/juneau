464 	void h01_classLevelAllParamDefaults() throws Exception {
465 		var h = MockRestClient.build(H.class);

468 		h.get("/query").run().assertContent("q=defaultQ");
469 		h.get("/query?q=custom").run().assertContent("q=custom");

472 		h.get("/header").run().assertContent("h=defaultHeader");
473 		h.get("/header").header("X-Custom", "custom").run().assertContent("h=custom");
474 
475 		// Test form data default
476 		h.post("/form").contentType("application/x-www-form-urlencoded").run().assertContent("f=defaultForm");
477 		h.post("/form").contentType("application/x-www-form-urlencoded").formData("f", "custom").run().assertContent("f=custom");
478 
479 		// Test method-level override
