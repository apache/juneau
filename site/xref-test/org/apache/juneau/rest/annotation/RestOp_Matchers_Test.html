13  package org.apache.juneau.rest.annotation;
14  
15  import static org.apache.juneau.common.internal.Utils.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.rest.matcher.*;
19  import org.apache.juneau.rest.mock.*;
20  import org.junit.jupiter.api.*;
21  
22  import jakarta.servlet.http.*;
23  
24  class RestOp_Matchers_Test extends TestBase {
25  
26  	//------------------------------------------------------------------------------------------------------------------
27  	// Overlapping matchers
28  	//------------------------------------------------------------------------------------------------------------------
29  
30  	@Rest
31  	public static class A {
32  		@RestOp(path="/one", matchers=A1.class)
33  		public String a() {
34  			return "OK-1a";
35  		}
36  		@RestGet(path="/one", matchers=A2.class)
37  		public String b() {
38  			return "OK-1b";
39  		}
40  		@RestGet(path="/one")
41  		public String c() {
42  			return "OK-1c";
43  		}
44  		@RestOp(path="/two")
45  		public String d() {
46  			return "OK-2a";
47  		}
48  		@RestGet(path="/two", matchers={A1.class, A2.class})
49  		public String e() {
50  			return "OK-2b";
51  		}
52  
53  		public static class A1 extends RestMatcher {
54  			@Override /* RestMatcher */
55  			public boolean matches(HttpServletRequest req) {
56  				return emptyIfNull(req.getQueryString()).contains("t1=1");
57  			}
58  		}
59  		public static class A2 extends RestMatcher {
60  			@Override /* RestMatcher */
61  			public boolean matches(HttpServletRequest req) {
62  				return emptyIfNull(req.getQueryString()).contains("t2=2");
63  			}
64  		}
65  	}
66  
67  	@Test void a01_overlapping() throws Exception {
68  		var a = MockRestClient.build(A.class);
69  		a.get("/one?t1=1").run().assertContent("OK-1a");
70  		a.get("/one?t2=2").run().assertContent("OK-1b");
71  		a.get("/one").run().assertContent("OK-1c");
72  		a.get("/two?t1=1").run().assertContent("OK-2b");
73  		a.get("/two?t2=2").run().assertContent("OK-2b");
74  		a.get("/two?t1=1&t2=2").run().assertContent("OK-2b");
75  		a.get("/two?tx=x").run().assertContent("OK-2a");
76  	}
77  }
