13  package org.apache.juneau.rest.annotation;
14  
15  import org.apache.juneau.*;
16  import org.apache.juneau.collections.*;
17  import org.apache.juneau.rest.*;
18  import org.apache.juneau.rest.mock.*;
19  import org.junit.jupiter.api.*;
20  
21  import jakarta.servlet.http.*;
22  
23  class RestStartCall_Test extends SimpleTestBase {
24  
25  	//------------------------------------------------------------------------------------------------------------------
26  	// @RestStartCall
27  	//------------------------------------------------------------------------------------------------------------------
28  
29  	@Rest
30  	public static class A extends A_Parent {
31  		private boolean start3Called;
32  		@RestStartCall
33  		public void start3() {
34  			start3Called = true;
35  		}
36  		@RestStartCall
37  		public void start4(HttpServletRequest req, HttpServletResponse res) {
38  			res.setHeader("start3-called", ""+start3Called);
39  			start3Called = false;
40  			if (res.getHeader("start4-called") != null)
41  				throw new IllegalArgumentException("start4 called multiple times.");
42  			res.setHeader("start4-called", "true");
43  		}
44  		@RestGet(path="/")
45  		public JsonMap a(RestRequest req, RestResponse res) {
46  			return JsonMap.create()
47  				.append("1", res.getHeader("start1-called"))
48  				.append("2", res.getHeader("start2-called"))
49  				.append("3", res.getHeader("start3-called"))
50  				.append("4", res.getHeader("start4-called"));
51  		}
52  	}
53  
54  	public static class A_Parent {
55  		private boolean start1Called;
56  		@RestStartCall
57  		public void start1() {
58  			start1Called = true;
59  		}
60  		@RestStartCall
61  		public void start2(HttpServletRequest req, HttpServletResponse res) {
62  			res.setHeader("start1-called", ""+start1Called);
63  			start1Called = false;
64  			if (res.getHeader("start2-called") != null)
65  				throw new IllegalArgumentException("start2 called multiple times.");
66  			res.setHeader("start2-called", "true");
67  		}
68  	}
69  
70  	@Test void a01_startCall() throws Exception {
71  		var a = MockRestClient.build(A.class);
72  		a.get("/").run().assertContent("{'1':'true','2':'true','3':'true','4':'true'}");
73  	}
74  }
