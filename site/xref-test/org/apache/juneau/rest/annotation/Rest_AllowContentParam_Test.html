13  package org.apache.juneau.rest.annotation;
14  
15  import org.apache.juneau.*;
16  import org.apache.juneau.collections.*;
17  import org.apache.juneau.http.annotation.*;
18  import org.apache.juneau.rest.mock.*;
19  import org.junit.jupiter.api.*;
20  
21  class Rest_AllowContentParam_Test extends SimpleTestBase {
22  
23  	//------------------------------------------------------------------------------------------------------------------
24  	// @Rest(disableBodyParam)
25  	//------------------------------------------------------------------------------------------------------------------
26  
27  	@Rest(disableContentParam="false")
28  	public static class A1 {
29  		@RestOp
30  		public JsonMap put(@Content JsonMap body) {
31  			return body;
32  		}
33  	}
34  	@Rest(disableContentParam="true")
35  	public static class A2 {
36  		@RestOp
37  		public JsonMap put(@Content JsonMap body) {
38  			return body;
39  		}
40  	}
41  	@Rest(disableContentParam="true")
42  	public static class A3 extends A1 {}
43  
44  	@Rest(disableContentParam="false")
45  	public static class A4 extends A2 {}
46  
47  	@Test void a01_basic() throws Exception {
48  		var a1 = MockRestClient.build(A1.class);
49  		a1.put("/", "{a:'b'}").run().assertContent("{a:'b'}");
50  		a1.put("/?content=(c=d)", "{a:'b'}").run().assertContent("{c:'d'}");
51  
52  		var a2 = MockRestClient.build(A2.class);
53  		a2.put("/", "{a:'b'}").run().assertContent("{a:'b'}");
54  		a2.put("/?content=(c=d)", "{a:'b'}").run().assertContent("{a:'b'}");
55  
56  		var a3 = MockRestClient.build(A3.class);
57  		a3.put("/", "{a:'b'}").run().assertContent("{a:'b'}");
58  		a3.put("/?content=(c=d)", "{a:'b'}").run().assertContent("{a:'b'}");
59  
60  		var a4 = MockRestClient.build(A4.class);
61  		a4.put("/", "{a:'b'}").run().assertContent("{a:'b'}");
62  		a4.put("/?content=(c=d)", "{a:'b'}").run().assertContent("{c:'d'}");
63  	}
64  }
