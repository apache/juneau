13  package org.apache.juneau.rest.annotation;
14  
15  import org.apache.juneau.*;
16  import org.apache.juneau.http.annotation.*;
17  import org.apache.juneau.rest.*;
18  import org.apache.juneau.rest.mock.*;
19  import org.junit.jupiter.api.*;
20  
21  class HasQuery_Test extends SimpleTestBase {
22  
23  	//------------------------------------------------------------------------------------------------------------------
24  	// Simple tests
25  	//------------------------------------------------------------------------------------------------------------------
26  
27  	@Rest
28  	public static class A {
29  		@RestGet
30  		public String a(RestRequest req, @HasQuery("p1") boolean p1, @HasQuery("p2") Boolean p2) {
31  			var q = req.getQueryParams();
32  			return "p1=["+p1+","+q.contains("p1")+"],p2=["+p2+","+q.contains("p2")+"]";
33  		}
34  		@RestPost
35  		public String b(RestRequest req, @HasQuery("p1") boolean p1, @HasQuery("p2") Boolean p2) {
36  			var q = req.getQueryParams();
37  			return "p1=["+p1+","+q.contains("p1")+"],p2=["+p2+","+q.contains("p2")+"]";
38  		}
39  	}
40  
41  	@Test void a01_basic() throws Exception {
42  		var a = MockRestClient.build(A.class);
43  
44  		a.get("/a?p1=p1&p2=2").run().assertContent("p1=[true,true],p2=[true,true]");
45  		a.get("/a?p1&p2").run().assertContent("p1=[true,true],p2=[true,true]");
46  		a.get("/a?p1=&p2=").run().assertContent("p1=[true,true],p2=[true,true]");
47  		a.get("/a").run().assertContent("p1=[false,false],p2=[false,false]");
48  		a.get("/a?p1").run().assertContent("p1=[true,true],p2=[false,false]");
49  		a.get("/a?p1=").run().assertContent("p1=[true,true],p2=[false,false]");
50  		a.get("/a?p2").run().assertContent("p1=[false,false],p2=[true,true]");
51  		a.get("/a?p2=").run().assertContent("p1=[false,false],p2=[true,true]");
52  		a.get("/a?p1=foo&p2").run().assertContent("p1=[true,true],p2=[true,true]");
53  		a.get("/a?p1&p2=1").run().assertContent("p1=[true,true],p2=[true,true]");
54  		String x1 = "a%2Fb%25c%3Dd+e"; // [x/y%z=a+b]
55  		a.get("/a?p1="+x1+"&p2=1").run().assertContent("p1=[true,true],p2=[true,true]");
56  
57  		a.post("/b?p1=p1&p2=2", null).run().assertContent("p1=[true,true],p2=[true,true]");
58  		a.post("/b?p1&p2", null).run().assertContent("p1=[true,true],p2=[true,true]");
59  		a.post("/b?p1=&p2=", null).run().assertContent("p1=[true,true],p2=[true,true]");
60  		a.post("/b", null).run().assertContent("p1=[false,false],p2=[false,false]");
61  		a.post("/b?p1", null).run().assertContent("p1=[true,true],p2=[false,false]");
62  		a.post("/b?p1=", null).run().assertContent("p1=[true,true],p2=[false,false]");
63  		a.post("/b?p2", null).run().assertContent("p1=[false,false],p2=[true,true]");
64  		a.post("/b?p2=", null).run().assertContent("p1=[false,false],p2=[true,true]");
65  		a.post("/b?p1=foo&p2", null).run().assertContent("p1=[true,true],p2=[true,true]");
66  		a.post("/b?p1&p2=1", null).run().assertContent("p1=[true,true],p2=[true,true]");
67  		String x2 = "a%2Fb%25c%3Dd+e"; // [x/y%z=a+b]
68  		a.post("/b?p1="+x2+"&p2=1", null).run().assertContent("p1=[true,true],p2=[true,true]");
69  	}
70  }
