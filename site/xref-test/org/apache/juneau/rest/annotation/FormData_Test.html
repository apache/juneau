37  	@Rest(parsers=UrlEncodingParser.class)
38  	public static class A {
39  		@RestPost
40  		public String a(RestRequest req, @FormData("p1") @Schema(allowEmptyValue=true) String p1, @FormData("p2") @Schema(allowEmptyValue=true) int p2) throws Exception {
41  			var f = req.getFormParams();
42  			return "p1=["+p1+","+f.get("p1").orElse(null)+","+f.get("p1").asString().orElse(null)+"],p2=["+p2+","+f.get("p2").orElse(null)+","+f.get("p2").as(int.class).orElse(null)+"]";
43  		}
44  	}

46  	@Test void a01_basic() throws Exception {
47  		var a = MockRestClient.build(A.class);
48  		a.post("/a", "p1=p1&p2=2").contentType("application/x-www-form-urlencoded").run().assertContent("p1=[p1,p1,p1],p2=[2,2,2]");
49  		a.post("/a", "p1&p2").contentType("application/x-www-form-urlencoded").run().assertContent("p1=[null,null,null],p2=[0,null,0]");
50  		a.post("/a", "p1=&p2=").contentType("application/x-www-form-urlencoded").run().assertContent("p1=[,,],p2=[0,,0]");
51  		a.post("/a", "").contentType("application/x-www-form-urlencoded").run().assertContent("p1=[null,null,null],p2=[0,null,0]");
52  		a.post("/a", "p1").contentType("application/x-www-form-urlencoded").run().assertContent("p1=[null,null,null],p2=[0,null,0]");
53  		a.post("/a", "p1=").contentType("application/x-www-form-urlencoded").run().assertContent("p1=[,,],p2=[0,null,0]");
54  		a.post("/a", "p2").contentType("application/x-www-form-urlencoded").run().assertContent("p1=[null,null,null],p2=[0,null,0]");
55  		a.post("/a", "p2=").contentType("application/x-www-form-urlencoded").run().assertContent("p1=[null,null,null],p2=[0,,0]");
56  		a.post("/a", "p1=foo&p2").contentType("application/x-www-form-urlencoded").run().assertContent("p1=[foo,foo,foo],p2=[0,null,0]");
57  		a.post("/a", "p1&p2=1").contentType("application/x-www-form-urlencoded").run().assertContent("p1=[null,null,null],p2=[1,1,1]");
58  		String x = "a%2Fb%25c%3Dd+e"; // [x/y%z=a+b]
59  		a.post("/a", "p1="+x+"&p2=1").contentType("application/x-www-form-urlencoded").run().assertContent("p1=[a/b%c=d e,a/b%c=d e,a/b%c=d e],p2=[1,1,1]");
60  	}

95  	public static class C {
96  		@RestPost(defaultRequestFormData={"f1:1","f2=2"," f3 : 3 "})
97  		public JsonMap a(RequestFormParams formData) {
98  			return JsonMap.create()
99  				.append("f1", formData.get("f1").asString())
100 				.append("f2", formData.get("f2").asString())
101 				.append("f3", formData.get("f3").asString());
102 		}
103 		@RestPost
104 		public JsonMap b(@FormData("f1") String f1, @FormData("f2") String f2, @FormData("f3") String f3) {
105 			return JsonMap.create()
106 				.append("f1", f1)
107 				.append("f2", f2)
108 				.append("f3", f3);
109 		}
110 		@RestPost
111 		public JsonMap c(@FormData("f1") @Schema(_default="1") String f1, @FormData("f2") @Schema(_default="2") String f2, @FormData("f3") @Schema(_default="3") String f3) {
112 			return JsonMap.create()
113 				.append("f1", f1)
114 				.append("f2", f2)
115 				.append("f3", f3);
116 		}
117 		@RestPost(defaultRequestFormData={"f1:1","f2=2"," f3 : 3 "})
118 		public JsonMap d(@FormData("f1") @Schema(_default="4") String f1, @FormData("f2") @Schema(_default="5") String f2, @FormData("f3") @Schema(_default="6") String f3) {
119 			return JsonMap.create()
120 				.append("f1", f1)
121 				.append("f2", f2)
122 				.append("f3", f3);
123 		}
124 	}

126 	@Test void c01_defaultFormData() throws Exception {
127 		var c = MockRestClient.build(C.class);
128 
129 		c.post("/a").contentType("application/x-www-form-urlencoded").run().assertContent("{f1:'1',f2:'2',f3:'3'}");
130 		c.post("/a").contentType("application/x-www-form-urlencoded").formData("f1",4).formData("f2",5).formData("f3",6).run().assertContent("{f1:'4',f2:'5',f3:'6'}");
131 
132 		c.post("/b").contentType("application/x-www-form-urlencoded").run().assertContent("{f1:null,f2:null,f3:null}");
133 		c.post("/b").contentType("application/x-www-form-urlencoded").formData("f1",4).formData("f2",5).formData("f3",6).run().assertContent("{f1:'4',f2:'5',f3:'6'}");
134 
135 		c.post("/c").contentType("application/x-www-form-urlencoded").run().assertContent("{f1:'1',f2:'2',f3:'3'}");
136 		c.post("/c").contentType("application/x-www-form-urlencoded").formData("f1",4).formData("f2",5).formData("f3",6).run().assertContent("{f1:'4',f2:'5',f3:'6'}");
137 
138 		c.post("/d").contentType("application/x-www-form-urlencoded").run().assertContent("{f1:'1',f2:'2',f3:'3'}");

170 		var d = MockRestClient.create(D.class).accept("application/json").contentType("application/x-www-form-urlencoded").build();
171 
172 		d.post("/a", "f1=123")
173 			.run()
174 			.assertStatus(200)
175 			.assertContent("123");
176 		d.post("/a", "null")
177 			.run()
178 			.assertStatus(200)
179 			.assertContent("null");
180 
181 		d.post("/b", "f1=a=1,b=foo")
182 			.run()
183 			.assertStatus(200)
184 			.assertContent("{a:1,b:'foo'}");
185 		d.post("/b", "null")
186 			.run()
187 			.assertStatus(200)
188 			.assertContent("null");
189 
190 		d.post("/c", "f1=@((a=1,b=foo))")
191 			.run()
192 			.assertStatus(200)
193 			.assertContent("[{a:1,b:'foo'}]");
194 		d.post("/c", "null")
195 			.run()
196 			.assertStatus(200)
197 			.assertContent("null");
198 
199 		d.post("/d", "f1=@((a=1,b=foo))")
