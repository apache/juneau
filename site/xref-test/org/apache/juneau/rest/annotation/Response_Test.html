142 		@RestGet
143 		public byte[] b() {
144 			return "foo".getBytes();
145 		}
146 		@RestGet
147 		public D1 c() {
148 			return new D1();
149 		}
150 		@RestGet
151 		public D2 d() {
152 			return new D2();
153 		}

156 			throw new D3();
157 		}
158 		@RestGet
159 		public String f() throws D4 {
160 			throw new D4();
161 		}
162 		@RestGet

164 			value.set(new String[]{"foo","bar"});
165 		}
166 		@RestGet
