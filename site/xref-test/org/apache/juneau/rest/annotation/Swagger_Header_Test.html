31  	@Rest
32  	public static class A {
33  
34  		@Header(
35  			name="H"
36  		)
37  		@Schema(description={"a","b"}, type="string")
38  		public static class A1 {
39  			public A1(String x) {}
40  		}
41  		@RestGet
42  		public void a(A1 h) { /* no-op */ }
43  
44  		@Header(
45  			name="H",
46  			schema=@Schema(description="a\nb",type="string")
47  		)
48  		public static class A2 {
49  			public A2(String x) {}
50  		}
51  		@RestPut
52  		public void b(A2 h) { /* no-op */ }
53  
54  		@Header(
55  			name="H",
56  			schema=@Schema(description="b\nc",type="string")
57  		)
58  		@Schema(description={"a","b"}, type="string")
59  		public static class A3 {
60  			public A3(String x) {}
61  		}
62  		@RestPost
63  		public void c(A3 h) { /* no-op */ }
64  	}
65  
66  	@Test void a01_fromPojo() {
67  		var s = getSwagger(A.class);
68  		var x = s.getParameterInfo("/a","get","header","H");
69  
70  		assertEquals("a\nb", x.getDescription());
71  		assertEquals("string", x.getType());
72  		x = s.getParameterInfo("/b","put","header","H");
73  		assertEquals("a\nb", x.getDescription());
74  		assertEquals("string", x.getType());
75  		x = s.getParameterInfo("/c","post","header","H");
76  		assertEquals("b\nc", x.getDescription());
77  		assertEquals("string", x.getType());	}

79  	@Rest
80  	public static class B {
81  
82  		@Header(name="H")
83  		public static class B1 {}
84  		@RestGet
85  		public void a(B1 h) { /* no-op */ }
86  
87  		@Header(name="H")
88  		public static class B2 {
89  			public String f1;
90  		}
91  		@RestPut
92  		public void b(B2 b) { /* no-op */ }
93  
94  		@Header(name="H")
95  		public static class B3 extends LinkedList<String> {
96  			private static final long serialVersionUID = 1L;
97  		}
98  		@RestPost
99  		public void c(B3 b) { /* no-op */ }
100 
101 		@Header(name="H")
102 		public static class B4 {}
103 		@RestDelete
104 		public void d(B4 b) { /* no-op */ }
105 	}
106 
107 	@Test void b01_schemaFromPojo() {
108 		var s = getSwagger(B.class);
109 		var x = s.getParameterInfo("/a","get","header","H");
110 
111 		assertBean(x, "in,name,type", "header,H,string");
112 
113 		x = s.getParameterInfo("/b","put","header","H");
114 		assertBean(x, "in,name,type,schema{properties{f1{type}}}", "header,H,object,{{{string}}}");
115 
116 		x = s.getParameterInfo("/c","post","header","H");
117 		assertBean(x, "in,name,type,items{type}", "header,H,array,{string}");
118 
119 		x = s.getParameterInfo("/d","delete","header","H");
120 		assertBean(x, "in,name,type", "header,H,string");
121 	}
