13  package org.apache.juneau.rest.annotation;
14  
15  import static org.apache.juneau.TestUtils.*;
16  import static org.junit.jupiter.api.Assertions.*;
17  
18  import org.apache.juneau.*;
19  import org.apache.juneau.annotation.*;
20  import org.junit.jupiter.api.*;
21  
22  class OpSwaggerAnnotation_Test extends SimpleTestBase {
23  
24  	//------------------------------------------------------------------------------------------------------------------
25  	// Basic tests
26  	//------------------------------------------------------------------------------------------------------------------
27  
28  	OpSwagger a1 = OpSwaggerAnnotation.create()
29  		.consumes("a")
30  		.deprecated("b")
31  		.description("c")
32  		.externalDocs(ExternalDocsAnnotation.DEFAULT)
33  		.operationId("d")
34  		.parameters("e")
35  		.produces("f")
36  		.responses("g")
37  		.schemes("h")
38  		.summary("i")
39  		.tags("j")
40  		.value("k")
41  		.build();
42  
43  	OpSwagger a2 = OpSwaggerAnnotation.create()
44  		.consumes("a")
45  		.deprecated("b")
46  		.description("c")
47  		.externalDocs(ExternalDocsAnnotation.DEFAULT)
48  		.operationId("d")
49  		.parameters("e")
50  		.produces("f")
51  		.responses("g")
52  		.schemes("h")
53  		.summary("i")
54  		.tags("j")
55  		.value("k")
56  		.build();
57  
58  	@Test void a01_basic() {
59  		assertBean(a1,
60  			"consumes,deprecated,description,externalDocs{description,url},operationId,parameters,produces,responses,schemes,summary,tags,value",
61  			"[a],b,[c],{[],},d,[e],[f],[g],[h],[i],[j],[k]");
62  	}
63  
64  	@Test void a02_testEquivalency() {
65  		assertEquals(a2, a1);
66  		assertNotEqualsAny(a1.hashCode(), 0, -1);
67  		assertEquals(a1.hashCode(), a2.hashCode());
68  	}
69  
70  	//------------------------------------------------------------------------------------------------------------------
71  	// PropertyStore equivalency.
72  	//------------------------------------------------------------------------------------------------------------------
73  
74  	@Test void b01_testEquivalencyInPropertyStores() {
75  		var bc1 = BeanContext.create().annotations(a1).build();
76  		var bc2 = BeanContext.create().annotations(a2).build();
77  		assertSame(bc1, bc2);
78  	}
79  
80  	//------------------------------------------------------------------------------------------------------------------
81  	// Other methods.
82  	//------------------------------------------------------------------------------------------------------------------
83  
84  	//------------------------------------------------------------------------------------------------------------------
85  	// Comparison with declared annotations.
86  	//------------------------------------------------------------------------------------------------------------------
87  
88  	@OpSwagger(
89  		consumes="a",
90  		deprecated="b",
91  		description="c",
92  		externalDocs=@ExternalDocs,
93  		operationId="d",
94  		parameters="e",
95  		produces="f",
96  		responses="g",
97  		schemes="h",
98  		summary="i",
99  		tags="j",
100 		value="k"
101 	)
102 	public static class D1 {}
103 	OpSwagger d1 = D1.class.getAnnotationsByType(OpSwagger.class)[0];
104 
105 	@OpSwagger(
106 		consumes="a",
107 		deprecated="b",
108 		description="c",
109 		externalDocs=@ExternalDocs,
110 		operationId="d",
111 		parameters="e",
112 		produces="f",
113 		responses="g",
114 		schemes="h",
115 		summary="i",
116 		tags="j",
117 		value="k"
118 	)
119 	public static class D2 {}
120 	OpSwagger d2 = D2.class.getAnnotationsByType(OpSwagger.class)[0];
121 
122 	@Test void d01_comparisonWithDeclarativeAnnotations() {
123 		assertEqualsAll(a1, d1, d2);
124 		assertNotEqualsAny(a1.hashCode(), 0, -1);
125 		assertEqualsAll(a1.hashCode(), d1.hashCode(), d2.hashCode());
126 	}
127 }
