29  class RestClient_Config_Context_Test extends TestBase {
30  
31  	@Rest
32  	public static class A extends BasicRestObject {
33  		@RestPost
34  		public Reader echoBody(org.apache.juneau.rest.RestRequest req) throws IOException {
35  			return req.getContent().getReader();
36  		}
37  	}
38  
39  	public static class A2 {
40  		public String foo;
41  		@Override
42  		public String toString() {
43  			return foo;
44  		}
45  		public static A2 fromString(String s) {
46  			var p2 = new A2();
47  			p2.foo = s;
48  			return p2;
49  		}
50  	}
51  
52  	@Test void a02_addToStringObject() throws Exception {
53  		client().notBeanClasses(A2.class).build().post("/echoBody",A2.fromString("bar")).run().cacheContent().assertContent("'bar'").getContent().as(A2.class);
54  	}
55  
56  	public static class A3a {
57  		public int foo;
58  		static A3a get() {
59  			var x = new A3a();
60  			x.foo = 1;
61  			return x;
62  		}
63  	}
64  
65  	public static class A3b extends ObjectSwap<A3a,Integer> {
66  		@Override
67  		public Integer swap(BeanSession session, A3a o) { return o.foo; }
68  		@Override
69  		public A3a unswap(BeanSession session, Integer f, ClassMeta<?> hint) {return A3a.get(); }
70  	}
71  
72  	@Test void a03_appendToStringObject() throws Exception {
73  		var x = client().swaps(A3b.class).build().post("/echoBody",A3a.get()).run().cacheContent().assertContent("1").getContent().as(A3a.class);
74  		assertEquals(1,x.foo);
75  	}
76  
77  	@Test void a04_prependToStringObject() throws Exception {
78  		var x = client().swaps(A3b.class).build().post("/echoBody",A3a.get()).run().cacheContent().assertContent("1").getContent().as(A3a.class);
79  		assertEquals(1,x.foo);
80  	}
81  
82  	public static class A6a {
83  		public int foo,bar,baz;
84  		static A6a get() {
85  			var x = new A6a();
86  			x.foo = 1;
87  			x.bar = 2;
88  			x.baz = 3;
89  			return x;
90  		}
91  	}
92  
93  	@org.apache.juneau.annotation.Bean(sort=true,on="A6a")
94  	public static class A6b {}
95  
96  	@BeanConfig(sortProperties="true")
97  	public static class A6c {}
98  
99  	public static class A6d {
100 		@BeanConfig(sortProperties="true")
101 		public void foo() { /* no-op */ }
102 	}
103 
104 	@Test void a06_applyAnnotations() throws Exception {
105 		new A6b();
106 		new A6c();
107 		new A6d().foo();
108 		client().applyAnnotations(A6b.class).build().post("/echoBody",A6a.get()).run().cacheContent().assertContent("{bar:2,baz:3,foo:1}").getContent().as(A6a.class);
109 		client().applyAnnotations(A6c.class).build().post("/echoBody",A6a.get()).run().cacheContent().assertContent("{bar:2,baz:3,foo:1}").getContent().as(A6a.class);
110 		client().applyAnnotations(A6d.class.getMethod("foo")).build().post("/echoBody",A6a.get()).run().cacheContent().assertContent("{bar:2,baz:3,foo:1}").getContent().as(A6a.class);
111 		var al = AnnotationWorkList.of(ClassInfo.of(A6c.class).getAnnotationList(CONTEXT_APPLY_FILTER));
112 		client().apply(al).build().post("/echoBody",A6a.get()).run().cacheContent().assertContent("{bar:2,baz:3,foo:1}").getContent().as(A6a.class);
113 	}
114 
115 	@Test void a09_annotations() throws Exception {
116 		client().annotations(BeanAnnotation.create(A6a.class).sort(true).build()).build().post("/echoBody",A6a.get()).run().cacheContent().assertContent("{bar:2,baz:3,foo:1}").getContent().as(A6a.class);
117 	}
