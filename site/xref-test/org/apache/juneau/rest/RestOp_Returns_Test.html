117 	@Rest
118 	public static class B {
119 		@RestGet
120 		public HttpResource a() {
121 			return stringResource("foo");
122 		}
123 		@RestGet
124 		public HttpResource b() {
125 			return readerResource(new StringReader("foo")).setHeader("Foo", "Bar");
126 		}
127 		@RestGet
128 		public HttpResource c() {
129 			return readerResource(new StringReader("foo")).setContentType("application/json");
130 		}
131 		@RestGet
132 		public HttpResource d(RestRequest req) {
133 			return stringResource(()->req.getVarResolverSession().resolve("$RQ{foo}"));
134 		}
135 		@RestGet
136 		public HttpResource e() {
137 			return streamResource(inputStream("foo"));
138 		}
139 		@RestGet
140 		public HttpResource f() {
141 			return readerResource(reader("foo"));
142 		}
143 	}

145 	@Test void b01_BasicHttpResource() throws Exception {
146 		var b = MockRestClient.build(B.class);
147 		b.get("/a")
148 			.run()
149 			.assertContent("foo");
150 		b.get("/b")
151 			.run()
152 			.assertHeader("Foo").is("Bar");
153 		b.get("/c")
154 			.run()
155 			.assertHeader("Content-Type").is("application/json");
156 		b.get("/d?foo=bar")
157 			.run()
158 			.assertContent("bar");
159 		b.get("/e")
160 			.run()
161 			.assertContent("foo");
162 		b.get("/f")
163 			.run()
164 			.assertContent("foo");
165 	}
