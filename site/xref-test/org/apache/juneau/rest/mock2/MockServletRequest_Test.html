13  package org.apache.juneau.rest.mock2;
14  
15  import static org.apache.juneau.TestUtils.*;
16  import static org.junit.jupiter.api.Assertions.*;
17  
18  import org.apache.juneau.*;
19  import org.apache.juneau.rest.mock.*;
20  import org.junit.jupiter.api.*;
21  
22  class MockServletRequest_Test extends TestBase {
23  
24  	//-----------------------------------------------------------------------------------------------------------------
25  	// URIs
26  	//-----------------------------------------------------------------------------------------------------------------
27  
28  	@Test void a01_uris_basic() {
29  		var req = MockServletRequest.create("GET", "/foo");
30  		assertBean(
31  			req,
32  			"contextPath,pathInfo,pathTranslated,queryString,requestURI,requestURL,servletPath",
33  			",/foo,/mock-path/foo,<null>,/foo,/foo,"
34  		);
35  	}
36  
37  	@Test void a02_uris_full() {
38  		var req = MockServletRequest.create("GET", "http://localhost:8080/foo?bar=baz#quz");
39  		assertBean(
40  			req,
41  			"contextPath,pathInfo,pathTranslated,queryString,requestURI,requestURL,servletPath",
42  			",/foo,/mock-path/foo,bar=baz,/foo,http://localhost:8080/foo,"
43  		);
44  	}
45  
46  	@Test void a03_uris_full2() {
47  		var req = MockServletRequest.create("GET", "http://localhost:8080/foo/bar/baz?bar=baz#quz");
48  		assertBean(
49  			req,
50  			"contextPath,pathInfo,pathTranslated,queryString,requestURI,requestURL,servletPath",
51  			",/foo/bar/baz,/mock-path/foo/bar/baz,bar=baz,/foo/bar/baz,http://localhost:8080/foo/bar/baz,"
52  		);
53  	}
54  
55  	@Test void a04_uris_contextPath() {
56  		var req = MockServletRequest.create("GET", "http://localhost:8080/foo/bar/baz?bar=baz#quz").contextPath("/foo");
57  		assertBean(
58  			req,
59  			"contextPath,pathInfo,pathTranslated,queryString,requestURI,requestURL,servletPath",
60  			"/foo,/bar/baz,/mock-path/bar/baz,bar=baz,/foo/bar/baz,http://localhost:8080/foo/bar/baz,"
61  		);
62  	}
63  
64  	@Test void a05_uris_servletPath() {
65  		assertBean(
66  			MockServletRequest.create("GET", "http://localhost:8080/foo/bar/baz?bar=baz#quz").servletPath("/foo"),
67  			"contextPath,pathInfo,pathTranslated,queryString,requestURI,requestURL,servletPath",
68  			",/bar/baz,/mock-path/bar/baz,bar=baz,/foo/bar/baz,http://localhost:8080/foo/bar/baz,/foo"
69  		);
70  	}
71  
72  	//-----------------------------------------------------------------------------------------------------------------
73  	// Query strings
74  	//-----------------------------------------------------------------------------------------------------------------
75  
76  	@Test void b01_query_basic() {
77  		var req = MockServletRequest.create("GET", "/foo?bar=baz&bing=qux");
78  		assertBean(
79  			req,
80  			"queryString,parameterMap,parameterNames",
81  			"bar=baz&bing=qux,{bar=[baz],bing=[qux]},[bar,bing]"
82  		);
83  		assertEquals("baz", req.getParameter("bar"));
84  		assertList(req.getParameterValues("bar"), "baz");
85  	}
86  
87  	@Test void b02_query_multivalues() {
88  		var req = MockServletRequest.create("GET", "/foo?bar=baz&bar=bing");
89  		assertBean(
90  			req,
91  			"queryString,parameterMap,parameterNames",
92  			"bar=baz&bar=bing,{bar=[baz,bing]},[bar]"
93  		);
94  		assertEquals("baz", req.getParameter("bar"));
95  		assertList(req.getParameterValues("bar"), "baz", "bing");
96  	}
97  }
