286 		var b = MockRestClient.build(B1.class);
287 
288 		b.get("/accept").accept("text/foo").run().assertContent("text/foo");
289 		b.get("/accept").accept("text/foo+bar").run().assertContent("text/foo+bar");
290 		b.get("/accept").accept("text/*").run().assertContent("text/*");
291 		b.get("/accept").accept("*/foo").run().assertContent("*/foo");
292 		b.get("/accept").accept("text/foo;q=1.0").run().assertContent("text/foo;q=1.0");
293 		b.get("/accept").accept("text/foo;q=0.9").run().assertContent("text/foo;q=0.9");
294 		b.get("/accept").accept("text/foo;x=X;q=0.9;y=Y").run().assertContent("text/foo;x=X;q=0.9;y=Y");
295 		b.get("/accept?Accept=text/foo").run().assertContent("text/foo");
296 		b.get("/acceptCharset").acceptCharset("UTF-8").run().assertContent("UTF-8");

302 		b.get("/contentLength?Content-Length=0").run().assertContent("0");
303 		b.get("/contentType").contentType("text/foo").run().assertContent("text/foo");
304 		b.get("/contentType?Content-Type=text/foo").run().assertContent("text/foo");

341 	public static class B2a implements RestOpArg {
342 
343 		public static B2a create(ParamInfo pi) {
344 			if (pi.isType(B2b.class))
345 				return new B2a();
346 			return null;
347 		}
348 
349 		@Override
350 		public Object resolve(RestOpSession opSession) throws Exception {
351 			return new B2b(opSession.getRequest().getHeaderParam("Custom").orElse(null));
352 		}
353 	}
354 
355 	public static class B2b {
356 		public String value;
357 		public B2b(String value) {
358 			this.value = value;
359 		}
360 		@Override
361 		public String toString() {
362 			return value;
363 		}
364 	}
