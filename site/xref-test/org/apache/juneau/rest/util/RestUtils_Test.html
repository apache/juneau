57  	@Test void c01_testGetServletURI() {
58  		var e = "http://hostname";
59  		var sp = "";
60  		var cp = "";
61  
62  		for (var s : a("http://hostname", "http://hostname/foo", "http://hostname?foo", "http://hostname/?foo"))
63  			assertEquals(e, trimPathInfo(new StringBuffer(s), cp, sp).toString());
64  
65  		for (var s : a("http:/hostname?foo")) {
66  			assertThrows(Exception.class, ()->trimPathInfo(new StringBuffer(s), "", ""));
67  		}
68  
69  		e = "http://hostname";
70  		sp = "/";
71  		cp = "/";
72  
73  		for (var s : a("http://hostname", "http://hostname/foo", "http://hostname?foo", "http://hostname/?foo"))
74  			assertEquals(e, trimPathInfo(new StringBuffer(s), cp, sp).toString());
75  
76  		e = "http://hostname/foo";
77  		sp = "/foo";
78  		cp = "/";
79

82  
83  		for (var s : a("http://hostname/foo2", "http://hostname/fo2", "http://hostname?foo", "http://hostname/fo?bar", "http:/hostname/foo")) {
84  			assertThrows(Exception.class, ()->trimPathInfo(new StringBuffer(s), "/", "/foo"));
85  		}
86  
87  		e = "http://hostname/foo/bar";
88  		sp = "/foo/bar";

141 	// Test URL-encoded strings parsed into plain-text values using UrlEncodingParser.parseIntoSimpleMap().
142 	//------------------------------------------------------------------------------------------------------------------
143 
144 	@Test void g01_testParseIntoSimpleMap() {
145 		var s = "?f1=,()=&f2a=$b(true)&f2b=true&f3a=$n(123)&f3b=123&f4=$s(foo)";
146 		var m = parseQuery(s);
147 		assertEquals(",()=", m.get("f1")[0]);
148 		assertEquals("$b(true)", m.get("f2a")[0]);
149 		assertEquals("true", m.get("f2b")[0]);
150 		assertEquals("$n(123)", m.get("f3a")[0]);
151 		assertEquals("123", m.get("f3b")[0]);
152 		assertEquals("$s(foo)", m.get("f4")[0]);
153 
154 		s = "f1=v1&=";
155 		m = parseQuery(s);
156 		assertEquals("v1", m.get("f1")[0]);
157 		assertEquals("", m.get("")[0]);
158 
159 		s = "f1=v1&f2&f3";
160 		m = parseQuery(s);
161 		assertEquals("v1", m.get("f1")[0]);
162 		assertTrue(m.containsKey("f2"));
163 		assertTrue(m.containsKey("f3"));
164 		assertNull(m.get("f2"));
165 		assertNull(m.get("f3"));
166 	}
