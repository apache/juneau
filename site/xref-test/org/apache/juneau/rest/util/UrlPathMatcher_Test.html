27  	private void check(UrlPathMatcher p, String path, String expected) {
28  		assertString(expected, p.match(UrlPath.of(path)));
29  	}
30  
31  	private void shouldNotMatch(UrlPathMatcher p, String...paths) {
32  		for (String path : paths)
33  			assertNull(p.match(UrlPath.of(path)));
34  	}

40  	@Test void a01_comparision() {
41  		var l = new LinkedList<UrlPathMatcher>();
42  
43  		l.add(of(""));
44  		l.add(of("*"));
45  		l.add(of("/"));
46  		l.add(of("/*"));
47  		l.add(of("/foo"));
48  		l.add(of("/foo/*"));
49  		l.add(of("/foo/bar"));
50  		l.add(of("/foo/bar/*"));
51  		l.add(of("/foo/{id}"));
52  		l.add(of("/foo/{id}/*"));
53  		l.add(of("/foo/{id}/bar"));
54  		l.add(of("/foo/{id}/bar/*"));
55  		l.add(of("foo.txt"));
56  		l.add(of("*.txt"));
57  		l.add(of("foo.*"));
58  
59  		Collections.sort(l);
60  		assertEquals("['foo.txt','foo.*','*.txt','/foo/bar','/foo/bar/*','/foo/{id}/bar','/foo/{id}/bar/*','/foo/{id}','/foo/{id}/*','/foo','/foo/*','/','/*','','*']", Json5Serializer.DEFAULT.toString(l));
61  	}
