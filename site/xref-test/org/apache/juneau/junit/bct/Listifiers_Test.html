69  			assertList(treeResult, "a", "m", "z"); // Natural order
70  		}

297 		@Test
298 		void f01_listifyMap() {
299 			var listifier = Listifiers.mapListifier();
300 			var input = Map.of("z", "value1", "a", "value2"); // Unordered input
301 			var result = listifier.apply(null, input);
302 
303 			assertEquals(2, result.size());
304 			// Result should contain Map.Entry objects
305 			assertTrue(result.stream().allMatch(obj -> obj instanceof Map.Entry));
306 
307 			// TreeMap conversion ensures natural key ordering
308 			var entries = result.stream()
309 				.map(obj -> (Map.Entry<?, ?>) obj)
310 				.toList();
311 			assertEquals("a", entries.get(0).getKey());
312 			assertEquals("z", entries.get(1).getKey());
313 		}
314 
315 		@Test
316 		void f01a_listifyMapTypes() {
317 			var listifier = Listifiers.mapListifier();
318 
319 			// HashMap (unordered) -> converted to TreeMap for natural key ordering
320 			var hashMap = new HashMap<String, String>();
321 			hashMap.put("z", "value1");
322 			hashMap.put("a", "value2");
323 			hashMap.put("m", "value3");
324 			var hashResult = listifier.apply(null, hashMap);
325 			var hashEntries = hashResult.stream()
326 				.map(obj -> (Map.Entry<?, ?>) obj)
327 				.toList();
328 			assertEquals("a", hashEntries.get(0).getKey());
329 			assertEquals("m", hashEntries.get(1).getKey());
330 			assertEquals("z", hashEntries.get(2).getKey());
331 
332 			// LinkedHashMap preserves insertion order
333 			var linkedMap = new LinkedHashMap<String, String>();
334 			linkedMap.put("z", "value1");
335 			linkedMap.put("a", "value2");
336 			linkedMap.put("m", "value3");
337 			var linkedResult = listifier.apply(null, linkedMap);
338 			var linkedEntries = linkedResult.stream()
339 				.map(obj -> (Map.Entry<?, ?>) obj)
340 				.toList();
341 			assertEquals("z", linkedEntries.get(0).getKey()); // Insertion order
342 			assertEquals("a", linkedEntries.get(1).getKey());
343 			assertEquals("m", linkedEntries.get(2).getKey());
344 
345 			// TreeMap already sorted, preserves its order
346 			var treeMap = new TreeMap<String, String>();
347 			treeMap.put("z", "value1");
348 			treeMap.put("a", "value2");
349 			treeMap.put("m", "value3");
350 			var treeResult = listifier.apply(null, treeMap);
351 			var treeEntries = treeResult.stream()
352 				.map(obj -> (Map.Entry<?, ?>) obj)
353 				.toList();
354 			assertEquals("a", treeEntries.get(0).getKey()); // Natural order
355 			assertEquals("m", treeEntries.get(1).getKey());
356 			assertEquals("z", treeEntries.get(2).getKey());
357 		}

368 		@Test
369 		void f03_listifyMapWithNullValues() {
370 			var listifier = Listifiers.mapListifier();
371 			var input = new HashMap<String, String>();
372 			input.put("key1", "value1");
373 			input.put("key2", null);
374 			var result = listifier.apply(null, input);
375 
376 			assertEquals(2, result.size());
377 			// Check that we have the expected entries
378 			var hasKey1Entry = result.stream()
379 				.filter(obj -> obj instanceof Map.Entry)
380 				.map(obj -> (Map.Entry<?, ?>) obj)
381 				.anyMatch(entry -> "key1".equals(entry.getKey()) && "value1".equals(entry.getValue()));
382 			assertTrue(hasKey1Entry);
383 
384 			var hasKey2Entry = result.stream()
385 				.filter(obj -> obj instanceof Map.Entry)
386 				.map(obj -> (Map.Entry<?, ?>) obj)
387 				.anyMatch(entry -> "key2".equals(entry.getKey()) && entry.getValue() == null);
388 			assertTrue(hasKey2Entry);
389 		}
