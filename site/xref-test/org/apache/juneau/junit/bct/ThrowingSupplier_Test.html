311 	@Nested
312 	class F_concurrency extends TestBase {
313 
314 		@Test
315 		void f01_threadSafeExecution() {
316 			ThrowingSupplier<String> supplier = () -> "thread_safe_value";
317 
318 			// Test concurrent execution
319 			var futures = new ArrayList<java.util.concurrent.CompletableFuture<String>>();
320 			for (int i = 0; i < 10; i++) {
321 				futures.add(java.util.concurrent.CompletableFuture.supplyAsync(() -> Utils.safe(supplier)));
322 			}
323 
324 			// All should complete successfully
325 			for (var future : futures) {
326 				assertEquals("thread_safe_value", future.join());
327 			}
328 		}
329 
330 		@Test
331 		void f02_concurrentExceptionHandling() {
332 			ThrowingSupplier<String> throwingSupplier = () -> {
333 				throw new RuntimeException("Concurrent exception");
334 			};
335 
336 			// Test concurrent exception handling
337 			var futures = new ArrayList<java.util.concurrent.CompletableFuture<RuntimeException>>();
338 			for (int i = 0; i < 5; i++) {
339 				futures.add(java.util.concurrent.CompletableFuture.supplyAsync(() -> {
340 					try {
341 						Utils.safe(throwingSupplier);
342 						return null; // Should not reach here
343 					} catch (RuntimeException e) {
344 						return e;
345 					}
346 				}));
347 			}
348 
349 			// All should throw the same exception type
350 			for (var future : futures) {
351 				var exception = future.join();
352 				assertNotNull(exception);
353 				assertEquals("Concurrent exception", exception.getMessage());
354 			}
355 		}
356 	}
