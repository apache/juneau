335 		@Test
336 		@DisplayName("extract() - null value setting handling")
337 		void c08_extract_nullValueSettingHandling() {
338 			// Test the specific line 238 logic where property name matches nullValue setting
339 			var map = new HashMap<String, Object>();
340 			map.put(null, "null key value"); // Map with actual null key
341 			map.put("other", "other value");
342 
343 			// When property name matches the nullValue setting ("<null>"), it should be converted to null
344 			assertEquals("null key value", extractor.extract(converter, map, "<null>"));
345 
346 			// Test with custom nullValue setting
347 			var customConverter = BasicBeanConverter.builder()
348 				.defaultSettings()
349 				.addSetting(BasicBeanConverter.SETTING_nullValue, "NULL_KEY")
350 				.build();
351 
352 			var customExtractor = new PropertyExtractors.MapPropertyExtractor();
353 			map.put("NULL_KEY", "literal null key string"); // Map with literal "NULL_KEY" string
354 
355 			// When property name matches custom nullValue setting, it should look for null key
356 			assertEquals("null key value", customExtractor.extract(customConverter, map, "NULL_KEY"));
357 		}

363 
364 			// Should fall back to HashMap.isEmpty() method since "empty" is not a key

366 		}
