60  		}

93  	class B_lambdaComposition extends TestBase {

95  		@Test
96  		void b01_andThenComposition() {
97  			Listifier<String> base = (converter, str) -> Arrays.asList(str.toLowerCase());
98  			Function<List<Object>, List<Object>> mapper = list -> {
99  				List<Object> result = new ArrayList<>(list);
100 				result.add("ADDED");
101 				return result;
102 			};
103 
104 			BiFunction<BeanConverter, String, List<Object>> composed = base.andThen(mapper);
105 
106 			var converter = BasicBeanConverter.DEFAULT;
107 			var result = composed.apply(converter, "TEST");
108 
109 			assertEquals(2, result.size());
110 			assertEquals("test", result.get(0));
111 			assertEquals("ADDED", result.get(1));
112 		}
113 
114 		@Test
115 		void b02_functionalComposition() {
116 			// Compose multiple listifiers
117 			Listifier<String> splitter = (converter, str) -> Arrays.asList((Object[])str.split(","));
118 			Listifier<String> trimmer = (converter, str) -> Arrays.asList((Object[])str.trim().split("\\s+"));
119 
120 			var converter = BasicBeanConverter.DEFAULT;
121 
122 			var splitResult = splitter.apply(converter, "a,b,c");
123 			assertEquals(3, splitResult.size());
124 
125 			var trimResult = trimmer.apply(converter, "  hello   world  ");
126 			assertEquals(2, trimResult.size());
127 			assertEquals("hello", trimResult.get(0));
128 			assertEquals("world", trimResult.get(1));
129 		}
130 	}

139 		@Test
140 		void c01_nullInputHandling() {
141 			Listifier<String> nullSafe = (converter, str) -> {
142 				if (str == null) return Arrays.asList("NULL_INPUT");
143 				return Arrays.asList(str);
144 			};
145 
146 			var converter = BasicBeanConverter.DEFAULT;
147 			var result = nullSafe.apply(converter, null);
148 
149 			assertEquals(1, result.size());
150 			assertEquals("NULL_INPUT", result.get(0));
151 		}
152 
153 		@Test
154 		void c02_emptyResultHandling() {
155 			Listifier<String> emptyReturner = (converter, str) -> new ArrayList<>();
156 
157 			var converter = BasicBeanConverter.DEFAULT;
158 			var result = emptyReturner.apply(converter, "anything");
159 
160 			assertNotNull(result);
161 			assertEquals(0, result.size());
162 		}
163 
164 		@Test
165 		void c03_exceptionHandling() {
166 			Listifier<String> throwing = (converter, str) -> {
167 				if ("ERROR".equals(str)) {
168 					throw new RuntimeException("Intentional test exception");
169 				}
170 				return Arrays.asList(str);
171 			};
172 
173 			var converter = BasicBeanConverter.DEFAULT;
174 
175 			// Normal case should work
176 			var normalResult = throwing.apply(converter, "normal");
177 			assertEquals(1, normalResult.size());
178 			assertEquals("normal", normalResult.get(0));
179 
180 			// Exception case should throw
181 			assertThrows(RuntimeException.class, () -> throwing.apply(converter, "ERROR"));
182 		}
183 
184 		@Test
185 		void c04_largeListHandling() {
186 			Listifier<Integer> largeListGenerator = (converter, count) -> {
187 				List<Object> result = new ArrayList<>();
188 				for (int i = 0; i < count; i++) {
189 					result.add("item_" + i);
190 				}
191 				return result;
192 			};
193 
194 			var converter = BasicBeanConverter.DEFAULT;
195 			var result = largeListGenerator.apply(converter, 1000);
196 
197 			assertEquals(1000, result.size());
198 			assertEquals("item_0", result.get(0));
199 			assertEquals("item_999", result.get(999));
200 		}
201 	}

211 		void d01_converterIntegration() {
212 			// Test integration with custom converter
213 			var customConverter = BasicBeanConverter.builder()

216 				Arrays.asList(obj.name, obj.value, "LISTIFIED"))

218

220 			var result = customConverter.listify(test);
221 
222 			assertEquals(3, result.size());
223 			assertEquals("test", result.get(0));
224 			assertEquals(42, result.get(1));
225 			assertEquals("LISTIFIED", result.get(2));

228 		@Test
229 		void d02_multipleListifierRegistration() {
230 			var customConverter = BasicBeanConverter.builder()
231 				.defaultSettings()
232 				.addListifier(String.class, (converter, str) -> Arrays.asList(str.toLowerCase()))
233 				.addListifier(Integer.class, (converter, num) -> Arrays.asList(num, num * 2))
234 				.build();
235 
236 			// Test string listifier
237 			var stringResult = customConverter.listify("TEST");
238 			assertEquals(1, stringResult.size());
239 			assertEquals("test", stringResult.get(0));
240 
241 			// Test integer listifier
242 			var intResult = customConverter.listify(5);
243 			assertEquals(2, intResult.size());
244 			assertEquals(5, intResult.get(0));
245 			assertEquals(10, intResult.get(1));
246 		}

248 		@Test
249 		void d03_converterPassthrough() {
250 			// Test that converter parameter is properly passed
251 			Listifier<String> converterUser = (converter, str) -> {
252 				// Use the converter parameter to stringify something
253 				String stringified = converter.stringify(Arrays.asList("nested", "call"));
254 				return Arrays.asList(str, stringified);
255 			};
256 
257 			var converter = BasicBeanConverter.DEFAULT;
258 			var result = converterUser.apply(converter, "test");
259 
260 			assertEquals(2, result.size());
261 			assertEquals("test", result.get(0));
262 			assertTrue(result.get(1).toString().contains("nested"));
263 		}

273 		@Test
274 		void e01_performanceWithLargeLists() {
275 			Listifier<Integer> rangeGenerator = (converter, count) -> {
276 				List<Object> result = new ArrayList<>();
277 				for (int i = 0; i < count; i++) {
278 					result.add(i);
279 				}
280 				return result;
281 			};
282 
283 			var converter = BasicBeanConverter.DEFAULT;
284 
285 			var start = System.currentTimeMillis();
286 			var result = rangeGenerator.apply(converter, 10000);
287 			var end = System.currentTimeMillis();
288 
289 			assertEquals(10000, result.size());
290 			assertTrue(end - start < 1000, "Should complete within 1 second");
291 		}
