199 	class D_integration extends TestBase {
200 
201 		@Test
202 		void d01_useInBasicBeanConverter() {
203 			// Test Optional swapping
204 			assertBean(Optional.of("test"), "<self>", "test");
205 			assertBean(Optional.empty(), "<self>", "<null>");
206 
207 			// Test Supplier swapping
208 			Supplier<String> supplier = () -> "supplied";
209 			assertBean(supplier, "<self>", "supplied");
210 
211 			// Test Future swapping
212 			var future = CompletableFuture.completedFuture("future-result");
213 			assertBean(future, "<self>", "future-result");
214 		}
215 
216 		@Test
217 		void d02_customSwapperRegistration() {
218 			// Test that custom registration works
219 			assertBean(Optional.of("custom"), "<self>", "custom");
220 		}
221 
222 		@Test
223 		void d03_nestedSwapping() {
224 			// Test nested Optional in Supplier
225 			Supplier<Optional<String>> nestedSupplier = () -> Optional.of("nested");
226 			assertBean(nestedSupplier, "<self>", "nested");
227 
228 			// Test Optional containing Supplier
229 			Supplier<String> supplier = () -> "inner";
230 			assertBean(Optional.of(supplier), "<self>", "inner");
