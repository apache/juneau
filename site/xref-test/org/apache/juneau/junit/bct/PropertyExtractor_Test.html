150 		void b02_nullHandlingExtractor() {
151 			var nullSafe = new PropertyExtractor() {
152 				@Override
153 				public boolean canExtract(BeanConverter converter, Object o, String key) {
154 					return true; // Always can extract
155 				}
156 
157 				@Override
158 				public Object extract(BeanConverter converter, Object o, String key) {
159 					if (o == null) return "NULL_OBJECT";
160 					if (key == null) return "NULL_PROPERTY";
161 					return o.toString() + ":" + key;
162 				}
163 			};
164 
165 			var converter = BasicBeanConverter.DEFAULT;
166 
167 			assertEquals("NULL_OBJECT", nullSafe.extract(converter, null, "any"));
168 			assertEquals("NULL_PROPERTY", nullSafe.extract(converter, "obj", null));
169 			assertEquals("test:prop", nullSafe.extract(converter, "test", "prop"));
170 		}

172 		@Test
173 		void b03_typeSpecificExtractor() {
174 			var numberExtractor = new PropertyExtractor() {
175 				@Override
176 				public boolean canExtract(BeanConverter converter, Object o, String key) {
177 					return o instanceof Number;
178 				}
179 
180 				@Override
181 				public Object extract(BeanConverter converter, Object o, String key) {
182 					if (o instanceof Number) {
183 						return switch (key) {
184 							case "doubled" -> ((Number) o).doubleValue() * 2;
185 							case "string" -> o.toString();
186 							case "type" -> o.getClass().getSimpleName();
187 							default -> "UNKNOWN_PROP:" + key;
188 						};
189 					}
190 					return "NOT_A_NUMBER";
191 				}
192 			};
193 
194 			var converter = BasicBeanConverter.DEFAULT;
195 
196 			assertEquals(84.0, numberExtractor.extract(converter, 42, "doubled"));
197 			assertEquals("42", numberExtractor.extract(converter, 42, "string"));
198 			assertEquals("Integer", numberExtractor.extract(converter, 42, "type"));
199 			assertEquals("UNKNOWN_PROP:other", numberExtractor.extract(converter, 42, "other"));
200 			assertEquals("NOT_A_NUMBER", numberExtractor.extract(converter, "string", "doubled"));
201 		}

237 		@Test
238 		void c02_recursiveExtraction() {
239 			var recursive = new PropertyExtractor() {
240 				@Override
241 				public boolean canExtract(BeanConverter converter, Object o, String key) {
242 					return o instanceof String && "recursive".equals(key);
243 				}
244 
245 				@Override
246 				public Object extract(BeanConverter converter, Object o, String key) {
247 					if ("recursive".equals(key) && o instanceof String) {
248 						// Use the converter recursively
249 						String str = (String) o;
250 						return "RECURSIVE[" + converter.stringify(str.length()) + "]";
251 					}
252 					return "NON_RECURSIVE:" + key;
253 				}
254 			};
255 
256 			var converter = BasicBeanConverter.DEFAULT;
257 
258 			assertEquals("RECURSIVE[4]", recursive.extract(converter, "test", "recursive"));
259 			assertEquals("NON_RECURSIVE:other", recursive.extract(converter, "test", "other"));
260 		}

262 		@Test
263 		void c03_complexObjectExtraction() {
264 			var complex = new PropertyExtractor() {
265 				@Override
266 				public boolean canExtract(BeanConverter converter, Object o, String key) {
267 					return o instanceof Map;
268 				}
269 
270 				@Override
271 				public Object extract(BeanConverter converter, Object o, String key) {
272 					if (o instanceof Map) {
273 						Map<?, ?> map = (Map<?, ?>) o;
274 						return switch (key) {
275 							case "keys" -> new ArrayList<>(map.keySet());
276 							case "values" -> new ArrayList<>(map.values());
277 							case "entries" -> map.entrySet().size();
278 							default -> map.get(key);
279 						};
280 					}
281 					return "NOT_A_MAP";
282 				}
283 			};
284 
285 			var converter = BasicBeanConverter.DEFAULT;
286 			var testMap = Map.of("a", "valueA", "b", "valueB");
287 
288 			var keys = (List<String>) complex.extract(converter, testMap, "keys");
289 			assertEquals(2, keys.size());
290 			assertTrue(keys.contains("a"));
291 			assertTrue(keys.contains("b"));
292 
293 			assertEquals(2, complex.extract(converter, testMap, "entries"));
294 			assertEquals("valueA", complex.extract(converter, testMap, "a"));
295 			assertEquals("NOT_A_MAP", complex.extract(converter, "string", "keys"));
296 		}
