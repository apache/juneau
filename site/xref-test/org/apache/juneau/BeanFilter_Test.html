29  	@Test void a01_subTypes() throws Exception {
30  		var s = Json5Serializer.DEFAULT.copy().addBeanTypes().addRootType().build();
31  		var p = JsonParser.DEFAULT;
32  
33  		var a1 = new A1();
34  		a1.f1 = "f1";
35  		a1.fb = new B2();
36  		((B2)a1.fb).f2 = "f2";
37  		var r = s.serialize(a1);
38  		assertEquals("{_type:'A1',f0:'f0',fb:{_type:'B2',f0b:'f0b',f2:'f2'},f1:'f1'}", r);
39  
40  		assertBean(
41  			p.parse(r, A.class),
42  			"class{simpleName},fb{class{simpleName}},f1,fb{f2}",
43  			"{A1},{{B2}},f1,{f2}"
44  		);
45  
46  		// Try out-of-order creation.
47  		assertBean(
48  			p.parse("{f0:'f0',f1:'f1',_type:'A1',fb:{f0b:'f0b',f2:'f2',_type:'B2'}}", A.class),
49  			"class{simpleName},fb{class{simpleName}},f1,fb{f2}",
50  			"{A1},{{B2}},f1,{f2}"
51  		);
52  	}

85  	@Test void a02_subTypes_usingConfig() throws Exception {
86  		var s = Json5Serializer.DEFAULT.copy().addBeanTypes().addRootType().applyAnnotations(EConfig.class).build();
87  		var p = JsonParser.create().applyAnnotations(EConfig2.class).build();
88  
89  		var e1 = new E1();
90  		e1.f1 = "f1";
91  		e1.fb = new F2();
92  		((F2)e1.fb).f2 = "f2";
93  		var r = s.serialize(e1);
94  		assertEquals("{_type:'E1',f0:'f0',fb:{_type:'F2',f0b:'f0b',f2:'f2'},f1:'f1'}", r);
95  
96  		assertBean(
97  			p.parse(r, E.class),
98  			"class{simpleName},fb{class{simpleName}},f1,fb{f2}",
99  			"{E1},{{F2}},f1,{f2}"
100 		);
101 
102 		// Try out-of-order creation.
103 		r = "{f0:'f0',f1:'f1',_type:'E1',fb:{f0b:'f0b',f2:'f2',_type:'F2'}}";
104 		assertBean(
105 			p.parse(r, E.class),
106 			"class{simpleName},fb{class{simpleName}},f1,fb{f2}",
107 			"{E1},{{F2}},f1,{f2}"
108 		);
109 	}

150 		var s = JsonSerializer.create().json5().interfaces(C1.class).build();

152 		var c1 = new C2();

154 		assertEquals("{f0:'f0'}", r);
155 
156 		var l = new LinkedList<>();
157 		l.add(new C2());
158 		r = s.serialize(l);
159 		assertEquals("[{f0:'f0'}]", r);
160 	}
