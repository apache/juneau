78  	private static void check(String expected, Object o) {
79  		if (o instanceof List) {
80  			var l = (List<?>)o;
81  			String actual = l
82  				.stream()
83  				.map(TO_STRING)
84  				.collect(Collectors.joining(","));
85  			assertEquals(expected, actual);
86  		} else if (o instanceof Iterable) {
87  			var actual = StreamSupport.stream(((Iterable<?>)o).spliterator(), false)
88  				.map(TO_STRING)
89  				.collect(Collectors.joining(","));
90  			assertEquals(expected, actual);
91  		} else {
92  			assertEquals(expected, TO_STRING.apply(o));
93  		}
94  	}

96  	private static final Function<Object,String> TO_STRING = t -> {
97  		if (t == null)
98  			return null;
99  		if (t instanceof Class)
100 			return ((Class<?>)t).getSimpleName();
101 		if (t instanceof Package)
102 			return ((Package)t).getName();
103 		if (t instanceof ClassInfo)
104 			return ((ClassInfo)t).getSimpleName();
105 		if (t instanceof MethodInfo)
106 			return ((MethodInfo)t).getDeclaringClass().getSimpleName() + '.' + ((MethodInfo)t).getShortName();
107 		if (t instanceof ConstructorInfo)
108 			return ((ConstructorInfo)t).getShortName();
109 		if (t instanceof FieldInfo)
110 			return ((FieldInfo)t).getDeclaringClass().getSimpleName() + '.' + ((FieldInfo)t).getName();
111 		if (t instanceof A)
112 			return "@A(" + ((A)t).value() + ")";
113 		if (t instanceof PA)
114 			return "@PA(" + ((PA)t).value() + ")";
115 		if (t instanceof AConfig)
116 			return "@AConfig(" + ((AConfig)t).value() + ")";
117 		if (t instanceof AnnotationInfo)
118 			return ClassInfo_Test.TO_STRING.apply(((AnnotationInfo<?>)t).inner());
119 		if (t instanceof AnnotationList) {
120 			AnnotationList al = (AnnotationList)t;
121 			return al.toString();
122 		}
123 		return t.toString();
124 	};
