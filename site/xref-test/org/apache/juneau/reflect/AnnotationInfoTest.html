13  package org.apache.juneau.reflect;
14  
15  import static java.lang.annotation.ElementType.*;
16  import static java.lang.annotation.RetentionPolicy.*;
17  import static org.apache.juneau.TestUtils.*;
18  
19  import java.lang.annotation.*;
20  
21  import org.apache.juneau.*;
22  import org.apache.juneau.annotation.*;
23  import org.junit.jupiter.api.*;
24  
25  class AnnotationInfoTest extends TestBase {
26  
27  	//-----------------------------------------------------------------------------------------------------------------
28  	// Is in group.
29  	//-----------------------------------------------------------------------------------------------------------------
30  
31  	@Target(TYPE)
32  	@Retention(RUNTIME)
33  	@AnnotationGroup(D1.class)
34  	public static @interface D1 {}
35  
36  	@Target(TYPE)
37  	@Retention(RUNTIME)
38  	@AnnotationGroup(D1.class)
39  	public static @interface D2 {}
40  
41  	@Target(TYPE)
42  	@Retention(RUNTIME)
43  	public static @interface D3 {}
44  
45  	@D1 @D2 @D3
46  	public static class D {}
47  
48  	@Test void d01_isInGroup() {
49  		var d = ClassInfo.of(D.class);
50  		var l = d.getAnnotationList(x -> x.isInGroup(D1.class));
51  		assertSize(2, l);
52  	}
53  }
