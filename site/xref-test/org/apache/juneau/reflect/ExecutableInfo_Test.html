38  	private static void check(String expected, Object o) {
39  		assertEquals(expected, TO_STRING.apply(o));
40  	}

42  	private static final Function<Object,String> TO_STRING = new Function<>() {
43  		@Override
44  		public String apply(Object t) {
45  			if (t == null)
46  				return null;
47  			if (t instanceof List)
48  				return ((List<?>)t).stream().map(this).collect(Collectors.joining(","));
49  			if (isArray(t))
50  				return StreamSupport.stream(toList(t, Object.class).spliterator(), false).map(this).collect(Collectors.joining(","));
51  			if (t instanceof Annotation)
52  				return "@" + ((Annotation)t).annotationType().getSimpleName() + "()";
53  			if (t instanceof Class)
54  				return ((Class<?>)t).getSimpleName();
55  			if (t instanceof ClassInfo)
56  				return ((ClassInfo)t).getSimpleName();
57  			if (t instanceof ConstructorInfo)
58  				return ((ConstructorInfo)t).getShortName();
59  			if (t instanceof ParamInfo)
60  				return apply(((ParamInfo)t).toString());
61  			return t.toString();
62  		}
63  	};
