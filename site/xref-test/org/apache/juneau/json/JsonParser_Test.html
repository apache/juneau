43  	@Test void a03_nonStringAsString() throws Exception {
44  		var json = "123";
45  		var s = p.parse(json, String.class);
46  
47  		// Strict mode does not allow unquoted values.
48  		assertThrowsWithMessage(Exception.class, "Did not find quote character", ()->sp.parse("123", String.class));
49  
50  		assertEquals("123", s);
51  
52  		json = " 123 ";
53  		// Strict mode does not allow unquoted values.
54  		assertThrowsWithMessage(Exception.class, "Did not find quote character", ()->sp.parse(" 123 ", String.class));
55  
56  		s = p.parse(json, String.class);
57  		assertEquals("123", s);
58  
59  		json = "{\"fa\":123}";
60  		assertThrowsWithMessage(Exception.class, "Did not find quote character", ()->sp.parse("{\"fa\":123}", A.class));
61  
62  		var a = p.parse(json, A.class);
63  		assertEquals("123", a.fa);
64  
65  		json = " { \"fa\" : 123 } ";
66  		assertThrowsWithMessage(Exception.class, "Did not find quote character", ()->sp.parse(" { \"fa\" : 123 } ", A.class));
67  
68  		a = p.parse(json, A.class);
69  		assertEquals("123", a.fa);
70  
71  		assertThrowsWithMessage(Exception.class, "Invalid quote character", ()->sp.parse("'123'", String.class));
72  	}

119 	// testInvalidJsonNumbers
120 	// Lax parser allows octal and hexadecimal numbers.  Strict parser does not.
121 	//====================================================================================================
122 	@Test void a06_invalidJsonNumbers() throws Exception {
123 		var p1 = JsonParser.DEFAULT;
124 		var p2 = JsonParser.DEFAULT_STRICT;
125 
126 		// Lax allows blank strings interpreted as 0, strict does not.
127 		var s = "\"\"";
128 		var r = p1.parse(s, Number.class);
129 		assertEquals(0, r.intValue());
130 		assertTrue(r instanceof Integer);
131 		assertThrowsWithMessage(Exception.class, "Invalid JSON number", ()->p2.parse("\"\"", Number.class));
132 
133 		// Either should allow 0 or -0.
134 		s = "0";
135 		r = p1.parse(s, Number.class);
136 		assertEquals(0, r.intValue());
137 		assertTrue(r instanceof Integer);
138 		r = p2.parse(s, Number.class);
139 		assertEquals(0, r.intValue());
140 		assertTrue(r instanceof Integer);
141 
142 		s = "-0";
143 		r = p1.parse(s, Number.class);
144 		assertEquals(0, r.intValue());
145 		assertTrue(r instanceof Integer);
146 		r = p2.parse(s, Number.class);
147 		assertEquals(0, r.intValue());
148 		assertTrue(r instanceof Integer);
149 
150 		// Lax allows 0123 and -0123, strict does not.
151 		s = "0123";
152 		r = p1.parse(s, Number.class);
153 		assertEquals(0123, r.intValue());
154 		assertTrue(r instanceof Integer);
155 		assertThrowsWithMessage(Exception.class, "Invalid JSON number", ()->p2.parse("0123", Number.class));
156 		s = "-0123";
157 		r = p1.parse(s, Number.class);
158 		assertEquals(-0123, r.intValue());
159 		assertTrue(r instanceof Integer);
160 		assertThrowsWithMessage(Exception.class, "Invalid JSON number", ()->p2.parse("-0123", Number.class));
161 
162 		// Lax allows 0x123 and -0x123, strict does not.
163 		s = "0x123";
164 		r = p1.parse(s, Number.class);
165 		assertEquals(0x123, r.intValue());
166 		assertTrue(r instanceof Integer);
167 		assertThrowsWithMessage(Exception.class, "Invalid JSON number", ()->p2.parse("0x123", Number.class));
168 		s = "-0x123";
169 		r = p1.parse(s, Number.class);
170 		assertEquals(-0x123, r.intValue());
171 		assertTrue(r instanceof Integer);
172 		assertThrowsWithMessage(Exception.class, "Invalid JSON number", ()->p2.parse("-0x123", Number.class));
173 	}

179 	@Test void a07_unquotedStrings() throws Exception {
180 		var p1 = JsonParser.DEFAULT;
181 		var p2 = JsonParser.DEFAULT_STRICT;

183 		var s = "foobar";
184 		var c = p1.parse(s, C.class);

186 
187 		assertThrows(ParseException.class, ()->p2.parse(s, C.class));

190 	public static class C {
191 		String f;
192 		public static C valueOf(String s) {
193 			var c = new C();
194 			c.f = s;
195 			return c;
196 		}
197 		@Override /* Object */
198 		public String toString() {
199 			return "f="+f;
200 		}
201 	}

217 	// testMultipleObjectsInStream
218 	// Validates that readers are not closed so that we can read streams of POJOs.
219 	//====================================================================================================
220 	@Test void a09_multipleObjectsInStream() throws Exception {
221 		var p2 = JsonParser.create().unbuffered().build();
222 		var r = reader("{foo:'bar'}{baz:'qux'}");
223 
224 		var x = (Object)p2.parse(r, JsonMap.class);
225 		assertBean(x, "foo", "bar");
226 		x = p2.parse(r, JsonMap.class);
227 		assertBean(x, "baz", "qux");
228 
229 		r = reader("[123][456]");
230 		x = p2.parse(r, JsonList.class);
231 		assertList(x, "123");
232 		x = p2.parse(r, JsonList.class);
233 		assertList(x, "456");
234 	}
