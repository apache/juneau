41  	private static void check(String expected, Object o) {
42  		assertEquals(expected, TO_STRING.apply(o));
43  	}
44  
45  	private static final Function<Object,String> TO_STRING = new Function<>() {
46  		@SuppressWarnings({ "rawtypes" })
47  		@Override
48  		public String apply(Object t) {
49  			if (t == null)
50  				return null;
51  			if (t instanceof List)
52  				return ((List<?>)t)
53  					.stream()
54  					.map(TO_STRING)
55  					.collect(Collectors.joining(","));
56  			if (isArray(t))
57  				return apply(toList(t, Object.class));
58  			if (t instanceof JsonMap)
59  				return ((JsonMap)t).toString();
60  			if (t instanceof Map)
61  				return ((Map<?,?>)t)
62  					.entrySet()
63  					.stream()
64  					.map(TO_STRING)
65  					.collect(Collectors.joining(","));
66  			if (t instanceof Map.Entry e) {
67  				return apply(e.getKey()) + "=" + apply(e.getValue());
68  			}
69  			if (t instanceof BeanFilter)
70  				return ((BeanFilter)t).getBeanClass().getSimpleName();
71  			if (t instanceof Class)
72  				return ((Class<?>)t).getSimpleName();
73  			if (t instanceof ClassInfo)
74  				return ((ClassInfo)t).getSimpleName();
75  			if (t instanceof PropertyNamer)
76  				return t.getClass().getSimpleName();
77  			if (t instanceof TimeZone)
78  				return ((TimeZone)t).getID();
79  			return t.toString();
80  		}
81  	};

115 		beanFieldVisibility="$X{PRIVATE}",

122 		typePropertyName="$X{foo}",
123 		debug="$X{true}",
124 		disableIgnoreUnknownNullBeanProperties="$X{true}",
125 		disableIgnoreMissingSetters="$X{true}",

136 		notBeanPackages_replace={"$X{foo1}","$X{foo2}","$X{foo3}"},

138 		swaps_replace={AB1.class,AB2.class,AB3.class},

149 		var al = AnnotationWorkList.of(sr, a.getAnnotationList());
150 		var bs = JsonSerializer.create().apply(al).build().getSession();
151 
152 		check("PRIVATE", bs.getBeanClassVisibility());
153 		check("PRIVATE", bs.getBeanConstructorVisibility());
154 		check("A1,A2,A3", bs.getBeanDictionary());

156 		check("true", bs.isBeanMapPutReturnsOldValue());
157 		check("PRIVATE", bs.getBeanMethodVisibility());

159 		check("true", bs.isBeansRequireSerializable());
160 		check("true", bs.isBeansRequireSettersForGetters());
161 		check("false", bs.isBeansRequireSomeProperties());
162 		check("foo", bs.getBeanTypePropertyName());
163 		check("true", bs.isDebug());
164 		check("true", bs.isFindFluentSetters());
165 		check("true", bs.isIgnoreInvocationExceptionsOnGetters());
166 		check("true", bs.isIgnoreInvocationExceptionsOnSetters());
167 		check("false", bs.isIgnoreMissingSetters());
168 		check("true", bs.isIgnoreUnknownBeanProperties());
169 		check("false", bs.isIgnoreUnknownNullBeanProperties());

174 		check("AB1<String,Integer>,AB2<String,Integer>,AB3<String,Integer>", bs.getSwaps());
