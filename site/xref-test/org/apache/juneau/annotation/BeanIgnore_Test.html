13  package org.apache.juneau.annotation;
14  
15  import static org.apache.juneau.TestUtils.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.json.*;
19  import org.junit.jupiter.api.*;
20  
21  class BeanIgnore_Test extends SimpleTestBase {
22  
23  	//------------------------------------------------------------------------------------------------------------------
24  	// Test @BeanIgnore on properties
25  	//------------------------------------------------------------------------------------------------------------------
26  
27  	public static class A {
28  		public String getA() {
29  			return "a";
30  		}
31  
32  		@BeanIgnore
33  		public String getB() {
34  			return "b";
35  		}
36  
37  		public String c = "c";
38  
39  		@BeanIgnore public String d = "d";
40  	}
41  
42  	@Test void a01_beanIgnoreOnProperties() {
43  		assertJson("{c:'c',a:'a'}", new A());
44  	}
45  
46  	@BeanIgnore(on="Ac.getB")
47  	@BeanIgnore(on="Ac.d")
48  	private static class AcConfig {}
49  
50  	public static class Ac {
51  		public String getA() {
52  			return "a";
53  		}
54  
55  		public String getB() {
56  			return "b";
57  		}
58  
59  		public String c = "c";
60  
61  		public String d = "d";
62  	}
63  
64  	@Test void a02_beanIgnoreOnProperties_usingConfig() {
65  		assertSerialized(new Ac(), Json5Serializer.DEFAULT.copy().applyAnnotations(AcConfig.class).build(), "{c:'c',a:'a'}");
66  	}
67  
68  	//------------------------------------------------------------------------------------------------------------------
69  	// Test @BeanIgnore on classes
70  	//------------------------------------------------------------------------------------------------------------------
71  
72  	@BeanIgnore
73  	public static class B1 {
74  		public int f = 1;
75  
76  		@Override
77  		public String toString() {
78  			return "xxx";
79  		}
80  	}
81  
82  	public static class B {
83  		public int f2 = 2;
84  		public B1 f3 = new B1();
85  
86  		public B1 getF4() {
87  			return new B1();
88  		}
89  	}
90  
91  	@Test void a03_beanIgnoreOnBean() {
92  		assertJson("{f2:2,f3:'xxx',f4:'xxx'}", new B());
93  	}
94  
95  	@BeanIgnore(on="B1c")
96  	private static class B1cConfig {}
97  
98  	public static class B1c {
99  		public int f = 1;
100 
101 		@Override
102 		public String toString() {
103 			return "xxx";
104 		}
105 	}
106 
107 	public static class Bc {
108 		public int f2 = 2;
109 		public B1c f3 = new B1c();
110 
111 		public B1c getF4() {
112 			return new B1c();
113 		}
114 	}
115 
116 	@Test void a04_beanIgnoreOnBean_usingConfig() {
117 		assertSerialized(new Bc(), Json5Serializer.DEFAULT.copy().applyAnnotations(B1cConfig.class).build(), "{f2:2,f3:'xxx',f4:'xxx'}");
118 	}
119 }
