189 	public static class A {
190 		public int a;
191 		boolean createdByBuilder;
192 
193 		public A(ABuilder x) {
194 			if (x != null)
195 				a = x.a;
196 		}
197 
198 		public A init() {
199 			a = 1;
200 			return this;
201 		}
202 
203 		public static ABuilder create() {
204 			return new ABuilder();
205 		}
206 	}
207 
208 	public static class ABuilder {
209 		public int a;
210 
211 		public A build() {
212 			var x = new A(this);
213 			x.createdByBuilder = true;
214 			return x;
215 		}
216 	}
