13  package org.apache.juneau.objecttools;
14  
15  import static org.apache.juneau.common.internal.Utils.*;
16  import static org.junit.jupiter.api.Assertions.*;
17  
18  import org.apache.juneau.*;
19  import org.junit.jupiter.api.*;
20  
21  class ObjectIntrospector_Test extends SimpleTestBase {
22  
23  	//====================================================================================================
24  	// testBasic
25  	//====================================================================================================
26  	@Test void a01_Basic() throws Exception {
27  		var in = n(String.class);
28  
29  		var r = new ObjectIntrospector(in, null).invokeMethod("substring(int,int)", "[3,6]");
30  		assertNull(r);
31  
32  		in = "foobar";
33  		r = new ObjectIntrospector(in).invokeMethod("substring(int,int)", "[3,6]");
34  		assertEquals("bar", r);
35  
36  		r = new ObjectIntrospector(in).invokeMethod("toString", null);
37  		assertEquals("foobar", r);
38  
39  		r = new ObjectIntrospector(in).invokeMethod("toString", "");
40  		assertEquals("foobar", r);
41  
42  		r = new ObjectIntrospector(in).invokeMethod("toString", "[]");
43  		assertEquals("foobar", r);
44  
45  		assertThrows(NoSuchMethodException.class, ()->new ObjectIntrospector("foobar").invokeMethod("noSuchMethod", "[3,6]"));
46  
47  		r = new ObjectIntrospector(null).invokeMethod(String.class.getMethod("toString"), null);
48  		assertNull(r);
49  
50  		r = new ObjectIntrospector("foobar").invokeMethod(String.class.getMethod("toString"), null);
51  		assertEquals("foobar", r);
52  	}
53  }
