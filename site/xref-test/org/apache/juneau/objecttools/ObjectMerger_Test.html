13  package org.apache.juneau.objecttools;
14  
15  import static org.junit.jupiter.api.Assertions.*;
16  
17  import org.apache.juneau.*;
18  import org.junit.jupiter.api.*;
19  
20  /**
21   * Test the PojoMerge class.
22   */
23  class ObjectMerger_Test extends SimpleTestBase {
24  
25  	//====================================================================================================
26  	// Basic tests
27  	//====================================================================================================
28  	@Test void a01_basic() {
29  		IA a1, a2, am;
30  
31  		a1 = new A("1"); a2 = new A("2");
32  		am = ObjectMerger.merge(IA.class, a1, a2);
33  		assertEquals("1", am.getA());
34  		am.setA("x");
35  		assertEquals("x", am.getA());
36  		assertEquals("x", a1.getA());
37  		assertEquals("2", a2.getA());
38  
39  		a1 = new A("1"); a2 = new A("2");
40  		am = ObjectMerger.merge(IA.class, true, a1, a2);
41  		assertEquals("1", am.getA());
42  		am.setA("x");
43  		assertEquals("x", am.getA());
44  		assertEquals("x", a1.getA());
45  		assertEquals("x", a2.getA());
46  
47  		a1 = new A(null); a2 = new A("2");
48  		am = ObjectMerger.merge(IA.class, a1, a2);
49  		assertEquals("2", am.getA());
50  		am.setA("x");
51  		assertEquals("x", am.getA());
52  		assertEquals("x", a1.getA());
53  		assertEquals("2", a2.getA());
54  
55  		a1 = new A(null); a2 = new A(null);
56  		am = ObjectMerger.merge(IA.class, a1, a2);
57  		assertEquals(null, am.getA());
58  
59  		a1 = new A(null); a2 = new A("2");
60  		am = ObjectMerger.merge(IA.class, null, a1, null, null, a2, null);
61  		assertEquals("2", am.getA());
62  	}
63  
64  	public interface IA {
65  		String getA();
66  		void setA(String a);
67  	}
68  
69  	public static class A implements IA {
70  		public A(String a) {
71  			this.a = a;
72  		}
73  
74  		private String a;
75  		@Override public String getA() { return a; }
76  		@Override public void setA(String v) { a = v; }
77  	}
78  }
