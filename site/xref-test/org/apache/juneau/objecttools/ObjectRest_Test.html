35  
36  		var model = ObjectRest.create(new JsonMap()); // An empty model.
37  
38  		// Do a PUT
39  		model.put("A", new JsonMap());
40  		model.put("A/B", new JsonMap());
41  		model.put("A/B/C", "A new string");
42  		assertEquals("{A:{B:{C:'A new string'}}}", model.toString());
43  
44  		// Do a POST to a list.
45  		model.put("A/B/C", new LinkedList());
46  		model.post("A/B/C", "String #1");
47  		model.post("A/B/C", "String #2");
48  		assertEquals("{A:{B:{C:['String #1\','String #2']}}}", model.toString());
49  
50  		// Do some GETs
51  		var s = (String) model.get("A/B/C/0");
52  		assertEquals("String #1", s);
53  
54  		var m = (Map) model.get("A/B");
55  		assertEquals("{C:['String #1','String #2']}", m.toString());
56  	}

61  	@Test void b01_beans() throws Exception {
62  		var model = ObjectRest.create(new JsonMap());
63  
64  		// Java beans.
65  		var p = new Person("some name", 123,
66  			new Address("street A", "city A", "state A", 12345, true),
67  			new Address("street B", "city B", "state B", 12345, false)
68  		);
69  		model.put("/person1", p);
70  
71  		// Make sure it got stored correctly.
72  		var serializer = JsonSerializer.create().json5().addRootType().build();
73  		assertEquals("{person1:{name:'some name',age:123,addresses:[{street:'street A',city:'city A',state:'state A',zip:12345,isCurrent:true},{street:'street B',city:'city B',state:'state B',zip:12345,isCurrent:false}]}}", serializer.serialize(model.getRootObject()));
74  
75  		// Get the original Person object back.
76  		p = (Person)model.get("/person1");
77  		assertEquals("city B", p.addresses[1].city);
78  
79  		// Look for deep information inside beans.
80  		var a3 = (Address)model.get("/person1/addresses/1");
81  		assertEquals("city B", a3.city);
82  
83  		serializer = Json5Serializer.DEFAULT.copy().addBeanTypes().addRootType().build();
84  		p = new Person("some name", 123,
85  			new Address("street A", "city A", "state A", 12345, true),
86  			new Address("street B", "city B", "state B", 12345, false)
87  		);
88  
89  		// Serialize it to JSON.
90  		var s = serializer.serialize(p);
91  		var expectedValue = "{_type:'Person',name:'some name',age:123,addresses:[{street:'street A',city:'city A',state:'state A',zip:12345,isCurrent:true},{street:'street B',city:'city B',state:'state B',zip:12345,isCurrent:false}]}";
92  		assertEquals(expectedValue, s);
93  
94  		// Parse it back to Java objects.
95  		p = (Person)JsonParser.create().beanDictionary(Person.class).build().parse(s, Object.class);
96  		expectedValue = "city B";
97  		s = p.addresses[1].city;
98  		assertEquals(expectedValue, s);
99  
100 		// Parse it back into JSON again.
101 		s = serializer.serialize(p);
102 		expectedValue = "{_type:'Person',name:'some name',age:123,addresses:[{street:'street A',city:'city A',state:'state A',zip:12345,isCurrent:true},{street:'street B',city:'city B',state:'state B',zip:12345,isCurrent:false}]}";
103 		assertEquals(expectedValue, s);
104 
105 		// Try adding an address
106 		model = ObjectRest.create(p);
107 		model.post("addresses", new Address("street C", "city C", "state C", 12345, true));
108 		s = ((Address)model.get("addresses/2")).toString();
109 		expectedValue = "Address(street=street C,city=city C,state=state C,zip=12345,isCurrent=true)";
110 		assertEquals(expectedValue, s);
111 
112 		// Try replacing addresses
113 		model.put("addresses/0", new Address("street D", "city D", "state D", 12345, false));
114 		model.put("addresses/1", new Address("street E", "city E", "state E", 12345, false));
115 		model.put("addresses/2", new Address("street F", "city F", "state F", 12345, false));
116 		serializer = JsonSerializer.create().json5().build();
117 		s = serializer.serialize(p);
118 		expectedValue = "{name:'some name',age:123,addresses:[{street:'street D',city:'city D',state:'state D',zip:12345,isCurrent:false},{street:'street E',city:'city E',state:'state E',zip:12345,isCurrent:false},{street:'street F',city:'city F',state:'state F',zip:12345,isCurrent:false}]}";
119 		assertEquals(expectedValue, s);
120 
121 		// Try removing an address
122 		model.delete("addresses/1");
123 		s = serializer.serialize(p);
124 		expectedValue = "{name:'some name',age:123,addresses:[{street:'street D',city:'city D',state:'state D',zip:12345,isCurrent:false},{street:'street F',city:'city F',state:'state F',zip:12345,isCurrent:false}]}";
125 		assertEquals(expectedValue, s);
126 
127 		model.delete("addresses/0");
128 		model.delete("addresses/0");
129 		s = serializer.serialize(p);
130 		expectedValue = "{name:'some name',age:123,addresses:[]}";
131 		assertEquals(expectedValue, s);
132 
133 		// Try adding an out-of-bounds address (should pad it with nulls)
134 		model.put("addresses/2", new Address("street A", "city A", "state A", 12345, true));
135 		s = serializer.serialize(p);
136 		expectedValue = "{name:'some name',age:123,addresses:[null,null,{street:'street A',city:'city A',state:'state A',zip:12345,isCurrent:true}]}";
137 		assertEquals(expectedValue, s);
138 
139 		// Try adding an address as a map (should be automatically converted to an Address)
140 		var m = new HashMap<String,Object>();
141 		m.put("street","street D");
142 		m.put("city","city D");
143 		m.put("state","state D");
144 		m.put("zip",Integer.valueOf(12345));
145 
146 		// Try the same for an address in an array.
147 		model.put("addresses/1", m);
148 		s = ((Address)model.get("addresses/1")).toString();
149 		expectedValue = "Address(street=street D,city=city D,state=state D,zip=12345,isCurrent=false)";
150 		assertEquals(expectedValue, s);
151 
152 		// Try setting some fields.
153 		model.put("addresses/1/zip", Integer.valueOf(99999));
154 		s = model.get("addresses/1/zip").toString();
155 		expectedValue = "99999";
156 		assertEquals(expectedValue, s);
157 
158 		// Make sure we can get non-existent branches without throwing any exceptions.
159 		// get() method should just return null.
160 		model = ObjectRest.create(new JsonMap());
161 		var o = model.get("xxx");
162 		assertEquals("null", (""+o));
163 
164 		// Make sure blanks and "/" returns the root object.
165 		s = model.get("").toString();
166 		assertEquals("{}", s);
167 		s = model.get("/").toString();
168 		assertEquals("{}", s);
169 
170 		// Make sure doing a PUT against "" or "/" replaces the root object.
171 		var m2 = JsonMap.ofJson("{x:1}");
172 		model.put("", m2);
173 		s = model.get("").toString();
174 		assertEquals("{x:1}", s);
175 		m2 = JsonMap.ofJson("{x:2}");
176 		model.put("/", m2);
177 		s = model.get("").toString();
178 		assertEquals("{x:2}", s);
179 
180 		// Make sure doing a POST against "" or "/" adds to the root object.
181 		model = ObjectRest.create(new JsonList());
182 		model.post("", Integer.valueOf(1));
183 		model.post("/", Integer.valueOf(2));
184 		s = model.get("").toString();
185 		assertEquals("[1,2]", s);
186 	}

223 		public String state;
224 		public int zip;
225 		public boolean isCurrent;
