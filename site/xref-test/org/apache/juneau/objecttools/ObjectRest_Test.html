40  
41  		var model = ObjectRest.create(new JsonMap()); // An empty model.
42  
43  		// Do a PUT
44  		model.put("A", new JsonMap());
45  		model.put("A/B", new JsonMap());
46  		model.put("A/B/C", "A new string");
47  		assertEquals("{A:{B:{C:'A new string'}}}", model.toString());
48  
49  		// Do a POST to a list.
50  		model.put("A/B/C", new LinkedList());
51  		model.post("A/B/C", "String #1");
52  		model.post("A/B/C", "String #2");
53  		assertEquals("{A:{B:{C:['String #1\','String #2']}}}", model.toString());
54  
55  		// Do some GETs
56  		var s = (String) model.get("A/B/C/0");
57  		assertEquals("String #1", s);
58  
59  		var m = (Map) model.get("A/B");
60  		assertEquals("{C:['String #1','String #2']}", m.toString());
61  	}

66  	@Test void b01_beans() throws Exception {
67  		var model = ObjectRest.create(new JsonMap());
68  
69  		// Java beans.
70  		var p = new Person("some name", 123,
71  			new Address("street A", "city A", "state A", 12345, true),
72  			new Address("street B", "city B", "state B", 12345, false)
73  		);
74  		model.put("/person1", p);
75  
76  		// Make sure it got stored correctly.
77  		var serializer = JsonSerializer.create().json5().addRootType().build();
78  		assertEquals("{person1:{name:'some name',age:123,addresses:[{street:'street A',city:'city A',state:'state A',zip:12345,isCurrent:true},{street:'street B',city:'city B',state:'state B',zip:12345,isCurrent:false}]}}", serializer.serialize(model.getRootObject()));
79  
80  		// Get the original Person object back.
81  		p = (Person)model.get("/person1");
82  		assertEquals("city B", p.addresses[1].city);
83  
84  		// Look for deep information inside beans.
85  		var a3 = (Address)model.get("/person1/addresses/1");
86  		assertEquals("city B", a3.city);
87  
88  		serializer = Json5Serializer.DEFAULT.copy().addBeanTypes().addRootType().build();
89  		p = new Person("some name", 123,
90  			new Address("street A", "city A", "state A", 12345, true),
91  			new Address("street B", "city B", "state B", 12345, false)
92  		);
93  
94  		// Serialize it to JSON.
95  		var s = serializer.serialize(p);
96  		var expectedValue = "{_type:'Person',name:'some name',age:123,addresses:[{street:'street A',city:'city A',state:'state A',zip:12345,isCurrent:true},{street:'street B',city:'city B',state:'state B',zip:12345,isCurrent:false}]}";
97  		assertEquals(expectedValue, s);
98  
99  		// Parse it back to Java objects.
100 		p = (Person)JsonParser.create().beanDictionary(Person.class).build().parse(s, Object.class);
101 		expectedValue = "city B";
102 		s = p.addresses[1].city;
103 		assertEquals(expectedValue, s);
104 
105 		// Parse it back into JSON again.
106 		s = serializer.serialize(p);
107 		expectedValue = "{_type:'Person',name:'some name',age:123,addresses:[{street:'street A',city:'city A',state:'state A',zip:12345,isCurrent:true},{street:'street B',city:'city B',state:'state B',zip:12345,isCurrent:false}]}";
108 		assertEquals(expectedValue, s);
109 
110 		// Try adding an address
111 		model = ObjectRest.create(p);
112 		model.post("addresses", new Address("street C", "city C", "state C", 12345, true));
113 		s = ((Address)model.get("addresses/2")).toString();
114 		expectedValue = "Address(street=street C,city=city C,state=state C,zip=12345,isCurrent=true)";
115 		assertEquals(expectedValue, s);
116 
117 		// Try replacing addresses
118 		model.put("addresses/0", new Address("street D", "city D", "state D", 12345, false));
119 		model.put("addresses/1", new Address("street E", "city E", "state E", 12345, false));
120 		model.put("addresses/2", new Address("street F", "city F", "state F", 12345, false));
121 		serializer = JsonSerializer.create().json5().build();
122 		s = serializer.serialize(p);
123 		expectedValue = "{name:'some name',age:123,addresses:[{street:'street D',city:'city D',state:'state D',zip:12345,isCurrent:false},{street:'street E',city:'city E',state:'state E',zip:12345,isCurrent:false},{street:'street F',city:'city F',state:'state F',zip:12345,isCurrent:false}]}";
124 		assertEquals(expectedValue, s);
125 
126 		// Try removing an address
127 		model.delete("addresses/1");
128 		s = serializer.serialize(p);
129 		expectedValue = "{name:'some name',age:123,addresses:[{street:'street D',city:'city D',state:'state D',zip:12345,isCurrent:false},{street:'street F',city:'city F',state:'state F',zip:12345,isCurrent:false}]}";
130 		assertEquals(expectedValue, s);
131 
132 		model.delete("addresses/0");
133 		model.delete("addresses/0");
134 		s = serializer.serialize(p);
135 		expectedValue = "{name:'some name',age:123,addresses:[]}";
136 		assertEquals(expectedValue, s);
137 
138 		// Try adding an out-of-bounds address (should pad it with nulls)
139 		model.put("addresses/2", new Address("street A", "city A", "state A", 12345, true));
140 		s = serializer.serialize(p);
141 		expectedValue = "{name:'some name',age:123,addresses:[null,null,{street:'street A',city:'city A',state:'state A',zip:12345,isCurrent:true}]}";
142 		assertEquals(expectedValue, s);
143 
144 		// Try adding an address as a map (should be automatically converted to an Address)
145 		var m = new HashMap<String,Object>();
146 		m.put("street","street D");
147 		m.put("city","city D");
148 		m.put("state","state D");
149 		m.put("zip",Integer.valueOf(12345));
150 
151 		// Try the same for an address in an array.
152 		model.put("addresses/1", m);
153 		s = ((Address)model.get("addresses/1")).toString();
154 		expectedValue = "Address(street=street D,city=city D,state=state D,zip=12345,isCurrent=false)";
155 		assertEquals(expectedValue, s);
156 
157 		// Try setting some fields.
158 		model.put("addresses/1/zip", Integer.valueOf(99999));
159 		s = model.get("addresses/1/zip").toString();
160 		expectedValue = "99999";
161 		assertEquals(expectedValue, s);
162 
163 		// Make sure we can get non-existent branches without throwing any exceptions.
164 		// get() method should just return null.
165 		model = ObjectRest.create(new JsonMap());
166 		var o = model.get("xxx");
167 		assertEquals("null", (""+o));
168 
169 		// Make sure blanks and "/" returns the root object.
170 		s = model.get("").toString();
171 		assertEquals("{}", s);
172 		s = model.get("/").toString();
173 		assertEquals("{}", s);
174 
175 		// Make sure doing a PUT against "" or "/" replaces the root object.
176 		var m2 = JsonMap.ofJson("{x:1}");
177 		model.put("", m2);
178 		s = model.get("").toString();
179 		assertEquals("{x:1}", s);
180 		m2 = JsonMap.ofJson("{x:2}");
181 		model.put("/", m2);
182 		s = model.get("").toString();
183 		assertEquals("{x:2}", s);
184 
185 		// Make sure doing a POST against "" or "/" adds to the root object.
186 		model = ObjectRest.create(new JsonList());
187 		model.post("", Integer.valueOf(1));
188 		model.post("/", Integer.valueOf(2));
189 		s = model.get("").toString();
190 		assertEquals("[1,2]", s);
191 	}
