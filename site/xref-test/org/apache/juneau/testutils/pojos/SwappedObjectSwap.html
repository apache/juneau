13  package org.apache.juneau.testutils.pojos;
14  
15  import static org.apache.juneau.common.utils.Utils.*;
16  import static org.apache.juneau.utest.utils.Constants.*;
17  
18  import org.apache.juneau.*;
19  import org.apache.juneau.parser.*;
20  import org.apache.juneau.serializer.*;
21  import org.apache.juneau.swap.*;
22  
23  public class SwappedObjectSwap extends ObjectSwap<SwappedObject,String> {
24  	@Override
25  	public String swap(BeanSession session, SwappedObject c) throws SerializeException {
26  		return SWAP;
27  	}
28  
29  	@Override
30  	public SwappedObject unswap(BeanSession session, String f, ClassMeta<?> hint) throws ParseException {
31  		var c = new SwappedObject();
32  		if (eq(f, SWAP))
33  			c.wasUnswapped = true;
34  		return c;
35  	}
36  }
