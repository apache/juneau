13  package org.apache.juneau.testutils.pojos;
14  
15  @org.apache.juneau.annotation.Bean(typeName="TypedBeanImpl", sort=true)
16  public class TypedBeanImpl implements TypedBean {
17  	public int a;
18  	public String b;
19  
20  	private TypedBeanImpl init() {
21  		this.a = 1;
22  		this.b = "foo";
23  		return this;
24  	}
25  
26  	public static TypedBeanImpl get() {
27  		return new TypedBeanImpl().init();
28  	}
29  
30  	@Override
31  	public String toString() {
32  		return "a:" + a + ";b:" + b;
33  	}
34  }
