126 	// testFluentSetters - Test fluent setter overrides
127 	//====================================================================================================
128 	@Test void a05_fluentSetters() {
129 		var vr = VarResolver.create().defaultVars().vars(XVar.class).build();
130 		var m = new ResolvingJsonMap(vr.createSession());
131 
132 		// Test inner() returns same instance for fluent chaining
133 		Map<String,Object> innerMap = new HashMap<>();
134 		innerMap.put("test", "$X{a}");
135 		assertSame(m, m.inner(innerMap));
136 		assertEquals("1", m.get("test"));
137 
138 		// Test session() returns same instance
139 		BeanSession session = BeanContext.DEFAULT.getSession();
140 		assertSame(m, m.session(session));
141 
142 		// Test append(String, Object) returns same instance
143 		assertSame(m, m.append("key1", "$X{b}"));
144 		assertEquals("2", m.get("key1"));
145 
146 		// Test append(Map) returns same instance
147 		Map<String,Object> appendMap = new HashMap<>();
148 		appendMap.put("key2", "$X{c}");
149 		assertSame(m, m.append(appendMap));
150 		assertEquals("3", m.get("key2"));
151 
152 		// Test appendIf() returns same instance
153 		assertSame(m, m.appendIf(true, "key3", "value3"));
154 		assertEquals("value3", m.get("key3"));
155 		assertSame(m, m.appendIf(false, "key4", "value4"));
156 		assertNull(m.get("key4"));
157 
158 		// Test filtered() returns same instance
159 		assertSame(m, m.filtered(x -> x != null));
160 
161 		// Test keepAll() returns same instance
162 		assertSame(m, m.keepAll("key1", "key2"));
163 
164 		// Test setBeanSession() returns same instance
165 		assertSame(m, m.setBeanSession(session));
166 
167 		// Test modifiable() returns a new instance when unmodifiable
168 		assertSame(m, m.modifiable());
169 
170 		// Test unmodifiable() returns same instance
171 		assertSame(m, m.unmodifiable());
172 	}

174 	@Test void a06_fluentChaining() {
175 		var vr = VarResolver.create().defaultVars().vars(XVar.class).build();
176 		// Test multiple fluent calls can be chained
177 		var m = new ResolvingJsonMap(vr.createSession())
178 			.append("key1", "$X{a}")
179 			.append("key2", "$X{b}")
180 			.appendIf(true, "key3", "$X{c}");
181 
182 		assertEquals("1", m.get("key1"));
183 		assertEquals("2", m.get("key2"));
184 		assertEquals("3", m.get("key3"));
