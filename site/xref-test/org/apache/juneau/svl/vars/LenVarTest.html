15  import static org.junit.jupiter.api.Assertions.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.svl.*;
19  import org.junit.jupiter.api.*;
20  
21  class LenVarTest extends TestBase {
22  
23  	//====================================================================================================
24  	// test - Basic tests
25  	//====================================================================================================
26  	@Test void a01_test() {
27  		var vr = VarResolver.create().vars(LenVar.class, SystemPropertiesVar.class).build();
28  
29  		System.setProperty("LenVarTest.test", "foo bar");
30  		System.setProperty("LenVarTest.test2", "1.2.3.4.5");
31  
32  		// $LN{stringArg} examples
33  		assertEquals("1", vr.resolve("$LN{J}"));
34  		assertEquals("len=7", vr.resolve("len=$LN{$S{LenVarTest.test}}"));
35  
36  		// $LN{stringArg, delimiter} examples
37  		assertEquals("5", vr.resolve("$LN{$S{LenVarTest.test2},.}"));
38  	}
39  }
