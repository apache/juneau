41  /**
42   * Comprehensive utility class for Bean-Centric Tests (BCT) and general testing operations.
43   *
44   * <p>This class provides the core testing infrastructure for Apache Juneau, with particular emphasis
45   * on the Bean-Centric Testing (BCT) framework. BCT enables sophisticated assertion patterns for
46   * testing object properties, collections, maps, and complex nested structures with minimal code.</p>
47   *
48   * <h5 class='section'>Bean-Centric Testing (BCT) Framework:</h5>
49   * <p>The BCT framework consists of several key components:</p>
50   * <ul>
51   * 	<li><b>{@link BeanConverter}:</b> Core interface for object conversion and property access</li>
52   * 	<li><b>{@link BasicBeanConverter}:</b> Default implementation with extensible type handlers</li>
53   * 	<li><b>Assertion Methods:</b> High-level testing methods that leverage the converter framework</li>
54   * </ul>
55   *
56   * <h5 class='section'>Primary BCT Assertion Methods:</h5>
57   * <dl>
58   * 	<dt><b>{@link #assertBean(Object, String, String)}</b></dt>
59   * 	<dd>Tests object properties with nested syntax support and collection iteration</dd>
60   *
61   * 	<dt><b>{@link #assertMap(Map, String, String)}</b></dt>
62   * 	<dd>Tests map entries with the same nested property syntax as assertBean</dd>
63   *
64   * 	<dt><b>{@link #assertMapped(Object, java.util.function.BiFunction, String, String)}</b></dt>
65   * 	<dd>Tests custom property access using BiFunction for non-standard objects</dd>
66   *
67   * 	<dt><b>{@link #assertList(List, Object...)}</b></dt>
68   * 	<dd>Tests list/collection elements with varargs for expected values</dd>
69   *
70   * 	<dt><b>{@link #assertBeans(Collection, String, String...)}</b></dt>
71   * 	<dd>Tests collections of objects by extracting and comparing specific fields</dd>
72   * </dl>
73   *
74   * <h5 class='section'>BCT Advanced Features:</h5>
75   * <ul>
76   * 	<li><b>Nested Property Syntax:</b> "address{street,city}" for testing nested objects</li>
77   * 	<li><b>Collection Iteration:</b> "#{property}" syntax for testing all elements</li>
78   * 	<li><b>Universal Size Properties:</b> "length" and "size" work on all collection types</li>
79   * 	<li><b>Array/List Access:</b> Numeric indices for element-specific testing</li>
80   * 	<li><b>Method Chaining:</b> Fluent setters can be tested directly</li>
81   * 	<li><b>Direct Field Access:</b> Public fields accessed without getters</li>
82   * 	<li><b>Map Key Access:</b> Including special "&lt;NULL&gt;" syntax for null keys</li>
83   * </ul>
84   *
85   * <h5 class='section'>Converter Extensibility:</h5>
86   * <p>The BCT framework is built on the extensible {@link BasicBeanConverter} which allows:</p>
87   * <ul>
88   * 	<li><b>Custom Stringifiers:</b> Type-specific string conversion logic</li>
89   * 	<li><b>Custom Listifiers:</b> Collection-type conversion for iteration</li>
90   * 	<li><b>Custom Swapifiers:</b> Object transformation before conversion</li>
91   * 	<li><b>Configurable Settings:</b> Formatting, delimiters, and display options</li>
92   * </ul>
93   *
94   * <h5 class='section'>Usage Examples:</h5>
95   *
96   * <p><b>Basic Property Testing:</b></p>
97   * <p class='bjava'>
98   * 	<jc>// Test multiple properties</jc>
99   * 	assertBean(user, <js>"name,age,active"</js>, <js>"John,30,true"</js>);
100  *
101  * 	<jc>// Test nested properties</jc>
102  * 	assertBean(user, <js>"address{street,city}"</js>, <js>"{123 Main St,Springfield}"</js>);
103  * </p>
104  *
105  * <p><b>Collection and Array Testing:</b></p>
106  * <p class='bjava'>
107  * 	<jc>// Test collection size and iterate over all elements</jc>
108  * 	assertBean(order, <js>"items{length,#{name}}"</js>, <js>"{3,[{Laptop},{Phone},{Tablet}]}"</js>);
109  *
110  * 	<jc>// Test specific array elements</jc>
111  * 	assertBean(data, <js>"values{0,1,2}"</js>, <js>"{100,200,300}"</js>);
112  * </p>
113  *
114  * <p><b>Map and Collection Testing:</b></p>
115  * <p class='bjava'>
116  * 	<jc>// Test map entries</jc>
117  * 	assertMap(config, <js>"timeout,retries"</js>, <js>"30000,3"</js>);
118  *
119  * 	<jc>// Test list elements</jc>
120  * 	assertList(tags, <js>"red"</js>, <js>"green"</js>, <js>"blue"</js>);
121  * </p>
122  *
123  * <p><b>Custom Property Access:</b></p>
124  * <p class='bjava'>
125  * 	<jc>// Test with custom accessor function</jc>
126  * 	assertMapped(myObject, (obj, prop) -> obj.getProperty(prop),
127  * 		<js>"prop1,prop2"</js>, <js>"value1,value2"</js>);
128  * </p>
129  *
130  * <h5 class='section'>Performance and Thread Safety:</h5>
131  * <p>The BCT framework is designed for high performance with:</p>
132  * <ul>
133  * 	<li><b>Caching:</b> Type-to-handler mappings cached for fast lookup</li>
134  * 	<li><b>Thread Safety:</b> All operations are thread-safe for concurrent testing</li>
135  * 	<li><b>Minimal Allocation:</b> Efficient object reuse and minimal temporary objects</li>
136  * </ul>
137  *
138  * @see BeanConverter
139  * @see BasicBeanConverter
140  */

204 	/**
205 	 * Validates that the whitespace is correct in the specified XML.
206 	 */
207 	public static final void checkXmlWhitespace(String out) throws SerializeException {
208 		if (out.indexOf('\u0000') != -1) {
209 			for (var s : out.split("\u0000"))
210 				checkXmlWhitespace(s);
211 			return;
212 		}
213 
214 		var indent = -1;
215 		var startTag = Pattern.compile("^(\\s*)<[^/>]+(\\s+\\S+=['\"]\\S*['\"])*\\s*>$");  // NOSONAR
216 		var endTag = Pattern.compile("^(\\s*)</[^>]+>$");
217 		var combinedTag = Pattern.compile("^(\\s*)<[^>/]+(\\s+\\S+=['\"]\\S*['\"])*\\s*/>$");  // NOSONAR
218 		var contentOnly = Pattern.compile("^(\\s*)[^\\s\\<]+$");
219 		var tagWithContent = Pattern.compile("^(\\s*)<[^>]+>.*</[^>]+>$");
220 		var lines = out.split("\n");
221 		try {
222 			for (var i = 0; i < lines.length; i++) {
223 				var line = lines[i];
224 				var m = startTag.matcher(line);
225 				if (m.matches()) {
226 					indent++;
227 					if (m.group(1).length() != indent)
228 						throw new SerializeException("Wrong indentation detected on start tag line ''{0}''", i+1);
229 					continue;
230 				}
231 				m = endTag.matcher(line);
232 				if (m.matches()) {
233 					if (m.group(1).length() != indent)
234 						throw new SerializeException("Wrong indentation detected on end tag line ''{0}''", i+1);
235 					indent--;
236 					continue;
237 				}
238 				m = combinedTag.matcher(line);
239 				if (m.matches()) {
240 					indent++;
241 					if (m.group(1).length() != indent)
242 						throw new SerializeException("Wrong indentation detected on combined tag line ''{0}''", i+1);
243 					indent--;
244 					continue;
245 				}
246 				m = contentOnly.matcher(line);
247 				if (m.matches()) {
248 					indent++;
249 					if (m.group(1).length() != indent)
250 						throw new SerializeException("Wrong indentation detected on content-only line ''{0}''", i+1);
251 					indent--;
252 					continue;
253 				}
254 				m = tagWithContent.matcher(line);
255 				if (m.matches()) {
256 					indent++;
257 					if (m.group(1).length() != indent)
258 						throw new SerializeException("Wrong indentation detected on tag-with-content line ''{0}''", i+1);
259 					indent--;
260 					continue;
261 				}
262 				throw new SerializeException("Unmatched whitespace line at line number ''{0}''", i+1);
263 			}
264 			if (indent != -1)
265 				throw new SerializeException("Possible unmatched tag.  indent=''{0}''", indent);
266 		} catch (SerializeException e) {
267 			printLines(lines);
268 			throw e;
269 		}
270 	}

272 	/**
273 	 * Returns the value of the specified field/property on the specified object.
274 	 * First looks for getter, then looks for field.
275 	 * Methods and fields can be any visibility.
276 	 */
277 	public static Object getBeanProp(Object o, String name) {
278 		return safe(() -> {
279 			var f = (Field)null;
280 			var c = o.getClass();
281 			var n = Character.toUpperCase(name.charAt(0)) + name.substring(1);
282 			var m = Arrays.stream(c.getMethods()).filter(x -> x.getName().equals("is"+n) && x.getParameterCount() == 0 && x.getAnnotation(BeanIgnore.class) == null).findFirst().orElse(null);
283 			if (m != null) {
284 				m.setAccessible(true);
285 				return m.invoke(o);
286 			}
287 			m = Arrays.stream(c.getMethods()).filter(x -> x.getName().equals("get"+n) && x.getParameterCount() == 0 && x.getAnnotation(BeanIgnore.class) == null).findFirst().orElse(null);
288 			if (m != null) {
289 				m.setAccessible(true);
290 				return m.invoke(o);
291 			}
292 			m = Arrays.stream(c.getMethods()).filter(x -> x.getName().equals("get") && x.getParameterCount() == 1 && x.getParameterTypes()[0] == String.class && x.getAnnotation(BeanIgnore.class) == null).findFirst().orElse(null);
293 			if (m != null) {
294 				m.setAccessible(true);
295 				return m.invoke(o, name);
296 			}
297 			var c2 = c;
298 			while (f == null && c2 != null) {
299 				f = Arrays.stream(c2.getDeclaredFields()).filter(x -> x.getName().equals(name)).findFirst().orElse(null);
300 				c2 = c2.getSuperclass();
301 			}
302 			if (f != null) {
303 				f.setAccessible(true);
304 				return f.get(o);
305 			}
306 			m = Arrays.stream(c.getMethods()).filter(x -> x.getName().equals(name) && x.getParameterCount() == 0).findFirst().orElse(null);
307 			if (m != null) {
308 				m.setAccessible(true);
309 				return m.invoke(o);
310 			}
311 			throw ThrowableUtils.runtimeException("Property {0} not found on object of type {1}", name, cn(o));
312 		});
313 	}

323 		try {

325 			var rc = RestContext.create(r.getClass(),null,null).init(()->r).build();
326 			var ctx = RestOpContext.create(TestUtils.class.getMethod("getSwagger", Class.class), rc).build();
327 			var session = RestSession.create(rc).resource(r).req(new MockServletRequest()).res(new MockServletResponse()).build();
328 			var req = ctx.createRequest(session);
329 			var ip = rc.getSwaggerProvider();
330 			return ip.getSwagger(rc, req.getLocale());
331 		} catch (Exception e) {
332 			throw new RuntimeException(e);
333 		}

425 	/**
426 	 * Extracts HTML/XML elements from a string based on element name and attributes.
427 	 *
428 	 * <p>Uses a depth-tracking parser to handle nested elements correctly, even with malformed HTML.</p>
429 	 *
430 	 * <h5 class='section'>Examples:</h5>
431 	 * <pre>
432 	 * // Extract all div elements with class='tag-block'
433 	 * List&lt;String&gt; blocks = extractXml(html, "div", Map.of("class", "tag-block"));
434 	 *
435 	 * // Extract all span elements (no attribute filtering)
436 	 * List&lt;String&gt; spans = extractXml(html, "span", null);
437 	 *
438 	 * // Extract divs with multiple attributes
439 	 * List&lt;String&gt; divs = extractXml(html, "div", Map.of("class", "header", "id", "main"));
440 	 * </pre>
441 	 *
442 	 * @param html The HTML/XML content to parse
443 	 * @param elementName The element name to extract (e.g., "div", "span")
444 	 * @param withAttributes Optional map of attribute name/value pairs that must match.
445 	 *                       Pass null or empty map to match all elements of the given name.
446 	 * @return List of HTML content strings (inner content of matching elements)
447 	 */
448 	public static List<String> extractXml(String html, String elementName, Map<String,String> withAttributes) {
449 		List<String> results = new ArrayList<>();
450 
451 		if (html == null || elementName == null) {
452 			return results;
453 		}
454 
455 		// Find all opening tags of the specified element
456 		String openTag = "<" + elementName;
457 		int searchPos = 0;
458 
459 		while ((searchPos = html.indexOf(openTag, searchPos)) != -1) {
460 			// Find the end of the opening tag
461 			int tagEnd = html.indexOf('>', searchPos);
462 			if (tagEnd == -1) break;
463 
464 			String fullOpenTag = html.substring(searchPos, tagEnd + 1);
465 
466 			// Check if attributes match
467 			boolean matches = true;
468 			if (withAttributes != null && !withAttributes.isEmpty()) {
469 				for (Map.Entry<String, String> entry : withAttributes.entrySet()) {
470 					String attrName = entry.getKey();
471 					String attrValue = entry.getValue();
472 
473 					// Look for attribute in the tag (handle both single and double quotes)
474 					String pattern1 = attrName + "=\"" + attrValue + "\"";
475 					String pattern2 = attrName + "='" + attrValue + "'";
476 
477 					if (!fullOpenTag.contains(pattern1) && !fullOpenTag.contains(pattern2)) {
478 						matches = false;
479 						break;
480 					}
481 				}
482 			}
483 
484 			if (matches) {
485 				// Find matching closing tag by tracking depth
486 				int contentStart = tagEnd + 1;
487 				int depth = 1;
488 				int pos = contentStart;
489 
490 				while (pos < html.length() && depth > 0) {
491 					// Look for next opening or closing tag of same element
492 					int nextOpen = html.indexOf("<" + elementName, pos);
493 					int nextClose = html.indexOf("</" + elementName + ">", pos);
494 
495 					// Validate that nextOpen is actually an opening tag
496 					if (nextOpen != -1 && (nextOpen < nextClose || nextClose == -1)) {
497 						if (nextOpen + elementName.length() + 1 < html.length()) {
498 							char nextChar = html.charAt(nextOpen + elementName.length() + 1);
499 							if (nextChar == ' ' || nextChar == '>' || nextChar == '/') {
500 								depth++;
501 								pos = nextOpen + elementName.length() + 1;
502 								continue;
503 							}
504 						}
505 						// Not a valid opening tag, skip it
506 						pos = nextOpen + 1;
507 						continue;
508 					}
509 
510 					if (nextClose != -1) {
511 						depth--;
512 						if (depth == 0) {
513 							// Found matching close tag
514 							results.add(html.substring(contentStart, nextClose));
515 							break;
516 						}
517 						pos = nextClose + elementName.length() + 3;
518 					} else {
519 						// No more closing tags
520 						break;
521 					}
522 				}
523 			}
524 
525 			searchPos = tagEnd + 1;
526 		}
527 
528 		return results;
529 	}
