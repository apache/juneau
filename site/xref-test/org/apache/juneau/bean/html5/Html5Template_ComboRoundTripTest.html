64  	@Bean(dictionary=HtmlBeanDictionary.class)
65  	public static class FormTemplate {
66  
67  		private String action;
68  		private int value1;
69  		private boolean value2;
70  
71  		public FormTemplate(Form f) {
72  			action = f.getAttr("action");
73  			value1 = f.getChild(Input.class, 0).getAttr(int.class, "value");
74  			value2 = f.getChild(Input.class, 1).getAttr(boolean.class, "value");
75  		}
76  
77  		public FormTemplate(String action, int value1, boolean value2) {
78  			this.action = action;
79  			this.value1 = value1;
80  			this.value2 = value2;
81  		}
82  
83  		public Form swap(BeanSession session) {
84  			return form(action,
85  				input("text").name("v1").value(value1),
86  				input("text").name("v2").value(value2)
87  			);
88  		}
89  	}
