419 			var x = swagger().addDefinition("a1", JsonMap.of("type", "a2"));
420 			assertBean(

422 				"type",

427 
428 			assertThrows(IllegalArgumentException.class, () -> x.findRef(null, JsonMap.class));
429 			assertThrows(IllegalArgumentException.class, () -> x.findRef("a", null));
430 			assertThrows(IllegalArgumentException.class, () -> x.findRef("", JsonMap.class));
431 			assertThrowsWithMessage(BasicRuntimeException.class, "Unsupported reference:  'invalid'", () -> x.findRef("invalid", JsonMap.class));
