83  	@Test void a08_otherGettersAndSetters() {
84  		// Test Collection variants of setters
85  		var x = bean()
86  			.setConsumes(list(
87  				MediaType.of("a1"),
88  				MediaType.of("a2")
89  			))
90  			.setProduces(list(
91  				MediaType.of("b1"),
92  				MediaType.of("b2")
93  			))
94  			.setSchemes(list("c1", "c2"))
95  			.setSecurity(list(
96  				map("d1", list("d2")),
97  				map("d3", list("d4"))
98  			));
99  
100 		assertBean(x,
101 			"consumes,produces,schemes,security{0{d1},1{d3}}",
102 			"[a1,a2],[b1,b2],[c1,c2],{{[d2]},{[d4]}}"
103 		);
104 
105 		// Test special getters
106 		x = bean()
107 			.addPath("a1", "get", operation().setSummary("a2"))
108 			.addPath("b1", "get", operation().addResponse("200", responseInfo("b2")).setParameters(parameterInfo("b3", "b4")))
109 			.addParameter("c1", parameterInfo("c2", "c3"));
110 
111 		assertBean(x.getPath("a1"), "get{summary}", "{a2}");
112 		assertBean(x.getOperation("a1", "get"), "summary", "a2");
113 		assertBean(x.getResponseInfo("b1", "get", "200"), "description", "b2");
114 		assertBean(x.getResponseInfo("b1", "get", 200), "description", "b2");
115 		assertBean(x.getParameterInfo("b1", "get", "b3", "b4"), "in,name", "b3,b4");
116 	}

344 		@Test void g02_findRef() {
345 			var x = swagger().addDefinition("a1", JsonMap.of("type", "a2"));
346 			assertBean(
347 				x.findRef("#/definitions/a1", JsonMap.class), 
348 				"type", 
349 				"a2"
350 			);
351 
352 			assertNull(x.findRef("#/definitions/notfound", JsonMap.class));
353 
354 			assertThrows(IllegalArgumentException.class, () -> x.findRef(null, JsonMap.class));
355 			assertThrows(IllegalArgumentException.class, () -> x.findRef("a", null));
356 			assertThrows(IllegalArgumentException.class, () -> x.findRef("", JsonMap.class));
357 			assertThrowsWithMessage(BasicRuntimeException.class, "Unsupported reference:  'invalid'", () -> x.findRef("invalid", JsonMap.class));
