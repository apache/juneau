92  					.setDescription("a")

94  					.set("x1", "x1a")
95  					.asMap(),
96  				"description,type,x1",
97  				"a,b,x1a"
98  			);
99  		}
100 
101 		@Test void a11_extraKeys() {
102 			var x = bean().set("x1", "x1a").set("x2", "x2a");
103 			assertList(x.extraKeys(), "x1", "x2");
104 			assertEmpty(bean().extraKeys());
105 		}
106 
107 		@Test void a12_strictMode() {
108 			assertThrows(RuntimeException.class, () -> bean().strict().set("foo", "bar"));
109 			assertDoesNotThrow(() -> bean().set("foo", "bar"));
110 
111 			assertFalse(bean().isStrict());
112 			assertTrue(bean().strict().isStrict());
113 			assertFalse(bean().strict(false).isStrict());
114 
115 			var x = bean().strict();
116 			var y = bean(); // not strict
117 
118 			assertThrowsWithMessage(RuntimeException.class, "Invalid value passed in to setType(String).  Value='invalid', valid values=['basic','apiKey','oauth2']", () -> x.setType("invalid"));
119 			assertDoesNotThrow(() -> x.setType("basic"));
120 			assertDoesNotThrow(() -> x.setType("apiKey"));
121 			assertDoesNotThrow(() -> x.setType("oauth2"));
122 			assertDoesNotThrow(() -> y.setType("invalid"));
123 		}
124 	}
