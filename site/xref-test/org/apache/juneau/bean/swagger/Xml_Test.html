28  	/**
29  	 * Test method for getters and setters.
30  	 */
31  	@Test void a01_gettersAndSetters() {
32  		var t = new Xml();
33  
34  		// Basic property setters
35  		assertBean(
36  			t.setName("a").setNamespace("b").setPrefix("c").setAttribute(true).setWrapped(true),
37  			"name,namespace,prefix,attribute,wrapped",
38  			"a,b,c,true,true"
39  		);
40  
41  		// Null values
42  		assertBean(
43  			t.setName(null).setNamespace(null).setPrefix(null),
44  			"name,namespace,prefix",
45  			"<null>,<null>,<null>"
46  		);
47  	}
48  
49  	/**
50  	 * Test method for {@link Xml#set(java.lang.String, java.lang.Object)}.
51  	 */
52  	@Test void b01_set() {
53  		var t = new Xml();
54  
55  		t
56  			.set("attribute", true)
57  			.set("name", "a")
58  			.set("namespace", "b")
59  			.set("prefix", "c")
60  			.set("wrapped", true)
61  			.set("$ref", "d");
62  
63  		// Comprehensive object state validation
64  		assertBean(t,
65  			"name,namespace,prefix,attribute,wrapped,$ref",
66  			"a,b,c,true,true,d");
67  
68  		t
69  			.set("attribute", "true")
70  			.set("name", "a")
71  			.set("namespace", "b")
72  			.set("prefix", "c")
73  			.set("wrapped", "true")
74  			.set("$ref", "d");
75  
76  		assertBean(t,
77  			"name,namespace,prefix,attribute,wrapped,$ref",
78  			"a,b,c,true,true,d");
79  
80  		t
81  			.set("attribute", Utils.sb("true"))
82  			.set("name", Utils.sb("a"))
83  			.set("namespace", Utils.sb("b"))
84  			.set("prefix", Utils.sb("c"))
85  			.set("wrapped", Utils.sb("true"))
86  			.set("$ref", Utils.sb("d"));
87  
88  		assertBean(t,
89  			"name,namespace,prefix,attribute,wrapped,$ref",
90  			"a,b,c,true,true,d");
91  
92  		assertMapped(t, (obj,prop) -> obj.get(prop, String.class),
93  			"attribute,name,namespace,prefix,wrapped,$ref",
94  			"true,a,b,c,true,d");
95  
96  		assertMapped(t, (obj,prop) -> obj.get(prop, Object.class).getClass().getSimpleName(),
97  			"attribute,name,namespace,prefix,wrapped,$ref",
98  			"Boolean,String,String,String,Boolean,StringBuilder");
99  
100 		t.set("null", null).set(null, "null");
101 		assertNull(t.get("null", Object.class));
102 		assertNull(t.get(null, Object.class));
103 		assertNull(t.get("foo", Object.class));
104 	}

114 		t = t.copy();

117 
118 		t
119 			.set("attribute", true)
120 			.set("name", "a")
121 			.set("namespace", "b")
122 			.set("prefix", "c")
123 			.set("wrapped", true)
124 			.set("$ref", "d")
125 			.copy();
126 
127 		assertBean(t,
128 			"name,namespace,prefix,attribute,wrapped,$ref",
129 			"a,b,c,true,true,d");
130 	}

132 	@Test void b04_keySet() {
133 		var t = new Xml();
134 
135 		assertEmpty(t.keySet());
136 
137 		t
138 			.set("attribute", true)
139 			.set("name", "a")
140 			.set("namespace", "b")
141 			.set("prefix", "c")
142 			.set("wrapped", true)
143 			.set("$ref", "d");
144 
145 		assertList(t.keySet(), "$ref", "attribute", "name", "namespace", "prefix", "wrapped");
146 	}
