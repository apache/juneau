34  	@Test void a01_gettersAndSetters() {
35  		var t = new SchemaInfo();
36  
37  		// General - Combined assertBean test
38  		assertBean(
39  			t.setAdditionalProperties(new SchemaInfo().setType("a")).setAllOf(set("b1","b2")).setDefault("c").setDescription("d")
40  				.setDiscriminator("e").setEnum(set("f1","f2")).setExample("g").setExclusiveMaximum(true).setExclusiveMinimum(true)
41  				.setExternalDocs(externalDocumentation("h")).setFormat("i").setItems(items("j")).setMaxItems(1).setMaxLength(2)
42  				.setMaxProperties(3).setMaximum(4).setMinItems(5).setMinLength(6).setMinProperties(7).setMinimum(8)
43  				.setMultipleOf(9).setPattern("k").setProperties(map("l",new SchemaInfo().setType("m"))).setReadOnly(true)
44  				.setRequiredProperties("n").setTitle("o").setType("p").setUniqueItems(true).setXml(xml().setName("q")),
45  			"additionalProperties{type},allOf,default,description,discriminator,enum,example,exclusiveMaximum,exclusiveMinimum,externalDocs{url},format,items{type},maxItems,maxLength,maxProperties,maximum,minItems,minLength,minProperties,minimum,multipleOf,pattern,properties{l{type}},readOnly,requiredProperties,title,type,uniqueItems,xml{name}",
46  			"{a},[b1,b2],c,d,e,[f1,f2],g,true,true,{h},i,{j},1,2,3,4,5,6,7,8,9,k,{{m}},true,[n],o,p,true,{q}"
47  		);
48  
49  		// Null cases
50  		assertBean(
51  			t.setAdditionalProperties((SchemaInfo)null).setAllOf((Collection<Object>)null).setDefault(null).setDescription(null)
52  				.setDiscriminator(null).setEnum((Collection<Object>)null).setExample(null).setFormat(null)
53  				.setPattern(null).setProperties((Map<String,SchemaInfo>)null).setRequiredProperties((Collection<String>)null)
54  				.setTitle(null).setType(null),
55  			"additionalProperties,allOf,default,description,discriminator,enum,example,format,pattern,properties,requiredProperties,title,type",
56  			"<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>"
57  		);
58  
59  		// Empty collections
60  		assertBean(
61  			t.setAllOf(set()).setProperties(map()),
62  			"allOf,properties",
63  			"[],{}"
64  		);
65  
66  		// Other
67  		assertBean(
68  			t.setDefault(Utils.sb("a")).setExample(1).setMaximum(2f).setMinimum(3f).setMultipleOf(4f),
69  			"default,example,maximum,minimum,multipleOf", "a,1,2.0,3.0,4.0");
70  		assertBean(t.setAdditionalProperties(new SchemaInfo()).getAdditionalProperties(),
71  			"default,enum,$ref,additionalProperties,allOf,description,discriminator,example,exclusiveMaximum,exclusiveMinimum,externalDocs,format,items,maximum,maxItems,maxLength,minimum,minItems,minLength,multipleOf,pattern,properties,readOnly,required,title,type,uniqueItems,xml",
72  			"<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>,<null>");
73  
74  		// addEnum and addAllOf
75  		assertList(t.addEnum("a","b").getEnum(), "a", "b");
76  		assertList(t.addEnum("c").getEnum(), "a", "b", "c");
77  		assertList(t.addAllOf("d","e").getAllOf(), "d", "e");
78  	}
