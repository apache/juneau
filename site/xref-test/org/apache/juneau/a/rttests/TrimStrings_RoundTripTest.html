24  /**
25   * Tests for the {@link Serializer#SERIALIZER_trimStrings} and {@link Parser#PARSER_trimStrings}.
26   */
27  class TrimStrings_RoundTripTest extends RoundTripTest_Base {
28  
29  	//====================================================================================================
30  	// test
31  	//====================================================================================================
32  
33  	@ParameterizedTest
34  	@MethodSource("testers")
35  	void a01_basic(RoundTrip_Tester t) throws Exception {
36  		if (t.isValidationOnly())
37  			return;
38  		var s = t.getSerializer();
39  		var p = t.getParser();
40  
41  		var s2 = s.copy().trimStrings().build();
42  		var p2 = p.copy().trimStrings().build();
43  
44  		var in = (Object)" foo bar ";
45  		var e = (Object)"foo bar";
46  		var a = (Object)p.parse(s2.serialize(in), String.class);
47  		assertEquals(json(a), json(e));
48  		a = p2.parse(s.serialize(in), String.class);
49  		assertEquals(json(a), json(e));
50  
51  		in = JsonMap.ofJson("{' foo ': ' bar '}");
52  		e = JsonMap.ofJson("{foo:'bar'}");
53  		a = p.parse(s2.serialize(in), JsonMap.class);
54  		assertEquals(json(a), json(e));
55  		a = p2.parse(s.serialize(in), JsonMap.class);
56  		assertEquals(json(a), json(e));
57  
58  		in = new JsonList("[' foo ', {' foo ': ' bar '}]");
59  		e = new JsonList("['foo',{foo:'bar'}]");
60  		a = p.parse(s2.serialize(in), JsonList.class);
61  		assertEquals(json(a), json(e));
62  		a = p2.parse(s.serialize(in), JsonList.class);
63  		assertEquals(json(a), json(e));
64  
65  		in = new A().init1();
66  		e = new A().init2();
67  		a = p.parse(s2.serialize(in), A.class);
68  		assertEquals(json(a), json(e));
69  		a = p2.parse(s.serialize(in), A.class);
70  		assertEquals(json(a), json(e));
71  	}
72  
73  	public static class A {
74  		public String f1;
75  		public String[] f2;
76  		public JsonList f3;
77  		public JsonMap f4;
78  
79  		public A init1() throws Exception {
80  			f1 = " f1 ";
81  			f2 = new String[]{" f2a ", " f2b "};
82  			f3 = JsonList.ofJson("[' f3a ',' f3b ']");
83  			f4 = JsonMap.ofJson("{' foo ':' bar '}");
84  			return this;
85  		}
86  
87  		public A init2() throws Exception {
88  			f1 = "f1";
89  			f2 = new String[]{"f2a", "f2b"};
90  			f3 = JsonList.ofJson("['f3a','f3b']");
91  			f4 = JsonMap.ofJson("{'foo':'bar'}");
92  			return this;
93  		}
94  	}
95  }
