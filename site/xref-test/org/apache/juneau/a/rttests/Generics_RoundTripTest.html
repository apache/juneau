15  import static org.apache.juneau.TestUtils.*;
16  import static org.junit.jupiter.api.Assertions.*;

18  import org.apache.juneau.annotation.*;

22  /**
23   * Tests designed to serialize and parse objects to make sure we end up
24   * with the same objects for all serializers and parsers.
25   */
26  class Generics_RoundTripTest extends RoundTripTest_Base {
27  
28  	//====================================================================================================
29  	// testBeansWithUnboundTypeVars
30  	//====================================================================================================
31  
32  	@ParameterizedTest
33  	@MethodSource("testers")
34  	void a01_beansWithUnboundTypeVars(RoundTrip_Tester t) throws Exception {
35  
36  		if (t.returnOriginalObject)
37  			return;
38  
39  		// Unbound type variables should be interpreted as Object.
40  		// During parsing, these become JsonMaps.
41  		var x = new Pair<Object,Object>(new Source().init(), new Target().init());
42  		x = t.roundTrip(x);
43  		assertBean(x, "s{s1},t{t1}", "{a1},{b1}");
44  		assertEquals("JsonMap", x.getS().getClass().getSimpleName());
45  		assertEquals("JsonMap", x.getT().getClass().getSimpleName());
46  
47  		// If you specify a concrete class, the type variables become bound and
48  		// the property types correctly resolve.
49  		x = t.roundTrip(x, RealPair.class);
50  		assertBean(x, "s{s1},t{t1}", "{a1},{b1}");
51  		assertEquals("Source", x.getS().getClass().getSimpleName());
52  		assertEquals("Target", x.getT().getClass().getSimpleName());
53  	}
54  
55  	// Class with unbound type variables.
56  	@Bean(p="s,t")
57  	public static class Pair<S,T> {
58  
59  		public Pair() {}
60  
61  		public Pair(S s, T t) {
62  			this.s = s;
63  			this.t = t;
64  		}
65  
66  		// Getters/setters
67  		private S s;
68  		public S getS() { return s; }
69  		public void setS(S v) { s = v; }
70  
71  		private T t;
72  		public T getT() { return t; }
73  		public void setT(T v) { t = v; }
74  	}
75  
76  	// Sublcass with bound type variables.
77  	public static class RealPair extends Pair<Source,Target> {}
78  
79  	public static class Source {
80  		public String s1;
81  		public Source init() {
82  			s1 = "a1";
83  			return this;
84  		}
85  	}
86  
87  	public static class Target {
88  		public String t1;
89  		public Target init() {
90  			t1 = "b1";
91  			return this;
92  		}
93  	}
94  }
