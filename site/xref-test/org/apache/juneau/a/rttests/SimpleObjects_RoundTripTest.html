506 		x.put("b", false);
507 		x.put("c", null);
508 		x.put("d", "foo");
509 		x.put(null, "bar");
510 		x.put("null", "null");
511 		x.put("true", "true");
512 		x.put("false", "false");
513 		x.put("a\"a", "a\"a");
514 		x.put("b'b", "b'b");
515 		x.put("\"cc\"", "\"cc\"");
516 		x.put("'dd'", "'dd'");
517 		x.put("<ee>", "<ee>");
518 		x.put("<ff/>", "<ff/>");
519 		x.put("</gg>", "</gg>");
520 		x.put("<>", "<>");
521 		x.put("{}", "{}");
522 		x.put("[]", "[]");
523 		x.put("&", "&");

525 		x.put("/", "/");
526 		x.put("\b", "\b");
527 		x.put("\\b", "\\b");
528 		x.put("\n", "\n");

537 		x.put("\\u2345", "\\u2345");
538 		x.put("\\\u2345", "\\\u2345");
539 		x.put("<>{}[]&?/\b\n\t\f\\\\\u2345", "<>{}[]&?/\b\n\t\f\\\\\u2345");

552 		assertEquals("'dd'", x.get("'dd'"));
