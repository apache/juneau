42  	private static String serialize(HttpPartSchema schema, Object in) throws Exception {
43  		return DS.serialize(null, schema, in);
44  	}
45  
46  	private static <T> T parse(HttpPartSchema schema, String in, Class<T> c, Class<?>...args) throws Exception {
47  		return DP.parse(null, schema, in, DP.getClassMeta(c, args));
48  	}
