13  package org.apache.juneau.parser;
14  
15  import static org.apache.juneau.TestUtils.*;
16  import static org.junit.jupiter.api.Assertions.*;
17  
18  import org.apache.juneau.*;
19  import org.apache.juneau.json.*;
20  import org.junit.jupiter.api.*;
21  
22  class ParserSet_Test extends SimpleTestBase {
23  
24  	//====================================================================================================
25  	// Test parser group matching
26  	//====================================================================================================
27  	@Test void a01_parserGroupMatching() {
28  
29  		var s = ParserSet.create().add(Parser1.class, Parser2.class, Parser3.class).build();
30  		assertInstanceOf(Parser1.class, s.getParser("text/foo"));
31  		assertInstanceOf(Parser1.class, s.getParser("text/foo_a"));
32  		assertInstanceOf(Parser1.class, s.getParser("text/foo_a+xxx"));
33  		assertInstanceOf(Parser1.class, s.getParser("text/xxx+foo_a"));
34  		assertInstanceOf(Parser2.class, s.getParser("text/foo+bar"));
35  		assertInstanceOf(Parser2.class, s.getParser("text/foo+bar_a"));
36  		assertInstanceOf(Parser2.class, s.getParser("text/bar+foo"));
37  		assertInstanceOf(Parser2.class, s.getParser("text/bar+foo+xxx"));
38  		assertInstanceOf(Parser3.class, s.getParser("text/baz"));
39  		assertInstanceOf(Parser3.class, s.getParser("text/baz_a"));
40  		assertInstanceOf(Parser3.class, s.getParser("text/baz+yyy"));
41  		assertInstanceOf(Parser3.class, s.getParser("text/baz_a+yyy"));
42  		assertInstanceOf(Parser3.class, s.getParser("text/yyy+baz"));
43  		assertInstanceOf(Parser3.class, s.getParser("text/yyy+baz_a"));
44  	}
45  
46  
47  	public static class Parser1 extends JsonParser { public Parser1(JsonParser.Builder b) { super(b.consumes("text/foo,text/foo_a")); }}
48  	public static class Parser2 extends JsonParser { public Parser2(JsonParser.Builder b) { super(b.consumes("text/foo+bar,text/foo+bar_a")); }}
49  	public static class Parser3 extends JsonParser { public Parser3(JsonParser.Builder b) { super(b.consumes("text/baz,text/baz_a")); }}
50  
51  	//====================================================================================================
52  	// Test inheritence
53  	//====================================================================================================
54  	@Test void a02_inheritence() {
55  		var sb = ParserSet.create().add(P1.class, P2.class);
56  		var s = sb.build();
57  		assertList(s.getSupportedMediaTypes(), "text/1", "text/2", "text/2a");
58  
59  		sb = ParserSet.create().add(P1.class, P2.class).add(P3.class, P4.class);
60  		s = sb.build();
61  		assertList(s.getSupportedMediaTypes(), "text/3", "text/4", "text/4a", "text/1", "text/2", "text/2a");
62  
63  		sb = ParserSet.create().add(P1.class, P2.class).add(P3.class, P4.class).add(P5.class);
64  		s = sb.build();
65  		assertList(s.getSupportedMediaTypes(), "text/5", "text/3", "text/4", "text/4a", "text/1", "text/2", "text/2a");
66  	}
67  
68  	public static class P1 extends JsonParser { public P1(JsonParser.Builder b) { super(b.consumes("text/1")); }}
69  	public static class P2 extends JsonParser { public P2(JsonParser.Builder b) { super(b.consumes("text/2,text/2a")); }}
70  	public static class P3 extends JsonParser { public P3(JsonParser.Builder b) { super(b.consumes("text/3")); }}
71  	public static class P4 extends JsonParser { public P4(JsonParser.Builder b) { super(b.consumes("text/4,text/4a"));} }
72  	public static class P5 extends JsonParser { public P5(JsonParser.Builder b) { super(b.consumes("text/5"));}}
73  }
