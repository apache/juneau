17  package org.apache.juneau.jsonschema.annotation;
18  
19  import static org.apache.juneau.TestUtils.*;
20  import static org.junit.jupiter.api.Assertions.*;
21  
22  import org.apache.juneau.*;
23  import org.apache.juneau.annotation.*;
24  import org.junit.jupiter.api.*;
25  
26  class ExternalDocsAnnotation_Test extends TestBase {
27  
28  	//------------------------------------------------------------------------------------------------------------------
29  	// Basic tests
30  	//------------------------------------------------------------------------------------------------------------------
31  
32  	ExternalDocs a1 = ExternalDocsAnnotation.create()
33  		.description("a")
34  		.url("b")
35  		.build();
36  
37  	ExternalDocs a2 = ExternalDocsAnnotation.create()
38  		.description("a")
39  		.url("b")
40  		.build();
41  
42  	@Test void a01_basic() {
43  		assertBean(a1, "description,url", "[a],b");
44  	}
45  
46  	@Test void a02_testEquivalency() {
47  		assertEquals(a2, a1);
48  		assertNotEqualsAny(a1.hashCode(), 0, -1);
49  		assertEquals(a1.hashCode(), a2.hashCode());
50  	}
51  
52  	//------------------------------------------------------------------------------------------------------------------
53  	// PropertyStore equivalency.
54  	//------------------------------------------------------------------------------------------------------------------
55  
56  	@Test void b01_testEquivalencyInPropertyStores() {
57  		var bc1 = BeanContext.create().annotations(a1).build();
58  		var bc2 = BeanContext.create().annotations(a2).build();
59  		assertSame(bc1, bc2);
60  	}
61  
62  	//------------------------------------------------------------------------------------------------------------------
63  	// Other methods.
64  	//------------------------------------------------------------------------------------------------------------------
65  
66  	//------------------------------------------------------------------------------------------------------------------
67  	// Comparison with declared annotations.
68  	//------------------------------------------------------------------------------------------------------------------
69  
70  	@ExternalDocs(
71  		description="a",
72  		url="b"
73  	)
74  	public static class D1 {}
75  	ExternalDocs d1 = D1.class.getAnnotationsByType(ExternalDocs.class)[0];
76  
77  	@ExternalDocs(
78  		description="a",
79  		url="b"
80  	)
81  	public static class D2 {}
82  	ExternalDocs d2 = D2.class.getAnnotationsByType(ExternalDocs.class)[0];
83  
84  	@Test void d01_comparisonWithDeclarativeAnnotations() {
85  		assertEqualsAll(a1, d1, d2);
86  		assertNotEqualsAny(a1.hashCode(), 0, -1);
87  		assertEqualsAll(a1.hashCode(), d1.hashCode(), d2.hashCode());
88  	}
89  }
