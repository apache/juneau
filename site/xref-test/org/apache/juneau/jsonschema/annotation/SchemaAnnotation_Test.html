186 		var c1 = SchemaAnnotation.create(C1.class).on(C2.class).build();
187 		var c2 = SchemaAnnotation.create("a").on("b").build();
188 		var c3 = SchemaAnnotation.create().on(C1.class.getField("f1")).on(C2.class.getField("f2")).build();
189 		var c4 = SchemaAnnotation.create().on(C1.class.getMethod("m1")).on(C2.class.getMethod("m2")).build();
190 
191 		assertBean(c1, "on", "["+CNAME+"$C1,"+CNAME+"$C2]");
192 		assertBean(c2, "on", "[a,b]");
193 		assertBean(c3, "on", "["+CNAME+"$C1.f1,"+CNAME+"$C2.f2]");
194 		assertBean(c4, "on", "["+CNAME+"$C1.m1(),"+CNAME+"$C2.m2()]");

445 	@Test void f01_backwardCompatibility_exclusiveMaxMin() {
446 		// Test that old boolean exclusiveMaximum/exclusiveMinimum still work
447 		Schema oldStyle = SchemaAnnotation.create()
448 			.exclusiveMaximum(true)
449 			.exclusiveMinimum(true)
450 			.maximum("100")
451 			.minimum("0")
452 			.build();
453 		
454 		assertBean(oldStyle, "exclusiveMaximum,exclusiveMinimum,maximum,minimum", "true,true,100,0");
455 		
456 		// Test that new numeric style takes precedence
457 		Schema newStyle = SchemaAnnotation.create()
458 			.exclusiveMaximumValue("100")
459 			.exclusiveMinimumValue("0")
460 			.build();
461 		
462 		assertBean(newStyle, "exclusiveMaximumValue,exclusiveMinimumValue", "100,0");
463 		
464 		// Test that new style takes precedence when both are set
465 		Schema mixed = SchemaAnnotation.create()
466 			.exclusiveMaximum(false)
467 			.exclusiveMinimum(false)
468 			.exclusiveMaximumValue("100")
469 			.exclusiveMinimumValue("0")
470 			.build();
471 		
472 		assertBean(mixed, "exclusiveMaximum,exclusiveMinimum,exclusiveMaximumValue,exclusiveMinimumValue", "false,false,100,0");
473 	}
