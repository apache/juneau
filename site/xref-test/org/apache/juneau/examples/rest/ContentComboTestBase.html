41  @FixMethodOrder(NAME_ASCENDING)
42  public class ContentComboTestBase extends RestTestcase {
43  
44  	// Reusable RestClients keyed by label that live for the duration of a testcase class.
45  	private static Map<String,RestClient> clients = CollectionUtils.map();
46  
47  	protected RestClient getClient(MediaType mediaType) {
48  		String mt = mediaType.toString();
49  		return switch (mt) {
50  			case "text/csv" -> getClient(mt, CsvSerializer.DEFAULT, CsvParser.DEFAULT);
51  			case "text/html" -> getClient(mt, HtmlSerializer.DEFAULT, HtmlParser.DEFAULT);
52  			case "application/json" -> getClient(mt, JsonSerializer.DEFAULT, JsonParser.DEFAULT);
53  			case "octal/msgpack" -> getClient(mt, MsgPackSerializer.DEFAULT, MsgPackParser.DEFAULT, x -> x.queryData("plainText","true"));
54  			case "text/plain" -> getClient(mt, PlainTextSerializer.DEFAULT, PlainTextParser.DEFAULT);
55  			case "text/uon" -> getClient(mt, UonSerializer.DEFAULT, UonParser.DEFAULT);
56  			case "application/x-www-form-urlencoded" -> getClient(mt, UrlEncodingSerializer.DEFAULT, UrlEncodingParser.DEFAULT);
57  			case "text/xml" -> getClient(mt, XmlSerializer.DEFAULT, XmlParser.DEFAULT);
58  			default -> throw new BasicRuntimeException("Client for mediaType ''{0}'' not found", mt);
59  		};
60  	}
61  
62  	protected RestClient getClient(String label, Serializer serializer, Parser parser, Consumer<RestClient.Builder>...postApply) {
63  		if (! clients.containsKey(label)) {
64  			RestClient.Builder b = SamplesMicroservice.client(serializer, parser);
65  			for (Consumer<RestClient.Builder> c : postApply)
66  				c.accept(b);
67  			clients.put(label, b.build());
68  		}
69  		return clients.get(label);
70  	}
71  
72  	@AfterClass
73  	public static void tearDown() {
74  		clients.values().forEach(rc -> {
75  			try {
76  				rc.close();
77  			} catch (IOException e) {
78  				e.printStackTrace();
79  			}
80  		});
81  		clients.clear();
82  	}
83  
84  	public static class ComboInput {
85  		public final String name, url;
86  		public final MediaType mediaType;
87  		public final String[] expectedResults;
88  
89  		public ComboInput(String name, String url, MediaType mediaType, String...expectedResults) {
90  			this.name = name;
91  			this.url = url;
92  			this.mediaType = mediaType;
93  			this.expectedResults = expectedResults;
94  		}
95  	}
96  
97  	private final ComboInput comboInput;
98  
99  	public ContentComboTestBase(ComboInput comboInput) {
100 		this.comboInput = comboInput;
101 	}
102 
103 	@Test
104 	public void doTest() throws Exception {
105 		RestClient rc = getClient(comboInput.mediaType);
106 		String s = rc.get(comboInput.url).run().getContent().asString();
107 		assertContains(s, comboInput.expectedResults);
108 	}
109 }
