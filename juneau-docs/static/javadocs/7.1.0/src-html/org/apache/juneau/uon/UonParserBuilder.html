055   /**
056    * Configuration property: Decode <js>"%xx"</js> sequences.
057    * 
058    * <p>
059    * Specify <jk>true</jk> if URI encoded characters should be decoded, <jk>false</jk> if they've already been
060    * decoded before being passed to this parser.
061    * 
062    * <h5 class='section'>See Also:</h5>
063    * <ul>
064    *    <li class='jf'>{@link UonParser#UON_decoding}
065    * </ul>
066    * 
067    * @param value 
068    *    The new value for this property.
069    *    <br>Default is <jk>false</jk> for {@link UonParser}, <jk>true</jk> for {@link UrlEncodingParser}
070    * @return This object (for method chaining).
071    */
072   public UonParserBuilder decoding(boolean value) {
073      return set(UON_decoding, value);
074   }
075
076   /**
077    * Configuration property: Decode <js>"%xx"</js> sequences.
078    * 
079    * <p>
080    * Shortcut for calling <code>decodeChars(<jk>true</jk>)</code>.
081    * 
082    * <h5 class='section'>See Also:</h5>
083    * <ul>
084    *    <li class='jf'>{@link UonParser#UON_decoding}
085    * </ul>
086    * 
087    * @return This object (for method chaining).
088    */
089   public UonParserBuilder decoding() {
090      return decoding(true);
091   }
092
093   /**
094    * Configuration property:  Validate end.
095    * 
096    * <p>
097    * If <jk>true</jk>, after parsing a POJO from the input, verifies that the remaining input in 
098    * the stream consists of only whitespace.
099    * 
100    * <h5 class='section'>See Also:</h5>
101    * <ul>
102    *    <li class='jf'>{@link UonParser#UON_validateEnd}
103    * </ul>
104    * 
105    * @param value 
106    *    The new value for this property.
107    *    <br>The default value is <jk>false</jk>.
108    * @return This object (for method chaining).
109    */
110   public UonParserBuilder validateEnd(boolean value) {
111      return set(UON_validateEnd, value);
112   }
113   
114   /**
115    * Configuration property:  Validate end.
116    * 
117    * <p>
118    * Shortcut for calling <code>validateEnd(<jk>true</jk>)</code>.
119    * 
120    * <h5 class='section'>See Also:</h5>
121    * <ul>
122    *    <li class='jf'>{@link UonParser#UON_validateEnd}
123    * </ul>
124    * 
125    * @return This object (for method chaining).
126    */
127   public UonParserBuilder validateEnd() {
128      return set(UON_validateEnd, true);
129   }
