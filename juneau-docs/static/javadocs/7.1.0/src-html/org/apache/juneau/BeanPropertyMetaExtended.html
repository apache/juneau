013package org.apache.juneau;
014
015/**
016 * Defines extended language-specific metadata associated with a bean property.
017 * 
018 * <p>
019 * Serializers and parsers can implement their own language-specific metadata on a bean property and retrieve
020 * the metadata using the {@link BeanPropertyMeta#getExtendedMeta(Class)} method.
021 */
022public class BeanPropertyMetaExtended {
023
024   private final BeanPropertyMeta bpm;
025
026   /**
027    * Constructor.
028    * 
029    * @param bpm The metadata of the bean property we're extending.
030    * @throws BeanRuntimeException If any error occurred trying to construct the metadata.
031    */
032   public BeanPropertyMetaExtended(BeanPropertyMeta bpm) throws BeanRuntimeException {
033      this.bpm = bpm;
034   }
035
036   /**
037    * Returns the bean property metadata that was passed into the constructor.
038    * 
039    * @return The bean property metadata that was passed into the constructor.
040    */
041   protected BeanPropertyMeta getBeanPropertyMeta() {
042      return bpm;
043   }
044}
