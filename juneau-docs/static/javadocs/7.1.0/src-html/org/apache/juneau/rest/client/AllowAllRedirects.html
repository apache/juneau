013package org.apache.juneau.rest.client;
014
015import org.apache.http.impl.client.*;
016
017/**
018 * Redirect strategy that allows for redirects on any request type, not just <code>GET</code> or <code>HEAD</code>.
019 * 
020 * <h5 class='section'>Notes:</h5>
021 * <ul class='spaced-list'>
022 *    <li>
023 *       This class is similar to <code>org.apache.http.impl.client.LaxRedirectStrategy</code>
024 *       in Apache HttpClient 4.2, but also allows for redirects on <code>PUTs</code> and <code>DELETEs</code>.
025 * </ul>
026 */
027public class AllowAllRedirects extends DefaultRedirectStrategy {
028
029   @Override /* DefaultRedirectStrategy */
030   protected boolean isRedirectable(final String method) {
031      return true;
032   }
033}
