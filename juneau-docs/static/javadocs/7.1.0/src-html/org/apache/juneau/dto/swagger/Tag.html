013package org.apache.juneau.dto.swagger;
014
015import static org.apache.juneau.internal.BeanPropertyUtils.*;
016import org.apache.juneau.annotation.*;
017
018/**
019 * Allows adding meta data to a single tag that is used by the <a class="doclink" href="http://swagger.io/specification/#operationObject">Operation Object</a>.
020 * 
021 * <p>
022 * It is not mandatory to have a Tag Object per tag used there.
023 * 
024 * <h5 class='section'>Example:</h5>
025 * <p class='bcode'>
026 *    <jc>// Construct using SwaggerBuilder.</jc>
027 *    Tag x = <jsm>tag</jsm>()
028 *       .name(<js>"pet"</js>)
029 *       .description(<js>"Pets operations"</js>)
030 * 
031 *    <jc>// Serialize using JsonSerializer.</jc>
032 *    String json = JsonSerializer.<jsf>DEFAULT</jsf>.toString(x);
033 * 
034 *    <jc>// Or just use toString() which does the same as above.</jc>
035 *    String json = x.toString();
036 * </p>
037 * <p class='bcode'>
038 *    <jc>// Output</jc>
039 *    {
040 *       <js>"name"</js>: <js>"pet"</js>,
041 *       <js>"description"</js>: <js>"Pets operations"</js>
042 *    }
043 * </p>
044 * 
045 * <h5 class='section'>See Also:</h5>
046 * <ul class='doctree'>
047 *    <li class='link'><a class='doclink' href='../../../../../overview-summary.html#juneau-dto.Swagger'>Overview &gt; juneau-dto &gt; Swagger</a>
048 * </ul>
049 */
050@Bean(properties="name,description,externalDocs,*")
051public class Tag extends SwaggerElement {
052
053   private String 
054      name,
055      description;
056   private ExternalDocumentation externalDocs;
057
058   /**
059    * Bean property getter:  <property>name</property>.
060    * 
061    * <p>
062    * The name of the tag.
063    * 
064    * @return The property value, or <jk>null</jk> if it is not set.
065    */
066   public String getName() {
067      return name;
068   }
069
070   /**
071    * Bean property setter:  <property>name</property>.
072    * 
073    * <p>
074    * The name of the tag.
075    * 
076    * @param value 
077    *    The new value for this property.
078    *    <br>Property value is required.
079    * @return This object (for method chaining).
080    */
081   public Tag setName(String value) {
082      name = value;
083      return this;
084   }
085
086   /**
087    * Same as {@link #setName(String)}.
088    * 
089    * @param value
090    *    The new value for this property.
091    *    <br>Non-String values will be converted to String using <code>toString()</code>.
092    *    <br>Can be <jk>null</jk> to unset the property.
093    * @return This object (for method chaining).
094    */
095   public Tag name(Object value) {
096      return setName(toStringVal(value));
097   }
098
099   /**
100    * Bean property getter:  <property>description</property>.
101    * 
102    * <p>
103    * A short description for the tag.
104    * 
105    * @return The property value, or <jk>null</jk> if it is not set.
106    */
107   public String getDescription() {
108      return description;
109   }
110
111   /**
112    * Bean property setter:  <property>description</property>.
113    * 
114    * <p>
115    * A short description for the tag.
116    * 
117    * @param value
118    *    The new value for this property.
119    *    <br><a class="doclink" href="https://help.github.com/articles/github-flavored-markdown">GFM syntax</a> can be used for rich text representation.
120    *    <br>Can be <jk>null</jk> to unset the property.
121    * @return This object (for method chaining).
122    */
123   public Tag setDescription(String value) {
124      description = value;
125      return this;
126   }
127
128   /**
129    * Same as {@link #setDescription(String)}.
130    * 
131    * @param value
132    *    The new value for this property.
133    *    <br>Non-String values will be converted to String using <code>toString()</code>.
134    *    <br>Can be <jk>null</jk> to unset the property.
135    * @return This object (for method chaining).
136    */
137   public Tag description(Object value) {
138      return setDescription(toStringVal(value));
139   }
140
141   /**
142    * Bean property getter:  <property>externalDocs</property>.
143    * 
144    * <p>
145    * Additional external documentation for this tag.
146    * 
147    * @return The property value, or <jk>null</jk> if it is not set.
148    */
149   public ExternalDocumentation getExternalDocs() {
150      return externalDocs;
151   }
152
153   /**
154    * Bean property setter:  <property>externalDocs</property>.
155    * 
156    * <p>
157    * Additional external documentation for this tag.
158    * 
159    * @param value
160    *    The new value for this property.
161    *    <br>Can be <jk>null</jk> to unset the property.
162    * @return This object (for method chaining).
163    */
164   public Tag setExternalDocs(ExternalDocumentation value) {
165      externalDocs = value;
166      return this;
167   }
168
169   /**
170    * Same as {@link #setExternalDocs(ExternalDocumentation)}.
171    * 
172    * @param value 
173    *    The new value for this property.
174    *    <br>Valid types:
175    *    <ul>
176    *       <li>{@link ExternalDocumentation}
177    *       <li><code>String</code> - JSON object representation of {@link ExternalDocumentation}
178    *          <h5 class='figure'>Example:</h5>
179    *          <p class='bcode'>
180    *    externalDocs(<js>"{description:'description',url:'url'}"</js>);
181    *          </p>
182    *    </ul>
183    *    <br>Can be <jk>null</jk> to unset the property.
184    * @return This object (for method chaining).
185    */
186   public Tag externalDocs(Object value) {
187      return setExternalDocs(toType(value, ExternalDocumentation.class));
188   }
189
190   @Override /* SwaggerElement */
191   public <T> T get(String property, Class<T> type) {
192      if (property == null)
193         return null;
194      switch (property) {
195         case "name": return toType(getName(), type);
196         case "description": return toType(getDescription(), type);
197         case "externalDocs": return toType(getExternalDocs(), type);
198         default: return super.get(property, type);
199      }
200   }
201
202   @Override /* SwaggerElement */
203   public Tag set(String property, Object value) {
204      if (property == null)
205         return this;
206      switch (property) {
207         case "name": return name(value);
208         case "description": return description(value);
209         case "externalDocs": return externalDocs(value);
210         default: 
211            super.set(property, value);
212            return this;
213      }
214   }
215}
