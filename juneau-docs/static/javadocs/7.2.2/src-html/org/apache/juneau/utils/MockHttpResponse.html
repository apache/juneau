013package org.apache.juneau.utils;
014
015import java.util.*;
016
017/**
018 * Represent the basic interface for an HTTP response.
019 *
020 * <p>
021 * Used as a shim between the server and client APIs that allow the <code>RestClient</code>
022 * class to send and receive mocked requests using the <code>MockRest</code> interface.
023 */
024public interface MockHttpResponse {
025
026   /**
027    * Returns the status code of the response.
028    *
029    * @return The status code of the response.
030    */
031   int getStatus();
032
033   /**
034    * Returns the status message of the response.
035    *
036    * @return The status message of the response.
037    */
038   String getMessage();
039
040   /**
041    * Returns the headers of the response.
042    *
043    * @return The headers of the response.
044    */
045   Map<String,String[]> getHeaders();
046
047   /**
048    * Returns the body of the response.
049    *
050    * @return The body of the response.
051    */
052   byte[] getBody();
053}
