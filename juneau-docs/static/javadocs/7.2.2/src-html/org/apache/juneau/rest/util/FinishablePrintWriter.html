013package org.apache.juneau.rest.util;
014
015import java.io.*;
016
017import org.apache.juneau.encoders.*;
018
019/**
020 * A wrapped {@link PrintWriter} with an added <code>finish()</code> method.
021 */
022public class FinishablePrintWriter extends PrintWriter implements Finishable {
023
024   final Finishable f;
025
026   /**
027    * Constructor.
028    *
029    * @param out The wrapped output stream.
030    * @param characterEncoding The character encoding of the output stream.
031    * @param autoFlush Automatically flush after every println.
032    * @throws IOException
033    */
034   public FinishablePrintWriter(OutputStream out, String characterEncoding, boolean autoFlush) throws IOException {
035      super(new OutputStreamWriter(out, characterEncoding), autoFlush);
036      f = (out instanceof Finishable ? (Finishable)out : null);
037   }
038
039   /**
040    * @deprecated Use {@link #FinishablePrintWriter(OutputStream, String, boolean)}
041    */
042   @SuppressWarnings("javadoc")
043   @Deprecated
044   public FinishablePrintWriter(OutputStream out, String characterEncoding) throws IOException {
045      this(out, characterEncoding, false);
046   }
047
048   /**
049    * Calls {@link Finishable#finish()} on the underlying output stream.
050    *
051    * <p>
052    * A no-op if the underlying output stream does not implement the {@link Finishable} interface.
053    */
054   @Override /* Finishable */
055   public void finish() throws IOException {
056      if (f != null)
057         f.finish();
058   }
059}
