013package org.apache.juneau.jsonschema.annotation;
014
015import static java.lang.annotation.ElementType.*;
016import static java.lang.annotation.RetentionPolicy.*;
017
018import java.lang.annotation.*;
019
020/**
021 * @deprecated Use {@link Schema}
022 */
023@Documented
024@Target({TYPE,METHOD,FIELD})
025@Retention(RUNTIME)
026@Inherited
027@Deprecated
028public @interface JsonSchema {
029
030   /**
031    * Defines the type for the class or property.
032    */
033   String type() default "";
034
035   /**
036    * Defines the format for the class or property.
037    */
038   String format() default "";
039
040   /**
041    * Defines the description for the class or property.
042    */
043   String description() default "";
044
045   /**
046    * Defines the example for the class or property.
047    *
048    * <p>
049    * The format of the value is Lax-JSON.
050    */
051   String example() default "";
052}
