013package org.apache.juneau.html;
014
015import org.apache.juneau.jsonschema.*;
016import org.apache.juneau.serializer.*;
017
018/**
019 * Context object that lives for the duration of a single serialization of {@link HtmlSchemaSerializer} and its subclasses.
020 *
021 * <p>
022 * This class is NOT thread safe.  It is meant to be discarded after one-time use.
023 */
024public class HtmlSchemaSerializerSession extends HtmlSerializerSession {
025
026   private final JsonSchemaGeneratorSession genSession;
027
028   /**
029    * Create a new session using properties specified in the context.
030    *
031    * @param ctx
032    *    The context creating this session object.
033    *    The context contains all the configuration settings for this object.
034    * @param args
035    *    Runtime arguments.
036    */
037   protected HtmlSchemaSerializerSession(HtmlSchemaSerializer ctx, SerializerSessionArgs args) {
038      super(ctx, args);
039      this.genSession = ctx.getGenerator().createSession(args);
040   }
041
042   @Override /* SerializerSession */
043   protected void doSerialize(SerializerPipe out, Object o) throws Exception {
044      super.doSerialize(out, genSession.getSchema(o));
045   }
046}
