013package org.apache.juneau.rest.client.remote;
014
015import static java.lang.annotation.ElementType.*;
016import static java.lang.annotation.RetentionPolicy.*;
017
018import java.lang.annotation.*;
019
020/**
021 * Identifies a proxy against a REST interface.
022 *
023 * <h5 class='section'>See Also:</h5>
024 * <ul class='doctree'>
025 *    <li class='link'>{@doc juneau-rest-client.RestProxies}
026 * </ul>
027 */
028@Documented
029@Target({TYPE})
030@Retention(RUNTIME)
031@Inherited
032public @interface RemoteResource {
033
034   /**
035    * REST service path.
036    *
037    * <p>
038    * The possible values are:
039    * <ul class='spaced-list'>
040    *    <li>An absolute URL.
041    *    <li>A relative URL interpreted as relative to the root URL defined on the <code>RestClient</code>
042    *    <li>No path interpreted as the class name (e.g. <js>"http://localhost/root-url/org.foo.MyInterface"</js>)
043    * </ul>
044    */
045   String path() default "";
046}
