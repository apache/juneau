013package org.apache.juneau.examples.rest.petstore;
014
015import org.apache.juneau.http.annotation.*;
016import org.apache.juneau.rest.exception.*;
017
018/**
019 * Exception thrown when trying to add an entry where the ID is already in use.
020 */
021@SuppressWarnings("serial")
022@Response(description="ID not found")
023public class IdNotFound extends NotFound {
024
025   /**
026    * Constructor.
027    *
028    * @param id The duplicate ID.
029    * @param c The object type..
030    */
031   public IdNotFound(Object id, Class<?> c) {
032      super("ID ''{0}'' not found for type ''{1}''", id, c.getSimpleName());
033   }
034}
