013package org.apache.juneau.examples.rest.petstore.rest;
014
015import static org.apache.juneau.http.HttpMethodName.*;
016
017import java.util.*;
018
019import org.apache.juneau.jsonschema.annotation.Items;
020import org.apache.juneau.*;
021import org.apache.juneau.examples.rest.petstore.*;
022import org.apache.juneau.examples.rest.petstore.dto.*;
023import org.apache.juneau.http.annotation.*;
024import org.apache.juneau.rest.*;
025import org.apache.juneau.rest.client.remote.*;
026import org.apache.juneau.rest.exception.*;
027import org.apache.juneau.rest.response.*;
028
029/**
030 * Defines the interface for both the server-side and client-side pet store application.
031 */
032@RemoteResource(path="/petstore")
033public interface PetStore {
034
035   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
036   // Pets
037   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
038
039   @RemoteMethod(method=GET, path="/pet")
040   public Collection<Pet> getPets() throws NotAcceptable;
041
042   @RemoteMethod(path="/pet/{petId}") /* method inferred from method name */
043   public Pet getPet(
044      @Path(
045         name="petId",
046         description="ID of pet to return",
047         example="123"
048      )
049      long petId
050   ) throws IdNotFound, NotAcceptable;
051
052   @RemoteMethod /* method and path inferred from method name */
053   public long postPet(
054      @Body(
055         description="Pet object to add to the store"
056      ) CreatePet pet
057   ) throws IdConflict, NotAcceptable, UnsupportedMediaType;
058
059   @RemoteMethod(method=PUT, path="/pet/{petId}")
060   public Ok updatePet(
061      @Body(
062         description="Pet object that needs to be added to the store"
063      ) UpdatePet pet
064   ) throws IdNotFound, NotAcceptable, UnsupportedMediaType;
065
066   @RemoteMethod(method=GET, path="/pet/findByStatus")
067   public Collection<Pet> findPetsByStatus(
068      @Query(
069         name="status",
070         description="Status values that need to be considered for filter.",
071         required=true,
072         type="array",
073         collectionFormat="csv",
074         items=@Items(
075            type="string",
076            _enum="AVAILABLE,PENDING,SOLD",
077            _default="AVAILABLE"
078         ),
079         example="AVALIABLE,PENDING"
080      )
081      PetStatus[] status
082   ) throws NotAcceptable;
083
084   @RemoteMethod(method=GET, path="/pet/findByTags")
085   @Deprecated
086   public Collection<Pet> findPetsByTags(
087      @Query(
088         name="tags",
089         description="Tags to filter by",
090         required=true,
091         example="['tag1','tag2']"
092      )
093      String[] tags
094   ) throws InvalidTag, NotAcceptable;
095
096   @RemoteMethod(method=DELETE, path="/pet/{petId}")
097   public Ok deletePet(
098      @Header(
099         name="api_key",
100         description="Security API key",
101         required=true,
102         example="foobar"
103      )
104      String apiKey,
105      @Path(
106         name="petId",
107         description="Pet id to delete",
108         example="123"
109      )
110      long petId
111   ) throws IdNotFound, NotAcceptable;
112
113   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
114   // Orders
115   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
116
117   @RemoteMethod(method=GET, path="/store/order")
118   public Collection<Order> getOrders() throws NotAcceptable;
119
120   @RemoteMethod(method=GET, path="/store/order/{orderId}")
121   public Order getOrder(
122      @Path(
123         name="orderId",
124         description="ID of order to fetch",
125         maximum="1000",
126         minimum="1",
127         example="123"
128      )
129      long orderId
130   ) throws InvalidId, IdNotFound, NotAcceptable;
131
132   @RemoteMethod(method=POST, path="/store/order")
133   public long placeOrder(
134      @FormData(
135         name="petId",
136         description="Pet ID"
137      )
138      long petId,
139      @FormData(
140         name="username",
141         description="The username of the user creating the order"
142      )
143      String username
144   ) throws IdConflict, NotAcceptable, UnsupportedMediaType;
145
146   @RemoteMethod(method=DELETE, path="/store/order/{orderId}")
147   public Ok deleteOrder(
148      @Path(
149         name="orderId",
150         description="ID of the order that needs to be deleted",
151         minimum="1",
152         example="5"
153      )
154      long orderId
155   ) throws InvalidId, IdNotFound, NotAcceptable;
156
157   @RemoteMethod(method=GET, path="/store/inventory")
158   public Map<PetStatus,Integer> getStoreInventory() throws NotAcceptable;
159
160   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
161   // Users
162   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
163
164   @RemoteMethod(method=GET, path="/user")
165   public Collection<User> getUsers() throws NotAcceptable;
166
167   @RemoteMethod(method=GET, path="/user/{username}")
168   public User getUser(
169      @Path(
170         name="username",
171         description="The name that needs to be fetched. Use user1 for testing."
172      )
173      String username
174   ) throws InvalidUsername, IdNotFound, NotAcceptable;
175
176   @RemoteMethod
177   public Ok postUser(
178      @Body(
179         description="Created user object"
180      )
181      User user
182   ) throws InvalidUsername, IdConflict, NotAcceptable, UnsupportedMediaType;
183
184   @RemoteMethod(method=POST, path="/user/createWithArray")
185   public Ok createUsers(
186      @Body(
187         description="List of user objects"
188      )
189      User[] users
190   ) throws InvalidUsername, IdConflict, NotAcceptable, UnsupportedMediaType;
191
192   @RemoteMethod(method=PUT, path="/user/{username}")
193   public Ok updateUser(
194      @Path(
195         name="username",
196         description="Name that need to be updated"
197      )
198      String username,
199      @Body(
200         description="Updated user object"
201      )
202      User user
203   ) throws InvalidUsername, IdNotFound, NotAcceptable, UnsupportedMediaType;
204
205   @RemoteMethod(method=DELETE, path="/user/{username}")
206   public Ok deleteUser(
207      @Path(
208         name="username",
209         description="The name that needs to be deleted"
210      )
211      String username
212   ) throws InvalidUsername, IdNotFound, NotAcceptable;
213
214   @RemoteMethod(method=GET, path="/user/login")
215   public Ok login(
216      @Query(
217         name="username",
218         description="The username for login",
219         required=true,
220         example="myuser"
221      )
222      String username,
223      @Query(
224         name="password",
225         description="The password for login in clear text",
226         required=true,
227         example="abc123"
228      )
229      String password,
230      @ResponseHeader(
231         name="X-Rate-Limit",
232         type="integer",
233         format="int32",
234         description="Calls per hour allowed by the user.",
235         example="123"
236      )
237      Value<Integer> rateLimit,
238      Value<ExpiresAfter> expiresAfter,
239      RestRequest req,
240      RestResponse res
241   ) throws InvalidLogin, NotAcceptable;
242
243   @RemoteMethod(method=GET, path="/user/logout")
244   public Ok logout() throws NotAcceptable;
245}
