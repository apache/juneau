013package org.apache.juneau.rest.vars;
014
015import org.apache.juneau.rest.*;
016import org.apache.juneau.svl.*;
017
018/**
019 * Swagger attribute variable resolver.
020 *
021 * <p>
022 * The format for this var is <js>"$SS{key1[,key2...]}"</js>.
023 * <br>When multiple keys are used, returns the first non-null/empty value.
024 *
025 * <p>
026 * Example values:
027 * <ul>
028 *    <li><js>"info/title"</js>
029 *    <li><js>"info/description"</js>
030 *    <li><js>"info/contact/name"</js>
031 * </ul>
032 *
033 * <h5 class='section'>Notes:</h5>
034 * <ul class='spaced-list'>
035 *    <li>
036 *       This variable resolver requires that a {@link RestRequest} object be set as a context object on the resolver
037 *       or a session object on the resolver session.
038 *    <li>
039 *       For security reasons, nested and recursive variables are not resolved.
040 * </ul>
041 *
042 * <h5 class='section'>See Also:</h5>
043 * <ul>
044 *    <li class='link'>{@doc juneau-rest-server.SvlVariables}
045 * </ul>
046 */
047public class SwaggerVar extends MultipartResolvingVar {
048
049   /**
050    * The name of the session or context object that identifies the {@link RestRequest} object.
051    */
052   public static final String SESSION_req = "req";
053
054
055   /** The name of this variable. */
056   public static final String NAME = "SS";
057
058   /**
059    * Constructor.
060    */
061   public SwaggerVar() {
062      super(NAME);
063   }
064
065   @Override /* Var */
066   protected boolean allowNested() {
067      return false;
068   }
069
070   @Override /* Var */
071   protected boolean allowRecurse() {
072      return false;
073   }
074
075   @Override /* Parameter */
076   public String resolve(VarResolverSession session, String key) {
077      RestRequest req = session.getSessionObject(RestRequest.class, SESSION_req, true);
078      return req.getProperties().getString(key);
079   }
080}
