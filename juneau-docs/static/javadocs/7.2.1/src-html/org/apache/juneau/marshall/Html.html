013package org.apache.juneau.marshall;
014
015import org.apache.juneau.html.*;
016
017/**
018 * A pairing of a {@link HtmlSerializer} and {@link HtmlParser} into a single class with convenience read/write methods.
019 *
020 * <p>
021 *    The general idea is to combine a single serializer and parser inside a simplified API for reading and writing POJOs.
022 *
023 * <h5 class='figure'>Examples:</h5>
024 * <p class='bcode w800'>
025 *    <jc>// Using instance.</jc>
026 *    Html html = <jk>new</jk> Html();
027 *    MyPojo myPojo = html.read(string, MyPojo.<jk>class</jk>);
028 *    String string = html.write(myPojo);
029 * </p>
030 * <p class='bcode w800'>
031 * <jc>// Using DEFAULT instance.</jc>
032 *    MyPojo myPojo = Html.<jsf>DEFAULT</jsf>.read(string, MyPojo.<jk>class</jk>);
033 *    String string = Html.<jsf>DEFAULT</jsf>.write(myPojo);
034 * </p>
035 *
036 * <h5 class='section'>See Also:</h5>
037 * <ul>
038 *    <li class='link'>{@doc juneau-marshall.Marshalls}
039 * </ul>
040 */
041public class Html extends CharMarshall {
042
043   /**
044    * Default reusable instance.
045    */
046   public static final Html DEFAULT = new Html();
047
048   /**
049    * Constructor.
050    *
051    * @param s
052    *    The serializer to use for serializing output.
053    *    <br>Must not be <jk>null</jk>.
054    * @param p
055    *    The parser to use for parsing input.
056    *    <br>Must not be <jk>null</jk>.
057    */
058   public Html(HtmlSerializer s, HtmlParser p) {
059      super(s, p);
060   }
061
062   /**
063    * Constructor.
064    *
065    * <p>
066    * Uses {@link HtmlSerializer#DEFAULT} and {@link HtmlParser#DEFAULT}.
067    */
068   public Html() {
069      this(HtmlSerializer.DEFAULT, HtmlParser.DEFAULT);
070   }
071}
