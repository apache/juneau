013package org.apache.juneau.plaintext;
014
015import static org.apache.juneau.internal.IOUtils.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.parser.*;
019
020/**
021 * Session object that lives for the duration of a single use of {@link PlainTextParser}.
022 *
023 * <p>
024 * This class is NOT thread safe.
025 * It is typically discarded after one-time use although it can be reused against multiple inputs.
026 */
027public class PlainTextParserSession extends ReaderParserSession {
028
029   /**
030    * Create a new session using properties specified in the context.
031    *
032    * @param ctx
033    *    The parser creating this session object.
034    *    The parser contains all the configuration settings for this object.
035    * @param args
036    *    Runtime session arguments.
037    */
038   protected PlainTextParserSession(PlainTextParser ctx, ParserSessionArgs args) {
039      super(ctx, args);
040   }
041
042   @Override /* ParserSession */
043   protected <T> T doParse(ParserPipe pipe, ClassMeta<T> type) throws Exception {
044      return convertToType(read(pipe.getReader()), type);
045   }
046}
