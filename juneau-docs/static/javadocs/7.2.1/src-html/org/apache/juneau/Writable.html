013package org.apache.juneau;
014
015import java.io.*;
016
017import org.apache.juneau.http.*;
018
019/**
020 * Interface that identifies that an object can be serialized directly to a writer.
021 *
022 * <p>
023 * Instances must identify the media type of the content by implementing the {@link #getMediaType()} method.
024 */
025public interface Writable {
026
027   /**
028    * Serialize this object to the specified writer.
029    *
030    * @param w The writer to write to.
031    * @return The same writer passed in.
032    * @throws IOException
033    */
034   Writer writeTo(Writer w) throws IOException;
035
036   /**
037    * Returns the serialized media type for this resource (e.g. <js>"text/html"</js>)
038    *
039    * @return The media type, or <jk>null</jk> if the media type is not known.
040    */
041   MediaType getMediaType();
042}
