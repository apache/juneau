023/**
024 * Sample REST resource showing how to implement a "router" resource page.
025 *
026 * <ul class='seealso'>
027 *    <li class='extlink'>{@source}
028 * </ul>
029 */
030@RestResource(
031   path="/*",
032   title="Root resources",
033   description="Example of a router resource page.",
034   children={
035      HelloWorldResource.class,
036      DtoExamples.class,
037      ConfigResource.class,
038      LogsResource.class,
039      ShutdownResource.class
040   }
041)
042@HtmlDocConfig(
043   widgets={
044      ContentTypeMenuItem.class,
045      ThemeMenuItem.class
046   },
047   navlinks={
048      "options: ?method=OPTIONS",
049      "$W{ContentTypeMenuItem}",
050      "$W{ThemeMenuItem}",
051      "source: $C{Source/gitHub}/org/apache/juneau/examples/rest/$R{servletClassSimple}.java"
052   },
053   aside={
054      "<div style='max-width:400px' class='text'>",
055      "  <p>This is an example of a 'router' page that serves as a jumping-off point to child resources.</p>",
056      "  <p>Resources can be nested arbitrarily deep through router pages.</p>",
057      "  <p>Note the <span class='link'>options</span> link provided that lets you see the generated swagger doc for this page.</p>",
058      "  <p>Also note the <span class='link'>sources</span> link on these pages to view the source code for the page.</p>",
059      "  <p>All content on pages in the UI are serialized POJOs.  In this case, it's a serialized array of beans with 2 properties, 'name' and 'description'.</p>",
060      "  <p>Other features (such as this aside) are added through annotations.</p>",
061      "</div>"
062   }
063)
064@SerializerConfig(
065   // For testing purposes, we want to use single quotes in all the serializers so it's easier to do simple
066   // String comparisons.
067   // You can apply any of the Serializer/Parser/BeanContext settings this way.
068   quoteChar="'"
069)
070public class RootResources extends BasicRestServletJenaGroup {
071   // IMPORTANT!  If you don't need RDF support, change the parent class to ResourceGroup.
072   // It allows you to remove the Jena prerequisite.
073
074   private static final long serialVersionUID = 1L;
075}
