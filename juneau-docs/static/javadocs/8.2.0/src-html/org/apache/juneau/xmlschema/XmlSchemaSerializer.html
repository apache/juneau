013package org.apache.juneau.xmlschema;
014
015import org.apache.juneau.*;
016import org.apache.juneau.annotation.*;
017import org.apache.juneau.collections.*;
018import org.apache.juneau.serializer.*;
019import org.apache.juneau.xml.*;
020
021/**
022 * Serializes POJO metadata to HTTP responses as XML.
023 *
024 * <h5 class='topic'>Media types</h5>
025 *
026 * Handles <c>Accept</c> types:  <bc>text/xml+schema</bc>
027 * <p>
028 * Produces <c>Content-Type</c> types:  <bc>text/xml</bc>
029 *
030 * <h5 class='topic'>Description</h5>
031 *
032 * Produces the XML-schema representation of the XML produced by the {@link XmlSerializer} class with the same properties.
033 */
034@ConfigurableContext
035@Deprecated
036public class XmlSchemaSerializer extends XmlSerializer {
037
038   //-------------------------------------------------------------------------------------------------------------------
039   // Configurable properties
040   //-------------------------------------------------------------------------------------------------------------------
041
042   static final String PREFIX = "XmlSchemaSerializer";
043
044   /**
045    * Constructor.
046    *
047    * @param ps Initialize with the specified config property store.
048    */
049   public XmlSchemaSerializer(PropertyStore ps) {
050      super(
051         ps.builder()
052            .setDefault(XML_enableNamespaces, true)
053            .build(),
054         "text/xml",
055         "text/xml+schema"
056      );
057   }
058
059   @Override /* Serializer */
060   public XmlSchemaSerializerSession createSession() {
061      return createSession(createDefaultSessionArgs());
062   }
063
064   @Override /* Serializer */
065   public XmlSchemaSerializerSession createSession(SerializerSessionArgs args) {
066      return new XmlSchemaSerializerSession(this, args);
067   }
068
069   //-----------------------------------------------------------------------------------------------------------------
070   // Other methods
071   //-----------------------------------------------------------------------------------------------------------------
072
073   @Override /* Context */
074   public OMap toMap() {
075      return super.toMap()
076         .a("XmlSchemaSerializer", new DefaultFilteringOMap()
077         );
078   }
079}
