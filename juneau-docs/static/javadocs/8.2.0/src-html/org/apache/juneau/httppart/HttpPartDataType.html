013package org.apache.juneau.httppart;
014
015/**
016 * Valid values for the <c>type</c> field.
017 */
018public enum HttpPartDataType {
019
020   /**
021    * String.
022    */
023   STRING,
024
025   /**
026    * Floating point number.
027    */
028   NUMBER,
029
030   /**
031    * Decimal number.
032    */
033   INTEGER,
034
035   /**
036    * Boolean.
037    */
038   BOOLEAN,
039
040   /**
041    * Array or collection.
042    */
043   ARRAY,
044
045   /**
046    * Map or bean.
047    */
048   OBJECT,
049
050   /**
051    * File.
052    */
053   FILE,
054
055   /**
056    * Not specified.
057    */
058   NO_TYPE;
059
060   /**
061    * Create from lowercase string.
062    * 
063    * @param value The enum name.
064    * @return The enum.
065    */
066   public static HttpPartDataType fromString(String value) {
067      return valueOf(value.toUpperCase());
068   }
069
070   @Override /* Object */
071   public String toString() {
072      return name().toLowerCase();
073   }
074}
