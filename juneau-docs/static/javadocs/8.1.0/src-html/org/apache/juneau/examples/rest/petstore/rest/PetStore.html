013package org.apache.juneau.examples.rest.petstore.rest;
014
015import static org.apache.juneau.http.HttpMethodName.*;
016
017import java.util.*;
018
019import org.apache.juneau.jsonschema.annotation.Items;
020import org.apache.juneau.*;
021import org.apache.juneau.examples.rest.petstore.*;
022import org.apache.juneau.examples.rest.petstore.dto.*;
023import org.apache.juneau.http.annotation.*;
024import org.apache.juneau.rest.*;
025import org.apache.juneau.rest.client.remote.*;
026import org.apache.juneau.rest.exception.*;
027import org.apache.juneau.rest.response.*;

029/**
030 * Defines the interface for both the server-side and client-side pet store application.
031 *
032 * <ul class='seealso'>
033 *    <li class='extlink'>{@source}
034 * </ul>
035 */
036@RemoteResource(path="/petstore")
037public interface PetStore {
038
039   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
040   // Pets
041   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
042
043   /**
044    * Returns all pets in the database.
045    *
046    * @return All pets in the database.
047    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
048    */
049   @RemoteMethod(method=GET, path="/pet")
050   public Collection<Pet> getPets() throws NotAcceptable;
051
052   /**
053    * Returns a pet from the database.
054    *
055    * @param petId The ID of the pet to retrieve.
056    * @return The pet.
057    * @throws IdNotFound Pet was not found.
058    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
059    */
060   @RemoteMethod(path="/pet/{petId}") /* method inferred from method name */
061   public Pet getPet(
062      @Path(
063         name="petId",
064         description="ID of pet to return",
065         example="123"
066      )
067      long petId
068   ) throws IdNotFound, NotAcceptable;
069
070   /**
071    * Adds a pet to the database.
072    *
073    * @param pet The pet data to add to the database.
074    * @return {@link Ok} if successful.
075    * @throws IdConflict ID already in use.
076    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
077    * @throws UnsupportedMediaType Unsupported <c>Content-Type</c> header specified.
078    */
079   @RemoteMethod /* method and path inferred from method name */
080   public long postPet(
081      @Body(
082         description="Pet object to add to the store"
083      ) CreatePet pet
084   ) throws IdConflict, NotAcceptable, UnsupportedMediaType;
085
086   /**
087    * Updates a pet in the database.
088    *
089    * @param pet The pet data to add to the database.
090    * @return {@link Ok} if successful.
091    * @throws IdNotFound ID not found.
092    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
093    * @throws UnsupportedMediaType Unsupported <c>Content-Type</c> header specified.
094    */
095   @RemoteMethod(method=PUT, path="/pet/{petId}")
096   public Ok updatePet(
097      @Body(
098         description="Pet object that needs to be added to the store"
099      ) UpdatePet pet
100   ) throws IdNotFound, NotAcceptable, UnsupportedMediaType;
101
102   /**
103    * Find all pets with the matching statuses.
104    *
105    * @param status The statuses to match against.
106    * @return The pets that match the specified statuses.
107    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
108    */
109   @RemoteMethod(method=GET, path="/pet/findByStatus")
110   public Collection<Pet> findPetsByStatus(
111      @Query(
112         name="status",
113         description="Status values that need to be considered for filter.",
114         required=true,
115         type="array",
116         collectionFormat="csv",
117         items=@Items(
118            type="string",
119            _enum="AVAILABLE,PENDING,SOLD",
120            _default="AVAILABLE"
121         ),
122         example="AVALIABLE,PENDING"
123      )
124      PetStatus[] status
125   ) throws NotAcceptable;
126
127   /**
128    * Find all pets with the specified tags.
129    *
130    * @param tags The tags to match against.
131    * @return The pets that match the specified tags.
132    * @throws InvalidTag Invalid tag was specified.
133    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
134    */
135   @RemoteMethod(method=GET, path="/pet/findByTags")
136   @Deprecated
137   public Collection<Pet> findPetsByTags(
138      @Query(
139         name="tags",
140         description="Tags to filter by",
141         required=true,
142         example="['tag1','tag2']"
143      )
144      String[] tags
145   ) throws InvalidTag, NotAcceptable;
146
147   /**
148    * Deletes the specified pet.
149    *
150    * @param apiKey Security key.
151    * @param petId ID of pet to delete.
152    * @return {@link Ok} if successful.
153    * @throws IdNotFound Pet not found.
154    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
155    */
156   @RemoteMethod(method=DELETE, path="/pet/{petId}")
157   public Ok deletePet(
158      @Header(
159         name="api_key",
160         description="Security API key",
161         required=true,
162         example="foobar"
163      )
164      String apiKey,
165      @Path(
166         name="petId",
167         description="Pet id to delete",
168         example="123"
169      )
170      long petId
171   ) throws IdNotFound, NotAcceptable;
172
173   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
174   // Orders
175   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
176
177   /**
178    * Returns all orders in the database.
179    *
180    * @return All orders in the database.
181    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
182    */
183   @RemoteMethod(method=GET, path="/store/order")
184   public Collection<Order> getOrders() throws NotAcceptable;
185
186   /**
187    * Returns an order from the database.
188    *
189    * @param orderId The ID of the order to retreieve.
190    * @return The retrieved order.
191    * @throws InvalidId ID was invalid.
192    * @throws IdNotFound Order was not found.
193    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
194    */
195   @RemoteMethod(method=GET, path="/store/order/{orderId}")
196   public Order getOrder(
197      @Path(
198         name="orderId",
199         description="ID of order to fetch",
200         maximum="1000",
201         minimum="1",
202         example="123"
203      )
204      long orderId
205   ) throws InvalidId, IdNotFound, NotAcceptable;
206
207   /**
208    * Adds an order to the database.
209    *
210    * @param petId Id of pet to order.
211    * @param username The username of the user placing the order.
212    * @return The ID of the order.
213    * @throws IdConflict ID was already in use.
214    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
215    * @throws UnsupportedMediaType Unsupported <c>Content-Type</c> header specified.
216    */
217   @RemoteMethod(method=POST, path="/store/order")
218   public long placeOrder(
219      @FormData(
220         name="petId",
221         description="Pet ID"
222      )
223      long petId,
224      @FormData(
225         name="username",
226         description="The username of the user creating the order"
227      )
228      String username
229   ) throws IdConflict, NotAcceptable, UnsupportedMediaType;
230
231   /**
232    * Deletes an order from the database.
233    *
234    * @param orderId The order ID.
235    * @return {@link Ok} if successful.
236    * @throws InvalidId ID not valid.
237    * @throws IdNotFound Order not found.
238    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
239    */
240   @RemoteMethod(method=DELETE, path="/store/order/{orderId}")
241   public Ok deleteOrder(
242      @Path(
243         name="orderId",
244         description="ID of the order that needs to be deleted",
245         minimum="1",
246         example="5"
247      )
248      long orderId
249   ) throws InvalidId, IdNotFound, NotAcceptable;
250
251   /**
252    * Returns an inventory of pet statuses and counts.
253    *
254    * @return An inventory of pet statuses and counts.
255    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
256    */
257   @RemoteMethod(method=GET, path="/store/inventory")
258   public Map<PetStatus,Integer> getStoreInventory() throws NotAcceptable;
259
260   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
261   // Users
262   //-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
263
264   /**
265    * Returns all users in the database.
266    *
267    * @return All users in the database.
268    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
269    */
270   @RemoteMethod(method=GET, path="/user")
271   public Collection<User> getUsers() throws NotAcceptable;
272
273   /**
274    * Returns a user from the database.
275    *
276    * @param username The username.
277    * @return The user.
278    * @throws InvalidUsername Invalid username.
279    * @throws IdNotFound username not found.
280    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
281    */
282   @RemoteMethod(method=GET, path="/user/{username}")
283   public User getUser(
284      @Path(
285         name="username",
286         description="The name that needs to be fetched. Use user1 for testing."
287      )
288      String username
289   ) throws InvalidUsername, IdNotFound, NotAcceptable;
290
291   /**
292    * Adds a new user to the database.
293    *
294    * @param user The user to add to the database.
295    * @return {@link Ok} if successful.
296    * @throws InvalidUsername Username was invalid.
297    * @throws IdConflict Username already in use.
298    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
299    * @throws UnsupportedMediaType Unsupported <c>Content-Type</c> header specified.
300    */
301   @RemoteMethod
302   public Ok postUser(
303      @Body(
304         description="Created user object"
305      )
306      User user
307   ) throws InvalidUsername, IdConflict, NotAcceptable, UnsupportedMediaType;
308
309   /**
310    * Bulk creates users.
311    *
312    * @param users The users to add to the database.
313    * @return {@link Ok} if successful.
314    * @throws InvalidUsername Username was invalid.
315    * @throws IdConflict Username already in use.
316    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
317    * @throws UnsupportedMediaType Unsupported <c>Content-Type</c> header specified.
318    */
319   @RemoteMethod(method=POST, path="/user/createWithArray")
320   public Ok createUsers(
321      @Body(
322         description="List of user objects"
323      )
324      User[] users
325   ) throws InvalidUsername, IdConflict, NotAcceptable, UnsupportedMediaType;
326
327   /**
328    * Updates a user in the database.
329    *
330    * @param username The username.
331    * @param user The updated information.
332    * @return {@link Ok} if successful.
333    * @throws InvalidUsername Username was invalid.
334    * @throws IdNotFound User was not found.
335    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
336    * @throws UnsupportedMediaType Unsupported <c>Content-Type</c> header specified.
337    */
338   @RemoteMethod(method=PUT, path="/user/{username}")
339   public Ok updateUser(
340      @Path(
341         name="username",
342         description="Name that need to be updated"
343      )
344      String username,
345      @Body(
346         description="Updated user object"
347      )
348      User user
349   ) throws InvalidUsername, IdNotFound, NotAcceptable, UnsupportedMediaType;
350
351   /**
352    * Deletes a user from the database.
353    *
354    * @param username The username.
355    * @return {@link Ok} if successful.
356    * @throws InvalidUsername Username was not valid.
357    * @throws IdNotFound User was not found.
358    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
359    */
360   @RemoteMethod(method=DELETE, path="/user/{username}")
361   public Ok deleteUser(
362      @Path(
363         name="username",
364         description="The name that needs to be deleted"
365      )
366      String username
367   ) throws InvalidUsername, IdNotFound, NotAcceptable;
368
369   /**
370    * User login.
371    *
372    * @param username The username for login.
373    * @param password The password for login in clear text.
374    * @param rateLimit Calls per hour allowed by the user.
375    * @param expiresAfter The <bc>Expires-After</bc> response header.
376    * @param req The servlet request.
377    * @param res The servlet response.
378    * @return {@link Ok} if successful.
379    * @throws InvalidLogin Login was unsuccessful.
380    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
381    */
382   @RemoteMethod(method=GET, path="/user/login")
383   public Ok login(
384      @Query(
385         name="username",
386         description="The username for login.",
387         required=true,
388         example="myuser"
389      )
390      String username,
391      @Query(
392         name="password",
393         description="The password for login in clear text.",
394         required=true,
395         example="abc123"
396      )
397      String password,
398      @ResponseHeader(
399         name="X-Rate-Limit",
400         type="integer",
401         format="int32",
402         description="Calls per hour allowed by the user.",
403         example="123"
404      )
405      Value<Integer> rateLimit,
406      Value<ExpiresAfter> expiresAfter,
407      RestRequest req,
408      RestResponse res
409   ) throws InvalidLogin, NotAcceptable;
410
411   /**
412    * User logout.
413    *
414    * @return {@link Ok} if successful.
415    * @throws NotAcceptable Unsupported <c>Accept</c> header specified.
416    */
417   @RemoteMethod(method=GET, path="/user/logout")
418   public Ok logout() throws NotAcceptable;
