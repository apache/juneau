087   /**
088    * Initializes the registry URL and rest client.
089    *
090    * @param builder The resource config.
091    */
092   @RestHook(INIT)
093   public void initConnection(RestContextBuilder builder) {
094      Config cf = builder.getConfig();
095
096      driver = cf.getString("SqlQueryResource/driver");
097      connectionUrl = cf.getString("SqlQueryResource/connectionUrl");
098      allowUpdates = cf.getBoolean("SqlQueryResource/allowUpdates", false);
099      allowTempUpdates = cf.getBoolean("SqlQueryResource/allowTempUpdates", false);
100      includeRowNums = cf.getBoolean("SqlQueryResource/includeRowNums", false);
101
102      try {
103         Class.forName(driver).newInstance();
104      } catch (Exception e) {
105         e.printStackTrace(System.err);
106         throw new RuntimeException(e);
107      }
108   }

168   /**
169    * Execute one or more queries.
170    *
171    * @param in
172    *    Query input
173    * @return
174    *    Query results.
175    *    <br>Each entry in the array is a result of one query.
176    *    <b>Each result can be a result set (for queries) or update count (for updates).
177    * @throws BadRequest Invalid SQL detected.
178    */
179   @RestMethod(
180      summary="Execute one or more queries"
181   )
182   @Response(
183      description="Query results.\nEach entry in the array is a result of one query.\nEach result can be a result set (for queries) or update count (for updates)."
184   )
185   public List<Object> post(
186         @Body(
187            description="Query input",
188            example="{sql:'select * from sys.systables',pos:1,limit:100}"
189         )
190         PostInput in
191      ) throws BadRequest {
192
193      List<Object> results = new LinkedList<>();
194
195      // Don't try to submit empty input.
196      if (isEmpty(in.sql))
197         return results;
198
199      if (in.pos < 1 || in.pos > 10000)
200         throw new BadRequest("Invalid value for position.  Must be between 1-10000");
201      if (in.limit < 1 || in.limit > 10000)
202         throw new BadRequest("Invalid value for limit.  Must be between 1-10000");
203
204      String sql = null;
205
206      // Create a connection and statement.
207      // If these fais, let the exception filter up as a 500 error.
208      try (Connection c = DriverManager.getConnection(connectionUrl)) {
209         c.setAutoCommit(false);
210         try (Statement st = c.createStatement()) {
211            for (String s : in.sql.split(";")) {
212               sql = s.trim();
213               if (! sql.isEmpty()) {
214                  Object o = null;
215                  if (allowUpdates || (allowTempUpdates && ! sql.matches("(?:i)commit.*"))) {
216                     if (st.execute(sql)) {
217                        try (ResultSet rs = st.getResultSet()) {
218                           o = new ResultSetList(rs, in.pos, in.limit, includeRowNums);
219                        }
220                     } else {
221                        o = st.getUpdateCount();
222                     }
223                  } else {
224                     try (ResultSet rs = st.executeQuery(sql)) {
225                        o = new ResultSetList(rs, in.pos, in.limit, includeRowNums);
226                     }
227                  }
228                  results.add(o);
229               }
230            }
231         }
232         if (allowUpdates)
233            c.commit();
234         else if (allowTempUpdates)
235            c.rollback();
236      } catch (SQLException e) {
237         throw new BadRequest(e, "Invalid query:  {0}", sql);
238      }
239
240      return results;
241   }

243   /** The parsed form post */
244   @SuppressWarnings("javadoc")
245   public static class PostInput {
246      public String sql = "";
247      public int pos = 1, limit = 100;
248   }
