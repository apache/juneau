013package org.apache.juneau.examples.rest.petstore;
014
015import org.apache.juneau.http.annotation.*;
016import org.apache.juneau.rest.exception.*;
017
018/**
019 * Exception thrown when trying to add an entry where the ID is already in use.
020 *
021 * <ul class='seealso'>
022 *    <li class='extlink'>{@source}
023 * </ul>
024 */
025@SuppressWarnings("serial")
026@Response(description="ID already in use")
027public class IdConflict extends Conflict {
028
029   /**
030    * Constructor.
031    *
032    * @param id The duplicate ID.
033    * @param c The object type..
034    */
035   public IdConflict(Object id, Class<?> c) {
036      super("ID ''{0}'' already in use for type ''{1}''", id, c.getSimpleName());
037   }
038}
