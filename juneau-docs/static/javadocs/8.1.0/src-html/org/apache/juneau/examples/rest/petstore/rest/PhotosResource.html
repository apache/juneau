051@RestResource(
052   path="/photos",
053   messages="nls/PhotosResource",
054   title="Photo REST service",
055   description="Sample resource that allows images to be uploaded and retrieved.",
056   swagger=@ResourceSwagger(

067      "options: servlet:/?method=OPTIONS",
068      "$W{UploadPhotoMenuItem}",
069      "source: $C{Source/gitHub}/org/apache/juneau/examples/rest/$R{servletClassSimple}.java"

086public class PhotosResource extends BasicRestServlet {
087   private static final long serialVersionUID = 1L;
088
089   // Our cache of photos
090   private Map<String,Photo> photos = new TreeMap<>();
091
092   @Override /* Servlet */
093   public void init() {
094      try (InputStream is = getClass().getResourceAsStream("photos/cat.jpg")) {
095         BufferedImage image = ImageIO.read(is);
096         Photo photo = new Photo("cat", image);
097         photos.put(photo.id, photo);
098      } catch (IOException e) {
099         throw new RuntimeException(e);
100      }
101   }
102
103   /** Our bean class for storing photos */
104   public static class Photo {
105      String id;
106      BufferedImage image;
107
108      Photo(String id, BufferedImage image) {
109         this.id = id;
110         this.image = image;
111      }
112
113      /**
114       * @return The URI of this photo.
115       * @throws URISyntaxException ID could not be converted to a URI.
116       */
117      public URI getURI() throws URISyntaxException {
118         return new URI("servlet:/" + id);
119      }
120   }

126   /**
127    * Show the list of all currently loaded photos
128    *
129    * @return The list of all currently loaded photos.
130    * @throws Exception Error occurred.
131    */
132   @RestMethod(
133      name=GET,
134      path="/",
135      summary="Show the list of all currently loaded photos"
136   )
137   public Collection<Photo> getAllPhotos() throws Exception {
138      return photos.values();
139   }
140
141   /**
142    * Shows how to use a custom serializer to serialize a BufferedImage object to a stream.
143    *
144    * @param id The photo ID.
145    * @return The image.
146    * @throws NotFound Image was not found.
147    */
148   @RestMethod(
149      name=GET,
150      path="/{id}",
151      serializers=ImageSerializer.class,
152      summary="Get a photo by ID",
153      description="Shows how to use a custom serializer to serialize a BufferedImage object to a stream."
154   )
155   @Response(
156      schema=@Schema(type="file")
157   )
158   public BufferedImage getPhoto(@Path("id") String id) throws NotFound {
159      Photo p = photos.get(id);
160      if (p == null)
161         throw new NotFound("Photo not found");
162      return p.image;
163   }
164
165   /**
166    * Shows how to use a custom parser to parse a stream into a BufferedImage object.
167    *
168    * @param id The photo ID.
169    * @param image Binary contents of image.
170    * @return <js>"OK"</jk> if successful.
171    * @throws Exception Error occurred.
172    */
173   @RestMethod(
174      name=PUT,
175      path="/{id}",
176      parsers=ImageParser.class,
177      summary="Add or overwrite a photo",
178      description="Shows how to use a custom parser to parse a stream into a BufferedImage object."
179   )
180   public String addPhoto(
181         @Path("id") String id,
182         @Body(
183            description="Binary contents of image.",
184            schema=@Schema(type="file")
185         )
186         BufferedImage image
187      ) throws Exception {
188      photos.put(id, new Photo(id, image));
189      return "OK";
190   }
191
192   /**
193    * Shows how to use a custom parser to parse a stream into a BufferedImage object.
194    *
195    * @param image Binary contents of image.
196    * @return The Photo bean.
197    * @throws Exception Error occurred.
198    */
199   @RestMethod(
200      name=POST,
201      path="/",
202      parsers=ImageParser.class,
203      summary="Add a photo",
204      description="Shows how to use a custom parser to parse a stream into a BufferedImage object."
205   )
206   public Photo setPhoto(
207         @Body(
208            description="Binary contents of image.",
209            schema=@Schema(type="file")
210         )
211         BufferedImage image
212      ) throws Exception {
213      Photo p = new Photo(UUID.randomUUID().toString(), image);
214      photos.put(p.id, p);
215      return p;
216   }
217
218   /**
219    * Upload a photo from a multipart form post.
220    *
221    * @param req HTTP request.
222    * @return Redirect to servlet root.
223    * @throws Exception Error occurred.
224    */
225   @RestMethod(
226      name=POST,
227      path="/upload",
228      matchers=MultipartFormDataMatcher.class,
229      summary="Upload a photo from a multipart form post",
230      description="Shows how to parse a multipart form post containing a binary field.",
231      swagger=@MethodSwagger(
232         parameters={
233            "{in:'formData', name:'id', description:'Unique identifier to assign to image.', type:'string', required:false},",
234            "{in:'formData', name:'file', description:'The binary contents of the image file.', type:'file', required:true}"
235         }
236      )
237   )
238    public SeeOtherRoot uploadFile(RestRequest req) throws Exception {
239       MultipartConfigElement multipartConfigElement = new MultipartConfigElement((String)null);
240       req.setAttribute("org.eclipse.jetty.multipartConfig", multipartConfigElement);
241       String id = UUID.randomUUID().toString();
242       BufferedImage img = null;
243       for (Part part : req.getParts()) {
244         switch (part.getName()) {
245            case "id":
246               id = IOUtils.read(part.getInputStream());
247               break;
248            case "file":
249               img = ImageIO.read(part.getInputStream());
250         }
251       }
252       addPhoto(id, img);
253       return new SeeOtherRoot(); // Redirect to the servlet root.
254    }
255
256   /**
257    * Removes a photo from the database.
258    *
259    * @param id ID of photo to remove.
260    * @return <js>"OK"</jk> if successful.
261    * @throws NotFound Photo was not found.
262    */
263   @RestMethod(
264      name=DELETE,
265      path="/{id}",
266      summary="Delete a photo by ID"
267   )
268   public String deletePhoto(@Path("id") String id) throws NotFound {
269      Photo p = photos.remove(id);
270      if (p == null)
271         throw new NotFound("Photo not found");
272      return "OK";
273   }

279   /** Serializer for converting images to byte streams */
280   public static class ImageSerializer extends OutputStreamSerializer {
281
282      /**
283       * Constructor.
284       * @param ps The property store containing all the settings for this object.
285       */
286      public ImageSerializer(PropertyStore ps) {
287         super(ps, null, "image/png,image/jpeg");
288      }
289
290      @Override /* Serializer */
291      public OutputStreamSerializerSession createSession(SerializerSessionArgs args) {
292         return new OutputStreamSerializerSession(args) {
293
294            @Override /* SerializerSession */
295            protected void doSerialize(SerializerPipe out, Object o) throws IOException, SerializeException {
296               RenderedImage image = (RenderedImage)o;
297               String mediaType = getProperty("mediaType", String.class, (String)null);
298               ImageIO.write(image, mediaType.substring(mediaType.indexOf('/')+1), out.getOutputStream());
299            }
300         };
301      }
302   }
303
304   /** Parser for converting byte streams to images */
305   public static class ImageParser extends InputStreamParser {
306
307      /**
308       * Constructor.
309       * @param ps The property store containing all the settings for this object.
310       */
311      public ImageParser(PropertyStore ps) {
312         super(ps, "image/png", "image/jpeg");
313      }
314
315      @Override /* Parser */
316      public InputStreamParserSession createSession(final ParserSessionArgs args) {
317         return new InputStreamParserSession(args) {
318
319            @Override /* ParserSession */
320            @SuppressWarnings("unchecked")
321            protected <T> T doParse(ParserPipe pipe, ClassMeta<T> type) throws IOException, ParseException, ExecutableException {
322               return (T)ImageIO.read(pipe.getInputStream());
323            }
324         };
325      }
326   }
