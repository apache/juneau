013package org.apache.juneau.transforms;
014
015import java.util.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.parser.*;
019import org.apache.juneau.transform.*;
020
021/**
022 * Transforms {@link Date Dates} to {@link Map Maps} of the format <tt>{value:long}</tt>.
023 *
024 * @deprecated Use {@link TemporalDateSwap}
025 */
026@Deprecated
027@SuppressWarnings("rawtypes")
028public class DateMapSwap extends PojoSwap<Date,Map> {
029
030   /**
031    * Converts the specified {@link Date} to a {@link Map}.
032    */
033   @Override /* PojoSwap */
034   public Map swap(BeanSession session, Date o) {
035      if (o == null)
036         return null;
037      ObjectMap m = new ObjectMap();
038      m.put("time", o.getTime());
039      return m;
040   }
041
042   /**
043    * Converts the specified {@link Map} to a {@link Date}.
044    */
045   @Override /* PojoSwap */
046   public Date unswap(BeanSession session, Map o, ClassMeta<?> hint) throws Exception {
047      Class<?> c = (hint == null ? java.util.Date.class : hint.getInnerClass());
048      long l = Long.parseLong(((Map<?,?>)o).get("time").toString());
049      if (c == java.util.Date.class)
050         return new java.util.Date(l);
051      if (c == java.sql.Date.class)
052         return new java.sql.Date(l);
053      if (c == java.sql.Time.class)
054         return new java.sql.Time(l);
055      if (c == java.sql.Timestamp.class)
056         return new java.sql.Timestamp(l);
057      throw new ParseException("DateMapSwap is unable to narrow object of type ''{0}''", c);
058   }
059}
