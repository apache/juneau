013package org.apache.juneau.jso;
014
015import org.apache.juneau.*;
016import org.apache.juneau.jso.annotation.*;
017
018/**
019 * Metadata on bean properties specific to the JSO serializers and parsers pulled from the {@link Jso @Jso}
020 * annotation on the bean property.
021 */
022public final class JsoBeanPropertyMeta extends ExtendedBeanPropertyMeta {
023
024   /**
025    * Default instance.
026    */
027   public static final JsoBeanPropertyMeta DEFAULT = new JsoBeanPropertyMeta();
028
029   /**
030    * Constructor.
031    *
032    * @param bpm The metadata of the bean property of this additional metadata.
033    * @param mp JSO metadata provider (for finding information about other artifacts).
034    */
035   public JsoBeanPropertyMeta(BeanPropertyMeta bpm, JsoMetaProvider mp) {
036      super(bpm);
037   }
038
039   private JsoBeanPropertyMeta() {
040      super(null);
041   }
042}
