013package org.apache.juneau.examples.rest.petstore.rest;
014
015import static org.apache.juneau.dto.html5.HtmlBuilder.*;
016import static org.apache.juneau.http.HttpMethodName.*;
017
018import org.apache.juneau.rest.*;
019import org.apache.juneau.rest.widget.*;
020
021/**
022 * Menu item for adding a Pet.
023 *
024 * <ul class='seealso'>
025 *    <li class='extlink'>{@source}
026 * </ul>
027 */
028public class AddPetMenuItem extends MenuItemWidget {
029
030   @Override /* MenuItemWidget */
031   public String getLabel(RestRequest req, RestResponse res) throws Exception {
032      return "add";
033   }
034
035   @Override /* Widget */
036   public Object getContent(RestRequest req, RestResponse res) throws Exception {
037      return div(
038         form().id("form").action("servlet:/pet").method(POST).children(
039            table(
040               tr(
041                  th("Name:"),
042                  td(input().name("name").type("text")),
043                  td(new Tooltip("&#x2753;", "The name of the pet.", br(), "e.g. 'Fluffy'"))
044               ),
045               tr(
046                  th("Species:"),
047                  td(
048                     select().name("species").children(
049                        option("CAT"), option("DOG"), option("BIRD"), option("FISH"), option("MOUSE"), option("RABBIT"), option("SNAKE")
050                     )
051                  ),
052                  td(new Tooltip("&#x2753;", "The kind of animal."))
053               ),
054               tr(
055                  th("Price:"),
056                  td(input().name("price").type("number").placeholder("1.0").step("0.01").min(1).max(100).value(9.99)),
057                  td(new Tooltip("&#x2753;", "The price to charge for this pet."))
058               ),
059               tr(
060                  th("Tags:"),
061                  td(input().name("tags").type("text")),
062                  td(new Tooltip("&#x2753;", "Arbitrary textual tags (comma-delimited).", br(), "e.g. 'fluffy,friendly'"))
063               ),
064               tr(
065                  td().colspan(2).style("text-align:right").children(
066                     button("reset", "Reset"),
067                     button("button","Cancel").onclick("window.location.href='/'"),
068                     button("submit", "Submit")
069                  )
070               )
071            ).style("white-space:nowrap")
072         )
073      );
074   }
075}
