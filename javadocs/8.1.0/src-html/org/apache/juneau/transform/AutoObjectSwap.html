013package org.apache.juneau.transform;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016import static org.apache.juneau.internal.ClassUtils.*;
017
018import java.lang.reflect.*;
019import java.util.*;
020
021import org.apache.juneau.*;
022import org.apache.juneau.annotation.*;
023import org.apache.juneau.parser.*;
024import org.apache.juneau.reflect.*;
025import org.apache.juneau.serializer.*;
026
027/**
028 * A dynamic POJO swap based on reflection of a Java class that converts POJOs to serializable objects.
029 *
030 * <p>
031 * Looks for methods on the class that can be called to swap-in surrogate objects before serialization and swap-out
032 * surrogate objects after parsing.
033 *
034 * <h5 class='figure'>Valid surrogate objects</h5>
035 * <ul>
036 *    <li class='jc'>{@link String}
037 *    <li class='jc'>{@link Number}
038 *    <li class='jc'>{@link Boolean}
039 *    <li class='jc'>{@link Map}
040 *    <li class='jc'>{@link Collection}
041 * </ul>
042 *
043 * <h5 class='figure'>Valid swap methods (S = Swapped type)</h5>
044 * <ul>
045 *    <li class='jm'><c><jk>public</jk> S swap()</c>
046 *    <li class='jm'><c><jk>public</jk> S swap(BeanSession)</c>
047 *    <li class='jm'><c><jk>public</jk> S toObject()</c>
048 *    <li class='jm'><c><jk>public</jk> S toObject(BeanSession)</c>
049 * </ul>
050 *
051 * <h5 class='figure'>Valid unswap methods (N = Normal type, S = Swapped type)</h5>
052 * <ul>
053 *    <li class='jm'><c><jk>public static</jk> N unswap(S)</c>
054 *    <li class='jm'><c><jk>public static</jk> N unswap(BeanSession, S)</c>
055 *    <li class='jm'><c><jk>public static</jk> N fromObject(S)</c>
056 *    <li class='jm'><c><jk>public static</jk> N fromObject(BeanSession, S)</c>
057 *    <li class='jm'><c><jk>public static</jk> N create(S)</c>
058 *    <li class='jm'><c><jk>public static</jk> N create(BeanSession, S)</c>
059 *    <li class='jm'><c><jk>public</jk> N(S)</c>
060 * </ul>
061 *
062 * <p>
063 * Classes are ignored if any of the following are true:
064 * <ul>
065 *    <li>Classes annotated with {@link BeanIgnore @BeanIgnore}.
066 *    <li>Non-static member classes.
067 * </ul>
068 *
069 * <p>
070 * Members/constructors are ignored if any of the following are true:
071 * <ul>
072 *    <li>Members/constructors annotated with {@link BeanIgnore @BeanIgnore}.
073 *    <li>Deprecated members/constructors.
074 * </ul>
075 *
076 * @param <T> The normal class type.
077 */
078public class AutoObjectSwap<T> extends PojoSwap<T,Object> {
079
080   private static final Set<String>
081      SWAP_METHOD_NAMES = newUnmodifiableHashSet("swap", "toObject"),
082      UNSWAP_METHOD_NAMES = newUnmodifiableHashSet("unswap", "create", "fromObject");
083
084   /**
085    * Inspects the specified class and returns a swap of this type if possible.
086    *
087    * @param ci The class to return a swap on.
088    * @return A POJO swap instance, or <jk>null</jk> if one could not be created.
089    */
090   @SuppressWarnings({ "rawtypes" })
091   public static PojoSwap<?,?> find(ClassInfo ci) {
092
093      if (shouldIgnore(ci))
094         return null;
095
096      // Find swap() method if present.
097      for (MethodInfo m : ci.getPublicMethods()) {
098         if (isSwapMethod(m)) {
099
100            ClassInfo rt = m.getReturnType();
101
102            for (MethodInfo m2 : ci.getPublicMethods())
103               if (isUnswapMethod(m2, ci, rt))
104                  return new AutoObjectSwap(ci, m, m2, null);
105
106            for (ConstructorInfo cs : ci.getPublicConstructors())
107               if (isUnswapConstructor(cs, rt))
108                  return new AutoObjectSwap(ci, m, null, cs);
109
110            return new AutoObjectSwap(ci, m, null, null);
111         }
112      }
113
114      return null;
115   }
116
117   private static boolean shouldIgnore(ClassInfo ci) {
118      return
119         ci.hasAnnotation(BeanIgnore.class)
120         || ci.isNonStaticMemberClass();
121   }
122
123   private static boolean isSwapMethod(MethodInfo mi) {
124      return
125         mi.isNotDeprecated()
126         && mi.isNotStatic()
127         && mi.hasName(SWAP_METHOD_NAMES)
128         && mi.hasFuzzyParamTypes(BeanSession.class)
129         && ! mi.hasAnnotation(BeanIgnore.class);
130   }
131
132   private static boolean isUnswapMethod(MethodInfo mi, ClassInfo ci, ClassInfo rt) {
133      return
134         mi.isNotDeprecated()
135         && mi.isStatic()
136         && mi.hasName(UNSWAP_METHOD_NAMES)
137         && mi.hasFuzzyParamTypes(BeanSession.class, rt.inner())
138         && mi.hasReturnTypeParent(ci)
139         && ! mi.hasAnnotation(BeanIgnore.class);
140   }
141
142   private static boolean isUnswapConstructor(ConstructorInfo cs, ClassInfo rt) {
143      return
144         cs.isNotDeprecated()
145         && cs.hasParamTypeParents(rt)
146         && ! cs.hasAnnotation(BeanIgnore.class);
147   }
148
149   //------------------------------------------------------------------------------------------------------------------
150
151   private final Method swapMethod, unswapMethod;
152   private final Constructor<?> unswapConstructor;
153
154   private AutoObjectSwap(ClassInfo ci, MethodInfo swapMethod, MethodInfo unswapMethod, ConstructorInfo unswapConstructor) {
155      super(ci.inner(), swapMethod.inner().getReturnType());
156      this.swapMethod = swapMethod.inner();
157      this.unswapMethod = unswapMethod == null ? null : unswapMethod.inner();
158      this.unswapConstructor = unswapConstructor == null ? null : unswapConstructor.inner();
159   }
160
161   @Override /* PojoSwap */
162   public Object swap(BeanSession session, Object o) throws SerializeException {
163      try {
164         return swapMethod.invoke(o, getMatchingArgs(swapMethod.getParameterTypes(), session));
165      } catch (Exception e) {
166         throw SerializeException.create(e);
167      }
168   }
169
170   @SuppressWarnings("unchecked")
171   @Override /* PojoSwap */
172   public T unswap(BeanSession session, Object f, ClassMeta<?> hint) throws ParseException {
173      try {
174         if (unswapMethod != null)
175            return (T)unswapMethod.invoke(null, getMatchingArgs(unswapMethod.getParameterTypes(), session, f));
176         if (unswapConstructor != null)
177            return (T)unswapConstructor.newInstance(f);
178         return super.unswap(session, f, hint);
179      } catch (Exception e) {
180         throw ParseException.create(e);
181      }
182   }
183}
