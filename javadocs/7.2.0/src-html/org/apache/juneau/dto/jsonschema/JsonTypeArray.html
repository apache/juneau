013package org.apache.juneau.dto.jsonschema;
014
015import java.util.*;
016
017/**
018 * Represents a list of {@link JsonType} objects.
019 *
020 * <h5 class='section'>See Also:</h5>
021 * <ul class='doctree'>
022 *    <li class='jp'>{@doc package-summary.html#TOC org.apache.juneau.dto.jsonschema}
023 * </ul>
024 */
025public final class JsonTypeArray extends LinkedList<JsonType> {
026
027   private static final long serialVersionUID = 1L;
028
029   /**
030    * Default constructor.
031    */
032   public JsonTypeArray() {}
033
034   /**
035    * Constructor with predefined types to add to this list.
036    *
037    * @param types The list of types to add to the list.
038    */
039   public JsonTypeArray(JsonType...types) {
040      addAll(types);
041   }
042
043   /**
044    * Convenience method for adding one or more {@link JsonType} objects to this array.
045    *
046    * @param types The {@link JsonType} objects to add to this array.
047    * @return This object (for method chaining).
048    */
049   public JsonTypeArray addAll(JsonType...types) {
050      for (JsonType t : types)
051         add(t);
052      return this;
053   }
054}
