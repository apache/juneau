013package org.apache.juneau.xml;
014
015import static org.apache.juneau.internal.StringUtils.*;
016
017import java.text.*;
018
019import javax.xml.stream.*;
020
021import org.apache.juneau.parser.ParseException;
022
023/**
024 * Exception that indicates invalid syntax encountered during XML parsing.
025 */
026@SuppressWarnings("serial")
027public class XmlParseException extends ParseException {
028
029   /**
030    * Constructor.
031    * 
032    * @param location The location of the exception.
033    * @param message The exception message containing {@link MessageFormat}-style arguments.
034    * @param args Optional {@link MessageFormat}-style arguments.
035    */
036   public XmlParseException(Location location, String message, Object...args) {
037      super(getMessage(location, message, args));
038   }
039
040   private static String getMessage(Location location, String msg, Object... args) {
041      if (args.length != 0)
042         msg = format(msg, args);
043      if (location != null)
044         msg = "Parse exception occurred at " + location + ".  " + msg;
045      return msg;
046   }
047}
