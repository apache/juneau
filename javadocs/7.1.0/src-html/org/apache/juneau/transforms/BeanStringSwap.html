013package org.apache.juneau.transforms;
014
015import org.apache.juneau.*;
016import org.apache.juneau.transform.*;
017
018/**
019 * Transforms beans into {@link String Strings} by simply calling the {@link Object#toString()} method.
020 * 
021 * <p>
022 * Allows you to specify classes that should just be converted to {@code Strings} instead of potentially
023 * being turned into Maps by the {@link BeanContext} (or worse, throwing
024 * {@link BeanRuntimeException BeanRuntimeExceptions}).
025 * 
026 * <p>
027 * This is usually a one-way transform.
028 * Beans serialized as strings cannot be reconstituted using a parser unless it is a
029 * <a class="doclink" href="../../../../overview-summary.html#juneau-marshall.PojoCategories">Type 5 POJO</a>.
030 * 
031 * @param <T> The class type of the bean.
032 */
033public class BeanStringSwap<T> extends StringSwap<T> {
034
035   /**
036    * Converts the specified bean to a {@link String}.
037    */
038   @Override /* PojoSwap */
039   public String swap(BeanSession session, T o) {
040      return o.toString();
041   }
042}
