013package org.apache.juneau.serializer;
014
015import org.apache.juneau.http.*;
016
017/**
018 * Represents a serializer and media type that matches an HTTP <code>Accept</code> header value.
019 */
020public final class SerializerMatch {
021
022   private final MediaType mediaType;
023   private final Serializer serializer;
024
025   SerializerMatch(MediaType mediaType, Serializer serializer) {
026      this.mediaType = mediaType;
027      this.serializer = serializer;
028   }
029
030   /**
031    * Returns the media type of the serializers that matched the HTTP <code>Accept</code> header value.
032    * 
033    * @return The media type of the match.
034    */
035   public MediaType getMediaType() {
036      return mediaType;
037   }
038
039   /**
040    * Returns the serializer that matched the HTTP <code>Accept</code> header value.
041    * 
042    * @return The serializer of the match.
043    */
044   public Serializer getSerializer() {
045      return serializer;
046   }
047}
