013package org.apache.juneau.httppart;
014
015import org.apache.juneau.internal.*;
016import org.apache.juneau.serializer.*;
017
018/**
019 * An implementation of {@link HttpPartSerializer} that simply serializes everything using {@link Object#toString()}.
020 *
021 * <p>
022 * More precisely, uses the {@link ClassUtils#toString(Object)} method to stringify objects.
023 */
024public class SimplePartSerializer extends BaseHttpPartSerializer {
025
026   //-------------------------------------------------------------------------------------------------------------------
027   // Predefined instances
028   //-------------------------------------------------------------------------------------------------------------------
029
030   /** Reusable instance of {@link SimplePartSerializer}, all default settings. */
031   public static final SimplePartSerializer DEFAULT = new SimplePartSerializer();
032
033   //-------------------------------------------------------------------------------------------------------------------
034   // Instance
035   //-------------------------------------------------------------------------------------------------------------------
036
037   @Override
038   public SimplePartSerializerSession createPartSession(SerializerSessionArgs args) {
039      return new SimplePartSerializerSession();
040   }
041
042   @Override /* HttpPartSerializer */
043   public String serialize(HttpPartType partType, HttpPartSchema schema, Object value) throws SchemaValidationException, SerializeException {
044      return createPartSession().serialize(partType, schema, value);
045   }
046
047   /**
048    * @deprecated Use {@link #serialize(HttpPartType, HttpPartSchema, Object)}
049    */
050   @Override
051   @Deprecated
052   public String serialize(HttpPartType type, Object value) {
053      try {
054         return serialize(type, null, value);
055      } catch (Exception e) {
056         throw new RuntimeException(e);
057      }
058   }
059}
