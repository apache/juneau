626   @Override
627   @RestMethod(
628      name=GET,
629      path="/user/login",
630      summary="Logs user into the system",
631      swagger=@MethodSwagger(
632         tags="user"
633      )
634   )
635   public Ok login(
636         String username,
637         String password,
638         Value<Integer> rateLimit,
639         Value<ExpiresAfter> expiresAfter,
640         RestRequest req,
641         RestResponse res
642      ) throws InvalidLogin, NotAcceptable {
643
644      if (! store.isValid(username, password))
645         throw new InvalidLogin();
646
647      Date d = new Date(System.currentTimeMillis() + 30 * 60 * 1000);
648      req.getSession().setAttribute("login-expires", d);
649      rateLimit.set(1000);
650      expiresAfter.set(new ExpiresAfter(d));
651      return OK;
652   }

661      )
662   )
663   public Ok logout() throws NotAcceptable {
664      getRequest().getSession().removeAttribute("login-expires");
665      return OK;
