013package org.apache.juneau.examples.rest.petstore.rest;
014
015import static org.apache.juneau.dto.html5.HtmlBuilder.*;
016import static org.apache.juneau.http.HttpMethodName.*;
017
018import org.apache.juneau.rest.*;
019import org.apache.juneau.rest.widget.*;
020
021/**
022 * Menu item for adding a Pet.
023 */
024public class AddOrderMenuItem extends MenuItemWidget {
025
026   @Override /* MenuItemWidget */
027   public String getLabel(RestRequest req) throws Exception {
028      return "add";
029   }
030
031   @Override /* MenuItemWidget */
032   public String getBeforeShowScript(RestRequest req) throws Exception {
033      return loadScript("AddOrderMenuItem_beforeShow.js");
034   }
035
036   @Override /* Widget */
037   public Object getContent(RestRequest req) throws Exception {
038
039      return div(
040         form().id("form").action("servlet:/store/order").method(POST).children(
041            table(
042               tr(
043                  th("Pet:"),
044                  td(
045                     select().id("addPet_names").name("petId")
046                  ),
047                  td(new Tooltip("&#x2753;", "The pet to purchase."))
048               ),
049               tr(
050                  th("Ship date:"),
051                  td(input().name("shipDate").type("date")),
052                  td(new Tooltip("&#x2753;", "The requested ship date."))
053               ),
054               tr(
055                  td().colspan(2).style("text-align:right").children(
056                     button("reset", "Reset"),
057                     button("submit", "Submit")
058                  )
059               )
060            ).style("white-space:nowrap")
061         )
062      );
063   }
064}
