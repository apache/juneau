013package org.apache.juneau.examples.rest.petstore.rest;
014
015import static org.apache.juneau.dto.html5.HtmlBuilder.*;
016import static org.apache.juneau.http.HttpMethodName.*;
017
018import org.apache.juneau.rest.*;
019import org.apache.juneau.rest.widget.*;
020
021/**
022 * Menu item for adding a Pet.
023 */
024public class AddPetMenuItem extends MenuItemWidget {
025
026   @Override /* MenuItemWidget */
027   public String getLabel(RestRequest req) throws Exception {
028      return "add";
029   }
030
031   @Override /* Widget */
032   public Object getContent(RestRequest req) throws Exception {
033      return div(
034         form().id("form").action("servlet:/pet").method(POST).children(
035            table(
036               tr(
037                  th("Name:"),
038                  td(input().name("name").type("text")),
039                  td(new Tooltip("&#x2753;", "The name of the pet.", br(), "e.g. 'Fluffy'"))
040               ),
041               tr(
042                  th("Species:"),
043                  td(
044                     select().name("species").children(
045                        option("CAT"), option("DOG"), option("BIRD"), option("FISH"), option("MOUSE"), option("RABBIT"), option("SNAKE")
046                     )
047                  ),
048                  td(new Tooltip("&#x2753;", "The kind of animal."))
049               ),
050               tr(
051                  th("Price:"),
052                  td(input().name("price").type("number").placeholder("1.0").step("0.01").min(1).max(100).value(9.99)),
053                  td(new Tooltip("&#x2753;", "The price to charge for this pet."))
054               ),
055               tr(
056                  th("Tags:"),
057                  td(input().name("tags").type("text")),
058                  td(new Tooltip("&#x2753;", "Arbitrary textual tags (comma-delimited).", br(), "e.g. 'fluffy,friendly'"))
059               ),
060               tr(
061                  td().colspan(2).style("text-align:right").children(
062                     button("reset", "Reset"),
063                     button("button","Cancel").onclick("window.location.href='/'"),
064                     button("submit", "Submit")
065                  )
066               )
067            ).style("white-space:nowrap")
068         )
069      );
070   }
071}
