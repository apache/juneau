013package org.apache.juneau.rest.client2;
014
015import java.io.*;
016
017import org.apache.http.*;
018import org.apache.http.client.methods.*;
019import org.apache.http.protocol.*;
020
021/**
022 * Default HTTP call handler.
023 *
024 * Can be subclasses and specified via {@link RestClient#RESTCLIENT_callHandler}.
025 */
026public class BasicRestCallHandler implements RestCallHandler {
027
028   private final RestClient client;
029
030   /**
031    * Constructor.
032    *
033    * @param client The client to use for handling requests.
034    */
035   public BasicRestCallHandler(RestClient client) {
036      this.client = client;
037   }
038
039   @Override /* RestCallHandler */
040   public HttpResponse run(HttpHost target, HttpRequest request, HttpContext context) throws IOException {
041      return target == null ? client.execute((HttpUriRequest)request, context) : client.execute(target, request, context);
042   }
043}
