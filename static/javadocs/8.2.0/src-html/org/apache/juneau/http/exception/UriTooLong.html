013package org.apache.juneau.http.exception;
014
015import static org.apache.juneau.http.exception.UriTooLong.*;
016
017import java.text.*;
018
019import org.apache.juneau.http.annotation.*;
020
021/**
022 * Exception representing an HTTP 414 (URI Too Long).
023 *
024 * <p>
025 * The URI provided was too long for the server to process.
026 * <br>Often the result of too much data being encoded as a query-string of a GET request, in which case it should be converted to a POST request.
027 */
028@Response(code=CODE, description=MESSAGE)
029public class UriTooLong extends HttpException {
030   private static final long serialVersionUID = 1L;
031
032   /** HTTP status code */
033   public static final int CODE = 414;
034
035   /** Default message */
036   public static final String MESSAGE = "URI Too Long";
037
038   /**
039    * Constructor.
040    *
041    * @param cause The cause.  Can be <jk>null</jk>.
042    * @param msg The message.  Can be <jk>null</jk>.
043    * @param args Optional {@link MessageFormat}-style arguments in the message.
044    */
045   public UriTooLong(Throwable cause, String msg, Object...args) {
046      super(cause, CODE, msg, args);
047   }
048
049   /**
050    * Constructor.
051    *
052    * @param msg The message.  Can be <jk>null</jk>.
053    */
054   public UriTooLong(String msg) {
055      this((Throwable)null, msg);
056   }
057
058   /**
059    * Constructor.
060    */
061   public UriTooLong() {
062      this((Throwable)null, MESSAGE);
063   }
064
065   /**
066    * Constructor.
067    *
068    * @param msg The message.  Can be <jk>null</jk>.
069    * @param args Optional {@link MessageFormat}-style arguments in the message.
070    */
071   public UriTooLong(String msg, Object...args) {
072      this(null, msg, args);
073   }
074
075   /**
076    * Constructor.
077    *
078    * @param cause The cause.  Can be <jk>null</jk>.
079    */
080   public UriTooLong(Throwable cause) {
081      this(cause, null);
082   }
083
084   //------------------------------------------------------------------------------------------------------------------
085   // Fluent setters.
086   //------------------------------------------------------------------------------------------------------------------
087
088   // <FluentSetters>
089
090   @Override /* GENERATED - HttpException */
091   public UriTooLong header(String name, Object val) {
092      super.header(name, val);
093      return this;
094   }
095
096   // </FluentSetters>
097}
