009package com.ibm.juno.server.labels;
010
011import com.ibm.juno.core.dto.*;
012import com.ibm.juno.server.*;
013
014/**
015 * Shortcut label for child resources.  Typically used in router resources.
016 *
017 * <h6 class='topic'>Examples</h6>
018 * <p class='bcode'>
019 *    <jc>// Instead of this...</jc>
020 *    <jk>new</jk> NameDescription(<jk>new</jk> Link(<js>"httpTool"</js>, uri + <js>"/httpTool"</js>), <js>"HTTP request test client"</js>);
021 *
022 *    <jc>// ...use this simpler equivalent...</jc>
023 *    <jk>new</jk> ResourceLink(uri, <js>"httpTool"</js>, <js>"HTTP request test client"</js>);
024 * </p>
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028public final class ResourceDescription extends NameDescription implements Comparable<ResourceDescription> {
029
030   /**
031    * Constructor.
032    *
033    * @param rootUrl The root URI of the child resource (e.g. the URI of the parent resource).
034    *       Must not end with <js>'/'</js>.
035    *       Must be URL-Encoded.
036    * @param name The name of the child resource.
037    *       This will be URL-encoded and appended onto the root URL to create the hyperlink for the resource.
038    * @param description The description of the child resource.
039    */
040   public ResourceDescription(String rootUrl, String name, String description) {
041      super(new Link(name, (rootUrl.equals("/") || rootUrl.isEmpty() ? "/" : rootUrl + "/") + RestUtils.encode(name)), description);
042   }
043
044   /**
045    * Constructor for resources that are children of a REST resource.
046    *
047    * @param req The HTTP request.
048    * @param childPath The childPath The path of the child resource relative to the servlet.
049    * @param description The description of the child resource.
050    */
051   public ResourceDescription(RestRequest req, String childPath, String description) {
052      super(new Link(calcName(childPath), calcHref(req, childPath)), description);
053   }
054
055   private static String calcName(String childPath) {
056      return RestUtils.decode(childPath.indexOf('/') == -1 ? childPath : childPath.substring(childPath.lastIndexOf('/')+1));
057   }
058
059   private static String calcHref(RestRequest req, String childPath) {
060      return req.getServletURIBuilder().append('/').append(childPath).toString();
061   }
062
063   /**
064    * Constructor.
065    *
066    * @param name The name of the child resource.
067    * @param description The description of the child resource.
068    */
069   public ResourceDescription(String name, String description) {
070      super(new Link(name, name), description);
071   }
072
073   /** No-arg constructor.  Used for JUnit testing of OPTIONS pages. */
074   public ResourceDescription() {}
075
076   @Override /* NameDescription */
077   public Link getName() {
078      return (Link)super.getName();
079   }
080
081   /**
082    * Overridden setter.
083    *
084    * @param name The new name.
085    */
086   public void setName(Link name) {
087      super.setName(name);
088   }
089
090   @Override /* Comparable */
091   public int compareTo(ResourceDescription o) {
092      return getName().compareTo(o.getName());
093   }
094
095   @Override /* Object */
096   public boolean equals(Object o) {
097      return (o instanceof ResourceDescription) && ((ResourceDescription)o).getName().equals(getName());
098   }
099
100   @Override /* Object */
101   public int hashCode() {
102      return getName().hashCode();
103   }
104}
