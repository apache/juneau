009package com.ibm.juno.server.labels;
010
011import com.ibm.juno.core.annotation.*;
012
013/**
014 * Simple bean with {@code name} and {@code description} properties.
015 * <p>
016 *    Primarily used for constructing tables with name/description columns on REST OPTIONS requests.
017 *
018 * @author James Bognar (jbognar@us.ibm.com)
019 */
020@Bean(properties={"name","description"})
021public class NameDescription {
022
023   private Object name;
024   private Object description;
025
026   /** No-arg constructor.  Used for JUnit testing of OPTIONS pages. */
027   public NameDescription() {}
028
029   /**
030    * Constructor.
031    * @param name A name.
032    * @param description A description.
033    */
034   public NameDescription(Object name, Object description) {
035      this.name = name;
036      this.description = description;
037   }
038
039   /**
040    * Returns the name field on this label.
041    * @return The name.
042    */
043   public Object getName() {
044      return name;
045   }
046
047   /**
048    * Sets the name field on this label to a new value.
049    * @param name The new name.
050    */
051   public void setName(Object name) {
052      this.name = name;
053   }
054
055   /**
056    * Returns the description field on this label.
057    * @return The description.
058    */
059   public Object getDescription() {
060      return description;
061   }
062
063   /**
064    * Sets the description field on this label to a new value.
065    * @param description The new description.
066    */
067   public void setDescription(Object description) {
068      this.description = description;
069   }
070}
