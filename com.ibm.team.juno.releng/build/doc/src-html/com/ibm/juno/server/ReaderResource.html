009package com.ibm.juno.server;
010
011import java.io.*;
012import java.util.*;
013
014import com.ibm.juno.core.*;
015import com.ibm.juno.core.utils.*;
016import com.ibm.juno.server.response.*;
017
018/**
019 * Represents the contents of a text file with convenience methods for resolving
020 *    {@link StringVar} variables and adding HTTP response headers.
021 * <p>
022 * This class is handled special by the {@link WritableHandler} class.
023 *
024 * @author James Bognar (jbognar@us.ibm.com)
025 */
026public class ReaderResource implements Writable {
027
028   private String contents;
029   private String mediaType;
030   private StringVarResolver vr;
031   private Map<String,String> headers = new LinkedHashMap<String,String>();
032
033   /**
034    * Constructor.
035    *
036    * @param contents The contents of this resource.
037    * @param mediaType The HTTP media type.
038    */
039   protected ReaderResource(String contents, String mediaType) {
040      this.contents = contents;
041      this.mediaType = mediaType;
042   }
043
044   /**
045    * Add an HTTP response header.
046    *
047    * @param name The header name.
048    * @param value The header value converted to a string using {@link Object#toString()}.
049    * @return This object (for method chaining).
050    */
051   public ReaderResource setHeader(String name, Object value) {
052      headers.put(name, value == null ? "" : value.toString());
053      return this;
054   }
055
056   /**
057    * Use the specified {@link StringVarResolver} to resolve any {@link StringVar StringVars} in the
058    * contents of this file when the {@link #writeTo(Writer)} or {@link #toString()} methods are called.
059    *
060    * @param vr The string variable resolver to use to resolve string variables.
061    * @return This object (for method chaining).
062    */
063   public ReaderResource setVarResolver(StringVarResolver vr) {
064      this.vr = vr;
065      return this;
066   }
067
068   /**
069    * Get the HTTP response headers.
070    *
071    * @return The HTTP response headers.
072    */
073   public Map<String,String> getHeaders() {
074      return headers;
075   }
076
077   @Override /* Writeable */
078   public void writeTo(Writer w) throws IOException {
079      if (vr != null)
080         vr.writeTo(contents, w);
081      else
082         w.write(contents);
083   }
084
085   @Override /* Streamable */
086   public String getMediaType() {
087      return mediaType;
088   }
089
090   @Override /* Object */
091   public String toString() {
092      if (vr != null)
093         return vr.resolve(contents);
094      return contents;
095   }
096}
