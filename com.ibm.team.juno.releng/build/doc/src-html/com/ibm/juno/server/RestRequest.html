036 * Represents an HTTP request for a REST resource.
037 * <p>
038 *    Equivalent to {@link HttpServletRequest} except with some additional convenience methods.

042 *    following methods return the following values....
043 * </p>
044 * <table class='styled'>
045 *    <tr><th>Method</th><th>Value</th></tr>
046 *    <tr><td>{@code getContextPath()}</td><td>{@code /contextRoot}</td></tr>

082   /**
083    * Constructor.
084    */
085   RestRequest(RestServlet servlet, HttpServletRequest req) throws ServletException {
086      super(req);
087
088      try {
089         this.servlet = servlet;
090         isPost = req.getMethod().equalsIgnoreCase("POST");
091
092         // If this is a POST, we want to parse the query parameters ourselves to prevent
093         // the servlet code from processing the HTTP body as URL-Encoded parameters.
094         if (isPost)
095            queryParams = servlet.getUrlEncodingParser().parseIntoSimpleMap(getQueryString());
096         else {
097            queryParams = req.getParameterMap();
098         }
099
100         // Get the HTTP method.
101         // Can be overridden through a "method" GET attribute.
102         method = super.getMethod();
103
104         String m = getQueryParameter("method");
105         if (! StringUtils.isEmpty(m) && (servlet.allowMethodParams.contains(m) || servlet.allowMethodParams.contains("*")))
106            method = m;
107
108         if (servlet.allowContentParam)
109            content = getQueryParameter("content");
110
111         defaultServletHeaders = servlet.getDefaultRequestHeaders();
112
113         debug = "true".equals(getQueryParameter("debug", "false"));
114
115         if (debug) {
116            servlet.log(Level.INFO, toString());
117         }
118
119      } catch (RestException e) {
120         throw e;
121      } catch (Exception e) {
122         throw new ServletException(e);
123      }
124   }

182   /**
183    * Returns the <code>Content-Type</code> header value on the request, stripped
184    *    of any parameters such as <js>";charset=X"</js>.
185    * <p>
186    * Example: <js>"text/json"</js>.
187    * <p>
188    * If the content type is not specified, and the content is specified via a
189    *    <code>&content</code> query parameter, the content type is assumed to be
190    *    <js>"text/uon"</js>.  Otherwise, the
191    *    content type is assumed to be <js>"text/json"</js>.
192    *
193    * @return The <code>Accept</code> media-type header values on the request.
194    */
195   public String getMediaType() {
196      String cm = getHeader("Content-Type");
197      if (cm == null) {
198         if (content != null)
199            return "text/uon";
200         return "text/json";
201      }
202      int j = cm.indexOf(';');
203      if (j != -1)
204         cm = cm.substring(0, j);
205      return cm;
206   }

217   /**
218    * Returns the charset specified on the <code>Content-Type</code> header, or
219    * <js>"UTF-8"</js> if not specified.
220    */
221   @Override /* ServletRequest */
222   public String getCharacterEncoding() {
223      if (charset == null) {
224         // Determine charset
225         // NOTE:  Don't use super.getCharacterEncoding() because the spec is implemented inconsistently.
226         // Jetty returns the default charset instead of null if the character is not specified on the request.
227         String h = getHeader("Content-Type");
228         if (h != null) {
229            int i = h.indexOf(";charset=");
230            if (i > 0)
231               charset = h.substring(i+9).trim();
232         }
233         if (charset == null)
234            charset = defaultCharset;
235         if (! RestServlet.availableCharsets.containsKey(charset))
236            throw new RestException(SC_UNSUPPORTED_MEDIA_TYPE, "Unsupported charset in header ''Content-Type'': ''{0}''", h);
237      }
238      return charset;
239   }

302   /**
303    * Returns the specified header value, or the specified default value if the
304    *    header value isn't present.
305    * <p>
306    *    If {@code allowHeaderParams} init parameter is <jk>true</jk>, then first looks
307    *    for {@code &HeaderName=x} in the URL query string.
308    *
309    * @param name The HTTP header name.
310    * @param def The default value to return if the header value isn't found.
311    * @return The header value, or the default value if the header isn't present.
312    */
313   public String getHeader(String name, String def) {
314      String h = getOverriddenHeader(name);
315      if (h != null)
316         return h;
317      h = super.getHeader(name);
318      if (h != null && ! h.isEmpty())
319         return h;
320      if (defaultMethodHeaders != null) {
321         h = defaultMethodHeaders.get(name);
322         if (h != null)
323            return h;
324      }
325      h = defaultServletHeaders.get(name);
326      if (h != null)
327         return h;
328      return def;
329   }

341    * to override actual header values on the request.
342    */
343   private String getOverriddenHeader(String name) {
344      String h = null;
345      if (servlet.allowHeaderParams)
346         h = getQueryParameter(name);
347      if (h != null)
348         return h;
349      if (overriddenHeaders != null) {
350         h = overriddenHeaders.get(name);
351         if (h != null)
352            return h;
353      }
354      return h;
355   }

368   @Override /* ServletRequest */
369   public Enumeration<Locale> getLocales() {
370      String h = getOverriddenHeader("Accept-Language");
371      if (h != null) {
372         MediaRange[] mr = MediaRange.parse(h);
373         if (mr.length > 0) {
374            List<Locale> l = new ArrayList<Locale>(mr.length);
375            for (MediaRange r : mr)
376               l.add(toLocale(r.getType()));
377            return enumeration(l);
378         }
379      }
380      return super.getLocales();
381   }

387      String country = "";
388      int i = lang.indexOf('-');
389      if (i > -1) {

391          lang = lang.substring(0,i).trim();
392      }
393      return new Locale(lang, country);

407   /**
408    * Returns the parameter with the specified name.
409    * <p>
410    * Returns <jk>null</jk> for parameters with no value (e.g. <js>"&foo"</js>).
411    * This is consistent with WAS, but differs from Tomcat behavior.
412    * The presence of parameter <js>"&foo"</js> in this case can be determined using {@link #hasParameter(String)}.
413    * <p>
414    * Parameter lookup is case-insensitive (consistent with WAS, but differs from Tomcat).
415    * <p>
416    * <i>Note:</i> Calling this method on URL-Encoded FORM posts causes the body content to be loaded and parsed by
417    *    the underlying servlet API.
418    * <p>
419    * <i>Note:</i> This method returns the raw unparsed value, and differs from calling <code>getParameter(name, String.<jk>class</js>)</code>
420    *    which will convert the value from UON notation:
421    * <ul>
422    *    <li><js>"\u0000"</js> =&gt; <jk>null</jk>
423    *    <li><js>"$s(foo)"</js> =&gt; <js>"foo"</js>
424    *    <li><js>"(foo)"</js> =&gt; <js>"foo"</js>
425    * </ul>
426    */
427   @Override /* ServletRequest */
428   public String getParameter(String name) {
429      String s = null;
430      if (overriddenParams != null)
431         s = overriddenParams.get(name);
432      if (s != null)
433         return s;
434
435      String val = super.getParameter(name);
436
437      // Fix for behavior difference between Tomcat and WAS.
438      // getParameter("foo") on "&foo" in Tomcat returns "".
439      // getParameter("foo") on "&foo" in WAS returns null.
440      if (val != null && val.isEmpty())
441         if (queryParams.containsKey(name))
442            val = null;
443
444      return val;
445   }

598   /**
599    * Same as {@link #getParameter(String, ClassMeta)} except for use on multi-part parameters
600    *    (e.g. <js>"&key=1&key=2&key=3"</js> instead of <js>"&key=(1,2,3)"</js>)
601    * <p>
602    *    This method must only be called when parsing into classes of type Collection or array.
603    *
604    * @param name The parameter name.
605    * @param cm The class type to convert the parameter value to.
606    * @param <T> The class type to convert the parameter value to.
607    * @return The parameter value converted to the specified class type.
608    * @throws ParseException
609    */
610   @SuppressWarnings("rawtypes")
611   public <T> T getParameters(String name, ClassMeta<T> cm) throws ParseException {
612      String[] p = getParameterValues(name);
613      if (p == null)
614         return null;
615      if (cm.isArray()) {
616         List c = new ArrayList();
617         for (int i = 0; i < p.length; i++)
618            c.add(parseParameter(p[i], cm.getElementType()));
619         return (T)ArrayUtils.toArray(c, cm.getElementType().getInnerClass());
620      } else if (cm.isCollection()) {
621         try {
622            Collection c = (Collection)(cm.canCreateNewInstance() ? cm.newInstance() : new ObjectList());
623            for (int i = 0; i < p.length; i++)
624               c.add(parseParameter(p[i], cm.getElementType()));
625            return (T)c;
626         } catch (ParseException e) {
627            throw e;
628         } catch (Exception e) {
629            // Typically an instantiation exception.
630            throw new ParseException(e);
631         }
632      }
633      throw new ParseException("Invalid call to getParameters(String, ClassMeta).  Class type must be a Collection or array.");
634   }

844            throw e;
845         } catch (Exception e) {
846            // Typically an instantiation exception.

848         }

916      if (c.getInnerClass() == String.class && val.length() > 0) {
917         char x = val.charAt(0);
918         if (x != '(' && x != '$' && x != '\u0000' && val.indexOf('~') == -1)
919            return (T)val;
920      }
921      return urlEncodingParser.parseParameter(val, c);
