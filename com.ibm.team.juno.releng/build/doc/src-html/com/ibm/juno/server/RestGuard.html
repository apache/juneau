009package com.ibm.juno.server;
010
011import static javax.servlet.http.HttpServletResponse.*;
012
013import com.ibm.juno.server.annotation.*;
014
015/**
016 * REST method guard.
017 *
018 *
019 * <h6 class='topic'>Description</h6>
020 * <p>
021 *    Implements a guard mechanism for REST method calls that allows requests to be
022 *       rejected before invocation of the REST method.
023 *    For example, guards can be used to ensure that only administrators can call certain methods.
024 * <p>
025 *    Guards are applied to REST methods declaratively through the {@link RestResource#guards()} or {@link RestMethod#guards()} annotations.
026 * <p>
027 *    If multiple guards are specified, ALL guards must pass in order for the request to proceed.
028 *
029 *
030 * <h6 class='topic'>How to implement</h6>
031 * <p>
032 *    Typically, guards will be used for permissions checking on the user making the request,
033 *       but it can also be used for other purposes like pre-call validation of a request.
034 * <p>
035 *    Implementers should simply throw a {@link RestException} from the {@link #guard(RestRequest, RestResponse)}
036 *       method to abort processing on the current request.
037 * <p>
038 *    Guards must implement a no-args constructor.
039 *
040 *
041 * <h6 class='topic'>Example usage</h6>
042 * <p class='bcode'>
043 *    <jk>public</jk> MyResource <jk>extends</jk> RestServlet {
044 *
045 *       <jc>// Delete method with guard that only allows Billy to call it.</jc>
046 *       <ja>@RestMethod</ja>(name=<js>"DELETE"</js>, guards=BillyGuard.<jk>class</jk>)
047 *       <jk>public</jk> doDelete(RestRequest req, RestResponse res) <jk>throws</jk> Exception {...}
048 *    }
049 * </p>
050 *
051 *
052 * <h6 class='topic'>Example implementation</h6>
053 * <p class='bcode'>
054 *    <jc>// Define a guard that only lets Billy make a request</jc>
055 *    <jk>public</jk> BillyGuard <jk>extends</jk> RestGuard {
056 *
057 *       <ja>@Override</ja>
058 *       <jk>public boolean</jk> isRequestAllowed(RestRequest req) {
059 *          return req.getUserPrincipal().getName().contains(<js>"Billy"</js>);
060 *       }
061 *    }
062 * </p>
063 */
064public abstract class RestGuard {
065
066   /**
067    * Checks the current HTTP request and throws a {@link RestException} if the guard
068    *    does not permit the request.
069    * <p>
070    *    By default, throws an <jsf>SC_FORBIDDEN</jsf> exception if {@link #isRequestAllowed(RestRequest)}
071    *    returns <jk>false</jk>.
072    * <p>
073    *    Subclasses are free to override this method to tailor the behavior of how to handle unauthorized
074    *    requests.
075    *
076    * @param req The servlet request.
077    * @param res The servlet response.
078    * @throws RestException Thrown to abort processing on current request.
079    * @return <jk>true</jk> if request can proceed.
080    *    Specify <jk>false</jk> if you're doing something like a redirection to a login page.
081    */
082   public boolean guard(RestRequest req, RestResponse res) throws RestException {
083      if (! isRequestAllowed(req))
084         throw new RestException(SC_FORBIDDEN, "Access denied by guard");
085      return true;
086   }
087
088   /**
089    * Returns <jk>true</jk> if the specified request can pass through this guard.
090    *
091    * @param req The servlet request.
092    * @return <jk>true</jk> if the specified request can pass through this guard.
093    */
094   public abstract boolean isRequestAllowed(RestRequest req);
095}
