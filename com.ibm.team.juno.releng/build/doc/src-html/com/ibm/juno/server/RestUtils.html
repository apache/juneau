076   /**
077    * Trims <js>'/'</js> characters from both the start and end of the specified string.
078    *
079    * @param s The string to trim.
080    * @return A new trimmed string, or the same string if no trimming was necessary.
081    */
082   public static String trimSlashes(String s) {
083      if (s == null)
084         return null;
085      while (StringUtils.endsWith(s, '/'))
086         s = s.substring(0, s.length()-1);
087      while (s.length() > 0 && s.charAt(0) == '/')
088         s = s.substring(1);
089      return s;
090   }

093    * Trims <js>'/'</js> characters from the end of the specified string.
094    *
095    * @param s The string to trim.
096    * @return A new trimmed string, or the same string if no trimming was necessary.
097    */
098   public static String trimTrailingSlashes(String s) {
099      if (s == null)
100         return null;
101      while (StringUtils.endsWith(s, '/'))
102         s = s.substring(0, s.length()-1);
103      return s;
104   }

120   /**
121    * Decodes a <code>application/x-www-form-urlencoded</code> string using <code>UTF-8</code> encoding scheme.
122    *
123    * @param s The string to decode.
124    * @return The decoded string, or <jk>null</jk> if input is <jk>null</jk>.
125    */
126   public static String decode(String s) {
127      if (s == null)
128         return s;
129      boolean needsDecode = false;
130      for (int i = 0; i < s.length() && ! needsDecode; i++) {
131         char c = s.charAt(i);
132         if (c == '+' || c == '%')
133            needsDecode = true;
134      }
135      if (needsDecode)
136      try {
137            return URLDecoder.decode(s, "UTF-8");
138         } catch (UnsupportedEncodingException e) {/* Won't happen */}
139      return s;
140   }

142   // Characters that do not need to be URL-encoded
143   private static final AsciiSet unencodedChars = new AsciiSet("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789-_.!~*'()\\");
144
145   /**
146    * Encodes a <code>application/x-www-form-urlencoded</code> string using <code>UTF-8</code> encoding scheme.
147    *
148    * @param s The string to encode.
149    * @return The encoded string, or <jk>null</jk> if input is <jk>null</jk>.
150    */
151   public static String encode(String s) {
152      if (s == null)
153         return null;
154      boolean needsEncode = false;
155      for (int i = 0; i < s.length() && ! needsEncode; i++)
156         needsEncode |= (! unencodedChars.contains(s.charAt(i)));
157      if (needsEncode)
158      try {
159            return URLEncoder.encode(s, "UTF-8");
160         } catch (UnsupportedEncodingException e) {/* Won't happen */}
161      return s;
162   }

164   /**
165    * Identical to {@link HttpServletRequest#getPathInfo()} but doesn't decode encoded characters.
166    *
167    * @param req The HTTP request
168    * @return The undecoded path info.
169    */
170   public static String getPathInfoUndecoded(HttpServletRequest req) {
171      String requestURI = req.getRequestURI();
172      String contextPath = req.getContextPath();
173      String servletPath = req.getServletPath();
174      int l = contextPath.length() + servletPath.length();
175      if (requestURI.length() == l)
176         return null;
177      return requestURI.substring(l);
178   }

180   /**
181    * Efficiently trims the path info part from a request URI.
182    * <p>
183    * The result is the URI of the servlet itself.
184    *
185    * @param requestURI The value returned by {@link HttpServletRequest#getRequestURL()}
186    * @param contextPath The value returned by {@link HttpServletRequest#getContextPath()}
187    * @param servletPath The value returned by {@link HttpServletRequest#getServletPath()}
188    * @return The same StringBuilder with remainder trimmed.
189    */
190   public static StringBuffer trimPathInfo(StringBuffer requestURI, String contextPath, String servletPath) {
191      if (servletPath.equals("/"))
192         servletPath = "";
193      if (contextPath.equals("/"))
194         contextPath = "";
195
196      try {
197         // Given URL:  http://hostname:port/servletPath/extra
198         // We want:    http://hostname:port/servletPath
199         int sc = 0;
200         for (int i = 0; i < requestURI.length(); i++) {
201            char c = requestURI.charAt(i);
202            if (c == '/') {
203               sc++;
204               if (sc == 3) {
205                  if (servletPath.isEmpty()) {
206                     requestURI.setLength(i);
207                     return requestURI;
208                  }
209
210                  // Make sure context path follows the authority.
211                  for (int j = 0; j < contextPath.length(); i++, j++)
212                     if (requestURI.charAt(i) != contextPath.charAt(j))
213                        throw new Exception("case=1");
214
215                  // Make sure servlet path follows the authority.
216                  for (int j = 0; j < servletPath.length(); i++, j++)
217                     if (requestURI.charAt(i) != servletPath.charAt(j))
218                        throw new Exception("case=2");
219
220                  // Make sure servlet path isn't a false match (e.g. /foo2 should not match /foo)
221                  c = (requestURI.length() == i ? '/' : requestURI.charAt(i));
222                  if (c == '/' || c == '?') {
223                     requestURI.setLength(i);
224                     return requestURI;
225                  }
226
227                  throw new Exception("case=3");
228               }
229            } else if (c == '?') {
230               if (sc != 2)
231                  throw new Exception("case=4");
232               if (servletPath.isEmpty()) {
233                  requestURI.setLength(i);
234                  return requestURI;
235               }
236               throw new Exception("case=5");
237            }
238         }
239         if (servletPath.isEmpty())
240            return requestURI;
241         throw new Exception("case=6");
242      } catch (Exception e) {
243         throw new RuntimeException("Could not find servlet path in request URI.  URI=["+requestURI+"], servletPath=["+servletPath+"]", e);
244      }
245   }
