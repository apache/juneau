009package com.ibm.juno.server.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016/**
017 * Annotation used in conjunction with {@link RestMethod#input()} and {@link Response#output()} to identify content and header descriptions
018 *    on specific method responses.
019 *
020 * <h6 class='topic'>Example</h6>
021 * <p class='bcode'>
022 *    <ja>@RestMethod</ja>(
023 *       name=<js>"*"</js>,
024 *       requestVars={
025 *             <ja>@Var</ja>(category=<js>"header"</js>,name=<js>"Range"</js>,description=<js>"$L{ContentRange.description}"</js>)
026 *       }
027 *       responses={
028 *          <ja>@Response</ja>(code=200,description=<js>"Everything was great."</js>,
029 *             responseVars={
030 *                <ja>@Var</ja>(category=<js>"header"</js>,name=<js>"Content-Range"</js>,description=<js>"$L{ContentRange.description}"</js>)
031 *             })
032 *          <ja>@Response</ja>(code=404,description=<js>"File was not found."</js>)
033 *       }
034 *    )
035 *    <jk>public void</jk> doAnything(RestRequest req, RestResponse res, <ja>@Method</ja> String method) {
036 *       ...
037 *    }
038 * </p>
039 *
040 * @author James Bognar (jbognar@us.ibm.com)
041 */
042@Documented
043@Target(PARAMETER)
044@Retention(RUNTIME)
045@Inherited
046public @interface Var {
047
048   /**
049    * Variable category (e.g. <js>"header"</js>, <js>"content"</js>).
050    * The {@link VarCategory} class contains predefined constants.
051    */
052   String category();
053
054   /**
055    * Variable name (e.g. <js>"Content-Range"</js>).
056    */
057   String name() default "";
058
059   /**
060    * Variable description (e.g. <js>"Indicates the range returned when Range header is present in the request"</js>).
061    * <p>
062    *    The default value pulls the description from the <code>description</code> entry in the servlet resource bundle.
063    *    (e.g. <js>"myMethod.res.[code].[category].[name] = foo"</js> or <js>"MyServlet.myMethod.res.[code].[category].[name] = foo"</js>).
064    */
065   String description() default "";
066}
