009package com.ibm.juno.server;
010
011import com.ibm.juno.server.annotation.*;
012
013/**
014 * Class used for defining method-level matchers using the {@link RestMethod#matchers()} annotation.
015 * <p>
016 * Matchers are used to allow multiple Java methods to handle requests assigned to the same
017 *    URL path pattern, but differing based on some request attribute, such as a specific header value.
018 * For example, matchers can be used to provide two different methods for handling requests
019 *    from two different client versions.
020 * <p>
021 * Java methods with matchers associated with them are always attempted before Java methods
022 *    without matchers.
023 * This allows a 'default' method to be defined to handle requests where no matchers match.
024 * <p>
025 * When multiple matchers are specified on a method, only one matcher is required to match.
026 * This is opposite from the {@link RestMethod#guards()} annotation, where all guards
027 *    are required to match in order to execute the method.
028 *
029 * <h6 class='topic'>Example</h6>
030 * <p class='bcode'>
031 *    <jk>public class</jk> MyResource <jk>extends</jk> RestServlet {
032 *
033 *       <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/foo"</js>, matchers=IsDNT.<jk>class</jk>)
034 *       <jk>public</jk> Object doGetWithDNT() {
035 *          <jc>// Handle request with Do-Not-Track specified</jc>
036 *       }
037 *
038 *       <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/foo"</js>)
039 *       <jk>public</jk> Object doGetWithoutDNT() {
040 *          <jc>// Handle request without Do-Not-Track specified</jc>
041 *       }
042 *    }
043 *
044 *    <jk>public class</jk> IsDNT <jk>extends</jk> RestMatcher {
045 *       <ja>@Override</ja>
046 *       <jk>public boolean</jk> matches(RestRequest req) {
047 *          <jk>return</jk> req.getHeader(<jk>int</jk>.<jk>class</jk>, <js>"DNT"</js>, 0) == 1;
048 *       }
049 *    }
050 * </p>
051 */
052public abstract class RestMatcher {
053
054   /**
055    * Returns <jk>true</jk> if the specified request matches this matcher.
056    *
057    * @param req The servlet request.
058    * @return <jk>true</jk> if the specified request matches this matcher.
059    */
060   public abstract boolean matches(RestRequest req);
061}
