009package com.ibm.juno.server.jaxrs;
010
011import static javax.servlet.http.HttpServletResponse.*;
012
013import java.io.*;
014import java.lang.annotation.*;
015import java.lang.reflect.*;
016
017import javax.ws.rs.*;
018import javax.ws.rs.core.*;
019import javax.ws.rs.ext.*;
020
021import com.ibm.juno.core.*;
022import com.ibm.juno.core.parser.*;
023import com.ibm.juno.core.serializer.*;
024import com.ibm.juno.core.utils.*;
025import com.ibm.juno.server.annotation.*;
026
027/**
028 * Base class for defining JAX-RS providers based on Juno serializers and parsers.
029 *
030 * @author James Bognar (jbognar@us.ibm.com)
031 */
032public class BaseProvider implements MessageBodyReader<Object>, MessageBodyWriter<Object> {
033
034   private SerializerGroup serializers = new SerializerGroup();
035   private ParserGroup parsers = new ParserGroup();
036   private ObjectMap properties = new ObjectMap();
037
038   /**
039    * Constructor.
040    */
041   protected BaseProvider() {
042      try {
043         properties = new ObjectMap();
044         JunoProvider jp = getClass().getAnnotation(JunoProvider.class);
045         serializers.append(jp.serializers());
046         parsers.append(jp.parsers());
047         for (Property p : jp.properties())
048            properties.put(p.name(), p.value());
049         serializers.addFilters(jp.filters());
050         parsers.addFilters(jp.filters());
051      } catch (Exception e) {
052         throw new RuntimeException(e);
053      }
054   }
055
056   /**
057    * Returns properties defined on the specified method through the {@link RestMethod#properties()}
058    *    annotation specified on the method and the {@link JunoProvider#properties()} annotation
059    *    specified on the provider class.
060    *
061    * @param a All annotations defined on the method.
062    * @return A map of all properties define on the method.
063    */
064   protected ObjectMap getMethodProperties(Annotation[] a) {
065      ObjectMap m = new ObjectMap().setInner(properties);
066      for (Annotation aa : a) {
067         if (aa instanceof RestMethod) {
068            for (Property p : ((RestMethod)aa).properties())
069               m.put(p.name(), p.value());
070         }
071      }
072      return m;
073   }
074
075   @Override /* MessageBodyWriter */
076   public long getSize(Object o, Class<?> type, Type gType, Annotation[] a, MediaType mediaType) {
077      return -1;
078   }
079
080   @Override /* MessageBodyWriter */
081   public boolean isWriteable(Class<?> type, Type gType, Annotation[] a, MediaType mediaType) {
082      return serializers.findMatch(mediaType.toString()) != null;
083   }
084
085   @Override /* MessageBodyWriter */
086   public void writeTo(Object o, Class<?> type, Type gType, Annotation[] a, MediaType mediaType,
087         MultivaluedMap<String,Object> headers, OutputStream out) throws IOException, WebApplicationException {
088      try {
089         String mt = serializers.findMatch(mediaType.toString());
090         if (mt == null)
091            throw new WebApplicationException(SC_NOT_ACCEPTABLE);
092         Serializer<?> s = serializers.getSerializer(mt);
093         ObjectMap mp = getMethodProperties(a);
094         mp.append("mediaType", mediaType.toString());
095         SerializerContext ctx = s.createContext(mp, null);
096         if (s.isWriterSerializer()) {
097            WriterSerializer s2 = (WriterSerializer)s;
098            OutputStreamWriter w = new OutputStreamWriter(out, IOUtils.UTF8);
099            s2.serialize(o, w, ctx);
100            w.flush();
101            w.close();
102         } else {
103            OutputStreamSerializer s2 = (OutputStreamSerializer)s;
104            s2.serialize(o, out, ctx);
105            out.flush();
106            out.close();
107         }
108      } catch (SerializeException e) {
109         throw new IOException(e);
110      }
111   }
112
113   @Override /* MessageBodyReader */
114   public boolean isReadable(Class<?> type, Type gType, Annotation[] a, MediaType mediaType) {
115      return parsers.findMatch(mediaType.toString()) != null;
116   }
117
118   @Override /* MessageBodyReader */
119   public Object readFrom(Class<Object> type, Type gType, Annotation[] a, MediaType mediaType,
120         MultivaluedMap<String,String> headers, InputStream in) throws IOException, WebApplicationException {
121      try {
122         String mt = parsers.findMatch(mediaType.toString());
123         if (mt == null)
124            throw new WebApplicationException(SC_UNSUPPORTED_MEDIA_TYPE);
125         Parser<?> p = parsers.getParser(mt);
126         BeanContext bc = p.getBeanContext();
127         ClassMeta<?> cm = bc.getClassMeta(gType);
128         ObjectMap mp = getMethodProperties(a);
129         mp.put("mediaType", mediaType.toString());
130         int length = IOUtils.getBufferSize(headers.getFirst("Content-Length"));
131         if (p.isReaderParser()) {
132            ReaderParser p2 = (ReaderParser)p;
133            InputStreamReader r = new InputStreamReader(in, IOUtils.UTF8);
134            ParserContext ctx = p2.createContext(mp, null, null);
135            return p2.parse(r, length, cm, ctx);
136         }
137         InputStreamParser p2 = (InputStreamParser)p;
138         ParserContext ctx = p2.createContext(mp, null, null);
139         return p2.parse(in, length, cm, ctx);
140      } catch (ParseException e) {
141         throw new IOException(e);
142      }
143   }
144}
