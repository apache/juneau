025/**
026 * Represents an HTTP response for a REST resource.
027 * <p>
028 * Essentially an extended {@link HttpServletResponse} with some special convenience methods
029 *    that allow you to easily output POJOs as responses.
030 * </p>
031 * <p>
032 * Since this class extends {@link HttpServletResponse}, developers are free to use these
033 *    convenience methods, or revert to using lower level methods like any other servlet response.
034 * </p>
035 *
036 * <h6 class='topic'>Examples</h6>
037 * <p class='bcode'>
038 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>)
039 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res) {
040 *       res.setProperty(HtmlSerializerProperties.<jsf>HTMLDOC_title</jsf>, <js>"My title"</js>)
041 *          .setOutput(<js>"Simple string response"</js>);
042 *    }
043 * </p>
044 * <p>
045 *    Refer to <a class='doclink' href='package-summary.html#TOC'>REST Servlet API</a> for information about using this class.
046 * </p>
047 *
048 * @author jbognar
049 */
050public final class RestResponse extends HttpServletResponseWrapper {
051
052   private final RestRequest request;
053   private Object output;                               // The POJO being sent to the output.
054   private boolean isNullOutput;                        // The output is null (as opposed to not being set at all)
055   private ObjectMap properties;                        // Response properties
056   SerializerGroup serializerGroup;
057   UrlEncodingSerializer urlEncodingSerializer;         // The serializer used to convert arguments passed into Redirect objects.
058   private EncoderGroup encoders;
059   private RestServlet servlet;
060   private ServletOutputStream os;
061
062   /**
063    * Constructor.
064    */
065   RestResponse(RestServlet servlet, RestRequest req, HttpServletResponse res) throws RestServletException {
066      super(res);
067      this.request = req;
068      this.servlet = servlet;
069
070      for (Map.Entry<String,Object> e : servlet.getDefaultResponseHeaders().entrySet())
071         setHeader(e.getKey(), e.getValue().toString());
072
073      try {
074         String passThroughHeaders = req.getHeader("x-response-headers");
075         if (passThroughHeaders != null) {
076            ObjectMap m = servlet.getUrlEncodingParser().parseParameter(passThroughHeaders, ObjectMap.class);
077            for (Map.Entry<String,Object> e : m.entrySet())
078               setHeader(e.getKey(), e.getValue().toString());
079         }
080      } catch (Exception e1) {
081         throw new RestException(SC_BAD_REQUEST, "Invalid format for header 'x-response-headers'.  Must be in URL-encoded format.").initCause(e1);
082      }
083   }

085   /*
086    * Called from RestServlet after a match has been made but before the guard or method invocation.
087    */
088   @SuppressWarnings("hiding")
089   final void init(ObjectMap properties, String defaultCharset, SerializerGroup mSerializers, UrlEncodingSerializer mUrlEncodingSerializer, EncoderGroup encoders) {
090      this.properties = properties;
091      this.serializerGroup = mSerializers;
092      this.urlEncodingSerializer = mUrlEncodingSerializer;
093      this.encoders = encoders;
094
095      // Find acceptable charset
096      String h = request.getHeader("accept-charset");
097      String charset = null;
098      if (h == null)
099         charset = defaultCharset;
100      else for (MediaRange r : MediaRange.parse(h)) {
101         if (r.getQValue() > 0) {
102            if (r.getType().equals("*"))
103               charset = defaultCharset;
104            else if (RestServlet.availableCharsets.containsKey(r.getType()))
105               charset = r.getType();
106            if (charset != null)
107               break;
108         }
109      }
110
111      if (charset == null)
112         throw new RestException(SC_NOT_ACCEPTABLE, "No supported charsets in header ''Accept-Charset'': ''{0}''", request.getHeader("Accept-Charset"));
113      super.setCharacterEncoding(charset);
114   }

251   /**
252    * Equivalent to {@link HttpServletResponse#getOutputStream()}, except
253    *    wraps the output stream if an {@link Encoder} was found that matched
254    *    the <code>Accept-Encoding</code> header.
255    *
256    * @return A negotiated output stream.
257    * @throws IOException
258    */
259   public ServletOutputStream getNegotiatedOutputStream() throws IOException {
260      if (os == null) {
261         Encoder encoder = null;
262
263         String ae = request.getHeader("Accept-Encoding");
264         if (! (ae == null || ae.isEmpty())) {
265            String match = encoders != null ? encoders.findMatch(ae) : null;
266            if (match == null) {
267               // Identity should always match unless "identity;q=0" or "*;q=0" is specified.
268               if (ae.matches(".*(identity|\\*)\\s*;\\s*q\\s*=\\s*(0(?!\\.)|0\\.0).*")) {
269                  throw new RestException(SC_NOT_ACCEPTABLE,
270                     "Unsupported encoding in request header ''Accept-Encoding'': ''{0}''\n\tSupported codings: {1}",
271                     ae, encoders.getSupportedEncodings()
272                  );
273               }
274            } else {
275               encoder = encoders.getEncoder(match);
276
277               // Some clients don't recognize identity as an encoding, so don't set it.
278               if (! match.equals("identity"))
279               setHeader("content-encoding", match);
280            }
281         }
282         os = getOutputStream();
283         if (encoder != null) {
284            final OutputStream os2 = encoder.getOutputStream(os);
285            os = new ServletOutputStream(){
286               @Override /* OutputStream */
287               public final void write(byte[] b, int off, int len) throws IOException {
288                  os2.write(b, off, len);
289               }
290               @Override /* OutputStream */
291               public final void write(int b) throws IOException {
292                  os2.write(b);
293               }
294               @Override /* OutputStream */
295               public final void flush() throws IOException {
296                  os2.flush();
297               }
298               @Override /* OutputStream */
299               public final void close() throws IOException {
300                  os2.close();
301               }
302            };
303         }
304      }
305      return os;
306   }

308   @Override /* ServletResponse */
309   public ServletOutputStream getOutputStream() throws IOException {
310      if (os == null)
311         os = super.getOutputStream();
312      return os;
313   }

334   /**
335    * Convenience method meant to be used when rendering directly to a browser with no buffering.
336    * Sets the header <js>"x-content-type-options=nosniff"</js> so that output is rendered
337    * immediately on IE and Chrome without any buffering for content-type sniffing.
338    *
339    * @param contentType The value to set as the <code>Content-Type</code> on the response.
340    * @return The raw writer.
341    * @throws IOException
342    */
343   public PrintWriter getDirectWriter(String contentType) throws IOException {
344      setContentType(contentType);
345      setHeader("x-content-type-options", "nosniff");
346      return getWriter();
347   }

349   /**
350    * Equivalent to {@link HttpServletResponse#getWriter()}, except
351    *    wraps the output stream if an {@link Encoder} was found that matched
352    *    the <code>Accept-Encoding</code> header and sets the <code>Content-Encoding</code>
353    *    header to the appropriate value.
354    *
355    * @return The negotiated writer.
356    * @throws IOException
357    */
358   public PrintWriter getNegotiatedWriter() throws IOException {
359      return getWriter(false);
360   }
361
362   private PrintWriter getWriter(boolean raw) throws IOException {
363      // If plain text requested, override it now.
364      if (request.isPlainText()) {
365         setHeader("Content-Type", "text/plain");
366      }
367
368      try {
369         OutputStream out = (raw ? getOutputStream() : getNegotiatedOutputStream());
370         return new PrintWriter(new OutputStreamWriter(out, getCharacterEncoding()));
371      } catch (UnsupportedEncodingException e) {
372         String ce = getCharacterEncoding();
373         setCharacterEncoding("UTF-8");
374         throw new RestException(SC_NOT_ACCEPTABLE, "Unsupported charset in request header ''Accept-Charset'': ''{0}''", ce);
375      }
376   }

378   /**
379    * Returns the <code>Content-Type</code> header stripped of the charset attribute if present.
380    *
381    * @return The <code>media-type</code> portion of the <code>Content-Type</code> header.
382    */
383   public String getMediaType() {
384      String contentType = getContentType();
385      if (contentType == null)
386         return null;
387      int i = contentType.indexOf(';');
388      if (i == -1)
389         return contentType;
390      return contentType.substring(0, i).trim();
391
392   }

394   /**
395    * Redirects to the specified URI.
396    * <p>
397    * Relative URIs are always interpreted as relative to the context root.
398    * This is similar to how WAS handles redirect requests, and is different from how Tomcat
399    *    handles redirect requests.
400    */
401   @Override /* ServletResponse */
402   public void sendRedirect(String uri) throws IOException {
403      char c = (uri.length() > 0 ? uri.charAt(0) : 0);
404      if (c != '/' && uri.indexOf("://") == -1)
405         uri = request.getContextPath() + '/' + uri;
406      super.sendRedirect(uri);
407   }

418   @Override /* ServletResponse */
419   public void setHeader(String name, String value) {
420      // Jetty doesn't set the content type correctly if set through this method.
421      // Tomcat/WAS does.
422      if (name.equalsIgnoreCase("Content-Type"))
423         super.setContentType(value);
424      else
425         super.setHeader(name, value);
426   }
