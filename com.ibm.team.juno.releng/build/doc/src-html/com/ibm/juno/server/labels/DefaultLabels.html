009package com.ibm.juno.server.labels;
010
011import static java.lang.String.*;
012
013import java.util.*;
014
015import com.ibm.juno.server.converters.*;
016
017/**
018 * Reusable label constructs for REST OPTIONS requests.
019 *
020 * @author James Bognar (jbognar@us.ibm.com)
021 */
022public final class DefaultLabels {
023
024   private static String
025      bundleName = DefaultLabels.class.getPackage().getName() + ".nls.DefaultLabels",
026      typeKey = "%s.%s.type", // "{category}.{name}.type"
027      descKey = "%s.%s.desc"; // "{catetory}.{name}.desc"
028
029   private static Map<Locale,ParamDescription[]> queryableParams = new HashMap<Locale,ParamDescription[]>();
030   private static Map<Locale,ParamDescription[]> headerParams = new HashMap<Locale,ParamDescription[]>();
031
032   /**
033    * OPTIONS request labels for query/view/sort/paging features on POJOs
034    *    when {@link Queryable} filter is associated with a resource method.
035    * @param locale The client locale.
036    * @return A list of localized parameter descriptions.
037    */
038   public static ParamDescription[] getQueryableParamDescriptions(Locale locale) {
039      if (! queryableParams.containsKey(locale)) {
040         ResourceBundle rb = ResourceBundle.getBundle(bundleName, locale);
041         String category = "QueryableParam";
042         String[] p = {"q","v","s","g","i","p","l"};
043         ParamDescription[] l = new ParamDescription[p.length];
044         for (int i = 0; i < p.length; i++)
045            l[i] = new ParamDescription(p[i],
046               rb.getString(format(typeKey, category, p[i])),
047               rb.getString(format(descKey, category, p[i]))
048            );
049         queryableParams.put(locale, l);
050      }
051      return queryableParams.get(locale);
052   }
053
054   /**
055    * OPTIONS request labels for header values that can be specified as GET parameters.
056    * @param locale The client locale.
057    * @return A list of localized parameter descriptions.
058    */
059   public static ParamDescription[] getHeaderParamDescriptions(Locale locale) {
060      if (! headerParams.containsKey(locale)) {
061         ResourceBundle rb = ResourceBundle.getBundle(bundleName, locale);
062         String category = "HeaderParam";
063         String[] p = {"Accept","Accept-Encoding","Method","Content","plainText"};
064         ParamDescription[] l = new ParamDescription[p.length];
065         for (int i = 0; i < p.length; i++)
066            l[i] = new ParamDescription(p[i],
067               rb.getString(format(typeKey, category, p[i])),
068               rb.getString(format(descKey, category, p[i]))
069            );
070         headerParams.put(locale, l);
071      }
072      return headerParams.get(locale);
073   }
074}
