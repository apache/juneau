009package com.ibm.juno.server.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016import com.ibm.juno.core.*;
017import com.ibm.juno.core.jena.*;
018import com.ibm.juno.core.json.*;
019import com.ibm.juno.core.parser.*;
020import com.ibm.juno.core.serializer.*;
021import com.ibm.juno.core.xml.*;
022
023/**
024 * Property name/value pair used in the {@link RestResource#properties()} annotation.
025 * <p>
026 *    Any of the following property names can be specified:
027 * <ul>
028 *    <li>{@link BeanContextProperties}
029 *    <li>{@link SerializerProperties}
030 *    <li>{@link ParserProperties}
031 *    <li>{@link JsonSerializerProperties}
032 *    <li>{@link RdfSerializerProperties}
033 *    <li>{@link RdfParserProperties}
034 *    <li>{@link RdfProperties}
035 *    <li>{@link XmlSerializerProperties}
036 *    <li>{@link XmlParserProperties}
037 * </ul>
038 * <p>
039 *    Property values types that are not <code>Strings</code> will automatically be converted to the
040 *       correct type (e.g. <code>Boolean</code>, etc...).
041 * <p>
042 *    See {@link RestResource#properties} for more information.
043 *
044 * @author James Bognar (jbognar@us.ibm.com)
045 */
046@Documented
047@Target(ANNOTATION_TYPE)
048@Retention(RUNTIME)
049@Inherited
050public @interface Property {
051
052   /**
053    * Property name.
054    */
055   String name();
056
057   /**
058    * Property value.
059    */
060   String value();
061}
