009package com.ibm.juno.server.response;
010
011import java.io.*;
012
013import com.ibm.juno.core.utils.*;
014import com.ibm.juno.server.*;
015
016/**
017 * Response handler for {@link Redirect} objects.
018 * <p>
019 * This handler is registered by default on {@link RestServlet RestServlets} via the
020 *    default implementation of the {@link RestServlet#createResponseHandlers} method.
021 *
022 * @author James Bognar (jbognar@us.ibm.com)
023 */
024public final class RedirectHandler implements ResponseHandler {
025
026   @Override /* ResponseHandler */
027   public boolean handle(RestRequest req, RestResponse res, Object output) throws IOException, RestException {
028      if (output instanceof Redirect) {
029         Redirect r = (Redirect)output;
030         String uri = r.toUrl(res.getUrlEncodingSerializer());
031         if (StringUtils.isEmpty(uri))
032            uri = req.getServletURI();
033         else {
034            char c = (uri.length() > 0 ? uri.charAt(0) : 0);
035            if (c != '/' && uri.indexOf("://") == -1)
036               uri = req.getServletURIBuilder().append('/').append(uri).toString();
037         }
038         int rc = r.getHttpResponseCode();
039         if (rc != 0)
040            res.setStatus(rc);   // TODO - This may get ignored by the call below.
041         res.sendRedirect(uri);
042         return true;
043      }
044      return false;
045   }
046}
