009package com.ibm.juno.server.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016import com.ibm.juno.server.*;
017
018/**
019 * Annotation that can be applied to a parameter of a {@link RestMethod} annotated method
020 *    to identify whether or not the request has the specified GET or POST parameter.
021 * <p>
022 * Note that this can be used to detect the existence of a parameter when it's not set to a particular value.
023 * <h6 class='topic'>Example</h6>
024 * <p class='bcode'>
025 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>)
026 *    <jk>public void</jk> doGet(<ja>@HasParam</ja>(<js>"p1"</js>) <jk>boolean</jk> p1) {
027 *       ...
028 *    }
029 * </p>
030 * <p>
031 *    This is functionally equivalent to the following code...
032 * </p>
033 * <p class='bcode'>
034 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>)
035 *    <jk>public void</jk> doGet(RestRequest req) {
036 *       <jk>boolean</jk> p1 = req.hasParameter(<js>"p1"</js>);
037 *       ...
038 *    }
039 * </p>
040 * <p>
041 * The following table shows the behavioral differences between <code>@HasParam</code> and <code>@Param</code>...
042 * <table class='styled'>
043 *    <tr>
044 *       <th><code>URL</code></th>
045 *       <th><code><ja>@HasParam</ja>(<js>"a"</js>)</code></th>
046 *       <th><code><ja>@Param</ja>(<js>"a"</js>)</code></th>
047 *    </tr>
048 *    <tr>
049 *       <td><code>?a=foo</code></td>
050 *       <td><code><jk>true</jk></td>
051 *       <td><code><js>"foo"</js></td>
052 *    </tr>
053 *    <tr>
054 *       <td><code>?a=</code></td>
055 *       <td><code><jk>true</jk></td>
056 *       <td><code><js>""</js></td>
057 *    </tr>
058 *    <tr>
059 *       <td><code>?a</code></td>
060 *       <td><code><jk>true</jk></td>
061 *       <td><code><jk>null</jk></td>
062 *    </tr>
063 *    <tr>
064 *       <td><code>?b=foo</code></td>
065 *       <td><code><jk>false</jk></td>
066 *       <td><code><jk>null</jk></td>
067 *    </tr>
068 * </table>
069 *
070 * <h6 class='topic'>Important note concerning FORM posts</h6>
071 * <p>
072 * This annotation should not be combined with the {@link Content @Content} annotation or {@link RestRequest#getInput(Class)} method
073 *    for <code>application/x-www-form-urlencoded POST</code> posts, since it will trigger the underlying servlet API to parse the body 
074 *    content as key-value pairs, resulting in empty content.
075 * <p>
076 * The {@link HasQParam @HasQParam} annotation can be used to check for the existing of a URL parameter
077 *    in the URL string without triggering the servlet to drain the body content.
078 *
079 * @author James Bognar (jbognar@us.ibm.com)
080 */
081@Documented
082@Target(PARAMETER)
083@Retention(RUNTIME)
084@Inherited
085public @interface HasParam {
086
087   /**
088    * URL parameter name.
089    */
090   String value();
091}
