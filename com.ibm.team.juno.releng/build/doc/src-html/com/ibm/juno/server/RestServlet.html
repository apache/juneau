147   @Override /* Servlet */
148   public synchronized void init(ServletConfig servletConfig) throws ServletException {
149      try {
150         log(FINE, "Servlet {0} init called.", getClass().getName());
151         this.servletConfig = servletConfig;
152
153         if (isInitialized)
154            return;
155
156         super.init(servletConfig);
157
158         // Find resource resource bundle location.
159         for (Map.Entry<Class<?>,RestResource> e : restResourceAnnotationsChildFirst.entrySet()) {
160            Class<?> c = e.getKey();
161            RestResource r = e.getValue();
162            if (! r.messages().isEmpty()) {
163               if (msgs == null)
164                  msgs = new MessageBundle(c, r.messages());
165               else
166                  msgs.addSearchPath(c, r.messages());
167            }
168            if (label.isEmpty())
169               label = r.label();
170            if (description.isEmpty())
171               description = r.description();
172         }
173         if (msgs == null)
174            msgs = new MessageBundle(this.getClass(), "");
175
176         styleSheet = createStyleSheet();
177         favIcon = createFavIcon();
178         staticFilesMap = Collections.unmodifiableMap(createStaticFilesMap());
179         staticFilesPrefixes = staticFilesMap.keySet().toArray(new String[0]);
180
181         ObjectMap p = getProperties();
182         renderResponseStackTraces = p.getBoolean(REST_renderResponseStackTraces, false);
183         useStackTraceHashes = p.getBoolean(REST_useStackTraceHashes, true);
184
185         allowHeaderParams = p.getBoolean(REST_allowHeaderParams, true);
186
187         allowMethodParams = new LinkedHashSet<String>();
188         for (String m : StringUtils.split(p.getString(REST_allowMethodParam, ""), ','))
189            if (m.equals("true"))  // For backwards compatibility when this was a boolean field.
190               allowMethodParams.add("*");
191            else
192               allowMethodParams.add(m.toUpperCase());
193
194         allowContentParam = p.getBoolean(REST_allowContentParam, true);
195         defaultCharset = p.getString(REST_defaultCharset, "utf-8");
196
197         // Discover the @RestMethod methods available on the resource.
198         List<String> methodsFound = new LinkedList<String>();   // Temporary to help debug transient duplicate method issue.
199         for (java.lang.reflect.Method method : this.getClass().getMethods()) {
200            if (method.isAnnotationPresent(RestMethod.class)) {
201               RestMethod a = method.getAnnotation(RestMethod.class);
202               methodsFound.add(method.getName() + "," + a.name() + "," + a.path());
203               try {
204                  if (! Modifier.isPublic(method.getModifiers()))
205                     throw new RestServletException("@RestMethod method {0}.{1} must be defined as public.", this.getClass().getName(), method.getName());
206
207                  MethodMeta sm = new MethodMeta(method);
208                  javaRestMethods.put(method.getName(), sm);
209                  ResourceMethod rm = restMethods.get(sm.httpMethod);
210                  if (rm == null)
211                     restMethods.put(sm.httpMethod, sm);
212                  else if (rm instanceof MultiMethod)
213                     ((MultiMethod)rm).addSimpleMethod(sm);
214                  else
215                     restMethods.put(sm.httpMethod, new MultiMethod((MethodMeta)rm, sm));
216               } catch (RestServletException e) {
217                  throw new RestServletException("Problem occurred trying to serialize methods on class {0}, methods={1}", this.getClass().getName(), JsonSerializer.DEFAULT_LAX.serialize(methodsFound)).initCause(e);
218               }
219            }
220         }
221
222         for (ResourceMethod m : restMethods.values())
223            m.complete();
224
225         // Discover the child resources.
226         childResources.putAll(createChildrenMap());
227
228         for (RestServlet child : childResources.values())
229            child.init(servletConfig);
230
231         // Make sure all lazy-loaded fields are loaded.
232         // This prevents us from needing synchronized blocks in all those methods.
233         getBeanContext();
234         getSerializers();
235         getParsers();
236         getUrlEncodingParser();
237         getConverters();
238         getEncoders();
239         getFilters();
240         getGuards();
241         getMimetypesFileTypeMap();
242         getDefaultRequestHeaders();
243         getDefaultResponseHeaders();
244         getResponseHandlers();
245
246      } catch (RestException e) {
247         // Thrown RestExceptions are simply caught and rethrown on subsequent calls to service().
248         initException = e;
249         log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
250         label = String.valueOf(initException.getLocalizedMessage());
251      } catch (ServletException e) {
252         initException = e;
253         log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
254         label = String.valueOf(initException.getLocalizedMessage());
255         throw e;
256      } catch (Exception e) {
257         initException = e;
258         log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
259         label = String.valueOf(initException.getLocalizedMessage());
260         throw new ServletException(e);
261      } catch (Throwable e) {
262         initException = new Exception(e);
263         log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
264         label = String.valueOf(initException.getLocalizedMessage());
265         throw new ServletException(e);
266      } finally {
267         isInitialized = true;
268      }
269   }

271   /**
272    * Creates the child resources of this resource.
273    * <p>
274    *    Default implementation calls {@link #createChildren()} and uses the {@link RestResource#path() @RestResource.path()} annotation
275    *       on each child to identify the subpath for the resource which become the keys in this map.
276    *    It then calls the {@link #setParent(RestServlet)} method on the child resource.
277    * </p>
278    * <p>
279    *    Subclasses can override this method to programatically create child resources
280    *       without using the {@link RestResource#children() @RestResource.children()} annotation.
281    *    When overridding this method, you are responsible for calling {@link #setParent(RestServlet)} on the
282    *       child resources.
283    * </p>
284    *
285    * @return The new mutable list of child resource instances.
286    * @throws Exception If an error occurred during servlet instantiation.
287    */
288   protected Map<String,RestServlet> createChildrenMap() throws Exception {
289      Map<String,RestServlet> m = new LinkedHashMap<String,RestServlet>();
290      for (RestServlet r : createChildren()) {
291         r.setParent(this);
292         String p = r.findPath();
293         if (p == null)
294            throw new RestServletException("Child resource ''{0}'' does not define a ''@RestResource.path'' attribute.", r.getClass().getName());
295         m.put(p, r);
296      }
297      return m;
298   }

301    * Creates instances of child resources for this servlet.
302    * <p>
303    *    Default implementation uses the {@link RestResource#children() @RestResource.children()} annotation to identify and
304    *       instantiate children.
305    * </p>
306    * <p>
307    *    Subclasses can override this method to programatically create child resources
308    *       without using the {@link RestResource#children() @RestResource.children()} annotation.
309    * </p>
310    *
311    * @return The new mutable list of child resource instances.
312    * @throws Exception If an error occurred during servlet instantiation.
313    */
314   protected List<RestServlet> createChildren() throws Exception {
315      List<RestServlet> l = new LinkedList<RestServlet>();
316      for (Class<?> c : getChildClasses()) {
317         if (isParentClass(RestServlet.class, c))
318            l.add((RestServlet)c.newInstance());
319         else
320            l.add(resolveChild(c));
321      }
322      return l;
323   }

435    * <p>

437    * </p>

439   @Override /* Servlet */
440   public void service(HttpServletRequest r1, HttpServletResponse r2) throws ServletException, IOException {
441
442      log(FINE, "HTTP: {0} {1}", r1.getMethod(), r1.getRequestURI());
443      long startTime = System.currentTimeMillis();
444
445      try {
446
447         if (initException != null) {
448            if (initException instanceof RestException)
449               throw (RestException)initException;
450            throw new RestException(SC_INTERNAL_SERVER_ERROR, initException);
451         }
452
453         if (! isInitialized)
454            throw new RestException(SC_INTERNAL_SERVER_ERROR, "Servlet has not been initialized");
455
456         String pathInfo = RestUtils.getPathInfoUndecoded(r1);  // Can't use r1.getPathInfo() because we don't want '%2F' resolved.
457
458         // If this resource has child resources, try to recursively call them.
459         if (pathInfo != null && (! childResources.isEmpty()) && (! pathInfo.equals("/"))) {
460            int i = pathInfo.indexOf('/', 1);
461            String pathInfoPart = i == -1 ? pathInfo.substring(1) : pathInfo.substring(1, i);
462            RestServlet childResource = childResources.get(pathInfoPart);
463            if (childResource != null) {
464               final String pathInfoRemainder = (i == -1 ? null : pathInfo.substring(i));
465               final String servletPath = r1.getServletPath() + "/" + pathInfoPart;
466               final HttpServletRequest childRequest = new HttpServletRequestWrapper(r1) {
467                  @Override /* ServletRequest */
468                  public String getPathInfo() {
469                     return RestUtils.decode(pathInfoRemainder);
470                  }
471                  @Override /* ServletRequest */
472                  public String getServletPath() {
473                     return servletPath;
474                  }
475               };
476               childResource.service(childRequest, r2);
477               return;
478            }
479         }
480
481         RestRequest req = createRequest(r1);
482         RestResponse res = createResponse(req, r2);
483         String method = req.getMethod();
484         String methodUC = method.toUpperCase(Locale.ENGLISH);
485
486         StreamResource r = null;
487         if (pathInfo != null) {
488            String p = pathInfo.substring(1);
489            if (p.equals("favicon.ico"))
490               r = favIcon;
491            else if (p.equals("style.css"))
492               r = styleSheet;
493            else if (StringUtils.pathStartsWith(p, staticFilesPrefixes))
494               r = resolveStaticFile(p);
495         }
496
497         if (r != null) {
498            res.setStatus(SC_OK);
499            res.setOutput(r);
500         } else {
501            // If the specified method has been defined in a subclass, invoke it.
502            int rc = SC_METHOD_NOT_ALLOWED;
503            if (restMethods.containsKey(methodUC)) {
504               rc = restMethods.get(methodUC).invoke(method, pathInfo, this, req, res);
505            } else if (restMethods.containsKey("*")) {
506               rc = restMethods.get("*").invoke(method, pathInfo, this, req, res);
507            }
508
509            // If not invoked above, see if it's an OPTIONs request
510            if (rc != SC_OK)
511               handleNotFound(rc, req, res);
512         }
513
514         if (res.hasOutput()) {
515            Object output = res.getOutput();
516
517            // Do any class-level filtering.
518            for (RestConverter converter : getConverters())
519               output = converter.convert(req, output, getBeanContext().getClassMetaForObject(output));
520
521            res.setOutput(output);
522
523            // Now serialize the output if there was any.
524            // Some subclasses may write to the OutputStream or Writer directly.
525            handleResponse(req, res, output);
526         }
527
528         onSuccess(req, res, System.currentTimeMillis() - startTime);
529
530      } catch (RestException e) {
531         handleError(r1, r2, e);
532      } catch (Throwable e) {
533         handleError(r1, r2, new RestException(SC_INTERNAL_SERVER_ERROR, e));
534      }
535      log(FINE, "HTTP: [{0} {1}] finished in {2}ms", r1.getMethod(), r1.getRequestURI(), System.currentTimeMillis()-startTime);
536   }

538   /**
539    * Handle the case where a matching method was not found.
540    * <p>
541    *    Subclasses can override this method to provide a 2nd-chance for specifying a response.
542    *    The default implementation will simply throw an exception with an appropriate message.
543    * </p>
544    *
545    * @param rc The HTTP response code.
546    * @param req The HTTP request.
547    * @param res The HTTP response.
548    * @throws Exception
549    */
550   protected void handleNotFound(int rc, RestRequest req, RestResponse res) throws Exception {
551      String pathInfo = req.getPathInfo();
552      String methodUC = req.getMethod();
553      String onPath = pathInfo == null ? " on no pathInfo"  : format(" on path '%s'", pathInfo);
554      if (rc == SC_NOT_FOUND)
555         throw new RestException(rc, "Method ''{0}'' not found on resource with matching pattern{1}.", methodUC, onPath);
556      else if (rc == SC_PRECONDITION_FAILED)
557         throw new RestException(rc, "Method ''{0}'' not found on resource{1} with matching matcher.", methodUC, onPath);
558      else if (rc == SC_METHOD_NOT_ALLOWED)
559         throw new RestException(rc, "Method ''{0}'' not found on resource.", methodUC);
560      else
561         throw new ServletException("Invalid method response: " + rc);
562   }

564   private synchronized void handleError(HttpServletRequest req, HttpServletResponse res, RestException e) throws IOException {
565      Integer c = 1;
566      if (useStackTraceHashes) {
567         int h = e.hashCode();
568         c = stackTraceHashes.get(h);
569         if (c == null)
570            c = 1;
571         else
572            c++;
573         stackTraceHashes.put(h, c);
574         e.setOccurrence(c);
575      }
576      onError(req, res, e);
577      renderError(req, res, e);
578   }

580   /**
581    * Method for rendering response errors.
582    * <p>
583    *    The default implementation renders a plain text English message, optionally with a stack trace
584    *       if {@link RestServletProperties#REST_renderResponseStackTraces} is enabled.
585    * </p>
586    * <p>
587    *    Subclasses can override this method to provide their own custom error response handling.
588    * </p>
589    *
590    * @param req The servlet request.
591    * @param res The servlet response.
592    * @param e The exception that occurred.
593    * @throws IOException Can be thrown if a problem occurred trying to write to the output stream.
594    */
595   protected void renderError(HttpServletRequest req, HttpServletResponse res, RestException e) throws IOException {
596
597      int status = e.getStatus();
598      res.setStatus(status);
599      res.setContentType("text/plain");
600      res.setHeader("Content-Encoding", "identity");
601      PrintWriter w = null;
602      try {
603         w = res.getWriter();
604      } catch (IllegalStateException e2) {
605         w = new PrintWriter(new OutputStreamWriter(res.getOutputStream(), IOUtils.UTF8));
606      }
607      String httpMessage = RestUtils.getHttpResponseText(status);
608      if (httpMessage != null)
609         w.append("HTTP ").append(String.valueOf(status)).append(": ").append(httpMessage).append("\n\n");
610      if (renderResponseStackTraces)
611         e.printStackTrace(w);
612      else
613         w.append(e.getFullStackMessage(true));
614      w.flush();
615      w.close();
616   }

619    * Callback method for logging errors during HTTP requests.
620    * <p>
621    *    Typically, subclasses will override this method and log errors themselves.
622    * <p>
623    * </p>
624    *    The default implementation simply logs errors to the <code>RestServlet</code> logger.
625    * </p>
626    * <p>
627    *    Here's a typical implementation showing how stack trace hashing can be used to reduce log file sizes...
628    * </p>
629    * <p class='bcode'>

866      if (! staticFilesCache.containsKey(pathInfo)) {
867         String p = RestUtils.decode(RestUtils.trimSlashes(pathInfo));
868         if (p.indexOf("..") != -1)
869            throw new RestException(SC_NOT_FOUND, "Invalid path");
870         for (Map.Entry<String,String> e : staticFilesMap.entrySet()) {
871            String key = RestUtils.trimSlashes(e.getKey());
872            if (p.startsWith(key)) {
873               String remainder = (p.equals(key) ? "" : p.substring(key.length()));
874               if (remainder.isEmpty() || remainder.startsWith("/")) {
875                  String p2 = RestUtils.trimSlashes(e.getValue()) + remainder;
876                  InputStream is = getResource(p2);
877                  if (is != null) {
878                     try {
879                        int i = p2.lastIndexOf('/');

881                        String mediaType = getMimetypesFileTypeMap().getContentType(name);
882                        staticFilesCache.put(pathInfo, new StreamResource(is, mediaType).setHeader("Cache-Control", "max-age=86400, public"));
883                        return staticFilesCache.get(pathInfo);
884                     } finally {
