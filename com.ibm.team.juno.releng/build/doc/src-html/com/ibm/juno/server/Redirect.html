009package com.ibm.juno.server;
010
011import java.net.*;
012import java.text.*;
013
014import com.ibm.juno.core.urlencoding.*;
015
016/**
017 * REST methods can return this object as a shortcut for performing <code>HTTP 302</code> redirects.
018 * <p>
019 * The following example shows the difference between handling redirects via the {@link RestRequest}/{@link RestResponse},
020 *    and the simplified approach of using this class.
021 * <p class='bcode'>
022 *    <jc>// Redirect to "/contextPath/servletPath/foobar"</jc>
023 *
024 *    <jc>// Using RestRequest and RestResponse</jc>
025 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/example1"</js>)
026 *    <jk>public void</jk> example1(RestRequest req, RestResponse res) <jk>throws</jk> IOException {
027 *       res.sendRedirect(req.getServletURI() + <js>"/foobar"</js>);
028 *    }
029 *
030 *    <jc>// Using Redirect</jc>
031 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/example2"</js>)
032 *    <jk>public</jk> Redirect example2() {
033 *       <jk>return new</jk> Redirect(<js>"foobar"</js>);
034 *    }
035 * </p>
036 * <p>
037 * The constructor can use a {@link MessageFormat}-style pattern with multiple arguments:
038 * <p class='bcode'>
039 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/example3"</js>)
040 *    <jk>public</jk> Redirect example3() {
041 *       <jk>return new</jk> Redirect(<js>"foo/{0}/bar/{1}"</js>, id1, id2);
042 *    }
043 * </p>
044 * <p>
045 * The arguments are serialized to strings using the servlet's {@link UrlEncodingSerializer},
046 *    so any filters defined on the serializer or REST method/class will be used when present.
047 * The arguments will also be automatically URL-encoded.
048 * <p>
049 * Redirecting to the servlet root can be accomplished by simply using the no-arg constructor.
050 * <p class='bcode'>
051 *    <jc>// Simply redirect to the servlet root.
052 *    // Equivalent to res.sendRedirect(req.getServletURI()).</jc>
053 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/example4"</js>)
054 *    <jk>public</jk> Redirect exmaple4() {
055 *       <jk>return new</jk> Redirect();
056 *    }
057 * </p>
058 * <p>
059 * This class is handled by {@link com.ibm.juno.server.response.RedirectHandler}, a built-in default
060 *    response handler created by {@link RestServlet#createResponseHandlers()}.
061 */
062public final class Redirect {
063
064   private int httpResponseCode;
065   private String url;
066   private Object[] args;
067
068   /**
069    * Redirect to the specified URL.
070    * Relative paths are interpreted as relative to the servlet path.
071    *
072    * @param url The URL to redirect to.
073    * @param args Optional {@link MessageFormat} arguments to replace in the URL string.
074    */
075   public Redirect(CharSequence url, Object...args) {
076      this.url = (url == null ? null : url.toString());
077      this.args = args;
078   }
079
080   /**
081    * Convenience method for redirecting to instance of {@link URL} and {@link URI}.
082    * Same as calling <code>toString()</code> on the object and using the other constructor.
083    *
084    * @param url The URL to redirect to.
085    */
086   public Redirect(Object url) {
087      this.url = (url == null ? null : url.toString());
088   }
089
090   /**
091    * Redirect to the specified URL.
092    * Relative paths are interpreted as relative to the servlet path.
093    *
094    * @param httpResponseCode The HTTP response code.
095    * @param url The URL to redirect to.
096    * @param args Optional {@link MessageFormat} arguments to replace in the URL string.
097    */
098   public Redirect(int httpResponseCode, CharSequence url, Object...args) {
099      this.httpResponseCode = httpResponseCode;
100      this.url = (url == null ? null : url.toString());
101      this.args = args;
102   }
103
104   /**
105    * Shortcut for redirecting to the servlet root.
106    */
107   public Redirect() {
108   }
109
110   /**
111    * Calculates the URL to redirect to.
112    *
113    * @param s Use this serializer to encode arguments using the {@link UrlEncodingSerializer#serializeUrlPart(Object)} method.
114    * @return The URL to redirect to.
115    */
116   public String toUrl(UrlEncodingSerializer s) {
117      if (url != null && args != null && args.length > 0) {
118         for (int i = 0; i < args.length; i++)
119            args[i] = s.serializeUrlPart(args[i]);
120         url = MessageFormat.format(url, args);
121      }
122      return url;
123   }
124
125   /**
126    * Returns the response code passed in through the constructor.
127    *
128    * @return The response code passed in through the constructor, or <code>0</code> if response code wasn't specified.
129    */
130   public int getHttpResponseCode() {
131      return httpResponseCode;
132   }
133}
