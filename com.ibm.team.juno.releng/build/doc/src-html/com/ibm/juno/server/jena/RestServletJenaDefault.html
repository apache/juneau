246   /**
247    * [OPTIONS /*] - Show resource options.
248    *
249    * @param req The HTTP request.
250    * @return The bean containing the contents of the OPTIONS page.
251    *
252    */
253   @RestMethod(name="OPTIONS", path="/*",
254      properties={
255         @Property(name=HTMLDOC_links, value="{back:'$R{servletURI}'}"),
256         @Property(name=HTMLDOC_description, value="Resource options")
257      },
258      description="Resource options"
259   )
260   public ResourceOptions getOptions(RestRequest req) {
261      return new ResourceOptions(this, req);
262   }
263
264   @Override /* RestServlet */
265   public boolean hasOptionsPage() {
266      return true;
267   }
