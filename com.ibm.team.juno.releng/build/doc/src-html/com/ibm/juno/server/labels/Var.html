009package com.ibm.juno.server.labels;
010
011import com.ibm.juno.core.annotation.*;
012import com.ibm.juno.core.utils.*;
013
014/**
015 * A request or response variable.
016 */
017@Bean(properties={"category","name","description"})
018public class Var implements Comparable<Var> {
019
020   /** Variable category (e.g. <js>"header"</js>, <js>"content"</js>) */
021   public String category;
022
023   /** Variable name (e.g. <js>"Content-Type"</js>) */
024   public String name;
025
026   /** Variable description */
027   public String description;
028
029   /** Bean constructor */
030   public Var() {}
031
032   /**
033    *
034    * Constructor.
035    * @param category Variable category (e.g. "ATTR", "PARAM").
036    * @param name Variable name.
037    */
038   public Var(String category, String name) {
039      this.category = category;
040      this.name = name;
041   }
042
043   @SuppressWarnings("hiding")
044   boolean matches(String category, String name) {
045      if (this.category.equals(category))
046         if (name == null || name.equals(this.name))
047            return true;
048      return false;
049   }
050
051   /**
052    * Sets the description for this variable.
053    *
054    * @param description The new description.
055    * @return This object (for method chaining).
056    */
057   public Var setDescription(String description) {
058      this.description = description;
059      return this;
060   }
061
062   @Override /* Comparable */
063   public int compareTo(Var o) {
064      int i = category.compareTo(o.category);
065      if (i == 0)
066         i = (name == null ? -1 : name.compareTo(o.name));
067      return i;
068   }
069
070   @Override /* Object */
071   public boolean equals(Object o) {
072      if (o instanceof Var) {
073         Var v = (Var)o;
074         return (v.category.equals(category) && StringUtils.isEquals(name, v.name));
075      }
076      return false;
077   }
078
079   @Override /* Object */
080   public int hashCode() {
081      return category.hashCode() + (name == null ? 0 : name.hashCode());
082   }
083}
