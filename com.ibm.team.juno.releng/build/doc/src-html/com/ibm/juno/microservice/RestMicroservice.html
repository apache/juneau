038/**
039 * Entry point for Juno microservice that implements a REST interface using Jetty on a single port.
040 * 
041 * <h6 class='topic'>Jetty Server Details</h6>
042 * <p>
043 * The Jetty server is created by the {@link #createServer()} method and started with the {@link #startServer()} method.
044 * These methods can be overridden to provided customized behavior.
045 * <p>
046 * 
047 * <h6 class='topic'>Defining REST Resources</h6>
048 * <p>
049 * Top-level REST resources are defined by the {@link #getResourceMap()} method.
050 * This method can be overridden to provide a customized list of REST resources.
051 * <p>
052 * 
053 * <h6 class='topic'>Logging</h6>
054 * <p>
055 * Logging is initialized by the {@link #initLogging()} method.
056 * This method can be overridden to provide customized logging behavior.
057 *
058 * <h6 class='topic'>Lifecycle Listener Methods</h6>
059 * Subclasses can optionally implement the following event listener methods:
060 * <ul>
061 *    <li>{@link #onStart()} - Gets executed before {@link #start()}.
062 *    <li>{@link #onStop()} - Gets executed before {@link #stop()}.
063 *    <li>{@link #onCreateServer()} - Gets executed before {@link #createServer()}.
064 *    <li>{@link #onStartServer()} - Gets executed before {@link #startServer()}.
065 *    <li>{@link #onPostStartServer()} - Gets executed after {@link #startServer()}.
066 *    <li>{@link #onStopServer()} - Gets executed before {@link #stop()}.
067 *    <li>{@link #onPostStopServer()} - Gets executed after {@link #stop()}.
068 * </ul>
069 *
070 * @author jbognar@us.ibm.com
071 */

111   @Override /* Microservice */
112   public void stop() {
113      Thread t = new Thread() {
114         @Override /* Thread */
115         public void run() {
116            try {
117               onStopServer();
118               logger.warning("Stopping server.");
119               System.out.println();
120               server.stop();
121               logger.warning("Server stopped.");
122               onPostStopServer();
123            } catch (Exception e) {
124               logger.log(Level.SEVERE, e.getLocalizedMessage(), e);
125            }
126         }
127      };
128      t.start();
129      try {
130         t.join();
131      } catch (InterruptedException e) {
132         e.printStackTrace();
133      }
134      super.stop();
135   }

141   /**
142    * Initialize the logging for this microservice.
143    * <p>
144    * Subclasses can override this method to provide customized logging.
145    * <p>
146    * The default implementation uses the <cs>Logging</cs> section in the config file to set up logging:
147    * <p class='bcode'>
148    *    <cc>#================================================================================
149    *    # Logger settings
150    *    # See FileHandler Java class for details.
151    *    #================================================================================</cc>
152    *    <cs>[Logging]</cs>
153    *    
154    *    <cc># The directory where to create the log file.
155    *    # Default is ".".</cc>
156    *    <ck>logDir</ck> = logs
157    *    
158    *    <cc># The name of the log file to create for the main logger.
159    *    # The logDir and logFile make up the pattern that's passed to the FileHandler
160    *    # constructor.
161    *    # If value is not specified, then logging to a file will not be set up.</cc>
162    *    <ck>logFile</ck> = microservice.%g.log
163    *    
164    *    <cc># Whether to append to the existing log file or create a new one.
165    *    # Default is false.</cc>
166    *    <ck>append</ck> =
167    *    
168    *    <cc># The SimpleDateFormat format to use for dates.
169    *    # Default is "yyyy.MM.dd hh:mm:ss".</cc>
170    *    <ck>dateFormat</ck> = 
171    *    
172    *    <cc># The log message format.
173    *    # The value can contain any of the following variables:
174    *    #  {date} - The date, formatted per dateFormat.
175    *    #  {class} - The class name.
176    *    #  {method} - The method name.
177    *    #  {logger} - The logger name.
178    *    #  {level} - The log level name.
179    *    #  {msg} - The log message.
180    *    #  {threadid} - The thread ID.
181    *    #  {exception} - The localized exception message.
182    *    # Default is "[{date} {level}] {msg}%n".</cc>
183    *    <ck>format</ck> = 
184    *    
185    *    <cc># The maximum log file size.
186    *    # Suffixes available for numbers.
187    *    # See ConfigFile.getInt(String,int) for details.
188    *    # Default is 1M.</cc>
189    *    <ck>limit</ck> = 10M
190    *    
191    *    <cc># Max number of log files.
192    *    # Default is 1.</cc>
193    *    <ck>count</ck> = 5
194    * 
195    *    <cc># Default log levels.
196    *    # Keys are logger names.
197    *    # Values are serialized Level POJOs.</cc>
198    *    <ck>levels</ck> = { com.ibm.juno:'INFO' }
199    *    
200    *    <cc># Only print unique stack traces once and then refer to them by a simple 8 character hash identifier.
201    *    # Useful for preventing log files from filling up with duplicate stack traces.
202    *    # Default is false.</cc>
203    *    <ck>useStackTraceHashes</ck> = true
204    *    
205    *    <cc># The default level for the console logger.
206    *    # Default is WARNING.</cc>
207    *    <ck>consoleLevel</ck> = WARNING
208    * </p>
209    * 
210    * @throws Exception 
211    */
212   protected void initLogging() throws Exception {
213      ConfigFile cf = getConfig();
214      logger = Logger.getLogger("");
215      String logFile = cf.getString("Logging/logFile");
216      if (! StringUtils.isEmpty(logFile)) {
217         LogManager.getLogManager().reset();
218         String logDir = cf.getString("Logging/logDir", ".");
219         FileUtils.mkdirs(new File(logDir), false);
220         boolean append = cf.getBoolean("Logging/append");
221         int limit = cf.getInt("Logging/limit", 1024*1024);
222         int count = cf.getInt("Logging/count", 1);
223         FileHandler fh = new FileHandler(logDir + '/' + logFile, limit, count, append);
224
225         boolean useStackTraceHashes = cf.getBoolean("Logging/useStackTraceHashes");
226         String format = cf.getString("Logging/format", "[{date} {level}] {msg}%n");
227         String dateFormat = cf.getString("Logging/dateFormat", "yyyy.MM.dd hh:mm:ss");
228         fh.setFormatter(new LogEntryFormatter(format, dateFormat, useStackTraceHashes));
229         logger.addHandler(fh);
230
231         ConsoleHandler ch = new ConsoleHandler();
232         ch.setLevel(Level.parse(cf.getString("Logging/consoleLevel", "WARNING")));
233         ch.setFormatter(new LogEntryFormatter(format, dateFormat, false));
234         logger.addHandler(ch);
235      }
236      ObjectMap loggerLevels = cf.getObject(ObjectMap.class, "Logging/levels");
237      if (loggerLevels != null)
238      for (String l : loggerLevels.keySet())
239         Logger.getLogger(l).setLevel(loggerLevels.get(Level.class, l));
240   }

242   /**
243    * Method used to create (but not start) an instance of a Jetty server.
244    * <p>
245    * Subclasses can override this method to customize the Jetty server before it is started.
246    * <p>
247    * The default implementation is configured by the following values in the config file:
248    * <p>
249    * <p class='bcode'>
250    *    <cc>#================================================================================
251    *    # REST settings
252    *    #================================================================================</cc>
253    *    <cs>[REST]</cs>
254    *    
255    *    <cc># The HTTP port number to use.
256    *    # Default is Rest-Port setting in manifest file, or 8000.</cc>
257    *    <ck>port</ck> = 10000
258    *    
259    *    <cc># The context root of the Jetty server.
260    *    # Default is Rest-ContextPath in manifest file, or "/".</cc>
261    *    <ck>contextPath</ck> = 10000
262    * 
263    *    <cc># Authentication:  NONE, BASIC.
264    *    # Default is Rest-AuthType in manifest file, or NONE.</cc>
265    *    <ck>authType</ck> = NONE
266    * 
267    *    <cc># The BASIC auth username.
268    *    # Default is Rest-LoginUser in manifest file.</cc>
269    *    <ck>loginUser</ck> = 
270    *    
271    *    <cc># The BASIC auth password.
272    *    # Default is Rest-LoginPassword in manifest file.</cc>
273    *    <ck>loginPassword</ck> = 
274    *    
275    *    <cc># The BASIC auth realm.
276    *    # Default is Rest-AuthRealm in manifest file.</cc>
277    *    <ck>authRealm</ck> = 
278    *    
279    *    <cc># Enable SSL support.</cc>
280    *    <ck>useSsl</ck> = false
281    * 
282    *    <cc>#================================================================================
283    *    # Bean properties on the org.eclipse.jetty.util.ssl.SslSocketFactory class
284    *    #--------------------------------------------------------------------------------
285    *    # Ignored if REST/useSsl is false.
286    *    #================================================================================</cc>
287    *    <cs>[REST-SslContextFactory]</cs>
288    *    <ck>keyStorePath</ck> = client_keystore.jks
289    *    <ck>keyStorePassword*</ck> = {HRAaRQoT}
290    *    <ck>excludeCipherSuites</ck> = TLS_DHE.*, TLS_EDH.*
291    *    <ck>excludeProtocols</ck> = SSLv3
292    *    <ck>allowRenegotiate</ck> = false
293    * </p>
294    * 
295    * @return The newly-created server. 
296    * @throws Exception 
297    */
298   protected Server createServer() throws Exception {
299      onCreateServer();
300
301      ConfigFile cf = getConfig();
302      ObjectMap mf = getManifest();
303      
304      port = cf.getInt("REST/port", mf.getInt("Rest-Port", 8000));
305      String contextPath = cf.getString("REST/contextPath", mf.getString("Rest-ContextPath", "/"));
306
307      if (cf.getBoolean("REST/useSsl")) {
308
309         SslContextFactory sslContextFactory = new SslContextFactory();
310
311         // Write the properties in REST-SslContextFactory to the bean setters on sslContextFactory.
312         // Throws an exception if section contains unknown properties.
313         // Only look for bean properties of type String/String/boolean/int since class has multiple
314         //    setters with the same name (e.g. setKeyStore(KeyStore) and setKeyStore(String)).
315         ObjectMap m = cf.writeProperties("REST-SslContextFactory", sslContextFactory, false, String.class, String[].class, boolean.class, int.class);
316
317         // We're using Jetty 8 that doesn't allow regular expression matching in SslContextFactory.setExcludeCipherSuites(), 
318         // so to prevent having the config file list all old cipher suites, exclude the known bad ones.
319         String[] excludeCipherSuites = ArrayUtils.combine(
320            StringUtils.split("SSL_RSA_WITH_DES_CBC_SHA,SSL_DHE_RSA_WITH_DES_CBC_SHA,SSL_DHE_DSS_WITH_DES_CBC_SHA,SSL_RSA_EXPORT_WITH_RC4_40_MD5,SSL_RSA_EXPORT_WITH_DES40_CBC_SHA,SSL_DHE_RSA_EXPORT_WITH_DES40_CBC_SHA,SSL_DHE_RSA_WITH_3DES_EDE_CBC_SHA,SSL_DHE_DSS_WITH_3DES_EDE_CBC_SHA,TLS_DHE_RSA_WITH_AES_256_CBC_SHA256,TLS_DHE_DSS_WITH_AES_256_CBC_SHA256,TLS_DHE_RSA_WITH_AES_256_CBC_SHA,TLS_DHE_DSS_WITH_AES_256_CBC_SHA,TLS_DHE_RSA_WITH_AES_128_CBC_SHA256,TLS_DHE_DSS_WITH_AES_128_CBC_SHA256,TLS_DHE_RSA_WITH_AES_128_CBC_SHA,TLS_DHE_DSS_WITH_AES_128_CBC_SHA", ','),
321            sslContextFactory.getExcludeCipherSuites()
322         );
323         sslContextFactory.setExcludeCipherSuites(excludeCipherSuites);
324         
325         logger.log(Level.WARNING, "SSL properties set: {0}", JsonSerializer.DEFAULT_LAX.toString(m));
326
327         SslSocketConnector connector = new SslSocketConnector(sslContextFactory);
328         connector.setPort(port);
329
330         server = new Server();
331         server.setConnectors(new Connector[] { connector });
332
333      } else {
334         server = new Server(port);
335      }
336
337      ServletContextHandler context = new ServletContextHandler(ServletContextHandler.SESSIONS);
338
339      String authType = cf.getString("REST/authType", mf.getString("Rest-AuthType", "NONE"));
340      if (authType.equals("BASIC"))
341         context.setSecurityHandler(basicAuth(cf, mf));
342
343      context.setContextPath(contextPath);
344      server.setHandler(context);
345
346      for (Map.Entry<String,Class<? extends Servlet>> e : getResourceMap().entrySet())
347         context.addServlet(e.getValue(), e.getKey()).setInitOrder(0);
348
349      return server;
350   }

367   /**
368    * Returns the resource map to use for this microservice.
369    * <p>
370    * <p>
371    * Subclasses can override this method to programmatically specify their resources.
372    * <p>
373    * The default implementation is configured by the following values in the config file:
374    * <p>
375    * <p class='bcode'>
376    * 
377    *    <cc>#================================================================================
378    *    # REST settings
379    *    #================================================================================</cc>
380    *    <cs>[REST]</cs>
381    *    
382    *    <cc># A JSON map of servlet paths to servlet classes.
383    *    # Example:  
384    *    #  resourceMap = {'/*':'com.ibm.MyServlet'}
385    *    # Either resourceMap or resources must be specified if it's not defined in 
386    *    # the manifest file.</cc>
387    *    <ck>resourceMap</ck> = 
388    * 
389    *    <cc># A comma-delimited list of names of classes that extend from Servlet.
390    *    # Resource paths are pulled from @RestResource.path() annotation, or
391    *    #  "/*" if annotation not specified.
392    *    # Example:  
393    *    #  resources = com.ibm.MyServlet
394    *     *    # Default is Rest-Resources in manifest file.
395    *    # Either resourceMap or resources must be specified if it's not defined in 
396    *    # the manifest file.</cc>
397    *    <ck>resources</ck> =
398    * </p>
399    * <p>
400    *    In most cases, the rest resources will be specified in the manifest file since
401    *    it's not likely to be a configurable property:
402    * <p>
403    * <p class='bcode'>
404    *    <mk>Rest-Resources:</mk> com.ibm.juno.microservice.sample.RootResources
405    * </p> 
406    * 
407    * @return The map of REST resources. 
408    * @throws ClassNotFoundException 
409    * @throws ParseException 
410    */
411   @SuppressWarnings("unchecked")
412   protected Map<String,Class<? extends Servlet>> getResourceMap() throws ClassNotFoundException, ParseException {
413      ConfigFile cf = getConfig();
414      ObjectMap mf = getManifest();
415      Map<String,Class<? extends Servlet>> rm = new LinkedHashMap<String,Class<? extends Servlet>>();
416
417      ObjectMap resourceMap = cf.getObject(ObjectMap.class, "REST/resourceMap");
418      String[] resources = cf.getStringArray("REST/resources", mf.getStringArray("Rest-Resources"));
419
420      if (resourceMap != null && ! resourceMap.isEmpty()) {
421         for (Map.Entry<String,Object> e : resourceMap.entrySet()) {
422            Class<?> c = Class.forName(e.getValue().toString());
423            if (! ClassUtils.isParentClass(Servlet.class, c))
424               throw new ClassNotFoundException("Invalid class specified as resource.  Must be a Servlet.  Class='"+c.getName()+"'");
425            rm.put(e.getKey(), (Class<? extends Servlet>)c);
426         }
427      } else if (resources.length > 0) {
428         for (String resource : resources) {
429            Class<?> c = Class.forName(resource);
430            if (! ClassUtils.isParentClass(Servlet.class, c))
431               throw new ClassNotFoundException("Invalid class specified as resource.  Must be a Servlet.  Class='"+c.getName()+"'");
432            RestResource rr = c.getAnnotation(RestResource.class);
433            String path = rr == null ? "/*" : rr.path();
434            if (! path.endsWith("*"))
435               path += (path.endsWith("/") ? "*" : "/*");
436            rm.put(path, (Class<? extends Servlet>)c);
437         }
438      }
439      return rm;
440   }

442   /**
443    * Called when {@link ConfigFile#save()} is called on the config file.
444    * <p>
445    * The default behavior is configured by the following value in the config file:
446    * <p>
447    * <p class='bcode'>
448    *    <cs>[REST]</cs>
449    *    
450    *    <cc># What to do when the config file is saved.
451    *    # Possible values:
452    *    #  NOTHING - Don't do anything. (default)
453    *    #  RESTART_SERVER - Restart the Jetty server.
454    *    #  RESTART_SERVICE - Shutdown and exit with code '3'.</cc>
455    *    <ck>saveConfigAction</ck> = RESTART_SERVER
456    * </p>
457    */
458   @Override /* Microservice */
459   protected void onConfigSave(ConfigFile cf) {
460      try {
461         String saveConfigAction = cf.getString("REST/saveConfigAction", "NOTHING");
462         if (saveConfigAction.equals("RESTART_SERVER")) {
463            new Thread() {
464               @Override /* Thread */
465               public void run() {
466                  try {
467                     RestMicroservice.this.stop();
468                     RestMicroservice.this.start();
469                  } catch (Exception e) {
470                     logger.log(Level.SEVERE, e.getLocalizedMessage(), e);
471                  }
472               }
473            }.start();
474         } else if (saveConfigAction.equals("RESTART_SERVICE")) {
475            stop();
476            System.exit(3);
477         }
478      } catch (Exception e) {
479         throw new RuntimeException(e);
480      }
481   }

526   private static final SecurityHandler basicAuth(ConfigFile cf, ObjectMap mf) {
527
528      HashLoginService l = new HashLoginService();
529      String user = cf.getString("REST/loginUser", mf.getString("Rest-LoginUser"));
530      String pw = cf.getString("REST/loginPassword", mf.getString("Rest-LoginPassword"));
531      String realm = cf.getString("REST/authRealm", mf.getString("Rest-AuthRealm", ""));
532      String ctx = cf.getString("REST/contextPath", mf.getString("Rest-ContextPath", "/"));
533
534      l.putUser(user, Credential.getCredential(pw), new String[] { "user" });
535      l.setName(realm);
536
537      Constraint constraint = new Constraint();
538      constraint.setName(Constraint.__BASIC_AUTH);
539      constraint.setRoles(new String[] { "user" });
540      constraint.setAuthenticate(true);
541
542      ConstraintMapping cm = new ConstraintMapping();
543      cm.setConstraint(constraint);
544      cm.setPathSpec(ctx);
545
546      ConstraintSecurityHandler csh = new ConstraintSecurityHandler();
547      csh.setAuthenticator(new BasicAuthenticator());
548      csh.setRealmName("myrealm");
549      csh.addConstraintMapping(cm);
550      csh.setLoginService(l);
551
552      return csh;
553   }
