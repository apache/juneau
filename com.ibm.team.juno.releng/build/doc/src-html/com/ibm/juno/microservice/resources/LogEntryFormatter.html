026/**
027 * Log entry formatter.
028 * <p>
029 *    Uses three simple parameter for configuring log entry formats:
030 *    <ul>
031 *       <li><code>dateFormat</code> - A {@link SimpleDateFormat} string describing the format for dates.
032 *       <li><code>format</code> - A string with <code>{...}</code> replacement variables representing predefined fields.
033 *       <li><code>useStackTraceHashes</code> - A setting that causes duplicate stack traces to be replaced with 8-character hash strings.
034 *    </ul>
035 * <p>
036 *    This class converts the format strings into a regular expression that can be used to parse the resulting log file.
037 *
038 * @author jbognar
039 */
040public class LogEntryFormatter extends Formatter {
041
042   private ConcurrentHashMap<String,AtomicInteger> hashes;
043   private DateFormat df;
044   private String format;
045   private Pattern rePattern;
046   private Map<String,Integer> fieldIndexes;
047
048   /**
049    * Create a new formatter.
050    * 
051    * @param format The log entry format.  e.g. <js>"[{date} {level}] {msg}%n"</js>
052    *    The string can contain any of the following variables:
053    *       <ol>
054    *          <li><js>"{date}"</js> - The date, formatted per <js>"Logging/dateFormat"</js>.
055    *          <li><js>"{class}"</js> - The class name.
056    *          <li><js>"{method}"</js> - The method name.
057    *          <li><js>"{logger}"</js> - The logger name.
058    *          <li><js>"{level}"</js> - The log level name.
059    *          <li><js>"{msg}"</js> - The log message.
060    *          <li><js>"{threadid}"</js> - The thread ID.
061    *          <li><js>"{exception}"</js> - The localized exception message.
062    *    </ol>
063    * @param dateFormat The {@link SimpleDateFormat} format to use for dates.  e.g. <js>"yyyy.MM.dd hh:mm:ss"</js>.
064    * @param useStackTraceHashes If <jk>true</jk>, only print unique stack traces once and then refer to them by a
065    *    simple 8 character hash identifier.
066    */
067   public LogEntryFormatter(String format, String dateFormat, boolean useStackTraceHashes) {
068      this.df = new SimpleDateFormat(dateFormat);
069      if (useStackTraceHashes)
070         hashes = new ConcurrentHashMap<String,AtomicInteger>();
071
072      fieldIndexes = new HashMap<String,Integer>();
073
074      format = format
075         .replaceAll("\\{date\\}", "%1\\$s")
076         .replaceAll("\\{class\\}", "%2\\$s")
077         .replaceAll("\\{method\\}", "%3\\$s")
078         .replaceAll("\\{logger\\}", "%4\\$s")
079         .replaceAll("\\{level\\}", "%5\\$s")
080         .replaceAll("\\{msg\\}", "%6\\$s")
081         .replaceAll("\\{threadid\\}", "%7\\$s")
082         .replaceAll("\\{exception\\}", "%8\\$s");
083
084      this.format = format;
085
086      // Construct a regular expression to match this log entry.
087      int index = 1;
088      StringBuilder re = new StringBuilder();
089      int S1 = 1; // Looking for %
090      int S2 = 2; // Found %, looking for number.
091      int S3 = 3; // Found number, looking for $.
092      int S4 = 4; // Found $, looking for s.
093      int state = 1;
094      int i1 = 0;
095      for (int i = 0; i < format.length(); i++) {
096         char c = format.charAt(i);
097         if (state == S1) {
098            if (c == '%')
099               state = S2;
100            else {
101               if (! (Character.isLetterOrDigit(c) || Character.isWhitespace(c)))
102                  re.append('\\');
103               re.append(c);
104            }
105         } else if (state == S2) {
106            if (Character.isDigit(c)) {
107               i1 = i;
108               state = S3;
109            } else {
110               re.append("\\%").append(c);
111               state = S1;
112            }
113         } else if (state == S3) {
114            if (c == '$') {
115               state = S4;
116            } else {
117               re.append("\\%").append(format.substring(i1, i));
118               state = S1;
119            }
120         } else if (state == S4) {
121            if (c == 's') {
122               int group = Integer.parseInt(format.substring(i1, i-1));
123               switch (group) {
124                  case 1:
125                     fieldIndexes.put("date", index++);
126                     re.append("(" + dateFormat.replaceAll("[mHhsSdMy]", "\\\\d").replaceAll("\\.", "\\\\.") + ")");
127                     break;
128                  case 2:
129                     fieldIndexes.put("class", index++);
130                     re.append("([\\p{javaJavaIdentifierPart}\\.]+)");
131                     break;
132                  case 3:
133                     fieldIndexes.put("method", index++);
134                     re.append("([\\p{javaJavaIdentifierPart}\\.]+)");
135                     break;
136                  case 4:
137                     fieldIndexes.put("logger", index++);
138                     re.append("([\\w\\d\\.\\_]+)");
139                     break;
140                  case 5:
141                     fieldIndexes.put("level", index++);
142                     re.append("(SEVERE|WARNING|INFO|CONFIG|FINE|FINER|FINEST)");
143                     break;
144                  case 6:
145                     fieldIndexes.put("msg", index++);
146                     re.append("(.*)");
147                     break;
148                  case 7:
149                     fieldIndexes.put("threadid", index++);
150                     re.append("(\\\\d+)");
151                     break;
152                  case 8:
153                     fieldIndexes.put("exception", index++);
154                     re.append("(.*)");
155                     break;
156               }
157            } else {
158               re.append("\\%").append(format.substring(i1, i));
159            }
160            state = S1;
161         }
162      }
163
164      // The log parser
165      String sre = re.toString();
166      if (sre.endsWith("\\%n"))
167         sre = sre.substring(0, sre.length()-3);
168
169      // Replace instances of %n.
170      sre = sre.replaceAll("\\\\%n", "\\\\n");
171
172      rePattern = Pattern.compile(sre);
173      fieldIndexes = Collections.unmodifiableMap(fieldIndexes);
174   }
175
176   /**
177    * Returns the regular expression pattern used for matching log entries.
178    * 
179    * @return The regular expression pattern used for matching log entries.
180    */
181   public Pattern getLogEntryPattern() {
182      return rePattern;
183   }
184
185   /**
186    * Returns the {@link DateFormat} used for matching dates.
187    * 
188    * @return The {@link DateFormat} used for matching dates.
189    */
190   public DateFormat getDateFormat() {
191      return df;
192   }
193
194   /**
195    * Given a matcher that has matched the pattern specified by {@link #getLogEntryPattern()},
196    * returns the field value from the match.
197    * 
198    * @param fieldName The field name.  Possible values are:
199    *    <ul>
200    *       <li><js>"date"</js>
201    *       <li><js>"class"</js>
202    *       <li><js>"method"</js>
203    *       <li><js>"logger"</js>
204    *       <li><js>"level"</js>
205    *       <li><js>"msg"</js>
206    *       <li><js>"threadid"</js>
207    *       <li><js>"exception"</js>
208    * </ul>
209    * @param m The matcher. 
210    * @return The field value, or <jk>null</jk> if the specified field does not exist.
211    */
212   public String getField(String fieldName, Matcher m) {
213      Integer i = fieldIndexes.get(fieldName);
214      return (i == null ? null : m.group(i));
215   }
216
217   @Override /* Formatter */
218   public String format(LogRecord r) {
219      String msg = formatMessage(r);
220      Throwable t = r.getThrown();
221      String hash = null;
222      int c = 0;
223      if (hashes != null && t != null) {
224         hash = hashCode(t);
225         hashes.putIfAbsent(hash, new AtomicInteger(0));
226         c = hashes.get(hash).incrementAndGet();
227         if (c == 1) {
228            msg = '[' + hash + '.' + c + "] " + msg;
229         } else {
230            msg = '[' + hash + '.' + c + "] " + msg + ", " + t.getLocalizedMessage();
231            t = null;
232         }
233      }
234      String s = String.format(format,
235         df.format(new Date(r.getMillis())),
236         r.getSourceClassName(),
237         r.getSourceMethodName(),
238         r.getLoggerName(),
239         r.getLevel(),
240         msg,
241         r.getThreadID(),
242         r.getThrown() == null ? "" : r.getThrown().getMessage());
243      if (t != null)
244         s += String.format("%n%s", StringUtils.getStackTrace(r.getThrown()));
245      return s;
246   }
247
248   private String hashCode(Throwable t) {
249      int i = 0;
250      while (t != null) {
251         for (StackTraceElement e : t.getStackTrace())
252            i ^= e.hashCode();
253         t = t.getCause();
254      }
255      return Integer.toHexString(i);
256   }
