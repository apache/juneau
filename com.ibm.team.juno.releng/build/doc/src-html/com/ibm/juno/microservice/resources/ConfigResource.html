009package com.ibm.juno.microservice.resources;
010
011import static com.ibm.juno.core.html.HtmlDocSerializerProperties.*;
012import static javax.servlet.http.HttpServletResponse.*;
013import static com.ibm.juno.server.annotation.VarCategory.*;
014
015import java.io.Reader;
016import java.io.StringReader;
017import java.util.Map;
018
019import com.ibm.juno.core.ObjectMap;
020import com.ibm.juno.core.ini.*;
021import com.ibm.juno.microservice.Resource;
022import com.ibm.juno.server.*;
023import com.ibm.juno.server.annotation.*;
024
025/**
026 * Shows contents of the microservice configuration file.
027 */
028@RestResource(
029   path="/config",
030   label="Configuration",
031   description="Contents of configuration file.",
032   properties={
033      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'$R{servletURI}?method=OPTIONS',edit:'$R{servletURI}/edit'}"),
034   }
035)
036public class ConfigResource extends Resource {
037   private static final long serialVersionUID = 1L;
038
039   /** 
040    * [GET /] - Show contents of config file.
041    *  
042    * @return The config file.  
043    * @throws Exception 
044    */
045   @RestMethod(name="GET", path="/", description="Show contents of config file.")
046   public ConfigFile getConfigContents() throws Exception {
047      return getConfig();
048   }
049
050   /** 
051    * [GET /edit] - Show config file edit page. 
052    * 
053    * @param req The HTTP request.
054    * @return The config file as a reader resource.
055    * @throws Exception 
056    */
057   @RestMethod(name="GET", path="/edit", description="Show config file edit page.")
058   public ReaderResource getConfigEditPage(RestRequest req) throws Exception {
059      // Note that we don't want variables in the config file to be resolved,
060      // so we need to escape any $ characters we see.
061      req.setAttribute("contents", getConfig().toString().replaceAll("\\$", "\\\\\\$"));
062      return req.getReaderResource("ConfigEdit.html", true);
063   }
064
065   /** 
066    * [GET /{section}] - Show config file section.
067    *  
068    * @param section The section name. 
069    * @return The config file section.
070    * @throws Exception 
071    */
072   @RestMethod(name="GET", path="/{section}",
073      description="Show config file section.",
074      input={
075         @Var(category=ATTR, name="section", description="Section name.")
076      }
077   )
078   public ObjectMap getConfigSection(@Attr("section") String section) throws Exception {
079      return getSection(section);
080   }
081
082   /** 
083    * [GET /{section}/{key}] - Show config file entry. 
084    * 
085    * @param section The section name. 
086    * @param key The section key.
087    * @return The value of the config file entry.
088    * @throws Exception 
089    */
090   @RestMethod(name="GET", path="/{section}/{key}",
091      description="Show config file entry.",
092      input={
093         @Var(category=ATTR, name="section", description="Section name."),
094         @Var(category=ATTR, name="key", description="Entry name.")
095      }
096   )
097   public String getConfigEntry(@Attr("section") String section, @Attr("key") String key) throws Exception {
098      return getSection(section).getString(key);
099   }
100
101   /** 
102    * [POST /] - Sets contents of config file from a FORM post. 
103    * 
104    * @param contents The new contents of the config file.
105    * @return The new config file contents.
106    * @throws Exception 
107    */
108   @RestMethod(name="POST", path="/",
109      description="Sets contents of config file from a FORM post.",
110      input={
111         @Var(category=PARAM, name="contents", description="New contents in INI file format.")
112      }
113   )
114   public ConfigFile setConfigContentsFormPost(@Param("contents") String contents) throws Exception {
115      return setConfigContents(new StringReader(contents));
116   }
117
118   /** 
119    * [PUT /] - Sets contents of config file. 
120    * 
121    * @param contents The new contents of the config file. 
122    * @return The new config file contents.
123    * @throws Exception 
124    */
125   @RestMethod(name="PUT", path="/",
126      description="Sets contents of config file.",
127      input={
128         @Var(category=CONTENT, description="New contents in INI file format.")
129      }
130   )
131   public ConfigFile setConfigContents(@Content Reader contents) throws Exception {
132      ConfigFile cf2 = ConfigMgr.DEFAULT.create().load(contents);
133      return getConfig().merge(cf2).save();
134   }
135
136   /** 
137    * [PUT /{section}] - Add or overwrite a config file section. 
138    * 
139    * @param section The section name. 
140    * @param contents The new contents of the config file section.
141    * @return The new section.
142    * @throws Exception 
143    */
144   @RestMethod(name="PUT", path="/{section}",
145      description="Add or overwrite a config file section.",
146      input={
147         @Var(category=ATTR, name="section", description="Section name."),
148         @Var(category=CONTENT, description="New contents for section as a simple map with string keys and values.")
149      }
150   )
151   public ObjectMap setConfigSection(@Attr("section") String section, @Content Map<String,String> contents) throws Exception {
152      getConfig().setSection(section, contents);
153      return getSection(section);
154   }
155
156   /** 
157    * [PUT /{section}/{key}] - Add or overwrite a config file entry. 
158    * 
159    * @param section The section name. 
160    * @param key The section key.
161    * @param value The new value.
162    * @return The new value.
163    * @throws Exception 
164    */
165   @RestMethod(name="PUT", path="/{section}/{key}",
166      description="Add or overwrite a config file entry.",
167      input={
168         @Var(category=ATTR, name="section", description="Section name."),
169         @Var(category=ATTR, name="key", description="Entry name."),
170         @Var(category=CONTENT, description="New value as a string.")
171      }
172   )
173   public String setConfigSection(@Attr("section") String section, @Attr("key") String key, @Content String value) throws Exception {
174      getConfig().put(section, key, value, false);
175      return getSection(section).getString(key);
176   }
177
178   private ObjectMap getSection(String name) {
179      ObjectMap m = getConfig().getSectionMap(name);
180      if (m == null)
181         throw new RestException(SC_NOT_FOUND, "Section not found.");
182      return m;
183   }
184}
