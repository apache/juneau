033/**
034 * REST resource that allows access to a file system directory.
035 * <p>
036 *    The root directory is specified in one of two ways:
037 * </p>
038 * <ul>
039 *    <li>Specifying the location via a <l>DirectoryResource.rootDir</l> property.
040 *    <li>Overriding the {@link #getRootDir()} method.
041 * </ul>
042 * <p>
043 *    Read/write access control is handled through the following properties:
044 * </p>
045 * <ul>
046 *    <li><l>DirectoryResource.allowViews</l> - If <jk>true</jk>, allows view and download access to files.
047 *    <li><l>DirectoryResource.allowPuts</l> - If <jk>true</jk>, allows files to be created or overwritten.
048 *    <li><l>DirectoryResource.allowDeletes</l> - If <jk>true</jk>, allows files to be deleted.
049 * </ul>
050 * <p>
051 *    Access can also be controlled by overriding the {@link #checkAccess(RestRequest)} method.
052 * </p>
053 */
054@RestResource(
055   label="File System Explorer",
056   description="Contents of $A{path}",
057   messages="nls/DirectoryResource",
058   properties={
059      @Property(name=HTML_uriAnchorText, value=PROPERTY_NAME),
060      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(com.ibm.juno.server.samples.DirectoryResource)'}"),
061      @Property(name=REST_allowMethodParam, value="*"),
062      @Property(name="DirectoryResource.rootDir", value=""),
063      @Property(name="DirectoryResource.allowViews", value="false"),
064      @Property(name="DirectoryResource.allowDeletes", value="false"),
065      @Property(name="DirectoryResource.allowPuts", value="false")
066   }
067)
068public class DirectoryResource extends Resource {
069   private static final long serialVersionUID = 1L;
070
071   private File rootDir;     // The root directory
072
073   // Settings enabled through servlet init parameters
074   private boolean allowDeletes, allowPuts, allowViews;
075
076   private static Logger logger = Logger.getLogger(DirectoryResource.class.getName());
077
078   @Override /* Servlet */
079   public void init() throws ServletException {
080      ObjectMap p = getProperties();
081      rootDir = new File(p.getString("DirectoryResource.rootDir"));
082      allowViews = p.getBoolean("DirectoryResource.allowViews", false);
083      allowDeletes = p.getBoolean("DirectoryResource.allowDeletes", false);
084      allowPuts = p.getBoolean("DirectoryResource.allowPuts", false);
085   }
086
087   /** 
088    * Returns the root directory defined by the 'rootDir' init parameter.
089    * Subclasses can override this method to provide their own root directory. 
090    * @return The root directory. 
091    */
092   protected File getRootDir() {
093      if (rootDir == null) {
094         rootDir = new File(getProperties().getString("rootDir"));
095         if (! rootDir.exists())
096            if (! rootDir.mkdirs())
097               throw new RuntimeException("Could not create root dir");
098      }
099      return rootDir;
100   }
101
102   /** 
103    * [GET /*]
104    *  On directories, returns a directory listing.
105    *  On files, returns information about the file.
106    *  
107    * @param req - The HTTP request.
108    * @return Either a FileResource or list of FileResources depending on whether it's a 
109    *    file or directory.
110    * @throws Exception - If file could not be read or access was not granted.
111    */
112   @RestMethod(name="GET", path="/*", 
113      description="On directories, returns a directory listing.\nOn files, returns information about the file.",
114      converters={Queryable.class}
115   )
116   public Object doGet(RestRequest req) throws Exception {
117      checkAccess(req);
118
119      String pathInfo = req.getPathInfo();
120      File f = pathInfo == null ? rootDir : new File(rootDir.getAbsolutePath() + pathInfo);
121
122      if (!f.exists())
123         throw new RestException(SC_NOT_FOUND, "File not found");
124
125      req.setAttribute("path", f.getAbsolutePath());
126
127      if (f.isDirectory()) {
128         List<FileResource> l = new LinkedList<FileResource>();
129         for (File fc : f.listFiles()) {
130            URL fUrl = new URL(req.getRequestURL().append("/").append(fc.getName()).toString());
131            l.add(new FileResource(fc, fUrl));
132         }
133         return l;
134      }
135
136      return new FileResource(f, new URL(req.getRequestURL().toString()));
137   }
138
139   /** 
140    * [DELETE /*]
141    *  Delete a file on the file system.
142    *  
143    * @param req - The HTTP request.
144    * @return The message <js>"File deleted"</js> if successful.
145    * @throws Exception - If file could not be read or access was not granted.
146    */
147   @RestMethod(name="DELETE", path="/*",
148      description="Delete a file on the file system."
149   )
150   public Object doDelete(RestRequest req) throws Exception {
151      checkAccess(req);
152
153      File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
154      deleteFile(f);
155
156      if (req.getHeader("Accept").contains("text/html"))
157         return new Redirect();
158      return "File deleted";
159   }
160
161   /** 
162    * [PUT /*]
163    * Add or overwrite a file on the file system.
164    *  
165    * @param req - The HTTP request.
166    * @return The message <js>"File added"</js> if successful.
167    * @throws Exception - If file could not be read or access was not granted.
168    */
169   @RestMethod(name="PUT", path="/*", 
170      description="Add or overwrite a file on the file system."
171   )
172   public Object doPut(RestRequest req) throws Exception {
173      checkAccess(req);
174
175      File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
176      String parentSubPath = f.getParentFile().getAbsolutePath().substring(rootDir.getAbsolutePath().length());
177      BufferedOutputStream bos = new BufferedOutputStream(new FileOutputStream(f));
178      IOPipe.create(req.getInputStream(), bos).closeOut().run();
179      if (req.getContentType().contains("html"))
180         return new Redirect(parentSubPath);
181      return "File added";
182   }
183
184   /** 
185    * [VIEW /*]
186    *    View the contents of a file.
187    *    Applies to files only.
188    *  
189    * @param req - The HTTP request.
190    * @param res - The HTTP response.
191    * @return A Reader containing the contents of the file.
192    * @throws Exception - If file could not be read or access was not granted.
193    */
194   @RestMethod(name="VIEW", path="/*",
195      description="View the contents of a file.\nApplies to files only."
196   )
197   public Reader doView(RestRequest req, RestResponse res) throws Exception {
198      checkAccess(req);
199
200      File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
201
202      if (!f.exists())
203         throw new RestException(SC_NOT_FOUND, "File not found");
204
205      if (f.isDirectory())
206         throw new RestException(SC_METHOD_NOT_ALLOWED, "VIEW not available on directories");
207
208      res.setContentType("text/plain");
209      return new FileReader(f);
210   }
211   
212   /** 
213    * [DOWNLOAD /*]
214    *    Download the contents of a file.
215    *    Applies to files only.
216    *  
217    * @param req - The HTTP request.
218    * @param res - The HTTP response.
219    * @return A Reader containing the contents of the file.
220    * @throws Exception - If file could not be read or access was not granted.
221    */
222   @RestMethod(name="DOWNLOAD", path="/*",
223      description="Download the contents of a file.\nApplies to files only."
224   )
225   public Reader doDownload(RestRequest req, RestResponse res) throws Exception {
226      checkAccess(req);
227
228      File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
229
230      if (!f.exists())
231         throw new RestException(SC_NOT_FOUND, "File not found");
232
233      if (f.isDirectory())
234         throw new RestException(SC_METHOD_NOT_ALLOWED, "DOWNLOAD not available on directories");
235
236      res.setContentType("application");
237      return new FileReader(f);
238   }
239   
240   /**
241    * Verify that the specified request is allowed.
242    * Subclasses can override this method to provide customized behavior.
243    * Method should throw a {@link RestException} if the request should be disallowed.
244    * 
245    * @param req The HTTP request.
246    */
247   protected void checkAccess(RestRequest req) {
248      String method = req.getMethod();
249      if (method.equals("VIEW") && ! allowViews)
250         throw new RestException(SC_METHOD_NOT_ALLOWED, "VIEW not enabled");
251      if (method.equals("PUT") && ! allowPuts)
252         throw new RestException(SC_METHOD_NOT_ALLOWED, "PUT not enabled");
253      if (method.equals("DELETE") && ! allowDeletes)
254         throw new RestException(SC_METHOD_NOT_ALLOWED, "DELETE not enabled");
255      if (method.equals("DOWNLOAD") && ! allowViews)
256         throw new RestException(SC_METHOD_NOT_ALLOWED, "DOWNLOAD not enabled");
257   }
258
259   /** File POJO */
260   public class FileResource {
261      private File f;
262      private URL url;
263
264      /** 
265       * Constructor.
266       * @param f - The file.
267       * @param url - The URL of the file resource. 
268       */
269      public FileResource(File f, URL url) {
270         this.f = f;
271         this.url = url;
272      }
273
274      // Bean property getters
275
276      /**
277       * @return The URL of the file resource.
278       */
279      public URL getUrl() {
280         return url;
281      }
282
283      /**
284       * @return The file type.
285       */
286      public String getType() {
287         return (f.isDirectory() ? "dir" : "file");
288      }
289
290      /**
291       * @return The file name.
292       */
293      public String getName() {
294         return f.getName();
295      }
296
297      /**
298       * @return The file size.
299       */
300      public long getSize() {
301         return f.length();
302      }
303
304      /**
305       * @return The file last modified timestamp.
306       */
307      @BeanProperty(filter=DateFilter.ISO8601DTP.class)
308      public Date getLastModified() {
309         return new Date(f.lastModified());
310      }
311
312      /**
313       * @return A hyperlink to view the contents of the file.
314       * @throws Exception If access is not allowed.
315       */
316      public URL getView() throws Exception {
317         if (allowViews && f.canRead() && ! f.isDirectory())
318            return new URL(url + "?method=VIEW");
319         return null;
320      }
321
322      /**
323       * @return A hyperlink to download the contents of the file.
324       * @throws Exception If access is not allowed.
325       */
326      public URL getDownload() throws Exception {
327         if (allowViews && f.canRead() && ! f.isDirectory())
328            return new URL(url + "?method=DOWNLOAD");
329         return null;
330      }
331
332      /**
333       * @return A hyperlink to delete the file.
334       * @throws Exception If access is not allowed.
335       */
336      public URL getDelete() throws Exception {
337         if (allowDeletes && f.canWrite())
338            return new URL(url + "?method=DELETE");
339         return null;
340      }
341   }
342
343   /** Utility method */
344   private void deleteFile(File f) {
345      try {
346         if (f.isDirectory())
347            for (File fc : f.listFiles())
348               deleteFile(fc);
349         f.delete();
350      } catch (Exception e) {
351         logger.log(WARNING, "Cannot delete file '" + f.getAbsolutePath() + "'", e);
352      }
353   }
