033/**
034 * REST resource for viewing and accessing log files.
035 */
036@RestResource(
037   path="/logs",
038   label="Log files",
039   description="Log files from this service",
040   properties={
041      @Property(name=HTML_uriAnchorText, value=PROPERTY_NAME),
042      @Property(name=REST_allowMethodParam, value="true")
043   },
044   filters={
045      IteratorFilter.class,       // Allows Iterators and Iterables to be serialized.
046      DateFilter.ISO8601DT.class  // Serialize Date objects as ISO8601 strings.
047   }
048)
049@SuppressWarnings("nls")
050public class LogsResource extends Resource {
051   private static final long serialVersionUID = 1L;
052
053   private ConfigFile cf = getConfig();
054
055   private File logDir = new File(cf.getString("Logging/logDir", "."));
056   private LogEntryFormatter leFormatter = new LogEntryFormatter(
057      cf.getString("Logging/format", "[{date} {level}] {msg}%n"),
058      cf.getString("Logging/dateFormat", "yyyy.MM.dd hh:mm:ss"),
059      cf.getBoolean("Logging/useStackTraceHashes")
060   );
061
062   private final FileFilter filter = new FileFilter() {
063      @Override /* FileFilter */
064      public boolean accept(File f) {
065         return f.isDirectory() || f.getName().endsWith(".log");
066      }
067   };

069   /** 
070    * [GET /*] - Get file details or directory listing. 
071    * 
072    * @param req The HTTP request
073    * @param properties The writable properties for setting the descriptions.
074    * @param path The log file path.
075    * @return The log file.
076    * @throws Exception 
077    */
078   @RestMethod(name="GET", path="/*", rc={200,404})
079   public Object getFileOrDirectory(RestRequest req, @Properties ObjectMap properties, @PathRemainder String path) throws Exception {
080
081      File f = getFile(path);
082
083      if (f.isDirectory()) {
084         Set<FileResource> l = new TreeSet<FileResource>(new FileResourceComparator());
085         for (File fc : f.listFiles(filter)) {
086            URL fUrl = new URL(req.getTrimmedRequestURL().append('/').append(fc.getName()).toString());
087            l.add(new FileResource(fc, fUrl));
088         }
089         properties.put(HTMLDOC_description, "Contents of " + f.getAbsolutePath());
090         return l;
091      }
092
093      properties.put(HTMLDOC_description, "File details on " + f.getAbsolutePath());
094      return new FileResource(f, new URL(req.getTrimmedRequestURL().toString()));
095   }

097   /**
098    * [VIEW /*] - Retrieve the contents of a log file.
099    * 
100    * @param req The HTTP request.
101    * @param res The HTTP response.
102    * @param path The log file path.
103    * @param properties The writable properties for setting the descriptions.
104    * @param highlight If <code>true</code>, add color highlighting based on severity.
105    * @param start Optional start timestamp.  Don't print lines logged before the specified timestamp.  Example:  "&start=2014-01-23 11:25:47".
106    * @param end Optional end timestamp.  Don't print lines logged after the specified timestamp.  Example:  "&end=2014-01-23 11:25:47".
107    * @param thread Optional thread name filter.  Only show log entries with the specified thread name.  Example: "&thread=pool-33-thread-1".
108    * @param loggers Optional logger filter.  Only show log entries if they were produced by one of the specified loggers (simple class name).  Example: "&loggers=(LinkIndexService,LinkIndexRestService)".
109    * @param severity Optional severity filter.  Only show log entries with the specified severity.  Example: "&severity=(ERROR,WARN)".
110    * @throws Exception 
111    */
112   @RestMethod(name="VIEW", path="/*", rc={200,404})
113   @SuppressWarnings("nls")
114   public void viewFile(RestRequest req, RestResponse res, @PathRemainder String path, @Properties ObjectMap properties, @Param("highlight") boolean highlight, @Param("start") String start, @Param("end") String end, @Param("thread") String thread, @Param("loggers") String[] loggers, @Param("severity") String[] severity) throws Exception {
115
116      File f = getFile(path);
117      if (f.isDirectory())
118         throw new RestException(SC_METHOD_NOT_ALLOWED, "View not available on directories");
119
120      Date startDate = StringUtils.parseISO8601Date(start), endDate = StringUtils.parseISO8601Date(end);
121
122      if (! highlight) {
123         Object o = getReader(f, startDate, endDate, thread, loggers, severity);
124         res.setContentType("text/plain");
125         if (o instanceof Reader)
126            res.setOutput(o);
127         else {
128            LogParser p = (LogParser)o;
129            Writer w = res.getNegotiatedWriter();
130            try {
131               p.writeTo(w);
132            } finally {
133               w.flush();
134               w.close();
135            }
136         }
137         return;
138      }
139
140      res.setContentType("text/html");
141      PrintWriter w = res.getNegotiatedWriter();
142      try {
143         w.println("<html><body style='font-family:monospace;font-size:8pt;white-space:pre;'>");
144         LogParser lp = getLogParser(f, startDate, endDate, thread, loggers, severity);
145         try {
146            if (! lp.hasNext())
147               w.append("<span style='color:gray'>[EMPTY]</span>");
148            else for (LogParser.Entry le : lp) {
149               char s = le.severity.charAt(0);
150               String color = "black";
151               //SEVERE|WARNING|INFO|CONFIG|FINE|FINER|FINEST
152               if (s == 'I')
153                  color = "#006400";
154               else if (s == 'W')
155                  color = "#CC8400";
156               else if (s == 'E' || s == 'S')
157                  color = "#DD0000";
158               else if (s == 'D' || s == 'F' || s == 'T')
159                  color = "#000064";
160               w.append("<span style='color:").append(color).append("'>");
161               le.appendHtml(w).append("</span>");
162            }
163            w.append("</body></html>");
164         } finally {
165            lp.close();
166         }
167      } finally {
168         w.close();
169      }
170   }

172   /**
173    * [VIEW /*] - Retrieve the contents of a log file as parsed entries.
174    * 
175    * @param req The HTTP request. 
176    * @param path The log file path.
177    * @param start Optional start timestamp.  Don't print lines logged before the specified timestamp.  Example:  "&start=2014-01-23 11:25:47".
178    * @param end Optional end timestamp.  Don't print lines logged after the specified timestamp.  Example:  "&end=2014-01-23 11:25:47".
179    * @param thread Optional thread name filter.  Only show log entries with the specified thread name.  Example: "&thread=pool-33-thread-1".
180    * @param loggers Optional logger filter.  Only show log entries if they were produced by one of the specified loggers (simple class name).  Example: "&loggers=(LinkIndexService,LinkIndexRestService)".
181    * @param severity Optional severity filter.  Only show log entries with the specified severity.  Example: "&severity=(ERROR,WARN)".
182    * @return The parsed contents of the log file.
183    * @throws Exception 
184    */
185   @RestMethod(name="PARSE", path="/*", converters=Queryable.class, rc={200,404})
186   public LogParser viewParsedEntries(RestRequest req, @PathRemainder String path, @Param("start") String start, @Param("end") String end, @Param("thread") String thread, @Param("loggers") String[] loggers, @Param("severity") String[] severity) throws Exception {
187
188      File f = getFile(path);
189      Date startDate = StringUtils.parseISO8601Date(start), endDate = StringUtils.parseISO8601Date(end);
190
191      if (f.isDirectory())
192         throw new RestException(SC_METHOD_NOT_ALLOWED, "View not available on directories");
193
194      return getLogParser(f, startDate, endDate, thread, loggers, severity);
195   }

197   /**
198    * [DOWNLOAD /*] - Download file.
199    * 
200    * @param res The HTTP response. 
201    * @param path The log file path. 
202    * @return The contents of the log file.
203    * @throws Exception 
204    */
205   @RestMethod(name="DOWNLOAD", path="/*", rc={200,404})
206   public Object downloadFile(RestResponse res, @PathRemainder String path) throws Exception {
207
208      File f = getFile(path);
209
210      if (f.isDirectory())
211         throw new RestException(SC_METHOD_NOT_ALLOWED, "Download not available on directories");
212
213      res.setContentType("application/octet-stream"); //$NON-NLS-1$
214      res.setContentLength((int)f.length());
215      return new FileInputStream(f);
216   }

218   /** 
219    * [DELETE /*] - Delete a file. 
220    * 
221    * @param path The log file path. 
222    * @return A redirect object to the root.
223    * @throws Exception 
224    */
225   @RestMethod(name="DELETE", path="/*", rc={200,404})
226   public Object deleteFile(@PathRemainder String path) throws Exception {
227
228      File f = getFile(path);
229
230      if (f.isDirectory())
231         throw new RestException(SC_BAD_REQUEST, "Delete not available on directories.");
232
233      if (f.canWrite())
234         if (! f.delete())
235            throw new RestException(SC_FORBIDDEN, "Could not delete file.");
236
237      return new Redirect(path + "/.."); //$NON-NLS-1$
238   }
239
240   private static BufferedReader getReader(File f) throws IOException {
241      return new BufferedReader(new InputStreamReader(new FileInputStream(f), Charset.defaultCharset()));
242   }
243
244   private File getFile(String path) {
245      if (path != null && path.indexOf("..") != -1)
246         throw new RestException(SC_NOT_FOUND, "File not found.");
247      File f = (path == null ? logDir : new File(logDir.getAbsolutePath() + '/' + path));
248      if (filter.accept(f))
249         return f;
250      throw new RestException(SC_NOT_FOUND, "File not found.");
251   }

253   /**
254    * File bean.
255    */
256   @SuppressWarnings("javadoc")
257   public static class FileResource {
258      private File f;
259      public String type;
260      public Object name;
261      public Long size;
262      @BeanProperty(filter=DateFilter.Medium.class) public Date lastModified;
263      public URL view, highlighted, parsed, download, delete;
264
265      public FileResource(File f, URL url) throws IOException {
266         this.f = f;
267         this.type = (f.isDirectory() ? "dir" : "file");
268         this.name = f.isDirectory() ? new Link(f.getName(), url.toString()) : f.getName();
269         this.size = f.isDirectory() ? null : f.length();
270         this.lastModified = new Date(f.lastModified());
271         if (f.canRead() && ! f.isDirectory()) {
272            this.view = new URL(url + "?method=VIEW");
273            this.highlighted = new URL(url + "?method=VIEW&highlight=true");
274            this.parsed = new URL(url + "?method=PARSE");
275            this.download = new URL(url + "?method=DOWNLOAD");
276            this.delete = new URL(url + "?method=DELETE");
277         }
278      }
279   }
280
281   private static class FileResourceComparator implements Comparator<FileResource>, Serializable {
282      private static final long serialVersionUID = 1L;
283      @Override /* Comparator */
284      public int compare(FileResource o1, FileResource o2) {
285         int c = o1.type.compareTo(o2.type);
286         return c != 0 ? c : o1.f.getName().compareTo(o2.f.getName());
287      }
288   }
289
290   private Object getReader(File f, final Date start, final Date end, final String thread, final String[] loggers, final String[] severity) throws IOException {
291      if (start == null && end == null && thread == null && loggers == null)
292         return getReader(f);
293      return getLogParser(f, start, end, thread, loggers, severity);
294   }
295
296   private LogParser getLogParser(File f, final Date start, final Date end, final String thread, final String[] loggers, final String[] severity) throws IOException {
297      return new LogParser(leFormatter, f, start, end, thread, loggers, severity);
298   }
