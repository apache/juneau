027/**
028 * Parent class for all microservices.
029 * <p>
030 * A microservice defines a simple API for starting and stopping simple Java services
031 *    contained in executable jars.
032 * <p>
033 * The general command for invoking these services is...
034 * <p class='bcode'>
035 *    java -jar mymicroservice.jar [mymicroservice.cfg]
036 * </p>
037 * <p>
038 * Your microservice class must be specified as the <jk>Main-Class</jk> entry in
039 *    the manifest file of your microservice jar file.
040 *
041 * <h6 class='topic'>Microservice Configuration</h6>
042 * 
043 * This class defines the following method for accessing configuration for your microservice:
044 * <p>
045 * <ul>
046 *    <li>{@link #getArgs()} - The command-line arguments passed to the jar file.
047 *    <li>{@link #getConfig()} - An external INI-style configuration file.
048 *    <li>{@link #getManifest()} - The manifest file for the main jar file.
049 * </ul>
050 *
051 * <h6 class='topic'>Entrypoint Method</h6>
052 * 
053 * Subclasses must implement a static void main method as the entry point for the microservice.
054 * Typically, this method will simply consist of the following...
055 * <p>
056 * <p class='bcode'>
057 *    <jk>public static void</jk> main(String[] args) <jk>throws</jk> Exception {
058 *       <jk>new</jk> MyMicroservice(args).start();
059 *    }
060 * </p>
061 *
062 * <h6 class='topic'>Lifecycle Methods</h6>
063 * 
064 * Subclasses must implement the following lifecycle methods:
065 * <p>
066 * <ul>
067 *    <li>{@link #start()} - Gets executed during startup.
068 *    <li>{@link #stop()} - Gets executed when 'exit' is typed in the console or an external shutdown signal is received.
069 *    <li>{@link #kill()} - Can be used to forcibly shut down the service.  Doesn't get called during normal operation.
070 * </ul>
071 *
072 * <h6 class='topic'>Lifecycle Listener Methods</h6>
073 * 
074 * Subclasses can optionally implement the following event listener methods:
075 * <p>
076 * <ul>
077 *    <li>{@link #onStart()} - Gets executed before {@link #start()}.
078 *    <li>{@link #onStop()} - Gets executed before {@link #stop()}.
079 *    <li>{@link #onConfigSave(ConfigFile)} - Gets executed after a config file has been saved.
080 *    <li>{@link #onConfigChange(ConfigFile, Set)} - Gets executed after a config file has been modified.
081 * </ul>
082 *
083 * <h6 class='topic'>Other Methods</h6>
084 * 
085 * Subclasses can optionally override the following methods to provide customized behavior:
086 * <p>
087 * <ul>
088 *    <li>{@link #createVarResolver()} - Creates the {@link StringVarResolver} used to resolve variables in the config file returned by {@link #getConfig()}.
089 * </ul>
090 * 
091 * @author jbognar@us.ibm.com
092 */
093public abstract class Microservice {
094
095   private static Args args;
096   private static ConfigFile cf;
097   private static ObjectMap mf;

099   /**
100    * Constructor.
101    * 
102    * @param args2 Command line arguments.
103    * @throws Exception 
104    */
105   protected Microservice(String[] args2) throws Exception {
106      Microservice.args = new Args(args2);
107
108      // --------------------------------------------------------------------------------
109      // Try to get the manifest file.
110      // --------------------------------------------------------------------------------
111      Manifest m = new Manifest();
112
113      // If running within an eclipse workspace, need to get it from the file system.
114      File f = new File("META-INF/MANIFEST.MF");
115      if (f.exists()) {
116         try {
117            m.read(new FileInputStream(f));
118         } catch (IOException e) {
119            System.err.println("Problem detected in MANIFEST.MF.  Contents below:\n" + IOUtils.read(f));
120            throw e;
121         }
122      } else {
123         // Otherwise, read from manifest file in the jar file containing the main class.
124         URLClassLoader cl = (URLClassLoader)getClass().getClassLoader();
125         URL url = cl.findResource("META-INF/MANIFEST.MF");
126         if (url != null) {
127            try {
128               m.read(url.openStream());
129            } catch (IOException e) {
130               System.err.println("Problem detected in MANIFEST.MF.  Contents below:\n" + IOUtils.read(url.openStream()));
131               throw e;
132            }
133         }
134      }
135      mf = new ObjectMap();
136      for (Map.Entry<Object,Object> e : m.getMainAttributes().entrySet())
137         mf.put(e.getKey().toString(), e.getValue().toString());
138      
139      // --------------------------------------------------------------------------------
140      // Find config file.
141      // Can either be passed in as first parameter, or we discover it using
142      // the 'sun.java.command' system property.
143      // --------------------------------------------------------------------------------
144      String cFile = null;
145      if (args.hasArg(0))
146         cFile = args.getArg(0);
147      else if (mf.containsKey("Main-ConfigFile"))
148         cFile = mf.getString("Main-ConfigFile");
149      else {
150         String cmd = System.getProperty("sun.java.command", "not_found").split("\\s+")[0];
151         if (cmd.endsWith(".jar"))
152            cFile = cmd.replace(".jar", ".cfg");
153      }
154
155      if (cFile == null) {
156         System.err.println("Running class ["+getClass().getSimpleName()+"] without a config file.");
157         cf = ConfigMgr.DEFAULT.create();
158      } else {
159         System.out.println("Running class ["+getClass().getSimpleName()+"] using config file ["+cFile+"]");
160         System.setProperty("juno.configFile", cFile);
161         cf = ConfigMgr.DEFAULT.get(cFile).getResolving(createVarResolver());
162      }
163
164      // --------------------------------------------------------------------------------
165      // Set system properties.
166      // --------------------------------------------------------------------------------
167      Set<String> spKeys = cf.getSectionKeys("SystemProperties");
168      if (spKeys != null)
169         for (String key : spKeys) 
170            System.setProperty(key, cf.get("SystemProperties", key));
171      
172      // --------------------------------------------------------------------------------
173      // Add a config file change listener.
174      // --------------------------------------------------------------------------------
175      cf.addListener(new ConfigFileListener() {
176         @Override /* ConfigFileListener */
177         public void onSave(ConfigFile cf) {
178            onConfigSave(cf);
179         }
180         @Override /* ConfigFileListener */
181         public void onChange(ConfigFile cf, Set<String> changes) {
182            onConfigChange(cf, changes);
183         }
184      });
185
186      // --------------------------------------------------------------------------------
187      // Add exit listeners.
188      // --------------------------------------------------------------------------------
189      new Thread() {
190         @Override /* Thread */
191         public void run() {
192            Console c = System.console();
193            if (c == null)
194               System.out.println("No available console.");
195            else {
196               while (true) {
197                  String l = c.readLine("\nEnter 'exit' to exit.\n");
198                  if (l == null || l.equals("exit")) {
199                     Microservice.this.stop();
200                     break;
201                  }
202               }
203            }
204         }
205      }.start();
206      Runtime.getRuntime().addShutdownHook(
207         new Thread() {
208            @Override /* Thread */
209            public void run() {
210               Microservice.this.stop();
211            }
212         }
213      );
214   }

216   /**
217    * Creates the {@link StringVarResolver} used to resolve variables in the 
218    * config file returned by {@link #getConfig()}.
219    * <p>
220    * The default implementation resolves the following variables:
221    * <ul>
222    *    <li><code>$S{key}</code>, <code>$S{key,default}</code> - System properties.
223    *    <li><code>$E{key}</code>, <code>$E{key,default}</code> - Environment variables.
224    *    <li><code>$C{key}</code>, <code>$C{key,default}</code> - Config file entries.
225    *    <li><code>$MF{key}</code>, <code>$MF{key,default}</code> - Manifest file entries.
226    *    <li><code>$ARG{key}</code>, <code>$ARG{key,default}</code> - Command-line arguments.
227    * </ul> 
228    * <p>
229    * Subclasses can override this method to provide their own variables.
230    * <dl>
231    *    <dt>Examples:</dt>
232    *    <dd>
233    *       <p class='bcode'>
234    *    <jd>/** 
235    *     * Augment default var resolver with a custom $B{...} variable that simply wraps strings inside square brackets.
236    *     * /</jd>
237    *    <ja>@Override</ja> <jc>// Microservice</jc>
238    *    <jk>protected</jk> StringVarResolver createVarResolver() {
239    *       <jk>return super</jk>.createVarResolver()
240    *          .addVar(<js>"B"</js>,
241    *             <jk>new</jk> StringVarWithDefault() {
242    *                <ja>@Override</ja> <jc>// StringVar</jc> 
243    *                <jk>public</jk> String resolve(String varVal) {
244    *                   <jk>return</jk> <js>'['</js> + varVal + <js>']'</js>;
245    *                }
246    *             }
247    *          );
248    *    }
249    *       </p>
250    *       <p class='bcode'>
251    *    <cc># Example config file</cc>
252    *    <cs>[MySection]</cs>
253    *    <ck>myEntry</ck> = $B{foo}
254    *       </p>
255    *       <p class='bcode'>
256    *    <jc>// Example java code</jc>
257    *    String myentry = getConfig().getString(<js>"MySection/myEntry"</js>); <jc>// == "[foo]"</js>
258    *       </p>
259    *    </dd>
260    * </dl>
261    * 
262    * @return A new {@link StringVarResolver}.
263    */
264   protected StringVarResolver createVarResolver() {
265      return new StringVarResolver(StringVarResolver.DEFAULT)
266         .addVar("C", 
267            new StringVarWithDefault() {
268               @Override /* StringVar */
269               public String resolve(String varVal) {
270                  return cf.getString(varVal);
271               }
272            }
273         )
274         .addVar("MF", 
275            new StringVarWithDefault() {
276               @Override /* StringVar */
277               public String resolve(String varVal) {
278                  if (mf == null)
279                     return null;
280                  return mf.getString(varVal);
281               }
282            }
283         )
284         .addVar("ARG", 
285            new StringVarWithDefault() {
286               @Override /* StringVar */
287               public String resolve(String varVal) {
288                  if (args == null)
289                     return null;
290                  return args.getString(varVal);
291               }
292            }
293         );
294   }

320   /**
321    * Returns the external INI-style configuration file that can be used to configure your microservice.
322    * <p>
323    * The config file location is determined in the following order:
324    * <ol class='spaced-list'>
325    *    <li>The first argument passed to the microservice jar.
326    *    <li>The <code>Main-ConfigFile</code> entry in the microservice jar manifest file.
327    *    <li>The name of the microservice jar with a <js>".cfg"</js> suffix (e.g. <js>"mymicroservice.jar"</js>-&gt;<js>"mymicroservice.cfg"</js>).
328    * </ol>
329    * <p>
330    * If all methods for locating the config file fail, then this method returns <jk>null</jk>.
331    * <p>
332    * Subclasses can set their own config file by calling the {@link #setConfig(ConfigFile)} method.
333    * <p>
334    * String variables defined by {@link #createVarResolver()} are automatically resolved when using this method.
335    * <p>
336    * This method can be called from the class constructor.
337    * <dl>
338    *    <dt>Examples:</dt>
339    *    <dd>
340    *       <p class='bcode'>
341    *    <cc>#--------------------------</cc>
342    *    <cc># My section</cc>
343    *    <cc>#--------------------------</cc>
344    *    <cs>[MySection]</cs>
345    *    
346    *    <cc># An integer</cc>
347    *    <ck>anInt</ck> = 1 
348    *    
349    *    <cc># A boolean</cc>
350    *    <ck>aBoolean</ck> = true 
351    *    
352    *    <cc># An int array</cc>
353    *    <ck>anIntArray</ck> = 1,2,3 
354    *    
355    *    <cc># A POJO that can be converted from a String</cc>
356    *    <ck>aURL</ck> = http://foo 
357    *    
358    *    <cc># A POJO that can be converted from JSON</cc>
359    *    <ck>aBean</ck> = {foo:'bar',baz:123}
360    *    
361    *    <cc># A system property</cc>
362    *    <ck>locale</ck> = $S{java.locale, en_US}
363    *    
364    *    <cc># An environment variable</cc>
365    *    <ck>path</ck> = $E{PATH, unknown}
366    *    
367    *    <cc># A manifest file entry</cc>
368    *    <ck>mainClass</ck> = $MF{Main-Class}
369    *    
370    *    <cc># Another value in this config file</cc>
371    *    <ck>sameAsAnInt</ck> = $C{MySection/anInt}
372    *    
373    *    <cc># A command-line argument in the form "myarg=foo"</cc>
374    *    <ck>myArg</ck> = $ARG{myarg}
375    *    
376    *    <cc># The first command-line argument</cc>
377    *    <ck>firstArg</ck> = $ARG{0}
378    * 
379    *    <cc># Look for system property, or env var if that doesn't exist, or command-line arg if that doesn't exist.</cc>
380    *    <ck>nested</ck> = $S{mySystemProperty,$E{MY_ENV_VAR,$ARG{0}}}
381    * 
382    *    <cc># A POJO with embedded variables</cc>
383    *    <ck>aBean2</ck> = {foo:'$ARG{0}',baz:$C{MySection/anInt}}
384    *    
385    *       </p>
386    *       <p class='bcode'>
387    *    <jc>// Java code for accessing config entries above.</jc>
388    *    ConfigFile cf = getConfig();
389    *    
390    *    <jk>int</jk> anInt = cf.getInt(<js>"MySection/anInt"</js>); 
391    *    <jk>boolean</jk> aBoolean = cf.getBoolean(<js>"MySection/aBoolean"</js>); 
392    *    <jk>int</jk>[] anIntArray = cf.getObject(<jk>int</jk>[].<jk>class</jk>, <js>"MySection/anIntArray"</js>); 
393    *    URL aURL = cf.getObject(URL.<jk>class</jk>, <js>"MySection/aURL"</js>); 
394    *    MyBean aBean = cf.getObject(MyBean.<jk>class</jk>, <js>"MySection/aBean"</js>); 
395    *    Locale locale = cf.getObject(Locale.<jk>class</jk>, <js>"MySection/locale"</js>); 
396    *    String path = cf.getString(<js>"MySection/path"</js>); 
397    *    String mainClass = cf.getString(<js>"MySection/mainClass"</js>); 
398    *    <jk>int</jk> sameAsAnInt = cf.getInt(<js>"MySection/sameAsAnInt"</js>); 
399    *    String myArg = cf.getString(<js>"MySection/myArg"</js>); 
400    *    String firstArg = cf.getString(<js>"MySection/firstArg"</js>); 
401    *       </p>
402    *    </dd>
403    * </dl>
404    * 
405    * @return The config file for this application, or <jk>null</jk> if no config file is configured. 
406    */
407   protected static ConfigFile getConfig() {
408      return cf;
409   }

422   /**
423    * Returns the main jar manifest file contents as a simple {@link ObjectMap}.
424    * <p>
425    * This map consists of the contents of {@link Manifest#getMainAttributes()} with the keys
426    *    and entries converted to simple strings.
427    * <p>
428    * This method can be called from the class constructor.
429    * <dl>
430    *    <dt>Examples:</dt>
431    *    <dd>
432    *       <p class='bcode'>
433    *    <jc>// Get Main-Class from manifest file.</jc>
434    *    String mainClass = Microservice.<jsm>getManifest</jsm>().getString(<js>"Main-Class"</js>, <js>"unknown"</js>);
435    * 
436    *    <jc>// Get Rest-Resources from manifest file.</jc>
437    *    String[] restResources = Microservice.<jsm>getManifest</jsm>().getStringArray(<js>"Rest-Resources"</js>);
438    *       </p>
439    *    </dd>
440    * </dl>
441    * 
442    * @return The manifest file from the main jar, or <jk>null</jk> if the manifest file could not be retrieved.
443    */
444   protected static ObjectMap getManifest() {
445      return mf;
446   }

452   /**
453    * Start this application.
454    * <p>
455    * Default implementation simply calls {@link #onStart()}.
456    * <p>
457    * Overridden methods MUST call this method FIRST so that the {@link #onStart()} method is called.
458    * 
459    * @throws Exception
460    */
461   protected void start() throws Exception {
462      onStart();
463   }
464
465   /**
466    * Stop this application.
467    * <p>
468    * Default implementation simply calls {@link #onStop()}.
469    * <p>
470    * Overridden methods MUST call this method LAST so that the {@link #onStop()} method is called.
471    */
472   protected void stop() {
473      onStop();
474   }
475
476   /**
477    * Kill the JVM by calling <code>System.exit(2);</code>.
478    */
479   protected void kill() {
480      // This triggers the shutdown hook.
481      System.exit(2);
482   }
