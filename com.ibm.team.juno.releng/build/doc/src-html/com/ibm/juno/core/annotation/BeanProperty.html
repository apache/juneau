011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015import java.util.*;
016
017import com.ibm.juno.core.*;
018import com.ibm.juno.core.filter.*;
019import com.ibm.juno.core.jena.*;
020import com.ibm.juno.core.xml.*;
021import com.ibm.juno.core.xml.annotation.*;

023/**
024 * Used tailor how bean properties get interpreted by the framework.
025 * <p>
026 *    Can be used to do the following:
027 * <ul>
028 *    <li>Override the name of a property.
029 *    <li>Identify a getter or setter with a non-standard naming convention.
030 *    <li>Identify a specific subclass for a property with a general class type.
031 *    <li>Identify class types of elements in properties of type <code>Collection</code> or <code>Map</code>.
032 *    <li>Hide properties during serialization.
033 *    <li>Associate filters with bean property values, such as a filter to convert a <code>Calendar</code> field to a string.
034 *    <li>Override the list of properties during serialization on child elements of a property of type <code>Collection</code> or <code>Map</code>.
035 *    <li>Identify a property as the URL for a bean.
036 *    <li>Identify a property as the ID for a bean.
037 * </ul>
038 * <p>
039 *    This annotation is applied to public fields and public getter/setter methods of beans.
040 *
041 * @author James Bognar (jbognar@us.ibm.com)
042 */
043@Documented
044@Target({FIELD,METHOD})
045@Retention(RUNTIME)
046@Inherited
047public @interface BeanProperty {
048
049   /**
050    * Identifies the name of the property.
051    * <p>
052    *    Normally, this is automatically inferred from the field name or getter method name
053    *    of the property.  However, this property can be used to assign a different
054    *    property name from the automatically inferred value.
055    * <p>
056    *    If the {@link BeanContextProperties#BEAN_beanFieldVisibility} setting on the bean context excludes this field (e.g. the visibility
057    *    is set to PUBLIC, but the field is PROTECTED), this annotation can be used to force the field to be identified as a property.
058    */
059   String name() default "";
060
061   /**
062    * Identifies a specialized class type for the property.
063    * <p>
064    *    Normally this can be inferred through reflection of the field type or getter return type.
065    *    However, you'll want to specify this value if you're parsing beans where the bean property class
066    *    is an interface or abstract class to identify the bean type to instantiate.  Otherwise, you may
067    *    cause an {@link InstantiationException} when trying to set these fields.
068    * <p>
069    *    This property must denote a concrete bean class with a no-arg constructor.
070    *
071    * <dl>
072    *    <dt>Example:</dt>
073    *    <dd>
074    *       <p class='bcode'>
075    *    <jk>public class</jk> MyBean {
076    *
077    *       <jc>// Identify concrete map type.</jc>
078    *       <ja>@BeanProperty</ja>(type=HashMap.<jk>class</jk>)
079    *       <jk>public</jk> Map <jf>p1</jf>;
080    *    }
081    *       </p>
082    *    </dd>
083    * </dl>
084    */
085   Class<?> type() default Object.class;
086
087   /**
088    * For bean properties of maps and collections, this annotation can be used to identify
089    * the class types of the contents of the bean property object when the generic parameter
090    * types are interfaces or abstract classes.
091    *
092    * <dl>
093    *    <dt>Example:</dt>
094    *    <dd>
095    *       <p class='bcode'>
096    *    <jk>public class</jk> MyBean {
097    *
098    *       <jc>// Identify concrete map type with String keys and Integer values.</jc>
099    *       <ja>@BeanProperty</ja>(type=HashMap.<jk>class</jk>, params={String.<jk>class</jk>,Integer.<jk>class</jk>})
100    *       <jk>public</jk> Map <jf>p1</jf>;
101    *    }
102    *       </p>
103    *    </dd>
104    * </dl>
105    */
106   Class<?>[] params() default {};
107
108   /**
109    * Associates an object filter with this bean property that will convert it
110    * to a different value during serialization and parsing.
111    * <p>
112    * This annotation supersedes any filter associated with the bean property type
113    *    class itself.
114    * <p>
115    * Typically used for rendering {@link Date Dates} and {@link Calendar Calendars}
116    *    as a particular string format.
117    *
118    * <dl>
119    *    <dt>Example:</dt>
120    *    <dd>
121    *       <p class='bcode'>
122    *    <jk>public class</jk> MyClass {
123    *
124    *       <jc>// During serialization, convert to ISO8601 date-time string.</jc>
125    *       <ja>@BeanProperty</ja>(filter=CalendarFilter.ISO8601DT.<jk>class</jk>)
126    *       <jk>public</jk> Calendar getTime();
127    *    }
128    *       </p>
129    *    </dd>
130    * </dl>
131    */
132   Class<? extends PojoFilter<?,?>> filter() default PojoFilter.NULL.class;
133
134   /**
135    * Used to limit which child properties are rendered by the serializers.
136    * <p>
137    * Can be used on any of the following bean property types:
138    * <ul>
139    *    <li>Beans - Only render the specified properties of the bean.
140    *    <li>Maps - Only render the specified entries in the map.
141    *    <li>Bean/Map arrays - Same, but applied to each element in the array.
142    *    <li>Bean/Map collections - Same, but applied to each element in the collection.
143    * </ul>
144    *
145    * <dl>
146    *    <dt>Example:</dt>
147    *    <dd>
148    *       <p class='bcode'>
149    *    <jk>public class</jk> MyClass {
150    *
151    *       <jc>// Only render 'f1' when serializing this bean property.</jc>
152    *       <ja>@BeanProperty</ja>(properties={<js>"f1"</js>})
153    *       <jk>public</jk> MyChildClass x1 = <jk>new</jk> MyChildClass();
154    *    }
155    *
156    *    <jk>public class</jk> MyChildClass {
157    *       <jk>public int</jk> f1 = 1;
158    *       <jk>public int</jk> f2 = 2;
159    *    }
160    *
161    *    <jc>// Renders "{x1:{f1:1}}"</jc>
162    *    String json = JsonSerializer.<jsf>DEFAULT</jsf>.serialize(<jk>new</jk> MyClass());
163    *       </p>
164    *    </dd>
165    * </dl>
166    */
167   String[] properties() default {};
168
169   /**
170    * Marks a bean property as a resource URI identifier for the bean.
171    * <p>
172    * Has the following effects on the following serializers:
173    * <ul>
174    *    <li>{@link XmlSerializer} - Will be rendered as an XML attribute on the bean element, unless
175    *       marked with a {@link Xml#format} value of {@link XmlFormat#ELEMENT}.
176    *    <li>{@link RdfSerializer} - Will be rendered as the value of the <js>"rdf:about"</js> attribute
177    *       for the bean.
178    * </ul>
179    */
180   boolean beanUri() default false;
181}
