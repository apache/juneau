009package com.ibm.juno.core.utils;
010
011import static com.ibm.juno.core.utils.ThrowableUtils.*;
012
013import java.io.*;
014
015/**
016 * File utilities.
017 */
018public class FileUtils {
019
020   /**
021    * Same as {@link File#mkdirs()} except throws a RuntimeExeption if directory could not be created.
022    *
023    * @param f The directory to create.  Must not be <jk>null</jk>.
024    * @param clean If <jk>true</jk>, deletes the contents of the directory if it already exists.
025    * @return The same file.
026    * @throws RuntimeException if directory could not be created.
027    */
028   public static File mkdirs(File f, boolean clean) {
029      assertFieldNotNull(f, "f");
030      if (f.exists()) {
031         if (clean) {
032            if (! delete(f))
033               throw new RuntimeException("Could not clean directory '"+f.getAbsolutePath()+"'");
034         } else {
035            return f;
036         }
037      }
038      if (! f.mkdirs())
039         throw new RuntimeException("Could not create directory '" + f.getAbsolutePath() + "'");
040      return f;
041   }
042
043   /**
044    * Same as {@link #mkdirs(String, boolean)} but uses String path.
045    *
046    * @param path The path of the directory to create.  Must not be <jk>null</jk>
047    * @param clean If <jk>true</jk>, deletes the contents of the directory if it already exists.
048    * @return The directory.
049    */
050   public static File mkdirs(String path, boolean clean) {
051      assertFieldNotNull(path, "path");
052      return mkdirs(new File(path), clean);
053   }
054
055   /**
056    * Recursively deletes a file or directory.
057    *
058    * @param f The file or directory to delete.
059    * @return <jk>true</jk> if file or directory was successfully deleted.
060    */
061   public static boolean delete(File f) {
062      if (f == null)
063         return true;
064      if (f.isDirectory()) {
065         File[] cf = f.listFiles();
066         if (cf != null)
067            for (File c : cf)
068               delete(c);
069      }
070      return f.delete();
071   }
072
073   /**
074    * Creates a file if it doesn't already exist using {@link File#createNewFile()}.
075    * Throws a {@link RuntimeException} if the file could not be created.
076    *
077    * @param f The file to create.
078    */
079   public static void create(File f) {
080      if (f.exists())
081         return;
082      try {
083         if (! f.createNewFile())
084            throw new RuntimeException("Could not create file '"+f.getAbsolutePath()+"'");
085      } catch (IOException e) {
086         throw new RuntimeException(e);
087      }
088   }
089
090   /**
091    * Updates the modified timestamp on the specified file.
092    * Method ensures that the timestamp changes even if it's been modified within the past millisecond.
093    *
094    * @param f The file to modify the modified timestamp on.
095    */
096   public static void modifyTimestamp(File f) {
097      long lm = f.lastModified();
098      long l = System.currentTimeMillis();
099      if (lm == l)
100         l++;
101      if (! f.setLastModified(l))
102         throw new RuntimeException("Could not modify timestamp on file '"+f.getAbsolutePath()+"'");
103
104      // Linux only gives 1s precision, so set the date 1s into the future.
105      if (lm == f.lastModified()) {
106         l += 1000;
107         if (! f.setLastModified(l))
108            throw new RuntimeException("Could not modify timestamp on file '"+f.getAbsolutePath()+"'");
109      }
110   }
111
112   /**
113    * Create a temporary file with the specified name.
114    * <p>
115    * The name is broken into file name and suffix, and the parts
116    * are passed to {@link File#createTempFile(String, String)}.
117    * <p>
118    * {@link File#deleteOnExit()} is called on the resulting file before being returned by this method.
119    *
120    * @param name The file name
121    * @return A newly-created temporary file.
122    * @throws IOException
123    */
124   public static File createTempFile(String name) throws IOException {
125      String[] parts = name.split("\\.");
126      File f = File.createTempFile(parts[0], "." + parts[1]);
127      f.deleteOnExit();
128      return f;
129   }
130}
