009package com.ibm.juno.core.dto.atom;
010
011import java.net.*;
012
013/**
014 * Represents an <code>atomPersonConstruct</code> construct in the RFC4287 specification.
015 * <p>
016 * <h6 class='figure'>Schema</h6>
017 * <p class='bcode'>
018 *    atomPersonConstruct =
019 *       atomCommonAttributes,
020 *       (element atom:name { text }
021 *       & element atom:uri { atomUri }?
022 *       & element atom:email { atomEmailAddress }?
023 *       & extensionElement*)
024 * </p>
025 * <p>
026 *    Refer to {@link com.ibm.juno.core.dto.atom} for further information about ATOM support.
027 * </p>
028 *
029 * @author James Bognar (jbognar@us.ibm.com)
030 */
031public class Person extends Common {
032
033   private String name;
034   private URI uri;
035   private String email;
036
037
038   /**
039    * Normal constructor.
040    *
041    * @param name The name of the person.
042    */
043   public Person(String name) {
044      this.name = name;
045   }
046
047   /** Bean constructor. */
048   public Person() {}
049
050
051   //--------------------------------------------------------------------------------
052   // Bean properties
053   //--------------------------------------------------------------------------------
054
055   /**
056    * Returns the name of the person.
057    *
058    * @return The name of the person.
059    */
060   public String getName() {
061      return name;
062   }
063
064   /**
065    * Sets the name of the person.
066    *
067    * @param name The name of the person.
068    * @return This object (for method chaining).
069    */
070   public Person setName(String name) {
071      this.name = name;
072      return this;
073   }
074
075   /**
076    * Returns the URI of the person.
077    *
078    * @return The URI of the person.
079    */
080   public URI getUri() {
081      return uri;
082   }
083
084   /**
085    * Sets the URI of the person.
086    *
087    * @param uri The URI of the person.
088    * @return This object (for method chaining).
089    */
090   public Person setUri(URI uri) {
091      this.uri = uri;
092      return this;
093   }
094
095   /**
096    * Returns the email address of the person.
097    *
098    * @return The email address of the person.
099    */
100   public String getEmail() {
101      return email;
102   }
103
104   /**
105    * Sets the email address of the person.
106    *
107    * @param email The email address of the person.
108    * @return This object (for method chaining).
109    */
110   public Person setEmail(String email) {
111      this.email = email;
112      return this;
113   }
114
115
116   //--------------------------------------------------------------------------------
117   // Overridden setters (to simplify method chaining)
118   //--------------------------------------------------------------------------------
119
120   @Override /* Common */
121   public Person setBase(URI base) {
122      super.setBase(base);
123      return this;
124   }
125
126   @Override /* Common */
127   public Person setLang(String lang) {
128      super.setLang(lang);
129      return this;
130   }
131}
