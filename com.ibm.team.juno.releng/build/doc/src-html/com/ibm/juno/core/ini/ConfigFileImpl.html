121   @Override /* ConfigFile */
122   public ConfigFileImpl load() throws IOException {
123      Reader r = null;
124      if (file != null && file.exists())
125         r = new InputStreamReader(new FileInputStream(file), charset);
126      else
127         r = new StringReader("");
128      try {
129         load(r);
130      } finally {
131         r.close();
132      }

136   @Override /* ConfigFile */
137   public ConfigFileImpl load(Reader r) throws IOException {
138      assertFieldNotNull(r, "r");
139      writeLock();
140      try {
141         this.sections = Collections.synchronizedMap(new LinkedHashMap<String,Section>());
142         BufferedReader in = new BufferedReader(r);
143         try {
144            writeLock();
145            hasBeenModified = false;
146            try {
147               sections.clear();
148               String line = null;
149               Section section = getSection(null, true);
150               ArrayList<String> lines = new ArrayList<String>();
151               boolean canAppend = false;
152               while ((line = in.readLine()) != null) {
153                  if (isSection(line)) {
154                     section.addLines(null, lines.toArray(new String[lines.size()]));
155                     lines.clear();
156                     canAppend = false;
157                     String sn = StringUtils.replaceUnicodeSequences(line.substring(line.indexOf('[')+1, line.indexOf(']')).trim());
158                     section = getSection(sn, true).addHeaderComments(section.removeTrailingComments());
159                  } else {
160                     char c = line.isEmpty() ? 0 : line.charAt(0);
161                     if ((c == ' ' || c == '\t') && canAppend && ! (isComment(line) || isAssignment(line)))
162                        lines.add(lines.remove(lines.size()-1) + '\n' + line.substring(1));
163                     else {
164                        lines.add(line);
165                        if (isAssignment(line))
166                           canAppend = true;
167                        else
168                           canAppend = canAppend && ! (StringUtils.isEmpty(line) || isComment(line));
169                     }
170                  }
171               }
172               section.addLines(null, lines.toArray(new String[lines.size()]));
173               in.close();
174               if (hasBeenModified)  // Set when values need to be encoded.
175                  save();
176               if (file != null)
177                  modifiedTimestamp = file.lastModified();
178            } finally {
179               writeUnlock();
180            }
181         } finally {
182            in.close();
183         }
184      } finally {
185         writeUnlock();
186      }
187      for (ConfigFileListener l : listeners)
188         l.onLoad(this);
189      return this;
190   }

555   @Override /* ConfigFile */
556   public ConfigFileImpl save() throws IOException {
557      writeLock();
558      try {
559         if (file == null)
560            throw new UnsupportedOperationException("No backing file specified for config file.");
561         Writer out = new OutputStreamWriter(new FileOutputStream(file), charset);
562         try {
563            serializeTo(out);
564            hasBeenModified = false;
565            modifiedTimestamp = file.lastModified();
566         } finally {
567            out.close();
568         }
569         for (ConfigFileListener l : listeners)
570            l.onSave(this);
571         return this;
572      } finally {
573         writeUnlock();
574      }
575   }

629   @Override /* ConfigFile */
630   public ConfigFile merge(ConfigFile cf) {
631      assertFieldNotNull(cf, "cf");
632      Set<String> changes = createChanges();
633      writeLock();
634      try {
635         for (String sectionName : this.keySet())
636            if (! cf.containsKey(sectionName))
637               remove(sectionName, changes);
638
639         for (Map.Entry<String,Section> e : cf.entrySet())
640            put(e.getKey(), e.getValue(), changes);
641
642      } finally {
643         writeUnlock();
644      }
645      signalChanges(changes);
646      return this;
647   }

704   /*
705    * Finds the keys that are different between the two sections and adds it to
706    * the specified set.
707    */
708   private void findChanges(Set<String> s, Section a, Section b) {
709      if (s == null)
710         return;
711      String sname = (a == null ? b.name : a.name);
712      if (a == null) {
713         for (String k : b.keySet())
714            s.add(getFullKey(sname, k));
715      } else if (b == null) {
716         for (String k : a.keySet())
717            s.add(getFullKey(sname, k));
718      } else {
719         for (String k : a.keySet())
720            addChange(s, sname, k, a.get(k), b.get(k));
721         for (String k : b.keySet())
722            addChange(s, sname, k, a.get(k), b.get(k));
723      }
724   }
725
726   private void addChange(Set<String> changes, String section, String key, String oldVal, String newVal) {
727      if (! StringUtils.isEquals(oldVal, newVal))
728         changes.add(getFullKey(section, key));
729   }

737         for (ConfigFileListener l : listeners)
