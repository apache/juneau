009package com.ibm.juno.core.utils;
010
011import static com.ibm.juno.core.utils.ThrowableUtils.*;
012
013import java.util.*;
014
015/**
016 * Wrapper around a map where the key names are overridden.
017 *
018 * @param <K> The key class type.
019 * @param <V> The value class type.
020 * @author James Bognar (jbognar@us.ibm.com)
021 */
022public final class FilteredMap<K,V> extends AbstractMap<K,V> {
023
024   private Map<K,V> innerMap;
025   private Set<Map.Entry<K,V>> entries;
026
027   /**
028    * Constructor.
029    *
030    * @param innerMap The map being wrapped.  Must not be <jk>null</jk>.
031    * @param keys The keys in the new map.  Must not be <jk>null</jk>.
032    */
033   public FilteredMap(Map<K,V> innerMap, K[] keys) {
034      assertFieldNotNull(innerMap, "innerMap");
035      assertFieldNotNull(keys, "keys");
036
037      this.innerMap = innerMap;
038         List<Map.Entry<K,V>> l = new ArrayList<Map.Entry<K,V>>(keys.length);
039         for (K k : keys)
040            if (innerMap.containsKey(k))
041               l.add(createEntry(k));
042         entries = new ListSet<Map.Entry<K,V>>(l);
043      }
044
045   private Map.Entry<K,V> createEntry(final K key) {
046      return new Map.Entry<K,V>() {
047
048         @Override /* Map.Entry */
049         public K getKey() {
050            return key;
051         }
052
053         @Override /* Map.Entry */
054         public V getValue() {
055            return innerMap.get(key);
056         }
057
058         @Override /* Map.Entry */
059         public V setValue(V v) {
060            return innerMap.put(key, v);
061         }
062      };
063   }
064
065
066   @Override /* Map */
067   public Set<Map.Entry<K,V>> entrySet() {
068      return entries;
069   }
070
071   /**
072    * A set with ordered entries (i.e. a List with a Set API).
073    */
074   private static class ListSet<E> extends AbstractSet<E> {
075
076      private List<E> entries;
077
078      public ListSet(List<E> entries) {
079         this.entries = entries;
080      }
081
082      @Override /* Set */
083      public Iterator<E> iterator() {
084         return entries.iterator();
085      }
086
087      @Override /* Set */
088      public int size() {
089         return entries.size();
090      }
091   }
092}
