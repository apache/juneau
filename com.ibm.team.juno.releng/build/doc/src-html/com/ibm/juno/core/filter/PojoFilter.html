009package com.ibm.juno.core.filter;
010
011import java.lang.reflect.*;
012import java.util.*;
013
014import com.ibm.juno.core.*;
015import com.ibm.juno.core.parser.*;
016import com.ibm.juno.core.serializer.*;
017import com.ibm.juno.core.utils.*;
018
019/**
020 * Used to convert non-serializable objects to a serializable form.
021 *
022 *
023 * <h6 class='topic'>Description</h6>
024 * <p>
025 *    <code>PojoFilters</code> are used to extend the functionality of the serializers and parsers to be able to handle POJOs
026 *    that aren't automatically handled by the serializers or parsers.  For example, JSON does not have a standard
027 *    representation for rendering dates.  By defining a special {@code Date} filter and associating it with a serializer and
028 *    parser, you can convert a {@code Date} object to a {@code String} during serialization, and convert that {@code String} object back into
029 *    a {@code Date} object during parsing.
030 * <p>
031 *    Object filters MUST declare a public no-arg constructor so that the bean context can instantiate them.
032 * <p>
033 *    <code>PojoFilters</code> are associated with instances of {@link BeanContext BeanContexts} by passing the filter class to
034 *    the {@link BeanContextFactory#addFilters(Class...)} method.<br>
035 *    When associated with a bean context, fields of the specified type will automatically be converted when the
036 *    {@link BeanMap#get(Object)} or {@link BeanMap#put(String, Object)} methods are called.<br>
037 * <p>
038 *    <code>PojoFilters</code> have two parameters:
039 *    <ol>
040 *       <li>{@code <F>} - The filtered representation of an object.
041 *       <li>{@code <T>} - The normal representation of an object.
042 *    </ol>
043 *    <br>
044 *    {@link Serializer Serializers} use object filters to convert objects of type T into objects of type F, and on calls to {@link BeanMap#get(Object)}.<br>
045 *    {@link Parser Parsers} use object filters to convert objects of type F into objects of type T, and on calls to {@link BeanMap#put(String,Object)}.
046 *
047 *
048 * <h6 class='topic'>Filtered Class Type {@code <F>}</h6>
049 * <p>
050 *    The filtered object representation of an object must be an object type that the serializers can
051 *    natively convert to JSON (or language-specific equivalent).  The list of valid filtered types are as follows...
052 *    <ul>
053 *       <li>{@link String}
054 *       <li>{@link Number}
055 *       <li>{@link Boolean}
056 *       <li>{@link Collection} containing anything on this list.
057 *       <li>{@link Map} containing anything on this list.
058 *       <li>A java bean with properties of anything on this list.
059 *       <li>An array of anything on this list.
060 *    </ul>
061 *
062 *
063 * <h6 class='topic'>Normal Class Type {@code <T>}</h6>
064 * <p>
065 *    The normal object representation of an object.<br>
066 *
067 *
068 * <h6 class='topic'>One-way vs. Two-way Serialization</h6>
069 * <p>
070 *    Note that while there is a unified interface for handling filtering during both serialization and parsing,
071 *    in many cases only one of the {@link #filter(Object)} or {@link #unfilter(Object, ClassMeta)} methods will be defined
072 *    because the filter is one-way.  For example, a filter may be defined to convert an {@code Iterator} to a {@code ObjectList}, but
073 *    it's not possible to unfilter an {@code Iterator}.  In that case, the {@code generalize(Object}} method would
074 *    be implemented, but the {@code narrow(ObjectMap)} object would not, and the filter would be associated on
075 *    the serializer, but not the parser.  Also, you may choose to serialize objects like {@code Dates} to readable {@code Strings},
076 *    in which case it's not possible to reparse it back into a {@code Date}, since there is no way for the {@code Parser} to
077 *    know it's a {@code Date} from just the JSON or XML text.
078 *
079 *
080 * <h6 class='topic'>Additional information</h6>
081 *    See {@link com.ibm.juno.core.filter} for more information.
082 *
083 *
084 * @author James Bognar (jbognar@us.ibm.com)
085 * @param <T> The normal form of the class.
086 * @param <F> The filtered form of the class.
087 */
088public abstract class PojoFilter<T,F> extends Filter {
089
090   /** Represents no filter. */
091   public static class NULL extends PojoFilter<Object,Object> {}
092
093   Class<T> normalClass;
094   Class<F> filteredClass;
095   ClassMeta<F> filteredClassMeta;
096
097   /**
098    * Constructor.
099    */
100   @SuppressWarnings("unchecked")
101   protected PojoFilter() {
102      super();
103
104      Class<?> c = this.getClass().getSuperclass();
105      Type t = this.getClass().getGenericSuperclass();
106      while (c != PojoFilter.class) {
107         t = c.getGenericSuperclass();
108         c = c.getSuperclass();
109      }
110
111      // Attempt to determine the T and G classes using reflection.
112      if (t instanceof ParameterizedType) {
113         ParameterizedType pt = (ParameterizedType)t;
114         Type[] pta = pt.getActualTypeArguments();
115         if (pta.length == 2) {
116            Type nType = pta[0];
117            if (nType instanceof Class) {
118               this.normalClass = (Class<T>)nType;
119
120            // <byte[],x> ends up containing a GenericArrayType, so it has to
121            // be handled as a special case.
122            } else if (nType instanceof GenericArrayType) {
123               Class<?> cmpntType = (Class<?>)((GenericArrayType)nType).getGenericComponentType();
124               this.normalClass = (Class<T>)Array.newInstance(cmpntType, 0).getClass();
125
126            // <Class<?>,x> ends up containing a ParameterizedType, so just use the raw type.
127            } else if (nType instanceof ParameterizedType) {
128               this.normalClass = (Class<T>)((ParameterizedType)nType).getRawType();
129
130            } else
131               throw new RuntimeException("Unsupported parameter type: " + nType);
132            if (pta[1] instanceof Class)
133               this.filteredClass = (Class<F>)pta[1];
134            else if (pta[1] instanceof ParameterizedType)
135               this.filteredClass = (Class<F>)((ParameterizedType)pta[1]).getRawType();
136            else
137               throw new RuntimeException("Unexpected filtered class type: " + pta[1].getClass().getName());
138         }
139      }
140   }
141
142   /**
143    * Constructor for when the normal and filtered classes are already known.
144    *
145    * @param normalClass The normal class (cannot be serialized).
146    * @param filteredClass The filtered class (serializable).
147    */
148   protected PojoFilter(Class<T> normalClass, Class<F> filteredClass) {
149      this.normalClass = normalClass;
150      this.filteredClass = filteredClass;
151   }
152
153   /**
154    * If this filter is to be used to serialize non-serializable POJOs, it must implement this method.
155    * <p>
156    *    The object must be converted into one of the following serializable types:
157    *    <ul>
158    *       <li>{@link String}
159    *       <li>{@link Number}
160    *       <li>{@link Boolean}
161    *       <li>{@link Collection} containing anything on this list.
162    *       <li>{@link Map} containing anything on this list.
163    *       <li>A java bean with properties of anything on this list.
164    *       <li>An array of anything on this list.
165    *    </ul>
166    *
167    * @param o The object to be filtered.
168    * @return The filtered object.
169    * @throws SerializeException If a problem occurred trying to convert the output.
170    */
171   public F filter(T o) throws SerializeException {
172      throw new SerializeException("Generalize method not implemented on filter ''{0}''", this.getClass().getName());
173   }
174
175   /**
176    * If this filter is to be used to reconstitute POJOs that aren't true Java beans, it must implement this method.
177    *
178    * @param f The filtered object.
179    * @param hint If possible, the parser will try to tell you the object type being created.  For example,
180    *    on a serialized date, this may tell you that the object being created must be of type {@code GregorianCalendar}.<br>
181    *    This may be <jk>null</jk> if the parser cannot make this determination.
182    * @return The narrowed object.
183    * @throws ParseException If this method is not implemented.
184    */
185   public T unfilter(F f, ClassMeta<?> hint) throws ParseException {
186      throw new ParseException("Narrow method not implemented on filter ''{0}''", this.getClass().getName());
187   }
188
189   /**
190    * Returns the T class, the normalized form of the class.
191    *
192    * @return The normal form of this class.
193    */
194   public Class<T> getNormalClass() {
195      return normalClass;
196   }
197
198   /**
199    * Returns the G class, the generialized form of the class.
200    * <p>
201    *    Subclasses must override this method if the generialized class is {@code Object},
202    *    meaning it can produce multiple generialized forms.
203    *
204    * @return The filtered form of this class.
205    */
206   public Class<F> getFilteredClass() {
207      return filteredClass;
208   }
209
210   /**
211    * Returns the {@link ClassMeta} of the filtered class type.
212    * This value is cached for quick lookup.
213    *
214    * @return The {@link ClassMeta} of the filtered class type.
215    */
216   public ClassMeta<F> getFilteredClassMeta() {
217      if (filteredClassMeta == null)
218         filteredClassMeta = beanContext.getClassMeta(filteredClass);
219      return filteredClassMeta;
220   }
221
222   /**
223    * Checks if the specified object is an instance of the normal class defined on this filter.
224    *
225    * @param o The object to check.
226    * @return <jk>true</jk> if the specified object is a subclass of the normal class defined on this filter.
227    *    <jk>null</jk> always return <jk>false</jk>.
228    */
229   public boolean isNormalObject(Object o) {
230      if (o == null)
231         return false;
232      return ClassUtils.isParentClass(normalClass, o.getClass());
233   }
234
235   /**
236    * Checks if the specified object is an instance of the filtered class defined on this filter.
237    *
238    * @param o The object to check.
239    * @return <jk>true</jk> if the specified object is a subclass of the filtered class defined on this filter.
240    *    <jk>null</jk> always return <jk>false</jk>.
241    */
242   public boolean isFilteredObject(Object o) {
243      if (o == null)
244         return false;
245      return ClassUtils.isParentClass(filteredClass, o.getClass());
246   }
247
248   //--------------------------------------------------------------------------------
249   // Overridden methods
250   //--------------------------------------------------------------------------------
251
252   @Override /* Filter */
253   public Class<?> forClass() {
254      return normalClass;
255   }
256
257   @Override /* Object */
258   public String toString() {
259      return getClass().getSimpleName() + '<' + getNormalClass().getSimpleName() + "," + getFilteredClass().getSimpleName() + '>';
260   }
261}
