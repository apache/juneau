011import java.util.*;

013import com.ibm.juno.core.*;

015/**
016 * Utility class to make it easier to work with command-line arguments pass in through a <code>main(String[] args)</code> method.
017 * <p>
018 * Used to parse command-line arguments of the form <js>"[zero or more main arguments] [zero or more optional arguments]"</js>.
019 * <p>
020 * The format of a main argument is a token that does not start with <js>'-'</js>.
021 * <p>
022 * The format of an optional argument is <js>"-argName [zero or more tokens]"</js>.
023 * <p>
024 * <h6 class='topic'>Command-line examples</h6>
025 * <ul>
026 *    <li><code>java com.sample.MyClass mainArg1</code>
027 *    <li><code>java com.sample.MyClass mainArg1 mainArg2</code>
028 *    <li><code>java com.sample.MyClass mainArg1 -optArg1</code>
029 *    <li><code>java com.sample.MyClass -optArg1</code>
030 *    <li><code>java com.sample.MyClass mainArg1 -optArg1 optArg1Val</code>
031 *    <li><code>java com.sample.MyClass mainArg1 -optArg1 optArg1Val1 optArg1Val2</code>
032 *    <li><code>java com.sample.MyClass mainArg1 -optArg1 optArg1Val1 -optArg1 optArg1Val2</code>
033 * </ul>
034 *
035 * <h6 class='topic'>Code examples</h6>
036 * <p class='bcode'>
037 *
038 *    <jc>// Main method with arguments</jc>
039 *    <jk>public static void</jk> <jsm>main</jsm>(String[] args) {
040 *
041 *       <jc>// Wrap in Args</jc>
042 *       Args a = new Args(args);
043 *
044 *       <jc>// One main argument</jc>
045 *       <jc>// a1</jc>
046 *       String a1 = a.getArg(0); <jc>// "a1"</jc>
047 *       String a2 = a.getArg(1); <jc>// null</jc>
048 *
049 *       <jc>// Two main arguments</jc>
050 *       <jc>// a1 a2</jc>
051 *       String a1 = a.getArg(0); <jc>// "a1"</jc>
052 *       String a2 = a.getArg(1); <jc>// "a2"</jc>
053 *
054 *       <jc>// One main argument and one optional argument with no value</jc>
055 *       <jc>// a1 -a2</jc>
056 *       String a1 = a.getArg(0);
057 *       <jk>boolean</jk> hasA2 = a.hasArg(<js>"a2"</js>); <jc>// true</jc>
058 *       <jk>boolean</jk> hasA3 = a.hasArg(<js>"a3"</js>); <jc>// false</jc>
059 *
060 *       <jc>// One main argument and one optional argument with one value</jc>
061 *       <jc>// a1 -a2 v2</jc>
062 *       String a1 = a.getArg(0);
063 *       String a2 = a.getArg(<js>"a2"</js>); <jc>// "v2"</jc>
064 *       String a3 = a.getArg(<js>"a3"</js>); <jc>// null</jc>
065 *
066 *       <jc>// One main argument and one optional argument with two values</jc>
067 *       <jc>// a1 -a2 v2a v2b</jc>
068 *       String a1 = a.getArg(0);
069 *       List&lt;String&gt; a2 = a.getArgs(<js>"a2"</js>); <jc>// Contains ["v2a","v2b"]</jc>
070 *       List&lt;String&gt; a3 = a.getArgs(<js>"a3"</js>); <jc>// Empty list</jc>
071 *
072 *       <jc>// Same as previous, except specify optional argument name multiple times</jc>
073 *       <jc>// a1 -a2 v2a -a2 v2b</jc>
074 *       String a1 = a.getArg(0);
075 *       List&lt;String&gt; a2 = a.getArgs(<js>"a2"</js>); <jc>// Contains ["v2a","v2b"]</jc>
076 *    }
077 * </p>
078 * <p>
079 * Main arguments are available through numeric string keys (e.g. <js>"0"</js>, <js>"1"</js>, ...).
080 * So you could use the {@link ObjectMap} API to convert main arguments directly to POJOs, such as an <code>Enum</code>
081 * <p class='bcode'>
082 *    <jc>// Get 1st main argument as an Enum</jc>
083 *    MyEnum e = a.get(MyEnum.<jk>class</jk>, <js>"0"</js>);
084 *
085 *    <jc>// Get 1st main argument as an integer</jc>
086 *    int i = a.get(<jk>int</jk>.<jk>class</jk>, <js>"0"</js>);
087 * </p>
088 * <p>
089 * Equivalent operations are available on optional arguments through the {@link #getArg(Class, String)} method.
090 *
091 * @author jbognar
092 */
093public final class Args extends ObjectMap {
094
095   private static final long serialVersionUID = 1L;
096   private transient volatile StringVar stringVar;
097
098   /**
099    * Constructor.
100    *
101    * @param args Arguments passed in through a <code>main(String[] args)</code> method.
102    */
103   public Args(String[] args) {
104      List<String> argList = new LinkedList<String>(Arrays.asList(args));
105
106      // Capture the main arguments.
107      Integer i = 0;
108      while (! argList.isEmpty()) {
109         String s = argList.get(0);
110         if (StringUtils.startsWith(s,'-'))
111            break;
112         put(i.toString(), argList.remove(0));
113         i++;
114      }
115
116      // Capture the mapped arguments.
117      String key = null;
118      while (! argList.isEmpty()) {
119         String s = argList.remove(0);
120         if (StringUtils.startsWith(s, '-')) {
121            key = s.substring(1);
122            if (key.matches("\\d*"))
123               throw new RuntimeException("Invalid optional key name '"+key+"'");
124            if (! containsKey(key))
125               put(key, new ObjectList());
126         } else {
127            ((ObjectList)get(key)).add(s);
128         }
129      }
130   }
131
132   /**
133    * Returns main argument at the specified index, or <jk>null</jk> if the index is out of range.
134    * <p>
135    * Can be used in conjuction with {@link #hasArg(int)} to check for existence of arg.
136    * <p class='bcode'>
137    *    <jc>// Check for no arguments</jc>
138    *    <jk>if</jk> (! args.hasArg(0))
139    *       printUsageAndExit();
140    *
141    *    <jc>// Get the first argument</jc>
142    *    String firstArg = args.getArg(0);
143    * </p>
144    * <p>
145    * Since main arguments are stored as numeric keys, this method is essentially equivalent to...
146    * <p class='bcode'>
147    *    <jc>// Check for no arguments</jc>
148    *    <jk>if</jk> (! args.containsKey(<js>"0"</js>))
149    *       printUsageAndExit();
150    *
151    *    <jc>// Get the first argument</jc>
152    *    String firstArg = args.getString("0");
153    * </p>
154    *
155    * @param i The index position of the main argument (zero-indexed).
156    * @return The main argument value, or <js>""</js> if argument doesn't exist at that position.
157    */
158   public String getArg(int i) {
159      return getString(Integer.toString(i));
160   }
161
162   /**
163    * Returns <jk>true</jk> if argument exists at specified index.
164    *
165    * @param i The zero-indexed position of the argument.
166    * @return <jk>true</jk> if argument exists at specified index.
167    */
168   public boolean hasArg(int i) {
169      return containsKey(Integer.toString(i));
170   }
171
172   /**
173    * Returns the optional argument value, or blank if the optional argument was not specified.
174    * <p>
175    * If the optional arg has multiple values, returns values as a comma-delimited list.
176    *
177    * @param name The optional argument name.
178    * @return The optional argument value, or blank if the optional argument was not specified.
179    */
180   public String getArg(String name) {
181      ObjectList l = (ObjectList)get(name);
182      if (l == null || l.size() == 0)
183         return null;
184      if (l.size() == 1)
185         return l.get(0).toString();
186      return Arrays.toString(l.toArray()).replaceAll("[\\[\\]]", "");
187   }
188
189   /**
190    * Returns the optional argument value converted to the specified object type.
191    * <p>
192    * If the optional arg has multiple values, returns only the first converted value.
193    *
194    * <dl>
195    *    <dt>Example:</dt>
196    *    <dd>
197    *       <p class='bcode'>
198    *    <jc>// Command:  java com.sample.MyClass -verbose true -debug 5</jc>
199    *    <jk>boolean</jk> b = args.getArg(<jk>boolean</jk>.<jk>class</jk>, <js>"verbose"</js>);
200    *    <jk>int</jk> i = args.getArg(<jk>int</jk>.<jk>class</jk>, <js>"debug"</js>);
201    *       </p>
202    *    </dd>
203    * </dl>
204    *
205    * @param c The class type to convert the value to.
206    * @param <T> The class type to convert the value to.
207    * @param name The optional argument name.
208    * @return The optional argument value, or blank if the optional argument was not specified.
209    */
210   public <T> T getArg(Class<T> c, String name) {
211      ObjectList l = (ObjectList)get(name);
212      if (l == null || l.size() == 0)
213         return null;
214      return l.get(c, 0);
215   }
216
217   /**
218    * Returns the optional argument values as a list of strings.
219    *
220    * <dl>
221    *    <dt>Example:</dt>
222    *    <dd>
223    *       <p class='bcode'>
224    *    <jc>// Command:  java com.sample.MyClass -extraArgs foo bar baz</jc>
225    *    List&lt;String&gt; l1 = args.getArgs(<js>"extraArgs"</js>); <jc>// ['foo','bar','baz']</jc>
226    *    List&lt;String&gt; l2 = args.getArgs(<js>"nonExistentArgs"</js>); <jc>// An empty list</jc>
227    *       </p>
228    *    </dd>
229    * </dl>
230    *
231    * @param name The optional argument name.
232    * @return The optional argument values, or an empty list if the optional argument was not specified.
233    */
234   @SuppressWarnings({"rawtypes", "unchecked"})
235   public List<String> getArgs(String name) {
236      List l = (ObjectList)get(name);
237      if (l == null)
238         return Collections.emptyList();
239      return l;
240   }
241
242   /**
243    * Create an instance of a {@link StringVarWithDefault} that returns values in this object.
244    *
245    * @return A new {@link StringVarWithDefault}.
246    */
247   public StringVar getStringVar() {
248      if (stringVar == null)
249         stringVar = new StringVarWithDefault() {
250         @Override /* StringVar */
251         public String resolve(String varVal) {
252            return getArg(varVal);
253         }
254      };
255      return stringVar;
256   }
