009package com.ibm.juno.core.filters;
010
011import java.util.*;
012
013import com.ibm.juno.core.*;
014import com.ibm.juno.core.filter.*;
015import com.ibm.juno.core.parser.*;
016
017/**
018 * Transforms {@link Calendar Calendars} to {@link Map Maps} of the format <code>{_class:String,value:long}</code>.
019 * <p>
020 *    TODO:  This class does not handle timezones correctly when parsing {@code GregorianCalendar} objects.
021 *
022 * @author James Bognar (jbognar@us.ibm.com)
023 */
024@SuppressWarnings("rawtypes")
025public class CalendarMapFilter extends PojoFilter<Calendar,Map> {
026
027   /**
028    * Converts the specified {@link Calendar} to a {@link Map}.
029    */
030   @Override /* PojoFilter */
031   public Map filter(Calendar o) {
032      ObjectMap m = new ObjectMap();
033      m.put("time", o.getTime().getTime());
034      m.put("timeZone", o.getTimeZone().getID());
035      return m;
036   }
037
038   /**
039    * Converts the specified {@link Map} to a {@link Calendar}.
040    */
041   @Override /* PojoFilter */
042   @SuppressWarnings("unchecked")
043   public Calendar unfilter(Map o, ClassMeta<?> hint) throws ParseException {
044      ClassMeta<? extends Calendar> tt;
045      try {
046         if (hint == null || ! hint.canCreateNewInstance())
047            hint = getBeanContext().getClassMeta(GregorianCalendar.class);
048         tt = (ClassMeta<? extends Calendar>)hint;
049         long time = Long.parseLong(o.get("time").toString());
050         String timeZone = o.get("timeZone").toString();
051         Date d = new Date(time);
052         Calendar c = tt.newInstance();
053         c.setTime(d);
054         c.setTimeZone(TimeZone.getTimeZone(timeZone));
055         return c;
056      } catch (Exception e) {
057         throw new ParseException(e);
058      }
059   }
060}
