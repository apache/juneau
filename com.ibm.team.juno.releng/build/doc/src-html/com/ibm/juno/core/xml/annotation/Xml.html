009package com.ibm.juno.core.xml.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016import com.ibm.juno.core.xml.*;
017
018/**
019 * Annotation for specifying various XML options for the XML and RDF/XML serializers.
020 * <p>
021 *    Can be applied to Java packages, types, fields, and methods.
022 * <p>
023 *    Can be used for the following:
024 * <ul>
025 *    <li>Override the element name on the XML representation of a bean or object.
026 *    <li>Override the child element name on the XML representation of collection or array properties.
027 *    <li>Specify the XML namespace on a package, class, or method.
028 *    <li>Override the XML format on a POJO.
029 * </ul>
030 *
031 * @author James Bognar (jbognar@us.ibm.com)
032 */
033@Documented
034@Target({TYPE,FIELD,METHOD})
035@Retention(RUNTIME)
036@Inherited
037public @interface Xml {
038
039   /**
040    * Sets the name of the XML element in cases where the XML element has no name (e.g. the root element).
041    * <p>
042    *    Applies only to {@link ElementType#TYPE}.
043    *
044    * <dl>
045    *    <dt>Example:</dt>
046    *    <dd>
047    *       <p class='bcode'>
048    *    <ja>@Xml</ja>(name=<js>"MyBean"</js>)
049    *    <jk>public class</jk> MyBean {
050    *       <jk>public int</jk> f1 = 123;
051    *    }
052    *       </p>
053    *       <p>
054    *          Without the <ja>@Xml</ja> annotations, serializing this bean as XML would have produced the following...
055    *       </p>
056    *       <p class='bcode'>
057    *    <xt>&lt;object&gt;</xt>
058    *       <xt>&lt;f1&gt;</xt>123<xt>&lt;/f1&gt;</xt>
059    *    <xt>&lt;/object&gt;</xt>
060    *       </p>
061    *       <p>
062    *          With the annotations, serializing this bean as XML produces the following...
063    *       </p>
064    *       <p class='bcode'>
065    *    <xt>&lt;MyBean&gt;</xt>
066    *       <xt>&lt;f1&gt;</xt>123<xt>&lt;/f1&gt;</xt>
067    *    <xt>&lt;/MyBean&gt;</xt>
068    *       </p>
069    *    </dd>
070    * </dl>
071    */
072   String name() default "";
073
074   /**
075    * Sets the name of the XML child elements for bean properties of type collection and array.
076    * <p>
077    *    Applies only to collection and array bean properties.
078    *
079    * <dl>
080    *    <dt>Example:</dt>
081    *    <dd>
082    *       <p class='bcode'>
083    *    <jk>public class</jk> MyBean {
084    *       <ja>@Xml</ja>(childName=<js>"child"</js>}
085    *       <jk>public</jk> String[] <jf>children</jf> = {<js>"foo"</js>,<js>"bar"</js>};
086    *    }
087    *       </p>
088    *       <p>
089    *          Without the <ja>@Xml</ja> annotation, serializing this bean as XML would have produced the following...
090    *       </p>
091    *       <p class='bcode'>
092    *    <xt>&lt;object&gt;</xt>
093    *       <xt>&lt;children&gt;</xt>
094    *          <xt>&lt;string&gt;</xt>foo<xt>&lt;/string&gt;</xt>
095    *          <xt>&lt;string&gt;</xt>bar<xt>&lt;/string&gt;</xt>
096    *       <xt>&lt;/children&gt;</xt>
097    *    <xt>&lt;/object&gt;</xt>
098    *       </p>
099    *       <p>
100    *          With the annotations, serializing this bean as XML produces the following...
101    *       </p>
102    *       <p class='bcode'>
103    *    <xt>&lt;object&gt;</xt>
104    *       <xt>&lt;children&gt;</xt>
105    *          <xt>&lt;child&gt;</xt>foo<xt>&lt;/child&gt;</xt>
106    *          <xt>&lt;child&gt;</xt>bar<xt>&lt;/child&gt;</xt>
107    *       <xt>&lt;/children&gt;</xt>
108    *    <xt>&lt;/object&gt;</xt>
109    *       </p>
110    *    </dd>
111    * </dl>
112    */
113   String childName() default "";
114
115   /**
116    * Sets the XML prefix of this property or class.
117    * <ul>
118    *    <li>When applied to a {@link ElementType#TYPE}, namespace is applied to all properties in the class, and all
119    *       subclasses of the class.
120    *    <li>When applied to bean properties on {@link ElementType#METHOD} and {@link ElementType#FIELD}, applies
121    *       to the bean property.
122    * </ul>
123    * <p>
124    *    Must either be matched to a {@link #namespace()} annotation on the same object, parent object, or a {@link XmlNs} with the same name
125    *    through the {@link XmlSchema#xmlNs()} annotation on the package.
126    * </p>
127    */
128   String prefix() default "";
129
130   /**
131    * Sets the namespace URI of this property or class.
132    * <p>
133    *    Must be matched with a {@link #prefix()} annotation on this object, a parent object, or a {@link XmlNs} with the same name
134    *    through the {@link XmlSchema#xmlNs()} annotation on the package.
135    */
136   String namespace() default "";
137
138   /**
139    * The {@link XmlFormat} to use for serializing this object type.
140    *
141    * <dl>
142    *    <dt>Example:</dt>
143    *    <dd>
144    *       <p class='bcode'>
145    *    <jk>public class</jk> MyBean {
146    *
147    *       <jc>// Normally, bean properties would be rendered as child elements of the bean element.</jc>
148    *       <jc>// Override so that it's rendered as a "f1='123'" attribute on the bean element instead.</jc>
149    *       <ja>@Xml</ja>(format=XmlFormat.<jsf>ATTR</jsf>}
150    *       <jk>public int</jk> f1 = 123;
151    *
152    *       <jc>// Normally, bean URL properties would be rendered as XML attributes on the bean element.</jc>
153    *       <jc>// Override so that it's rendered as an &lt;href&gt;http://foo&lt;/href&gt; child element instead.</jc>
154    *       <ja>@BeanProperty</ja>(uri=<jk>true</jk>)
155    *       <ja>@Xml</ja>(format=XmlFormat.<jsf>ELEMENT</jsf>}
156    *       <jk>public</jk> URL <jf>href</jf> = <jk>new</jk> URL(<js>"http://foo"</js>);
157    *
158    *       <jc>// Normally, collection properties would be grouped under a single &lt;children&gt; child element on the bean element.</jc>
159    *       <jc>// Override so that entries are directly children of the bean element with each entry having an element name of &lt;child&gt;.</jc>
160    *       <ja>@Xml</ja>(format=XmlFormat.<jsf>COLLAPSED</jsf>, childName=<js>"child"</js>}
161    *       <jk>public</jk> String[] <jf>children</jf> = <js>"foo"</js>,<js>"bar"</js>};
162    *    }
163    *       </p>
164    *       <p>
165    *          Without the <ja>@Xml</ja> annotations, serializing this bean as XML would have produced the following...
166    *       </p>
167    *       <p class='bcode'>
168    *    <xt>&lt;object</xt> <xa>href</xa>=<js>'http://foo'</js><xt>&gt;</xt>
169    *       <xt>&lt;f1&gt;</xt>123<xt>&lt;/f1&gt;</xt>
170    *       <xt>&lt;children&gt;</xt>
171    *          <xt>&lt;string&gt;</xt>foo<xt>&lt;/string&gt;</xt>
172    *          <xt>&lt;string&gt;</xt>bar<xt>&lt;/string&gt;</xt>
173    *       <xt>&lt;/children&gt;</xt>
174    *    <xt>&lt;/object&gt;</xt>
175    *       </p>
176    *       <p>
177    *          With the annotations, serializing this bean as XML produces the following...
178    *       </p>
179    *       <p class='bcode'>
180    *    <xt>&lt;object</xt> <xa>f1</xa>=<js>'123'</js><xt>&gt;</xt>
181    *       <xt>&lt;href&gt;</xt>http://foo<xt>&lt;/href&gt;</xt>
182    *       <xt>&lt;child&gt;</xt>foo<xt>&lt;/child&gt;</xt>
183    *       <xt>&lt;child&gt;</xt>bar<xt>&lt;/child&gt;</xt>
184    *    <xt>&lt;/object&gt;</xt>
185    *       </p>
186    *    </dd>
187    * </dl>
188    */
189   XmlFormat format() default XmlFormat.NORMAL;
190
191   /**
192    * Associates a content handler with a bean class or bean property.
193    * <p>
194    *    Refer to {@link XmlContentHandler} for more information.
195    */
196   Class<? extends XmlContentHandler<?>> contentHandler() default XmlContentHandler.NULL.class;
197}
