009package com.ibm.juno.core;
010
011import java.util.*;
012import java.util.Map.Entry;
013
014/**
015 * Describes a single type used in content negotiation between an HTTP client and server, as described in
016 * Section 14.1 and 14.7 of RFC2616 (the HTTP/1.1 specification).
017 */
018public final class MediaRange implements Comparable<MediaRange>  {
019
020   private final String type;                      // The media type (e.g. "text" for Accept, "utf-8" for Accept-Charset)
021   private final String subType;                   // The media sub-type (e.g. "json" for Accept, not used for Accept-Charset)
022   private final Float qValue;
023   private final Map<String,Set<String>> parameters, extensions;
024
025   /**
026    * Returns the media type enclosed by this media range.
027    * <p>
028    * Examples:
029    * <ul>
030    *    <li><js>"text/html"</js>
031    *    <li><js>"text/*"</js>
032    *    <li><js>"*\/*"</js>
033    * </ul>
034    *
035    * @return The media type of this media range, lowercased, never <jk>null</jk>.
036    */
037   public String getMediaType() {
038      return type + "/" + subType;
039   }
040
041   /**
042    * Return just the type portion of this media range.
043    *
044    * @return The type portion of this media range.
045    */
046   public String getType() {
047      return type;
048   }
049
050   /**
051    * Returns the <js>'q'</js> (quality) value for this type, as described in Section 3.9 of RFC2616.
052    * <p>
053    * The quality value is a float between <code>0.0</code> (unacceptable) and <code>1.0</code> (most acceptable).
054    * <p>
055    * If 'q' value doesn't make sense for the context (e.g. this range was extracted from a <js>"content-*"</js> header, as opposed to <js>"accept-*"</js>
056    * header, its value will always be <js>"1"</js>.
057    *
058    * @return The 'q' value for this type, never <jk>null</jk>.
059    */
060   public Float getQValue() {
061      return qValue;
062   }
063
064   /**
065    * Returns the optional set of parameters associated to the type as returned by {@link #getMediaType()}.
066    * <p>
067    * The parameters are those values as described in standardized MIME syntax.
068    * An example of such a parameter in string form might be <js>"level=1"</js>.
069    * <p>
070    * Values are lowercase and never <jk>null</jk>.
071    *
072    * @return The optional list of parameters, never <jk>null</jk>.
073    */
074   public Map<String,Set<String>> getParameters() {
075      return parameters;
076   }
077
078   /**
079    * Returns the optional set of custom extensions defined for this type.
080    * <p>
081    * Values are lowercase and never <jk>null</jk>.
082    *
083    * @return The optional list of extensions, never <jk>null</jk>.
084    */
085   public Map<String,Set<String>> getExtensions() {
086      return extensions;
087   }
088
089   /**
090    * Provides a string representation of this media range, suitable for use as an <code>Accept</code> header value.
091    * <p>
092    * The literal text generated will be all lowercase.
093    *
094    * @return A media range suitable for use as an Accept header value, never <code>null</code>.
095    */
096   @Override /* Object */
097   public String toString() {
098      StringBuffer sb = new StringBuffer().append(type).append('/').append(subType);
099
100      if (! parameters.isEmpty())
101         for (Entry<String,Set<String>> e : parameters.entrySet()) {
102            String k = e.getKey();
103            for (String v : e.getValue())
104               sb.append(';').append(k).append('=').append(v);
105         }
106
107      // '1' is equivalent to specifying no qValue. If there's no extensions, then we won't include a qValue.
108      if (qValue.floatValue() == 1.0) {
109         if (! extensions.isEmpty()) {
110            sb.append(";q=").append(qValue);
111            for (Entry<String,Set<String>> e : extensions.entrySet()) {
112               String k = e.getKey();
113               for (String v : e.getValue())
114                  sb.append(';').append(k).append('=').append(v);
115            }
116         }
117      } else {
118         sb.append(";q=").append(qValue);
119         for (Entry<String,Set<String>> e : extensions.entrySet()) {
120            String k = e.getKey();
121            for (String v : e.getValue())
122               sb.append(';').append(k).append('=').append(v);
123         }
124      }
125      return sb.toString();
126   }
127
128   /**
129    * Returns <jk>true</jk> if the specified object is also a <code>MediaType</code>, and has the same qValue, type, parameters, and extensions.
130    *
131    * @return <jk>true</jk> if object is equivalent.
132    */
133   @Override /* Object */
134   public boolean equals(Object o) {
135
136      if (o == null || !(o instanceof MediaRange))
137         return false;
138
139      if (this == o)
140         return true;
141
142      MediaRange o2 = (MediaRange) o;
143      return qValue.equals(o2.qValue)
144         && type.equals(o2.type)
145         && subType.equals(o2.subType)
146         && parameters.equals(o2.parameters)
147         && extensions.equals(o2.extensions);
148   }
149
150   /**
151    * Returns a hash based on this instance's <code>media-type</code>.
152    *
153    * @return A hash based on this instance's <code>media-type</code>.
154    */
155   @Override /* Object */
156   public int hashCode() {
157      return type.hashCode() + subType.hashCode();
158   }
159
160   /**
161    * Creates a <code>MediaRange</code> object with the referenced values.
162    *
163    * @param type The MIME type of this media range (e.g. <js>"application"</js> in <js>"application/json"</js>)
164    * @param subType The MIME subtype of this media range (e.g. <js>"json"</js> in <js>"application/json"</js>).
165    * @param parameters The optional parameters for this range.
166    * @param qValue The quality value of this range.  Must be between <code>0</code> and <code>1.0</code>.
167    * @param extensions The optional extensions to this quality value.
168    */
169   private MediaRange(String type, String subType, Map<String,Set<String>> parameters, Float qValue, Map<String,Set<String>> extensions) {
170      this.type = type;
171      this.subType = subType;
172      this.parameters = (parameters == null ? new TreeMap<String,Set<String>>() : parameters);
173      this.extensions = (extensions == null ? new TreeMap<String,Set<String>>() : extensions);
174      this.qValue = qValue;
175   }
176
177   /**
178    * Parses an <code>Accept</code> header value into an array of media ranges.
179    * <p>
180    * The returned media ranges are sorted such that the most acceptable media is available at ordinal position <js>'0'</js>, and the least acceptable at position n-1.
181    * <p>
182    * The syntax expected to be found in the referenced <code>value</code> complies with the syntax described in RFC2616, Section 14.1, as described below:
183    * <p class='bcode'>
184    *    Accept         = "Accept" ":"
185    *                      #( media-range [ accept-params ] )
186    *
187    *    media-range    = ( "*\/*"
188    *                      | ( type "/" "*" )
189    *                      | ( type "/" subtype )
190    *                      ) *( ";" parameter )
191    *    accept-params  = ";" "q" "=" qvalue *( accept-extension )
192    *    accept-extension = ";" token [ "=" ( token | quoted-string ) ]
193    * </p>
194    * This method can also be used on other headers such as <code>Accept-Charset</code> and <code>Accept-Encoding</code>...
195    * <p class='bcode'>
196    *    Accept-Charset = "Accept-Charset" ":"
197    *    1#( ( charset | "*" )[ ";" "q" "=" qvalue ] )
198    * </p>
199    *
200    * @param value The value to parse.  If <jk>null</jk> or empty, returns a single <code>MediaRange</code> is returned that represents all types.
201    * @return The media ranges described by the string.
202    *    The ranges are sorted such that the most acceptable media is available at ordinal position <js>'0'</js>, and the least acceptable at position n-1.
203    */
204   public static MediaRange[] parse(String value) {
205
206      Set<MediaRange> ranges = new TreeSet<MediaRange>();
207
208      if (value == null || value.length() == 0)
209         return new MediaRange[]{new MediaRange("*", "*", null, 1f, null)};
210
211      value = value.toLowerCase(Locale.ENGLISH);
212
213      for (String r : value.trim().split("\\s*,\\s*")) {
214         r = r.trim();
215
216         if (r.isEmpty())
217            continue;
218
219         String[] tokens = r.split("\\s*;\\s*");
220
221         tokens[0] = tokens[0].replace(' ', '+');
222
223         // There is at least a type.
224         String[] t = tokens[0].split("/");
225         String type = t[0], subType = (t.length == 1 ? "*" : t[1]);
226
227         // Only the type of the range is specified
228         if (tokens.length == 1) {
229            ranges.add(new MediaRange(type, subType, null, 1f, null));
230            continue;
231         }
232
233         Float qValue = 1f;
234         Map<String,Set<String>> params = new TreeMap<String,Set<String>>();
235         Map<String,Set<String>> exts = new TreeMap<String,Set<String>>();
236
237         boolean isInExtensions = false;
238         for (int i = 1; i < tokens.length; i++) {
239            String[] parm = tokens[i].split("\\s*=\\s*");
240            if (parm.length == 2) {
241               String k = parm[0], v = parm[1];
242               if (isInExtensions) {
243                  if (! exts.containsKey(parm[0]))
244                     exts.put(parm[0], new TreeSet<String>());
245                  exts.get(parm[0]).add(parm[1]);
246               } else if (k.equals("q")) {
247                  qValue = new Float(v);
248                  isInExtensions = true;
249               } else /*(! isInExtensions)*/ {
250                  if (! params.containsKey(parm[0]))
251                     params.put(parm[0], new TreeSet<String>());
252                  params.get(parm[0]).add(parm[1]);
253               }
254            }
255         }
256
257         ranges.add(new MediaRange(type, subType, params, qValue, exts));
258      }
259
260      return ranges.toArray(new MediaRange[ranges.size()]);
261   }
262
263   /**
264    * Compares two MediaRanges for equality.
265    * <p>
266    * The values are first compared according to <code>qValue</code> values.
267    * Should those values be equal, the <code>type</code> is then lexicographically compared (case-insensitive) in ascending order,
268    *    with the <js>"*"</js> type demoted last in that order.
269    * <code>MediaRanges</code> with the same type but different sub-types are compared - a more specific subtype is
270    *    promoted over the 'wildcard' subtype.
271    * <code>MediaRanges</code> with the same types but with extensions are promoted over those same types with no extensions.
272    *
273    * @param o The range to compare to.  Never <jk>null</jk>.
274    */
275   @Override /* Comparable */
276   public int compareTo(MediaRange o) {
277
278      // Compare q-values.
279      int qCompare = Float.compare(o.qValue, qValue);
280      if (qCompare != 0)
281         return qCompare;
282
283      // Compare media-types.
284      // Note that '*' comes alphabetically before letters, so just do a reverse-alphabetical comparison.
285      int i = o.type.compareTo(type);
286      if (i == 0)
287         i = o.subType.compareTo(subType);
288      return i;
289   }
290
291   /**
292    * Returns <jk>true</jk> if the specified <code>MediaRange</code> matches this range.
293    * <p>
294    * This implies the types and subtypes are the same as or encompasses the other (e.g. <js>'application/xml'</js> and <js>'application/*'</js>).
295    *
296    * @param o The other media rage.
297    * @return <jk>true</jk> if the media ranges are the same or one encompasses the other.
298    */
299   public boolean matches(MediaRange o) {
300      if (this == o)
301         return true;
302
303      if (qValue == 0 || o.qValue == 0)
304         return false;
305
306      if (type.equals(o.type) || (type.equals("*")) || (o.type.equals("*")))
307         if (subType.equals(o.subType) || subType.equals("*") || o.subType.equals("*"))
308            return true;
309
310      return false;
311   }
