032/**
033 * A wrapper class around the {@link Class} object that provides cached information
034 * about that class.
035 *
036 * <p>
037 *    Instances of this class can be created through the {@link BeanContext#getClassMeta(Class)} method.
038 * <p>
039 *    The {@link BeanContext} class will cache and reuse instances of this class except for the following class types:
040 * <ul>
041 *    <li>Arrays
042 *    <li>Maps with non-Object key/values.
043 *    <li>Collections with non-Object key/values.
044 * </ul>
045 * <p>
046 *    This class is tied to the {@link BeanContext} class because it's that class that makes the determination
047 *    of what is a bean.
048 *
049 * @author James Bognar (jbognar@us.ibm.com)
050 * @param <T> The class type of the wrapped class.
051 */
052@Bean(properties={"innerClass","classCategory","elementType","keyType","valueType","notABeanReason","initException","beanMeta"})
053public final class ClassMeta<T> implements Type {
054
055   /** Class categories. */
056   enum ClassCategory {
057      MAP, COLLECTION, NUMBER, DECIMAL, BOOLEAN, CHAR, DATE, ARRAY, ENUM, BEAN, UNKNOWN, OTHER, CHARSEQ, STR, OBJ, URI, BEANMAP, READER, INPUTSTREAM
058   }
059
060   final BeanContext beanContext;                    // The bean context that created this object.
061   ClassCategory classCategory = UNKNOWN;            // The class category.
062   final Class<T> innerClass;                        // The class being wrapped.
063   ClassMeta<?>
064      filteredClassMeta,                             // The filtered class type (in class has filter associated with it.
065      elementType = null,                            // If ARRAY or COLLECTION, the element class type.
066      keyType = null,                                // If MAP, the key class type.
067      valueType = null;                              // If MAP, the value class type.
068   InvocationHandler invocationHandler;              // The invocation handler for this class (if it has one).
069   volatile BeanMeta<T> beanMeta;                    // The bean meta for this bean class (if it's a bean).
070   Method valueOfMethod;                             // The static valueOf(String) or fromString(String) method (if it has one).
071   Constructor<? extends T> noArgConstructor;        // The no-arg constructor for this class (if it has one).
072   Constructor<T> stringConstructor;                 // The X(String) constructor (if it has one).
073   Constructor<T> objectMapConstructor;              // The X(ObjectMap) constructor (if it has one).
074   Method toObjectMapMethod;                         // The toObjectMap() method (if it has one).
075   Method namePropertyMethod;                        // The method to set the name on an object (if it has one).
076   Method parentPropertyMethod;                      // The method to set the parent on an object (if it has one).
077   String notABeanReason;                            // If this isn't a bean, the reason why.
078   PojoFilter<T,?> pojoFilter;                       // The object filter associated with this bean (if it has one).
079   BeanFilter<? extends T> beanFilter;               // The bean filter associated with this bean (if it has one).
080   boolean
081      isDelegate,                                    // True if this class extends Delegate.
082      isAbstract,                                    // True if this class is abstract.
083      isMemberClass;                                 // True if this is a non-static member class.
084
085   private XmlClassMeta xmlMeta;                      // Class-related metadata from the @Xml annotation found on this class or parent class.
086   private JsonClassMeta jsonMeta;                    // Class-related metadata from the @Json annotation found on this class or parent class.
087   private HtmlClassMeta htmlMeta;                    // Class-related metadata from the @Html annotation found on this class or parent class.
088   private UrlEncodingClassMeta urlEncodingMeta;      // Class-related metadata from the @UrlEncoding annotation found on this class or parent class.
089   private RdfClassMeta rdfMeta;                      // Class-related metadata from the @Rdf annotation found on this class or parent class.
090
091   private Throwable initException;                   // Any exceptions thrown in the init() method.
092   private boolean hasChildPojoFilters;               // True if this class or any subclass of this class has a PojoFilter associated with it.
093   private Object primitiveDefault;                   // Default value for primitive type classes.
094   private Map<String,Method> remoteableMethods,      // Methods annotated with @Remoteable.  Contains all public methods if class is annotated with @Remotable.
095      publicMethods;                                 // All public methods, including static methods.
096
097   private static final Boolean BOOLEAN_DEFAULT = false;
098   private static final Character CHARACTER_DEFAULT = (char)0;
099   private static final Short SHORT_DEFAULT = (short)0;
100   private static final Integer INTEGER_DEFAULT = 0;
101   private static final Long LONG_DEFAULT = 0l;
102   private static final Float FLOAT_DEFAULT = 0f;
103   private static final Double DOUBLE_DEFAULT = 0d;
104   private static final Byte BYTE_DEFAULT = (byte)0;

129   @SuppressWarnings({ "unchecked", "rawtypes" })
130   ClassMeta init() {
131
132      try {
133         Filter filter = findFilter(beanContext);
134         if (filter != null) {
135            if (filter.getType() == Filter.FilterType.BEAN)
136               beanFilter = (BeanFilter)filter;
137            else
138               pojoFilter = (PojoFilter)filter;
139            filteredClassMeta = (pojoFilter == null ? this : beanContext.getClassMeta(pojoFilter.getFilteredClass()));
140         }
141         if (filteredClassMeta == null)
142            filteredClassMeta = this;
143
144         if (innerClass != Object.class) {
145            this.noArgConstructor = beanContext.getImplClassConstructor(innerClass, Visibility.PUBLIC);
146            if (noArgConstructor == null)
147               noArgConstructor = findNoArgConstructor(innerClass, Visibility.PUBLIC);
148         }
149
150         this.hasChildPojoFilters = beanContext.hasChildPojoFilters(innerClass);
151
152         this.xmlMeta = new XmlClassMeta(innerClass);
153         this.jsonMeta = new JsonClassMeta(innerClass);
154         this.htmlMeta = new HtmlClassMeta(innerClass);
155         this.urlEncodingMeta = new UrlEncodingClassMeta(innerClass);
156         this.rdfMeta = new RdfClassMeta(innerClass);
157
158         Class c = innerClass;
159
160         if (c.isPrimitive()) {
161            if (c == Boolean.TYPE)
162               classCategory = BOOLEAN;
163            else if (c == Byte.TYPE || c == Short.TYPE || c == Integer.TYPE || c == Long.TYPE || c == Float.TYPE || c == Double.TYPE) {
164               if (c == Float.TYPE || c == Double.TYPE)
165                  classCategory = DECIMAL;
166               else
167                  classCategory = NUMBER;
168            }
169            else if (c == Character.TYPE)
170               classCategory = CHAR;
171         } else {
172            if (isParentClass(Delegate.class, c))
173               isDelegate = true;
174            if (c == Object.class)
175               classCategory = OBJ;
176            else if (c.isEnum())
177               classCategory = ENUM;
178            else if (isParentClass(CharSequence.class, c)) {
179               if (c.equals(String.class))
180                  classCategory = STR;
181               else
182                  classCategory = CHARSEQ;
183            }
184            else if (isParentClass(Number.class, c)) {
185               if (isParentClass(Float.class, c) || isParentClass(Double.class, c))
186                  classCategory = DECIMAL;
187               else
188                  classCategory = NUMBER;
189            }
190            else if (isParentClass(Collection.class, c))
191               classCategory = COLLECTION;
192            else if (isParentClass(Map.class, c)) {
193               if (isParentClass(BeanMap.class, c))
194                  classCategory = BEANMAP;
195               else
196                  classCategory = MAP;
197            }
198            else if (c == Character.class)
199               classCategory = CHAR;
200            else if (c == Boolean.class)
201               classCategory = BOOLEAN;
202            else if (isParentClass(Date.class, c) || isParentClass(Calendar.class, c))
203               classCategory = DATE;
204            else if (c.isArray())
205               classCategory = ARRAY;
206            else if (isParentClass(URL.class, c) || isParentClass(URI.class, c) || c.isAnnotationPresent(com.ibm.juno.core.annotation.URI.class))
207               classCategory = URI;
208            else if (isParentClass(Reader.class, c))
209               classCategory = READER;
210            else if (isParentClass(InputStream.class, c))
211               classCategory = INPUTSTREAM;
212         }
213
214         isMemberClass = c.isMemberClass() && ! Modifier.isStatic(c.getModifiers());
215
216         // Find static fromString(String) or equivalent method.
217         // fromString() must be checked before valueOf() so that Enum classes can create their own
218         //    specialized fromString() methods to override the behavior of Enum.valueOf(String).
219         // valueOf() is used by enums.
220         // parse() is used by the java logging Level class.
221         // forName() is used by Class and Charset
222         for (String methodName : new String[]{"fromString","valueOf","parse","parseString","forName"}) {
223            if (this.valueOfMethod == null) {
224               for (Method m : c.getMethods()) {
225                  int mod = m.getModifiers();
226                  if (Modifier.isStatic(mod) && Modifier.isPublic(mod)) {
227                     String mName = m.getName();
228                     if (mName.equals(methodName) && m.getReturnType() == innerClass) {
229                        Class<?>[] args = m.getParameterTypes();
230                        if (args.length == 1 && args[0] == String.class) {
231                           this.valueOfMethod = m;
232                           break;
233                        }
234                     }
235                  }
236               }
237            }
238         }
239
240         // Find toObjectMap() method if present.
241         for (Method m : c.getMethods()) {
242            int mod = m.getModifiers();
243            if (Modifier.isPublic(mod) && ! Modifier.isStatic(mod)) {
244               String mName = m.getName();
245               if (mName.equals("toObjectMap")) {
246                  if (m.getParameterTypes().length == 0 && m.getReturnType() == ObjectMap.class) {
247                     this.toObjectMapMethod = m;
248                     break;
249                  }
250               }
251            }
252         }
253
254         // Find @NameProperty and @ParentProperty methods if present.
255         for (Method m : c.getDeclaredMethods()) {
256            if (m.isAnnotationPresent(ParentProperty.class) && m.getParameterTypes().length == 1) {
257               m.setAccessible(true);
258               parentPropertyMethod = m;
259            }
260            if (m.isAnnotationPresent(NameProperty.class) && m.getParameterTypes().length == 1) {
261               m.setAccessible(true);
262               namePropertyMethod = m;
263            }
264         }
265
266         // Find constructor(String) method if present.
267         for (Constructor cs : c.getConstructors()) {
268            int mod = cs.getModifiers();
269            if (Modifier.isPublic(mod)) {
270               Class<?>[] args = cs.getParameterTypes();
271               if (args.length == (isMemberClass ? 2 : 1)) {
272                  Class<?> arg = args[(isMemberClass ? 1 : 0)];
273                  if (arg == String.class)
274                     this.stringConstructor = cs;
275                  else if (ObjectMap.class.isAssignableFrom(arg))
276                     this.objectMapConstructor = cs;
277               }
278            }
279         }
280
281         // Note:  Primitive types are normally abstract.
282         isAbstract = Modifier.isAbstract(c.getModifiers()) && ! isPrimitive();
283
284         // If this is an array, get the element type.
285         if (classCategory == ARRAY)
286            elementType = beanContext.getClassMeta(innerClass.getComponentType());
287
288         // If this is a MAP, see if it's parameterized (e.g. AddressBook extends HashMap<String,Person>)
289         else if (classCategory == MAP) {
290            ClassMeta[] parameters = beanContext.findParameters(innerClass, innerClass);
291            if (parameters != null && parameters.length == 2) {
292               keyType = parameters[0];
293               valueType = parameters[1];
294            } else {
295               keyType = beanContext.getClassMeta(Object.class);
296               valueType = beanContext.getClassMeta(Object.class);
297            }
298         }
299
300         // If this is a COLLECTION, see if it's parameterized (e.g. AddressBook extends LinkedList<Person>)
301         else if (classCategory == COLLECTION) {
302            ClassMeta[] parameters = beanContext.findParameters(innerClass, innerClass);
303            if (parameters != null && parameters.length == 1) {
304               elementType = parameters[0];
305            } else {
306               elementType = beanContext.getClassMeta(Object.class);
307            }
308         }
309
310         // If the category is unknown, see if it's a bean.
311         // Note that this needs to be done after all other initialization has been done.
312         else if (classCategory == UNKNOWN) {
313
314            BeanMeta newMeta = new BeanMeta(this, beanContext, beanFilter);
315            try {
316               notABeanReason = newMeta.init();
317            } catch (RuntimeException e) {
318               notABeanReason = e.getMessage();
319               throw e;
320            }
321            if (notABeanReason != null)
322               classCategory = OTHER;
323            else {
324               beanMeta = newMeta;
325               classCategory = BEAN;
326            }
327         }
328
329         if (c.isPrimitive()) {
330            if (c == Boolean.TYPE)
331               primitiveDefault = BOOLEAN_DEFAULT;
332            else if (c == Character.TYPE)
333               primitiveDefault = CHARACTER_DEFAULT;
334            else if (c == Short.TYPE)
335               primitiveDefault = SHORT_DEFAULT;
336            else if (c == Integer.TYPE)
337               primitiveDefault = INTEGER_DEFAULT;
338            else if (c == Long.TYPE)
339               primitiveDefault = LONG_DEFAULT;
340            else if (c == Float.TYPE)
341               primitiveDefault = FLOAT_DEFAULT;
342            else if (c == Double.TYPE)
343               primitiveDefault = DOUBLE_DEFAULT;
344            else if (c == Byte.TYPE)
345               primitiveDefault = BYTE_DEFAULT;
346         } else {
347            if (c == Boolean.class)
348               primitiveDefault = BOOLEAN_DEFAULT;
349            else if (c == Character.class)
350               primitiveDefault = CHARACTER_DEFAULT;
351            else if (c == Short.class)
352               primitiveDefault = SHORT_DEFAULT;
353            else if (c == Integer.class)
354               primitiveDefault = INTEGER_DEFAULT;
355            else if (c == Long.class)
356               primitiveDefault = LONG_DEFAULT;
357            else if (c == Float.class)
358               primitiveDefault = FLOAT_DEFAULT;
359            else if (c == Double.class)
360               primitiveDefault = DOUBLE_DEFAULT;
361            else if (c == Byte.class)
362               primitiveDefault = BYTE_DEFAULT;
363         }
364      } catch (NoClassDefFoundError e) {
365         this.initException = e;
366      } catch (RuntimeException e) {
367         this.initException = e;
368         throw e;
369      }
370
371      if (innerClass.getAnnotation(Remoteable.class) != null) {
372         remoteableMethods = getPublicMethods();
373      } else {
374         for (Method m : innerClass.getMethods()) {
375            if (m.getAnnotation(Remoteable.class) != null) {
376               if (remoteableMethods == null)
377                  remoteableMethods = new LinkedHashMap<String,Method>();
378               remoteableMethods.put(ClassUtils.getMethodSignature(m), m);
379            }
380         }
381      }
382      if (remoteableMethods != null)
383         remoteableMethods = Collections.unmodifiableMap(remoteableMethods);
384
385      return this;
386   }

438   private Filter findFilter(BeanContext context) {
439      try {
440         com.ibm.juno.core.annotation.Filter b = innerClass.getAnnotation(com.ibm.juno.core.annotation.Filter.class);
441         if (b != null) {
442            Class<? extends Filter> c = b.value();
443            if (c != Filter.NULL.class) {
444               Filter f = c.newInstance();
445               f.setBeanContext(context);
446               return f;
447            }
448         }
449         if (context == null)
450            return null;
451         Filter f = context.findBeanFilter(innerClass);
452         if (f != null)
453            return f;
454         f = context.findPojoFilter(innerClass);
455         if (f != null)
456            return f;
457         List<Bean> ba = ReflectionUtils.findAnnotations(Bean.class, innerClass);
458         if (! ba.isEmpty())
459            f = new AnnotationBeanFilter<T>(innerClass, ba);
460         return f;
461      } catch (Exception e) {
462         throw new RuntimeException(e);
463      }
464   }

466   /**
467    * Locates the no-arg constructor for the specified class.
468    * Constructor must match the visibility requirements specified by parameter 'v'.
469    * If class is abstract, always returns <jk>null</jk>.
470    * Note that this also returns the 1-arg constructor for non-static member classes.
471    *
472    * @param c The class from which to locate the no-arg constructor.
473    * @param v The minimum visibility.
474    * @return The constructor, or <jk>null</jk> if no no-arg constructor exists with the required visibility.
475    */
476   @SuppressWarnings({"rawtypes","unchecked"})
477   protected static <T> Constructor<? extends T> findNoArgConstructor(Class<T> c, Visibility v) {
478      int mod = c.getModifiers();
479      if (Modifier.isAbstract(mod))
480         return null;
481      boolean isMemberClass = c.isMemberClass() && ! Modifier.isStatic(mod);
482      for (Constructor cc : c.getConstructors()) {
483         mod = cc.getModifiers();
484         if (cc.getParameterTypes().length == (isMemberClass ? 1 : 0) && v.isVisible(mod))
485            return v.filter(cc);
486      }
487      return null;
488   }

1201   /**
1202    * Appends this object as a readable string to the specified string builder.
1203    *
1204    * @param sb The string builder to append this object to.
1205    * @param simple Print simple class names only (no package).
1206    * @return The same string builder passed in (for method chaining).
1207    */
1208   protected StringBuilder toString(StringBuilder sb, boolean simple) {
1209      String name = innerClass.getName();
1210      if (simple) {
1211         int i = name.lastIndexOf('.');
1212         name = name.substring(i == -1 ? 0 : i+1).replace('$', '.');
1213      }
1214      switch(classCategory) {
1215         case ARRAY:
1216            return elementType.toString(sb, simple).append('[').append(']');
1217         case MAP:
1218            return sb.append(name).append(keyType.isObject() && valueType.isObject() ? "" : "<"+keyType.toString(simple)+","+valueType.toString(simple)+">");
1219         case BEANMAP:
1220            return sb.append(BeanMap.class.getName()).append('<').append(name).append('>');
1221         case COLLECTION:
1222            return sb.append(name).append(elementType.isObject() ? "" : "<"+elementType.toString(simple)+">");
1223         case OTHER:
1224            if (simple)
1225               return sb.append(name);
1226            sb.append("OTHER-").append(name).append(",notABeanReason=").append(notABeanReason);
1227            if (initException != null)
1228               sb.append(",initException=").append(initException);
1229            return sb;
1230         default:
1231            return sb.append(name);
1232      }
1233   }
