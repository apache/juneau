009package com.ibm.juno.core.urlencoding;
010
011import java.io.*;
012
013import com.ibm.juno.core.parser.*;
014
015/**
016 * Same functionality as {@link ParserReader} except automatically decoded <code>%xx</code> escape sequences.
017 * <p>
018 * Escape sequences are assumed to be encoded UTF-8.  Extended Unicode (&gt;\u10000) is supported.
019 * <p>
020 * If decoding is enabled, the following character replacements occur so that boundaries are not lost:
021 * <ul>
022 *    <li><js>'&'</js> -&gt; <js>'\u0001'</js>
023 *    <li><js>'='</js> -&gt; <js>'\u0002'</js>
024 * </ul>
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028public final class UonParserReader extends ParserReader {
029
030   private final boolean decodeChars;
031   private final char[] buff;
032   private int iCurrent, iEnd;
033
034   /**
035    * Constructor for input from a {@link CharSequence}.
036    *
037    * @param in The character sequence being read from.
038    * @param decodeChars If <jk>true</jk>, decode <code>%xx</code> escape sequences.
039    */
040   public UonParserReader(CharSequence in, boolean decodeChars) {
041      super(in);
042      this.decodeChars = decodeChars;
043      if (in == null || ! decodeChars)
044         this.buff = new char[0];
045      else
046         this.buff = new char[in.length() < 1024 ? in.length() : 1024];
047   }
048
049   /**
050    * Constructor for input from a {@link Reader}).
051    *
052    * @param r The Reader being wrapped.
053    * @param buffSize Buffer size.
054    * @param decodeChars If <jk>true</jk>, decode <code>%xx</code> escape sequences.
055    */
056   public UonParserReader(Reader r, int buffSize, boolean decodeChars) {
057      super(r, buffSize);
058      this.decodeChars = decodeChars;
059      buffSize = decodeChars ? (buffSize <= 0 ? 1024 : Math.max(buffSize, 20)) : 0;
060      this.buff = new char[buffSize];
061   }
062
063   @Override /* Reader */
064   public final int read(char[] cbuf, int off, int len) throws IOException {
065
066      if (! decodeChars)
067         return super.read(cbuf, off, len);
068
069      // Copy any remainder to the beginning of the buffer.
070      int remainder = iEnd - iCurrent;
071      if (remainder > 0)
072         System.arraycopy(buff, iCurrent, buff, 0, remainder);
073      iCurrent = 0;
074
075      int expected = buff.length - remainder;
076
077      int x = super.read(buff, remainder, expected);
078      if (x == -1 && remainder == 0)
079         return -1;
080
081      iEnd = remainder + (x == -1 ? 0 : x);
082
083      int i = 0;
084      while (i < len) {
085         if (iCurrent >= iEnd)
086            return i;
087         char c = buff[iCurrent++];
088         if (c == '+') {
089            cbuf[off + i++] = ' ';
090         } else if (c == '&') {
091            cbuf[off + i++] = '\u0001';
092         } else if (c == '=') {
093            cbuf[off + i++] = '\u0002';
094         } else if (c != '%') {
095            cbuf[off + i++] = c;
096         } else {
097            int iMark = iCurrent-1;  // Keep track of current position.
098
099            // Stop if there aren't at least two more characters following '%' in the buffer,
100            // or there aren't at least two more positions open in cbuf to handle double-char chars.
101            if (iMark+2 >= iEnd || i+2 > len) {
102               iCurrent--;
103               return i;
104            }
105
106            int b0 = readEncodedByte();
107            int cx;
108
109            // 0xxxxxxx
110            if (b0 < 128) {
111               cx = b0;
112
113            // 10xxxxxx
114            } else if (b0 < 192) {
115               throw new IOException("Invalid hex value for first escape pattern in UTF-8 sequence:  " + b0);
116
117            // 110xxxxx 10xxxxxx
118            // 11000000(192) - 11011111(223)
119            } else if (b0 < 224) {
120               cx = readUTF8(b0-192, 1);
121               if (cx == -1) {
122                  iCurrent = iMark;
123                  return i;
124               }
125
126            // 1110xxxx 10xxxxxx 10xxxxxx
127            // 11100000(224) - 11101111(239)
128            } else if (b0 < 240) {
129               cx = readUTF8(b0-224, 2);
130               if (cx == -1) {
131                  iCurrent = iMark;
132                  return i;
133               }
134
135            // 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
136            // 11110000(240) - 11110111(247)
137            } else if (b0 < 248) {
138               cx = readUTF8(b0-240, 3);
139               if (cx == -1) {
140                  iCurrent = iMark;
141                  return i;
142               }
143
144            } else
145               throw new IOException("Invalid hex value for first escape pattern in UTF-8 sequence:  " + b0);
146
147            if (cx < 0x10000)
148               cbuf[off + i++] = (char)cx;
149            else {
150               cx -= 0x10000;
151               cbuf[off + i++] = (char)(0xd800 + (cx >> 10));
152               cbuf[off + i++] = (char)(0xdc00 + (cx & 0x3ff));
153            }
154         }
155      }
156      return i;
157   }
158
159   private final int readUTF8(int n, final int numBytes) throws IOException {
160      if (iCurrent + numBytes*3 > iEnd)
161         return -1;
162      for (int i = 0; i < numBytes; i++) {
163         n <<= 6;
164         n += readHex()-128;
165      }
166      return n;
167   }
168
169   private final int readHex() throws IOException {
170      int c = buff[iCurrent++];
171      if (c != '%')
172         throw new IOException("Did not find expected '%' character in UTF-8 sequence.");
173      return readEncodedByte();
174   }
175
176   private final int readEncodedByte() throws IOException {
177      if (iEnd <= iCurrent + 1)
178         throw new IOException("Incomplete trailing escape pattern");
179      int h = buff[iCurrent++];
180      int l = buff[iCurrent++];
181      h = fromHexChar(h);
182      l = fromHexChar(l);
183      return (h << 4) + l;
184   }
185
186   private final int fromHexChar(int c) throws IOException {
187      if (c >= '0' && c <= '9')
188         return c - '0';
189      if (c >= 'a' && c <= 'f')
190         return 10 + c - 'a';
191      if (c >= 'A' && c <= 'F')
192         return 10 + c - 'A';
193      throw new IOException("Invalid hex character '"+c+"' found in escape pattern.");
194   }
195}
