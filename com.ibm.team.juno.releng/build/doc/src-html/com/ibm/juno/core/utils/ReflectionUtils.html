009package com.ibm.juno.core.utils;
010
011import static com.ibm.juno.core.utils.CollectionUtils.*;
012
013import java.io.*;
014import java.lang.annotation.*;
015import java.util.*;
016
017/**
018 * Reflection utilities.
019 *
020 * @author James Bognar (jbognar@us.ibm.com)
021 */
022public final class ReflectionUtils {
023
024   /**
025    * Similar to {@link Class#getAnnotation(Class)} except also searches annotations on interfaces.
026    *
027    * @param <T> The annotation class type.
028    * @param a The annotation class.
029    * @param c The annotated class.
030    * @return The annotation, or <jk>null</jk> if not found.
031    */
032   public static <T extends Annotation> T getAnnotation(Class<T> a, Class<?> c) {
033      if (c == null)
034         return null;
035
036      T t = getDeclaredAnnotation(a, c);
037      if (t != null)
038         return t;
039
040      t = getAnnotation(a, c.getSuperclass());
041      if (t != null)
042         return t;
043
044      for (Class<?> c2 : c.getInterfaces()) {
045         t = getAnnotation(a, c2);
046         if (t != null)
047            return t;
048      }
049      return null;
050   }
051
052   /**
053    * Returns the specified annotation only if it's been declared on the specified class.
054    * <p>
055    *    More efficient than calling {@link Class#getAnnotation(Class)} since it doesn't
056    *    recursively look for the class up the parent chain.
057    *
058    * @param <T> The annotation class type.
059    * @param a The annotation class.
060    * @param c The annotated class.
061    * @return The annotation, or <jk>null</jk> if not found.
062    */
063   @SuppressWarnings("unchecked")
064   public static <T extends Annotation> T getDeclaredAnnotation(Class<T> a, Class<?> c) {
065      for (Annotation a2 : c.getDeclaredAnnotations())
066         if (a2.annotationType() == a)
067            return (T)a2;
068      return null;
069   }
070
071   /**
072    * Returns all instances of the specified annotation on the specified class.
073    * <p>
074    * Searches all superclasses and superinterfaces.
075    * <p>
076    * Results are ordered child-to-parent.
077    *
078    * @param <T> The annotation class type.
079    * @param a The annotation class type.
080    * @param c The class being searched.
081    * @return The found matches, or an empty array if annotation was not found.
082    */
083   public static <T extends Annotation> List<T> findAnnotations(Class<T> a, Class<?> c) {
084      List<T> l = new LinkedList<T>();
085      appendAnnotations(a, c, l);
086      return l;
087   }
088
089   /**
090    * Sames as {@link #findAnnotations(Class, Class)} except returns the annotations as a map
091    * with the keys being the class on which the annotation was found.
092    * <p>
093    * Results are ordered child-to-parent.
094    *
095    * @param <T> The annotation class type.
096    * @param a The annotation class type.
097    * @param c The class being searched.
098    * @return The found matches, or an empty array if annotation was not found.
099    */
100   public static <T extends Annotation> LinkedHashMap<Class<?>,T> findAnnotationsMap(Class<T> a, Class<?> c) {
101      LinkedHashMap<Class<?>,T> m = new LinkedHashMap<Class<?>,T>();
102      findAnnotationsMap(a, c, m);
103      return m;
104   }
105
106   private static <T extends Annotation> void findAnnotationsMap(Class<T> a, Class<?> c, Map<Class<?>,T> m) {
107      if (c == null)
108         return;
109
110      T t = getDeclaredAnnotation(a, c);
111      if (t != null)
112         m.put(c, t);
113
114      findAnnotationsMap(a, c.getSuperclass(), m);
115
116      for (Class<?> c2 : c.getInterfaces())
117         findAnnotationsMap(a, c2, m);
118   }
119
120   /**
121    * Finds and appends the specified annotation on the specified class and superclasses/interfaces to the specified list.
122    *
123    * @param a The annotation.
124    * @param c The class.
125    * @param l The list of annotations.
126    */
127   public static <T extends Annotation> void appendAnnotations(Class<T> a, Class<?> c, List<T> l) {
128      if (c == null)
129         return;
130
131      addIfNotNull(l, getDeclaredAnnotation(a, c));
132
133      if (c.getPackage() != null)
134         addIfNotNull(l, c.getPackage().getAnnotation(a));
135
136      appendAnnotations(a, c.getSuperclass(), l);
137
138      for (Class<?> c2 : c.getInterfaces())
139         appendAnnotations(a, c2, l);
140   }
141
142   /**
143    * Similar to {@link Class#getResourceAsStream(String)} except looks up the
144    * parent hierarchy for the existence of the specified resource.
145    *
146    * @param c The class to return the resource on.
147    * @param name The resource name.
148    * @return An input stream on the specified resource, or <jk>null</jk> if the resource could not be found.
149    */
150   public static InputStream getResource(Class<?> c, String name) {
151      while (c != null) {
152         InputStream is = c.getResourceAsStream(name);
153         if (is != null)
154            return is;
155         c = c.getSuperclass();
156      }
157      return null;
158   }
159}
