009package com.ibm.juno.core.urlencoding;
010
011import com.ibm.juno.core.*;
012
013/**
014 * Configurable properties on the {@link UrlEncodingSerializer} and {@link UrlEncodingParser} classes.
015 * <p>
016 *    Use the {@link UrlEncodingSerializer#setProperty(String, Object)} and
017 *    {@link UrlEncodingParser#setProperty(String, Object)} methods to set property values.
018 *
019 * @author James Bognar (jbognar@us.ibm.com)
020 */
021public final class UrlEncodingProperties implements Cloneable {
022
023   /**
024    * Serialize bean property collections/arrays as separate key/value pairs ({@link Boolean}, default=<jk>false</jk>).
025    * <p>
026    *    If <jk>false</jk>, serializing the array <code>[1,2,3]</code> results in <code>?key=$a(1,2,3)</code>.
027    *    If <jk>true</jk>, serializing the same array results in <code>?key=1&key=2&key=3</code>.
028    * <p>
029    *    Example:
030    * <p class='bcode'>
031    *    <jk>public class</jk> A {
032    *       <jk>public</jk> String[] f1 = {<js>"a"</js>,<js>"b"</js>};
033    *       <jk>public</jk> List&lt;String&gt; f2 = <jk>new</jk> LinkedList&lt;String&gt;(Arrays.<jsm>asList</jsm>(<jk>new</jk> String[]{<js>"c"</js>,<js>"d"</js>}));
034    *    }
035    *
036    *    UrlEncodingSerializer s1 = <jk>new</jk> UrlEncodingParser();
037    *    UrlEncodingSerializer s2 = <jk>new</jk> UrlEncodingParser().setProperty(UrlEncodingProperties.<jsf>URLENC_expandedParams</jsf>, <jk>true</jk>);
038    *
039    *    String s1 = p1.serialize(<jk>new</jk> A()); <jc>// Produces "f1=(a,b)&f2=(c,d)"</jc>
040    *    String s2 = p2.serialize(<jk>new</jk> A()); <jc>// Produces "f1=a&f1=b&f2=c&f2=d"</jc>
041    * </p>
042    * <p>
043    *    <b>Important note:</b>  If parsing multi-part parameters, it's highly recommended to use Collections or Lists
044    *    as bean property types instead of arrays since arrays have to be recreated from scratch every time a value
045    *    is added to it.
046    * <p>
047    *    This option only applies to beans.
048    */
049   public static final String URLENC_expandedParams = "UrlEncoding.expandedParams";
050
051   boolean
052      expandedParams = false;
053
054   /**
055    * Sets the specified property value.
056    * @param property The property name.
057    * @param value The property value.
058    * @return <jk>true</jk> if property name was valid and property was set.
059    */
060   public boolean setProperty(String property, Object value) {
061      BeanContext bc = BeanContext.DEFAULT;
062      if (property.equals(URLENC_expandedParams))
063         expandedParams = bc.convertToType(value, Boolean.class);
064      else
065         return false;
066      return true;
067   }
068
069
070   //--------------------------------------------------------------------------------
071   // Overridden methods
072   //--------------------------------------------------------------------------------
073
074   @Override /* Cloneable */
075   public UrlEncodingProperties clone() {
076      try {
077         return (UrlEncodingProperties)super.clone();
078      } catch (CloneNotSupportedException e) {
079         throw new RuntimeException(e); // Shouldn't happen
080      }
081   }
082}
