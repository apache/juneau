034   /**
035    * Encodes invalid XML text characters.
036    * <p>
037    * Encodes <js>'&'</js>, <js>'&lt;'</js>, and <js>'&gt;'</js> as XML entities.<br>
038    * Encodes any other invalid XML text characters to <code>_x####_</code> sequences.
039    *
040    * @param o The object being encoded.
041    * @return The encoded string.
042    */
043   public static final String encodeText(Object o) {
044
045      if (o == null)
046         return "_x0000_";
047
048      String s = o.toString();
049
050      try {
051         if (needsTextEncoding(s))
052            return encodeTextInner(new StringBuilderWriter(s.length()*2), s).toString();
053      } catch (IOException e) {
054         throw new RuntimeException(e); // Never happens
055      }
056
057      return s;
058   }

156   private static final Writer encodeTextInner(Writer w, String s) throws IOException {
157      final int len = s.length();
158      for (int i = 0; i < len; i++) {
159         char c = s.charAt(i);
160         if (c == '&')
161            w.append("&amp;");
162         else if (c == '<')
163            w.append("&lt;");
164         else if (c == '>')
165            w.append("&gt;");
166         else if (c == '_' && isEscapeSequence(s,i))
167            appendPaddedHexChar(w, c);
168         else if ((i == 0 || i == len-1) && Character.isWhitespace(c))
169            appendPaddedHexChar(w, c);
170         else if (isValidXmlCharacter(c))
171            w.append(c);
172         else if (c == 0x09 || c == 0x0A || c == 0x0D)
173            w.append("&#x000").append(Integer.toHexString(c)).append(";");
174         else
175            appendPaddedHexChar(w, c);
176      }
177      return w;
178   }

180   private static final Writer encodeTextInvalidCharsInner(Writer w, String s) throws IOException {
181      final int len = s.length();
182      for (int i = 0; i < len; i++) {
183         char c = s.charAt(i);
184         if ((i == 0 || i == len-1) && Character.isWhitespace(c))
185            appendPaddedHexChar(w, c);
186         else if (c == '_' && isEscapeSequence(s,i))
187            appendPaddedHexChar(w, c);
188         else if (isValidXmlCharacter(c))
189            w.append(c);
190         else
191            appendPaddedHexChar(w, c);
192      }
193      return w;
194   }

203            w.append("&lt;");

206         else

212   private static final boolean needsTextEncoding(String s) {
213      // See if we need to convert the string.
214      // Conversion is somewhat expensive, so make sure we need to do so before hand.
215      final int len = s.length();
216      for (int i = 0; i < len; i++) {
217         char c = s.charAt(i);
218         if ((i == 0 || i == len-1) && Character.isWhitespace(c))
219            return true;
220         if (c == '&' || c == '<' || c == '>' || c == '\n' || ! isValidXmlCharacter(c) || (c == '_' && isEscapeSequence(s,i)))
221            return true;
222      }
223      return false;
224   }

231   /**
232    * Translates any _x####_ sequences (introduced by the various encode methods) back into their original characters.
233    *
234    * @param s The string being decoded.
235    * @return The decoded string.
236    */
237   public static final String decode(String s) {
238      if (s == null) return null;
239      if (s.length() == 0)
240         return s;
241      if (s.indexOf('_') == -1)
242         return s;
243
244      StringBuffer sb = new StringBuffer(s.length());
245      for (int i = 0; i < s.length(); i++) {
246         char c = s.charAt(i);
247         if (c == '_' && isEscapeSequence(s,i)) {
248
249            int x = Integer.parseInt(s.substring(i+2, i+6), 16);
250
251            // If we find _x0000_, then that means a null.
252            if (x == 0)
253               return null;
254
255            sb.append((char)x);
256            i+=6;
257         } else {
258            sb.append(c);
259         }
260      }
261      return sb.toString();
262   }

269   /**
270    * Serializes and encodes the specified object as valid XML attribute name.
271    *
272    * @param w The writer to send the output to.
273    * @param o The object being serialized.
274    * @return This object (for method chaining).
275    * @throws IOException If a problem occurred.

277   public static final Writer encodeAttr(Writer w, Object o) throws IOException {
278
279      if (o == null)
280         return w.append("_x0000_");
281
282      String s = o.toString();
283
284      if (needsAttributeEncoding(s))
285         return encodeAttrInner(w, s);
286
287      w.append(s);
288      return w;
289   }
290
291   private static final Writer encodeAttrInner(Writer w, String s) throws IOException {
292      for (int i = 0; i < s.length(); i++) {
293         char c = s.charAt(i);
294         if (c == '&')
295            w.append("&amp;");
296         else if (c == '<')
297            w.append("&lt;");
298         else if (c == '>')
299            w.append("&gt;");
300         else if (c == '\'')
301            w.append("&apos;");
302         else if (c == '"')
303            w.append("&quot;");
304         else if (c == '_' && isEscapeSequence(s,i))
305            appendPaddedHexChar(w, c);
306         else if (isValidXmlCharacter(c))
307            w.append(c);
308         else
309            appendPaddedHexChar(w, c);
310      }
311      return w;
312   }
313
314   private static boolean needsAttributeEncoding(String s) {
315      // See if we need to convert the string.
316      // Conversion is somewhat expensive, so make sure we need to do so before hand.
317      for (int i = 0; i < s.length(); i++) {
318         char c = s.charAt(i);
319         if (c == '&' || c == '<' || c == '>' || c == '\n' || c == '\'' || c == '"' || ! isValidXmlCharacter(c))
320            return true;
321      }
322      return false;
323   }

354    *
355    * @param o The object being encoded.
356    * @return The encoded element name string.
357    */
358   public static final String encodeElementName(Object o) {
359      if (o == null)
360         return "_x0000_";
361
362      String s = o.toString();
363
364      try {
365         if (needsElementNameEncoding(s))
366            return encodeElementNameInner(new StringBuilderWriter(s.length() * 2), s).toString();
367      } catch (IOException e) {
368         throw new RuntimeException(e); // Never happens
369      }
370
371      return s;

374   private static final Writer encodeElementNameInner(Writer w, String s) throws IOException {
375      for (int i = 0; i < s.length(); i++) {
376         char c = s.charAt(i);
377         if ((c >= 'A' && c <= 'Z')
378               || (c == '_' && ! isEscapeSequence(s,i))
379               || (c >= 'a' && c <= 'z')
380               || (i != 0 && (
381                     c == '-'
382                     || c == '.'
383                     || (c >= '0' && c <= '9')
384                     || c == '\u00b7'
385                     || (c >= '\u0300' && c <= '\u036f')
386                     || (c >= '\u203f' && c <= '\u2040')
387                  ))
388               || (c >= '\u00c0' && c <= '\u00d6')
389               || (c >= '\u00d8' && c <= '\u00f6')
390               || (c >= '\u00f8' && c <= '\u02ff')
391               || (c >= '\u0370' && c <= '\u037d')
392               || (c >= '\u037f' && c <= '\u1fff')
393               || (c >= '\u200c' && c <= '\u200d')
394               || (c >= '\u2070' && c <= '\u218f')
395               || (c >= '\u2c00' && c <= '\u2fef')
396               || (c >= '\u3001' && c <= '\ud7ff')
397               || (c >= '\uf900' && c <= '\ufdcf')
398               || (c >= '\ufdf0' && c <= '\ufffd')) {
399            w.append(c);
400         }  else {
401            appendPaddedHexChar(w, c);
402         }
403      }
404      return w;
405   }
406
407   private static final boolean needsElementNameEncoding(String s) {
408      // Note that this doesn't need to be perfect, just fast.
409      for (int i = 0; i < s.length(); i++) {
410         char c = s.charAt(i);
411         if (! (c >= '0' && c <= '9' || c >= 'a' && c <= 'z' || c >= 'A' && c <= 'Z'))
412            return true;
413         if (i == 0 && (c >= '0' && c <= '9'))
414            return true;
415      }
416      return false;
417   }

424   /**
425    * Utility method for reading XML mixed content from an XML element and returning it as text.
426    *
427    * @param r The reader to read from.
428    * @return The contents read as a string.
429    * @throws XMLStreamException
430    * @throws IOException
431    */
432   public static String readXmlContents(XMLStreamReader r) throws XMLStreamException, IOException {
433      StringWriter sw = new StringWriter();
434      XmlSerializerWriter w = new XmlSerializerWriter(sw, false, false, '"', null, null, false, null);
435      try {
436         int depth = 0;
437         do {
438            int event = r.next();
439            if (event == START_ELEMENT) {
440               depth++;
441               QName n = r.getName();
442               w.oTag(n.getPrefix(), n.getLocalPart());
443               for (int i = 0; i < r.getNamespaceCount(); i++)
444                  w.attr(r.getNamespacePrefix(i), "xmlns", r.getNamespaceURI(i));
445               for (int i = 0; i < r.getAttributeCount(); i++)
446                  w.attr(r.getAttributePrefix(i), r.getAttributeLocalName(i), r.getAttributeValue(i));
447               w.append('>');
448            } else if (r.hasText()) {
449               w.encodeTextXmlChars(r.getText());
450            } else if (event == ATTRIBUTE) {
451               // attributes handled above.
452            } else if (event == END_ELEMENT) {
453               QName n = r.getName();
454               if (depth > 0)
455                  w.eTag(n.getPrefix(), n.getLocalPart());
456               depth--;
457            }
458            if (depth < 0)
459               return sw.toString();
460         } while (true);
461      } finally {
462         w.close();
463      }
464   }

466   // Returns true if the specified character can safely be used in XML text or an attribute.
467   private static final boolean isValidXmlCharacter(char c) {
468      return (c >= 0x20 && c <= 0xD7FF) /*|| c == 0xA || c == 0xD*/ || (c >= 0xE000 && c <= 0xFFFD);
469   }

471   // Returns true if the string at the specified position is of the form "_x####_"
472   // where '#' are hexadecimal characters.
473   private static final boolean isEscapeSequence(String s, int i) {
474      return s.length() > i+6
475         && s.charAt(i) == '_'
476         && s.charAt(i+1) == 'x'
477         && isHexCharacter(s.charAt(i+2))
478         && isHexCharacter(s.charAt(i+3))
479         && isHexCharacter(s.charAt(i+4))
480         && isHexCharacter(s.charAt(i+5))
481         && s.charAt(i+6) == '_';
482   }
483
484   // Returns true if the character is a hexadecimal character
485   private static final boolean isHexCharacter(char c) {
486      return (c >= '0' && c <= '9') || (c >= 'A' && c <= 'F');
487   }

489   // Converts an integer to a hexadecimal string padded to 4 places.
490   private static final Writer appendPaddedHexChar(Writer out, int num) throws IOException {
491      out.append("_x");
492      char[] n = new char[4];
493      int a = num%16;
494      n[3] = (char)(a > 9 ? 'A'+a-10 : '0'+a);
495      int base = 16;
496      for (int i = 1; i < 4; i++) {
497         a = (num/base)%16;
498         base <<= 4;
499         n[3-i] = (char)(a > 9 ? 'A'+a-10 : '0'+a);
500      }
501      for (int i = 0; i < 4; i++)
502         out.append(n[i]);
503      return out.append('_');
504   }

506   /**
507    * Find the namespace given a list of <ja>@Xml</ja> and <ja>@XmlSchema</ja> annotations.
508    * The annotations should be a child-to-parent ordering of annotations found on
509    *    a class or method.
510    *
511    * @param xmls The list of <ja>@Xml</ja> annotations.
512    * @param schemas The list of <ja>@XmlSchema</ja> annotations.
513    * @return The namespace, or <jk>null</jk> if it couldn't be found.
514    */
515   public static Namespace findNamespace(List<Xml> xmls, List<XmlSchema> schemas) {
516
517      for (Xml xml : xmls) {
518         Namespace ns = findNamespace(xml.prefix(), xml.namespace(), xmls, schemas);
519         if (ns != null)
520            return ns;
521      }
522
523      for (XmlSchema schema : schemas) {
524         Namespace ns = findNamespace(schema.prefix(), schema.namespace(), null, schemas);
525         if (ns != null)
526            return ns;
527      }
528
529      return null;
530   }
531
532   private static Namespace findNamespace(String prefix, String ns, List<Xml> xmls, List<XmlSchema> schemas) {
533
534      // If both prefix and namespace specified, use that Namespace mapping.
535      if (! (prefix.isEmpty() || ns.isEmpty()))
536         return NamespaceFactory.get(prefix, ns);
537
538      // If only prefix specified, need to search for namespaceURI.
539      if (! prefix.isEmpty()) {
540         if (xmls != null)
541            for (Xml xml2 : xmls)
542               if (xml2.prefix().equals(prefix) && ! xml2.namespace().isEmpty())
543                  return NamespaceFactory.get(prefix, xml2.namespace());
544         for (XmlSchema schema : schemas) {
545            if (schema.prefix().equals(prefix) && ! schema.namespace().isEmpty())
546               return NamespaceFactory.get(prefix, schema.namespace());
547            for (XmlNs xmlNs : schema.xmlNs())
548               if (xmlNs.prefix().equals(prefix))
549                  return NamespaceFactory.get(prefix, xmlNs.namespaceURI());
550         }
551         throw new BeanRuntimeException("Found @Xml.prefix annotation with no matching URI.  prefix='"+prefix+"'");
552      }
553
554      // If only namespaceURI specified, need to search for prefix.
555      if (! ns.isEmpty()) {
556         if (xmls != null)
557            for (Xml xml2 : xmls)
558               if (xml2.namespace().equals(ns) && ! xml2.prefix().isEmpty())
559                  return NamespaceFactory.get(xml2.prefix(), ns);
560         for (XmlSchema schema : schemas) {
561            if (schema.namespace().equals(ns) && ! schema.prefix().isEmpty())
562               return NamespaceFactory.get(schema.prefix(), ns);
563            for (XmlNs xmlNs : schema.xmlNs())
564               if (xmlNs.namespaceURI().equals(ns))
565                  return NamespaceFactory.get(xmlNs.prefix(), ns);
566         }
567      }
568
569      return null;
570   }
