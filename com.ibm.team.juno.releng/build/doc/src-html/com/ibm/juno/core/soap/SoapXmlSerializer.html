009package com.ibm.juno.core.soap;

011import static com.ibm.juno.core.soap.SoapXmlSerializerProperties.*;
012
013import java.io.*;
014
015import com.ibm.juno.core.*;
016import com.ibm.juno.core.annotation.*;
017import com.ibm.juno.core.serializer.*;
018import com.ibm.juno.core.xml.*;

020/**
021 * Serializes POJOs to HTTP responses as XML+SOAP.
022 *
023 *
024 * <h6 class='topic'>Media types</h6>
025 * <p>
026 *    Handles <code>Accept</code> types: <code>text/xml+soap</code>
027 * <p>
028 *    Produces <code>Content-Type</code> types: <code>text/xml+soap</code>
029 *
030 *
031 * <h6 class='topic'>Description</h6>
032 * <p>
033 *    Essentially the same output as {@link XmlDocSerializer}, except wrapped in a standard SOAP envelope.
034 *
035 *
036 * <h6 class='topic'>Configurable properties</h6>
037 * <p>
038 *    This class has the following properties associated with it:
039 * <ul>
040 *    <li>{@link SoapXmlSerializerProperties}
041 *    <li>{@link XmlSerializerProperties}
042 *    <li>{@link SerializerProperties}
043 *    <li>{@link BeanContextProperties}
044 * </ul>
045 *
046 *
047 * @author James Bognar (jbognar@us.ibm.com)
048 */
049@Produces(value="text/xml+soap",contentType="text/xml")
050public final class SoapXmlSerializer extends XmlSerializer {
051
052   //--------------------------------------------------------------------------------
053   // Overridden methods
054   //--------------------------------------------------------------------------------
055
056   @Override /* Serializer */
057   protected void doSerialize(Object o, Writer out, SerializerContext ctx) throws IOException, SerializeException {
058      XmlSerializerContext xctx = (XmlSerializerContext)ctx;
059      XmlSerializerWriter w = xctx.getWriter(out);
060      w.append("<?xml")
061         .attr("version", "1.0")
062         .attr("encoding", "UTF-8")
063         .appendln("?>");
064      w.oTag("soap", "Envelope")
065         .attr("xmlns", "soap", xctx.getProperties().getString(SOAPXML_SOAPAction, "http://www.w3.org/2003/05/soap-envelope"))
066         .appendln(">");
067      w.sTag(1, "soap", "Body").nl();
068      super.serialize(o, w, ctx);
069      w.eTag(1, "soap", "Body").nl();
070      w.eTag("soap", "Envelope").nl();
071   }
072
073   @Override /* Serializer */
074   public ObjectMap getResponseHeaders(ObjectMap properties) {
075      return super.getResponseHeaders(properties)
076         .append("SOAPAction", properties.getString(SOAPXML_SOAPAction, "http://www.w3.org/2003/05/soap-envelope"));
077   }
078}
