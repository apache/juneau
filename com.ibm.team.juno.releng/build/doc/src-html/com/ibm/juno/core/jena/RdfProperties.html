343   /**
344    * Specifies that collections should be serialized and parsed as loose collections.  ({@link Boolean}, default=<jk>false</jk>).
345    * <p>
346    * When specified, collections of resources are handled as loose collections of resources in RDF instead of
347    * resources that are children of an RDF collection (e.g. Sequence, Bag).
348    * <p>
349    * Note that this setting is specialized for RDF syntax, and is incompatible with the concept of
350    * losslessly representing POJO models, since the tree structure of these POJO models are lost
351    * when serialized as loose collections.
352    * <p>
353    * This setting is typically only useful if the beans being parsed into do not have a bean property
354    * annotated with {@link BeanProperty#beanUri @BeanProperty(beanUri=true)}.
355    *
356    * <dl>
357    *    <dt>Example:</dt>
358    *    <dd>
359    * <p class='bcode'>
360    *    WriterSerializer s = <jk>new</jk> RdfSerializer.XmlAbbrev().setProperty(<jsf>RDF_looseCollection</jsf>, <jk>true</jk>);
361    *    ReaderParser p = <jk>new</jk> RdfParser.Xml().setProperty(<jsf>RDF_looseCollection</jsf>, <jk>true</jk>);
362    *
363    *    List&lt;MyBean&gt; l = createListOfMyBeans();
364    *
365    *    <jc>// Serialize to RDF/XML as loose resources</jc>
366    *    String rdfXml = s.serialize(l);
367    *
368    *    <jc>// Parse back into a Java collection</jc>
369    *    l = p.parseCollection(rdfXml, LinkedList.<jk>class</jk>, MyBean.<jk>class</jk>);
370    *
371    *    MyBean[] b = createArrayOfMyBeans();
372    *
373    *    <jc>// Serialize to RDF/XML as loose resources</jc>
374    *    String rdfXml = s.serialize(b);
375    *
376    *    <jc>// Parse back into a bean array</jc>
377    *    b = p.parse(rdfXml, MyBean[].<jk>class</jk>);
378    * </p>
379    *    </dd>
380    * </dl>
381    */
382   public static final String RDF_looseCollection = "Rdf.looseCollection";

390   /**
391    * Sets the specified property value.
392    * @param property The property name.
393    * @param value The property value.
394    * @return <jk>true</jk> if property name was valid and property was set.
395    */
396   public boolean setProperty(String property, Object value) {
397      if (property.equals(RDF_language))
398         rdfLanguage = value.toString();
399      else if (property.equals(RDF_junoNs))
400         junoNs = NamespaceFactory.parseNamespace(value);
401      else if (property.equals(RDF_junoBpNs))
402         junoBpNs = NamespaceFactory.parseNamespace(value);
403      else if (property.startsWith("Rdf.jena."))
404         jenaSettings.put(property.substring(9), value);
405      else if (property.equals(RDF_collectionFormat))
406         collectionFormat = RdfCollectionFormat.valueOf(value.toString());
407      else if (property.equals(RDF_useXmlNamespaces))
408         useXmlNamespaces = Boolean.valueOf(value.toString());
409      else if (property.equals(RDF_looseCollection))
410         looseCollection = Boolean.valueOf(value.toString());
411      else
412         return false;
413      return true;
414   }
