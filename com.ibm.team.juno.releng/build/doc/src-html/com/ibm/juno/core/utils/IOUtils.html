026   /**
027    * Reads the contents of a file into a string.
028    *
029    * @param path The path of the file to read using default character encoding.
030    * @return The contents of the reader as a string, or <jk>null</jk> if file does not exist.
031    * @throws IOException If a problem occurred trying to read from the reader.

037   /**
038    * Reads the contents of a file into a string.
039    *
040    * @param in The file to read using default character encoding.
041    * @return The contents of the reader as a string, or <jk>null</jk> if file does not exist.
042    * @throws IOException If a problem occurred trying to read from the reader.
043    */
044   public static String read(File in) throws IOException {
045      if (in == null || ! in.exists())
046         return null;
047      Reader r = new InputStreamReader(new FileInputStream(in), Charset.defaultCharset());
048      return read(r, 0, 1024);
049   }

051   /**
052    * Writes the contents of the specified <code>Reader</code> to the specified file.
053    *
054    * @param out The file to write the output to.
055    * @param in The reader to pipe from.
056    * @return The number of characters written to the file.
057    * @throws IOException
058    */
059   public static int write(File out, Reader in) throws IOException {
060      assertFieldNotNull(out, "out");
061      assertFieldNotNull(in, "in");
062      Writer w = new OutputStreamWriter(new FileOutputStream(out), Charset.defaultCharset());
063      try {
064         return IOPipe.create(in, w).closeOut().run();
065      } finally {
066         w.close();
067      }
068   }

108   /**
109    * Read the specified input stream into a byte array and closes the stream.
110    *
111    * @param in The input stream.
112    * @param bufferSize The expected size of the buffer.
113    * @return The contents of the stream as a byte array.
114    * @throws IOException Thrown by underlying stream.
115    */
116   public static byte[] readBytes(InputStream in, int bufferSize) throws IOException {
117      if (in == null)
118         return null;
119      ByteArrayOutputStream buff = new ByteArrayOutputStream(bufferSize);
120      int nRead;
121      byte[] b = new byte[Math.min(bufferSize, 8192)];
122
123      try {
124         while ((nRead = in.read(b, 0, b.length)) != -1)
125              buff.write(b, 0, nRead);
126
127            buff.flush();
128
129            return buff.toByteArray();
130      } finally {
131         in.close();
132      }
133   }

136   /**
137    * Reads the specified input into a {@link String} until the end of the input is reached.
138    * <p>
139    *    The {@code Reader} is automatically closed.
140    * <p>
141    *    If the {@code Reader} is not an instance of a {@code BufferedReader}, then it gets wrapped in a {@code BufferedReader}.
142    *
143    * @param in The input reader.
144    * @param length Specify a positive number if the length of the input is known.
145    * @param bufferSize Specify the buffer size to use.
146    * @return The contents of the reader as a string.  <jk>null</jk> if reader was null.
147    * @throws IOException If a problem occurred trying to read from the reader.
148    */
149   public static String read(Reader in, int length, int bufferSize) throws IOException {
150      if (in == null)
151         return null;
152      length = (length <= 0 ? bufferSize : length);
153      StringBuilder sb = new StringBuilder(length); // Assume they're ASCII characters.
154      try {
155         char[] buf = new char[Math.min(bufferSize, length)];
156         int i = 0;
157         while ((i = in.read(buf)) != -1)
158            sb.append(buf, 0, i);
159         return sb.toString();
160      } finally {
161         in.close();
162      }
163   }

179   /**
180    * Wraps the specified reader in a buffered reader.
181    * <p>
182    * Returns the original reader if it's already one of the following:
183    * <ul>
184    *    <li>{@link BufferedReader}
185    *    <li>{@link StringReader}
186    *    <li>{@link CharSequenceReader}
187    * </ul>
188    *
189    * @param r The reader being wrapped.
190    * @param buffSize The expected size of the input.
191    * @param minBuffSize The minimum buffer size to use if buffSize is too small.
192    * @param maxBuffSize The maximum buffer size to use if buffSize is too large.
193    * @return The wrapped reader.
194    */
195   public static Reader getBufferedReader(Reader r, int buffSize, int minBuffSize, int maxBuffSize) {
196      assertFieldNotNull(r, "r");
197      if (r instanceof BufferedReader || r instanceof StringReader)
198         return r;
199      if (buffSize <= 0)
200         buffSize = 1024;
201      if (buffSize < minBuffSize)
202         buffSize = minBuffSize;
203      else if (buffSize > maxBuffSize)
204         buffSize = maxBuffSize;
205      return new BufferedReader(r, buffSize);
206   }
207
208   /**
209    * Shortcut for {@code getBufferedReader(r, buffSize, 128, 8096)}
210    *
211    * @param r The reader being wrapped.
212    * @param buffSize The expected size of the input.
213    * @return The reader wrapped in a {@link BufferedReader}, or the original {@link Reader} if it's already
214    *    a buffered reader.
215    */
216   public static Reader getBufferedReader(Reader r, int buffSize) {
217      return getBufferedReader(r, buffSize, 128, 8096);
218   }

220   /**
221    * Counts the number of bytes in the input stream and then closes the stream.
222    *
223    * @param is The input stream to read from.
224    * @return The number of bytes read.
225    * @throws IOException
226    */
227   public static long count(InputStream is) throws IOException {
228      assertFieldNotNull(is, "is");
229      long c = 0;
230      long i;
231      try {
232         while ((i = is.skip(1024)) != 0)
233            c += i;
234      } finally {
235         is.close();
236      }
237      return c;
238   }

260   /**
261    * Given the specified <js>"Content-Length"</js> header value, return an appropriate buffer size.
262    * The maximum buffer size is 1MB.
263    *
264    * @param contentLength The value of the <js>"Content-Length"</js> header.
265    * @return The appropriate buffer size.
266    */
267   public static int getBufferSize(String contentLength) {
268      try {
269         if (! StringUtils.isEmpty(contentLength)) {
270            long l = Long.decode(contentLength);
271            if (l > 1048576)
272               return 1048576;
273            if (l <= 0)
274               return 8192;
275            return (int)l;
276         }
277      } catch (Exception e) {
278         return 8192;
279      }
280      return 8192;
281   }

335   /**
336    * Quietly close all specified input streams, output streams, readers, and writers.
337    *
338    * @param o The list of all objects to quietly close.
339    */
340   public static void closeQuietly(Object...o) {
341      for (Object o2 : o) {
342         if (o2 instanceof InputStream)
343            closeQuietly((InputStream)o2);
344         if (o2 instanceof OutputStream)
345            closeQuietly((OutputStream)o2);
346         if (o2 instanceof Reader)
347            closeQuietly((Reader)o2);
348         if (o2 instanceof Writer)
349            closeQuietly((Writer)o2);
350      }
351   }
