011import java.io.*;

015/**
016 * Similar to a {@link java.io.PushbackReader} with a pushback buffer of 1 character.
017 * <p>
018 *    Code is optimized to work with a 1 character buffer.
019 * <p>
020 *    Additionally keeps track of current line and column number, and provides the ability to set
021 *    mark points and capture characters from the previous mark point.
022 * <p>
023 *    <b>Warning:</b>  Not thread safe.
024 *
025 * @author James Bognar (jbognar@us.ibm.com)
026 */
027public class ParserReader extends Reader {
028
029   /** Wrapped reader */
030   protected Reader r;
031
032   private char[] buff;       // Internal character buffer
033   private int line = 1;      // Current line number
034   private int column;        // Current column number
035   private int iCurrent = 0;  // Current pointer into character buffer
036   private int iMark = -1;    // Mark position in buffer
037   private int iEnd = 0;      // The last good character position in the buffer
038   private boolean endReached, holesExist;
039
040   ParserReader() {}
041
042   /**
043    * Constructor for input from a {@link CharSequence}.
044    *
045    * @param in The character sequence being read from.
046    */
047   public ParserReader(CharSequence in) {
048      this.r = new CharSequenceReader(in);
049      if (in == null)
050         this.buff = new char[0];
051      else
052         this.buff = new char[in.length() < 1024 ? in.length() : 1024];
053   }
054
055   /**
056    * Constructor for input from a {@link Reader}).
057    *
058    * @param r The Reader being wrapped.
059    * @param buffSize The buffer size to use for the buffered reader.
060    */
061   public ParserReader(Reader r, int buffSize) {
062      if (r instanceof ParserReader)
063         this.r = ((ParserReader)r).r;
064      else
065         this.r = r;
066      this.buff = new char[buffSize <= 0 ? 1024 : Math.max(buffSize, 20)];
067   }
068
069   /**
070    * Returns the current line number position in this reader.
071    *
072    * @return The current line number.
073    */
074   public final int getLine() {
075      return line;
076   }
077
078   /**
079    * Returns the current column number position in this reader.
080    *
081    * @return The current column number.
082    */
083   public final int getColumn() {
084      return column;
085   }

087   /**
088    * Reads a single character.
089    * Note that this method does NOT process extended unicode characters (i.e. characters
090    *    above 0x10000), but rather returns them as two <jk>char</jk>s.
091    * Use {@link #readCodePoint()} to ensure proper handling of extended unicode.
092    *
093    * @return The character read, or -1 if the end of the stream has been reached.
094    * @throws IOException If a problem occurred trying to read from the reader.
095    */
096   @Override /* Reader */
097   public final int read() throws IOException {
098      int c = readFromBuff();
099      if (c == -1)
100         return -1;
101      if (c == '\n') {
102         line++;
103         column = 0;
104      } else {
105         column++;
106      }
107      return c;
108   }
109
110   /**
111    * Same as {@link #read()} but detects and combines extended unicode characters (i.e. characters
112    *    above 0x10000).
113    *
114    * @return The character read, or -1 if the end of the stream has been reached.
115    * @throws IOException If a problem occurred trying to read from the reader.
116    */
117   public final int readCodePoint() throws IOException {
118      int c = read();
119
120      // Characters that take up 2 chars.
121      if (c >= 0xd800 && c <= 0xdbff) {
122         int low = read();
123         if (low >= 0xdc00 && low <= 0xdfff)
124            c = 0x10000 + ((c - 0xd800) << 10) + (low - 0xdc00);
125      }
126
127      return c;
128   }
129
130   private final int readFromBuff() throws IOException {
131      while (iCurrent >= iEnd) {
132         if (endReached)
133            return -1;
134
135         // If there's still space at the end of this buffer, fill it.
136         // Make sure there's at least 2 character spaces free for extended unicode characters.
137         //if (false) {
138         if (iEnd+1 < buff.length) {
139            int x = read(buff, iCurrent, buff.length-iEnd);
140            if (x == -1) {
141               endReached = true;
142               return -1;
143            }
144            iEnd += x;
145
146         } else {
147            // If we're currently marking, then we want to copy from the current mark point
148            // to the beginning of the buffer and then fill in the remainder of buffer.
149            if (iMark >= 0) {
150
151               // If we're marking from the beginning of the array, we double the size of the
152               // buffer.  This isn't likely to occur often.
153               if (iMark == 0) {
154                  char[] buff2 = new char[buff.length<<1];
155                  System.arraycopy(buff, 0, buff2, 0, buff.length);
156                  buff = buff2;
157
158               // Otherwise, we copy what's currently marked to the beginning of the buffer.
159               } else {
160                  int copyBuff = iMark;
161                  System.arraycopy(buff, copyBuff, buff, 0, buff.length - copyBuff);
162                  iCurrent -= copyBuff;
163                  iMark -= copyBuff;
164               }
165               int expected = buff.length - iCurrent;
166
167               int x = read(buff, iCurrent, expected);
168               if (x == -1) {
169                  endReached = true;
170                  iEnd = iCurrent;
171                  return -1;
172               }
173               iEnd = iCurrent + x;
174            } else {
175               // Copy the last 10 chars in the buffer to the beginning of the buffer.
176               int copyBuff = Math.min(iCurrent, 10);
177               System.arraycopy(buff, iCurrent-copyBuff, buff, 0, copyBuff);
178
179               // Number of characters we expect to copy on the next read.
180               int expected = buff.length - copyBuff;
181               int x = read(buff, copyBuff, expected);
182               iCurrent = copyBuff;
183               if (x == -1) {
184                  endReached = true;
185                  iEnd = iCurrent;
186                  return -1;
187               }
188               iEnd = iCurrent + x;
189            }
190         }
191      }
192      return buff[iCurrent++];
193   }
194
195   /**
196    * Start buffering the calls to read() so that the text can be gathered from the mark
197    * point on calling {@code getFromMarked()}.
198    */
199   public final void mark() {
200      iMark = iCurrent;
201   }
202
203
204   /**
205    * Peeks the next character in the stream.
206    * <p>
207    *    This is equivalent to doing a {@code read()} followed by an {@code unread()}.
208    *
209    * @return The peeked character, or (char)-1 if the end of the stream has been reached.
210    * @throws IOException If a problem occurred trying to read from the reader.
211    */
212   public final int peek() throws IOException {
213      int c = read();
214      if (c != -1)
215         unread();
216      return c;
217   }
218
219   /**
220    * Read the specified number of characters off the stream.
221    *
222    * @param num The number of characters to read.
223    * @return The characters packaged as a String.
224    * @throws IOException If a problem occurred trying to read from the reader.
225    */
226   public final String read(int num) throws IOException {
227      char[] c = new char[num];
228      for (int i = 0; i < num; i++) {
229         int c2 = read();
230         if (c2 == -1)
231            return new String(c, 0, i);
232         c[i] = (char)c2;
233      }
234      return new String(c);
235   }
236
237   /**
238    * Pushes the last read character back into the stream.
239    *
240    * @return This object (for method chaining).
241    * @throws IOException If a problem occurred trying to read from the reader.
242    */
243   public final ParserReader unread() throws IOException {
244      if (iCurrent <= 0)
245         throw new IOException("Buffer underflow.");
246      iCurrent--;
247      column--;
248      return this;
249   }

251   /**
252    * Close this reader and the underlying reader.
253    *
254    * @throws IOException If a problem occurred trying to read from the reader.
255    */
256   @Override /* Reader */
257   public void close() throws IOException {
258      r.close();
259   }

261   /**
262    * Returns the contents of the reusable character buffer as a string, and
263    * resets the buffer for next usage.
264    *
265    * @return The contents of the reusable character buffer as a string.
266    */
267   public final String getMarked() {
268      return getMarked(0, 0);
269   }
270
271   /**
272    * Same as {@link #getMarked()} except allows you to specify offsets
273    *    into the buffer.
274    * <p>
275    * For example, to return the marked string, but trim the first and last characters,
276    *    call the following:
277    * <p class='bcode'>
278    *    getFromMarked(1, -1);
279    * </p>
280    *
281    * @param offsetStart The offset of the start position.
282    * @param offsetEnd The offset of the end position.
283    * @return The contents of the reusable character buffer as a string.
284    */
285   public final String getMarked(int offsetStart, int offsetEnd) {
286      int offset = 0;
287
288      // Holes are \u00FF 'delete' characters that we need to get rid of now.
289      if (holesExist) {
290         for (int i = iMark; i < iCurrent; i++) {
291            char c = buff[i];
292            if (c == 127)
293               offset++;
294            else
295               buff[i-offset] = c;
296         }
297         holesExist = false;
298      }
299      int start = iMark + offsetStart, len = iCurrent - iMark + offsetEnd - offsetStart - offset;
300      String s = new String(buff, start, len);
301      iMark = -1;
302      return s;
303   }
304
305   /**
306    * Trims off the last character in the marking buffer.
307    * Useful for removing escape characters from sequences.
308    *
309    * @return This object (for method chaining).
310    */
311   public final ParserReader delete() {
312      return delete(1);
313   }
314
315   /**
316    * Trims off the specified number of last characters in the marking buffer.
317    * Useful for removing escape characters from sequences.
318    *
319    * @param count The number of characters to delete.
320    * @return This object (for method chaining).
321    */
322   public final ParserReader delete(int count) {
323      for (int i = 0; i < count; i++)
324         buff[iCurrent-i-1] = 127;
325      holesExist = true;
326      return this;
327   }
328
329   /**
330    * Replaces the last character in the marking buffer with the specified character.
331    * <code>offset</code> must be at least <code>1</code> for normal characters, and
332    * <code>2</code> for extended unicode characters in order for the replacement
333    * to fit into the buffer.
334    *
335    * @param c The new character.
336    * @param offset The offset.
337    * @return This object (for method chaining).
338    * @throws IOException
339    */
340   public final ParserReader replace(int c, int offset) throws IOException {
341      if (c < 0x10000) {
342         if (offset < 1)
343            throw new IOException("Buffer underflow.");
344         buff[iCurrent-offset] = (char)c;
345      } else {
346         if (offset < 2)
347            throw new IOException("Buffer underflow.");
348         c -= 0x10000;
349         buff[iCurrent-offset] = (char)(0xd800 + (c >> 10));
350         buff[iCurrent-offset+1] = (char)(0xdc00 + (c & 0x3ff));
351         offset--;
352      }
353      // Fill in the gap with DEL characters.
354      for (int i = 1; i < offset; i++)
355         buff[iCurrent-i] = 127;
356      holesExist |= (offset > 1);
357      return this;
358   }
359
360   /**
361    * Replace the last read character in the buffer with the specified character.
362    *
363    * @param c The new character.
364    * @return This object (for method chaining).
365    * @throws IOException
366    */
367   public final ParserReader replace(char c) throws IOException {
368      return replace(c, 1);
369   }
370
371   /**
372    * Subclasses can override this method to provide additional filtering (e.g. {#link UrlEncodingParserReader}).
373    * Default implementation simply calls the same method on the underlying reader.
374    */
375   @Override /* Reader */
376   public int read(char[] cbuf, int off, int len) throws IOException {
377      return r.read(cbuf, off, len);
378   }
