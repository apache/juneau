009package com.ibm.juno.core.xml;
010
011import java.util.*;
012
013import com.ibm.juno.core.*;
014import com.ibm.juno.core.xml.annotation.*;
015
016/**
017 * Metadata on bean properties specific to the XML serializers and parsers pulled from the {@link Xml @Xml} annotation on the bean property.
018 *
019 * @author James Bognar (jbognar@us.ibm.com)
020 * @param <T> The bean class.
021 */
022public class XmlBeanPropertyMeta<T> {
023
024   private Namespace namespace = null;
025   private XmlFormat xmlFormat = XmlFormat.NORMAL;
026   private XmlContentHandler<T> xmlContentHandler = null;
027   private String childName;
028   private final BeanPropertyMeta<T> beanPropertyMeta;
029
030   /**
031    * Constructor.
032    *
033    * @param beanPropertyMeta The metadata of the bean property of this additional metadata.
034    */
035   public XmlBeanPropertyMeta(BeanPropertyMeta<T> beanPropertyMeta) {
036      this.beanPropertyMeta = beanPropertyMeta;
037
038      if (beanPropertyMeta.getField() != null)
039         findXmlInfo(beanPropertyMeta.getField().getAnnotation(Xml.class));
040      if (beanPropertyMeta.getGetter() != null)
041         findXmlInfo(beanPropertyMeta.getGetter().getAnnotation(Xml.class));
042      if (beanPropertyMeta.getSetter() != null)
043         findXmlInfo(beanPropertyMeta.getSetter().getAnnotation(Xml.class));
044
045      if (namespace == null)
046         namespace = beanPropertyMeta.getBeanMeta().getClassMeta().getXmlMeta().getNamespace();
047
048      if (beanPropertyMeta.isBeanUri() && xmlFormat != XmlFormat.ELEMENT)
049         xmlFormat = XmlFormat.ATTR;
050   }
051
052   /**
053    * Returns the XML namespace associated with this bean property.
054    * <p>
055    *    Namespace is determined in the following order:
056    * <ol>
057    *    <li>{@link Xml#prefix()} annotation defined on bean property field.
058    *    <li>{@link Xml#prefix()} annotation defined on bean getter.
059    *    <li>{@link Xml#prefix()} annotation defined on bean setter.
060    *    <li>{@link Xml#prefix()} annotation defined on bean.
061    *    <li>{@link Xml#prefix()} annotation defined on bean package.
062    *    <li>{@link Xml#prefix()} annotation defined on bean superclasses.
063    *    <li>{@link Xml#prefix()} annotation defined on bean superclass packages.
064    *    <li>{@link Xml#prefix()} annotation defined on bean interfaces.
065    *    <li>{@link Xml#prefix()} annotation defined on bean interface packages.
066    * </ol>
067    *
068    * @return The namespace associated with this bean property, or <jk>null</jk> if no namespace is
069    *    associated with it.
070    */
071   public Namespace getNamespace() {
072      return namespace;
073   }
074
075   /**
076    * Returns the XML format of this property from the {@link Xml#format} annotation on this bean property.
077    *
078    * @return The XML format, or {@link XmlFormat#NORMAL} if annotation not specified.
079    */
080   protected XmlFormat getXmlFormat() {
081      return xmlFormat;
082   }
083
084   /**
085    * Returns the XML content handler of this property from the {@link Xml#contentHandler} annotation on this bean property.
086    *
087    * @return The XML content handler, or <jk>null</jk> if annotation not specified.
088    */
089   protected XmlContentHandler<T> getXmlContentHandler() {
090      return xmlContentHandler;
091   }
092
093   /**
094    * Returns the child element of this property from the {@link Xml#childName} annotation on this bean property.
095    *
096    * @return The child element, or <jk>null</jk> if annotation not specified.
097    */
098   protected String getChildName() {
099      return childName;
100   }
101
102   /**
103    * Returns the bean property metadata that this metadata belongs to.
104    *
105    * @return The bean property metadata.  Never <jk>null</jk>.
106    */
107   protected BeanPropertyMeta<T> getBeanPropertyMeta() {
108      return beanPropertyMeta;
109   }
110
111   @SuppressWarnings("unchecked")
112   private void findXmlInfo(Xml xml) {
113      if (xml == null)
114         return;
115      ClassMeta<?> cmProperty = beanPropertyMeta.getClassMeta();
116      ClassMeta<?> cmBean = beanPropertyMeta.getBeanMeta().getClassMeta();
117      String name = beanPropertyMeta.getName();
118      if (! xml.name().isEmpty())
119         throw new BeanRuntimeException(cmBean.getInnerClass(), "Annotation error on property ''{0}''.  Found @Xml.name annotation can only be specified on types.", name);
120
121      List<Xml> xmls = beanPropertyMeta.findAnnotations(Xml.class);
122      List<XmlSchema> schemas = beanPropertyMeta.findAnnotations(XmlSchema.class);
123      namespace = XmlUtils.findNamespace(xmls, schemas);
124
125      if (xmlFormat == XmlFormat.NORMAL)
126         xmlFormat = xml.format();
127
128      boolean isCollection = cmProperty.isCollection() || cmProperty.isArray();
129
130      String cen = xml.childName();
131      if ((! cen.isEmpty()) && (! isCollection))
132         throw new BeanRuntimeException(cmProperty.getInnerClass(), "Annotation error on property ''{0}''.  @Xml.childName can only be specified on collections and arrays.", name);
133
134      if (xmlFormat == XmlFormat.COLLAPSED) {
135         if (isCollection) {
136            if (cen.isEmpty())
137               cen = cmProperty.getXmlMeta().getChildName();
138            if (cen == null || cen.isEmpty())
139               cen = cmProperty.getElementType().getXmlMeta().getElementName();
140            if (cen == null || cen.isEmpty())
141               cen = name;
142         } else {
143            throw new BeanRuntimeException(cmBean.getInnerClass(), "Annotation error on property ''{0}''.  @Xml.format=COLLAPSED can only be specified on collections and arrays.", name);
144         }
145         if (cen.isEmpty() && isCollection)
146            cen = cmProperty.getXmlMeta().getElementName();
147      }
148
149      try {
150         if (xmlFormat == XmlFormat.CONTENT && xml.contentHandler() != XmlContentHandler.NULL.class)
151            xmlContentHandler = (XmlContentHandler<T>) xml.contentHandler().newInstance();
152      } catch (Exception e) {
153         throw new BeanRuntimeException(cmBean.getInnerClass(), "Could not instantiate content handler ''{0}''", xml.contentHandler().getName()).initCause(e);
154      }
155
156      if (! cen.isEmpty())
157         childName = cen;
158   }
159}
