009package com.ibm.juno.core.ini;
010
011import java.util.*;
012
013
014/**
015 * Listener that can be used to listen for change events for a specific entry in a config file.
016 * <p>
017 * Use the {@link ConfigFile#addListener(ConfigFileListener)} method to register listeners.
018 */
019public class EntryListener extends ConfigFileListener {
020
021   private String fullKey;
022
023   /**
024    * Constructor.
025    *
026    * @param fullKey The key in the config file to listen for changes on.
027    */
028   public EntryListener(String fullKey) {
029      this.fullKey = fullKey;
030   }
031
032   @Override /* ConfigFileListener */
033   public void onChange(ConfigFile cf, Set<String> changes) {
034      if (changes.contains(fullKey))
035         onChange(cf);
036   }
037
038   /**
039    * Signifies that the config file entry changed.
040    *
041    * @param cf The config file being changed.
042    */
043   public void onChange(ConfigFile cf) {}
044}
