009package com.ibm.juno.core.annotation;

011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015import java.net.*;
016
017import com.ibm.juno.core.serializer.*;
018
019/**
020 * Used to identify a class or bean property as a URI.
021 * <p>
022 *    By default, instances of {@link URL} and {@link URI} are considered URIs during serialization, and are
023 *       handled differently depending on the serializer (e.g. <code>HtmlSerializer</code> creates a hyperlink,
024 *       <code>RdfXmlSerializer</code> creates an <code>rdf:resource</code> object, etc...).
025 * <p>
026 *    This annotation allows you to identify other classes that return URIs via <code>toString()</code> as URI objects.
027 * <p>
028 *    Relative URIs are automatically prepended with {@link SerializerProperties#SERIALIZER_absolutePathUriBase} and {@link SerializerProperties#SERIALIZER_relativeUriBase}
029 *       during serialization just like relative <code>URIs</code>.
030 * <p>
031 *    This annotation can be applied to classes, interfaces, or bean property methods for fields.
032 *
033 * <h6 class='topic'>Examples</h6>
034 * <p class='bcode'>
035 *
036 *    <jc>// Applied to a class whose toString() method returns a URI.</jc>
037 *    <ja>@URI</ja>
038 *    <jk>public class</jk> MyURI {
039 *       <ja>@Override</ja>
040 *       <jk>public</jk> String toString() {
041 *          <jk>return</jk> <js>"http://localhost:9080/foo/bar"</js>;
042 *       }
043 *    }
044 *
045 *    <jc>// Applied to bean properties</jc>
046 *    <jk>public class</jk> MyBean {
047 *
048 *       <ja>@URI</ja>
049 *       <jk>public</jk> String <jf>beanUri</jf>;
050 *
051 *       <ja>@URI</ja>
052 *       <jk>public</jk> String getParentUri() {
053 *          ...
054 *       }
055 *    }
056 * </p>
057 *
058 * @author James Bognar (jbognar@us.ibm.com)
059 */
060@Documented
061@Target({TYPE,FIELD,METHOD})
062@Retention(RUNTIME)
063@Inherited
064public @interface URI {}
