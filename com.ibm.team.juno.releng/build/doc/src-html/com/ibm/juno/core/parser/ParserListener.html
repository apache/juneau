013/**
014 * Class for listening for certain parse events during a document parse.
015 * <p>
016 *    Listeners can be registered with parsers through the {@link Parser#addListener(ParserListener)} method.

018 *    It should be noted that listeners are not automatically copied over to new parsers when a parser is cloned.
019 *
020 * @author James Bognar (jbognar@us.ibm.com)
021 */
022public class ParserListener {
023
024   /**
025    * Gets called when an unknown bean property is detected in a document.
026    * <p>
027    *    This method only gets called if {@link BeanContextProperties#BEAN_ignoreUnknownBeanProperties} setting is <jk>true</jk>.
028    *    Otherwise, the parser will throw a {@link ParseException}.
029    *
030    * @param <T> The class type of the bean.
031    * @param propertyName The property name encountered in the document.
032    * @param beanClass The bean class.
033    * @param bean The bean.
034    * @param line The line number where the unknown property was found (-1 if parser doesn't support line/column indicators).
035    * @param col The column number where the unknown property was found (-1 if parser doesn't support line/column indicators).
036    */
037   public <T> void onUnknownProperty(String propertyName, Class<T> beanClass, T bean, int line, int col) {
038      // Do something with information
039   }
040}
