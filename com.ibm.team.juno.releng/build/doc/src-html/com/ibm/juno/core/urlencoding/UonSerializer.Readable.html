023/**
024 * Serializes POJO models to UON (a notation for URL-encoded query parameter values).
025 *
026 *
027 * <h6 class='topic'>Media types</h6>
028 * <p>
029 *    Handles <code>Accept</code> types: <code>text/uon</code>
030 * <p>
031 *    Produces <code>Content-Type</code> types: <code>text/uon</code>
032 *
033 *
034 * <h6 class='topic'>Description</h6>
035 * <p>
036 *    This serializer provides several serialization options.  Typically, one of the predefined DEFAULT serializers will be sufficient.
037 *    However, custom serializers can be constructed to fine-tune behavior.
038 *
039 *
040 * <h6 class='topic'>Configurable properties</h6>
041 * <p>
042 *    This class has the following properties associated with it:
043 * <ul>
044 *    <li>{@link UonSerializerProperties}
045 *    <li>{@link SerializerProperties}
046 *    <li>{@link BeanContextProperties}
047 * </ul>
048 * <p>
049 *    The following shows a sample object defined in Javascript:
050 * </p>
051 * <p class='bcode'>
052 *    {
053 *       id: 1,
054 *       name: <js>'John Smith'</js>,
055 *       uri: <js>'http://sample/addressBook/person/1'</js>,
056 *       addressBookUri: <js>'http://sample/addressBook'</js>,
057 *       birthDate: <js>'1946-08-12T00:00:00Z'</js>,
058 *       otherIds: <jk>null</jk>,
059 *       addresses: [
060 *          {
061 *             uri: <js>'http://sample/addressBook/address/1'</js>,
062 *             personUri: <js>'http://sample/addressBook/person/1'</js>,
063 *             id: 1,
064 *             street: <js>'100 Main Street'</js>,
065 *             city: <js>'Anywhereville'</js>,
066 *             state: <js>'NY'</js>,
067 *             zip: 12345,
068 *             isCurrent: <jk>true</jk>,
069 *          }
070 *       ]
071 *    }
072 * </p>
073 * <p>
074 *    Using the "strict" syntax defined in this document, the equivalent
075 *       UON notation would be as follows:
076 * </p>
077 * <p class='bcode'>
078 *    $o(
079 *       <xa>id</xa>=$n(<xs>1</xs>),
080 *       <xa>name</xa>=<xs>John+Smith</xs>,
081 *       <xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
082 *       <xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
083 *       <xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
084 *       <xa>otherIds</xa>=<xs>%00</xs>,
085 *       <xa>addresses</xa>=$a(
086 *          $o(
087 *             <xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
088 *             <xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
089 *             <xa>id</xa>=$n(<xs>1</xs>),
090 *             <xa>street</xa>=<xs>100+Main+Street</xs>,
091 *             <xa>city</xa>=<xs>Anywhereville</xs>,
092 *             <xa>state</xa>=<xs>NY</xs>,
093 *             <xa>zip</xa>=$n(<xs>12345</xs>),
094 *             <xa>isCurrent</xa>=$b(<xs>true</xs>)
095 *          )
096 *       )
097 *    )
098 * </p>
099 * <p>
100 *    A secondary "lax" syntax is available when the data type of the
101 *       values are already known on the receiving end of the transmission:
102 * </p>
103 * <p class='bcode'>
104 *    (
105 *       <xa>id</xa>=<xs>1</xs>,
106 *       <xa>name</xa>=<xs>John+Smith</xs>,
107 *       <xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
108 *       <xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
109 *       <xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
110 *       <xa>otherIds</xa>=<xs>%00</xs>,
111 *       <xa>addresses</xa>=(
112 *          (
113 *             <xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
114 *             <xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
115 *             <xa>id</xa>=<xs>1</xs>,
116 *             <xa>street</xa>=<xs>100+Main+Street</xs>,
117 *             <xa>city</xa>=<xs>Anywhereville</xs>,
118 *             <xa>state</xa>=<xs>NY</xs>,
119 *             <xa>zip</xa>=<xs>12345</xs>,
120 *             <xa>isCurrent</xa>=<xs>true</xs>
121 *          )
122 *       )
123 *    )
124 * </p>
125 *
126 *
127 * <h6 class='topic'>Examples</h6>
128 * <p class='bcode'>
129 *    <jc>// Serialize a Map</jc>
130 *    Map m = <jk>new</jk> ObjectMap(<js>"{a:'b',c:1,d:false,e:['f',1,false],g:{h:'i'}}"</js>);
131 *
132 *    <jc>// Serialize to value equivalent to JSON.</jc>
133 *    <jc>// Produces "$o(a=b,c=$n(1),d=$b(false),e=$a(f,$n(1),$b(false)),g=$o(h=i))"</jc>
134 *    String s = UonSerializer.<jsf>DEFAULT</jsf>.serialize(s);
135 *
136 *    <jc>// Serialize to simplified value (for when data type is already known by receiver).</jc>
137 *    <jc>// Produces "(a=b,c=1,d=false,e=(f,1,false),g=(h=i))"</jc>
138 *    String s = UonSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
139 *
140 *    <jc>// Serialize a bean</jc>
141 *    <jk>public class</jk> Person {
142 *       <jk>public</jk> Person(String s);
143 *       <jk>public</jk> String getName();
144 *       <jk>public int</jk> getAge();
145 *       <jk>public</jk> Address getAddress();
146 *       <jk>public boolean</jk> deceased;
147 *    }
148 *
149 *    <jk>public class</jk> Address {
150 *       <jk>public</jk> String getStreet();
151 *       <jk>public</jk> String getCity();
152 *       <jk>public</jk> String getState();
153 *       <jk>public int</jk> getZip();
154 *    }
155 *
156 *    Person p = <jk>new</jk> Person(<js>"John Doe"</js>, 23, <js>"123 Main St"</js>, <js>"Anywhere"</js>, <js>"NY"</js>, 12345, <jk>false</jk>);
157 *
158 *    <jc>// Produces "$o(name=John Doe,age=23,address=$o(street=123 Main St,city=Anywhere,state=NY,zip=$n(12345)),deceased=$b(false))"</jc>
159 *    String s = UonSerializer.<jsf>DEFAULT</jsf>.serialize(s);
160 *
161 *    <jc>// Produces "(name=John Doe,age=23,address=(street=123 Main St,city=Anywhere,state=NY,zip=12345),deceased=false)"</jc>
162 *    String s = UonSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
163 * </p>
164 *
165 * @author James Bognar (jbognar@us.ibm.com)
166 */

236   /**
237    * Workhorse method. Determines the type of object, and then calls the
238    * appropriate type-specific serialization method.
239    *
240    * @param out The writer to serialize to.
241    * @param o The object being serialized.
242    * @param eType The expected type of the object if this is a bean property.
243    * @param ctx The context that exist for the duration of a serialize.
244    * @param attrName The bean property name if this is a bean property.  <jk>null</jk> if this isn't a bean property being serialized.
245    * @param pMeta The bean property metadata.
246    * @param quoteEmptyStrings <jk>true</jk> if this is the first entry in an array.
247    * @param isTop If we haven't recursively called this method.
248    * @return The same writer passed in.
249    * @throws SerializeException
250    */
251   @SuppressWarnings({ "rawtypes", "unchecked" })
252   protected SerializerWriter serializeAnything(UonSerializerWriter out, Object o, ClassMeta<?> eType, UonSerializerContext ctx,
253         String attrName, BeanPropertyMeta pMeta, boolean quoteEmptyStrings, boolean isTop) throws SerializeException {
254      try {
255
256         BeanContext bc = ctx.getBeanContext();
257
258         if (o == null) {
259            out.appendObject(null, false, false, isTop);
260            return out;
261         }
262
263         if (eType == null)
264            eType = object();
265
266         boolean addClassAttr;      // Add "_class" attribute to element?
267         ClassMeta<?> aType;        // The actual type
268         ClassMeta<?> gType;        // The generic type
269
270         aType = ctx.push(attrName, o, eType);
271         boolean isRecursion = aType == null;
272
273         // Handle recursion
274         if (aType == null) {
275            o = null;
276            aType = object();
277         }
278
279         gType = aType.getFilteredClassMeta();
280         addClassAttr = (ctx.isAddClassAttrs() && ! eType.equals(aType));
281
282         // Filter if necessary
283         PojoFilter filter = aType.getPojoFilter();            // The filter
284         if (filter != null) {
285            o = filter.filter(o);
286
287            // If the filter's getFilteredClass() method returns Object, we need to figure out
288            // the actual type now.
289            if (gType.isObject())
290               gType = bc.getClassMetaForObject(o);
291         }
292
293         // '\0' characters are considered null.
294         if (o == null || (gType.isChar() && ((Character)o).charValue() == 0))
295            out.appendObject(null, false, false, isTop);
296         else if (gType.hasToObjectMapMethod())
297            serializeMap(out, gType.toObjectMap(o), eType, ctx);
298         else if (gType.isBean())
299            serializeBeanMap(out, bc.forBean(o), addClassAttr, ctx);
300         else if (gType.isUri() || (pMeta != null && (pMeta.isUri() || pMeta.isBeanUri())))
301            out.appendUri(o, isTop);
302         else if (gType.isMap()) {
303            if (o instanceof BeanMap)
304               serializeBeanMap(out, (BeanMap)o, addClassAttr, ctx);
305            else
306               serializeMap(out, (Map)o, eType, ctx);
307         }
308         else if (gType.isCollection()) {
309            if (addClassAttr)
310               serializeCollectionMap(out, (Collection)o, gType, ctx);
311            else
312               serializeCollection(out, (Collection) o, eType, ctx);
313         }
314         else if (gType.isArray()) {
315            if (addClassAttr)
316               serializeCollectionMap(out, toList(gType.getInnerClass(), o), gType, ctx);
317            else
318               serializeCollection(out, toList(gType.getInnerClass(), o), eType, ctx);
319         }
320         else {
321            out.appendObject(o, quoteEmptyStrings, false, isTop);
322         }
323
324         if (! isRecursion)
325            ctx.pop();
326         return out;
327      } catch (SerializeException e) {
328         throw e;
329      } catch (StackOverflowError e) {
330         throw e;
331      } catch (Throwable e) {
332         throw new SerializeException("Exception occured trying to process object of type ''{0}''", (o == null ? null : o.getClass().getName())).initCause(e);
333      }
334   }

336   @SuppressWarnings({ "rawtypes", "unchecked" })
337   private SerializerWriter serializeMap(UonSerializerWriter out, Map m, ClassMeta<?> type, UonSerializerContext ctx) throws IOException, SerializeException {
338
339      m = ctx.sort(m);
340
341      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
342
343      int depth = ctx.getIndent();
344      out.startFlag('o');
345
346      Iterator mapEntries = m.entrySet().iterator();
347
348      while (mapEntries.hasNext()) {
349         Map.Entry e = (Map.Entry) mapEntries.next();
350         Object value = e.getValue();
351         Object key = ctx.generalize(e.getKey(), keyType);
352         out.cr(depth).appendObject(key, ctx.isUseWhitespace(), false, false).append('=');
353         serializeAnything(out, value, valueType, ctx, (key == null ? null : key.toString()), null, ctx.isUseWhitespace(), false);
354         if (mapEntries.hasNext())
355            out.append(',');
356      }
357
358      if (m.size() > 0)
359         out.cr(depth-1);
360      out.append(')');
361
362      return out;
363   }

365   @SuppressWarnings({ "rawtypes" })
366   private SerializerWriter serializeCollectionMap(UonSerializerWriter out, Collection o, ClassMeta<?> type, UonSerializerContext ctx) throws IOException, SerializeException {
367      int i = ctx.getIndent();
368      out.startFlag('o').nl();
369      out.append(i, "_class=").appendObject(type, false, false, false).append(',').nl();
370      out.append(i, "items=");
371      ctx.indent++;
372      serializeCollection(out, o, type, ctx);
373      ctx.indent--;
374
375      if (o.size() > 0)
376         out.cr(i-1);
377      out.append(')');
378
379      return out;
380   }

382   @SuppressWarnings({ "rawtypes" })
383   private SerializerWriter serializeBeanMap(UonSerializerWriter out, BeanMap m, boolean addClassAttr, UonSerializerContext ctx) throws IOException, SerializeException {
384      int depth = ctx.getIndent();
385
386      out.startFlag('o');
387
388      Iterator mapEntries = m.entrySet().iterator();
389
390      // Print out "_class" attribute on this bean if required.
391      if (addClassAttr) {
392         String attr = "_class";
393         out.cr(depth).appendObject(attr, false, false, false).append('=').append(m.getClassMeta().getInnerClass().getName());
394         if (mapEntries.hasNext())
395            out.append(',');
396      }
397
398      boolean addComma = false;
399
400      while (mapEntries.hasNext()) {
401         BeanMapEntry p = (BeanMapEntry)mapEntries.next();
402         BeanPropertyMeta pMeta = p.getMeta();
403
404         String key = p.getKey();
405         Object value = null;
406         try {
407            value = p.getValue();
408         } catch (StackOverflowError e) {
409            throw e;
410         } catch (Throwable t) {
411            ctx.addBeanGetterWarning(pMeta, t);
412         }
413
414         if (ctx.canIgnoreValue(pMeta.getClassMeta(), key, value))
415            continue;
416
417         if (addComma)
418            out.append(',');
419
420         out.cr(depth).appendObject(key, false, false, false).append('=');
421
422         serializeAnything(out, value, pMeta.getClassMeta(), ctx, key, pMeta, false, false);
423
424         addComma = true;
425      }
426
427      if (m.size() > 0)
428         out.cr(depth-1);
429      out.append(')');
430
431      return out;
432   }

434   @SuppressWarnings({ "rawtypes", "unchecked" })
435   private SerializerWriter serializeCollection(UonSerializerWriter out, Collection c, ClassMeta<?> type, UonSerializerContext ctx) throws IOException, SerializeException {
436
437      ClassMeta<?> elementType = type.getElementType();
438
439      c = ctx.sort(c);
440
441      out.startFlag('a');
442
443      int depth = ctx.getIndent();
444      boolean quoteEmptyString = (c.size() == 1 || ctx.isUseWhitespace());
445
446      for (Iterator i = c.iterator(); i.hasNext();) {
447         out.cr(depth);
448         serializeAnything(out, i.next(), elementType, ctx, "<iterator>", null, quoteEmptyString, false);
449         if (i.hasNext())
450            out.append(',');
451      }
452
453      if (c.size() > 0)
454         out.cr(depth-1);
455      out.append(')');
456
457      return out;
458   }

464   @Override /* Serializer */
465   public UonSerializerContext createContext(ObjectMap properties, Method javaMethod) {
466      return new UonSerializerContext(getBeanContext(), sp, usp, uep, properties, javaMethod);
467   }

469   @Override /* Serializer */
470   protected void doSerialize(Object o, Writer out, SerializerContext ctx) throws IOException, SerializeException {
471      UonSerializerContext uctx = (UonSerializerContext)ctx;
472      serializeAnything(uctx.getWriter(out), o, null, uctx, "root", null, false, true);
473   }

475   @Override /* CoreApi */
476   public UonSerializer setProperty(String property, Object value) throws LockedException {
477      checkLock();
478      if (! usp.setProperty(property, value))
479         if (! uep.setProperty(property, value))
480            super.setProperty(property, value);
481      return this;
482   }
483
484   @Override /* CoreApi */
485   public UonSerializer setProperties(ObjectMap properties) throws LockedException {
486      for (Map.Entry<String,Object> e : properties.entrySet())
487         setProperty(e.getKey(), e.getValue());
488      return this;
489   }
