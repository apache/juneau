009package com.ibm.juno.core.xml;
010
011import java.util.*;
012
013import com.ibm.juno.core.utils.*;
014import com.ibm.juno.core.xml.annotation.*;
015
016
017/**
018 * Metadata on classes specific to the XML serializers and parsers pulled from the {@link Xml @Xml} annotation on the class.
019 *
020 * @author James Bognar (jbognar@us.ibm.com)
021 */
022public class XmlClassMeta {
023
024   private final Namespace namespace;
025   private final Xml xml;
026   private final XmlFormat format;
027   private final String elementName;
028   private final String childName;
029
030   /**
031    * Constructor.
032    *
033    * @param c The class that this annotation is defined on.
034    */
035   public XmlClassMeta(Class<?> c) {
036      this.namespace = findNamespace(c);
037      this.xml =  ReflectionUtils.getAnnotation(Xml.class, c);
038      if (xml != null) {
039         this.format = xml.format();
040         this.elementName = StringUtils.nullIfEmpty(xml.name());
041         this.childName = StringUtils.nullIfEmpty(xml.childName());
042
043      } else {
044         this.format = XmlFormat.NORMAL;
045         this.elementName = null;
046         this.childName = null;
047      }
048   }
049
050   /**
051    * Returns the {@link Xml} annotation defined on the class.
052    *
053    * @return The value of the {@link Xml} annotation defined on the class, or <jk>null</jk> if annotation is not specified.
054    */
055   protected Xml getAnnotation() {
056      return xml;
057   }
058
059   /**
060    * Returns the {@link Xml#format()} annotation defined on the class.
061    *
062    * @return The value of the {@link Xml#format()} annotation, or {@link XmlFormat#NORMAL} if not specified.
063    */
064   protected XmlFormat getFormat() {
065      return format;
066   }
067
068   /**
069    * Returns the {@link Xml#name()} annotation defined on the class.
070    *
071    * @return The value of the {@link Xml#name()} annotation, or <jk>null</jk> if not specified.
072    */
073   protected String getElementName() {
074      return elementName;
075   }
076
077   /**
078    * Returns the {@link Xml#childName()} annotation defined on the class.
079    *
080    * @return The value of the {@link Xml#childName()} annotation, or <jk>null</jk> if not specified.
081    */
082   protected String getChildName() {
083      return childName;
084   }
085
086   /**
087    * Returns the XML namespace associated with this class.
088    * <p>
089    *    Namespace is determined in the following order:
090    * <ol>
091    *    <li>{@link Xml#prefix()} annotation defined on class.
092    *    <li>{@link Xml#prefix()} annotation defined on package.
093    *    <li>{@link Xml#prefix()} annotation defined on superclasses.
094    *    <li>{@link Xml#prefix()} annotation defined on superclass packages.
095    *    <li>{@link Xml#prefix()} annotation defined on interfaces.
096    *    <li>{@link Xml#prefix()} annotation defined on interface packages.
097    * </ol>
098    *
099    * @return The namespace associated with this class, or <jk>null</jk> if no namespace is
100    *    associated with it.
101    */
102   protected Namespace getNamespace() {
103      return namespace;
104   }
105
106   private Namespace findNamespace(Class<?> c) {
107      if (c == null)
108         return null;
109
110      List<Xml> xmls = ReflectionUtils.findAnnotations(Xml.class, c);
111      List<XmlSchema> schemas = ReflectionUtils.findAnnotations(XmlSchema.class, c);
112      return XmlUtils.findNamespace(xmls, schemas);
113   }
114}
