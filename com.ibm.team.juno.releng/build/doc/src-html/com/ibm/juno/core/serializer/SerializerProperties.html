009package com.ibm.juno.core.serializer;
010
011import com.ibm.juno.core.*;
012
013/**
014 * Configurable properties common to all {@link Serializer} classes.
015 * <p>
016 *    Use the {@link Serializer#setProperty(String, Object)} method to set property values.
017 *
018 * @author James Bognar (jbognar@us.ibm.com)
019 */
020public final class SerializerProperties implements Cloneable {
021
022   /**
023    * Max depth ({@link Integer}, default=<code>100</code>).
024    * <p>
025    * Abort serialization if specified depth is reached in the POJO tree.
026    * If this depth is exceeded, an exception is thrown.
027    * This prevents stack overflows from occurring when trying to serialize models with recursive references.
028    */
029   public static final String SERIALIZER_maxDepth = "Serializer.maxDepth";
030
031   /**
032    * Initial depth ({@link Integer}, default=<code>0</code>).
033    * <p>
034    * The initial indentation level at the root.
035    * Useful when constructing document fragments that need to be indented at a certain level.
036    */
037   public static final String SERIALIZER_initialDepth = "Serializer.initialDepth";
038
039   /**
040    * Automatically detect POJO recursions ({@link Boolean}, default=<jk>false</jk>).
041    * <p>
042    * Specifies that recursions should be checked for during serialization.
043    * <p>
044    * Recursions can occur when serializing models that aren't true trees, but rather contain loops.
045    * <p>
046    * The behavior when recursions are detected depends on the value for {@link #SERIALIZER_ignoreRecursions}.
047    * <p>
048    * For example, if a model contains the links A-&gt;B-&gt;C-&gt;A, then the JSON generated will look like
049    *    the following when <jsf>SERIALIZER_ignoreRecursions</jsf> is <jk>true</jk>...
050    * <code>{A:{B:{C:null}}}</code><br>
051    * <p>
052    * Note:  Checking for recursion can cause a small performance penalty.
053    */
054   public static final String SERIALIZER_detectRecursions = "Serializer.detectRecursions";
055
056   /**
057    * Ignore recursion errors ({@link Boolean}, default=<jk>false</jk>).
058    * <p>
059    * Used in conjunction with {@link #SERIALIZER_detectRecursions}.
060    * Setting is ignored if <jsf>SERIALIZER_detectRecursions</jsf> is <jk>false</jk>.
061    * <p>
062    * If <jk>true</jk>, when we encounter the same object when serializing a tree,
063    *    we set the value to <jk>null</jk>.
064    * Otherwise, an exception is thrown.
065    */
066   public static final String SERIALIZER_ignoreRecursions = "Serializer.ignoreRecursions";
067
068   /**
069    * Debug mode ({@link Boolean}, default=<jk>false</jk>).
070    * <p>
071    * Enables the following additional information during serialization:
072    * <ul>
073    *    <li>When bean getters throws exceptions, the exception includes the object stack information
074    *       in order to determine how that method was invoked.
075    *    <li>Enables {#link SERIALIZER_detectRecursions}.
076    * </ul>
077    */
078   public static final String SERIALIZER_debug = "Serializer.debug";
079
080   /**
081    * Use indentation in output ({@link Boolean}, default=<jk>false</jk>).
082    * <p>
083    * If <jk>true</jk>, newlines and indentation is added to the output to improve readability.
084    */
085   public static final String SERIALIZER_useIndentation = "Serializer.useIndentation";
086
087   /**
088    * Add class attributes to output ({@link Boolean}, default=<jk>false</jk>).
089    * <p>
090    * If <jk>true</jk>, then <js>"_class"</js> attributes will be added to beans if their type cannot be inferred through reflection.
091    * This is used to recreate the correct objects during parsing if the object types cannot be inferred.
092    * For example, when serializing a {@code Map<String,Object>} field, where the bean class cannot be determined from the value type.
093    */
094   public static final String SERIALIZER_addClassAttrs = "Serializer.addClassAttrs";
095
096   /**
097    * Quote character ({@link Character}, default=<js>'"'</js>.
098    * <p>
099    * This is the character used for quoting attributes and values.
100    */
101   public static final String SERIALIZER_quoteChar = "Serializer.quoteChar";
102
103   /**
104    * Boolean.  Trim null bean property values from output.
105    * <p>
106    *    If <jk>true</jk>, null bean values will not be serialized to the output.
107    * <p>
108    *    Note that enabling this setting has the following effects on parsing:
109    *    <ul>
110    *       <li>Map entries with <jk>null</jk> values will be lost.
111    *    </ul>
112    * <p>
113    *    Default is <jk>true</jk>.
114    */
115   public static final String SERIALIZER_trimNullProperties = "Serializer.trimNullProperties";
116
117   /**
118    * Boolean.  Trim empty lists and arrays from output.
119    * <p>
120    *    If <jk>true</jk>, empty list values will not be serialized to the output.
121    * <p>
122    *    Note that enabling this setting has the following effects on parsing:
123    *    <ul>
124    *       <li>Map entries with empty list values will be lost.
125    *       <li>Bean properties with empty list values will not be set.
126    *    </ul>
127    * <p>
128    *    Default is <jk>false</jk>.
129    */
130   public static final String SERIALIZER_trimEmptyLists = "Serializer.trimEmptyLists";
131
132   /**
133    * Boolean.  Trim empty maps from output.
134    * <p>
135    *    If <jk>true</jk>, empty map values will not be serialized to the output.
136    * <p>
137    *    Note that enabling this setting has the following effects on parsing:
138    *    <ul>
139    *       <li>Bean properties with empty map values will not be set.
140    *    </ul>
141    * <p>
142    *    Default is <jk>false</jk>.
143    */
144   public static final String SERIALIZER_trimEmptyMaps = "Serializer.trimEmptyMaps";
145
146   /**
147    * Boolean.  Trim strings in output.
148    * <p>
149    *    If <jk>true</jk>, string values will be trimmed of whitespace using {@link String#trim()} before being serialized.
150    * <p>
151    *    Default is <jk>false</jk>.
152    */
153   public static final String SERIALIZER_trimStrings = "Serializer.trimStrings";
154
155   /**
156    * String.  URI base for relative URIs.
157    * <p>
158    *    Prepended to relative URIs during serialization (along with the {@link #SERIALIZER_absolutePathUriBase} if specified.
159    *    (i.e. URIs not containing a schema and not starting with <js>'/'</js>).
160    *    (e.g. <js>"foo/bar"</js>)
161    * <p>
162    *    Default is <js>""</js>.
163    *
164    * <dl>
165    *    <dt>Examples:</dt>
166    *    <dd>
167    *       <table class='styled'>
168    *          <tr><th>SERIALIZER_relativeUriBase</th><th>URI</th><th>Serialized URI</th></tr>
169    *          <tr>
170    *             <td><code>http://foo:9080/bar/baz</code></td>
171    *             <td><code>mywebapp</code></td>
172    *             <td><code>http://foo:9080/bar/baz/mywebapp</code></td>
173    *          </tr>
174    *          <tr>
175    *             <td><code>http://foo:9080/bar/baz</code></td>
176    *             <td><code>/mywebapp</code></td>
177    *             <td><code>/mywebapp</code></td>
178    *          </tr>
179    *          <tr>
180    *             <td><code>http://foo:9080/bar/baz</code></td>
181    *             <td><code>http://mywebapp</code></td>
182    *             <td><code>http://mywebapp</code></td>
183    *          </tr>
184    *       </table>
185    *    </dd>
186    * </dl>
187    */
188   public static final String SERIALIZER_relativeUriBase = "Serializer.relativeUriBase";
189
190   /**
191    * Boolean.  Sort arrays and collections alphabetically before serializing.
192    * <p>
193    *    Note that this introduces a performance penalty.
194    * <p>
195    *    Default is <jk>false</jk>.
196    */
197   public static final String SERIALIZER_sortCollections = "Serializer.sortCollections";
198
199   /**
200    * Boolean.  Sort maps alphabetically before serializing.
201    * <p>
202    *    Note that this introduces a performance penalty.
203    * <p>
204    *    Default is <jk>false</jk>.
205    */
206   public static final String SERIALIZER_sortMaps = "Serializer.sortMaps";
207
208   /**
209    * String.  URI base for relative URIs with absolute paths.
210    * <p>
211    *    Prepended to relative absolute-path URIs during serialization.
212    *    (i.e. URIs starting with <js>'/'</js>).
213    *    (e.g. <js>"/foo/bar"</js>)
214    * <p>
215    *    Default is <js>""</js>.
216    *
217    * <dl>
218    *    <dt>Examples:</dt>
219    *    <dd>
220    *       <table class='styled'>
221    *          <tr><th>SERIALIZER_absolutePathUriBase</th><th>URI</th><th>Serialized URI</th></tr>
222    *          <tr>
223    *             <td><code>http://foo:9080/bar/baz</code></td>
224    *             <td><code>mywebapp</code></td>
225    *             <td><code>mywebapp</code></td>
226    *          </tr>
227    *          <tr>
228    *             <td><code>http://foo:9080/bar/baz</code></td>
229    *             <td><code>/mywebapp</code></td>
230    *             <td><code>http://foo:9080/bar/baz/mywebapp</code></td>
231    *          </tr>
232    *          <tr>
233    *             <td><code>http://foo:9080/bar/baz</code></td>
234    *             <td><code>http://mywebapp</code></td>
235    *             <td><code>http://mywebapp</code></td>
236    *          </tr>
237    *       </table>
238    *    </dd>
239    * </dl>
240    */
241   public static final String SERIALIZER_absolutePathUriBase = "Serializer.absolutePathUriBase";
242
243   int maxDepth = 100, initialDepth = 0;
244   boolean
245      debug = false,
246      detectRecursions = false,
247      ignoreRecursions = false,
248      useIndentation = false,
249      addClassAttrs = false,
250      trimNulls = true,
251      trimEmptyLists = false,
252      trimEmptyMaps = false,
253      trimStrings = false,
254      sortCollections = false,
255      sortMaps = false;
256   char quoteChar = '"';
257   String relativeUriBase="", absolutePathUriBase="";
258
259   /**
260    * Sets the specified property value.
261    * @param property The property name.
262    * @param value The property value.
263    * @return <jk>true</jk> if property name was valid and property was set.
264    */
265   public boolean setProperty(String property, Object value) {
266      BeanContext bc = BeanContext.DEFAULT;
267      if (property.equals(SERIALIZER_maxDepth))
268         maxDepth = bc.convertToType(value, Integer.class);
269      else if (property.equals(SERIALIZER_debug))
270         debug = bc.convertToType(value, Boolean.class);
271      else if (property.equals(SERIALIZER_detectRecursions))
272         detectRecursions = bc.convertToType(value, Boolean.class);
273      else if (property.equals(SERIALIZER_ignoreRecursions))
274         ignoreRecursions = bc.convertToType(value, Boolean.class);
275      else if (property.equals(SERIALIZER_useIndentation))
276         useIndentation = bc.convertToType(value, Boolean.class);
277      else if (property.equals(SERIALIZER_addClassAttrs))
278         addClassAttrs = bc.convertToType(value, Boolean.class);
279      else if (property.equals(SERIALIZER_quoteChar))
280         quoteChar = bc.convertToType(value, Character.class);
281      else if (property.equals(SERIALIZER_trimNullProperties))
282         trimNulls = bc.convertToType(value, Boolean.class);
283      else if (property.equals(SERIALIZER_trimEmptyLists))
284         trimEmptyLists = bc.convertToType(value, Boolean.class);
285      else if (property.equals(SERIALIZER_trimEmptyMaps))
286         trimEmptyMaps = bc.convertToType(value, Boolean.class);
287      else if (property.equals(SERIALIZER_trimStrings))
288         trimStrings = bc.convertToType(value, Boolean.class);
289      else if (property.equals(SERIALIZER_relativeUriBase))
290         relativeUriBase = value == null ? null : value.toString();
291      else if (property.equals(SERIALIZER_absolutePathUriBase))
292         absolutePathUriBase = value == null ? null : value.toString();
293      else if (property.equals(SERIALIZER_sortCollections))
294         sortCollections = bc.convertToType(value, Boolean.class);
295      else if (property.equals(SERIALIZER_sortMaps))
296         sortMaps = bc.convertToType(value, Boolean.class);
297      else
298         return false;
299      return true;
300   }

307   @Override /* Cloneable */
308   public SerializerProperties clone() {
309      try {
310         return (SerializerProperties)super.clone();
311      } catch (CloneNotSupportedException e) {
312         throw new RuntimeException(e); // Won't happen.
313      }
314   }
