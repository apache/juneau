261   /**
262    * Same as {@link #getObject(Class, String, Object)}, but value is referenced through section name and key instead of full key.
263    *
264    * @param c The class to convert the value to.
265    * @param sectionName - The section name.  Must not be <jk>null</jk>.
266    * @param sectionKey - The section key.  Must not be <jk>null</jk>.
267    * @param def The default value if section or key does not exist.
268    * @throws ParseException If parser could not parse the value or if a parser is not registered with this config file.
269    * @return The value, or the default value if the section or value doesn't exist.
270    */
271   @SuppressWarnings("unchecked")
272   public <T> T getObject(Class<T> c, String sectionName, String sectionKey, T def) throws ParseException {
273      String s = get(sectionName, sectionKey);
274      if (s == null)
275         return def;
276      if (c == String.class)
277         return (T)s;
278      if (c == Integer.class || c == int.class)
279         return (T)(StringUtils.isEmpty(s) ? def : Integer.valueOf(parseIntWithSuffix(s)));
280      if (c == Boolean.class || c == boolean.class)
281         return (T)(StringUtils.isEmpty(s) ? def : Boolean.valueOf(Boolean.parseBoolean(s)));
282      if (c == String[].class) {
283         String[] r = StringUtils.isEmpty(s) ? new String[0] : StringUtils.split(s, ',');
284         return (T)(r.length == 0 ? def : r);
285      }
286      if (c.isArray()) {
287         Class<?> ce = c.getComponentType();
288         if (StringUtils.isEmpty(s))
289            return def;
290         String[] r = StringUtils.split(s, ',');
291         Object o = Array.newInstance(ce, r.length);
292         for (int i = 0; i < r.length; i++)
293            Array.set(o, i, getParser().parse(r[i], ce));
294         return (T)o;
295      }
296      if (StringUtils.isEmpty(s))
297         return def;
298      return getParser().parse(s, c);
299   }

415   /**
416    * Adds or replaces an entry with the specified key with the specified value.
417    * <p>
418    * The format of the entry depends on the data type of the value.
419    * <ul>
420    *    <li>Simple types (<code>String</code>, <code>Number</code>, <code>Boolean</code>, primitives)
421    *       are serialized as plain strings.
422    *    <li>Arrays and collections of simple types are serialized as comma-delimited lists of plain strings.
423    *    <li>Other types (e.g. beans) are serialized using the serializer registered with this config file.
424    *    <li>Arrays and collections of other types are serialized as comma-delimited lists of serialized strings of each entry.
425    * </ul>
426    *
427    * @param key The key.  See {@link #getString(String)} for a description of the key.
428    * @param value The new value.
429    * @param encoded If <jk>true</jk>, value is encoded by the registered encoder when the config file is persisted to disk.
430    * @return The previous value, or <jk>null</jk> if the section or key did not previously exist.
431    * @throws SerializeException If serializer could not serialize the value or if a serializer is not registered with this config file.
432    * @throws UnsupportedOperationException If config file is read only.
433    */
434   public final String put(String key, Object value, boolean encoded) throws SerializeException {
435      assertFieldNotNull(key, "key");
436      if (value == null)
437         value = "";
438      Class<?> c = value.getClass();
439      if (isSimpleType(c))
440         return put(getSectionName(key), getSectionKey(key), value.toString(), encoded);
441      if (c.isAssignableFrom(Collection.class)) {
442         Collection<?> c2 = (Collection<?>)value;
443         String[] r = new String[c2.size()];
444         int i = 0;
445         for (Object o2 : c2) {
446            boolean isSimpleType = o2 == null ? true : isSimpleType(o2.getClass());
447            r[i++] = (isSimpleType ? Array.get(value, i).toString() : getSerializer().toString(Array.get(value, i)));
448         }
449         return put(getSectionName(key), getSectionKey(key), StringUtils.join(r, ','), encoded);
450      } else if (c.isArray()) {
451         boolean isSimpleType = isSimpleType(c.getComponentType());
452         String[] r = new String[Array.getLength(value)];
453         for (int i = 0; i < r.length; i++) {
454            r[i] = (isSimpleType ? Array.get(value, i).toString() : getSerializer().toString(Array.get(value, i)));
455         }
456         return put(getSectionName(key), getSectionKey(key), StringUtils.join(r, ','), encoded);
457      }
458      return put(getSectionName(key), getSectionKey(key), getSerializer().toString(value), encoded);
459   }

461   private final boolean isSimpleType(Class<?> c) {
462      return (c == String.class || c.isPrimitive() || c.isAssignableFrom(Number.class) || c == Boolean.class);
463   }

465   /**
466    * Returns the specified section as a map of key/value pairs.
467    *
468    * @param sectionName The section name to retrieve.
469    * @return A map of the section, or <jk>null</jk> if the section was not found.
470    */
471   public final ObjectMap getSectionMap(String sectionName) {
472      readLock();
473      try {
474         Set<String> keys = getSectionKeys(sectionName);
475         if (keys == null)
476            return null;
477         ObjectMap m = new ObjectMap();
478         for (String key : keys)
479            m.put(key, get(sectionName, key));
480         return m;
481      } finally {
482         readUnlock();
483      }
484   }

486   /**
487    * Copies the entries in a section to the specified bean by calling the public setters on that bean.
488    *
489    * @param sectionName The section name to write from.
490    * @param bean The bean to set the properties on.
491    * @param ignoreUnknownProperties If <jk>true</jk>, don't throw an {@link IllegalArgumentException} if this section
492    *    contains a key that doesn't correspond to a setter method.
493    * @param permittedPropertyTypes If specified, only look for setters whose property types
494    *    are those listed.  If not specified, use all setters.
495    * @return An object map of the changes made to the bean.
496    * @throws ParseException If parser was not set on this config file or invalid properties were found in the section.
497    * @throws IllegalArgumentException
498    * @throws IllegalAccessException
499    * @throws InvocationTargetException
500    */
501   public final ObjectMap writeProperties(String sectionName, Object bean, boolean ignoreUnknownProperties, Class<?>...permittedPropertyTypes) throws ParseException, IllegalArgumentException, IllegalAccessException, InvocationTargetException {
502      assertFieldNotNull(bean, "bean");
503      ObjectMap om = new ObjectMap();
504      readLock();
505      try {
506         Set<String> keys = getSectionKeys(sectionName);
507         if (keys == null)
508            throw new IllegalArgumentException("Section not found");
509         keys = new LinkedHashSet<String>(keys);
510         for (Method m : bean.getClass().getMethods()) {
511            int mod = m.getModifiers();
512            if (isPublic(mod) && (!isStatic(mod)) && m.getName().startsWith("set") && m.getParameterTypes().length == 1) {
513               Class<?> pt = m.getParameterTypes()[0];
514               if (permittedPropertyTypes == null || permittedPropertyTypes.length == 0 || ArrayUtils.contains(pt, permittedPropertyTypes)) {
515                  String propName = Introspector.decapitalize(m.getName().substring(3));
516                  Object value = getObject(pt, sectionName, propName, null);
517                  if (value != null) {
518                     m.invoke(bean, value);
519                     om.put(propName, value);
520                     keys.remove(propName);
521                  }
522               }
523            }
524         }
525         if (! (ignoreUnknownProperties || keys.isEmpty()))
526            throw new ParseException("Invalid properties found in config file section ["+sectionName+"]: " + JsonSerializer.DEFAULT_LAX.toString(keys));
527         return om;
528      } finally {
529         readUnlock();
530      }
531   }

545   /**
546    * Converts this config file section to the specified bean instance.
547    *
548    * @param sectionName The section name to write from.
549    * @param c The bean class to create.
550    * @param ignoreUnknownProperties If <jk>false</jk>, throws a {@link ParseException} if
551    *    the section contains an entry that isn't a bean property name.
552    * @return A new bean instance.
553    * @throws ParseException
554    */
555   public final <T> T getSectionAsBean(String sectionName, Class<T> c, boolean ignoreUnknownProperties) throws ParseException {
556      assertFieldNotNull(c, "c");
557      readLock();
558      try {
559         BeanMap<T> bm = getParser().getBeanContext().newBeanMap(c);
560         for (String k : getSectionKeys(sectionName)) {
561            BeanPropertyMeta<?> bpm = bm.getPropertyMeta(k);
562            if (bpm == null) {
563               if (! ignoreUnknownProperties)
564                  throw new ParseException("Unknown property {0} encountered", k);
565            } else {
566               bm.put(k, getObject(bpm.getClassMeta().getInnerClass(), sectionName + '/' + k));
567            }
568         }
569         return bm.getBean();
570      } finally {
571         readUnlock();
572      }
573   }

757    * Create an instance of a {@link StringVarWithDefault} that returns values in this config file.
758    *
759    * @return A new {@link StringVarWithDefault}.
760    */
761   public StringVar getStringVar() {
762      if (stringVar == null) {
763         synchronized(this) {
764            if (stringVar == null) {
765               stringVar = new StringVarWithDefault() {
766                  @Override /* StringVar */
767                  public String resolve(String varVal) {
768                     return ConfigFile.this.getString(varVal);
769                  }
770               };
771            }
772         }
773      }
774      return stringVar;
775   }

777   private int parseIntWithSuffix(String s) {
778      assertFieldNotNull(s, "s");
779      int m = 1;
780      if (s.endsWith("M")) {
781         m = 1024*1024;
782         s = s.substring(0, s.length()-1).trim();
783      } else if (s.endsWith("K")) {
784         m = 1024;
785         s = s.substring(0, s.length()-1).trim();
786      }
787      return Integer.parseInt(s) * m;
788   }
