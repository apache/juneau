009package com.ibm.juno.core.utils;
010
011import java.io.*;
012
013/**
014 * Similar to {@link StringWriter}, but uses a {@link StringBuilder} instead to avoid synchronization overhead.
015 * <p>
016 * Note that this class is NOT thread safe.
017 *
018 * @author James Bognar (jbognar@us.ibm.com)
019 */
020public final class StringBuilderWriter extends Writer {
021
022   private StringBuilder sb;
023
024   /**
025    * Create a new string writer using the default initial string-builder size.
026    */
027   public StringBuilderWriter() {
028      sb = new StringBuilder();
029      lock = null;
030   }
031
032   /**
033    * Create a new string writer using the specified initial string-builder size.
034    *
035    * @param initialSize The number of <tt>char</tt> values that will fit into this buffer before it is automatically expanded
036    * @throws IllegalArgumentException If <tt>initialSize</tt> is negative
037    */
038   public StringBuilderWriter(int initialSize) {
039      sb = new StringBuilder(initialSize);
040      lock = null;
041   }
042
043   @Override /* Writer */
044   public void write(int c) {
045      sb.append((char) c);
046   }
047
048   @Override /* Writer */
049   public void write(char cbuf[], int start, int length) {
050      sb.append(cbuf, start, length);
051   }
052
053   @Override /* Writer */
054   public void write(String str) {
055      sb.append(str);
056   }
057
058   @Override /* Writer */
059   public void write(String str, int off, int len) {
060      sb.append(str.substring(off, off + len));
061   }
062
063   @Override /* Writer */
064   public StringBuilderWriter append(CharSequence csq) {
065      if (csq == null)
066         write("null");
067      else
068         write(csq.toString());
069      return this;
070   }
071
072   @Override /* Writer */
073   public StringBuilderWriter append(CharSequence csq, int start, int end) {
074      CharSequence cs = (csq == null ? "null" : csq);
075      write(cs.subSequence(start, end).toString());
076      return this;
077   }
078
079   @Override /* Writer */
080   public StringBuilderWriter append(char c) {
081      write(c);
082      return this;
083   }
084
085   @Override /* Object */
086   public String toString() {
087      return sb.toString();
088   }
089
090   @Override /* Writer */
091   public void flush() {}
092
093   @Override /* Writer */
094   public void close() throws IOException {}
095}
