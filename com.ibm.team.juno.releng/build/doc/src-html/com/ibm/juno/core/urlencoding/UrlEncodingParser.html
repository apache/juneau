023/**
024 * Parses URL-encoded text into POJO models.
025 *
026 *
027 * <h6 class='topic'>Media types</h6>
028 * <p>
029 *    Handles <code>Content-Type</code> types: <code>application/x-www-form-urlencoded</code>
030 *
031 *
032 * <h6 class='topic'>Description</h6>
033 * <p>
034 *    Parses URL-Encoded text (e.g. <js>"foo=bar&baz=bing"</js>) into POJOs.
035 * <p>
036 *    Expects parameter values to be in UON notation.
037 * <p>
038 *    This parser uses a state machine, which makes it very fast and efficient.
039 *
040 *
041 * <h6 class='topic'>Configurable properties</h6>
042 * <p>
043 *    This class has the following properties associated with it:
044 * <ul>
045 *    <li>{@link UonParserProperties}
046 *    <li>{@link ParserProperties}
047 *    <li>{@link BeanContextProperties}
048 * </ul>
049 *
050 *
051 * @author James Bognar (jbognar@us.ibm.com)
052 */
053@SuppressWarnings({ "rawtypes", "unchecked", "hiding" })
054@Consumes("application/x-www-form-urlencoded")
055public class UrlEncodingParser extends UonParser {

070   private <T> T parseAnything(ClassMeta<T> nt, UonParserContext ctx, ParserReader r, Object outer, Object name) throws ParseException {
071
072      BeanContext bc = ctx.getBeanContext();
073      if (nt == null)
074         nt = (ClassMeta<T>)object();
075      PojoFilter<T,Object> filter = (PojoFilter<T,Object>)nt.getPojoFilter();
076      ClassMeta<?> ft = nt.getFilteredClassMeta();
077
078      try {
079         int c = r.peek();
080         if (c == '?')
081            r.read();
082
083         Object o;
084
085         if (ft.isObject()) {
086            ObjectMap m = new ObjectMap(bc);
087            parseIntoMap(ctx, r, m, bc.string(), bc.object());
088            o = m.cast();
089         } else if (ft.isMap()) {
090            Map m = (ft.canCreateNewInstance() ? (Map)ft.newInstance() : new ObjectMap(bc));
091            o = parseIntoMap(ctx, r, m, ft.getKeyType(), ft.getValueType());
092         } else if (ft.canCreateNewInstanceFromObjectMap(outer)) {
093            ObjectMap m = new ObjectMap(bc);
094            parseIntoMap(ctx, r, m, string(), object());
095            o = ft.newInstanceFromObjectMap(outer, m);
096         } else if (ft.canCreateNewBean(outer)) {
097            BeanMap m = bc.newBeanMap(outer, ft.getInnerClass());
098            m = parseIntoBeanMap(ctx, r, m);
099            o = m == null ? null : m.getBean();
100         } else {
101            // It could be a non-bean with _class attribute.
102            ObjectMap m = new ObjectMap(bc);
103            ClassMeta<Object> valueType = object();
104            parseIntoMap(ctx, r, m, string(), valueType);
105            if (m.containsKey("_class"))
106               o = m.cast();
107            else if (m.containsKey("_value"))
108               o = ctx.getBeanContext().convertToType(m.get("_value"), ft);
109            else if (ft.isCollection()) {
110               // ?1=foo&2=bar...
111               Collection c2 = ft.canCreateNewInstance() ? (Collection)ft.newInstance() : new ObjectList(bc);
112               Map<Integer,Object> t = new TreeMap<Integer,Object>();
113               for (Map.Entry<String,Object> e : m.entrySet()) {
114                  String k = e.getKey();
115                  if (StringUtils.isNumeric(k))
116                     t.put(Integer.valueOf(k), bc.convertToType(e.getValue(), ft.getElementType()));
117               }
118               c2.addAll(t.values());
119               o = c2;
120            } else {
121               if (ft.getNotABeanReason() != null)
122                  throw new ParseException("Class ''{0}'' could not be instantiated as application/x-www-form-urlencoded.  Reason: ''{1}''", ft, ft.getNotABeanReason());
123               throw new ParseException("Malformed application/x-www-form-urlencoded input for class ''{0}''.", ft);
124            }
125         }
126
127         if (filter != null && o != null)
128            o = filter.unfilter(o, nt);
129
130         if (outer != null)
131            setParent(nt, o, outer);
132
133         if (name != null)
134            setName(nt, o, name);
135
136         return (T)o;
137
138      } catch (RuntimeException e) {
139         throw e;
140      } catch (Exception e) {
141         throw new ParseException("Error occurred trying to parse into class ''{0}''", ft).initCause(e);
142      }
143   }

145   private <K,V> Map<K,V> parseIntoMap(UonParserContext ctx, ParserReader r, Map<K,V> m, ClassMeta<K> keyType, ClassMeta<V> valueType) throws ParseException, IOException {
146
147      if (keyType == null)
148         keyType = (ClassMeta<K>)string();
149
150      int c = r.peek();
151      if (c == -1)
152         return m;
153
154      final int S1=1; // Looking for attrName start.
155      final int S2=2; // Found attrName end, looking for =.
156      final int S3=3; // Found =, looking for valStart.
157      final int S4=4; // Looking for & or end.
158      boolean isInEscape = false;
159
160      int state = S1;
161      K currAttr = null;
162      while (c != -1) {
163         c = r.read();
164         if (! isInEscape) {
165            if (state == S1) {
166               if (c == -1)
167                  return m;
168               r.unread();
169               Object attr = parseAttr(r, true, ctx);
170               currAttr = ctx.trim(ctx.getBeanContext().convertToType(attr, keyType));
171               state = S2;
172               c = 0; // Avoid isInEscape if c was '\'
173            } else if (state == S2) {
174               if (c == '\u0002')
175                  state = S3;
176               else if (c == -1 || c == '\u0001') {
177                  m.put(currAttr, null);
178                  if (c == -1)
179                     return m;
180                  state = S1;
181               }
182            } else if (state == S3) {
183               if (c == -1 || c == '\u0001') {
184                  V value = convertAttrToType(m, "", valueType);
185                  m.put(currAttr, value);
186                  if (c == -1)
187                     return m;
188                  state = S1;
189               } else  {
190                  // For performance, we bypass parseAnything for string values.
191                  V value = (V)(valueType.isString() ? super.parseString(r.unread(), true, ctx) : super.parseAnything(valueType, ctx, r.unread(), null, m, true, null));
192
193                  // If we already encountered this parameter, turn it into a list.
194                  if (m.containsKey(currAttr) && valueType.isObject()) {
195                     Object v2 = m.get(currAttr);
196                     if (! (v2 instanceof ObjectList)) {
197                        v2 = new ObjectList(v2);
198                        m.put(currAttr, (V)v2);
199                     }
200                     ((ObjectList)v2).add(value);
201                  } else {
202                     m.put(currAttr, value);
203                  }
204                  state = S4;
205                  c = 0; // Avoid isInEscape if c was '\'
206               }
207            } else if (state == S4) {
208               if (c == '\u0001')
209                  state = S1;
210               else if (c == -1) {
211                  return m;
212               }
213            }
214         }
215         isInEscape = (c == '\\' && ! isInEscape);
216      }
217      if (state == S1)
218         throw new ParseException("Could not find attribute name on object.");
219      if (state == S2)
220         throw new ParseException("Could not find '=' following attribute name on object.");
221      if (state == S3)
222         throw new ParseException("Dangling '=' found in object entry");
223      if (state == S4)
224         throw new ParseException("Could not find end of object.");
225
226      return null; // Unreachable.
227   }

229   private <T> BeanMap<T> parseIntoBeanMap(UonParserContext ctx, ParserReader r, BeanMap<T> m) throws ParseException, IOException {
230      int line = r.getLine();
231      int column = r.getColumn();
232
233      int c = r.peek();
234      if (c == -1)
235         return m;
236
237      final int S1=1; // Looking for attrName start.
238      final int S2=2; // Found attrName end, looking for =.
239      final int S3=3; // Found =, looking for valStart.
240      final int S4=4; // Looking for , or }
241      boolean isInEscape = false;
242
243      int state = S1;
244      String currAttr = "";
245      int currAttrLine = -1, currAttrCol = -1;
246      while (c != -1) {
247         c = r.read();
248         if (! isInEscape) {
249            if (state == S1) {
250               if (c == -1) {
251                  return m;
252               }
253               r.unread();
254               currAttrLine= r.getLine();
255               currAttrCol = r.getColumn();
256               currAttr = parseAttrName(ctx, r, true);
257               if (currAttr == null)  // Value was '%00'
258                  return null;
259               state = S2;
260            } else if (state == S2) {
261               if (c == '\u0002')
262                  state = S3;
263               else if (c == -1 || c == '\u0001') {
264                  m.put(currAttr, null);
265                  if (c == -1)
266                     return m;
267                  state = S1;
268               }
269            } else if (state == S3) {
270               if (c == -1 || c == '\u0001') {
271                  if (! currAttr.equals("_class")) {
272                     BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
273                     if (pMeta == null) {
274                        if (m.getMeta().isSubTyped()) {
275                           m.put(currAttr, "");
276                        } else {
277                           onUnknownProperty(ctx, currAttr, m, currAttrLine, currAttrCol);
278                        }
279                     } else {
280                        try {
281                           // In cases of "&foo=", create an empty instance of the value if createable.
282                           // Otherwise, leave it null.
283                           ClassMeta<?> cm = pMeta.getClassMeta();
284                           if (cm.canCreateNewInstance())
285                              pMeta.set(m, cm.newInstance());
286                        } catch (Exception e) {
287                           throw new ParseException(e);
288                        }
289                     }
290                  }
291                  if (c == -1)
292                     return m;
293                  state = S1;
294               } else {
295                  if (! currAttr.equals("_class")) {
296                     BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
297                     if (pMeta == null) {
298                        if (m.getMeta().isSubTyped()) {
299                           m.put(currAttr, parseAnything(object(), ctx, r.unread(), null, m.getBean(false), true, currAttr));
300                        } else {
301                           onUnknownProperty(ctx, currAttr, m, currAttrLine, currAttrCol);
302                           parseAnything(object(), ctx, r.unread(), null, m.getBean(false), true, null); // Read content anyway to ignore it
303                        }
304                     } else {
305                        if (ctx.shouldUseExpandedParams(pMeta)) {
306                           ClassMeta cm = pMeta.getClassMeta();
307                           Object value = parseAnything(cm.getElementType(), ctx, r.unread(), pMeta, m.getBean(false), true, currAttr);
308                           pMeta.add(m, value);
309                        } else {
310                           Object value = parseAnything(pMeta.getClassMeta(), ctx, r.unread(), pMeta, m.getBean(false), true, currAttr);
311                           pMeta.set(m, value);
312                        }
313                     }
314                  }
315                  state = S4;
316               }
317            } else if (state == S4) {
318               if (c == '\u0001')
319                  state = S1;
320               else if (c == -1) {
321                  return m;
322               }
323            }
324         }
325         isInEscape = (c == '\\' && ! isInEscape);
326      }
327      if (state == S1)
328         throw new ParseException(line, column, "Could not find attribute name on object.");
329      if (state == S2)
330         throw new ParseException(line, column, "Could not find '=' following attribute name on object.");
331      if (state == S3)
332         throw new ParseException(line, column, "Could not find value following '=' on object.");
333      if (state == S4)
334         throw new ParseException(line, column, "Could not find end of object.");
335
336      return null; // Unreachable.
337   }

339   /**
340    * Parse a URL query string into a simple map of key/value pairs.
341    *
342    * @param qs The query string to parse.
343    * @return A sorted {@link TreeMap} of query string entries.
344    * @throws IOException
345    */
346   public Map<String,String[]> parseIntoSimpleMap(String qs) throws IOException {
347
348      Map<String,String[]> m = new TreeMap<String,String[]>();
349
350      if (StringUtils.isEmpty(qs))
351         return m;
352
353      UonParserReader r = new UonParserReader(qs, true);
354
355      final int S1=1; // Looking for attrName start.
356      final int S2=2; // Found attrName start, looking for = or & or end.
357      final int S3=3; // Found =, looking for valStart.
358      final int S4=4; // Found valStart, looking for & or end.
359
360      try {
361         int c = r.peek();
362         if (c == '?')
363            r.read();
364
365         int state = S1;
366         String currAttr = null;
367         while (c != -1) {
368            c = r.read();
369            if (state == S1) {
370               if (c != -1) {
371                  r.unread();
372                  r.mark();
373                  state = S2;
374               }
375            } else if (state == S2) {
376               if (c == -1) {
377                  add(m, r.getMarked(), null);
378               } else if (c == '\u0001') {
379                  m.put(r.getMarked(0,-1), null);
380                  state = S1;
381               } else if (c == '\u0002') {
382                  currAttr = r.getMarked(0,-1);
383                  state = S3;
384               }
385            } else if (state == S3) {
386               if (c == -1 || c == '\u0001') {
387                  add(m, currAttr, "");
388               } else {
389                  if (c == '\u0002')
390                     r.replace('=');
391                  r.unread();
392                  r.mark();
393                  state = S4;
394               }
395            } else if (state == S4) {
396               if (c == -1) {
397                  add(m, currAttr, r.getMarked());
398               } else if (c == '\u0001') {
399                  add(m, currAttr, r.getMarked(0,-1));
400                  state = S1;
401               } else if (c == '\u0002') {
402                  r.replace('=');
403               }
404            }
405         }
406      } finally {
407         r.close();
408      }
409
410      return m;
411   }

413   private static void add(Map<String,String[]> m, String key, String val) {
414      boolean b = m.containsKey(key);
415      if (val == null) {
416         if (! b)
417            m.put(key, null);
418      } else if (b && m.get(key) != null) {
419         m.put(key, ArrayUtils.append(m.get(key), val));
420      } else {
421         m.put(key, new String[]{val});
422      }
423   }

425   private Object[] parseArgs(UonParserContext ctx, ParserReader r, ClassMeta<?>[] argTypes) throws ParseException {
426      // TODO - This can be made more efficient.
427      BeanContext bc = ctx.getBeanContext();
428      ClassMeta<TreeMap<Integer,String>> cm = bc.getMapClassMeta(TreeMap.class, Integer.class, String.class);
429      TreeMap<Integer,String> m = parseAnything(cm, ctx, r, ctx.getOuter(), null);
430      Object[] vals = m.values().toArray(new Object[m.size()]);
431      if (vals.length != argTypes.length)
432         throw new ParseException("Argument lengths don't match.  vals={0}, argTypes={1}", vals.length, argTypes.length);
433      for (int i = 0; i < vals.length; i++) {
434         String s = String.valueOf(vals[i]);
435         vals[i] = super.parseAnything(argTypes[i], ctx, ctx.getUrlEncodingParserReader(new StringReader(s), s.length()), null, ctx.getOuter(), true, null);
436      }
437
438      return vals;
439   }

441   /**
442    * Parses a single query parameter value into the specified class type.
443    *
444    * @param in The input query string value.
445    * @param type The class type of the object to create.
446    * @return A new instance of the specified type.
447    * @throws ParseException
448    */
449   public <T> T parseParameter(CharSequence in, ClassMeta<T> type) throws ParseException {
450      if (in == null)
451         return null;
452      UonParserContext uctx = createParameterContext();
453      UonParserReader r = uctx.getUrlEncodingParserReader(wrapReader(in), in.length());
454      return super.parseAnything(type, uctx, r, null, null, true, null);
455   }

465   public <T> T parseParameter(CharSequence in, Class<T> type) throws ParseException {
466      return parseParameter(in, getBeanContext().getClassMeta(type));

473   @Override /* Parser */
474   protected <T> T doParse(Reader in, int estimatedSize, ClassMeta<T> type, ParserContext ctx) throws ParseException, IOException {
475      UonParserContext uctx = (UonParserContext)ctx;
476      type = ctx.getBeanContext().normalizeClassMeta(type);
477      UonParserReader r = uctx.getUrlEncodingParserReader(in, estimatedSize);
478      T o = parseAnything(type, uctx, r, ctx.getOuter(), null);
479      return o;
480   }

482   @Override /* ReaderParser */
483   protected Object[] doParseArgs(Reader in, int estimatedSize, ClassMeta<?>[] argTypes, ParserContext ctx) throws ParseException, IOException {
484      UonParserContext uctx = (UonParserContext)ctx;
485      UonParserReader r = uctx.getUrlEncodingParserReader(in, estimatedSize);
486      Object[] a = parseArgs(uctx, r, argTypes);
487      return a;
488   }

490   @Override /* ReaderParser */
491   protected <K,V> Map<K,V> doParseIntoMap(Reader in, int estimatedSize, Map<K,V> m, Type keyType, Type valueType, ParserContext ctx) throws ParseException, IOException {
492      UonParserContext uctx = (UonParserContext)ctx;
493      UonParserReader r = uctx.getUrlEncodingParserReader(in, estimatedSize);
494      if (r.peek() == '?')
495         r.read();
496      m = parseIntoMap(uctx, r, m, ctx.getBeanContext().getClassMeta(keyType), ctx.getBeanContext().getClassMeta(valueType));
497      return m;
498   }
