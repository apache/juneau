021/**
022 * Transforms {@link Date Dates} to {@link String Strings}.
023 *
024 *
025 * <h6 class='topic'>Behavior-specific subclasses</h6>
026 * <p>
027 *    The following direct subclasses are provided for convenience:
028 * <ul>
029 *    <li>{@link ToString} - Transforms to {@link String Strings} using the {@code Date.toString()} method.
030 *    <li>{@link ISO8601DT} - Transforms to ISO8601 date-time strings.
031 *    <li>{@link ISO8601DTP} - Transforms to ISO8601 date-time strings with millisecond precision.
032 *    <li>{@link ISO8601DTZ} - Same as {@link ISO8601DT}, except always serializes in GMT.
033 *    <li>{@link ISO8601DTZ} - Same as {@link ISO8601DTZ}, except with millisecond precision.
034 *    <li>{@link RFC2822DT} - Transforms to RFC2822 date-time strings.
035 *    <li>{@link RFC2822DTZ} - Same as {@link RFC2822DT}, except always serializes in GMT.
036 *    <li>{@link RFC2822D} - Transforms to RFC2822 date strings.
037 *    <li>{@link Simple} - Transforms to simple <js>"yyyy/MM/dd HH:mm:ss"</js> strings.
038 *    <li>{@link Medium} - Transforms to {@link DateFormat#MEDIUM} strings.
039 * </ul>
040 *
041 *
042 * @author James Bognar (jbognar@us.ibm.com)
043 */
044public class DateFilter extends PojoFilter<Date,String> {

065   /**
066    * Transforms {@link Date Dates} to ISO8601 date-time strings.
067    *
068    * <dl>
069    *    <dt>Example output:</dt>
070    *    <dd>
071    * <ul>
072    *    <li><js>"2001-07-04T15:30:45-05:00"</js>
073    *    <li><js>"2001-07-04T15:30:45Z"</js>
074    * </ul>
075    *    </dd>
076    *    <dt>Example input:</dt>
077    *    <dd>
078    * <ul>
079    *    <li><js>"2001-07-04T15:30:45-05:00"</js>
080    *    <li><js>"2001-07-04T15:30:45Z"</js>
081    *    <li><js>"2001-07-04T15:30:45.1Z"</js>
082    *    <li><js>"2001-07-04T15:30Z"</js>
083    *    <li><js>"2001-07-04"</js>
084    *    <li><js>"2001-07"</js>
085    *    <li><js>"2001"</js>
086    * </ul>
087    *    </dd>
088    * </dl>
089    */
090   public static class ISO8601DT extends DateFilter {
091      private SimpleDateFormat tzFormat = new SimpleDateFormat("Z");
092
093      /** Constructor */
094      public ISO8601DT() {
095         this("yyyy-MM-dd'T'HH:mm:ss");
096      }
097
098      /**
099       * Constructor with specific pattern.
100       *
101       * @param pattern The {@link MessageFormat}-style format string.
102       */
103      protected ISO8601DT(String pattern) {
104         super(pattern);
105      }
106
107      @Override /* PojoFilter */
108      public Date unfilter(String o, ClassMeta<?> hint) throws ParseException {
109         try {
110            if (StringUtils.isEmpty(o))
111               return null;
112            return convert(DatatypeConverter.parseDateTime(o).getTime(), hint);
113         } catch (ParseException e) {
114            throw e;
115         } catch (Exception e) {
116            throw new ParseException(e);
117         }
118      }
119
120      @Override /* PojoFilter */
121      public String filter(Date o) {
122         String s = super.filter(o);
123         String tz = tzFormat.format(o);
124         if (tz.equals("+0000"))
125            return s + "Z";
126         return s + tz.substring(0,3) + ':' + tz.substring(3);
127      }
128   }

266   /** The formatter to convert dates to Strings. */
267   private DateFormat format;
268
269   /**
270    * Construct a filter using the specified date format string that will be
271    *    used to construct a {@link SimpleDateFormat} that will be used to convert
272    *    dates to strings.
273    *
274    * @param simpleDateFormat The {@link SimpleDateFormat} pattern.
275    */
276   public DateFilter(String simpleDateFormat) {
277      this(new SimpleDateFormat(simpleDateFormat));
278   }
279
280   /**
281    * Construct a filter using the specified date format string that will be
282    *    used to construct a {@link SimpleDateFormat} that will be used to convert
283    *    dates to strings.
284    *
285    * @param simpleDateFormat The {@link SimpleDateFormat} pattern.
286    * @param timeZone The time zone to associate with the date pattern.
287    */
288   public DateFilter(String simpleDateFormat, String timeZone) {
289      this(new SimpleDateFormat(simpleDateFormat));
290      format.setTimeZone(TimeZone.getTimeZone(timeZone));
291   }
292
293   /**
294    * Construct a filter using the specified {@link DateFormat} that will be used to convert
295    *    dates to strings.
296    *
297    * @param format The format to use to convert dates to strings.
298    */
299   public DateFilter(DateFormat format) {
300      super();
301      this.format = format;
302   }
303
304   /**
305    * Converts the specified {@link Date} to a {@link String}.
306    */
307   @Override /* PojoFilter */
308   public String filter(Date o) {
309      return format.format(o);
310   }
311
312   /**
313    * Converts the specified {@link String} to a {@link Date}.
314    */
315   @Override /* PojoFilter */
316   public Date unfilter(String o, ClassMeta<?> hint) throws ParseException {
317      try {
318         if (StringUtils.isEmpty(o))
319            return null;
320         Date d = format.parse(o);
321         return convert(d, hint);
322      } catch (ParseException e) {
323         throw e;
324      } catch (Exception e) {
325         throw new ParseException(e);
326      }
327   }
328
329   private static Date convert(Date in, ClassMeta<?> hint) throws Exception {
330      if (in == null)
331         return null;
332      if (hint == null || hint.isInstance(in))
333         return in;
334      Class<?> c = hint.getInnerClass();
335      if (c == java.util.Date.class)
336         return in;
337      if (c == java.sql.Date.class)
338         return new java.sql.Date(in.getTime());
339      if (c == java.sql.Time.class)
340         return new java.sql.Time(in.getTime());
341      if (c == java.sql.Timestamp.class)
342         return new java.sql.Timestamp(in.getTime());
343      throw new ParseException("DateFilter is unable to narrow object of type ''{0}''", c);
344   }
