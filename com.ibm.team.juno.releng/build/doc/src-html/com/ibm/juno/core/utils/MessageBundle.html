009package com.ibm.juno.core.utils;

011import java.text.*;
012import java.util.*;
013import java.util.concurrent.*;
014
015import com.ibm.juno.core.*;
016import com.ibm.juno.core.annotation.*;

018/**
019 * Wraps a {@link ResourceBundle} to provide some useful additional functionality.
020 * <ul>
021 *    <li>Instead of throwing {@link MissingResourceException}, the {@link #getString(String)} method
022 *       will return <js>"{!!key}"</js> if the bundle was not found, and <js>"{!key}"</js> if bundle
023 *       was found but the key is not in the bundle.
024 *    <li>A client locale can be set as a {@link ThreadLocal} object using the static {@link #setClientLocale(Locale)} so that client localized
025 *       messages can be retrieved using the {@link #getClientString(String, Object...)} method on all instances of this class.
026 *    <li>Resource bundles on parent classes can be added to the search path for this class by using the
027 *       {@link #addSearchPath(Class, String)} method.  This allows messages to be retrieved from the resource bundles of parent classes.
028 *    <li>Locale-specific bundles can be retrieved by using the {@link #getBundle(Locale)} method.
029 *    <li>The {@link #getString(Locale, String, Object...)} method can be used to retrieve locale-specific messages.
030 *    <li>Messages in the resource bundle can optionally be prefixed with the simple class name.
031 *       For example, if the class is <code>MyClass</code> and the properties file contains <js>"MyClass.myMessage"</js>,
032 *       the message can be retrieved using <code>getString(<js>"myMessage"</js>)</code>.
033 * </ul>
034 *
035 * @author James Bognar (jbognar@us.ibm.com)
036 */
037@ThreadSafe
038public class MessageBundle extends ResourceBundle {
039
040   private static final ThreadLocal<Locale> clientLocale = new ThreadLocal<Locale>();
041
042   private final ResourceBundle rb;
043   private final String bundlePath, className;
044   private final Class<?> forClass;
045   private final long creationThreadId;
046
047   // A map that contains all keys [shortKeyName->keyName] and [keyName->keyName], where shortKeyName
048   // refers to keys prefixed and stripped of the class name (e.g. "foobar"->"MyClass.foobar")
049   private final ConcurrentHashMap<String,String> keyMap = new ConcurrentHashMap<String,String>();
050
051   // Contains all keys present in all bundles in searchBundles.
052   private final ConcurrentSkipListSet<String> allKeys = new ConcurrentSkipListSet<String>();
053
054   // Bundles to search through to find properties.
055   // Typically this will be a list of resource bundles for each class up the class hierarchy chain.
056   private final CopyOnWriteArrayList<MessageBundle> searchBundles = new CopyOnWriteArrayList<MessageBundle>();
057
058   // Cache of message bundles per locale.
059   private final ConcurrentHashMap<Locale,MessageBundle> localizedBundles = new ConcurrentHashMap<Locale,MessageBundle>();
060
061   /**
062    * Sets the locale for this thread so that calls to {@link #getClientString(String, Object...)} return
063    * messages in that locale.
064    *
065    * @param locale The new client locale.
066    */
067   public static void setClientLocale(Locale locale) {
068      MessageBundle.clientLocale.set(locale);
069   }
070
071   /**
072    * Constructor.
073    *
074    * @param forClass The class using this resource bundle.
075    * @param bundlePath The path of the resource bundle to wrap.
076    *    This can be an absolute path (e.g. <js>"com.ibm.MyMessages"</js>) or a path
077    *    relative to the package of the <l>forClass</l> (e.g. <js>"MyMessages"</js> if <l>forClass</l> is <js>"com.ibm.MyClass"</js>).
078    */
079   public MessageBundle(Class<?> forClass, String bundlePath) {
080      this(forClass, bundlePath, Locale.getDefault());
081   }
082
083   private MessageBundle(Class<?> forClass, String bundlePath, Locale locale) {
084      this.forClass = forClass;
085      this.className = forClass.getSimpleName();
086      this.bundlePath = bundlePath;
087      this.creationThreadId = Thread.currentThread().getId();
088      ClassLoader cl = forClass.getClassLoader();
089      ResourceBundle trb = null;
090      try {
091         trb = ResourceBundle.getBundle(bundlePath, locale, cl);
092      } catch (MissingResourceException e) {
093         try {
094            trb = ResourceBundle.getBundle(forClass.getPackage().getName() + '.' + bundlePath, locale, cl);
095         } catch (MissingResourceException e2) {
096         }
097      }
098      this.rb = trb;
099      if (rb != null) {
100
101         // Populate keyMap with original mappings.
102         for (Enumeration<String> e = getKeys(); e.hasMoreElements();) {
103            String key = e.nextElement();
104            keyMap.put(key, key);
105         }
106
107         // Override/augment with shortname mappings (e.g. "foobar"->"MyClass.foobar")
108         String c = className + '.';
109         for (Enumeration<String> e = getKeys(); e.hasMoreElements();) {
110            String key = e.nextElement();
111            if (key.startsWith(c)) {
112               String shortKey = key.substring(className.length() + 1);
113               keyMap.put(shortKey, key);
114            }
115         }
116
117         allKeys.addAll(keyMap.keySet());
118      }
119      searchBundles.add(this);
120   }
121
122
123   /**
124    * Add another bundle path to this resource bundle.
125    * Order of property lookup is first-to-last.
126    * <p>
127    * This method must be called from the same thread as the call to the constructor.
128    * This eliminates the need for synchronization.
129    *
130    * @param forClass The class using this resource bundle.
131    * @param bundlePath The bundle path.
132    * @return This object (for method chaining).
133    */
134   @SuppressWarnings("hiding")
135   public MessageBundle addSearchPath(Class<?> forClass, String bundlePath) {
136      ThrowableUtils.assertSameThread(creationThreadId, "This method can only be called from the same thread that created the object.");
137      MessageBundle srb = new MessageBundle(forClass, bundlePath);
138      if (srb.rb != null) {
139         allKeys.addAll(srb.keySet());
140         searchBundles.add(srb);
141      }
142      return this;
143   }
144
145   @Override /* ResourceBundle */
146   public boolean containsKey(String key) {
147      return allKeys.contains(key);
148   }
149
150   /**
151    * Similar to {@link ResourceBundle#getString(String)} except allows you to pass in {@link MessageFormat} objects.
152    *
153    * @param key The resource bundle key.
154    * @param args Optional variable replacement arguments.
155    * @return The resolved value.  Never <jk>null</jk>.  <js>"{!!key}"</j> if the bundle is missing.  <js>"{!key}"</j> if the key is missing.
156    */
157   public String getString(String key, Object...args) {
158      String s = getString(key);
159      if (s.length() > 0 && s.charAt(0) == '{')
160         return s;
161      if (args.length > 0)
162         return MessageFormat.format(s, args);
163      return s;
164   }
165
166   /**
167    * Same as {@link #getString(String, Object...)} but allows you to specify the locale.
168    *
169    * @param locale The locale of the resource bundle to retrieve message from.
170    * @param key The resource bundle key.
171    * @param args Optional variable replacement arguments.
172    * @return The resolved value.  Never <jk>null</jk>.  <js>"{!!key}"</j> if the bundle is missing.  <js>"{!key}"</j> if the key is missing.
173    */
174   public String getString(Locale locale, String key, Object...args) {
175      if (locale == null)
176         return getString(key, args);
177      return getBundle(locale).getString(key, args);
178   }
179
180   /**
181    * Same as {@link #getString(String, Object...)} but uses the locale specified on the call to {@link #setClientLocale(Locale)}.
182    *
183    * @param key The resource bundle key.
184    * @param args Optional variable replacement arguments.
185    * @return The resolved value.  Never <jk>null</jk>.  <js>"{!!key}"</j> if the bundle is missing.  <js>"{!key}"</j> if the key is missing.
186    */
187   public String getClientString(String key, Object...args) {
188      return getString(clientLocale.get(), key, args);
189   }
190
191   /**
192    * Looks for all the specified keys in the resource bundle and returns the first value that exists.
193    *
194    * @param keys The list of possible keys.
195    * @return The resolved value, or <jk>null</jk> if no value is found or the resource bundle is missing.
196    */
197   public String findFirstString(String...keys) {
198      if (rb == null)
199         return null;
200      for (String k : keys) {
201         if (containsKey(k))
202            return getString(k);
203      }
204      return null;
205   }
206
207   /**
208    * Same as {@link #findFirstString(String...)}, but uses the specified locale.
209    *
210    * @param locale The locale of the resource bundle to retrieve message from.
211    * @param keys The list of possible keys.
212    * @return The resolved value, or <jk>null</jk> if no value is found or the resource bundle is missing.
213    */
214   public String findFirstString(Locale locale, String...keys) {
215      MessageBundle srb = getBundle(locale);
216      return srb.findFirstString(keys);
217   }
218
219   @Override /* ResourceBundle */
220   public Set<String> keySet() {
221      return Collections.unmodifiableSet(allKeys);
222   }
223
224   /**
225    * Returns all keys in this resource bundle with the specified prefix.
226    *
227    * @param prefix The prefix.
228    * @return The set of all keys in the resource bundle with the prefix.
229    */
230   public Set<String> keySet(String prefix) {
231      Set<String> set = new HashSet<String>();
232      for (String s : keySet()) {
233         if (s.equals(prefix) || (s.startsWith(prefix) && s.charAt(prefix.length()) == '.'))
234            set.add(s);
235      }
236      return set;
237   }
238
239   @Override /* ResourceBundle */
240   public Enumeration<String> getKeys() {
241      if (rb == null)
242         return new Vector<String>(0).elements();
243      return rb.getKeys();
244   }
245
246   @Override /* ResourceBundle */
247   protected Object handleGetObject(String key) {
248      for (MessageBundle srb : searchBundles) {
249         if (srb.rb != null) {
250            String key2 = srb.keyMap.get(key);
251            if (key2 != null) {
252               try {
253                  return srb.rb.getObject(key2);
254               } catch (Exception e) {
255                  return "{!"+key+"}";
256               }
257            }
258         }
259      }
260      if (rb == null)
261         return "{!!"+key+"}";
262      return "{!"+key+"}";
263   }
264
265   /**
266    * Returns this resource bundle as an {@link ObjectMap}.
267    * <p>
268    * Useful for debugging purposes.
269    * Note that any class that implements a <code>toObjectMap()</code> method will automatically be serialized by
270    *    calling this method and serializing the result.
271    * <p>
272    * This method always constructs a new {@link ObjectMap} on each call.
273    *
274    * @return A new map containing all the keys and values in this bundle.
275    */
276   public ObjectMap toObjectMap() {
277      ObjectMap om = new ObjectMap();
278      for (String k : allKeys)
279         om.put(k, getString(k));
280      return om;
281   }
282
283   /**
284    * Returns the resource bundle for the specified locale.
285    *
286    * @param locale The client locale.
287    * @return The resource bundle for the specified locale.  Never <jk>null</jk>.
288    */
289   public MessageBundle getBundle(Locale locale) {
290      MessageBundle mb = localizedBundles.get(locale);
291      if (mb != null)
292         return mb;
293      mb = new MessageBundle(forClass, bundlePath, locale);
294      List<MessageBundle> l = new ArrayList<MessageBundle>(searchBundles.size()-1);
295      for (int i = 1; i < searchBundles.size(); i++) {
296         MessageBundle srb = searchBundles.get(i);
297         srb = new MessageBundle(srb.forClass, srb.bundlePath, locale);
298         l.add(srb);
299         mb.allKeys.addAll(srb.keySet());
300      }
301      mb.searchBundles.addAll(l);
302      localizedBundles.putIfAbsent(locale, mb);
303      return localizedBundles.get(locale);
304   }
