009package com.ibm.juno.core.plaintext;
010
011import java.io.*;
012
013import com.ibm.juno.core.*;
014import com.ibm.juno.core.annotation.*;
015import com.ibm.juno.core.filter.*;
016import com.ibm.juno.core.parser.*;
017import com.ibm.juno.core.utils.*;
018
019/**
020 * Parsers HTTP plain text request bodies into <a href='../package-summary.html#PojoCategories'>Group 5</a> POJOs.
021 *
022 *
023 * <h6 class='topic'>Media types</h6>
024 * <p>
025 *    Handles <code>Accept</code> types: <code>text/plain</code>
026 * <p>
027 *    Produces <code>Content-Type</code> types: <code>text/plain</code>
028 *
029 *
030 * <h6 class='topic'>Description</h6>
031 * <p>
032 *    Essentially just converts plain text to POJOs via static <code>fromString()</code> or <code>valueOf()</code>, or
033 *    through constructors that take a single string argument.
034 * <p>
035 *    Also parses objects using a filter if the object class has an {@link PojoFilter PojoFilter&lt;?,String&gt;} filter defined on it.
036 *
037 *
038 * <h6 class='topic'>Configurable properties</h6>
039 * <p>
040 *    This class has the following properties associated with it:
041 * <ul>
042 *    <li>{@link BeanContextProperties}
043 * </ul>
044 *
045 *
046 * @author James Bognar (jbognar@us.ibm.com)
047 */
048@Consumes("text/plain")
049public final class PlainTextParser extends ReaderParser {
050
051   //--------------------------------------------------------------------------------
052   // Overridden methods
053   //--------------------------------------------------------------------------------
054
055   @Override /* Parser */
056   protected <T> T doParse(Reader in, int estimatedSize, ClassMeta<T> type, ParserContext ctx) throws IOException, ParseException {
057      return ctx.getBeanContext().convertToType(IOUtils.read(in), type);
058   }
059
060   @Override /* Lockable */
061   public PlainTextParser clone() {
062      try {
063         return (PlainTextParser)super.clone();
064      } catch (CloneNotSupportedException e) {
065         throw new RuntimeException(e); // Shouldn't happen.
066      }
067   }
068}
