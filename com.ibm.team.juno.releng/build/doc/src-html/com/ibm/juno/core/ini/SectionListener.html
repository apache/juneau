009package com.ibm.juno.core.ini;
010
011import java.util.*;
012
013import com.ibm.juno.core.utils.*;
014
015
016/**
017 * Listener that can be used to listen for change events for a specific section in a config file.
018 * <p>
019 * Use the {@link ConfigFile#addListener(ConfigFileListener)} method to register listeners.
020 */
021public class SectionListener extends ConfigFileListener {
022
023   private boolean isDefault;
024   private String prefix;
025
026   /**
027    * Constructor.
028    *
029    * @param section The name of the section in the config file to listen to.
030    */
031   public SectionListener(String section) {
032      isDefault = StringUtils.isEmpty(section);
033      prefix = isDefault ? null : (section + '/');
034   }
035
036   @Override /* ConfigFileListener */
037   public void onChange(ConfigFile cf, Set<String> changes) {
038      for (String c : changes) {
039         if (isDefault) {
040            if (c.indexOf('/') == -1) {
041               onChange(cf);
042               return;
043            }
044         } else {
045            if (c.startsWith(prefix)) {
046               onChange(cf);
047               return;
048            }
049         }
050      }
051   }
052
053   /**
054    * Signifies that the config file entry changed.
055    *
056    * @param cf The config file being modified.
057    */
058   public void onChange(ConfigFile cf) {}
059}
