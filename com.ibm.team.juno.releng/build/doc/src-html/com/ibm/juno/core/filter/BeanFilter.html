018/**
019 * Parent class for all bean filters.
020 * <p>
021 *    Bean filters are used to control aspects of how beans are handled during serialization and parsing.
022 * <p>
023 *    This class can be considered a programmatic equivalent to using the {@link Bean @Bean} annotation on bean classes.
024 *    Thus, it can be used to perform the same function as the <code>@Bean</code> annotation when you don't have
025 *       the ability to annotate those classes (e.g. you don't have access to the source code).
026 * <p>
027 *    Note that value returned by the {@link Filter#forClass()} method is automatically determined through reflection
028 *       when the no-arg constructor is used.
029 *
030 * <p>
031 *    When defining bean filters, you can either call the setters in the contructor, or override getters.
032 *
033 * <h6 class='topic'>Example</h6>
034 * <p class='bcode'>
035 *    <jc>// Create our serializer with a filter.</jc>
036 *    WriterSerializer s = <jk>new</jk> JsonSerializer().addFilters(AddressFilter.<jk>class</jk>);
037 *
038 *    Address a = <jk>new</jk> Address();
039 *    String json = s.serialize(a); <jc>// Serializes only street, city, state.</jc>
040 *
041 *    <jc>// Filter class defined via setters</jc>
042 *    <jk>public class</jk> AddressFilter <jk>extends</jk> BeanFilter&lt;Address&gt; {
043 *       <jk>public</jk> AddressFilter() {
044 *          setProperties(<js>"street"</js>,<js>"city"</js>,<js>"state"</js>);
045 *       }
046 *    }
047 *
048 *    <jc>// Filter class defined by overriding getters</jc>
049 *    <jk>public class</jk> AddressFilter <jk>extends</jk> BeanFilter&lt;Address&gt; {
050 *       <jk>public</jk> String[] getProperties() {
051 *          <jk>return new</jk> String[]{<js>"street"</js>,<js>"city"</js>,<js>"state"</js>};
052 *       }
053 *    }
054 * </p>
055 * <p>
056 *    The examples in this class use the setters approach.
057 *
058 * <h6 class='topic'>Additional information</h6>
059 *    See {@link com.ibm.juno.core.filter} for more information.
060 *
061 *
062 * @author James Bognar (jbognar@us.ibm.com)
063 * @param <T> The class type that this filter applies to.
064 */
065public abstract class BeanFilter<T> extends Filter {
066
067   private String[] properties, excludeProperties;
068   private LinkedHashMap<Class<?>, String> subTypes;
069   private String subTypeAttr;
070   private Class<? extends PropertyNamer> propertyNamer;
071   private Class<?> interfaceClass, stopClass;

073   /**
074    * Constructor that determines the for-class value using reflection.
075    */
076   @SuppressWarnings("unchecked")
077   public BeanFilter() {
078      super();
079      this.type = FilterType.BEAN;
080
081      Class<?> c = this.getClass().getSuperclass();
082      Type t = this.getClass().getGenericSuperclass();
083      while (c != BeanFilter.class) {
084         t = c.getGenericSuperclass();
085         c = c.getSuperclass();
086      }
087
088      // Attempt to determine the T and G classes using reflection.
089      if (t instanceof ParameterizedType) {
090         ParameterizedType pt = (ParameterizedType)t;
091         Type[] pta = pt.getActualTypeArguments();
092         if (pta.length > 0) {
093            Type nType = pta[0];
094            if (nType instanceof Class)
095               this.forClass = (Class<T>)nType;
096
097            else
098               throw new RuntimeException("Unsupported parameter type: " + nType);
099         }
100      }
101   }

103   /**
104    * Constructor that specifies the for-class explicitly.
105    * <p>
106    * This constructor only needs to be called when the class type cannot be inferred through reflection.
107    *
108    * <dl>
109    *    <dt>Example:</dt>
110    *    <dd>
111    *       <p class='bcode'>
112    *    <jk>public class</jk> SomeArbitraryFilter <jk>extends</jk> BeanFilter&lt?&gt; {
113    *       <jk>public</jk> SomeArbitraryFilter(Class&lt?&gt; forClass) {
114    *          <jk>super</jk>(forClass);
115    *          ...
116    *       }
117    *    }
118    *       </p>
119    *    </dd>
120    * </dl>
121    *
122    * @param forClass The class that this bean filter applies to.
123    */
124   public BeanFilter(Class<T> forClass) {
125      super(forClass);
126      this.type = FilterType.BEAN;
127   }

254   /**
255    * Defines a virtual property on a superclass that identifies bean subtype classes.
256    * <p>
257    *    In the following example, the abstract class has two subclasses that are differentiated
258    *       by a property called <code>subType</code>
259    *
260    * <p class='bcode'>
261    *    <jc>// Abstract superclass</jc>
262    *    <jk>public abstract class</jk> A {
263    *       <jk>public</jk> String <jf>f0</jf> = <js>"f0"</js>;
264    *    }
265    *
266    *    <jc>// Subclass 1</jc>
267    *    <jk>public class</jk> A1 <jk>extends</jk> A {
268    *       <jk>public</jk> String <jf>f1</jf>;
269    *    }
270    *
271    *    <jc>// Subclass 2</jc>
272    *    <jk>public class</jk> A2 <jk>extends</jk> A {
273    *       <jk>public</jk> String <jf>f2</jf>;
274    *    }
275    *
276    *    <jc>// Filter for defining subtypes</jc>
277    *    <jk>public class</jk> AFilter <jk>extends</jk> BeanFilter&lt;A&gt; {
278    *       <jk>public</jk> AFilter() {
279    *          setSubTypeProperty(<js>"subType"</js>);
280    *          addSubType(Al.<jk>class</jk>, <js>"A1"</js>);
281    *          addSubType(A2.<jk>class</jk>, <js>"A2"</js>);
282    *       }
283    *    }
284    * </p>
285    * <p>
286    *    The following shows what happens when serializing a subclassed object to JSON:
287    * <p class='bcode'>
288    *    JsonSerializer s = <jk>new</jk> JsonSerializer().addFilters(AFilter.<jk>class</jk>);
289    *    A1 a1 = <jk>new</jk> A1();
290    *    a1.<jf>f1</jf> = <js>"f1"</js>;
291    *    String r = s.serialize(a1);
292    *    <jsm>assertEquals</jsm>(<js>"{subType:'A1',f1:'f1',f0:'f0'}"</js>, r);
293    * </p>
294    * <p>
295    *    The following shows what happens when parsing back into the original object.
296    * <p class='bcode'>
297    *    JsonParser p = <jk>new</jk> JsonParser().addFilters(AFilter.<jk>class</jk>);
298    *    A a = p.parse(r, A.<jk>class</jk>);
299    *    <jsm>assertTrue</jsm>(a <jk>instanceof</jk> A1);
300    * </p>
301    * <p>
302    *    This method is an alternative to using the {@link Bean#subTypeProperty()} annotation on a class.
303    *
304    * @param subTypeAttr The name of the attribute representing the subtype.
305    * @return This object (for method chaining).
306    */
307   public BeanFilter<T> setSubTypeProperty(String subTypeAttr) {
308      this.subTypeAttr = subTypeAttr;
309      return this;
310   }

313    * Returns the subtypes associated with the bean class.
314    *

317    */

319      return subTypes;

322   /**
323    * Specifies the set of subclasses of this bean class in addition to a string identifier for that subclass.
324    *
325    * @see #setSubTypeProperty(String)
326    * @param subTypes the map of subtype classes to subtype identifier strings.
327    * @return This object (for method chaining).
328    */
329   public BeanFilter<T> setSubTypes(LinkedHashMap<Class<?>, String> subTypes) {
330      this.subTypes = subTypes;
331      return this;
332   }
333
334   /**
335    * Convenience method for adding a single subtype in leu of using {@link #setSubTypes(LinkedHashMap)} in one call.
336    *
337    * @see #setSubTypeProperty(String)
338    * @param c The subtype class.
339    * @param id The subtype identifier string for the specified subtype class.
340    * @return This object (for method chaining).
341    */
342   public BeanFilter<T> addSubType(Class<?> c, String id) {
343      if (subTypes == null)
344         subTypes = new LinkedHashMap<Class<?>, String>();
345      subTypes.put(c, id);
346      return this;
347   }

359   /**
360    * Identifies a class to be used as the interface class for this and all subclasses.
361    * <p>
362    *    Functionally equivalent to using the {@link Bean#interfaceClass()} annotation.
363    * <p>
364    *    When specified, only the list of properties defined on the interface class will be used during serialization.
365    *    Additional properties on subclasses will be ignored.
366    * <p class='bcode'>
367    *    <jc>// Parent class</jc>
368    *    <jk>public abstract class</jk> A {
369    *       <jk>public</jk> String <jf>f0</jf> = <js>"f0"</js>;
370    *    }
371    *
372    *    <jc>// Sub class</jc>
373    *    <jk>public class</jk> A1 <jk>extends</jk> A {
374    *       <jk>public</jk> String <jf>f1</jf> = <js>"f1"</js>;
375    *    }
376    *
377    *    <jc>// Filter class</jc>
378    *    <jk>public class</jk> AFilter <jk>extends</jk> BeanFilter&lt;A&gt; {
379    *       <jk>public</jk> AFilter() {
380    *          setInterfaceClass(A.<jk>class</jk>);
381    *       }
382    *    }
383    *
384    *    JsonSerializer s = new JsonSerializer().addFilters(AFilter.<jk>class</jk>);
385    *    A1 a1 = <jk>new</jk> A1();
386    *    String r = s.serialize(a1);
387    *    <jsm>assertEquals</jsm>(<js>"{f0:'f0'}"</js>, r);  <jc>// Note f1 is not serialized</jc>
388    * </p>
389    * <p>
390    *    Note that this filter can be used on the parent class so that it filters to all child classes,
391    *       or can be set individually on the child classes.
392    * <p>
393    *    This method is an alternative to using the {@link Bean#interfaceClass()}} annotation.
394    *
395    * @param interfaceClass The interface class.
396    * @return This object (for method chaining).
397    */
398   public BeanFilter<T> setInterfaceClass(Class<?> interfaceClass) {
399      this.interfaceClass = interfaceClass;
400      return this;
401   }

413   /**
414    * Identifies a stop class for this class and all subclasses.
415    * <p>
416    *    Functionally equivalent to using the {@link Bean#stopClass()} annotation.
417    * <p>
418    *    Identical in purpose to the stop class specified by {@link Introspector#getBeanInfo(Class, Class)}.
419    *    Any properties in the stop class or in its baseclasses will be ignored during analysis.
420    * <p>
421    *    For example, in the following class hierarchy, instances of <code>C3</code> will include property <code>p3</code>, but
422    *       not <code>p1</code> or <code>p2</code>.
423    * <p class='bcode'>
424    *    <jk>public class</jk> C1 {
425    *       <jk>public int</jk> getP1();
426    *    }
427    *
428    *    <jk>public class</jk> C2 <jk>extends</jk> C1 {
429    *       <jk>public int</jk> getP2();
430    *    }
431    *
432    *    <ja>@Bean</ja>(stopClass=C2.<jk>class</jk>)
433    *    <jk>public class</jk> C3 <jk>extends</jk> C2 {
434    *       <jk>public int</jk> getP3();
435    *    }
436    * </p>
437    *
438    * @param stopClass The stop class.
439    * @return This object (for method chaining).
440    */
441   public BeanFilter<T> setStopClass(Class<?> stopClass) {
442      this.stopClass = stopClass;
443      return this;
444   }
