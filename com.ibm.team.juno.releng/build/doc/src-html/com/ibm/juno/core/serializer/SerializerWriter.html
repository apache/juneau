009package com.ibm.juno.core.serializer;
010
011import java.io.*;
012import java.net.*;
013
014import com.ibm.juno.core.utils.*;
015
016/**
017 * Simple wrapper around a standard {@link Writer} with additional methods.
018 * <p>
019 * Modeled after the Java ProcessBuilder class so that you can chain commands to reduce
020 *    the need for string concatenation for performance reasons.
021 *
022 * <h6 class='topic'>Example</h6>
023 * <p class='bcode'>
024 *    writer.append(<js>"foo"</js>).nl().i(5).append(<js>"bar"</js>);
025 * </p>
026 *
027 * @author James Bognar (jbognar@us.ibm.com)
028 */
029public class SerializerWriter extends Writer {
030
031   /** The underlying writer. */
032   protected final Writer out;
033
034   /** Use-indentation flag. */
035   protected final boolean useIndentation;
036
037   /** Use-whitespace flag. */
038   protected final boolean useWhitespace;
039
040   /** Use-whitespace flag. */
041   protected final boolean trimStrings;
042
043   /** The quote character being used by this writer. */
044   protected final char quoteChar;
045
046   /** The base (e.g. <js>https://localhost:9443/contextPath"</js>) for relative URIs (e.g. <js>"my/path"</js>). */
047   protected final String relativeUriBase;
048
049   /** The base (e.g. <js>https://localhost:9443"</js>) for relative URIs with absolute paths (e.g. <js>"/contextPath/my/path"</js>). */
050   protected final String absolutePathUriBase;
051
052   /**
053    * @param out The writer being wrapped.
054    * @param useIndentation If <jk>true</jk>, calling {@link #cr(int)} will create an indentation.
055    * @param useWhitespace If <jk>true</jk>, calling {@link #s()} will write a space character.
056    * @param trimStrings If <jk>true</jk>, strings should be trimmed before they're serialized.
057    * @param quoteChar The character to write when {@link #q()} is called.
058    * @param relativeUriBase The base (e.g. <js>https://localhost:9443/contextPath"</js>) for relative URIs (e.g. <js>"my/path"</js>).
059    * @param absolutePathUriBase The base (e.g. <js>https://localhost:9443"</js>) for relative URIs with absolute paths (e.g. <js>"/contextPath/my/path"</js>).
060    */
061   public SerializerWriter(Writer out, boolean useIndentation, boolean useWhitespace, boolean trimStrings, char quoteChar, String relativeUriBase, String absolutePathUriBase) {
062      this.out = out;
063      this.useIndentation = useIndentation;
064      this.useWhitespace = useWhitespace;
065      this.trimStrings = trimStrings;
066      this.quoteChar = quoteChar;
067      this.relativeUriBase = relativeUriBase;
068      this.absolutePathUriBase = absolutePathUriBase;
069   }
070
071   /**
072    * Performs a carriage return.
073    * <p>
074    *    Adds a newline and the specified number of tabs (if the {@code useIndentation} setting is enabled) to the output.
075    *
076    * @param depth The indentation.
077    * @throws IOException If a problem occurred trying to write to the writer.
078    * @return This object (for method chaining).
079    */
080   public SerializerWriter cr(int depth) throws IOException {
081      if (useIndentation)
082         return nl().i(depth);
083      return this;
084   }
085
086   /**
087    * Writes an indent (if the {@code useIndentation} setting is enabled), followed by text,
088    *    followed by a newline (if the {@code useIndentation} setting is enabled).
089    *
090    * @param indent The number of tabs to indent.
091    * @param text The text to write.
092    * @throws IOException If a problem occurred trying to write to the writer.
093    * @return This object.
094    */
095   public SerializerWriter appendln(int indent, String text) throws IOException {
096      return append(indent, true, text);
097   }
098
099   /**
100    * Writes the specified text followed by a newline (if the {@code useIndentation} setting is enabled).
101    *
102    * @param text The text to write.
103    * @throws IOException If a problem occurred trying to write to the writer.
104    * @return This object.
105    */
106   public SerializerWriter appendln(String text) throws IOException {
107      return append(0, true, text);
108   }
109
110   /**
111    * Writes an indent (if the {@code useIndentation} setting is enabled), followed by text.
112    *
113    * @param indent The number of tabs to indent.
114    * @param text The text to write.
115    * @throws IOException If a problem occurred trying to write to the writer.
116    * @return This object.
117    */
118   public SerializerWriter append(int indent, String text) throws IOException {
119      return append(indent, false, text);
120   }
121
122   /**
123    * Writes an indent (if the {@code useIndentation} setting is enabled), followed by text.
124    *
125    * @param indent The number of tabs to indent.
126    * @param c The character to write.
127    * @throws IOException If a problem occurred trying to write to the writer.
128    * @return This object.
129    */
130   public SerializerWriter append(int indent, char c) throws IOException {
131      return i(indent).append(c);
132   }
133
134   /**
135    * Writes an indent (if the {@code useIndentation} setting is enabled), followed by text,
136    *    optionally followed by a newline (if the {@code useIndentation} setting is enabled).
137    *
138    * @param indent The number of tabs to indent.
139    * @param newline If <jk>true</jk>, then a newline is written.
140    * @param text The text to write.
141    * @throws IOException If a problem occurred trying to write to the writer.
142    * @return This object (for method chaining).
143    */
144   private SerializerWriter append(int indent, boolean newline, String text) throws IOException {
145      i(indent);
146      out.write(text);
147      if (newline)
148         nl();
149      return this;
150   }
151
152   /**
153    * Appends the specified object as a URI.
154    * <p>
155    * Object is converted to a <code>String</code> using <code>toString()</code>, so this will work on {@link URL} or {@link URI} objects,
156    * or any other type that returns a URI via it's <code>toString()</code> method.
157    * <p>
158    * If the URI is relative (i.e. without a schema and not prepended with <js>'/'</js>) the URI
159    * will be prepended with {@link #absolutePathUriBase} and {@link #relativeUriBase}.
160    * <p>
161    * If the URI is context-absolute (i.e. without a schema, but prepended with <js>'/'</js>)
162    * the URI will be prepended with {@link #absolutePathUriBase}.
163    *
164    * @param uri The URI to serialize.
165    * @return This object (for method chaining).
166    * @throws IOException If a problem occurred trying to write to the writer.
167    */
168   public SerializerWriter appendUri(Object uri) throws IOException {
169      String s = uri.toString();
170      if (s.indexOf("://") == -1) {
171         if (StringUtils.startsWith(s, '/')) {
172            if (absolutePathUriBase != null)
173               append(absolutePathUriBase);
174         } else {
175            if (relativeUriBase != null) {
176               append(relativeUriBase);
177               if (! relativeUriBase.equals("/"))
178                  append("/");
179
180            }
181         }
182      }
183      return append(s);
184   }
185
186   /**
187    * Adds a whitespace character to the output if the {@code useWhitespace} setting is enabled.
188    *
189    * @return This object (for method chaining).
190    * @throws IOException If a problem occurred trying to write to the writer.
191    */
192   public SerializerWriter s() throws IOException {
193      if (useWhitespace)
194         out.write(' ');
195      return this;
196   }
197
198   /**
199    * Adds the quote character specified by the {@code quoteChar} setting to the output.
200    *
201    * @return This object (for method chaining).
202    * @throws IOException If a problem occurred trying to write to the writer.
203    */
204   public SerializerWriter q() throws IOException {
205      out.write(quoteChar);
206      return this;
207   }
208
209   /**
210    * Writes an indent to the writer if the {@code useIndentation} setting is enabled.
211    *
212    * @param indent The number of tabs to indent.
213    * @throws IOException If a problem occurred trying to write to the writer.
214    * @return This object (for method chaining).
215    */
216   public SerializerWriter i(int indent) throws IOException {
217      if (useIndentation)
218         for (int i = 0; i < indent; i++)
219            out.write('\t');
220      return this;
221   }
222
223   /**
224    * Writes a newline to the writer if the {@code useIndentation} setting is enabled.
225    *
226    * @throws IOException If a problem occurred trying to write to the writer.
227    * @return This object (for method chaining).
228    */
229   public SerializerWriter nl() throws IOException {
230      if (useIndentation)
231         out.write('\n');
232      return this;
233   }
234
235   /**
236    * Writes the specified text to the writer.
237    *
238    * @param text The text to write.
239    * @throws IOException If a problem occurred trying to write to the writer.
240    * @return This object (for method chaining).
241    */
242   public SerializerWriter append(Object text) throws IOException {
243      out.append(text == null ? null : text.toString());
244      return this;
245   }
246
247   /**
248    * Writes the specified text to the writer.
249    *
250    * @param text The text to write.
251    * @throws IOException If a problem occurred trying to write to the writer.
252    * @return This object (for method chaining).
253    */
254   public SerializerWriter append(String text) throws IOException {
255      if (text != null)
256         out.append(text);
257      return this;
258   }
259
260   /**
261    * Writes the specified text to the writer if b is true.
262    *
263    * @param b Boolean flag.
264    * @param text The text to write.
265    * @throws IOException If a problem occurred trying to write to the writer.
266    * @return This object (for method chaining).
267    */
268   public SerializerWriter appendIf(boolean b, String text) throws IOException {
269      if (b)
270         out.write(text);
271      return this;
272   }
273
274   /**
275    * Writes the specified text to the writer if b is true.
276    *
277    * @param b Boolean flag.
278    * @param c The text to write.
279    * @throws IOException If a problem occurred trying to write to the writer.
280    * @return This object (for method chaining).
281    */
282   public SerializerWriter appendIf(boolean b, char c) throws IOException {
283      if (b)
284         out.write(c);
285      return this;
286   }
287
288
289   //--------------------------------------------------------------------------------
290   // Overridden methods
291   //--------------------------------------------------------------------------------
292
293   @Override /* Writer */
294   public SerializerWriter append(char c) throws IOException {
295      out.write(c);
296      return this;
297   }
298
299   @Override /* Writer */
300   public void write(char[] cbuf, int off, int len) throws IOException {
301      out.write(cbuf, off, len);
302   }
303
304   @Override /* Writer */
305   public void flush() throws IOException {
306      out.flush();
307   }
308
309   @Override /* Writer */
310   public void close() throws IOException {
311      out.close();
312   }
