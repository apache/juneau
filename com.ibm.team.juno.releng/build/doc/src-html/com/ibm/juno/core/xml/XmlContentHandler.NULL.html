009package com.ibm.juno.core.xml;
010
011import javax.xml.stream.*;
012
013import com.ibm.juno.core.dto.atom.*;
014import com.ibm.juno.core.xml.annotation.*;
015
016/**
017 * Customization class that allows a bean (or parts of a bean) to be serialized as XML text or mixed content.
018 * <p>
019 *    For example, the ATOM specification allows text elements (e.g. title, subtitle...)
020 *       to be either plain text or XML depending on the value of a <xa>type</xa> attribute.
021 *    The behavior of text escaping thus depends on that attribute.
022 *
023 * <p class='bcode'>
024 *    <xt>&lt;feed</xt> <xa>xmlns</xa>=<xs>"http://www.w3.org/2005/Atom"</xs><xt>&gt;</xt>
025 *       <xt>&lt;title</xt> <xa>type</xa>=<xs>"html"</xs><xt>&gt;</xt>
026 *          &amp;lt;p&amp;gt;&amp;lt;i&amp;gt;This is the title&amp;lt;/i&amp;gt;&amp;lt;/p&amp;gt;
027 *       <xt>&lt;/title&gt;</xt>
028 *       <xt>&lt;title</xt> <xa>type</xa>=<xs>"xhtml"</xs><xt>&gt;</xt>
029 *          <xt>&lt;div</xt> <xa>xmlns</xa>=<xs>"http://www.w3.org/1999/xhtml"</xs><xt>&gt;</xt>
030 *             <xt>&lt;p&gt;&lt;i&gt;</xt>This is the subtitle<xt>&lt;/i&gt;&lt;/p&gt;</xt>
031 *          <xt>&lt;/div&gt;</xt>
032 *       <xt>&lt;/title&gt;</xt>
033 *    <xt>&lt;/feed&gt;</xt>
034 * </p>
035 *
036 * <p>
037 *    The ATOM {@link Text} class (the implementation for both the <xt>&lt;title&gt;</xt> and <xt>&lt;subtitle&gt;</xt>
038 *       tags shown above) then associates a content handler through the {@link Xml#contentHandler()} annotation
039 *       on the bean property containing the text, like so...
040 *
041 * <p class='bcode'>
042 *    <ja>@Xml</ja>(format=<jsf>ATTR</jsf>)
043 *    <jk>public</jk> String getType() {
044 *       <jk>return</jk> <jf>type</jf>;
045 *    }
046 *
047 *    <ja>@Xml</ja>(format=<jsf>CONTENT</jsf>, contentHandler=TextContentHandler.<jk>class</jk>)
048 *    <jk>public</jk> String getText() {
049 *       <jk>return</jk> <jf>text</jf>;
050 *    }
051 *
052 *    <jk>public void</jk> setText(String text) {
053 *       <jk>this</jk>.<jf>text</jf> = text;
054 *    }
055 * </p>
056 *
057 * <p>
058 *    The content handler that transforms the output is shown below...
059 *
060 * <p class='bcode'>
061 *    <jk>public static class</jk> TextContentHandler <jk>implements</jk> XmlContentHandler&lt;Text&gt; {
062 *
063 *       <ja>@Override</ja>
064 *       <jk>public void</jk> parse(XMLStreamReader r, Text text) <jk>throws</jk> Exception {
065 *          String type = text.<jf>type</jf>;
066 *          <jk>if</jk> (type != <jk>null</jk> && type.equals(<js>"xhtml"</js>))
067 *             text.<jf>text</jf> = <jsm>decode</jsm>(readXmlContents(r).trim());
068 *          <jk>else</jk>
069 *             text.<jf>text</jf> = <jsm>decode</jsm>(r.getElementText().trim());
070 *       }
071 *
072 *       <ja>@Override</ja>
073 *       <jk>public void</jk> serialize(XmlSerializerWriter w, Text text) <jk>throws</jk> Exception {
074 *          String type = text.<jf>type</jf>;
075 *          String content = text.<jf>text</jf>;
076 *          <jk>if</jk> (type != <jk>null</jk> && type.equals(<js>"xhtml"</js>))
077 *             w.encodeTextInvalidChars(content);
078 *          <jk>else</jk>
079 *             w.encodeText(content);
080 *       }
081 *    }
082 * </p>
083 *
084 * <h6 class='topic'>Notes</h6>
085 * <ul>
086 *    <li>The {@link Xml#contentHandler()} annotation can only be specified on a bean class, or a bean property
087 *       of format {@link XmlFormat#CONTENT}.
088 * </ul>
089 *
090 *
091 * @author James Bognar (jbognar@us.ibm.com)
092 * @param <T> The class type of the bean
093 */
094public interface XmlContentHandler<T> {
095
096   /**
097    * Represents <jk>null</jk> on the {@link Xml#contentHandler()} annotation.
098    */
099   public static interface NULL extends XmlContentHandler<Object> {}
100
101   /**
102    * Reads XML element content the specified reader and sets the appropriate value on the specified bean.
103    * <p>
104    *    When this method is called, the attributes have already been parsed and set on the bean.
105    *    Therefore, if the content handling is different based on some XML attribute (e.g.
106    *       <code><xa>type</xa>=<xs>"text/xml"</xs></code> vs <code><xa>type</xa>=<xs>"text/plain"</xs></code>)
107    *       then that attribute value can be obtained via the set bean property.
108    *
109    * @param r The XML stream reader.
110    *    When called, the reader is positioned on the element containing the text to read.
111    *    For example, calling <code>r.getElementText()</code> can be called immediately
112    *    to return the element text if the element contains only characters and whitespace.
113    *    However typically, the stream is going to contain XML elements that need to
114    *    be handled special (otherwise you wouldn't need to use an <code>XmlContentHandler</code>
115    *    to begin with).
116    * @param bean The bean where the parsed contents are going to be placed.
117    *    Subclasses determine how the content maps to values in the bean.
118    *    However, typically the contents map to a single property on the bean.
119    * @throws Exception If any problem occurs.  Causes parse to fail.
120    */
121   public void parse(XMLStreamReader r, T bean) throws Exception;
122
123   /**
124    * Writes XML element content from values in the specified bean.
125    *
126    * @param w The XML output writer.
127    *    When called, the XML element/attributes and
128    *       whitespace/indentation (if enabled) have already been written to the stream.
129    *    Subclasses must simply write the contents of the element.
130    * @param bean The bean whose values will be converted to XML content.
131    * @throws Exception If any problems occur.  Causes serialize to fail.
132    */
133   public void serialize(XmlSerializerWriter w, T bean) throws Exception;
134
135}
