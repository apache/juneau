021/**
022 * Transforms {@link Calendar Calendars} to {@link String Strings}.
023 *
024 *
025 * <h6 class='topic'>Behavior-specific subclasses</h6>
026 * <p>
027 *    The following direct subclasses are provided for convenience:
028 * <ul>
029 *    <li>{@link ToString} - Transforms to {@link String Strings} using the {@code Date.toString()} method.
030 *    <li>{@link ISO8601DT} - Transforms to ISO8601 date-time strings.
031 *    <li>{@link ISO8601DTZ} - Same as {@link ISO8601DT}, except always serializes in GMT.
032 *    <li>{@link RFC2822DT} - Transforms to RFC2822 date-time strings.
033 *    <li>{@link RFC2822DTZ} - Same as {@link RFC2822DT}, except always serializes in GMT.
034 *    <li>{@link RFC2822D} - Transforms to RFC2822 date strings.
035 *    <li>{@link Simple} - Transforms to simple <js>"yyyy/MM/dd HH:mm:ss"</js> strings.
036 *    <li>{@link Medium} - Transforms to {@link DateFormat#MEDIUM} strings.
037 * </ul>
038 *
039 *
040 * @author James Bognar (jbognar@us.ibm.com)
041 */
042public class CalendarFilter extends PojoFilter<Calendar,String> {
043
044   private static final TimeZone GMT = TimeZone.getTimeZone("GMT");
045
046   /**
047    * Transforms {@link Calendar Calendars} to {@link String Strings} using the {@code Date.toString()} method.
048    *
049    * <dl>
050    *    <dt>Example output:</dt>
051    *    <dd>
052    *       <ul>
053    *          <li><js>"Wed Jul 04 15:30:45 EST 2001"</js>
054    *       </ul>
055    *    </dd>
056    * </dl>
057    */
058   public static class ToString extends CalendarFilter {
059      /** Constructor */
060      public ToString() {
061         super("EEE MMM dd HH:mm:ss zzz yyyy");
062      }
063   }
064
065   /**
066    * Transforms {@link Calendar Calendars} to ISO8601 date-time strings.
067    *
068    * <dl>
069    *    <dt>Example output:</dt>
070    *    <dd>
071    *       <ul>
072    *          <li><js>"2001-07-04T15:30:45-05:00"</js>
073    *          <li><js>"2001-07-04T15:30:45Z"</js>
074    *       </ul>
075    *    </dd>
076    *    <dt>Example input:</dt>
077    *    <dd>
078    *       <ul>
079    *          <li><js>"2001-07-04T15:30:45-05:00"</js>
080    *          <li><js>"2001-07-04T15:30:45Z"</js>
081    *          <li><js>"2001-07-04T15:30:45.1Z"</js>
082    *          <li><js>"2001-07-04T15:30Z"</js>
083    *          <li><js>"2001-07-04"</js>
084    *          <li><js>"2001-07"</js>
085    *          <li><js>"2001"</js>
086    *        </ul>
087    *    </dd>
088    * </dl>
089    */
090   public static class ISO8601DT extends CalendarFilter {
091
092      /** Constructor */
093      public ISO8601DT() {}
094
095      @Override /* PojoFilter */
096      public Calendar unfilter(String o, ClassMeta<?> hint) throws ParseException {
097         try {
098            if (StringUtils.isEmpty(o))
099               return null;
100            return convert(DatatypeConverter.parseDateTime(o), hint);
101         } catch (Exception e) {
102            throw new ParseException(e);
103         }
104      }
105
106      @Override /* PojoFilter */
107      public String filter(Calendar o) {
108         return DatatypeConverter.printDateTime(o);
109      }
110   }
111
112   /**
113    * Same as {@link ISO8601DT}, except always serializes in GMT.
114    * <p>
115    * Example output: <js>"2001-07-04T15:30:45Z"</js>
116    */
117   public static class ISO8601DTZ extends CalendarFilter {
118
119      /** Constructor */
120      public ISO8601DTZ() {}
121
122      @Override /* PojoFilter */
123      public Calendar unfilter(String o, ClassMeta<?> hint) throws ParseException {
124         try {
125            if (StringUtils.isEmpty(o))
126               return null;
127            return convert(DatatypeConverter.parseDateTime(o), hint);
128         } catch (Exception e) {
129            throw new ParseException(e);
130         }
131      }
132
133      @Override /* PojoFilter */
134      public String filter(Calendar o) {
135         if (o.getTimeZone().getRawOffset() != 0) {
136            Calendar c = Calendar.getInstance(GMT);
137            c.setTime(o.getTime());
138            o = c;
139         }
140         return DatatypeConverter.printDateTime(o);
141      }
142   }
143
144   /**
145    * Transforms {@link Calendar Calendars} to RFC2822 date-time strings.
146    */
147   public static class RFC2822DT extends CalendarFilter {
148      /** Constructor */
149      public RFC2822DT() {
150         super("EEE, dd MMM yyyy HH:mm:ss Z");
151      }
152   }
153
154   /**
155    * Same as {@link RFC2822DT}, except always serializes in GMT.
156    * <p>
157    * Example output: <js>"Wed, 31 Jan 2001 12:34:56 +0000"</js>
158    */
159   public static class RFC2822DTZ extends CalendarFilter {
160      /** Constructor */
161      public RFC2822DTZ() {
162         super("EEE, dd MMM yyyy HH:mm:ss 'GMT'", GMT);
163      }
164   }
165
166   /**
167    * Transforms {@link Calendar Calendars} to RFC2822 date strings.
168    */
169   public static class RFC2822D extends CalendarFilter {
170      /** Constructor */
171      public RFC2822D() {
172         super("dd MMM yyyy");
173      }
174   }
175
176   /**
177    * Transforms {@link Calendar Calendars} to simple <js>"yyyy/MM/dd HH:mm:ss"</js> strings.
178    */
179   public static class Simple extends CalendarFilter {
180      /** Constructor */
181      public Simple() {
182         super("yyyy/MM/dd HH:mm:ss");
183      }
184   }
185
186   /**
187    * Transforms {@link Calendar Calendars} to {@link DateFormat#MEDIUM} strings.
188    */
189   public static class Medium extends CalendarFilter {
190      /** Constructor */
191      public Medium() {
192         super(DateFormat.getDateInstance(DateFormat.MEDIUM));
193      }
194   }
195
196   /** The formatter to convert dates to Strings. */
197   private DateFormat format;
198
199   private TimeZone timeZone;
200
201   /**
202    * Default constructor.
203    * <p>
204    *    This constructor is used when <code>filter()</code> and <code>unfilter()</code> are overridden by subclasses.
205    */
206   public CalendarFilter() {}
207
208   /**
209    * Construct a filter using the specified date format string that will be
210    *    used to construct a {@link SimpleDateFormat} that will be used to convert
211    *    dates to strings.
212    *
213    * @param simpleDateFormat The {@link SimpleDateFormat} pattern.
214    */
215   public CalendarFilter(String simpleDateFormat) {
216      this(new SimpleDateFormat(simpleDateFormat));
217   }
218
219   /**
220    * Construct a filter using the specified date format string that will be
221    *    used to construct a {@link SimpleDateFormat} that will be used to convert
222    *    dates to strings.
223    *
224    * @param simpleDateFormat The {@link SimpleDateFormat} pattern.
225    * @param timeZone The time zone to associate with the date pattern.
226    */
227   public CalendarFilter(String simpleDateFormat, TimeZone timeZone) {
228      this(new SimpleDateFormat(simpleDateFormat));
229      format.setTimeZone(timeZone);
230      this.timeZone = timeZone;
231   }
232
233   /**
234    * Construct a filter using the specified {@link DateFormat} that will be used to convert
235    *    dates to strings.
236    *
237    * @param format The format to use to convert dates to strings.
238    */
239   public CalendarFilter(DateFormat format) {
240      super();
241      this.format = format;
242   }
243
244   /**
245    * Converts the specified {@link Calendar} to a {@link String}.
246    */
247   @Override /* PojoFilter */
248   public String filter(Calendar o) {
249      DateFormat df = format;
250      TimeZone tz1 = o.getTimeZone();
251      TimeZone tz2 = format.getTimeZone();
252      if (timeZone == null && ! tz1.equals(tz2)) {
253         df = (DateFormat)format.clone();
254         df.setTimeZone(tz1);
255      }
256      return df.format(o.getTime());
257   }
258
259   /**
260    * Converts the specified {@link String} to a {@link Calendar}.
261    */
262   @Override /* PojoFilter */
263   public Calendar unfilter(String o, ClassMeta<?> hint) throws ParseException {
264      try {
265         if (StringUtils.isEmpty(o))
266            return null;
267         return convert(format.parse(o), hint);
268      } catch (Exception e) {
269         throw new ParseException(e);
270      }
271   }
272
273   private static Calendar convert(Calendar in, ClassMeta<?> hint) throws Exception {
274      if (hint.isInstance(in) || ! hint.canCreateNewInstance())
275         return in;
276      Calendar c = (Calendar)hint.newInstance();
277      c.setTime(in.getTime());
278      c.setTimeZone(in.getTimeZone());
279      return c;
280   }
281
282   private static Calendar convert(Date in, ClassMeta<?> hint) throws Exception {
283      if (hint == null || ! hint.canCreateNewInstance())
284         hint = BeanContext.DEFAULT.getClassMeta(GregorianCalendar.class);
285      Calendar c = (Calendar)hint.newInstance();
286      c.setTime(in);
287      return c;
288   }
