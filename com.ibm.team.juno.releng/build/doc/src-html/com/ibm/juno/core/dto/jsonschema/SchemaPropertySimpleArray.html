009package com.ibm.juno.core.dto.jsonschema;
010
011/**
012 * Convenience class for representing a property that's an array of simple types.
013 * <p>
014 *    An instance of this object is equivalent to calling...
015 *
016 * <p class='bcode'>
017 *    SchemaProperty p = <jk>new</jk> SchemaProperty(name)
018 *       .setType(JsonType.<jsf>ARRAY</jsf>)
019 *       .setItems(
020 *          <jk>new</jk> Schema().setType(elementType)
021 *       );
022 * </p>
023 *
024 * @author James Bognar (jbognar@us.ibm.com)
025 */
026public class SchemaPropertySimpleArray extends SchemaProperty {
027
028   /**
029    * Constructor.
030    *
031    * @param name The name of the schema property.
032    * @param elementType The JSON type of the elements in the array.
033    */
034   public SchemaPropertySimpleArray(String name, JsonType elementType) {
035      setName(name);
036      setType(JsonType.ARRAY);
037      setItems(
038         new Schema().setType(elementType)
039      );
040   }
041}
