089   @Override /* Map */
090   public void clear() {
091      Set<String> changes = createChanges();
092      writeLock();
093      try {
094         if (changes != null)
095            for (String k : keySet())
096               changes.add(getFullKey(name, k));
097         entries.clear();
098         lines.clear();
099         headerComments.clear();
100      } finally {
101         writeUnlock();
102      }
103      signalChanges(changes);
104   }

239   String put(String key, String value, boolean encoded, Set<String> changes) {
240      writeLock();
241      try {
242         addLine(key, encoded);
243         String prev = entries.put(key, value);
244         addChange(changes, key, prev, value);
245         return prev;
246      } finally {
247         writeUnlock();
248      }
249   }

251   @Override /* Map */
252   public void putAll(Map<? extends String,? extends String> map) {
253      Set<String> changes = createChanges();
254      for (Map.Entry<? extends String,? extends String> e : map.entrySet())
255         put(e.getKey(), e.getValue(), false, changes);
256      signalChanges(changes);
257   }

262      String old = remove(key, changes);

265   }

267   String remove(Object key, Set<String> changes) {
268      writeLock();
269      try {
270         String prev = entries.remove(key);
271         addChange(changes, key.toString(), prev, null);
272         removeLine(key.toString());
273         return prev;
274      } finally {
275         writeUnlock();
276      }
277   }
278
279   private void removeLine(String key) {
280      for (Iterator<String> i = lines.iterator(); i.hasNext();) {
281         String k = i.next();
282         if (k.startsWith("*") || k.startsWith(">")) {
283            if (k.substring(1).equals(key)) {
284               i.remove();
285               break;
286            }
287         }
288      }
289   }

305   /**
306    * Returns <jk>true</jk> if the specified entry is encoded.
307    *
308    * @param key The key.
309    * @return <jk>true</jk> if the specified entry is encoded.
310    */
311   public boolean isEncoded(String key) {
312      readLock();
313      try {
314         for (String s : lines)
315            if (s.length() > 1)
316               if (s.substring(1).equals(key))
317                  return s.charAt(0) == '*';
318         return false;
319      } finally {
320         readUnlock();
321      }
322   }

324   /**
325    * Adds header comments to this section.
326    * @see ConfigFile#addHeaderComments(String, String...) for a description.
327    * @param comments The comment lines to add to this section.
328    * @return This object (for method chaining).
329    */
330   public Section addHeaderComments(List<String> comments) {
331      writeLock();
332      try {
333         for (String c : comments) {
334            if (c == null)
335               c = "";
336            if (! c.startsWith("#"))
337               c = "#" + c;
338            this.headerComments.add(c);
339         }
340         return this;
341      } finally {
342         writeUnlock();
343      }
344   }

347    * Removes all header comments from this section.
348    */
349   public void clearHeaderComments() {
350      writeLock();
351      try {
352         this.headerComments.clear();
353      } finally {
354         writeUnlock();
355      }
356   }

358   /**
359    * Serialize this section.
360    * @param out What to serialize to.
361    * @param format The format (e.g. INI, BATCH, SHELL).
362    */
363   public void writeTo(PrintWriter out, ConfigFileFormat format) {
364      readLock();
365      try {
366         if (format == INI) {
367            for (String s : headerComments)
368               out.append(s).println();
369            if (! name.equals("default"))
370               out.append('[').append(name).append(']').println();
371            for (String l : lines) {
372               char c = (l.length() > 0 ? l.charAt(0) : 0);
373               if (c == '>' || c == '*'){
374                  boolean encode = c == '*';
375                  String key = l.substring(1);
376                  String val = entries.get(key);
377                  if (val.indexOf('\n') != -1)
378                     val = val.replaceAll("(\\r?\\n)", "$1\t");
379                  if (val.indexOf('=') != -1)
380                     val = val.replace("=", "\\u003D");
381                  if (val.indexOf('#') != -1)
382                     val = val.replace("#", "\\u0023");
383                  out.append(key);
384                  if (encode)
385                     out.append('*');
386                  out.append(" = ");
387                  if (encode)
388                     out.append('{').append(configFile.getEncoder().encode(key, val)).append('}');
389                  else
390                     out.append(val);
391                  out.println();
392               } else {
393                  out.append(l).println();
394               }
395            }
396
397         } else if (format == BATCH) {
398            String section = name.replaceAll("\\.\\/", "_");
399            for (String l : headerComments) {
400               l = trimComment(l);
401               if (! l.isEmpty())
402                  out.append("rem ").append(l);
403               out.println();
404            }
405            for (String l : lines) {
406               char c = (l.length() > 0 ? l.charAt(0) : 0);
407               if (c == '>' || c == '*') {
408                  String key = l.substring(1);
409                  String val = entries.get(key);
410                  out.append("set ");
411                  if (! name.equals("default"))
412                     out.append(section).append('_');
413                  out.append(key.replaceAll("\\.\\/", "_")).append(" = ").append(val).println();
414               } else {
415                  l = trimComment(l);
416                  if (! l.isEmpty())
417                     out.append("rem ").append(l);
418                  out.println();
419               }
420            }
421
422         } else if (format == SHELL) {
423            String section = name.replaceAll("\\.\\/", "_");
424            for (String l : headerComments) {
425               l = trimComment(l);
426               if (! l.isEmpty())
427                  out.append("# ").append(l);
428               out.println();
429            }
430            for (String l : lines) {
431               char c = (l.length() > 0 ? l.charAt(0) : 0);
432               if (c == '>' || c == '*'){
433                  String key = l.substring(1);
434                  String val = entries.get(key).replaceAll("\\\\", "\\\\\\\\");
435                  out.append("export ");
436                  if (! name.equals("default"))
437                     out.append(section).append('_');
438                  out.append(key.replaceAll("\\.\\/", "_")).append('=').append('"').append(val).append('"').println();
439               } else {
440                  l = trimComment(l);
441                  if (! l.isEmpty())
442                     out.append("# ").append(l);
443                  out.println();
444               }
445            }
446         }
447      } finally {
448         readUnlock();
449      }
450   }

456   /*
457    * Add lines to this section.
458    */
459   Section addLines(Set<String> changes, String...l) {
460      writeLock();
461      try {
462         if (l == null)
463            l = new String[0];
464         for (int i = 0; i < l.length; i++) {
465            String line = l[i];
466            if (line == null)
467               line = "";
468            if (isComment(line))
469               this.lines.add(line);
470            else if (isAssignment(line)) {
471               // Key/value pairs are stored as either ">key" or "*key";
472               String key = StringUtils.replaceUnicodeSequences(line.substring(0, line.indexOf('=')).trim());
473               String val = StringUtils.replaceUnicodeSequences(line.substring(line.indexOf('=')+1).trim());
474               boolean encoded = key.length() > 1 && key.endsWith("*");
475               if (encoded) {
476                  key = key.substring(0, key.lastIndexOf('*'));
477                  String v = val.toString().trim();
478                  if (v.startsWith("{") && v.endsWith("}"))
479                     val = configFile.getEncoder().decode(key, v.substring(1, v.length()-1));
480                  else
481                     configFile.setHasBeenModified();
482               }
483               if (containsKey(key)) {
484                  entries.remove(key);
485                  lines.remove('*' + key);
486                  lines.remove('>' + key);
487               }
488               lines.add((encoded ? '*' : '>') + key);
489               addChange(changes, key, entries.put(key, val), val);
490            } else {
491               this.lines.add(line);
492            }
493         }
494         return this;
495      } finally {
496         writeUnlock();
497      }
498   }

500   /*
501    * Remove all "#*" lines at the end of this section so they can
502    * be associated with the next section.
503    */
504   List<String> removeTrailingComments() {
505      LinkedList<String> l = new LinkedList<String>();
506      while ((! lines.isEmpty()) && lines.getLast().startsWith("#"))
507         l.addFirst(lines.removeLast());
508      return l;
509   }

515   private void addLine(String key, boolean encoded) {
516      for (Iterator<String> i = lines.iterator(); i.hasNext();) {
517         String k = i.next();
518         if ((k.startsWith("*") || k.startsWith(">")) && k.substring(1).equals(key)) {
519            if (k.startsWith("*") && encoded || k.startsWith(">") && ! encoded)
520               return;
521            i.remove();
522         }
523      }
524      lines.add((encoded ? "*" : ">") + key);
525   }
526
527   private void readLock() {
528      lock.readLock().lock();
529   }
530
531   private void readUnlock() {
532      lock.readLock().unlock();
533   }
534
535   private void writeLock() {
536      if (readOnly)
537         throw new UnsupportedOperationException("Cannot modify read-only ConfigFile.");
538      lock.writeLock().lock();
539   }
540
541   private void writeUnlock() {
542      lock.writeLock().unlock();
543   }
544
545   private String trimComment(String s) {
546      return s.replaceAll("^\\s*\\#\\s*", "").trim();
547   }
548
549   private Set<String> createChanges() {
550      return (configFile != null && configFile.getListeners().size() > 0 ? new LinkedHashSet<String>() : null);
551   }
552
553   private void signalChanges(Set<String> changes) {
554      if (changes != null && ! changes.isEmpty())
555         for (ConfigFileListener l : configFile.getListeners())
556            l.onChange(configFile, changes);
557   }
558
559   private void addChange(Set<String> changes, String key, String oldVal, String newVal) {
560      if (changes != null)
561         if (! StringUtils.isEquals(oldVal, newVal))
562            changes.add(getFullKey(name, key));
563   }
