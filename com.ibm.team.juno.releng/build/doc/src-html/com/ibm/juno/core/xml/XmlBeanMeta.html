009package com.ibm.juno.core.xml;
010
011import java.util.*;
012
013import com.ibm.juno.core.*;
014import com.ibm.juno.core.xml.annotation.*;
015
016/**
017 * Metadata on beans specific to the XML serializers and parsers pulled from the {@link Xml @Xml} annotation on the class.
018 *
019 * @author James Bognar (jbognar@us.ibm.com)
020 * @param <T> The bean class type.
021 */
022public class XmlBeanMeta<T> {
023
024   // XML related fields
025   private final Map<String,BeanPropertyMeta<T>> xmlAttrs;                        // Map of bean properties that are represented as XML attributes.
026   private final BeanPropertyMeta<T> xmlContent;                                  // Bean property that is represented as XML content within the bean element.
027   private final XmlContentHandler<T> xmlContentHandler;                          // Class used to convert bean to XML content.
028   private final Map<String,BeanPropertyMeta<T>> childElementProperties;          // Properties defined with @Xml.childName annotation.
029   private final BeanMeta<T> beanMeta;
030
031   /**
032    * Constructor.
033    *
034    * @param beanMeta The metadata on the bean that this metadata applies to.
035    * @param pNames Only look at these property names.  If <jk>null</jk>, apply to all bean properties.
036    */
037   public XmlBeanMeta(BeanMeta<T> beanMeta, String[] pNames) {
038      this.beanMeta = beanMeta;
039      Class<T> c = beanMeta.getClassMeta().getInnerClass();
040
041      Map<String,BeanPropertyMeta<T>> tXmlAttrs = new LinkedHashMap<String,BeanPropertyMeta<T>>();
042      BeanPropertyMeta<T> tXmlContent = null;
043      XmlContentHandler<T> tXmlContentHandler = null;
044      Map<String,BeanPropertyMeta<T>> tChildElementProperties = new LinkedHashMap<String,BeanPropertyMeta<T>>();
045
046      for (BeanPropertyMeta<T> p : beanMeta.getPropertyMetas(pNames)) {
047         XmlFormat xf = p.getXmlMeta().getXmlFormat();
048         if (xf == XmlFormat.ATTR)
049            tXmlAttrs.put(p.getName(), p);
050         else if (xf == XmlFormat.CONTENT) {
051            if (tXmlContent != null)
052               throw new BeanRuntimeException(c, "Multiple instances of CONTENT properties defined on class.  Only one property can be designated as such.");
053            tXmlContent = p;
054            tXmlContentHandler = p.getXmlMeta().getXmlContentHandler();
055         }
056         // Look for any properties that are collections with @Xml.childName specified.
057         String n = p.getXmlMeta().getChildName();
058         if (n != null) {
059            if (tChildElementProperties.containsKey(n))
060               throw new BeanRuntimeException(c, "Multiple properties found with the name ''{0}''.", n);
061            tChildElementProperties.put(n, p);
062         }
063      }
064
065      xmlAttrs = Collections.unmodifiableMap(tXmlAttrs);
066      xmlContent = tXmlContent;
067      xmlContentHandler = tXmlContentHandler;
068      childElementProperties = (tChildElementProperties.isEmpty() ? null : Collections.unmodifiableMap(tChildElementProperties));
069   }
070
071   /**
072    * Returns the list of properties annotated with an {@link Xml#format()} of {@link XmlFormat#ATTR}.
073    * In other words, the list of properties that should be rendered as XML attributes instead of child elements.
074    *
075    * @return Metadata on the XML attribute properties of the bean.
076    */
077   protected Map<String,BeanPropertyMeta<T>> getXmlAttrProperties() {
078      return xmlAttrs;
079   }
080
081   /**
082    * Returns the bean property annotated with an {@link Xml#format()} value of {@link XmlFormat#CONTENT}
083    *
084    * @return The bean property, or <jk>null</jk> if annotation is not specified.
085    */
086   protected BeanPropertyMeta<T> getXmlContentProperty() {
087      return xmlContent;
088   }
089
090   /**
091    * Return the XML content handler for this bean.
092    *
093    * @return The XML content handler for this bean, or <jk>null</jk> if no content handler is defined.
094    */
095   protected XmlContentHandler<T> getXmlContentHandler() {
096      return xmlContentHandler;
097   }
098
099   /**
100    * Returns the child element properties for this bean.
101    * See {@link Xml#childName()}
102    *
103    * @return The child element properties for this bean, or <jk>null</jk> if no child element properties are defined.
104    */
105   protected Map<String,BeanPropertyMeta<T>> getChildElementProperties() {
106      return childElementProperties;
107   }
108
109   /**
110    * Returns bean property meta with the specified name.
111    * This is identical to calling {@link BeanMeta#getPropertyMeta(String)} except it first retrieves
112    *    the bean property meta based on the child name (e.g. a property whose name is "people", but whose child name is "person").
113    *
114    * @param fieldName The bean property name.
115    * @return The property metadata.
116    */
117   protected BeanPropertyMeta<T> getPropertyMeta(String fieldName) {
118      if (childElementProperties != null) {
119         BeanPropertyMeta<T> bpm = childElementProperties.get(fieldName);
120         if (bpm != null)
121            return bpm;
122      }
123      return beanMeta.getPropertyMeta(fieldName);
124   }
125}
