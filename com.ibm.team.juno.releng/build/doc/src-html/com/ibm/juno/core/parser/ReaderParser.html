019/**
020 * Subclass of {@link Parser} for characters-based parsers.
021 *
022 *
023 * <h6 class='topic'>Description</h6>
024 * <p>
025 *    This class is typically the parent class of all character-based parsers.
026 *    It has 1 abstract method to implement...
027 * <ul>
028 *    <li><code>parse(Reader, ClassMeta, ParserContext)</code>
029 * </ul>
030 *
031 *
032 * <h6 class='topic'>@Consumes annotation</h6>
033 * <p>
034 *    The media types that this parser can handle is specified through the {@link Consumes @Consumes} annotation.
035 * <p>
036 *    However, the media types can also be specified programmatically by overriding the {@link #getMediaTypes()} method.
037 *
038 *
039 * @author James Bognar (jbognar@us.ibm.com)
040 */
041public abstract class ReaderParser extends Parser<Reader> {

054   /**
055    * Same as <code>parse(Reader, Class)</code> except parses from a <code>CharSequence</code>.
056    *
057    * @param in The string containing the input.
058    * @param type The class type of the object to create.
059    *    If <jk>null</jk> or <code>Object.<jk>class</jk></code>, object type is based on what's being parsed.
060    * @param <T> The class type of the object to create.
061    * @return The parsed object.
062    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
063    */
064   public <T> T parse(CharSequence in, Class<T> type) throws ParseException {
065      return parse(in, getBeanContext().getClassMeta(type));
066   }
067
068   /**
069    * Same as <code>parse(Reader, ClassMeta)</code> except parses from a <code>CharSequence</code>.
070    *
071    * @param in The string containing the input.
072    * @param type The class type of the object to create.
073    *    If <jk>null</jk> or <code>Object.<jk>class</jk></code>, object type is based on what's being parsed.
074    * @param <T> The class type of the object to create.
075    * @return The parsed object.
076    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
077    */
078   public <T> T parse(CharSequence in, ClassMeta<T> type) throws ParseException {
079      try {
080         if (in == null)
081            return null;
082         return parse(wrapReader(in), in.length(), type, createContext());
083      } catch (IOException e) {
084         throw new ParseException(e); // Won't happen since it's a StringReader.
085      }
086   }

090    *
091    * @param <T> The map class type.

093    * @param <V> The class type of the map values.

095    * @param mapClass The map class type.

124    * Subclasses can override this method to implement their own readers.

137   /**
138    * Parses the contents of the specified reader and loads the results into the specified map.
139    * <p>
140    *    Reader must contain something that serializes to a map (such as text containing a JSON object).
141    * <p>
142    *    Used in the following locations:
143    * <ul>
144    *    <li>The various character-based constructors in {@link ObjectMap} (e.g. {@link ObjectMap#ObjectMap(CharSequence, ReaderParser)}).
145    * </ul>
146    *
147    * @param <K> The key class type.
148    * @param <V> The value class type.
149    * @param in The reader containing the input.
150    * @param estimatedSize The estimated size of the input, or <code>-1</code> if unknown.
151    * @param m The map being loaded.
152    * @param keyType The class type of the keys, or <jk>null</jk> to default to <code>String.<jk>class</jk></code>.<br>
153    * @param valueType The class type of the values, or <jk>null</jk> to default to whatever is being parsed.<br>
154    * @return The same map that was passed in to allow this method to be chained.
155    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
156    * @throws IOException If a problem occurred trying to read from the reader.
157    * @throws UnsupportedOperationException If not implemented.
158    */
159   public final <K,V> Map<K,V> parseIntoMap(Reader in, int estimatedSize, Map<K,V> m, Type keyType, Type valueType) throws ParseException, IOException {
160      ParserContext ctx = createContext();
161      try {
162         if (in == null)
163            throw new IOException("Null input stream or reader passed to parser.");
164         return doParseIntoMap(in, estimatedSize, m, keyType, valueType, ctx);
165      } finally {
166         ctx.close();
167      }
168   }

227   public final <E> Collection<E> parseIntoCollection(Reader in, int estimatedSize, Collection<E> c, Type elementType) throws ParseException, IOException {

273   /**
274    * Parses the specified array input with each entry in the object defined by the {@code argTypes}
275    * argument.
276    * <p>
277    *    Used for converting arrays (e.g. <js>"[arg1,arg2,...]"</js>) into an {@code Object[]} that can be passed
278    *    to the {@code Method.invoke(target, args)} method.
279    * <p>
280    *    Used in the following locations:
281    * <ul>
282    *    <li>Used to parse argument strings in the {@link PojoIntrospector#invokeMethod(Method, Reader)} method.
283    * </ul>
284    *
285    * @param in The input.  Must represent an array.
286    * @param estimatedSize The estimated size of the input, or <code>-1</code> if unknown.
287    * @param argTypes Specifies the type of objects to create for each entry in the array.
288    * @return An array of parsed objects.
289    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
290    * @throws IOException If a problem occurred trying to read from the reader.
291    * @throws UnsupportedOperationException If not implemented.
292    */
293   public final Object[] parseArgs(Reader in, int estimatedSize, ClassMeta<?>[] argTypes) throws ParseException, IOException {
294      if (in == null)
295         throw new IOException("Null reader passed to parser.");
296      if (argTypes == null || argTypes.length == 0)
297         return new Object[0];
298      ParserContext ctx = createContext();
299      try {
300         return doParseArgs(in, estimatedSize, argTypes, ctx);
301      } finally {
302         ctx.close();
303      }
304   }
