025/**
026 * Serializes POJO models to HTML.
027 *
028 *
029 * <h6 class='topic'>Media types</h6>
030 * <p>
031 *    Handles <code>Accept</code> types: <code>text/html</code>
032 * <p>
033 *    Produces <code>Content-Type</code> types: <code>text/html</code>
034 *
035 *
036 * <h6 class='topic'>Description</h6>
037 * <p>
038 *    The conversion is as follows...
039 *    <ul>
040 *       <li>{@link Map Maps} (e.g. {@link HashMap}, {@link TreeMap}) and beans are converted to HTML tables with 'key' and 'value' columns.
041 *       <li>{@link Collection Collections} (e.g. {@link HashSet}, {@link LinkedList}) and Java arrays are converted to HTML ordered lists.
042 *       <li>{@code Collections} of {@code Maps} and beans are converted to HTML tables with keys as headers.
043 *       <li>Everything else is converted to text.
044 *    </ul>
045 * <p>
046 *    This serializer provides several serialization options.  Typically, one of the predefined <jsf>DEFAULT</jsf> serializers will be sufficient.
047 *    However, custom serializers can be constructed to fine-tune behavior.
048 * <p>
049 *    The {@link HtmlLink} annotation can be used on beans to add hyperlinks to the output.
050 *
051 *
052 * <h6 class='topic'>Configurable properties</h6>
053 * <p>
054 *    This class has the following properties associated with it:
055 * <ul>
056 *    <li>{@link HtmlSerializerProperties}
057 *    <li>{@link SerializerProperties}
058 *    <li>{@link BeanContextProperties}
059 * </ul>
060 *
061 *
062 * <h6 class='topic'>Behavior-specific subclasses</h6>
063 * <p>
064 *    The following direct subclasses are provided for convenience:
065 * <ul>
066 *    <li>{@link Sq} - Default serializer, single quotes.
067 *    <li>{@link SqReadable} - Default serializer, single quotes, whitespace added.
068 * </ul>
069 *
070 *
071 * <h6 class='topic'>Examples</h6>
072 * <p class='bcode'>
073 *    <jc>// Use one of the default serializers to serialize a POJO</jc>
074 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(someObject);
075 *
076 *       <jc>// Create a custom serializer that doesn't use whitespace and newlines</jc>
077 *       HtmlSerializer serializer = <jk>new</jk> HtmlSerializer()
078 *          .setProperty(SerializerProperties.<jsf>SERIALIZER_useIndentation</jsf>, <jk>false</jk>);
079 *
080 *       <jc>// Same as above, except uses cloning</jc>
081 *       HtmlSerializer serializer = HtmlSerializer.<jsf>DEFAULT</jsf>.clone()
082 *          .setProperty(SerializerProperties.<jsf>SERIALIZER_useIndentation</jsf>, <jk>false</jk>);
083 *
084 *       <jc>// Serialize POJOs to HTML</jc>
085 *
086 *       <jc>// Produces: </jc>
087 *       <jc>// &lt;ul&gt;&lt;li&gt;1&lt;li&gt;2&lt;li&gt;3&lt;/ul&gt;</jc>
088 *       List l = new ObjectList(1, 2, 3);
089 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(l);
090 *
091 *       <jc>// Produces: </jc>
092 *       <jc>//    &lt;table&gt; </jc>
093 *       <jc>//       &lt;tr&gt;&lt;th&gt;firstName&lt;/th&gt;&lt;th&gt;lastName&lt;/th&gt;&lt;/tr&gt; </jc>
094 *       <jc>//       &lt;tr&gt;&lt;td&gt;Bob&lt;/td&gt;&lt;td&gt;Costas&lt;/td&gt;&lt;/tr&gt; </jc>
095 *       <jc>//       &lt;tr&gt;&lt;td&gt;Billy&lt;/td&gt;&lt;td&gt;TheKid&lt;/td&gt;&lt;/tr&gt; </jc>
096 *       <jc>//       &lt;tr&gt;&lt;td&gt;Barney&lt;/td&gt;&lt;td&gt;Miller&lt;/td&gt;&lt;/tr&gt; </jc>
097 *       <jc>//    &lt;/table&gt; </jc>
098 *       l = <jk>new</jk> ObjectList();
099 *       l.add(<jk>new</jk> ObjectMap(<js>"{firstName:'Bob',lastName:'Costas'}"</js>));
100 *       l.add(<jk>new</jk> ObjectMap(<js>"{firstName:'Billy',lastName:'TheKid'}"</js>));
101 *       l.add(<jk>new</jk> ObjectMap(<js>"{firstName:'Barney',lastName:'Miller'}"</js>));
102 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(l);
103 *
104 *       <jc>// Produces: </jc>
105 *       <jc>//    &lt;table&gt; </jc>
106 *       <jc>//       &lt;tr&gt;&lt;th&gt;key&lt;/th&gt;&lt;th&gt;value&lt;/th&gt;&lt;/tr&gt; </jc>
107 *       <jc>//       &lt;tr&gt;&lt;td&gt;foo&lt;/td&gt;&lt;td&gt;bar&lt;/td&gt;&lt;/tr&gt; </jc>
108 *       <jc>//       &lt;tr&gt;&lt;td&gt;baz&lt;/td&gt;&lt;td&gt;123&lt;/td&gt;&lt;/tr&gt; </jc>
109 *       <jc>//    &lt;/table&gt; </jc>
110 *       Map m = <jk>new</jk> ObjectMap(<js>"{foo:'bar',baz:123}"</js>);
111 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(m);
112 *
113 *       <jc>// HTML elements can be nested arbitrarily deep</jc>
114 *       <jc>// Produces: </jc>
115 *       <jc>//   &lt;table&gt; </jc>
116 *       <jc>//      &lt;tr&gt;&lt;th&gt;key&lt;/th&gt;&lt;th&gt;value&lt;/th&gt;&lt;/tr&gt; </jc>
117 *       <jc>//      &lt;tr&gt;&lt;td&gt;foo&lt;/td&gt;&lt;td&gt;bar&lt;/td&gt;&lt;/tr&gt; </jc>
118 *       <jc>//      &lt;tr&gt;&lt;td&gt;baz&lt;/td&gt;&lt;td&gt;123&lt;/td&gt;&lt;/tr&gt; </jc>
119 *       <jc>//      &lt;tr&gt;&lt;td&gt;someNumbers&lt;/td&gt;&lt;td&gt;&lt;ul&gt;&lt;li&gt;1&lt;li&gt;2&lt;li&gt;3&lt;/ul&gt;&lt;/td&gt;&lt;/tr&gt; </jc>
120 *       <jc>//      &lt;tr&gt;&lt;td&gt;someSubMap&lt;/td&gt;&lt;td&gt; </jc>
121 *       <jc>//         &lt;table&gt; </jc>
122 *       <jc>//            &lt;tr&gt;&lt;th&gt;key&lt;/th&gt;&lt;th&gt;value&lt;/th&gt;&lt;/tr&gt; </jc>
123 *       <jc>//            &lt;tr&gt;&lt;td&gt;a&lt;/td&gt;&lt;td&gt;b&lt;/td&gt;&lt;/tr&gt; </jc>
124 *       <jc>//         &lt;/table&gt; </jc>
125 *       <jc>//      &lt;/td&gt;&lt;/tr&gt; </jc>
126 *       <jc>//   &lt;/table&gt; </jc>
127 *       Map m = <jk>new</jk> ObjectMap(<js>"{foo:'bar',baz:123}"</js>);
128 *       m.put("someNumbers", new ObjectList(1, 2, 3));
129 *       m.put(<js>"someSubMap"</js>, new ObjectMap(<js>"{a:'b'}"</js>));
130 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(m);
131 * </p>
132 *
133 *
134 * @author James Bognar (jbognar@us.ibm.com)
135 */

169   /**
170    * Main serialization routine.
171    *
172    * @param o The object being serialized.
173    * @param w The writer to serialize to.
174    * @param ctx The serialization context object.
175    *
176    * @return The same writer passed in.
177    * @throws IOException If a problem occurred trying to send output to the writer.
178    */
179   private HtmlSerializerWriter doSerialize(Object o, HtmlSerializerWriter w, HtmlSerializerContext ctx) throws SerializeException, IOException {
180      serializeAnything(w, o, null, ctx, null, ctx.getInitialDepth()-1, null);
181      return w;
182   }

184   /**
185    * Serialize the specified object to the specified writer.
186    *
187    * @param out The writer.
188    * @param o The object to serialize.
189    * @param eType The expected type of the object if this is a bean property.
190    * @param ctx The context object that lives for the duration of this serialization.
191    * @param name The attribute name of this object if this object was a field in a JSON object (i.e. key of a {@link java.util.Map.Entry} or property name of a bean).
192    * @param indent The current indentation value.
193    * @param pMeta The bean property being serialized, or <jk>null</jk> if we're not serializing a bean property.
194    * @throws IOException
195    * @throws SerializeException If a problem occurred trying to convert the output.
196    */
197   @SuppressWarnings({ "rawtypes", "unchecked" })
198   protected void serializeAnything(HtmlSerializerWriter out, Object o, ClassMeta<?> eType, HtmlSerializerContext ctx, String name, int indent, BeanPropertyMeta pMeta) throws IOException, SerializeException {
199
200      BeanContext bc = ctx.getBeanContext();
201      ClassMeta<?> aType = null;       // The actual type
202      ClassMeta<?> gType = object();   // The generic type
203
204      if (eType == null)
205         eType = object();
206
207      aType = ctx.push(name, o, eType);
208
209      // Handle recursion
210      if (aType == null) {
211         o = null;
212         aType = object();
213      }
214
215      ctx.indent += indent;
216      int i = ctx.indent;
217
218      // Determine the type.
219      if (o == null || (aType.isChar() && ((Character)o).charValue() == 0))
220         out.tag(i, "null").nl();
221      else {
222
223         gType = aType.getFilteredClassMeta();
224         String classAttr = null;
225         if (ctx.isAddClassAttrs() && ! eType.equals(aType))
226            classAttr = aType.toString();
227
228         // Filter if necessary
229         PojoFilter filter = aType.getPojoFilter();
230         if (filter != null) {
231            o = filter.filter(o);
232
233            // If the filter's getFilteredClass() method returns Object, we need to figure out
234            // the actual type now.
235            if (gType.isObject())
236               gType = bc.getClassMetaForObject(o);
237         }
238
239         HtmlClassMeta html = gType.getHtmlMeta();
240
241         if (html.isAsXml() || (pMeta != null && pMeta.getHtmlMeta().isAsXml()))
242            super.serializeAnything(out, o, null, ctx, null, null, false, XmlFormat.NORMAL, null);
243         else if (html.isAsPlainText() || (pMeta != null && pMeta.getHtmlMeta().isAsPlainText()))
244            out.write(o == null ? "null" : o.toString());
245         else if (o == null || (gType.isChar() && ((Character)o).charValue() == 0))
246            out.tag(i, "null").nl();
247         else if (gType.hasToObjectMapMethod())
248            serializeMap(out, gType.toObjectMap(o), eType, ctx, classAttr, pMeta);
249         else if (gType.isBean())
250            serializeBeanMap(out, bc.forBean(o), ctx, classAttr, pMeta);
251         else if (gType.isNumber())
252            out.sTag(i, "number").append(o).eTag("number").nl();
253         else if (gType.isBoolean())
254            out.sTag(i, "boolean").append(o).eTag("boolean").nl();
255         else if (gType.isMap()) {
256            if (o instanceof BeanMap)
257               serializeBeanMap(out, (BeanMap)o, ctx, classAttr, pMeta);
258            else
259               serializeMap(out, (Map)o, eType, ctx, classAttr, pMeta);
260         }
261         else if (gType.isCollection()) {
262            if (classAttr != null)
263               serializeCollection(out, (Collection)o, gType, ctx, name, classAttr, pMeta);
264            else
265               serializeCollection(out, (Collection)o, eType, ctx, name, null, pMeta);
266         }
267         else if (gType.isArray()) {
268            if (classAttr != null)
269               serializeCollection(out, toList(gType.getInnerClass(), o), gType, ctx, name, classAttr, pMeta);
270            else
271               serializeCollection(out, toList(gType.getInnerClass(), o), eType, ctx, name, null, pMeta);
272         }
273         else if (gType.isUri() || (pMeta != null && (pMeta.isUri() || pMeta.isBeanUri()))) {
274            String label = null;
275            String at = ctx.getUriAnchorText();
276            if (at != null) {
277               if (at.equals(LAST_TOKEN)) {
278                  label = o.toString();
279                  if (label.indexOf('/') != -1)
280                     label = label.substring(label.lastIndexOf('/')+1);
281               } else if (at.equals(PROPERTY_NAME)) {
282                  label = (pMeta != null ? pMeta.getName() : null);
283               } else {
284                  label = o.toString();
285               }
286            }
287            if (label == null)
288               label = o.toString();
289            out.oTag(i, "a").attrUri("href", o).append('>');
290            if (at != null && at.equals(URI))
291               out.appendUri(label);
292            else
293               out.append(label);
294            out.eTag("a").nl();
295         }
296         else
297            out.sTag(i, "string").encodeText(o).eTag("string").nl();
298      }
299      ctx.pop();
300      ctx.indent -= indent;
301   }

303   @SuppressWarnings({ "rawtypes", "unchecked" })
304   private void serializeMap(HtmlSerializerWriter out, Map m, ClassMeta<?> type, HtmlSerializerContext ctx, String classAttr, BeanPropertyMeta<?> ppMeta) throws IOException, SerializeException {
305      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
306
307      int i = ctx.getIndent();
308      out.oTag(i, "table").attr("type", "object");
309      if (classAttr != null)
310         out.attr("class", classAttr);
311      out.appendln(">");
312      if (! (ppMeta != null && ppMeta.getHtmlMeta().isNoTableHeaders())) {
313         out.sTag(i+1, "tr").nl();
314         out.sTag(i+2, "th").nl().appendln(i+3, "<string>key</string>").eTag(i+2, "th").nl();
315         out.sTag(i+2, "th").nl().appendln(i+3, "<string>value</string>").eTag(i+2, "th").nl();
316         out.eTag(i+1, "tr").nl();
317      }
318      for (Map.Entry e : (Set<Map.Entry>)m.entrySet()) {
319
320         Object key = ctx.generalize(e.getKey(), keyType);
321         Object value = null;
322         try {
323            value = e.getValue();
324         } catch (StackOverflowError t) {
325            throw t;
326         } catch (Throwable t) {
327            ctx.addWarning("Could not call getValue() on property ''{0}'', {1}", e.getKey(), t.getLocalizedMessage());
328         }
329
330         out.sTag(i+1, "tr").nl();
331         out.sTag(i+2, "td").nl();
332         serializeAnything(out, key, keyType, ctx, null, 2, null);
333         out.eTag(i+2, "td").nl();
334         out.sTag(i+2, "td").nl();
335         serializeAnything(out, value, valueType, ctx, (key == null ? "_x0000_" : key.toString()), 2, null);
336         out.eTag(i+2, "td").nl();
337         out.eTag(i+1, "tr").nl();
338      }
339      out.eTag(i, "table").nl();
340   }

342   @SuppressWarnings({ "rawtypes" })
343   private void serializeBeanMap(HtmlSerializerWriter out, BeanMap m, HtmlSerializerContext ctx, String classAttr, BeanPropertyMeta<?> ppMeta) throws IOException, SerializeException {
344      int i = ctx.getIndent();
345
346      Object o = m.getBean();
347
348      Class<?> c = o.getClass();
349      if (c.isAnnotationPresent(HtmlLink.class)) {
350         HtmlLink h = o.getClass().getAnnotation(HtmlLink.class);
351         Object urlProp = m.get(h.hrefProperty());
352         Object nameProp = m.get(h.nameProperty());
353         out.oTag(i, "a").attrUri("href", urlProp).append('>').encodeText(nameProp).eTag("a").nl();
354         return;
355      }
356
357      out.oTag(i, "table").attr("type", "object");
358      if (classAttr != null)
359         out.attr("_class", classAttr);
360      out.append('>').nl();
361      if (! (m.getClassMeta().getHtmlMeta().isNoTableHeaders() || (ppMeta != null && ppMeta.getHtmlMeta().isNoTableHeaders()))) {
362         out.sTag(i+1, "tr").nl();
363         out.sTag(i+2, "th").nl().appendln(i+3, "<string>key</string>").eTag(i+2, "th").nl();
364         out.sTag(i+2, "th").nl().appendln(i+3, "<string>value</string>").eTag(i+2, "th").nl();
365         out.eTag(i+1, "tr").nl();
366      }
367
368      Iterator mapEntries = m.entrySet().iterator();
369
370      while (mapEntries.hasNext()) {
371         BeanMapEntry p = (BeanMapEntry)mapEntries.next();
372         BeanPropertyMeta pMeta = p.getMeta();
373
374         String key = p.getKey();
375         Object value = null;
376         try {
377            value = p.getValue();
378         } catch (StackOverflowError e) {
379            throw e;
380         } catch (Throwable t) {
381            ctx.addBeanGetterWarning(pMeta, t);
382         }
383
384         if (ctx.canIgnoreValue(pMeta.getClassMeta(), key, value))
385            continue;
386
387         out.sTag(i+1, "tr").nl();
388         out.sTag(i+2, "td").nl();
389         out.sTag(i+3, "string").encodeText(key).eTag("string").nl();
390         out.eTag(i+2, "td").nl();
391         out.sTag(i+2, "td").nl();
392         try {
393            serializeAnything(out, value, p.getMeta().getClassMeta(), ctx, key, 2, pMeta);
394         } catch (SerializeException t) {
395            throw t;
396         } catch (StackOverflowError t) {
397            throw t;
398         } catch (Throwable t) {
399            ctx.addBeanGetterWarning(pMeta, t);
400         }
401         out.eTag(i+2, "td").nl();
402         out.eTag(i+1, "tr").nl();
403      }
404      out.eTag(i, "table").nl();
405   }

407   @SuppressWarnings({ "rawtypes", "unchecked" })
408   private void serializeCollection(HtmlSerializerWriter out, Collection c, ClassMeta<?> type, HtmlSerializerContext ctx, String name, String classAttr, BeanPropertyMeta<?> ppMeta) throws IOException, SerializeException {
409
410      BeanContext bc = ctx.getBeanContext();
411      ClassMeta<?> elementType = type.getElementType();
412
413      int i = ctx.getIndent();
414      if (c.isEmpty()) {
415         out.appendln(i, "<ul></ul>");
416         return;
417      }
418
419      c = ctx.sort(c);
420
421      // Look at the objects to see how we're going to handle them.  Check the first object to see how we're going to handle this.
422      // If it's a map or bean, then we'll create a table.
423      // Otherwise, we'll create a list.
424      String[] th = getTableHeaders(ctx, c, ppMeta);
425
426      if (th != null) {
427
428         out.oTag(i, "table").attr("type", "array");
429         if (classAttr != null)
430            out.attr("_class", classAttr);
431         out.append('>').nl();
432         out.sTag(i+1, "tr").nl();
433         for (String key : th)
434            out.sTag(i+2, "th").append(key).eTag("th").nl();
435         out.eTag(i+1, "tr").nl();
436
437         for (Object o : c) {
438            ClassMeta<?> cm = bc.getClassMetaForObject(o);
439
440            if (cm != null && cm.getPojoFilter() != null) {
441               PojoFilter f = cm.getPojoFilter();
442               o = f.filter(o);
443               cm = cm.getFilteredClassMeta();
444            }
445
446            if (cm != null && ctx.isAddClassAttrs() && elementType.getInnerClass() != o.getClass())
447               out.oTag(i+1, "tr").attr("_class", o.getClass().getName()).append('>').nl();
448            else
449               out.sTag(i+1, "tr").nl();
450
451            if (cm == null) {
452               serializeAnything(out, o, null, ctx, null, 1, null);
453
454            } else if (cm.isMap() && ! (cm.isBeanMap())) {
455               Map m2 = ctx.sort((Map)o);
456
457               Iterator mapEntries = m2.entrySet().iterator();
458               while (mapEntries.hasNext()) {
459                  Map.Entry e = (Map.Entry)mapEntries.next();
460                  out.sTag(i+2, "td").nl();
461                  serializeAnything(out, e.getValue(), elementType, ctx, e.getKey().toString(), 2, null);
462                  out.eTag(i+2, "td").nl();
463               }
464            } else {
465               BeanMap m2 = null;
466               if (o instanceof BeanMap)
467                  m2 = (BeanMap)o;
468               else
469                  m2 = bc.forBean(o);
470
471               Iterator mapEntries = m2.entrySet().iterator();
472               while (mapEntries.hasNext()) {
473                  BeanMapEntry p = (BeanMapEntry)mapEntries.next();
474                  BeanPropertyMeta pMeta = p.getMeta();
475                  out.sTag(i+2, "td").nl();
476                  serializeAnything(out, p.getValue(), pMeta.getClassMeta(), ctx, p.getKey().toString(), 2, pMeta);
477                  out.eTag(i+2, "td").nl();
478               }
479            }
480            out.eTag(i+1, "tr").nl();
481         }
482         out.eTag(i, "table").nl();
483
484      } else {
485         out.sTag(i, "ul").nl();
486         for (Object o : c) {
487            out.sTag(i+1, "li").nl();
488            serializeAnything(out, o, elementType, ctx, name, 1, null);
489            out.eTag(i+1, "li").nl();
490         }
491         out.eTag(i, "ul").nl();
492      }
493   }

495   /*
496    * Returns the table column headers for the specified collection of objects.
497    * Returns null if collection should not be serialized as a 2-dimensional table.
498    * 2-dimensional tables are used for collections of objects that all have the same set of property names.
499    */
500   @SuppressWarnings({ "rawtypes", "unchecked" })
501   private String[] getTableHeaders(SerializerContext ctx, Collection c, BeanPropertyMeta<?> pMeta) throws SerializeException {
502      BeanContext bc = ctx.getBeanContext();
503      if (c.size() == 0)
504         return null;
505      c = ctx.sort(c);
506      String[] th;
507      Set<String> s = new TreeSet<String>();
508      Set<ClassMeta> prevC = new HashSet<ClassMeta>();
509      Object o1 = null;
510      for (Object o : c)
511         if (o != null) {
512            o1 = o;
513            break;
514         }
515      if (o1 == null)
516         return null;
517      ClassMeta cm = bc.getClassMetaForObject(o1);
518      if (cm.getPojoFilter() != null) {
519         PojoFilter f = cm.getPojoFilter();
520         o1 = f.filter(o1);
521         cm = cm.getFilteredClassMeta();
522      }
523      if (cm == null || ! (cm.isMap() || cm.isBean()))
524         return null;
525      if (cm.getInnerClass().isAnnotationPresent(HtmlLink.class))
526         return null;
527      HtmlClassMeta h = cm.getHtmlMeta();
528      if (h.isNoTables() || (pMeta != null && pMeta.getHtmlMeta().isNoTables()))
529         return null;
