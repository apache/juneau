020/**
021 * Java bean wrapper class.
022 *
023 *
024 * <h6 class='topic'>Description</h6>
025 * <p>
026 *    A wrapper that wraps Java bean instances inside of a {@link Map} interface that allows
027 *    properties on the wrapped object can be accessed using the {@link Map#get(Object) get()} and {@link Map#put(Object,Object) put()} methods.
028 * <p>
029 *    Use the {@link BeanContext} class to create instances of this class.
030 *
031 *
032 * <h6 class='topic'>Bean property order</h6>
033 * <p>
034 *    The order of the properties returned by the {@link Map#keySet() keySet()} and {@link Map#entrySet() entrySet()} methods are as follows:
035 *    <ul>
036 *       <li>If {@link Bean @Bean} annotation is specified on class, then the order is the same as the list of properties in the annotation.
037 *       <li>If {@link Bean @Bean} annotation is not specified on the class, then the order is the same as that returned
038 *          by the {@link java.beans.BeanInfo} class (i.e. ordered by definition in the class).
039 *    </ul>
040 *    <br>
041 *    The order can also be overridden through the use of a {@link BeanFilter}.
042 *
043 *
044 * <h6 class='topic'>POJO filters</h6>
045 * <p>
046 *    If {@link PojoFilter PojoFilters} are defined on the class types of the properties of this bean or the bean properties themselves, the
047 *    {@link #get(Object)} and {@link #put(Object, Object)} methods will automatically
048 *    transform the property value to and from the serialized form.
049 *
050 * @author Barry M. Caceres
051 * @author James Bognar (jbognar@us.ibm.com)
052 * @param <T> Specifies the type of object that this map encapsulates.
053 */
054public class BeanMap<T> extends AbstractMap<String,Object> implements Delegate<T> {
055
056   /** The wrapped object. */
057   protected T bean;
058
059   /** Temporary holding cache for beans with read-only properties.  Normally null. */
060   protected Map<String,Object> propertyCache;
061
062   /** Temporary holding cache for bean properties of array types when the add() method is being used. */
063   protected Map<String,List<?>> arrayPropertyCache;
064
065   /** The BeanMeta associated with the class of the object. */
066   protected BeanMeta<T> meta;
067
068   /**
069    * Instance of this class are instantiated through the BeanContext class.
070    *
071    * @param bean The bean to wrap inside this map.
072    * @param meta The metadata associated with the bean class.
073    */
074   protected BeanMap(T bean, BeanMeta<T> meta) {
075      this.bean = bean;
076      this.meta = meta;
077      if (meta.constructorArgs.length > 0)
078         propertyCache = new TreeMap<String,Object>();
079   }
080
081   /**
082    * Returns the metadata associated with this bean map.
083    *
084    * @return The metadata associated with this bean map.
085    */
086   public BeanMeta<T> getMeta() {
087      return meta;
088   }
089
090   /**
091    * Returns the wrapped bean object.
092    * Triggers bean creation if bean has read-only properties set through a constructor
093    *    defined by the {@link BeanConstructor} annotation.
094    *
095    * @return The inner bean object.
096    */
097   public T getBean() {
098      T b = getBean(true);
099
100      // If we have any arrays that need to be constructed, do it now.
101      if (arrayPropertyCache != null) {
102         for (Map.Entry<String,List<?>> e : arrayPropertyCache.entrySet()) {
103            String key = e.getKey();
104            List<?> value = e.getValue();
105            BeanPropertyMeta<T> bpm = getPropertyMeta(key);
106            try {
107               bpm.setArray(b, value);
108            } catch (Exception e1) {
109               throw new RuntimeException(e1);
110            }
111         }
112         arrayPropertyCache = null;
113      }
114      return b;
115   }
116
117   /**
118    * Returns the wrapped bean object.
119    * <p>
120    * If <code>create</code> is <jk>false</jk>, then this method may return <jk>null</jk>
121    *    if the bean has read-only properties set through a constructor
122    *    defined by the {@link BeanConstructor} annotation.
123    * <p>
124    * This method does NOT always return the bean in it's final state.
125    *    Array properties temporary stored as ArrayLists are not finalized
126    *    until the {@link #getBean()} method is called.
127    *
128    * @param create If bean hasn't been instantiated yet, then instantiate it.
129    * @return The inner bean object.
130    */
131   public T getBean(boolean create) {
132      /** If this is a read-only bean, then we need to create it. */
133      if (bean == null && create && meta.constructorArgs.length > 0) {
134         String[] props = meta.constructorArgs;
135         Constructor<T> c = meta.constructor;
136         Object[] args = new Object[props.length];
137         for (int i = 0; i < props.length; i++)
138            args[i] = propertyCache.remove(props[i]);
139         try {
140            bean = c.newInstance(args);
141            for (Map.Entry<String,Object> e : propertyCache.entrySet())
142               put(e.getKey(), e.getValue());
143            propertyCache = null;
144         } catch (Exception e) {
145            throw new BeanRuntimeException(e);
146         }
147      }
148      return bean;
149   }
150
151   /**
152    * Returns the value of the property identified as the URI property (annotated with {@link BeanProperty#beanUri()} as <jk>true</jk>).
153    *
154    * @return The URI value, or <jk>null</jk> if no URI property exists on this bean.
155    */
156   public Object getBeanUri() {
157      BeanMeta<T> bm = getMeta();
158      return bm.hasBeanUriProperty() ? bm.getBeanUriProperty().get(this) : null;
159   }
160
161   /**
162    * Sets the bean URI property if the bean has a URI property.
163    * Ignored otherwise.
164    *
165    * @param o The bean URI object.
166    * @return If the bean context setting {@code beanMapPutReturnsOldValue} is <jk>true</jk>, then the old value of the property is returned.
167    *       Otherwise, this method always returns <jk>null</jk>.
168    */
169   public Object putBeanUri(Object o) {
170      BeanMeta<T> bm = getMeta();
171      return bm.hasBeanUriProperty() ? bm.getBeanUriProperty().set(this, o) : null;
172   }

174   /**
175    * Sets a property on the bean.
176    * <p>
177    * If there is a {@link PojoFilter} associated with this bean property or bean property type class, then 
178    *    you must pass in a filtered value.
179    * For example, if the bean property type class is a {@link Date} and the bean property has the 
180    *    {@link com.ibm.juno.core.filters.DateFilter.ISO8601DT} filter associated with it through the 
181    *    {@link BeanProperty#filter() @BeanProperty.filter()} annotation, the value being passed in must be 
182    *    a String containing an ISO8601 date-time string value. 
183    *
184    * <dl>
185    *    <dt>Example:</dt>
186    *    <dd>
187    * <p class='bcode'>
188    *    <jc>// Construct a bean with a 'birthDate' Date field</jc>
189    *    Person p = <jk>new</jk> Person();
190    *
191    *    <jc>// Create a bean context and add the ISO8601 date-time filter</jc>
192    *    BeanContext beanContext = <jk>new</jk> BeanContext().addFilter(DateFilter.ISO8601DT.<jk>class</jk>);
193    *
194    *    <jc>// Wrap our bean in a bean map</jc>
195    *    BeanMap&lt;Person&gt; b = beanContext.forBean(p);
196    *
197    *    <jc>// Set the field</jc>
198    *    myBeanMap.put(<js>"birthDate"</js>, <js>"'1901-03-03T04:05:06-5000'"</js>);
199    * </p>
200    *    </dd>
201    * </dl>
202    *
203    * @param property The name of the property to set.
204    * @param value The value to set the property to.
205    * @return If the bean context setting {@code beanMapPutReturnsOldValue} is <jk>true</jk>, then the old value of the property is returned.
206    *       Otherwise, this method always returns <jk>null</jk>.
207    * @throws RuntimeException if any of the following occur.
208    *    <ul>
209    *       <li>BeanMapEntry does not exist on the underlying object.
210    *       <li>Security settings prevent access to the underlying object setter method.
211    *       <li>An exception occurred inside the setter method.
212    *    </ul>
213    */
214   @Override /* Map */
215   public Object put(String property, Object value) {
216      BeanPropertyMeta<T> p = meta.properties.get(property);
217      if (p == null) {
218         if (meta.ctx.ignoreUnknownBeanProperties)
219            return null;
220         if (property.equals("<uri>") && meta.uriProperty != null)
221            return meta.uriProperty.set(this, value);
222
223         // If this bean has subtypes, and we haven't set the subtype yet,
224         // store the property in a temporary cache until the bean can be instantiated.
225         // This eliminates the need for requiring that the sub type attribute be provided first.
226         if (meta.subTypeIdProperty != null) {
227            if (propertyCache == null)
228               propertyCache = new TreeMap<String,Object>();
229            return propertyCache.put(property, value);
230         }
231
232         throw new BeanRuntimeException(meta.c, "Bean property ''{0}'' not found.", property);
233      }
234      if (meta.filter != null)
235         if (meta.filter.writeProperty(this.bean, property, value))
236            return null;
237      return p.set(this, value);
238   }

240   /**
241    * Add a value to a collection or array property.
242    * <p>
243    *    As a general rule, adding to arrays is not recommended since the array must be recreate each time
244    *    this method is called.
245    *
246    * @param property Property name or child-element name (if {@link Xml#childName()} is specified).
247    * @param value The value to add to the collection or array.
248    */
249   public void add(String property, Object value) {
250      BeanPropertyMeta<T> p = meta.properties.get(property);
251      if (p == null) {
252         if (meta.ctx.ignoreUnknownBeanProperties)
253            return;
254         throw new BeanRuntimeException(meta.c, "Bean property ''{0}'' not found.", property);
255      }
256      p.add(this, value);
257   }

260   /**
261    * Gets a property on the bean.
262    * <p>
263    * If there is a {@link PojoFilter} associated with this bean property or bean property type class, then 
264    *    this method will return the filtered value.
265    * For example, if the bean property type class is a {@link Date} and the bean property has the 
266    *    {@link com.ibm.juno.core.filters.DateFilter.ISO8601DT} filter associated with it through the 
267    *    {@link BeanProperty#filter() @BeanProperty.filter()} annotation, this method will return a String 
268    *    containing an ISO8601 date-time string value. 
269    *
270    * <dl>
271    *    <dt>Example:</dt>
272    *    <dd>
273    * <p class='bcode'>
274    *    <jc>// Construct a bean with a 'birthDate' Date field</jc>
275    *    Person p = <jk>new</jk> Person();
276    *    p.setBirthDate(<jk>new</jk> Date(1, 2, 3, 4, 5, 6));
277    *
278    *    <jc>// Create a bean context and add the ISO8601 date-time filter</jc>
279    *    BeanContext beanContext = <jk>new</jk> BeanContext().addFilter(DateFilter.ISO8601DT.<jk>class</jk>);
280    *
281    *    <jc>// Wrap our bean in a bean map</jc>
282    *    BeanMap&lt;Person&gt; b = beanContext.forBean(p);
283    *
284    *    <jc>// Get the field as a string (i.e. "'1901-03-03T04:05:06-5000'")</jc>
285    *    String s = myBeanMap.get(<js>"birthDate"</js>);
286    * </p>
287    *    </dd>
288    * </dl>
289    *
290    * @param property The name of the property to get.
291    * @throws RuntimeException if any of the following occur.
292    *    <ol>
293    *       <li>BeanMapEntry does not exist on the underlying object.
294    *       <li>Security settings prevent access to the underlying object getter method.
295    *       <li>An exception occurred inside the getter method.
296    *    </ol>
297    */
298   @Override /* Map */
299   public Object get(Object property) {
300      BeanPropertyMeta<T> p = meta.properties.get(property);
301      if (p == null)
302         return null;
303      if (meta.filter != null && property != null)
304         return meta.filter.readProperty(this.bean, property.toString(), p.get(this));
305      return p.get(this);
306   }

308   /**
309    * Convenience method for setting multiple property values by passing in JSON (or other) text.
310    * <p>
311    * Typically the input is going to be JSON, although the actual data type
312    *    depends on the default parser specified by the {@link BeanContextFactory#setDefaultParser(ReaderParser)} method
313    *    on the bean context that created this map.
314    *
315    * <dl>
316    *    <dt>Example:</dt>
317    *    <dd>
318    * <p class='bcode'>
319    *    aPersonBean.load(<js>"{name:'John Smith',age:21}"</js>)
320    * </p>
321    *    </dd>
322    * </dl>
323    *
324    * @param input The text that will get parsed into a map and then added to this map.
325    * @return This object (for method chaining).
326    * @throws ParseException If the input contains a syntax error or is malformed.
327    */
328   public BeanMap<T> load(String input) throws ParseException {
329      putAll(new ObjectMap(input, this.meta.ctx.defaultParser));
330      return this;
331   }
332
333   /**
334    * Convenience method for setting multiple property values by passing in a reader.
335    *
336    * @param r The text that will get parsed into a map and then added to this map.
337    * @param p The parser to use to parse the text.
338    * @return This object (for method chaining).
339    * @throws ParseException If the input contains a syntax error or is malformed.
340    * @throws IOException Thrown by <code>Reader</code>.
341    */
342   public BeanMap<T> load(Reader r, ReaderParser p) throws ParseException, IOException {
343      putAll(new ObjectMap(r, p));
344      return this;
345   }

347   /**
348    * Convenience method for loading this map with the contents of the specified map.
349    * <p>
350    * Identical to {@link #putAll(Map)} except as a fluent-style method.
351    *
352    * @param entries The map containing the entries to add to this map.
353    * @return This object (for method chaining).
354    */
355   @SuppressWarnings({"unchecked","rawtypes"})
356   public BeanMap<T> load(Map entries) {
357      putAll(entries);
358      return this;
359   }

371   /**
372    * Returns the specified property on this bean map.
373    * <p>
374    *    Allows you to get and set an individual property on a bean without having a
375    *    handle to the bean itself by using the {@link BeanMapEntry#getValue()}
376    *    and {@link BeanMapEntry#setValue(Object)} methods.
377    * <p>
378    *    This method can also be used to get metadata on a property by
379    *    calling the {@link BeanMapEntry#getMeta()} method.
380    *
381    * @param propertyName The name of the property to look up.
382    * @return The bean property, or null if the bean has no such property.
383    */
384   public BeanMapEntry<T> getProperty(String propertyName) {
385      BeanPropertyMeta<T> p = meta.properties.get(propertyName);
386      if (p == null)
387         return null;
388      return new BeanMapEntry<T>(this, p);
389   }

411   /**
412    * Returns all the properties associated with the bean.
413    */
414   @Override /* Map */
415   public Set<Entry<String,Object>> entrySet() {
416
417      // Construct our own anonymous set to implement this function.
418      Set<Entry<String,Object>> s = new AbstractSet<Entry<String,Object>>() {
419
420         // Get the list of properties from the meta object.
421         // Note that the HashMap.values() method caches results, so this collection
422         // will really only be constructed once per bean type since the underlying
423         // map never changes.
424         final Collection<BeanPropertyMeta<T>> pSet = meta.properties.values();
425
426         @Override /* Set */
427         public Iterator<java.util.Map.Entry<String, Object>> iterator() {
428
429            // Construct our own anonymous iterator that uses iterators against the meta.properties
430            // map to maintain position.  This prevents us from having to construct any of our own
431            // collection objects.
432            return new Iterator<Entry<String,Object>>() {
433
434               final Iterator<BeanPropertyMeta<T>> pIterator = pSet.iterator();
435
436               @Override /* Iterator */
437               public boolean hasNext() {
438                  return pIterator.hasNext();
439               }
440
441               @Override /* Iterator */
442               public Map.Entry<String, Object> next() {
443                  return new BeanMapEntry<T>(BeanMap.this, pIterator.next());
444               }
445
446               @Override /* Iterator */
447               public void remove() {
448                  throw new UnsupportedOperationException("Cannot remove item from iterator.");
449               }
450            };
451         }
452
453         @Override /* Set */
454         public int size() {
455            return pSet.size();
456         }
457      };
458
459      return s;
460   }
