009package com.ibm.juno.core.utils;
010
011import java.io.*;
012import java.util.*;
013
014
015/**
016 * Writer that can send output to multiple writers.
017 *
018 * @author James Bognar (jbognar@us.ibm.com)
019 */
020public class TeeWriter extends Writer {
021   private Writer[] writers = new Writer[0];
022   private Map<String,Writer> writerMap;
023
024   /**
025    * Constructor.
026    *
027    * @param writers The list of writers.
028    */
029   public TeeWriter(Writer...writers) {
030      this.writers = writers;
031   }
032
033   /**
034    * Constructor.
035    *
036    * @param writers The list of writers.
037    */
038   public TeeWriter(Collection<Writer> writers) {
039      this.writers = writers.toArray(new Writer[writers.size()]);
040   }
041
042   /**
043    * Adds a writer to this tee writer.
044    *
045    * @param w The writer to add to this tee writer.
046    * @param close If <jk>false</jk>, then calling {@link #close()} on this tee writer
047    *    will not filter to the specified writer.
048    * @return This object (for method chaining).
049    */
050   public TeeWriter add(Writer w, boolean close) {
051      if (w == null)
052         return this;
053      if (! close)
054         w = new NoCloseWriter(w);
055      if (w == this)
056         throw new RuntimeException("Cannot add this writer to itself.");
057      for (Writer w2 : writers)
058         if (w2 == w)
059            throw new RuntimeException("Cannot add this writer again.");
060      if (w instanceof TeeWriter) {
061         for (Writer w2 : ((TeeWriter)w).writers)
062            add(w2, true);
063      } else {
064         writers = ArrayUtils.append(writers, w);
065      }
066      return this;
067   }
068
069   /**
070    * Same as {@link #add(Writer, boolean)} but associates the writer with an identifier
071    * so the writer can be retrieved through {@link #getWriter(String)}.
072    *
073    * @param id The ID to associate the writer with.
074    * @param w The writer to add.
075    * @param close Close the specified writer afterwards.
076    * @return This object (for method chaining).
077    */
078   public TeeWriter add(String id, Writer w, boolean close) {
079      if (id != null) {
080         if (writerMap == null)
081            writerMap = new TreeMap<String,Writer>();
082         writerMap.put(id, w);
083      }
084      return add(w, close);
085   }
086
087   /**
088    * Returns the number of inner writers in this tee writer.
089    *
090    * @return The number of writers.
091    */
092   public int size() {
093      return writers.length;
094   }
095
096   /**
097    * Returns the writer identified through the <code>id</code> parameter
098    * passed in through the {@link #add(String, Writer, boolean)} method.
099    *
100    * @param id The ID associated with the writer.
101    * @return The writer, or <jk>null</jk> if no identifier was specified when the writer was added.
102    */
103   public Writer getWriter(String id) {
104      if (writerMap != null)
105         return writerMap.get(id);
106      return null;
107   }
108
109   @Override /* Writer */
110   public void write(char[] cbuf, int off, int len) throws IOException {
111      for (Writer w : writers)
112         if (w != null)
113         w.write(cbuf, off, len);
114   }
115
116   @Override /* Writer */
117   public void flush() throws IOException {
118      for (Writer w : writers)
119         if (w != null)
120         w.flush();
121   }
122
123   @Override /* Writer */
124   public void close() throws IOException {
125      IOException e = null;
126      for (Writer w : writers) {
127         if (w != null) {
128            try {
129         w.close();
130            } catch (IOException e2) {
131               e = e2;
132            }
133         }
134      }
135      if (e != null)
136         throw e;
137   }
138
139   private static class NoCloseWriter extends Writer {
140      private Writer writer;
141
142      private NoCloseWriter(Writer writer) {
143         this.writer = writer;
144      }
145
146      @Override /* Writer */
147      public void write(char[] cbuf, int off, int len) throws IOException {
148         writer.write(cbuf, off, len);
149      }
150
151      @Override /* Writer */
152      public void flush() throws IOException {
153         writer.flush();
154      }
155
156      @Override /* Writer */
157      public void close() throws IOException {
158         // Do nothing.
159      }
160   }
161}
