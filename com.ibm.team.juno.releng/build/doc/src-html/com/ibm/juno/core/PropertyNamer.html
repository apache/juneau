009package com.ibm.juno.core;
010
011import com.ibm.juno.core.annotation.*;
012
013/**
014 * Defines an API for converting conventional bean property names to some other form.
015 * <p>
016 * For example, given the bean property <js>"fooBarURL"</js>, the {@link PropertyNamerDashedLC}
017 *    property namer will convert this to <js>"foo-bar-url"</js>.
018 * <p>
019 * Property namers are associated with beans through the {@link Bean#propertyNamer} annotation.
020 *
021 * @author James Bognar (jbognar@us.ibm.com)
022 */
023public interface PropertyNamer {
024
025   /**
026    * Convert the specified default property name to some other value.
027    * @param name The original bean property name.
028    * @return The converted property name.
029    */
030   public String getPropertyName(String name);
031}
