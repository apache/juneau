026/**
027 * Serializes POJOs to RDF.
028 *
029 *
030 * <h6 class='topic'>Configurable properties</h6>
031 * <p>
032 *    Refer to <a class='doclink' href='package-summary.html#SerializerConfigurableProperties'>Configurable Properties</a>
033 *       for the entire list of configurable properties.
034 *
035 *
036 * <h6 class='topic'>Behavior-specific subclasses</h6>
037 * <p>
038 *    The following direct subclasses are provided for language-specific serializers:
039 * <ul>
040 *    <li>{@link RdfSerializer.Xml} - RDF/XML.
041 *    <li>{@link RdfSerializer.XmlAbbrev} - RDF/XML-ABBREV.
042 *    <li>{@link RdfSerializer.NTriple} - N-TRIPLE.
043 *    <li>{@link RdfSerializer.Turtle} - TURTLE.
044 *    <li>{@link RdfSerializer.N3} - N3.
045 * </ul>
046 *
047 *
048 * <h6 class='topic'>Additional Information</h6>
049 * <p>
050 *    See <a class='doclink' href='package-summary.html#TOC'>RDF Overview</a> for an overview of RDF support in Juno.
051 *
052 *
053 * @author James Bognar (jbognar@us.ibm.com)
054 */

128   @Override /* Serializer */
129   protected void doSerialize(Object o, Writer out, SerializerContext ctx) throws IOException, SerializeException {
130
131      RdfSerializerContext rctx = (RdfSerializerContext)ctx;
132
133      Model model = rctx.getModel();
134      Resource r = null;
135
136      ClassMeta<?> cm = rctx.getBeanContext().getClassMetaForObject(o);
137      if (rctx.isLooseCollection() && cm != null && (cm.isCollection() || cm.isArray())) {
138         Collection c = ctx.sort(cm.isCollection() ? (Collection)o : toList(cm.getInnerClass(), o));
139         for (Object o2 : c)
140            serializeAnything(o2, false, object(), rctx, "root", null, null);
141      } else {
142         RDFNode n = serializeAnything(o, false, object(), rctx, "root", null, null);
143         if (n.isLiteral()) {
144            r = model.createResource();
145            r.addProperty(rctx.getValueProperty(), n);
146         } else {
147            r = n.asResource();
148         }
149
150         if (rctx.isAddRootProperty())
151            r.addProperty(rctx.getRootProperty(), "true");
152      }
153
154      rctx.getWriter().write(model, out, "http://unknown/");
155   }

157   private RDFNode serializeAnything(Object o, boolean isURI, ClassMeta<?> eType, RdfSerializerContext ctx, String attrName, BeanPropertyMeta bpm, Resource parentResource) throws SerializeException {
158      Model m = ctx.getModel();
159      BeanContext bc = ctx.getBeanContext();
160
161      ClassMeta<?> aType = null;       // The actual type
162      ClassMeta<?> wType = null;       // The wrapped type
163      ClassMeta<?> gType = object();   // The generic type
164
165      aType = ctx.push(attrName, o, eType);
166
167      if (eType == null)
168         eType = object();
169
170      // Handle recursion
171      if (aType == null) {
172         o = null;
173         aType = object();
174      }
175
176      if (o != null) {
177
178         if (aType.isDelegate()) {
179            wType = aType;
180            aType = ((Delegate)o).getClassMeta();
181         }
182
183         gType = aType.getFilteredClassMeta();
184
185         // Filter if necessary
186         PojoFilter filter = aType.getPojoFilter();
187         if (filter != null) {
188            o = filter.filter(o);
189
190            // If the filter's getFilteredClass() method returns Object, we need to figure out
191            // the actual type now.
192            if (gType.isObject())
193               gType = bc.getClassMetaForObject(o);
194         }
195      } else {
196         gType = eType.getFilteredClassMeta();
197      }
198
199      RDFNode n = null;
200
201      if (o == null || gType.isChar() && ((Character)o).charValue() == 0) {
202         if (bpm != null) {
203            if (! ctx.isTrimNulls()) {
204               n = m.createResource(RDF_NIL);
205            }
206         } else {
207            n = m.createResource(RDF_NIL);
208         }
209
210      } else if (gType.isUri() || isURI) {
211         n = m.createResource(getUri(o, null, ctx));
212
213      } else if (gType.isCharSequence() || gType.isChar()) {
214         n = m.createLiteral(ctx.encodeTextInvalidChars(o));
215
216      } else if (gType.isNumber() || gType.isBoolean()) {
217         if (! ctx.isAddLiteralTypes())
218            n = m.createLiteral(o.toString());
219         else
220            n = m.createTypedLiteral(o);
221
222      } else if (gType.isMap() || (wType != null && wType.isMap())) {
223         if (o instanceof BeanMap) {
224            BeanMap bm = (BeanMap)o;
225            String uri = getUri(bm.getBeanUri(), null, ctx);
226            n = m.createResource(uri);
227            serializeBeanMap(bm, (Resource)n, ctx);
228         } else {
229            Map m2 = (Map)o;
230            n = m.createResource();
231            serializeMap(m2, (Resource)n, gType, ctx);
232         }
233
234      } else if (gType.hasToObjectMapMethod()) {
235         Map m2 = gType.toObjectMap(o);
236         n = m.createResource();
237         serializeMap(m2, (Resource)n, gType, ctx);
238
239      } else if (gType.isBean()) {
240         BeanMap bm = bc.forBean(o);
241         String uri = getUri(bm.getBeanUri(), null, ctx);
242         n = m.createResource(uri);
243         serializeBeanMap(bm, (Resource)n, ctx);
244
245      } else if (gType.isCollection() || gType.isArray() || (wType != null && wType.isCollection())) {
246         Collection c = ctx.sort(gType.isCollection() ? (Collection)o : toList(gType.getInnerClass(), o));
247         RdfCollectionFormat f = ctx.getCollectionFormat();
248         if (gType.getRdfMeta().getCollectionFormat() != RdfCollectionFormat.DEFAULT)
249            f = gType.getRdfMeta().getCollectionFormat();
250         if (bpm != null && bpm.getRdfMeta().getCollectionFormat() != RdfCollectionFormat.DEFAULT)
251            f = bpm.getRdfMeta().getCollectionFormat();
252         switch (f) {
253            case BAG: n = serializeToContainer(c, gType, m.createBag(), ctx); break;
254            case LIST: n = serializeToList(c, gType, ctx); break;
255            case MULTI_VALUED: serializeToMultiProperties(c, gType, bpm, ctx, attrName, parentResource); break;
256            default: n = serializeToContainer(c, gType, m.createSeq(), ctx);
257         }
258      } else {
259         n = m.createLiteral(ctx.encodeTextInvalidChars(o));
260      }
261
262      if (ctx.isAddClassAttrs() && n != null && n.isResource()) {
263         if (o != null && ! eType.equals(aType))
264            n.asResource().addProperty(ctx.getClassProperty(), aType.toString());
265         else if (o == null)
266            n.asResource().addProperty(ctx.getClassProperty(), eType.toString());
267      }
268
269      ctx.pop();
270
271      return n;
272   }

274   private String getUri(Object uri, Object uri2, RdfSerializerContext ctx) {
275      String s = null;
276      if (uri != null)
277         s = uri.toString();
278      if ((s == null || s.isEmpty()) && uri2 != null)
279         s = uri2.toString();
280      if (s == null)
281         return null;
282      if (s.indexOf("://") == -1) {
283         String aUri = ctx.getAbsolutePathUriBase();
284         String rUri = ctx.getRelativeUriBase();
285         if (StringUtils.startsWith(s, '/')) {
286            if (aUri != null)
287               return aUri + s;
288         } else {
289            if (rUri != null) {
290               if (rUri.equals("/"))
291                  return '/' + s;
292               return rUri + '/' + s;
293            }
294         }
295      }
296      return s;
297   }

299   private void serializeMap(Map m, Resource r, ClassMeta<?> type, RdfSerializerContext ctx) throws SerializeException {
300
301      m = ctx.sort(m);
302
303      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
304
305      ArrayList<Map.Entry<Object,Object>> l = new ArrayList<Map.Entry<Object,Object>>(m.entrySet());
306      Collections.reverse(l);
307      for (Map.Entry<Object,Object> me : l) {
308         Object value = me.getValue();
309
310         Object key = ctx.generalize(me.getKey(), keyType);
311
312         Namespace ns = ctx.getJunoBpNs();
313         Model model = ctx.getModel();
314         Property p = model.createProperty(ns.getUri(), ctx.encodeElementName(key));
315         RDFNode n = serializeAnything(value, false, valueType, ctx, key == null ? null : key.toString(), null, r);
316         if (n != null)
317            r.addProperty(p, n);
318      }
319   }

321   private void serializeBeanMap(BeanMap m, Resource r, RdfSerializerContext ctx) throws SerializeException {
322      ArrayList<BeanMapEntry> l = new ArrayList<BeanMapEntry>(m.entrySet());
323      Collections.reverse(l);
324      for (BeanMapEntry bme : l) {
325         BeanPropertyMeta pMeta = bme.getMeta();
326         ClassMeta<?> cm = pMeta.getClassMeta();
327
328         if (pMeta.isBeanUri())
329            continue;
330
331         String key = bme.getKey();
332         Object value = null;
333         try {
334            value = bme.getValue();
335         } catch (StackOverflowError e) {
336            throw e;
337         } catch (Throwable t) {
338            ctx.addBeanGetterWarning(pMeta, t);
339         }
340
341         if (ctx.canIgnoreValue(cm, key, value))
342            continue;
343
344         BeanPropertyMeta bpm = bme.getMeta();
345         Namespace ns = bpm.getRdfMeta().getNamespace();
346         if (ns == null && ctx.isUseXmlNamespaces())
347            ns = bpm.getXmlMeta().getNamespace();
348         if (ns == null)
349            ns = ctx.getJunoBpNs();
350         else if (ctx.isAutoDetectNamespaces())
351            ctx.addModelPrefix(ns);
352
353         Property p = ctx.getModel().createProperty(ns.getUri(), ctx.encodeElementName(key));
354         RDFNode n = serializeAnything(value, pMeta.isUri(), cm, ctx, key, pMeta, r);
355         if (n != null)
356            r.addProperty(p, n);
357      }
358   }

361   private Container serializeToContainer(Collection c, ClassMeta<?> type, Container list, RdfSerializerContext ctx) throws SerializeException {
362
363      ClassMeta<?> elementType = type.getElementType();
364      for (Object e : c) {
365         RDFNode n = serializeAnything(e, false, elementType, ctx, null, null, null);
366         list = list.add(n);
367      }
368      return list;
369   }
370
371   private RDFList serializeToList(Collection c, ClassMeta<?> type, RdfSerializerContext ctx) throws SerializeException {
372      ClassMeta<?> elementType = type.getElementType();
373      List<RDFNode> l = new ArrayList<RDFNode>(c.size());
374      for (Object e : c) {
375         l.add(serializeAnything(e, false, elementType, ctx, null, null, null));
376      }
377      return ctx.getModel().createList(l.iterator());
378   }
379
380   private void serializeToMultiProperties(Collection c, ClassMeta<?> gType, BeanPropertyMeta bpm, RdfSerializerContext ctx, String attrName, Resource parentResource) throws SerializeException {
381      ClassMeta<?> elementType = gType.getElementType();
382      for (Object e : c) {
383         Namespace ns = null;
384         if (bpm != null) {
385            ns = bpm.getRdfMeta().getNamespace();
386            if (ns == null && ctx.isUseXmlNamespaces())
387               ns = bpm.getXmlMeta().getNamespace();
388         }
389         if (ns == null)
390            ns = ctx.getJunoBpNs();
391         else if (ctx.isAutoDetectNamespaces())
392            ctx.addModelPrefix(ns);
393         RDFNode n2 = serializeAnything(e, false, elementType, ctx, null, null, null);
394         Property p = ctx.getModel().createProperty(ns.getUri(), ctx.encodeElementName(attrName));
395         parentResource.addProperty(p, n2);
396      }

409   @Override /* CoreApi */
410   public RdfSerializer setProperty(String property, Object value) throws LockedException {
411      checkLock();
412      if (! rsp.setProperty(property, value))
413         if (! xsp.setProperty(property, value))
414            super.setProperty(property, value);
415      return this;
416   }
