009package com.ibm.juno.core.ini;
010import static com.ibm.juno.core.utils.ThrowableUtils.*;
011
012import java.io.*;
013import java.util.*;
014
015import com.ibm.juno.core.*;
016import com.ibm.juno.core.parser.*;

020/**
021 * Wraps an instance of {@link ConfigFileImpl} in an interface that will
022 *    automatically replace {@link StringVarResolver} variables.
023 * <p>
024 * The {@link ConfigFile#getResolving(StringVarResolver)} returns an instance of this class.
025 * <p>
026 * This class overrides the {@link #getString(String, String)} to resolve string variables.
027 * All other method calls are passed through to the inner config file.
028 *
029 * @author James Bognar (jbognar@us.ibm.com)
030 */
031public final class ConfigFileWrapped extends ConfigFile {
032
033   private final ConfigFileImpl cf;
034   private final StringVarResolver vr;
035
036   ConfigFileWrapped(ConfigFileImpl cf, StringVarResolver vr) {
037      this.cf = cf;
038      this.vr = vr;
039   }
040
041   @Override /* ConfigFile */
042   public void clear() {
043      cf.clear();
044   }
045
046   @Override /* ConfigFile */
047   public boolean containsKey(Object key) {
048      return cf.containsKey(key);
049   }
050
051   @Override /* ConfigFile */
052   public boolean containsValue(Object value) {
053      return cf.containsValue(value);
054   }
055
056   @Override /* ConfigFile */
057   public Set<java.util.Map.Entry<String,Section>> entrySet() {
058      return cf.entrySet();
059   }
060
061   @Override /* ConfigFile */
062   public Section get(Object key) {
063      return cf.get(key);
064   }
065
066   @Override /* ConfigFile */
067   public boolean isEmpty() {
068      return cf.isEmpty();
069   }
070
071   @Override /* ConfigFile */
072   public Set<String> keySet() {
073      return cf.keySet();
074   }
075
076   @Override /* ConfigFile */
077   public Section put(String key, Section value) {
078      return cf.put(key, value);
079   }
080
081   @Override /* ConfigFile */
082   public void putAll(Map<? extends String,? extends Section> map) {
083      cf.putAll(map);
084   }
085
086   @Override /* ConfigFile */
087   public Section remove(Object key) {
088      return cf.remove(key);
089   }
090
091   @Override /* ConfigFile */
092   public int size() {
093      return cf.size();
094   }
095
096   @Override /* ConfigFile */
097   public Collection<Section> values() {
098      return cf.values();
099   }
100
101   @Override /* ConfigFile */
102   public ConfigFile loadIfModified() throws IOException {
103      cf.loadIfModified();
104      return this;
105   }
106
107   @Override /* ConfigFile */
108   public ConfigFile load() throws IOException {
109      cf.load();
110      return this;
111   }
112
113   @Override /* ConfigFile */
114   public ConfigFile load(Reader r) throws IOException {
115      cf.load(r);
116      return this;
117   }
118
119
120   @Override /* ConfigFile */
121   public boolean isEncoded(String key) {
122      return cf.isEncoded(key);
123   }
124
125   @Override /* ConfigFile */
126   public ConfigFile addLines(String section, String... lines) {
127      cf.addLines(section, lines);
128      return this;
129   }
130
131   @Override /* ConfigFile */
132   public ConfigFile addHeaderComments(String section, String... headerComments) {
133      cf.addHeaderComments(section, headerComments);
134      return this;
135   }
136
137   @Override /* ConfigFile */
138   public ConfigFile clearHeaderComments(String section) {
139      cf.clearHeaderComments(section);
140      return this;
141   }
142
143   @Override /* ConfigFile */
144   public Section getSection(String name) {
145      return cf.getSection(name);
146   }
147
148   @Override /* ConfigFile */
149   public Section getSection(String name, boolean create) {
150      return cf.getSection(name, create);
151   }
152
153   @Override /* ConfigFile */
154   public ConfigFile addSection(String name) {
155      cf.addSection(name);
156      return this;
157   }
158
159   @Override /* ConfigFile */
160   public ConfigFile setSection(String name, Map<String,String> contents) {
161      cf.setSection(name, contents);
162      return this;
163   }
164
165   @Override /* ConfigFile */
166   public ConfigFile removeSection(String name) {
167      cf.removeSection(name);
168      return this;
169   }
170
171   @Override /* ConfigFile */
172   public ConfigFile save() throws IOException {
173      cf.save();
174      return this;
175   }
176
177   @Override /* ConfigFile */
178   public ConfigFile serializeTo(Writer out, ConfigFileFormat format) throws IOException {
179      cf.serializeTo(out, format);
180      return this;
181   }
182
183   @Override /* ConfigFile */
184   public String toString() {
185      return cf.toString();
186   }
187
188   @Override /* ConfigFile */
189   @SuppressWarnings("hiding")
190   public ConfigFile getResolving(StringVarResolver vr) {
191      assertFieldNotNull(vr, "vr");
192      return new ConfigFileWrapped(cf, vr);
193   }
194
195   @Override /* ConfigFile */
196   @SuppressWarnings("hiding")
197   public ConfigFile getResolving(StringVarResolver vr, boolean augment) {
198      assertFieldNotNull(vr, "vr");
199      if (augment)
200         vr.setParent(this.vr, false);
201      return new ConfigFileWrapped(cf, vr);
202   }
203
204   @Override /* ConfigFile */
205   public ConfigFile getResolving() {
206      return new ConfigFileWrapped(cf, StringVarResolver.DEFAULT);
207   }
208
209   @Override /* ConfigFile */
210   public ConfigFile addListener(ConfigFileListener listener) {
211      cf.addListener(listener);
212      return this;
213   }
214
215   @Override /* ConfigFile */
216   public Writable toWritable() {
217      return cf.toWritable();
218   }
219
220   @Override /* ConfigFile */
221   public ConfigFile merge(ConfigFile newCf) {
222      cf.merge(newCf);
223      return this;
224   }
225
226   @Override /* ConfigFile */
227   protected WriterSerializer getSerializer() throws SerializeException {
228      return cf.getSerializer();
229   }
230
231   @Override /* ConfigFile */
232   protected ReaderParser getParser() throws ParseException {
233      return cf.getParser();
234   }
235
236   @Override /* ConfigFile */
237   public String get(String sectionName, String sectionKey) {
238      String s = cf.get(sectionName, sectionKey);
239      if (s == null)
240         return null;
241      return vr.resolve(s);
242   }
243
244   @Override /* ConfigFile */
245   public String put(String sectionName, String sectionKey, Object value, boolean encoded) {
246      return cf.put(sectionName, sectionKey, value, encoded);
247   }
248
249   @Override /* ConfigFile */
250   public String remove(String sectionName, String sectionKey) {
251      return cf.remove(sectionName, sectionKey);
252   }
253
254   @Override /* ConfigFile */
255   public Set<String> getSectionKeys(String sectionName) {
256      return cf.getSectionKeys(sectionName);
257   }
258
259   @Override /* ConfigFile */
260   protected void readLock() {
261      cf.readLock();
262   }
263
264   @Override /* ConfigFile */
265   protected void readUnlock() {
266      cf.readUnlock();
267   }
268
269   @Override /* ConfigFile */
270   protected StringVarResolver getVarResolver() {
271      return vr;
272   }
273}
