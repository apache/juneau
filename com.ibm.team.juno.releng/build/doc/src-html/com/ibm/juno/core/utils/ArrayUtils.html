023   /**
024    * Appends one or more elements to an array.
025    *
026    * @param <T> The element type.
027    * @param array The array to append to.
028    * @param newElements The new elements to append to the array.
029    * @return A new array with the specified elements appended.
030    */
031   @SuppressWarnings("unchecked")
032   public static <T> T[] append(T[] array, T...newElements) {
033      assertNotNull(array, "Array cannot be null.");
034      if (newElements.length == 0)
035         return array;
036      T[] a = (T[])Array.newInstance(array.getClass().getComponentType(), array.length + newElements.length);
037      for (int i = 0; i < array.length; i++)
038         a[i] = array[i];
039      for (int i = 0; i < newElements.length; i++)
040         a[i+array.length] = newElements[i];
041      return a;
042   }

044   /**
045    * Appends one or more elements to an array.
046    *
047    * @param <T> The element type.
048    * @param array The array to append to.
049    * @param newElements The new elements to append to the array.
050    * @return A new array with the specified elements appended.
051    */
052   @SuppressWarnings("unchecked")
053   public static <T> T[] append(T[] array, Collection<T> newElements) {
054      assertFieldNotNull(array, "array");
055      if (newElements.size() == 0)
056         return array;
057      T[] a = (T[])Array.newInstance(array.getClass().getComponentType(), array.length + newElements.size());
058      for (int i = 0; i < array.length; i++)
059         a[i] = array[i];
060      int l = array.length;
061      for (T t : newElements)
062         a[l++] = t;
063      return a;
064   }

066   /**
067    * Combine an arbitrary number of arrays into a single array.
068    *
069    * @param arrays Collection of arrays to combine.
070    * @return A new combined array, or <jk>null</jk> if all arrays are <jk>null</jk>.
071    */
072   @SuppressWarnings("unchecked")
073   public static <T> T[] combine(T[]...arrays) {
074      assertFieldNotNull(arrays, "arrays");
075      int l = 0;
076      T[] a1 = null;
077      for (T[] a : arrays) {
078         if (a1 == null && a != null)
079            a1 = a;
080         l += (a == null ? 0 : a.length);
081      }
082      if (a1 == null)
083         return null;
084      T[] a = (T[])Array.newInstance(a1.getClass().getComponentType(), l);
085      int i = 0;
086      for (T[] aa : arrays)
087         if (aa != null)
088            for (T t : aa)
089               a[i++] = t;
090      return a;
091   }

093   /**
094    * Creates a new array with reversed entries.
095    *
096    * @param <T> The class type of the array.
097    * @param array The array to reverse.
098    * @return A new array with reversed entries, or <jk>null</jk> if the array was <jk>null</jk>.
099    */
100   @SuppressWarnings("unchecked")
101   public static <T> T[] reverse(T[] array) {
102      assertFieldNotNull(array, "array");
103      Class<T> c = (Class<T>)array.getClass().getComponentType();
104      T[] a2 = (T[])Array.newInstance(c, array.length);
105      for (int i = 0; i < array.length; i++)
106         a2[a2.length-i-1] = array[i];
107      return a2;
108   }

110   /**
111    * Converts the specified array to a <code>Set</code>.
112    * <p>
113    *    The order of the entries in the set are the same as the array.
114    *
115    * @param <T> The entry type of the array.
116    * @param array The array being wrapped in a <code>Set</code> interface.
117    * @return The new set.
118    */
119   public static <T> Set<T> asSet(final T[] array) {
120      assertFieldNotNull(array, "array");
121      return new AbstractSet<T>() {
122
123         @Override /* Set */
124         public Iterator<T> iterator() {
125            return new Iterator<T>() {
126               int i = 0;
127
128               @Override /* Iterator */
129               public boolean hasNext() {
130                  return i < array.length;
131               }
132
133               @Override /* Iterator */
134               public T next() {
135                  if (i >= array.length)
136                     throw new NoSuchElementException();
137                  T t = array[i];
138                  i++;
139                  return t;
140               }
141
142               @Override /* Iterator */
143               public void remove() {
144                  throw new UnsupportedOperationException();
145               }
146            };
147         }
148
149         @Override /* Set */
150         public int size() {
151            return array.length;
152         }
153      };
154   }

156   /**
157    * Returns an iterator against an array.
158    * This works with any array type (e.g. <code>String[]</code>, <code>Object[]</code>, <code><jk>int</jk>[]</code>, etc...).
159    *
160    * @param array The array to create an iterator over.
161    * @return An iterator over the specified array.
162    */
163   public static Iterator<Object> iterator(final Object array) {
164      return new Iterator<Object>() {
165         int i = 0;
166         int length = array == null ? 0 : Array.getLength(array);
167
168         @Override /* Iterator */
169         public boolean hasNext() {
170            return i < length;
171         }
172
173         @Override /* Iterator */
174         public Object next() {
175            if (i >= length)
176               throw new NoSuchElementException();
177            return Array.get(array, i++);
178         }
179
180         @Override /* Iterator */
181         public void remove() {
182            throw new UnsupportedOperationException();
183         }
184      };
185   }

187   /**
188    * Converts the specified collection to an array.
189    * Works on both object and primitive arrays.
190    *
191    * @param c The collection to convert to an array.
192    * @param componentType The component type of the collection.
193    * @return A new array.
194    */
195   public static <T> Object toArray(Collection<T> c, Class<T> componentType) {
196      Object a = Array.newInstance(componentType, c.size());
197      Iterator<T> it = c.iterator();
198      int i = 0;
199      while (it.hasNext())
200         Array.set(a, i++, it.next());
201      return a;
202   }

204   /**
205    * Copies the specified array into the specified list.
206    * Works on both object and primitive arrays.
207    *
208    * @param array The array to copy into a list.
209    * @param list The list to copy the values into.
210    */
211   @SuppressWarnings({"unchecked","rawtypes"})
212   public static void copyToList(Object array, List list) {
213      if (array != null) {
214         int length = Array.getLength(array);
215         for (int i = 0; i < length; i++)
216            list.add(Array.get(array, i));
217      }
218   }

220   /**
221    * Returns <jk>true</jk> if the specified array contains the specified element
222    *    using the {@link Object#equals(Object)} method.
223    *
224    * @param element The element to check for.
225    * @param array The array to check.
226    * @return <jk>true</jk> if the specified array contains the specified element,
227    *    <jk>false</jk> if the array or element is <jk>null</jk>.
228    */
229   public static <T> boolean contains(T element, T[] array) {
230      return indexOf(element, array) != -1;
231   }
232
233   /**
234    * Returns the index position of the element in the specified array
235    *    using the {@link Object#equals(Object)} method.
236    *
237    * @param element The element to check for.
238    * @param array The array to check.
239    * @return The index position of the element in the specified array, or
240    *    <code>-1</code> if the array doesn't contain the element, or the array or element is <jk>null</jk>.
241    */
242   public static <T> int indexOf(T element, T[] array) {
243      if (element == null)
244         return -1;
245      if (array == null)
246         return -1;
247      for (int i = 0; i < array.length; i++)
248         if (element.equals(array[i]))
249            return i;
250      return -1;
251   }

253   /**
254    * Converts a primitive wrapper array (e.g. <code>Integer[]</code>) to a primitive array (e.g. <code><jk>int</jk>[]</code>).
255    *
256    * @param o The array to convert.  Must be a primitive wrapper array.
257    * @return A new array.
258    * @throws IllegalArgumentException If object is not a wrapper object array.
259    */
260   public static Object toPrimitiveArray(Object o) {
261      Class<?> c = o.getClass();
262      if (! c.isArray())
263         throw new IllegalArgumentException("Cannot pass non-array objects to toPrimitiveArray()");
264      int l = Array.getLength(o);
265      Class<?> tc = ClassUtils.getPrimitiveForWrapper(c.getComponentType());
266      if (tc == null)
267         throw new IllegalArgumentException("Array type is not a primitive wrapper array.");
268      Object a = Array.newInstance(tc, l);
269      for (int i = 0; i < l; i++)
270         Array.set(a, i, Array.get(o, i));
271      return a;
272   }
