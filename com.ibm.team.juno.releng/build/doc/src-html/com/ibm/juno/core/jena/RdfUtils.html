009package com.ibm.juno.core.jena;
010
011import java.util.*;
012
013import com.ibm.juno.core.*;
014import com.ibm.juno.core.jena.annotation.*;
015import com.ibm.juno.core.xml.*;
016
017/**
018 * Utility classes.
019 */
020public class RdfUtils {
021
022   /**
023    * Find the namespace given a list of <ja>@Rdf</ja> and <ja>@RdfSchema</ja> annotations.
024    * The annotations should be a child-to-parent ordering of annotations found on
025    *    a class or method.
026    *
027    * @param rdfs The <code>@Rdf</code> annotations to search.
028    * @param schemas The list of known RDF schemas.
029    * @return The resolved namespace, or <jk>null</jk> if the namespace could not be resolved.
030    */
031   public static Namespace findNamespace(List<Rdf> rdfs, List<RdfSchema> schemas) {
032
033      for (Rdf rdf : rdfs) {
034         Namespace ns = findNamespace(rdf.prefix(), rdf.namespace(), rdfs, schemas);
035         if (ns != null)
036            return ns;
037      }
038
039      for (RdfSchema schema : schemas) {
040         Namespace ns = findNamespace(schema.prefix(), schema.namespace(), null, schemas);
041         if (ns != null)
042            return ns;
043      }
044
045      return null;
046   }
047
048   private static Namespace findNamespace(String prefix, String ns, List<Rdf> rdfs, List<RdfSchema> schemas) {
049
050      // If both prefix and namespace specified, use that Namespace mapping.
051      if (! (prefix.isEmpty() || ns.isEmpty()))
052         return NamespaceFactory.get(prefix, ns);
053
054      // If only prefix specified, need to search for namespaceURI.
055      if (! prefix.isEmpty()) {
056         if (rdfs != null)
057            for (Rdf rdf2 : rdfs)
058               if (rdf2.prefix().equals(prefix) && ! rdf2.namespace().isEmpty())
059                  return NamespaceFactory.get(prefix, rdf2.namespace());
060         for (RdfSchema schema : schemas) {
061            if (schema.prefix().equals(prefix) && ! schema.namespace().isEmpty())
062               return NamespaceFactory.get(prefix, schema.namespace());
063            for (RdfNs rdfNs : schema.rdfNs())
064               if (rdfNs.prefix().equals(prefix))
065                  return NamespaceFactory.get(prefix, rdfNs.namespaceURI());
066         }
067         throw new BeanRuntimeException("Found @Rdf.prefix annotation with no matching URI.  prefix='"+prefix+"'");
068      }
069
070      // If only namespaceURI specified, need to search for prefix.
071      if (! ns.isEmpty()) {
072         if (rdfs != null)
073            for (Rdf rdf2 : rdfs)
074               if (rdf2.namespace().equals(ns) && ! rdf2.prefix().isEmpty())
075                  return NamespaceFactory.get(rdf2.prefix(), ns);
076         for (RdfSchema schema : schemas) {
077            if (schema.namespace().equals(ns) && ! schema.prefix().isEmpty())
078               return NamespaceFactory.get(schema.prefix(), ns);
079            for (RdfNs rdfNs : schema.rdfNs())
080               if (rdfNs.namespaceURI().equals(ns))
081                  return NamespaceFactory.get(rdfNs.prefix(), ns);
082         }
083      }
084
085      return null;
086   }
087}
