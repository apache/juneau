016/**
017 * Specialized writer for serializing UON-encoded text.
018 * <p>
019 *    <b>Note:  This class is not intended for external use.</b>
020 *
021 * @author James Bognar (jbognar@us.ibm.com)
022 */
023public final class UonSerializerWriter extends SerializerWriter {
024
025   private final boolean simpleMode, encodeChars;
026
027   // Characters that do not need to be URL-encoded in strings.
028   private static final AsciiSet unencodedChars = new AsciiSet("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789;/?:@-_.!*'$(),~=");
029
030   // Characters that do not need to be URL-encoded in attribute names.
031   // Identical to unencodedChars, but excludes '='.
032   private static final AsciiSet unencodedCharsAttrName = new AsciiSet("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789;/?:@-_.!*'$(),~");
033
034   // Characters that need to be preceeded with an escape character.
035   private static final AsciiSet escapedChars = new AsciiSet(",()~=");
036
037   // AsciiSet that maps no characters.
038   private static final AsciiSet emptyCharSet = new AsciiSet("");
039
040   private static char[] hexArray = "0123456789ABCDEF".toCharArray();
041
042   /**
043    * Constructor.
044    *
045    * @param out The writer being wrapped.
046    * @param useIndentation If <jk>true</jk>, tabs will be used in output.
047    * @param simpleMode If <jk>true</jk>, type flags will not be generated in output.
048    * @param encodeChars If <jk>true</jk>, special characters should be encoded.
049    * @param trimStrings If <jk>true</jk>, strings should be trimmed before they're serialized.
050    * @param relativeUriBase The base (e.g. <js>https://localhost:9443/contextPath"</js>) for relative URIs (e.g. <js>"my/path"</js>).
051    * @param absolutePathUriBase The base (e.g. <js>https://localhost:9443"</js>) for relative URIs with absolute paths (e.g. <js>"/contextPath/my/path"</js>).
052    */
053   protected UonSerializerWriter(Writer out, boolean useIndentation, boolean simpleMode, boolean encodeChars, boolean trimStrings, String relativeUriBase, String absolutePathUriBase) {
054      super(out, useIndentation, false, trimStrings, '\'', relativeUriBase, absolutePathUriBase);
055      this.simpleMode = simpleMode;
056      this.encodeChars = encodeChars;
057   }
058
059   /**
060    * Serializes the specified simple object as a UON string value.
061    *
062    * @param o The object being serialized.
063    * @param quoteEmptyStrings Special case where we're serializing an array containing an empty string.
064    * @param isTopAttrName If this is a top-level attribute name we're serializing.
065    * @param isTop If this is a top-level value we're serializing.
066    * @return This object (for method chaining).
067    * @throws IOException Should never happen.
068    */
069   protected UonSerializerWriter appendObject(Object o, boolean quoteEmptyStrings, boolean isTopAttrName, boolean isTop) throws IOException {
070
071      char typeFlag = 0;
072
073      if (o == null)
074         o = "\u0000";
075      else if (o.equals("\u0000"))
076         typeFlag = 's';
077
078      String s = o.toString();
079      if (trimStrings)
080         s = s.trim();
081      if (s.isEmpty()) {
082         if (quoteEmptyStrings)
083            typeFlag = 's';
084      } else if (s.charAt(0) == '(' || s.charAt(0) == '$') {
085         typeFlag = 's';
086      } else if (useIndentation && (s.indexOf('\n') != -1 || (s.charAt(0) <= ' ' && s.charAt(0) != 0))) {
087         // Strings containing newline characters must always be quoted so that they're not confused with whitespace.
088         // Also, strings starting with whitespace must be quoted so that the contents are not ignored when whitespace is ignored.
089         typeFlag = 's';
090      } else if (! simpleMode) {
091         if (o instanceof Boolean)
092            typeFlag = 'b';
093         else if (o instanceof Number)
094            typeFlag = 'n';
095      }
096
097      if (typeFlag != 0)
098         startFlag(typeFlag);
099
100      AsciiSet unenc = (isTopAttrName ? unencodedCharsAttrName : unencodedChars);
101      AsciiSet esc = (isTop && typeFlag == 0 ? emptyCharSet : escapedChars);
102
103      for (int i = 0; i < s.length(); i++) {
104         char c = s.charAt(i);
105         if (esc.contains(c))
106            append('~');
107         if ((!encodeChars) || unenc.contains(c))
108            append(c);
109         else {
110            if (c == ' ')
111               append('+');
112            else {
113               int p = s.codePointAt(i);
114               if (p < 0x0080)
115                  appendHex(p);
116               else if (p < 0x0800) {
117                  int p1=p>>>6;
118                  appendHex(p1+192).appendHex((p&63)+128);
119               } else if (p < 0x10000) {
120                  int p1=p>>>6, p2=p1>>>6;
121                  appendHex(p2+224).appendHex((p1&63)+128).appendHex((p&63)+128);
122               } else {
123                  i++;  // Two-byte codepoint...skip past surrogate pair lower byte.
124                  int p1=p>>>6, p2=p1>>>6, p3=p2>>>6;
125                  appendHex(p3+240).appendHex((p2&63)+128).appendHex((p1&63)+128).appendHex((p&63)+128);
126               }
127            }
128         }
129      }
130
131      if (typeFlag != 0)
132         append(')');
133
134      return this;
135   }
136
137   /**
138    * Prints <code>$f(</code> in normal mode, and <code>(</code> in simple mode.
139    *
140    * @param f The flag character.
141    * @return This object (for method chaining).
142    * @throws IOException
143    */
144   protected UonSerializerWriter startFlag(char f) throws IOException {
145      if (f != 's' && ! simpleMode)
146         append('$').append(f);
147      append('(');
148      return this;
149   }
150
151   /**
152    * Prints out a two-byte %xx sequence for the given byte value.
153    */
154   private UonSerializerWriter appendHex(int b) throws IOException {
155      if (b > 255)
156         throw new IOException("Invalid value passed to appendHex.  Must be in the range 0-255.  Value=" + b);
157      append('%').append(hexArray[b>>>4]).append(hexArray[b&0x0F]);
158      return this;
159   }
160
161   /**
162    * Appends a URI to the output.
163    *
164    * @param uri The URI to append to the output.
165    * @param isTop If this is a top-level value we're serializing.
166    * @return This object (for method chaining).
167    * @throws IOException
168    */
169   public SerializerWriter appendUri(Object uri, boolean isTop) throws IOException {
170      String s = uri.toString();
171      if (s.indexOf("://") == -1) {
172         if (StringUtils.startsWith(s, '/')) {
173            if (absolutePathUriBase != null)
174               append(absolutePathUriBase);
175         } else {
176            if (relativeUriBase != null) {
177               append(relativeUriBase);
178               if (! relativeUriBase.equals("/"))
179                  append("/");
180
181            }
182         }
183      }
184      return appendObject(s, false, false, isTop);
185   }
