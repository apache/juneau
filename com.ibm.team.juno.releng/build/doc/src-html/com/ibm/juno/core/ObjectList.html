019/**
020 * Java implementation of a JSON array.
021 * <p>
022 *    An extension of {@link LinkedList}, so all methods available to in that class are also available
023 *    to this class.
024 * <p>
025 *    Note that the use of this class is optional.  The serializers will accept any objects that implement
026 *    the {@link Collection} interface.  But this class provides some useful additional functionality
027 *    when working with JSON models constructed from Java Collections Framework objects.  For example, a
028 *    constructor is provided for converting a JSON array string directly into a {@link List}.  It also contains
029 *    accessor methods for to avoid common typecasting when accessing elements in a list.
030 *
031 * <h6 class='topic'>Examples</h6>
032 * <p class='bcode'>
033 *    <jc>// Construct an empty List</jc>
034 *    List l = <jk>new</jk> ObjectList();
035 *
036 *    <jc>// Construct a list of objects using various methods</jc>
037 *    l = <jk>new</jk> ObjectList().append(<js>"foo"</js>).append(123).append(<jk>true</jk>);
038 *    l = <jk>new</jk> ObjectList().append(<js>"foo"</js>, 123, <jk>true</jk>);  <jc>// Equivalent</jc>
039 *    l = <jk>new</jk> ObjectList(<js>"foo"</js>, 123, <jk>true</jk>);  <jc>// Equivalent</jc>
040 *
041 *    <jc>// Construct a list of integers from JSON</jc>
042 *    l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
043 *
044 *    <jc>// Construct a list of generic ObjectMap objects from JSON</jc>
045 *    l = <jk>new</jk> ObjectList(<js>"[{foo:'bar'},{baz:'bing'}]"</js>);
046 *
047 *    <jc>// Construct a list of integers from XML</jc>
048 *    String xml = <js>"&lt;array&gt;&lt;number&gt;1&lt;/number&gt;&lt;number&gt;2&lt;/number&gt;&lt;number&gt;3&lt;/number&gt;&lt;/array&gt;"</js>;
049 *    l = <jk>new</jk> ObjectList(xml, DataFormat.<jsf>XML</jsf>);
050 *    l = (List)XmlParser.<jsf>DEFAULT</jsf>.parse(xml);  <jc>// Equivalent</jc>
051 *    l = (List)XmlParser.<jsf>DEFAULT</jsf>.parse(Object.<jk>class</jk>, xml);  <jc>// Equivalent</jc>
052 *    l = XmlParser.<jsf>DEFAULT</jsf>.parse(List.<jk>class</jk>, xml);  <jc>// Equivalent</jc>
053 *    l = XmlParser.<jsf>DEFAULT</jsf>.parse(ObjectList.<jk>class</jk>, xml);  <jc>// Equivalent</jc>
054 *
055 *    <jc>// Construct JSON from ObjectList</jc>
056 *    l = <jk>new</jk> ObjectList(<js>"[{foo:'bar'},{baz:'bing'}]"</js>);
057 *    String json = l.toString();  <jc>// Produces "[{foo:'bar'},{baz:'bing'}]"</jc>
058 *    json = l.toString(JsonSerializer.<jsf>DEFAULT_CONDENSED</jsf>);  <jc>// Equivalent</jc>
059 *    json = JsonSerializer.<jsf>DEFAULT_CONDENSED</jsf>.serialize(l);  <jc>// Equivalent</jc>
060 *
061 *    <jc>// Get one of the entries in the list as an Integer</jc>
062 *    l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
063 *    Integer i = l.getInt(1);
064 *    i = l.get(Integer.<jk>class</jk>, 1);  <jc>// Equivalent</jc>
065 *
066 *    <jc>// Get one of the entries in the list as an Float</jc>
067 *    l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
068 *    Float f = l.getFloat(1); <jc>// Returns 2f </jc>
069 *    f = l.get(Float.<jk>class</jk>, 1);  <jc>// Equivalent</jc>
070 *
071 *    <jc>// Same as above, except converted to a String</jc>
072 *    l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
073 *    String s = l.getString(1); <jc>// Returns "2" </jc>
074 *    s = l.get(String.<jk>class</jk>, 1);  <jc>// Equivalent</jc>
075 *
076 *    <jc>// Get one of the entries in the list as a bean (converted to a bean if it isn't already one)</jc>
077 *    l = <jk>new</jk> ObjectList(<js>"[{name:'John Smith',age:45}]"</js>);
078 *    Person p = l.get(Person.<jk>class</jk>, 0);
079 *
080 *    <jc>// Iterate over a list of beans using the elements() method</jc>
081 *    ObjectList ObjectList = <jk>new</jk> ObjectList(<js>"[{name:'John Smith',age:45}]"</js>);
082 *    <jk>for</jk> (Person p : ObjectList.elements(Person.<jk>class</jk>) {
083 *       <jc>// Do something with p</jc>
084 *    }
085 * </p>
086 * <p>
087 *    This class is not thread safe.
088 * </p>
089 *
090 * @author James Bognar (jbognar@us.ibm.com)
091 */

130   /**
131    * Construct a JSON array directly from text using the specified parser.
132    *
133    * @param s The string being parsed.
134    * @param p The parser to use to parse the input.
135    * @throws ParseException If the input contains a syntax error or is malformed.
136    */
137   public ObjectList(CharSequence s, ReaderParser p) throws ParseException {
138      this(p == null ? BeanContext.DEFAULT : p.getBeanContext());
139      try {
140         if (p == null)
141            p = JsonParser.DEFAULT;
142         if (s != null)
143            p.parseIntoCollection(new CharSequenceReader(s), s.length(), this, beanContext.object());
144      } catch (IOException e) {
145         throw new ParseException(e);
146      }
147   }

154    */
155   public ObjectList(CharSequence s) throws ParseException {
156      this(s, null);
157   }

159   /**
160    * Construct a JSON array directly from a reader using the specified parser.
161    *
162    * @param r The reader to read from.  Will automatically be wrapped in a {@link BufferedReader} if it isn't already a BufferedReader.
163    * @param p The parser to use to parse the input.
164    * @throws ParseException If the input contains a syntax error or is malformed.
165    * @throws IOException If a problem occurred trying to read from the reader.
166    */
167   public ObjectList(Reader r, ReaderParser p) throws ParseException, IOException {
168      parseReader(r, p);
169   }
170
171   private void parseReader(Reader r, ReaderParser p) throws IOException, ParseException {
172      if (p == null)
173         p = JsonParser.DEFAULT;
174      p.parseIntoCollection(r, -1, this, beanContext.object());
175   }

342    * <p>

358    *
359    * @param <T> The class type.
360    * @param type The class type.

362    * @return The value, or <jk>null</jk> if the entry doesn't exist.

451   /**
452    * Creates an {@link Iterable} with elements of the specified child type.
453    * <p>
454    * Attempts to convert the child objects to the correct type if they aren't already the correct type.
455    * <p>
456    * The <code>next()</code> method on the returned iterator may throw a {@link InvalidDataConversionException} if
457    *    the next element cannot be converted to the specified type.
458    * <p>
459    * See {@link BeanContext#convertToType(Object, ClassMeta)} for a description of valid conversions.
460    *
461    * <dl>
462    *    <dt>Example:</dt>
463    *    <dd>
464    *       <p class='bcode'>
465    *    <jc>// Iterate over a list of ObjectMaps.</jc>
466    *    ObjectList l = <jk>new</jk> ObjectList(<js>"[{foo:'bar'},{baz:123}]"</js>);
467    *    for (ObjectMap m : l.elements(ObjectMap.<jk>class</jk>)) {
468    *       <jc>// Do something with m.</jc>
469    *    }
470    *
471    *    <jc>// Iterate over a list of ints.</jc>
472    *    ObjectList l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
473    *    for (Integer i : l.elements(Integer.<jk>class</jk>)) {
474    *       <jc>// Do something with i.</jc>
475    *    }
476    *
477    *    <jc>// Iterate over a list of beans.</jc>
478    *    <jc>// Automatically converts to beans.</jc>
479    *    ObjectList l = <jk>new</jk> ObjectList(<js>"[{name:'John Smith',age:45}]"</js>);
480    *    for (Person p : l.elements(Person.<jk>class</jk>)) {
481    *       <jc>// Do something with p.</jc>
482    *    }
483    *       </p>
484    *    </dd>
485    * </dl>
486    *
487    * @param <E> The child object type.
488    * @param childType The child object type.
489    * @return A new <code>Iterable</code> object over this list.
490    */
491   public <E> Iterable<E> elements(final Class<E> childType) {
492      final Iterator<?> i = iterator();
493      return new Iterable<E>() {
494
495         @Override /* Iterable */
496         public Iterator<E> iterator() {
497            return new Iterator<E>() {
498
499               @Override /* Iterator */
500               public boolean hasNext() {
501                  return i.hasNext();
502               }
503
504               @Override /* Iterator */
505               public E next() {
506                  return beanContext.convertToType(i.next(), childType);
507               }
508
509               @Override /* Iterator */
510               public void remove() {
511                  i.remove();
512               }
513
514            };
515         }
516      };
517   }
