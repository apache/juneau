009package com.ibm.juno.core.xml;
010
011import java.util.*;
012import java.util.concurrent.*;
013
014import com.ibm.juno.core.*;
015import com.ibm.juno.core.parser.*;
016import com.ibm.juno.core.utils.*;
017
018/**
019 * Factory class for getting unique instances of {@link Namespace} objects.
020 * <p>
021 *    For performance reasons, {@link Namespace} objects are stored in {@link IdentityList IdentityLists}.
022 *    For this to work property, namespaces with the same name and URI must only be represented by a single
023 *    {@link Namespace} instance.
024 *    This factory class ensures this identity uniqueness.
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028public final class NamespaceFactory {
029
030   private static ConcurrentHashMap<String,Namespace> cache = new ConcurrentHashMap<String,Namespace>();
031
032   /**
033    * Get the {@link Namespace} with the specified name and URI, and create a new one
034    *    if this is the first time it's been encountered.
035    *
036    * @param name The namespace name.  See {@link Namespace#getName()}.
037    * @param uri The namespace URI.  See {@link Namespace#getUri()}.
038    * @return The namespace object.
039    */
040   public static Namespace get(String name, String uri) {
041      String key = name + "+" + uri;
042      Namespace n = cache.get(key);
043      if (n == null) {
044         n = new Namespace(name, uri);
045         Namespace n2 = cache.putIfAbsent(key, n);
046         return (n2 == null ? n : n2);
047      }
048      return n;
049   }
050
051   /**
052    * Converts the specified object into a {@link Namespace} object.
053    * <p>
054    *    Can be any of following types:
055    * <ul>
056    *    <li>A {@link Namespace} object
057    *    <li>A JSON string containing a single key/value pair indicating the name/URI mapping.
058    *    <li>A <code>Map</code> containing a single key/value pair indicating the name/URI mapping.
059    * </ul>
060    *
061    * @param o The input.
062    * @return The namespace object, or <jk>null</jk> if the input was <jk>null</jk> or an empty JSON object.
063    */
064   @SuppressWarnings("rawtypes")
065   public static Namespace parseNamespace(Object o) {
066      if (o == null)
067         return null;
068      if (o instanceof Namespace)
069         return (Namespace)o;
070      try {
071         Map<?,?> m = (o instanceof Map ? (Map)o : new ObjectMap(o.toString()));
072         if (m.size() == 0)
073            return null;
074         if (m.size() > 1)
075            throw new RuntimeException("Too many namespaces specified.  Only one allowed. '"+o+"'");
076         Map.Entry<?,?> e = m.entrySet().iterator().next();
077         return get(e.getKey().toString(), e.getValue().toString());
078      } catch (ParseException e) {
079         throw new RuntimeException(e);
080      }
081   }
082
083   /**
084    * Converts the specified object into an array of {@link Namespace} object.
085    * <p>
086    *    Can be any of following types:
087    * <ul>
088    *    <li>A {@link Namespace} array
089    *    <li>A JSON string with key/value pairs indicating name/URI pairs.
090    *    <li>A <code>Map</code> with key/value pairs indicating name/URI pairs.
091    *    <li>A <code>Collection</code> containing any of object that can be passed to {@link #parseNamespace(Object)}.
092    * </ul>
093    *
094    * @param o The input.
095    * @return The namespace objects, or <jk>null</jk> if the input was <jk>null</jk> or an empty JSON object.
096    */
097   @SuppressWarnings("rawtypes")
098   public static Namespace[] parseNamespaces(Object o) {
099      try {
100         if (o instanceof Namespace[])
101            return (Namespace[])o;
102
103         if (o instanceof CharSequence)
104            o = new ObjectMap(o.toString());
105
106         Namespace[] n;
107         int i = 0;
108         if (o instanceof Collection) {
109            Collection c = (Collection)o;
110            n = new Namespace[c.size()];
111            for (Object o2 : c)
112               n[i++] = parseNamespace(o2);
113         } else if (o instanceof Map) {
114            Map<?,?> m = (Map<?,?>)o;
115            n = new Namespace[m.size()];
116            for (Map.Entry e : m.entrySet())
117               n[i++] = get(e.getKey().toString(), e.getValue().toString());
118         } else {
119            throw new RuntimeException("Invalid type passed to NamespaceFactory.listFromObject: '"+o+"'");
120         }
121         return n;
122      } catch (ParseException e) {
123         throw new RuntimeException(e);
124      }
125   }
126}
