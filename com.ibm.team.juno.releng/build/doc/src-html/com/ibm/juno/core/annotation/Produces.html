009package com.ibm.juno.core.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016import com.ibm.juno.core.serializer.*;
017
018/**
019 * Annotation used on subclasses of {@link Serializer} to identify the media types that it produces.
020 *
021 *
022 * <h6 class='topic'>Description</h6>
023 * <p>
024 *    Provides a way to define the contents of {@link Serializer#getMediaTypes()} through an annotation.
025 * <p>
026 *    The {@link Serializer#getMediaTypes()} default implementation gathers the media types by looking
027 *       for this annotation.
028 *    It should be noted that this annotation is optional and that the {@link Serializer#getMediaTypes()} method can
029 *       be overridden by subclasses to return the media types programmatically.
030 *
031 *
032 * <h6 class='topic'>Examples</h6>
033 * <p>
034 *    Standard example:
035 * <p class='bcode'>
036 *    <ja>@Produces</ja>({<js>"application/json"</js>,<js>"text/json"</js>})
037 *    <jk>public class</jk> JsonSerializer <jk>extends</jk> WriterSerializer {...}
038 * </p>
039 * <p>
040 *    The media types can also be <code>media-range</code> values per
041 *       <a href='http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.1'>RFC2616/14.1</a>.
042 *    When meta-characters are used, you should specify the {@link #contentType()} value to
043 *       indicate the real media type value that can be set on the <code>Content-Type</code> response header.
044 *
045 * <p class='bcode'>
046 *    <jc>// Produces any text</jc>
047 *    <ja>@Produces</ja>(value=<js>"text\/*"</js>, contentType=<js>"text/plain"</js>)
048 *    <jk>public class</jk> AnythingSerializer <jk>extends</jk> WriterSerializer {...}
049 *
050 *    <jc>// Produces anything</jc>
051 *    <ja>@Produces</ja>(value=<js>"*\/*"</js>, contentType=<js>"text/plain"</js>)
052 *    <jk>public class</jk> AnythingSerializer <jk>extends</jk> WriterSerializer {...}
053 * </p>
054 *
055 *
056 * @author James Bognar (jbognar@us.ibm.com)
057 */
058@Documented
059@Target(TYPE)
060@Retention(RUNTIME)
061@Inherited
062public @interface Produces {
063
064   /**
065    * The media types that the serializer can handle.
066    * <p>
067    *    Can contain meta-characters per the <code>media-type</code> specification of
068    *    <a href='http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.1'>RFC2616/14.1</a>
069    * @return The media types that the parser can handle.
070    */
071   String[] value() default {};
072
073   /**
074    * The content type that this serializer produces.
075    * <p>
076    *    Can be used to override the <code>Content-Type</code> response type if the media types
077    *       are <code>media-ranges</code> with meta-characters, or the <code>Content-Type</code>
078    *       differs from the media type for some reason.
079    * @return The content type that this serializer produces, or blank if no overriding value exists.
080    */
081   String contentType() default "";
082}
