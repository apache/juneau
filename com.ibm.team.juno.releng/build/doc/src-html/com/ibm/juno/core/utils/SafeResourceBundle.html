009package com.ibm.juno.core.utils;
010
011import java.text.*;
012import java.util.*;
013
014/**
015 * Wraps a {@link ResourceBundle} to gracefully handle missing bundles and entries.
016 * <p>
017 * If the bundle isn't found, <code>getString(key)</code> returns <js>"{!!key}"</js>.
018 * <p>
019 * If the key isn't found, <code>getString(key)</code> returns <js>"{!key}"</js>.
020 *
021 * @author James Bognar (jbognar@us.ibm.com)
022 */
023public class SafeResourceBundle extends ResourceBundle {
024
025   private ResourceBundle rb;
026   private String className = null;
027   Map<String,String> classPrefixedKeys = new HashMap<String,String>();
028
029   SafeResourceBundle() {}
030
031   /**
032    * Constructor.
033    *
034    * @param rb The resource bundle to wrap.  Can be <jk>null</jk> to indicate a missing bundle.
035    * @param forClass The class using this resource bundle.
036    */
037   public SafeResourceBundle(ResourceBundle rb, Class<?> forClass) {
038      this.rb = rb;
039      if (forClass != null)
040         className = forClass.getSimpleName();
041      if (rb != null) {
042         String c = className + '.';
043         for (Enumeration<String> e = getKeys(); e.hasMoreElements();) {
044            String key = e.nextElement();
045            if (key.startsWith(c))
046               classPrefixedKeys.put(key.substring(className.length() + 1), key);
047         }
048      }
049   }
050
051   @Override /* ResourceBundle */
052   public boolean containsKey(String key) {
053      return rb != null && (rb.containsKey(key) || classPrefixedKeys.containsKey(key));
054   }
055
056   /**
057    * Similar to {@link ResourceBundle#getString(String)} except allows you to pass in {@link MessageFormat} objects.
058    *
059    * @param key The resource bundle key.
060    * @param args Optional variable replacement arguments.
061    * @return The resolved value.  Never <jk>null</jk>.  <js>"{!!key}"</j> if the bundle is missing.  <js>"{!key}"</j> if the key is missing.
062    */
063   public String getString(String key, Object...args) {
064      if (rb == null)
065         return "{!!"+key+"}";
066      if (! containsKey(key))
067         return "{!" + key + "}";
068      String val = getString(key);
069      if (args.length > 0)
070         return MessageFormat.format(val, args);
071      return val;
072   }
073
074   /**
075    * Looks for all the specified keys in the resource bundle and returns the first value that exists.
076    *
077    * @param keys
078    * @return The resolved value, or <jk>null</jk> if no value is found or the resource bundle is missing.
079    */
080   public String findFirstString(String...keys) {
081      if (rb == null)
082         return null;
083      for (String k : keys) {
084         if (containsKey(k))
085            return getString(k);
086      }
087      return null;
088   }
089
090   @SuppressWarnings("unchecked")
091   @Override /* ResourceBundle */
092   public Set<String> keySet() {
093      if (rb == null)
094         return Collections.emptySet();
095      return new MultiSet<String>(rb.keySet(), classPrefixedKeys.keySet()) ;
096   }
097
098   /**
099    * Returns all keys in this resource bundle with the specified prefix.
100    *
101    * @param prefix The prefix.
102    * @return The set of all keys in the resource bundle with the prefix.
103    */
104   public Set<String> keySet(String prefix) {
105      Set<String> set = new HashSet<String>();
106      for (String s : keySet()) {
107         if (s.equals(prefix) || (s.startsWith(prefix) && s.charAt(prefix.length()) == '.'))
108            set.add(s);
109      }
110      return set;
111   }
112
113   @Override /* ResourceBundle */
114   public Enumeration<String> getKeys() {
115      if (rb == null)
116         return new Vector<String>(0).elements();
117      return rb.getKeys();
118   }
119
120   @Override /* ResourceBundle */
121   protected Object handleGetObject(String key) {
122      if (rb == null)
123         return "{!!"+key+"}";
124      try {
125         if (classPrefixedKeys.containsKey(key))
126            key = classPrefixedKeys.get(key);
127         return rb.getObject(key);
128      } catch (Exception e) {
129         return "{!"+key+"}";
130      }
131   }
132}
