016/**
017 * Specialized writer for serializing JSON.
018 * <p>
019 *    <b>Note:  This class is not intended for external use.</b>
020 *
021 * @author James Bognar (jbognar@us.ibm.com)
022 */
023public final class JsonSerializerWriter extends SerializerWriter {
024
025   private final boolean laxMode, escapeSolidus;
026
027   // Characters that trigger special handling of serializing attribute values.
028   private static final AsciiSet
029      encodedChars = new AsciiSet("\n\t\b\f\r'\"\\"),
030      encodedChars2 = new AsciiSet("\n\t\b\f\r'\"\\/");
031
032   private static final KeywordSet reservedWords = new KeywordSet(
033      "arguments","break","case","catch","class","const","continue","debugger","default","delete",
034      "do","else","enum","eval","export","extends","false","finally","for","function","if",
035      "implements","import","in","instanceof","interface","let","new","null","package",
036      "private","protected","public","return","static","super","switch","this","throw",
037      "true","try","typeof","var","void","while","with","undefined","yield"
038   );
039
040
041   // Characters that represent attribute name characters that don't trigger quoting.
042   // These are actually more strict than the actual Javascript specification, but
043   // can be narrowed in the future if necessary.
044   // For example, we quote attributes that start with $ even though we don't need to.
045   private static final AsciiSet validAttrChars = new AsciiSet("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789_");
046   private static final AsciiSet validFirstAttrChars = new AsciiSet("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ_");
047
048   private final AsciiSet ec;
049
050   /**
051    * Constructor.
052    * @param out The writer being wrapped.
053    * @param useIndentation If <jk>true</jk>, tabs will be used in output.
054    * @param useWhitespace If <jk>true</jk>, whitespace will be used in output.
055    * @param escapeSolidus If <jk>true</jk>, forward slashes should be escaped in the output.
056    * @param quoteChar The quote character to use (i.e. <js>'\''</js> or <js>'"'</js>)
057    * @param laxMode If <jk>true</jk>, JSON attributes will only be quoted when necessary.
058    * @param trimStrings If <jk>true</jk>, strings will be trimmed before being serialized.
059    * @param relativeUriBase The base (e.g. <js>https://localhost:9443/contextPath"</js>) for relative URIs (e.g. <js>"my/path"</js>).
060    * @param absolutePathUriBase The base (e.g. <js>https://localhost:9443"</js>) for relative URIs with absolute paths (e.g. <js>"/contextPath/my/path"</js>).
061    */
062   protected JsonSerializerWriter(Writer out, boolean useIndentation, boolean useWhitespace, boolean escapeSolidus, char quoteChar, boolean laxMode, boolean trimStrings, String relativeUriBase, String absolutePathUriBase) {
063      super(out, useIndentation, useWhitespace, trimStrings, quoteChar, relativeUriBase, absolutePathUriBase);
064      this.laxMode = laxMode;
065      this.escapeSolidus = escapeSolidus;
066      this.ec = escapeSolidus ? encodedChars2 : encodedChars;
067   }
068
069   /**
070    * Serializes the specified object as a JSON string value.
071    * @param o The object being serialized.
072    * @return This object (for method chaining).
073    * @throws IOException Should never happen.
074    */
075   public JsonSerializerWriter stringValue(Object o) throws IOException {
076       /*
077        * Fixes up a Java string so that it can be used as a JSON string.<br>
078        * Does the following:<br>
079        * <ul>
080        *  <li> Replaces {@code \r?\n} with {@code \\n}<br>
081        *  <li> Replaces {@code \t} with {@code \\t}<br>
082        *  <li> Replaces {@code '} with {@code \\'}<br>
083        *  <li> Replaces {@code "} with {@code \\"}<br>
084        * </ul>
085        */
086      if (o == null)
087         return this;
088      String s = o.toString();
089      if (trimStrings)
090         s = s.trim();
091      boolean doConvert = false;
092      for (int i = 0; i < s.length() && ! doConvert; i++) {
093         char c = s.charAt(i);
094         doConvert |= ec.contains(c);
095      }
096      q();
097      if (! doConvert) {
098         out.append(s);
099      } else {
100         for (int i = 0; i < s.length(); i++) {
101            char c = s.charAt(i);
102            if (ec.contains(c)) {
103               if (c == '\n')
104                  out.append('\\').append('n');
105               else if (c == '\t')
106                  out.append('\\').append('t');
107               else if (c == '\b')
108                  out.append('\\').append('b');
109               else if (c == '\f')
110                  out.append('\\').append('f');
111               else if (c == quoteChar)
112                  out.append('\\').append(quoteChar);
113               else if (c == '\\')
114                  out.append('\\').append('\\');
115               else if (c == '/' && escapeSolidus)
116                  out.append('\\').append('/');
117               else if (c != '\r')
118                  out.append(c);
119            } else {
120               out.append(c);
121            }
122         }
123      }
124      q();
125      return this;
126   }
127
128   /**
129    * Serializes the specified object as a JSON attribute name.
130    * @param o The object being serialized.
131    * @return This object (for method chaining).
132    * @throws IOException Should never happen.
133    */
134   public JsonSerializerWriter attr(Object o) throws IOException {
135      /*
136       * Converts a Java string to an acceptable JSON attribute name. If
137       * useStrictJson is false, then quotes will only be used if the attribute
138       * name consists of only alphanumeric characters.
139       */
140      boolean doConvert = trimStrings || ! laxMode;      // Always convert when not in lax mode.
141
142      String s = null;
143
144      // If the attribute is null, it must always be printed as null without quotes.
145      // Technically, this isn't part of the JSON spec, but it does allow for null key values.
146      if (o == null) {
147         s = "null";
148         doConvert = false;
149
150      } else {
151         s = o.toString();
152
153         // Look for characters that would require the attribute to be quoted.
154         // All possible numbers should be caught here.
155         if (! doConvert) {
156            for (int i = 0; i < s.length() && ! doConvert; i++) {
157               char c = s.charAt(i);
158               doConvert |= ! (i == 0 ? validFirstAttrChars.contains(c) : validAttrChars.contains(c));
159            }
160         }
161
162         // Reserved words and blanks must be quoted.
163         if (! doConvert) {
164            if (s.isEmpty() || reservedWords.contains(s))
165               doConvert = true;
166         }
167      }
168
169      // If no conversion necessary, just print the attribute as-is.
170      if (doConvert)
171         stringValue(s);
172      else
173         out.append(s);
174
175      return this;
176   }
