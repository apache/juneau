023/**
024 * Context object that lives for the duration of a single serialization of {@link Serializer} and its subclasses.
025 * <p>
026 *    Used by serializers for the following purposes:
027 *    <ul>
028 *       <li>Keeping track of how deep it is in a model for indentation purposes.
029 *       <li>Ensuring infinite loops don't occur by setting a limit on how deep to traverse a model.
030 *       <li>Ensuring infinite loops don't occur from loops in the model (when detectRecursions is enabled.
031 *       <li>Allowing serializer properties to be overridden on method calls.
032 *    </ul>
033 * </p>
034 * <p>
035 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
036 * </p>
037 *
038 * @author James Bognar (jbognar@us.ibm.com)
039 */
040public class SerializerContext {

082   /**
083    * Create a new HashStack with the specified options.
084    *
085    * @param beanContext The bean context being used by the serializer.
086    * @param sp The default serializer properties.
087    * @param op The override properties.
088    * @param javaMethod Java method that invoked this serializer.
089    *    When using the REST API, this is the Java method invoked by the REST call.
090    *    Can be used to access annotations defined on the method or class.
091    */
092   public SerializerContext(BeanContext beanContext, SerializerProperties sp, ObjectMap op, Method javaMethod) {
093      this.beanContext = beanContext;
094      this.javaMethod = javaMethod;
095      if (op == null || op.isEmpty()) {
096         overrideProperties = new ObjectMap();
097         maxDepth = sp.maxDepth;
098         initialDepth = sp.initialDepth;
099         debug = sp.debug;
100         detectRecursions = sp.detectRecursions;
101         ignoreRecursions = sp.ignoreRecursions;
102         useIndentation = sp.useIndentation;
103         addClassAttrs = sp.addClassAttrs;
104         trimNulls = sp.trimNulls;
105         trimEmptyLists = sp.trimEmptyLists;
106         trimEmptyMaps = sp.trimEmptyMaps;
107         trimStrings = sp.trimStrings;
108         quoteChar = sp.quoteChar;
109         relativeUriBase = resolveRelativeUriBase(sp.relativeUriBase);
110         absolutePathUriBase = resolveAbsolutePathUriBase(sp.absolutePathUriBase);
111         sortCollections = sp.sortCollections;
112         sortMaps = sp.sortMaps;
113      } else {
114         overrideProperties = op;
115         maxDepth = op.getInt(SERIALIZER_maxDepth, sp.maxDepth);
116         initialDepth = op.getInt(SERIALIZER_initialDepth, sp.initialDepth);
117         debug = op.getBoolean(SERIALIZER_debug, sp.debug);
118         detectRecursions = op.getBoolean(SERIALIZER_detectRecursions, sp.detectRecursions);
119         ignoreRecursions = op.getBoolean(SERIALIZER_ignoreRecursions, sp.ignoreRecursions);
120         useIndentation = op.getBoolean(SERIALIZER_useIndentation, sp.useIndentation);
121         addClassAttrs = op.getBoolean(SERIALIZER_addClassAttrs, sp.addClassAttrs);
122         trimNulls = op.getBoolean(SERIALIZER_trimNullProperties, sp.trimNulls);
123         trimEmptyLists = op.getBoolean(SERIALIZER_trimEmptyLists, sp.trimEmptyLists);
124         trimEmptyMaps = op.getBoolean(SERIALIZER_trimEmptyMaps, sp.trimEmptyMaps);
125         trimStrings = op.getBoolean(SERIALIZER_trimStrings, sp.trimStrings);
126         quoteChar = op.getString(SERIALIZER_quoteChar, ""+sp.quoteChar).charAt(0);
127         relativeUriBase = resolveRelativeUriBase(op.getString(SERIALIZER_relativeUriBase, sp.relativeUriBase));
128         absolutePathUriBase = resolveAbsolutePathUriBase(op.getString(SERIALIZER_absolutePathUriBase, sp.absolutePathUriBase));
129         sortCollections = op.getBoolean(SERIALIZER_sortCollections, sp.sortMaps);
130         sortMaps = op.getBoolean(SERIALIZER_sortMaps, sp.sortMaps);
131      }
132
133      this.indent = initialDepth;
134      if (detectRecursions || debug) {
135         set = new IdentityHashMap<Object,Object>();
136         stack = new LinkedList<StackElement>();
137      }
138   }

140   private String resolveRelativeUriBase(String s) {
141      if (StringUtils.isEmpty(s))
142         return null;
143      if (s.equals("/"))
144         return s;
145      else if (StringUtils.endsWith(s, '/'))
146         s = s.substring(0, s.length()-1);
147      return s;
148   }
149
150   private String resolveAbsolutePathUriBase(String s) {
151      if (StringUtils.isEmpty(s))
152         return null;
153      if (StringUtils.endsWith(s, '/'))
154         s = s.substring(0, s.length()-1);
155      return s;
156   }

332   /**
333    * Push the specified object onto the stack.
334    *
335    * @param attrName The attribute name.
336    * @param o The current object being serialized.
337    * @param eType The expected class type.
338    * @return The {@link ClassMeta} of the object so that <code>instanceof</code> operations
339    *    only need to be performed once (since they can be expensive).<br>
340    * @throws SerializeException
341    */
342   public ClassMeta<?> push(String attrName, Object o, ClassMeta<?> eType) throws SerializeException {
343      indent++;
344      isBottom = true;
345      if (o == null)
346         return null;
347      Class<?> c = o.getClass();
348      ClassMeta<?> cm = (eType != null && c == eType.getInnerClass()) ? eType : beanContext.getClassMeta(c);
349      if (cm.isCharSequence() || cm.isNumber() || cm.isBoolean())
350         return cm;
351      if (detectRecursions || debug) {
352         if (stack.size() > maxDepth)
353            return null;
354         if (willRecurse(attrName, o, cm))
355            return null;
356         isBottom = false;
357         stack.add(new StackElement(stack.size(), attrName, o, cm));
358         if (debug)
359            logger.info(getStack(false));
360         set.put(o, o);
361      }
362      return cm;
363   }

365   /**
366    * Returns <jk>true</jk> if {@link SerializerProperties#SERIALIZER_detectRecursions} is enabled, and the specified
367    *    object is already higher up in the serialization chain.
368    *
369    * @param attrName The bean property attribute name, or some other identifier.
370    * @param o The object to check for recursion.
371    * @param cm The metadata on the object class.
372    * @return <jk>true</jk> if recursion detected.
373    * @throws SerializeException
374    */
375   public boolean willRecurse(String attrName, Object o, ClassMeta<?> cm) throws SerializeException {
376      if (! (detectRecursions || debug))
377         return false;
378      if (! set.containsKey(o))
379         return false;
380      if (ignoreRecursions && ! debug)
381         return true;
382
383      stack.add(new StackElement(stack.size(), attrName, o, cm));
384      throw new SerializeException("Recursion occurred, stack={0}", getStack(true));
385   }

387   /**
388    * Pop an object off the stack.
389    */
390   public void pop() {
391      indent--;
392      if ((detectRecursions || debug) && ! isBottom)  {
393         Object o = stack.removeLast().o;
394         Object o2 = set.remove(o);
395         if (o2 == null)
396            addWarning("Couldn't remove object of type ''{0}'' on attribute ''{1}'' from object stack.", o.getClass().getName(), stack);
397      }
398      isBottom = false;
399   }

448   /**
449    * Generalize the specified object if a filter is associated with it.
450    *
451    * @param o The object to generalize.
452    * @param type The type of object.
453    * @return The generalized object, or <jk>null</jk> if the object is <jk>null</jk>.
454    * @throws SerializeException If a problem occurred trying to convert the output.
455    */
456   @SuppressWarnings({ "rawtypes", "unchecked" })
457   public final Object generalize(Object o, ClassMeta<?> type) throws SerializeException {
458      if (o == null)
459         return null;
460      PojoFilter f = (type == null || type.isObject() ? getBeanContext().getClassMeta(o.getClass()).getPojoFilter() : type.getPojoFilter());
461      if (f == null)
462         return o;
463      return f.filter(o);
464   }

466   /**
467    * Returns <jk>true</jk> if the specified value should not be serialized.
468    *
469    * @param cm The class type of the object being serialized.
470    * @param attrName The bean attribute name, or <jk>null</jk> if this isn't a bean attribute.
471    * @param value The object being serialized.
472    * @return <jk>true</jk> if the specified value should not be serialized.
473    * @throws SerializeException
474    */
475   public final boolean canIgnoreValue(ClassMeta<?> cm, String attrName, Object value) throws SerializeException {
476
477      if (trimNulls && value == null)
478         return true;
479
480      if (value == null)
481         return false;
482
483      if (cm == null)
484         cm = getBeanContext().object();
485
486      if (trimEmptyLists) {
487         if (cm.isArray() || (cm.isObject() && value.getClass().isArray())) {
488            if (((Object[])value).length == 0)
489               return true;
490         }
491         if (cm.isCollection() || (cm.isObject() && isParentClass(Collection.class, value.getClass()))) {
492            if (((Collection<?>)value).isEmpty())
493               return true;
494         }
495      }
496
497      if (trimEmptyMaps) {
498         if (cm.isMap() || (cm.isObject() && isParentClass(Map.class, value.getClass()))) {
499            if (((Map<?,?>)value).isEmpty())
500               return true;
501         }
502      }
503
504      if (trimNulls && willRecurse(attrName, value, cm))
505         return true;
506
507      return false;
508   }

511    * Sorts the specified map if {@link SerializerContext#isSortMaps()} returns <jk>true</jk>.
512    *
513    * @param m The map being sorted.
514    * @return A new sorted {@link TreeMap}.

517      if (sortMaps && m != null && (! m.isEmpty()) && m.keySet().iterator().next() instanceof Comparable<?>)
518         return new TreeMap<K,V>(m);
519      return m;
520   }

534   /**

543
544   private static class StackElement {
545      private int depth;
546      private String name;
547      private Object o;
548      private ClassMeta<?> aType;
549
550      private StackElement(int depth, String name, Object o, ClassMeta<?> aType) {
551         this.depth = depth;
552         this.name = name;
553         this.o = o;
554         this.aType = aType;
555      }
556
557      private String toString(boolean simple) {
558         StringBuilder sb = new StringBuilder().append('[').append(depth).append(']');
559         sb.append(StringUtils.isEmpty(name) ? "<noname>" : name).append(':');
560         sb.append(aType.toString(simple));
561         if (aType != aType.getFilteredClassMeta())
562            sb.append('/').append(aType.getFilteredClassMeta().toString(simple));
563         return sb.toString();
564      }
565   }
566
567   private String getStack(boolean full) {
568      StringBuilder sb = new StringBuilder();
569      for (StackElement e : stack) {
570         if (full) {
571            sb.append("\n\t");
572            for (int i = 1; i < e.depth; i++)
573               sb.append("  ");
574            if (e.depth > 0)
575               sb.append("->");
576            sb.append(e.toString(false));
577         } else {
578            sb.append(" > ").append(e.toString(true));
579         }
580      }
581      return sb.toString();
582   }
