009package com.ibm.juno.core;
010
011/**
012 * Superclass of all classes that have a locked state.
013 * <p>
014 *    Used to mark bean contexts, serializers, and parsers as read-only so that
015 *    settings can no longer be modified.
016 * <p>
017 *    Also keeps track of when the object has been cloned and allows for lazy cloning through
018 *    the {@link #onUnclone()} method.  The idea behind this is that certain expensive fields don't
019 *    need to be cloned unless the object is actually being modified.
020 * <p>
021 *    Calling {@link #lock()} on the object causes it to be put into a read-only state.
022 *    Once called, subsequent calls to {@link #checkLock()} will cause {@link LockedException LockedExceptions}
023 *       to be thrown.
024 * <p>
025 *    As a rule, cloned objects are unlocked by default.
026 *
027 * @author James Bognar (jbognar@us.ibm.com)
028 */
029public abstract class Lockable implements Cloneable {
030
031   private boolean isLocked = false;
032   private boolean isCloned = false;
033
034   /**
035    * Locks this object so that settings on it cannot be modified.
036    *
037    * @return This object (for method chaining).
038    */
039   public Lockable lock() {
040      isLocked = true;
041      return this;
042   }
043
044   /**
045    * @return <code><jk>true</jk></code> if this object has been locked.
046    */
047   public boolean isLocked() {
048      return isLocked;
049   }
050
051   /**
052    * Causes a {@link LockedException} to be thrown if this object has been locked.
053    * <p>
054    *    Also calls {@link #onUnclone()} if this is the first time this method has been called since cloning.
055    *
056    * @throws LockedException If {@link #lock()} has been called on this object.
057    */
058   public void checkLock() throws LockedException {
059      if (isLocked)
060         throw new LockedException();
061      if (isCloned)
062         onUnclone();
063      isCloned = false;
064   }
065
066   /**
067    * Subclass can override this method to handle lazy-cloning on the first time {@link #checkLock()} is called after
068    * the object has been cloned.
069    */
070   public void onUnclone() {}
071
072   /**
073    * Creates an unlocked clone of this object.
074    *
075    * @throws CloneNotSupportedException If class cannot be cloned.
076    */
077   @Override /* Object */
078   public Lockable clone() throws CloneNotSupportedException {
079      Lockable c = (Lockable)super.clone();
080      c.isLocked = false;
081      c.isCloned = true;
082      return c;
083   }
084}
