009package com.ibm.juno.core;
010
011import java.util.*;
012
013import com.ibm.juno.core.annotation.*;
014import com.ibm.juno.core.xml.*;
015
016/**
017 * Sames as {@link BeanMeta}, except the list of bean properties are limited
018 * by a {@link BeanProperty#properties()} annotation.
019 *
020 * @param <T> The class type that this metadata applies to.
021 * @author James Bognar (jbognar@us.ibm.com)
022 */
023public final class BeanMetaFiltered<T> extends BeanMeta<T> {
024
025   private final BeanMeta<T> innerMeta;
026
027   /**
028    * Wrapper constructor.
029    *
030    * @param innerMeta The unfiltered bean meta of the bean property.
031    * @param pNames The list of filtered property names.
032    */
033   public BeanMetaFiltered(BeanMeta<T> innerMeta, String[] pNames) {
034      this.innerMeta = innerMeta;
035      this.properties = new LinkedHashMap<String,BeanPropertyMeta<T>>();
036      for (String p : pNames)
037         properties.put(p, innerMeta.getPropertyMeta(p));
038      this.xmlMeta = new XmlBeanMeta<T>(innerMeta, pNames);
039   }
040
041   /**
042    * Wrapper constructor.
043    *
044    * @param innerMeta The unfiltered bean meta of the bean property.
045    * @param pNames The list of filtered property names.
046    */
047   public BeanMetaFiltered(BeanMeta<T> innerMeta, Collection<String> pNames) {
048      this(innerMeta, pNames.toArray(new String[pNames.size()]));
049   }
050
051   @Override /* Delagate */
052   public ClassMeta<T> getClassMeta() {
053      return innerMeta.classMeta;
054   }
055
056   @Override /* BeanMeta */
057   public Collection<BeanPropertyMeta<T>> getPropertyMetas() {
058      return properties.values();
059   }
060
061   @Override /* BeanMeta */
062   public BeanPropertyMeta<T> getPropertyMeta(String name) {
063      return properties.get(name);
064   }
065
066   @Override /* Object */
067   public String toString() {
068      return innerMeta.c.getName();
069   }
070}
