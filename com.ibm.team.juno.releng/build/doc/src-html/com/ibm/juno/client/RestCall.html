031/**
032 * Represents a connection to a remote REST resource.
033 * <p>
034 *    Instances of this class are created by the various {@code doX()} methods on the {@link RestClient} class.
035 * <p>
036 *    This class uses only Java standard APIs.  Requests can be built up using a fluent interface with method chaining, like so...
037 *
038 * <p class='bcode'>
039 *    RestClient client = <jk>new</jk> RestClient();
040 *    RestCall c = client.doPost(<jsf>URL</jsf>).setInput(o).setHeader(x,y);
041 *    MyBean b = c.getResponse(MyBean.<jk>class</jk>);
042 * </p>
043 * <p>
044 *    The actual connection and request/response transaction occurs when calling one of the <code>getResponseXXX()</code> methods.
045 *
046 * <h6 class='topic'>Additional Information</h6>
047 * <ul>
048 *    <li><a class='doclink' href='package-summary.html#RestClient'>com.ibm.juno.client &gt; REST client API</a> for more information and code examples.
049 * </ul>
050 *
051 * @author James Bognar (jbognar@us.ibm.com)
052 */

082   protected RestCall(RestClient client, HttpRequestBase request) throws RestCallException {

089   /**
090    * Sets the input for this REST call.
091    *
092    * @param input The input to be sent to the REST resource (only valid for PUT and POST) requests. <br>
093    *    Can be of the following types:
094    *    <ul>
095    *       <li>{@link Reader} - Raw contents of {@code Reader} will be serialized to remote resource.
096    *       <li>{@link InputStream} - Raw contents of {@code InputStream} will be serialized to remote resource.
097    *       <li>{@link Object} - POJO to be converted to text using the {@link Serializer} registered with the {@link RestClient}.
098    *       <li>{@link HttpEntity} - Bypass Juno serialization and pass HttpEntity directly to HttpClient.
099    *    </ul>
100    * @return This object (for method chaining).
101    * @throws RestCallException If a retry was attempted, but the entity was not repeatable.
102    */
103   public RestCall setInput(final Object input) throws RestCallException {
104      if (! (request instanceof HttpEntityEnclosingRequestBase))
105         throw new RestCallException(0, "Method does not support content entity.", request.getMethod(), request.getURI(), null);
106      HttpEntity entity = (input instanceof HttpEntity ? (HttpEntity)input : new RestRequestEntity(input, client.serializer));
107      ((HttpEntityEnclosingRequestBase)request).setEntity(entity);
108      if (retries > 1 && ! entity.isRepeatable())
109         throw new RestCallException("Rest call set to retryable, but entity is not repeatable.");
110      return this;
111   }

127   /**
128    * Make this call retryable if an error response (>=400) is received.
129    *
130    * @param retries The number of retries to attempt.
131    * @param interval The time in milliseconds between attempts.
132    * @param retryOn Optional object used for determining whether a retry should be attempted.
133    *    If <jk>null</jk>, uses {@link RetryOn#DEFAULT}.
134    * @return This object (for method chaining).
135    * @throws RestCallException If current entity is not repeatable.
136    */
137   public RestCall setRetryable(int retries, long interval, RetryOn retryOn) throws RestCallException {
138      if (request instanceof HttpEntityEnclosingRequestBase) {
139      HttpEntity e = ((HttpEntityEnclosingRequestBase)request).getEntity();
140      if (e != null && ! e.isRepeatable())
141         throw new RestCallException("Attempt to make call retryable, but entity is not repeatable.");
142      }
143      this.retries = retries;
144      this.retryInterval = interval;
145      this.retryOn = (retryOn == null ? RetryOn.DEFAULT : retryOn);
146      return this;
147

449   /**
450    * Method used to execute an HTTP response where you're only interested in the HTTP response code.
451    * <p>
452    * The response entity is discarded unless one of the pipe methods have been specified to pipe the
453    *     output to an output stream or writer.
454    *
455    * <dl>
456    *    <dt>Example:</dt>
457    *    <dd>
458    * <p class='bcode'>
459    *    <jk>try</jk> {
460    *       RestClient client = <jk>new</jk> RestClient();
461    *       <jk>int</jk> rc = client.doGet(url).execute();
462    *       <jc>// Succeeded!</jc>
463    *    } <jk>catch</jk> (RestCallException e) {
464    *       <jc>// Failed!</jc>
465    *    }
466    * </p>
467    *    </dd>
468    * </dl>
469    *
470    * @return This object (for method chaining).
471    * @throws RestCallException If an exception or non-200 response code occurred during the connection attempt.
472    */
473   public int run() throws RestCallException {
474      connect();
475      try {
476         StatusLine status = response.getStatusLine();
477         int sc = status.getStatusCode();
478         if (sc >= 400 && ! ignoreErrors)
479            throw new RestCallException(sc, status.getReasonPhrase(), request.getMethod(), request.getURI(), getResponseAsString()).setHttpResponse(response);
480         if (outputStreams.size() > 0 || writers.size() > 0)
481            getReader();
482         return sc;
483      } catch (RestCallException e) {
484         isFailed = true;
485         throw e;
486      } catch (IOException e) {
487         isFailed = true;
488         throw new RestCallException(e).setHttpResponse(response);
489      } finally {
490         close();
491      }
492   }

494   /**
495    * Connects to the REST resource.
496    * <p>
497    *    If this is a <code>PUT</code> or <code>POST</code>, also sends the input to the remote resource.<br>
498    * <p>
499    *    Typically, you would only call this method if you're not interested in retrieving the body of the HTTP response.
500    *    Otherwise, you're better off just calling one of the {@link #getReader()}/{@link #getResponse(Class)}/{@link #pipeTo(Writer)}
501    *    methods directly which automatically call this method already.
502    *
503    * @return This object (for method chaining).
504    * @throws RestCallException If an exception or <code>400+</code> HTTP status code occurred during the connection attempt.
505    */
506   public RestCall connect() throws RestCallException {
507
508      if (isConnected)
509         return this;
510      isConnected = true;
511
512      try {
513         int sc = 0;
514         while (retries > 0) {
515            retries--;
516            Exception ex = null;
517            try {
518         response = client.execute(request);
519               sc = (response == null || response.getStatusLine() == null) ? -1 : response.getStatusLine().getStatusCode();
520            } catch (Exception e) {
521               ex = e;
522               sc = -1;
523               if (response != null)
524                  EntityUtils.consumeQuietly(response.getEntity());
525            }
526            if (! retryOn.onCode(sc))
527               retries = 0;
528            if (retries > 0) {
529               for (RestCallInterceptor rci : interceptors)
530                  rci.onRetry(this, sc, request, response, ex);
531               request.reset();
532               long w = retryInterval;
533               synchronized(this) {
534                  wait(w);
535               }
536            } else if (ex != null) {
537               throw ex;
538            }
539         }
540         for (RestCallInterceptor rci : interceptors)
541            rci.onConnect(this, sc, request, response);
542         if (response == null)
543            throw new RestCallException("HttpClient returned a null response");
544         StatusLine sl = response.getStatusLine();
545         String method = request.getMethod();
546         sc = sl.getStatusCode(); // Read it again in case it was changed by one of the interceptors.
547         if (sc >= 400 && ! ignoreErrors)
548            throw new RestCallException(sc, sl.getReasonPhrase(), method, request.getURI(), getResponseAsString()).setHttpResponse(response);
549         if ((sc == 307 || sc == 302) && allowRedirectsOnPosts && method.equalsIgnoreCase("POST")) {
550            if (redirectOnPostsTries-- < 1)
551               throw new RestCallException(sc, "Maximum number of redirects occurred.  Location header: " + response.getFirstHeader("Location"), method, request.getURI(), getResponseAsString());
552            Header h = response.getFirstHeader("Location");
553            if (h != null) {
554               reset();
555               request.setURI(URI.create(h.getValue()));
556               retries++;  // Redirects should affect retries.
557               connect();
558            }
559         }
560
561      } catch (RestCallException e) {
562         isFailed = true;
563         try {
564         close();
565         } catch (RestCallException e2) { /* Ignore */ }
566         throw e;
567      } catch (Exception e) {
568         isFailed = true;
569         close();
570         throw new RestCallException(e).setHttpResponse(response);
571      }
572
573      return this;
574   }
575
576   private void reset() {
577      if (response != null)
578         EntityUtils.consumeQuietly(response.getEntity());
579      request.reset();
580      isConnected = false;
581      isClosed = false;
582      isFailed = false;
583      if (capturedResponseWriter != null)
584         capturedResponseWriter.getBuffer().setLength(0);
585   }

587   /**
588    * Connects to the remote resource (if <code>connect()</code> hasn't already been called) and returns the HTTP response message body as a reader.
589    * <p>
590    *    If an {@link Encoder} has been registered with the {@link RestClient}, then the underlying input stream
591    *       will be wrapped in the encoded stream (e.g. a <code>GZIPInputStream</code>).
592    * <p>
593    *    If present, automatically handles the <code>charset</code> value in the <code>Content-Type</code> response header.
594    * <p>
595    *    <b>IMPORTANT:</b>  It is your responsibility to close this reader once you have finished with it.
596    *
597    * @return The HTTP response message body reader.  <jk>null</jk> if response was successful but didn't contain a body (e.g. HTTP 204).
598    * @throws IOException If an exception occurred while streaming was already occurring.
599    */
600   public Reader getReader() throws IOException {
601      InputStream is = getInputStream();
602      if (is == null)
603         return null;
604
605      // Figure out what the charset of the response is.
606      String cs = null;
607      Header contentType = response.getLastHeader("Content-Type");
608      String ct = contentType == null ? null : contentType.getValue();
609
610      // First look for "charset=" in Content-Type header of response.
611      if (ct != null && ct.contains("charset="))
612         cs = ct.substring(ct.indexOf("charset=")+8).trim();
613
614      if (cs == null)
615         cs = "UTF-8";
616
617      Reader isr = new InputStreamReader(is, cs);
618
619      if (writers.size() > 0) {
620         StringWriter sw = new StringWriter();
621         writers.add(sw, true);
622         IOPipe.create(isr, writers).byLines(byLines).run();
623         return new StringReader(sw.toString());
624      }
625
626      return new InputStreamReader(is, cs);
627   }

629   /**
630    * Returns the response text as a string if {@link #captureResponse()} was called on this object.
631    * <p>
632    * Note that while similar to {@link #getResponseAsString()}, this method can be called multiple times
633    *    to retrieve the response text multiple times.
634    * <p>
635    * Note that this method returns <jk>null</jk> if you have not called one of the methods that cause
636    *    the response to be processed.  (e.g. {@link #run()}, {@link #getResponse()}, {@link #getResponseAsString()}.
637    *
638    * @return The captured response, or <jk>null</jk> if {@link #captureResponse()} has not been called.
639    * @throws IllegalStateException If trying to call this method before the response is consumed.
640    */
641   public String getCapturedResponse() {
642      if (! isClosed)
643         throw new IllegalStateException("This method cannot be called until the response has been consumed.");
644      if (capturedResponse == null && capturedResponseWriter != null && capturedResponseWriter.getBuffer().length() > 0)
645         capturedResponse = capturedResponseWriter.toString();
646      return capturedResponse;
647   }

649   /**
650    * Returns the parser specified on the client to use for parsing HTTP response bodies.
651    *
652    * @return The parser.
653    * @throws RestCallException If no parser was defined on the client.
654    */
655   protected Parser<?> getParser() throws RestCallException {
656      if (client.parser == null)
657         throw new RestCallException(0, "No parser defined on client", request.getMethod(), request.getURI(), null);

674    * Returns the value of the <code>Content-Length</code> header.
675    *
676    * @return The value of the <code>Content-Length</code> header, or <code>-1</code> if header is not present.
677    * @throws IOException
678    */
679   public int getContentLength() throws IOException {
680      connect();
681      Header h = response.getLastHeader("Content-Length");
682      if (h == null)
683         return -1;
684      long l = Long.parseLong(h.getValue());

686         return Integer.MAX_VALUE;
687      return (int)l;

690   /**
691    * Connects to the remote resource (if <code>connect()</code> hasn't already been called) and returns the HTTP response message body as an input stream.
692    * <p>
693    *    If an {@link Encoder} has been registered with the {@link RestClient}, then the underlying input stream
694    *       will be wrapped in the encoded stream (e.g. a <code>GZIPInputStream</code>).
695    * <p>
696    *    <b>IMPORTANT:</b>  It is your responsibility to close this reader once you have finished with it.
697    *
698    * @return The HTTP response message body input stream. <jk>null</jk> if response was successful but didn't contain a body (e.g. HTTP 204).
699    * @throws IOException If an exception occurred while streaming was already occurring.
700    * @throws IllegalStateException If an attempt is made to read the response more than once.
701    */
702   public InputStream getInputStream() throws IOException {
703      if (isClosed)
704         throw new IllegalStateException("Method cannot be called.  Response has already been consumed.");
705      connect();
706      if (response == null)
707         throw new RestCallException("Response was null");
708      if (response.getEntity() == null)  // HTTP 204 results in no content.
709         return null;
710      InputStream is = response.getEntity().getContent();
711
712      if (outputStreams.size() > 0) {
713         ByteArrayInOutStream baios = new ByteArrayInOutStream();
714         outputStreams.add(baios, true);
715         IOPipe.create(is, baios).run();
716         return baios.getInputStream();
717      }
718      return is;
719   }

847   <T> T getResponse(ClassMeta<T> type) throws IOException, ParseException {
848      try {
849      Parser<?> p = getParser();
850      T o = null;
851      int contentLength = getContentLength();
852         if (! p.isReaderParser()) {
853         InputStream is = getInputStream();
854         o = ((InputStreamParser)p).parse(is, contentLength, type);
855      } else {
856         Reader r = getReader();
857         o = ((ReaderParser)p).parse(r, contentLength, type);
858         }
859      return o;
860      } catch (ParseException e) {
861         isFailed = true;
862         throw e;
863      } catch (IOException e) {
864         isFailed = true;
865         throw e;
866      } finally {
867         close();
868      }
869   }

924      if (response != null)
