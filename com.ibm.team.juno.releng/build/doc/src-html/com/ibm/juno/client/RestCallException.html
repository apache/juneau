009package com.ibm.juno.client;
010
011import static java.lang.String.*;
012
013import java.io.*;
014import java.net.*;
015import java.util.regex.*;
016
017import org.apache.http.*;
018import org.apache.http.client.*;
019import org.apache.http.util.*;
020
021import com.ibm.juno.core.utils.*;

023/**
024 * Exception representing a <code>400+</code> HTTP response code against a remote resource.
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028public final class RestCallException extends IOException {
029
030   private static final long serialVersionUID = 1L;
031
032   private int responseCode;
033   private String response, responseStatusMessage;
034   HttpResponseException e;
035   private HttpResponse httpResponse;
036
037
038   /**
039    * Constructor.
040    *
041    * @param msg The exception message.
042    */
043   public RestCallException(String msg) {
044      super(msg);
045   }
046
047   /**
048    * Constructor.
049    *
050    * @param e The inner cause of the exception.
051    */
052   public RestCallException(Exception e) {
053      super(e.getLocalizedMessage(), e);
054      if (e instanceof FileNotFoundException) {
055         responseCode = 404;
056      } else if (e.getMessage() != null) {
057         Pattern p = Pattern.compile("[^\\d](\\d{3})[^\\d]");
058         Matcher m = p.matcher(e.getMessage());
059         if (m.find())
060            responseCode = Integer.parseInt(m.group(1));
061      }
062      setStackTrace(e.getStackTrace());
063   }
064
065   /**
066    * Create an exception with a simple message and the status code and body of the specified response.
067    *
068    * @param msg The exception message.
069    * @param response The HTTP response object.
070    * @throws ParseException
071    * @throws IOException
072    */
073   public RestCallException(String msg, HttpResponse response) throws ParseException, IOException {
074      super(format("%s%nstatus='%s'%nResponse: %n%s%n", msg, response.getStatusLine().getStatusCode(), EntityUtils.toString(response.getEntity(), IOUtils.UTF8)));
075   }
076
077   /**
078    * Constructor.
079    *
080    * @param responseCode The response code.
081    * @param responseMsg The response message.
082    * @param method The HTTP method (for message purposes).
083    * @param url The HTTP URL (for message purposes).
084    * @param response The reponse from the server.
085    */
086   public RestCallException(int responseCode, String responseMsg, String method, URI url, String response) {
087      super(format("HTTP method '%s' call to '%s' caused response code '%s,%s'.%nResponse: %n%s%n", method, url, responseCode, responseMsg, response));
088      this.responseCode = responseCode;
089      this.responseStatusMessage = responseMsg;
090      this.response = response;
091   }
092
093   /**
094    * Sets the HTTP response object that caused this exception.
095    *
096    * @param httpResponse The HTTP respose object.
097    * @return This object (for method chaining).
098    */
099   protected RestCallException setHttpResponse(HttpResponse httpResponse) {
100      this.httpResponse = httpResponse;
101      return this;
102   }
103
104   /**
105    * Returns the HTTP response object that caused this exception.
106    *
107    * @return The HTTP response object that caused this exception, or <jk>null</jk> if no response was created yet when the exception was thrown.
108    */
109   public HttpResponse getHttpResponse() {
110      return this.httpResponse;
111   }
112
113   /**
114    * Returns the HTTP response status code.
115    *
116    * @return The response status code.  If a connection could not be made at all, returns <code>0<code>.
117    */
118   public int getResponseCode() {
119      return responseCode;
120   }
121
122   /**
123    * Returns the HTTP response message body text.
124    *
125    * @return The response message body text.
126    */
127   public String getResponseMessage() {
128      return response;
129   }
130
131   /**
132    * Returns the response status message as a plain string.
133    *
134    * @return The response status message.
135    */
136   public String getResponseStatusMessage() {
137      return responseStatusMessage;
138   }
139
140   /**
141    * Sets the inner cause for this exception.
142    * @param cause The inner cause.
143    * @return This object (for method chaining).
144    */
145   @Override /* Throwable */
146   public synchronized RestCallException initCause(Throwable cause) {
147      super.initCause(cause);
148      return this;
149   }
150}
