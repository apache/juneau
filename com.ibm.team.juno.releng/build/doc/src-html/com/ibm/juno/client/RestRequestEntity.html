009package com.ibm.juno.client;
010
011import java.io.*;
012
013import org.apache.http.entity.*;
014import org.apache.http.message.*;
015
016import com.ibm.juno.core.serializer.*;
017import com.ibm.juno.core.utils.*;
018
019/**
020 * HttpEntity for serializing POJOs as the body of HTTP requests.
021 *
022 * @author James Bognar (jbognar@us.ibm.com)
023 */
024public final class RestRequestEntity extends BasicHttpEntity {
025   final Object output;
026   final Serializer<?> serializer;
027   byte[] outputBytes;
028
029   /**
030    * Constructor.
031    * @param input The POJO to serialize.  Can also be a {@link Reader} or {@link InputStream}.
032    * @param serializer The serializer to use to serialize this response.
033    */
034   public RestRequestEntity(Object input, Serializer<?> serializer) {
035      this.output = input;
036      this.serializer = serializer;
037      if (serializer != null)
038         setContentType(new BasicHeader("Content-Type", serializer.getResponseContentType()));
039   }
040
041   @Override /* BasicHttpEntity */
042   public void writeTo(OutputStream os) throws IOException {
043      if (output instanceof InputStream) {
044         IOPipe.create(output, os).closeOut().run();
045      } else if (output instanceof Reader) {
046         IOPipe.create(output, new OutputStreamWriter(os, IOUtils.UTF8)).closeOut().run();
047      } else {
048         try {
049            if (serializer == null) {
050               // If no serializer specified, just close the stream.
051               os.close();
052            } else if (! serializer.isWriterSerializer()) {
053               OutputStreamSerializer s2 = (OutputStreamSerializer)serializer;
054               s2.serialize(output, os);
055               os.close();
056            } else {
057               Writer w = new OutputStreamWriter(os, IOUtils.UTF8);
058               WriterSerializer s2 = (WriterSerializer)serializer;
059               s2.serialize(output, w);
060               w.close();
061            }
062         } catch (SerializeException e) {
063            throw new com.ibm.juno.client.RestCallException(e);
064         }
065      }
066   }
067
068   @Override /* BasicHttpEntity */
069   public boolean isRepeatable() {
070      return true;
071   }
072
073   @Override /* BasicHttpEntity */
074   public InputStream getContent() {
075      if (outputBytes == null) {
076         ByteArrayOutputStream baos = new ByteArrayOutputStream();
077         try {
078            writeTo(baos);
079            outputBytes = baos.toByteArray();
080         } catch (IOException e) {
081            throw new RuntimeException(e);
082         }
083      }
084      return new ByteArrayInputStream(outputBytes);
085   }
086}
