009package com.ibm.juno.client;
010
011import java.io.*;
012import java.util.regex.*;
013
014/**
015 * Used to find regular expression matches in REST responses made through {@link RestCall}.
016 * <p>
017 * Response patterns are applied to REST calls through the {@link RestCall#addResponsePattern(ResponsePattern)} method.
018 * <p>
019 * <h6 class='topic'>Example</h6>
020 * This example shows how to use a response pattern finder to find and capture patterns for <js>"x=number"</js> and <js>"y=string"</js>
021 *    from a response body.
022 * <p>
023 * <p class='bcode'>
024 *    <jk>final</jk> List&lt;Number&gt; xList = <jk>new</jk> ArrayList&lt;Number&gt;();
025 *    <jk>final</jk> List&lt;String&gt; yList = <jk>new</jk> ArrayList&lt;String&gt;();
026 *
027 *    restClient.doGet(<jsf>URL</jsf>)
028 *       .addResponsePattern(
029 *          <jk>new</jk> ResponsePattern(<js>"x=(\\d+)"</js>) {
030 *             <ja>@Override</ja>
031 *             <jk>public void</jk> onMatch(RestCall restCall, Matcher m) <jk>throws</jk> RestCallException {
032 *                xList.add(Integer.<jsm>parseInt</jsm>(m.group(1)));
033 *             }
034 *             <ja>@Override</ja>
035 *             <jk>public void</jk> onNoMatch(RestCall restCall) <jk>throws</jk> RestCallException {
036 *                <jk>throw new</jk> RestCallException(<js>"No X's found!"</js>);
037 *             }
038 *          }
039 *       )
040 *       .addResponsePattern(
041 *          <jk>new</jk> ResponsePattern(<js>"y=(\\S+)"</js>) {
042 *             <ja>@Override</ja>
043 *             <jk>public void</jk> onMatch(RestCall restCall, Matcher m) <jk>throws</jk> RestCallException {
044 *                yList.add(m.group(1));
045 *             }
046 *             <ja>@Override</ja>
047 *             <jk>public void</jk> onNoMatch(RestCall restCall) <jk>throws</jk> RestCallException {
048 *                <jk>throw new</jk> RestCallException(<js>"No Y's found!"</js>);
049 *             }
050 *          }
051 *       )
052 *       .run();
053 * </p>
054 * <p>
055 * <h5 class='notes'>Important Notes:</h5>
056 * <ol class='notes'>
057 *    <li><p>
058 *       Using response patterns does not affect the functionality of any of the other methods
059 *       used to retrieve the response such as {@link RestCall#getResponseAsString()} or {@link RestCall#getResponse(Class)}.<br>
060 *       HOWEVER, if you want to retrieve the entire text of the response from inside the match methods,
061 *       use {@link RestCall#getCapturedResponse()} since this method will not absorb the response for those other methods.
062 *    </p>
063 *    <li><p>
064 *       Response pattern methods are NOT executed if a REST exception occurs during the request.
065 *    </p>
066 *    <li><p>
067 *       The {@link RestCall#successPattern(String)} and {@link RestCall#failurePattern(String)} methods use instances of
068 *       this class to throw {@link RestCallException RestCallExceptions} when success patterns are not found or failure patterns
069 *       are found.
070 *    </p>
071 *    <li><p>
072 *       {@link ResponsePattern} objects are reusable and thread-safe.
073 *    </p>
074 * </ol>
075 *
076 * @author James Bognar (jbognar@us.ibm.com)
077 */
078public abstract class ResponsePattern {
079
080   private Pattern pattern;
081
082   /**
083    * Constructor.
084    *
085    * @param pattern Regular expression pattern.
086    */
087   public ResponsePattern(String pattern) {
088      this.pattern = Pattern.compile(pattern);
089   }
090
091   void match(RestCall rc) throws RestCallException {
092      try {
093         Matcher m = pattern.matcher(rc.getCapturedResponse());
094         boolean found = false;
095         while (m.find()) {
096            onMatch(rc, m);
097            found = true;
098         }
099         if (! found)
100            onNoMatch(rc);
101      } catch (IOException e) {
102         throw new RestCallException(e);
103      }
104   }
105
106   /**
107    * Returns the pattern passed in through the constructor.
108    *
109    * @return The pattern passed in through the constructor.
110    */
111   protected String getPattern() {
112      return pattern.pattern();
113   }
114
115   /**
116    * Instances can override this method to handle when a regular expression pattern matches
117    *    on the output.
118    * <p>
119    * This method is called once for every pattern match that occurs in the response text.
120    *
121    * @param rc The {@link RestCall} that this pattern finder is being used on.
122    * @param m The regular expression {@link Matcher}.  Can be used to retrieve group matches in the pattern.
123    * @throws RestCallException Instances can throw an exception if a failure condition is detected.
124    */
125   public void onMatch(RestCall rc, Matcher m) throws RestCallException {}
126
127   /**
128    * Instances can override this method to handle when a regular expression pattern doesn't match on the output.
129    *
130    * @param rc The {@link RestCall} that this pattern finder is being used on.
131    * @throws RestCallException Instances can throw an exception if a failure condition is detected.
132    */
133   public void onNoMatch(RestCall rc) throws RestCallException {}
134}
