054 *
055 * <h6 class='topic'>Features</h6>

202   /**
203    * Creates an instance of an {@link HttpClient} to be used to handle all HTTP communications with the target server.
204    * <p>
205    * This HTTP client is used when the HTTP client is not specified through one of the constructors or the
206    *    {@link #setHttpClient(CloseableHttpClient)} method.
207    * <p>
208    * Subclasses can override this method to provide specially-configured HTTP clients to handle
209    *    stuff such as SSL/TLS certificate handling, authentication, etc.
210    * <p>
211    * The default implementation returns an instance of {@link HttpClient} using the client builder
212    *    returned by {@link #createHttpClientBuilder()}.
213    *
214    * @return The HTTP client to use.
215    * @throws Exception
216    */
217   protected CloseableHttpClient createHttpClient() throws Exception {
218      // Don't call createConnectionManager() if RestClient.setConnectionManager() was called.
219      if (httpClientConnectionManager == null)
220         httpClientBuilder.setConnectionManager(createConnectionManager());
221      return httpClientBuilder.build();
222   }

224   /**
225    * Creates an instance of an {@link HttpClientBuilder} to be used to create
226    *    the {@link HttpClient}.
227    * <p>
228    *    Subclasses can override this method to provide their own client builder.
229    * </p>
230    * <p>
231    *    The predefined method returns an {@link HttpClientBuilder} with the following settings:
232    * </p>
233    * <ul>
234    *    <li>Lax redirect strategy.
235    *    <li>The connection manager returned by {@link #createConnectionManager()}.
236    * </ul>
237    *
238    * @return The HTTP client builder to use to create the HTTP client.
239    */
240   protected HttpClientBuilder createHttpClientBuilder() {
241      HttpClientBuilder b = HttpClientBuilder.create();
242      b.setRedirectStrategy(new AllowAllRedirects());
243      return b;
244   }

246   /**
247    * Creates the {@link HttpClientConnectionManager} returned by {@link #createConnectionManager()}.
248    * <p>
249    *    Subclasses can override this method to provide their own connection manager.
250    * </p>
251    * <p>
252    *    The default implementation returns an instance of a {@link PoolingHttpClientConnectionManager}.
253    * </p>
254    *
255    * @return The HTTP client builder to use to create the HTTP client.
256    */
257   @SuppressWarnings("resource")
258   protected HttpClientConnectionManager createConnectionManager() {
259      if (sslOpts != null) {
260         HostnameVerifier hv = null;
261         switch (sslOpts.getHostVerify()) {
262            case LAX: hv = new NoopHostnameVerifier(); break;
263            case DEFAULT: hv = new DefaultHostnameVerifier(); break;
264         }
265
266         for (String p : StringUtils.split(sslOpts.getProtocols(), ',')) {
267            try {
268               TrustManager tm = new SimpleX509TrustManager(sslOpts.getCertValidate() == SSLOpts.CertValidate.LAX);
269
270               SSLContext ctx = SSLContext.getInstance(p);
271               ctx.init(null, new TrustManager[] { tm }, null);
272
273               // Create a socket to ensure this algorithm is acceptable.
274               // This will correctly disallow certain configurations (such as SSL_TLS under FIPS)
275               ctx.getSocketFactory().createSocket().close();
276               SSLConnectionSocketFactory sf = new SSLConnectionSocketFactory(ctx, hv);
277               setSSLSocketFactory(sf);
278
279               Registry<ConnectionSocketFactory> r = RegistryBuilder.<ConnectionSocketFactory> create().register("https", sf).build();
280
281               return (pooled ? new PoolingHttpClientConnectionManager(r) : new BasicHttpClientConnectionManager(r));
282            } catch (Throwable t) {}
283         }
284      }
285
286         // Using pooling connection so that this client is threadsafe.
287      return (pooled ? new PoolingHttpClientConnectionManager() : new BasicHttpClientConnectionManager());
288   }

291    * Set up this client to use BASIC auth.
292    *
293    * @param host The auth scope hostname.
294    * @param port The auth scope port.
295    * @param user The username.
296    * @param pw The password.
297    * @return This object (for method chaining).
298    */
299   public RestClient setBasicAuth(String host, int port, String user, String pw) {
300      AuthScope scope = new AuthScope(host, port);
301      Credentials up = new UsernamePasswordCredentials(user, pw);
302      CredentialsProvider p = new BasicCredentialsProvider();
303      p.setCredentials(scope, up);
304      setDefaultCredentialsProvider(p);
305      return this;

659    * Performs a REST call where the entire call is specified in a simple string.
660    * <p>
661    * This method is useful for performing callbacks when the target of a callback is passed in
662    * on an initial request, for example to signal when a long-running process has completed.
663    * <p>
664    * The call string can be any of the following formats:
665    * <ul>
666    *    <li><js>"[method] [url]"</js> - e.g. <js>"GET http://localhost/callback"</js>
667    *    <li><js>"[method] [url] [payload]"</js> - e.g. <js>"POST http://localhost/callback some text payload"</js>
668    *    <li><js>"[method] [headers] [url] [payload]"</js> - e.g. <js>"POST {'Content-Type':'text/json'} http://localhost/callback {'some':'json'}"</js>
669    * </ul>
670    * <p>
671    * The payload will always be sent using a simple {@link StringEntity}.
672    *
673    * @param callString The call string.
674    * @return A {@link RestCall} object that can be further tailored before executing the request
675    *    and getting the response as a parsed object.
676    * @throws RestCallException
677    */
678   public RestCall doCallback(String callString) throws RestCallException {
679      String s = callString;
680      try {
681         RestCall rc = null;
682         String method = null, uri = null, content = null;
683         ObjectMap h = null;
684         int i = s.indexOf(' ');
685         if (i != -1) {
686            method = s.substring(0, i).trim();
687            s = s.substring(i).trim();
688            if (s.length() > 0) {
689               if (s.charAt(0) == '{') {
690                  i = s.indexOf('}');
691                  if (i != -1) {
692                     String json = s.substring(0, i+1);
693                     h = JsonParser.DEFAULT.parse(json, ObjectMap.class);
694                     s = s.substring(i+1).trim();
695                  }
696               }
697               if (s.length() > 0) {
698                  i = s.indexOf(' ');
699                  if (i == -1)
700                     uri = s;
701                  else {
702                     uri = s.substring(0, i).trim();
703                     s = s.substring(i).trim();
704                     if (s.length() > 0)
705                        content = s;
706                  }
707               }
708            }
709         }
710         if (method != null && uri != null) {
711            rc = doCall(method, uri, content != null);
712            if (content != null)
713               rc.setInput(new StringEntity(content));
714            if (h != null)
715               for (Map.Entry<String,Object> e : h.entrySet())
716                  rc.setHeader(e.getKey(), e.getValue());
717            return rc;
718         }
719      } catch (Exception e) {
720         throw new RestCallException(e);
721      }
722      throw new RestCallException("Invalid format for call string.");
723   }

726    * Perform a generic REST call.
727    *
728    * @param method The HTTP method.
729    * @param url The URL of the remote REST resource.  Can be any of the following:  {@link String}, {@link URI}, {@link URL}.
730    * @param content The HTTP body content.
731    *    Can be of the following types:
732    *    <ul>
733    *       <li>{@link Reader} - Raw contents of {@code Reader} will be serialized to remote resource.
734    *       <li>{@link InputStream} - Raw contents of {@code InputStream} will be serialized to remote resource.
735    *       <li>{@link Object} - POJO to be converted to text using the {@link Serializer} registered with the {@link RestClient}.
736    *       <li>{@link HttpEntity} - Bypass Juno serialization and pass HttpEntity directly to HttpClient.
737    *    </ul>
738    *    This parameter is IGNORED if {@link HttpMethod#hasContent()} is <jk>false</jk>.
739    * @return A {@link RestCall} object that can be further tailored before executing the request
740    *    and getting the response as a parsed object.
741    * @throws RestCallException If any authentication errors occurred.
742    */
743   public RestCall doCall(HttpMethod method, Object url, Object content) throws RestCallException {
744      RestCall rc = doCall(method.name(), url, method.hasContent());
745      if (method.hasContent())
746         rc.setInput(content);
747      return rc;
748   }

764      if (hasContent) {
765         req = new HttpEntityEnclosingRequestBase() {
766            @Override /* HttpRequest */
767            public String getMethod() {
768               return methodUC;
769            }
770         };
771         restCall = new RestCall(this, req);
772         if (contentType != null)
773            restCall.setHeader("Content-Type", contentType);
774      } else {
775         req = new HttpRequestBase() {
776            @Override /* HttpRequest */
777            public String getMethod() {
778               return methodUC;
779            }
780         };
781         restCall = new RestCall(this, req);
782      }
783      try {
784         req.setURI(toURI(url));
785      } catch (URISyntaxException e) {
786         throw new RestCallException(e);
787      }
788      if (accept != null)
789         restCall.setHeader("Accept", accept);
790      for (Map.Entry<String,? extends Object> e : headers.entrySet())
791         restCall.setHeader(e.getKey(), e.getValue());
792      return restCall;

795   /**
796    * Create a new proxy interface for the specified remoteable service interface.
797    *
798    * @param interfaceClass The interface to create a proxy for.
799    * @return The new proxy interface.
800    * @throws RuntimeException If the Remotable service URI has not been specified on this
801    *    client by calling {@link #setRemoteableServletUri(String)}.
802    */
803   @SuppressWarnings("unchecked")
804   public <T> T getRemoteableProxy(final Class<T> interfaceClass) {
805      if (remoteableServletUri == null)
806         throw new RuntimeException("Remoteable service URI has not been specified.");
807      return (T)Proxy.newProxyInstance(
808         interfaceClass.getClassLoader(),
809         new Class[] { interfaceClass },
810         new InvocationHandler() {
811            @Override /* InvocationHandler */
812            public Object invoke(Object proxy, Method method, Object[] args) {
813               try {
814                  String uri = remoteableServiceUriMap.get(method);
815                  if (uri == null) {
816                     // Constructing this string each time can be time consuming, so cache it.
817                     uri = remoteableServletUri + '/' + interfaceClass.getName() + '/' + ClassUtils.getMethodSignature(method);
818                     remoteableServiceUriMap.put(method, uri);
819                  }
820                  return doPost(uri, args).getResponse(method.getReturnType());
821               } catch (Exception e) {
822                  throw new RuntimeException(e);
823               }
824            }
825      });
826   }

828   private Pattern absUrlPattern = Pattern.compile("^\\w+\\:\\/\\/.*");
829
830   private URI toURI(Object url) throws URISyntaxException {
831      assertFieldNotNull(url, "url");
832      if (url instanceof URI)
833         return (URI)url;
834      if (url instanceof URL)
835         ((URL)url).toURI();
836      String s = url.toString();
837      if (rootUrl != null && ! absUrlPattern.matcher(s).matches()) {
838         if (s.isEmpty())
839            s = rootUrl;
840         else {
841            StringBuilder sb = new StringBuilder(rootUrl);
842            if (! s.startsWith("/"))
843               sb.append('/');
844            sb.append(s);
845            s = sb.toString();
846         }
847      }
848      return new URI(s);
849   }
