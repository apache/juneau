009package com.ibm.juno.client;
010
011import com.ibm.juno.core.utils.*;
012
013/**
014 * SSL configuration options that get passed to {@link RestClient#enableSSL(SSLOpts)}.
015 *
016 * @author James Bognar (jbognar@us.ibm.com)
017 */
018public class SSLOpts {
019
020   private String protocols = getDefaultProtocols();
021   private CertValidate certValidate = CertValidate.DEFAULT;
022   private HostVerify hostVerify = HostVerify.DEFAULT;
023
024   /**
025    * Reusable SSL options for lenient SSL (no cert validation or hostname verification).
026    */
027   public static final SSLOpts LAX = new SSLOpts(null, CertValidate.LAX, HostVerify.LAX);
028
029   /**
030    * Reusable SSL options for normal SSL (default cert validation and hostname verification).
031    */
032   public static final SSLOpts DEFAULT = new SSLOpts(null, CertValidate.DEFAULT, HostVerify.DEFAULT);
033
034   /**
035    * Constructor.
036    */
037   public SSLOpts() {}
038
039   /**
040    * Constructor.
041    *
042    * @param protocols A comma-delimited list of supported SSL protocols.
043    *    If <jk>null</jk>, uses the value returned by {@link #getDefaultProtocols()}.
044    * @param certValidate Certificate validation setting.
045    * @param hostVerify Host verification setting.
046    */
047   public SSLOpts(String protocols, CertValidate certValidate, HostVerify hostVerify) {
048      if (protocols != null)
049         this.protocols = protocols;
050      this.certValidate = certValidate;
051      this.hostVerify = hostVerify;
052   }
053
054   /**
055    * Returns the default list of SSL protocols to support when the <code>protocols</code>
056    *    parameter on the constructor is <jk>null</jk>.
057    * <p>
058    * The default value is <jk>"SSL_TLS,TLS,SSL"</js> unless overridden by one of the following
059    *    system properties:
060    * <ul>
061    *    <li><js>"com.ibm.team.repository.transport.client.protocol"</js>
062    *    <li><js>"transport.client.protocol"</js>
063    * </ul>
064    * <p>
065    * Subclasses can override this method to provide their own logic for determining default supported protocols.
066    *
067    * @return The comma-delimited list of supported protocols.
068    */
069   protected String getDefaultProtocols() {
070      String sp = System.getProperty("com.ibm.team.repository.transport.client.protocol");
071      if (StringUtils.isEmpty(sp))
072         sp = System.getProperty("transport.client.protocol");
073      if (StringUtils.isEmpty(sp))
074         sp = "SSL_TLS,TLS,SSL";
075      return sp;
076   }
077
078
079   //--------------------------------------------------------------------------------
080   // Bean properties
081   //--------------------------------------------------------------------------------
082
083   /**
084    * Bean property getter:  <property>protocols</property>.
085    *
086    * @return The value of the <property>protocols</property> property on this bean, or <jk>null</jk> if it is not set.
087    */
088   public String getProtocols() {
089      return protocols;
090   }
091
092   /**
093    * Bean property setter:  <property>protocols</property>.
094    *
095    * @param protocols The new value for the <property>properties</property> property on this bean.
096    * @return This object (for method chaining).
097    */
098   public SSLOpts setProtocols(String protocols) {
099      this.protocols = protocols;
100      return this;
101   }
102
103   /**
104    * Bean property getter:  <property>certValidate</property>.
105    *
106    * @return The value of the <property>certValidate</property> property on this bean, or <jk>null</jk> if it is not set.
107    */
108   public CertValidate getCertValidate() {
109      return certValidate;
110   }
111
112   /**
113    * Bean property setter:  <property>certValidate</property>.
114    *
115    * @param certValidate The new value for the <property>properties</property> property on this bean.
116    * @return This object (for method chaining).
117    */
118   public SSLOpts setCertValidate(CertValidate certValidate) {
119      this.certValidate = certValidate;
120      return this;
121   }
122
123   /**
124    * Bean property getter:  <property>hostVerify</property>.
125    *
126    * @return The value of the <property>hostVerify</property> property on this bean, or <jk>null</jk> if it is not set.
127    */
128   public HostVerify getHostVerify() {
129      return hostVerify;
130   }
131
132   /**
133    * Bean property setter:  <property>hostVerify</property>.
134    *
135    * @param hostVerify The new value for the <property>properties</property> property on this bean.
136    * @return This object (for method chaining).
137    */
138   public SSLOpts setHostVerify(HostVerify hostVerify) {
139      this.hostVerify = hostVerify;
140      return this;
141   }
142
143
144   //--------------------------------------------------------------------------------
145   // Enums
146   //--------------------------------------------------------------------------------
147
148   /**
149    * Certificate validation options.
150    * <p>
151    * Used as enum for {@link SSLOpts#getCertValidate()} property.
152    */
153   @SuppressWarnings("hiding")
154   public static enum CertValidate {
155
156      /**
157       * Verify that the certificate is valid, but allow for self-signed certificates.
158       */
159      LAX,
160
161      /**
162       * Do normal certificate chain validation.
163       */
164      DEFAULT
165   }
166
167   /**
168    * Certificate host verification options.
169    * <p>
170    * Used as enum for {@link SSLOpts#getHostVerify()} property.
171    */
172   @SuppressWarnings("hiding")
173   public enum HostVerify {
174
175      /**
176       * Don't verify the hostname in the certificate.
177       */
178      LAX,
179
180      /**
181       * Do normal hostname verification.
182       */
183      DEFAULT
184   }
